  136.     public function all_deps( $handles, $recursion = false, $group = false ) { 
  137.         if ( !$handles = (array) $handles ) 
  138.             return false; 
  139.  
  140.         foreach ( $handles as $handle ) { 
  141.             $handle_parts = explode('?', $handle); 
  142.             $handle = $handle_parts[0]; 
  143.             $queued = in_array($handle, $this->to_do, true); 
  144.  
  145.             if ( in_array($handle, $this->done, true) ) // Already done 
  146.                 continue; 
  147.  
  148.             $moved = $this->set_group( $handle, $recursion, $group ); 
  149.             $new_group = $this->groups[ $handle ]; 
  150.  
  151.             if ( $queued && !$moved ) // already queued and in the right group 
  152.                 continue; 
  153.  
  154.             $keep_going = true; 
  155.             if ( !isset($this->registered[$handle]) ) 
  156.                 $keep_going = false; // Item doesn't exist. 
  157.             elseif ( $this->registered[$handle]->deps && array_diff($this->registered[$handle]->deps, array_keys($this->registered)) ) 
  158.                 $keep_going = false; // Item requires dependencies that don't exist. 
  159.             elseif ( $this->registered[$handle]->deps && !$this->all_deps( $this->registered[$handle]->deps, true, $new_group ) ) 
  160.                 $keep_going = false; // Item requires dependencies that don't exist. 
  161.  
  162.             if ( ! $keep_going ) { // Either item or its dependencies don't exist. 
  163.                 if ( $recursion ) 
  164.                     return false; // Abort this branch. 
  165.                 else 
  166.                     continue; // We're at the top level. Move on to the next one. 
  167.             } 
  168.  
  169.             if ( $queued ) // Already grabbed it and its dependencies. 
  170.                 continue; 
  171.  
  172.             if ( isset($handle_parts[1]) ) 
  173.                 $this->args[$handle] = $handle_parts[1]; 
  174.
