  1741.     public static function MPEGaudioHeaderDecode($Header4Bytes) { 
  1742.         // AAAA AAAA  AAAB BCCD  EEEE FFGH  IIJJ KLMM 
  1743.         // A - Frame sync (all bits set) 
  1744.         // B - MPEG Audio version ID 
  1745.         // C - Layer description 
  1746.         // D - Protection bit 
  1747.         // E - Bitrate index 
  1748.         // F - Sampling rate frequency index 
  1749.         // G - Padding bit 
  1750.         // H - Private bit 
  1751.         // I - Channel Mode 
  1752.         // J - Mode extension (Only if Joint stereo) 
  1753.         // K - Copyright 
  1754.         // L - Original 
  1755.         // M - Emphasis 
  1756.  
  1757.         if (strlen($Header4Bytes) != 4) { 
  1758.             return false; 
  1759.         } 
  1760.  
  1761.         $MPEGrawHeader['synch'] = (getid3_lib::BigEndian2Int(substr($Header4Bytes, 0, 2)) & 0xFFE0) >> 4; 
  1762.         $MPEGrawHeader['version'] = (ord($Header4Bytes{1}) & 0x18) >> 3; //    BB 
  1763.         $MPEGrawHeader['layer'] = (ord($Header4Bytes{1}) & 0x06) >> 1; //      CC 
  1764.         $MPEGrawHeader['protection'] = (ord($Header4Bytes{1}) & 0x01);      //        D 
  1765.         $MPEGrawHeader['bitrate'] = (ord($Header4Bytes{2}) & 0xF0) >> 4; // EEEE 
  1766.         $MPEGrawHeader['sample_rate'] = (ord($Header4Bytes{2}) & 0x0C) >> 2; //     FF 
  1767.         $MPEGrawHeader['padding'] = (ord($Header4Bytes{2}) & 0x02) >> 1; //       G 
  1768.         $MPEGrawHeader['private'] = (ord($Header4Bytes{2}) & 0x01);      //        H 
  1769.         $MPEGrawHeader['channelmode'] = (ord($Header4Bytes{3}) & 0xC0) >> 6; // II 
  1770.         $MPEGrawHeader['modeextension'] = (ord($Header4Bytes{3}) & 0x30) >> 4; //   JJ 
  1771.         $MPEGrawHeader['copyright'] = (ord($Header4Bytes{3}) & 0x08) >> 3; //     K 
  1772.         $MPEGrawHeader['original'] = (ord($Header4Bytes{3}) & 0x04) >> 2; //      L 
  1773.         $MPEGrawHeader['emphasis'] = (ord($Header4Bytes{3}) & 0x03);      //       MM 
  1774.  
  1775.         return $MPEGrawHeader; 
  1776.     }
