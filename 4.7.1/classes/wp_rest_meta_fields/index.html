  211.

  214.  
  215.         foreach ( $to_add as $add_key => $value ) { 
  216.             $remove_keys = array_keys( $to_remove, $value, true ); 
  217.  
  218.             if ( empty( $remove_keys ) ) { 
  219.                 continue; 
  220.             } 
  221.  
  222.             if ( count( $remove_keys ) > 1 ) { 
  223.                 // To remove, we need to remove first, then add, so don't touch.

  231.         } 
  232.  
  233.         // `delete_metadata` removes _all_ instances of the value, so only call once. 
  234.         $to_remove = array_unique( $to_remove );

  237.             if ( ! delete_metadata( $meta_type, $object_id, wp_slash( $meta_key ), wp_slash( $value ) ) ) { 
  238.                 return new WP_Error( 
  239.                     'rest_meta_database_error',  
  240.                     __( 'Could not update meta value in database.' ),  
  241.                     array( 'key' => $name, 'status' => WP_Http::INTERNAL_SERVER_ERROR ) 
  242.  );

  339.  ); 
  340.  
  341.             $rest_args = array_merge( $default_args, $rest_args ); 
  342.             $rest_args['schema'] = array_merge( $default_schema, $rest_args['schema'] );
