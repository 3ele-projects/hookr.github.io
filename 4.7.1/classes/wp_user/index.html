  626.     /** 
  627.      * Remove capability from user. 
  628.      * 
  629.      * @since 2.0.0 
  630.      * @access public 
  631.      * 
  632.      * @param string $cap Capability name. 
  633.      */ 
  634.     public function remove_cap( $cap ) { 
  635.         if ( ! isset( $this->caps[ $cap ] ) ) { 
  636.             return; 
  637.         } 
  638.         unset( $this->caps[ $cap ] ); 
  639.         update_user_meta( $this->ID, $this->cap_key, $this->caps ); 
  640.         $this->get_role_caps(); 
  641.         $this->update_user_level_from_caps(); 
  642.     } 
  643.  
  644.     /** 
  645.      * Remove all of the capabilities of the user. 
  646.      * 
  647.      * @since 2.1.0 
  648.      * @access public 
  649.      * 
  650.      * @global wpdb $wpdb WordPress database abstraction object. 
  651.      */ 
  652.     public function remove_all_caps() { 
  653.         global $wpdb; 
  654.         $this->caps = array(); 
  655.         delete_user_meta( $this->ID, $this->cap_key ); 
  656.         delete_user_meta( $this->ID, $wpdb->get_blog_prefix() . 'user_level' ); 
  657.         $this->get_role_caps(); 
  658.     }

  660.     /** 
  661.      * Whether user has capability or role name. 
  662.      * 
  663.      * While checking against particular roles in place of a capability is supported 
  664.      * in part, this practice is discouraged as it may produce unreliable results. 
  665.      * 
  666.      * @since 2.0.0 
  667.      * @access public 
  668.      * 
  669.      * @see map_meta_cap() 
  670.      * 
  671.      * @param string $cap           Capability name. 
  672.      * @param int    $object_id, ... Optional. ID of the specific object to check against if `$cap` is a "meta" cap. 
  673.      *                              "Meta" capabilities, e.g. 'edit_post', 'edit_user', etc., are capabilities used 
  674.      *                              by map_meta_cap() to map to other "primitive" capabilities, e.g. 'edit_posts',  
  675.      *                              'edit_others_posts', etc. The parameter is accessed via func_get_args() and passed 
  676.      *                              to map_meta_cap(). 
  677.      * @return bool Whether the current user has the given capability. If `$cap` is a meta cap and `$object_id` is 
  678.      *              passed, whether the current user has the given meta capability for the given object. 
  679.      */ 
  680.     public function has_cap( $cap ) { 
  681.         if ( is_numeric( $cap ) ) { 
  682.             _deprecated_argument( __FUNCTION__, '2.0.0', __('Usage of user levels by plugins and themes is deprecated. Use roles and capabilities instead.') ); 
  683.             $cap = $this->translate_level_to_cap( $cap ); 
  684.         } 
  685.  
  686.         $args = array_slice( func_get_args(), 1 ); 
  687.         $args = array_merge( array( $cap, $this->ID ), $args ); 
  688.         $caps = call_user_func_array( 'map_meta_cap', $args ); 
  689.  
  690.         // Multisite super admin has all caps by definition, Unless specifically denied. 
  691.         if ( is_multisite() && is_super_admin( $this->ID ) ) { 
  692.             if ( in_array('do_not_allow', $caps) ) 
  693.                 return false; 
  694.             return true; 
  695.         } 
  696.  
  697.         /** 
  698.          * Dynamically filter a user's capabilities. 
  699.          * 
  700.          * @since 2.0.0 
  701.          * @since 3.7.0 Added the user object. 
  702.          * 
  703.          * @param array   $allcaps An array of all the user's capabilities. 
  704.          * @param array   $caps    Actual capabilities for meta capability. 
  705.          * @param array   $args    Optional parameters passed to has_cap(), typically object ID. 
  706.          * @param WP_User $user    The user object. 
  707.          */ 
  708.         $capabilities = apply_filters( 'user_has_cap', $this->allcaps, $caps, $args, $this ); 
  709.  
  710.         // Everyone is allowed to exist. 
  711.         $capabilities['exist'] = true; 
  712.  
  713.         // Must have ALL requested caps. 
  714.         foreach ( (array) $caps as $cap ) { 
  715.             if ( empty( $capabilities[ $cap ] ) ) 
  716.                 return false; 
  717.         } 
  718.  
  719.         return true; 
  720.     } 
  721.  
  722.     /** 
  723.      * Convert numeric level to level capability name. 
  724.      * 
  725.      * Prepends 'level_' to level number. 
  726.      * 
  727.      * @since 2.0.0 
  728.      * @access public 
  729.      * 
  730.      * @param int $level Level number, 1 to 10. 
  731.      * @return string 
  732.      */ 
  733.     public function translate_level_to_cap( $level ) { 
  734.         return 'level_' . $level; 
  735.     }

  739.      * 
  740.      * @since 3.0.0

  746.     public function for_blog( $blog_id = '' ) {

  749.             $cap_key = $wpdb->get_blog_prefix( $blog_id ) . 'capabilities'; 
  750.         else 
  751.             $cap_key = ''; 
  752.         $this->_init_caps( $cap_key ); 
  753.     }
