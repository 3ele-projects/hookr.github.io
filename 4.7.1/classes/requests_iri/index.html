  616.     /**

  621.      */ 
  622.     public function is_valid() { 
  623.         $isauthority = $this->iuserinfo !== null || $this->ihost !== null || $this->port !== null; 
  624.         if ($this->ipath !== '' && 
  625.             ( 
  626.                 $isauthority && $this->ipath[0] !== '/' || 
  627.                 ( 
  628.                     $this->scheme === null && 
  629.                     !$isauthority && 
  630.                     strpos($this->ipath, ':') !== false && 
  631.                     (strpos($this->ipath, '/') === false ? true : strpos($this->ipath, ':') < strpos($this->ipath, '/')) 
  632.  ) 
  633.  ) 
  634.  ) {

  636.         }

  639.     }

  641.     /**

  646.      * @return bool 
  647.      */

  650.         if (!$cache) { 
  651.             $cache = array(); 
  652.         } 
  653.  
  654.         if ($iri === null) {

  678.                              $this->iuserinfo,  
  679.                              $this->ihost,  
  680.                              $this->port,  
  681.                              $this->ipath,  
  682.                              $this->iquery,  
  683.                              $this->ifragment,  
  684.                              $return); 
  685.         return $return; 
  686.     }

  697.             $this->scheme = null; 
  698.         }

  701.             return false; 
  702.         } 
  703.         else {

  705.         } 
  706.         return true; 
  707.     }

  709.     /** 
  710.      * Set the authority. Returns true on success, false on failure (if there are

  712.      * 
  713.      * @param string $authority 
  714.      * @return bool 
  715.      */ 
  716.     protected function set_authority($authority) { 
  717.         static $cache; 
  718.         if (!$cache) { 
  719.             $cache = array(); 
  720.         } 
  721.  
  722.         if ($authority === null) { 
  723.             $this->iuserinfo = null; 
  724.             $this->ihost = null; 
  725.             $this->port = null; 
  726.             return true; 
  727.         } 
  728.         if (isset($cache[$authority])) { 
  729.             list($this->iuserinfo,  
  730.                  $this->ihost,  
  731.                  $this->port,  
  732.                  $return) = $cache[$authority]; 
  733.  
  734.             return $return; 
  735.         } 
  736.  
  737.         $remaining = $authority; 
  738.         if (($iuserinfo_end = strrpos($remaining, '@')) !== false) { 
  739.             $iuserinfo = substr($remaining, 0, $iuserinfo_end); 
  740.             $remaining = substr($remaining, $iuserinfo_end + 1); 
  741.         } 
  742.         else { 
  743.             $iuserinfo = null; 
  744.         } 
  745.         if (($port_start = strpos($remaining, ':', strpos($remaining, ']'))) !== false) { 
  746.             $port = substr($remaining, $port_start + 1); 
  747.             if ($port === false || $port === '') { 
  748.                 $port = null; 
  749.             } 
  750.             $remaining = substr($remaining, 0, $port_start); 
  751.         } 
  752.         else { 
  753.             $port = null; 
  754.         } 
  755.  
  756.         $return = $this->set_userinfo($iuserinfo) && 
  757.                   $this->set_host($remaining) && 
  758.                   $this->set_port($port); 
  759.  
  760.         $cache[$authority] = array($this->iuserinfo,  
  761.                                    $this->ihost,  
  762.                                    $this->port,  
  763.                                    $return); 
  764.  
  765.         return $return; 
  766.     }

  774.     protected function set_userinfo($iuserinfo) {

  776.             $this->iuserinfo = null; 
  777.         } 
  778.         else {

  786.     /** 
  787.      * Set the ihost. Returns true on success, false on failure (if there are 
  788.      * any invalid characters). 
  789.      * 
  790.      * @param string $ihost 
  791.      * @return bool 
  792.      */ 
  793.     protected function set_host($ihost) { 
  794.         if ($ihost === null) { 
  795.             $this->ihost = null; 
  796.             return true; 
  797.         } 
  798.         if (substr($ihost, 0, 1) === '[' && substr($ihost, -1) === ']') { 
  799.             if (Requests_IPv6::check_ipv6(substr($ihost, 1, -1))) { 
  800.                 $this->ihost = '[' . Requests_IPv6::compress(substr($ihost, 1, -1)) . ']'; 
  801.             } 
  802.             else { 
  803.                 $this->ihost = null; 
  804.                 return false; 
  805.             } 
  806.         } 
  807.         else { 
  808.             $ihost = $this->replace_invalid_with_pct_encoding($ihost, '!$&\'()*+, ;='); 
  809.  
  810.             // Lowercase, but ignore pct-encoded sections (as they should 
  811.             // remain uppercase). This must be done after the previous step 
  812.             // as that can add unescaped characters. 
  813.             $position = 0; 
  814.             $strlen = strlen($ihost); 
  815.             while (($position += strcspn($ihost, 'ABCDEFGHIJKLMNOPQRSTUVWXYZ%', $position)) < $strlen) { 
  816.                 if ($ihost[$position] === '%') { 
  817.                     $position += 3; 
  818.                 } 
  819.                 else { 
  820.                     $ihost[$position] = strtolower($ihost[$position]); 
  821.                     $position++; 
  822.                 } 
  823.             } 
  824.  
  825.             $this->ihost = $ihost; 
  826.         } 
  827.  
  828.         $this->scheme_normalization(); 
  829.  
  830.         return true; 
  831.     }

  926.             return false; 
  927.         } 
  928.

  932.         }

  936.         while (($position += strcspn($string, $non_ascii, $position)) < $strlen) {

  938.             $position += 3; 
  939.             $strlen += 2; 
  940.         } 
  941.

  943.     }
