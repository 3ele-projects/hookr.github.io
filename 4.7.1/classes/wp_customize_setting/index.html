  561.      * @access protected 
  562.      * 
  563.      * @param mixed $default Value to return if root does not exist. 
  564.      * @return mixed 
  565.      */ 
  566.     protected function get_root_value( $default = null ) { 
  567.         $id_base = $this->id_data['base']; 
  568.         if ( 'option' === $this->type ) { 
  569.             return get_option( $id_base, $default ); 
  570.         } else if ( 'theme_mod' ) { 
  571.             return get_theme_mod( $id_base, $default ); 
  572.         } else { 
  573.             /** 
  574.              * Any WP_Customize_Setting subclass implementing aggregate multidimensional 
  575.              * will need to override this method to obtain the data from the appropriate 
  576.              * location. 
  577.              */ 
  578.             return $default; 
  579.         } 
  580.     }

  582.     /** 
  583.      * Set the root value for a setting, especially for multidimensional ones. 
  584.      * 
  585.      * @since 4.4.0 
  586.      * @access protected 
  587.      * 
  588.      * @param mixed $value Value to set as root of multidimensional setting. 
  589.      * @return bool Whether the multidimensional root was updated successfully. 
  590.      */ 
  591.     protected function set_root_value( $value ) { 
  592.         $id_base = $this->id_data['base']; 
  593.         if ( 'option' === $this->type ) { 
  594.             $autoload = true; 
  595.             if ( isset( self::$aggregated_multidimensionals[ $this->type ][ $this->id_data['base'] ]['autoload'] ) ) { 
  596.                 $autoload = self::$aggregated_multidimensionals[ $this->type ][ $this->id_data['base'] ]['autoload']; 
  597.             } 
  598.             return update_option( $id_base, $value, $autoload ); 
  599.         } else if ( 'theme_mod' ) { 
  600.             set_theme_mod( $id_base, $value ); 
  601.             return true; 
  602.         } else { 
  603.             /** 
  604.              * Any WP_Customize_Setting subclass implementing aggregate multidimensional 
  605.              * will need to override this method to obtain the data from the appropriate 
  606.              * location. 
  607.              */ 
  608.             return false; 
  609.         } 
  610.     }

  612.     /** 
  613.      * Save the value of the setting, using the related API. 
  614.      * 
  615.      * @since 3.4.0 
  616.      * 
  617.      * @param mixed $value The value to update. 
  618.      * @return bool The result of saving the value. 
  619.      */ 
  620.     protected function update( $value ) { 
  621.         $id_base = $this->id_data['base']; 
  622.         if ( 'option' === $this->type || 'theme_mod' === $this->type ) { 
  623.             if ( ! $this->is_multidimensional_aggregated ) { 
  624.                 return $this->set_root_value( $value ); 
  625.             } else { 
  626.                 $root = self::$aggregated_multidimensionals[ $this->type ][ $id_base ]['root_value']; 
  627.                 $root = $this->multidimensional_replace( $root, $this->id_data['keys'], $value ); 
  628.                 self::$aggregated_multidimensionals[ $this->type ][ $id_base ]['root_value'] = $root; 
  629.                 return $this->set_root_value( $root ); 
  630.             } 
  631.         } else { 
  632.             /** 
  633.              * Fires when the WP_Customize_Setting::update() method is called for settings 
  634.              * not handled as theme_mods or options. 
  635.              * 
  636.              * The dynamic portion of the hook name, `$this->type`, refers to the type of setting. 
  637.              * 
  638.              * @since 3.4.0 
  639.              * 
  640.              * @param mixed                $value Value of the setting. 
  641.              * @param WP_Customize_Setting $this  WP_Customize_Setting instance. 
  642.              */ 
  643.             do_action( "customize_update_{$this->type}", $value, $this ); 
  644.  
  645.             return has_action( "customize_update_{$this->type}" ); 
  646.         } 
  647.     }

  669.     /** 
  670.      * Fetch the value of the setting. 
  671.      * 
  672.      * @since 3.4.0 
  673.      * 
  674.      * @return mixed The value. 
  675.      */ 
  676.     public function value() { 
  677.         $id_base = $this->id_data['base']; 
  678.         $is_core_type = ( 'option' === $this->type || 'theme_mod' === $this->type ); 
  679.  
  680.         if ( ! $is_core_type && ! $this->is_multidimensional_aggregated ) { 
  681.  
  682.             // Use post value if previewed and a post value is present. 
  683.             if ( $this->is_previewed ) { 
  684.                 $value = $this->post_value( null ); 
  685.                 if ( null !== $value ) { 
  686.                     return $value; 
  687.                 } 
  688.             } 
  689.  
  690.             $value = $this->get_root_value( $this->default ); 
  691.  
  692.             /** 
  693.              * Filters a Customize setting value not handled as a theme_mod or option. 
  694.              * 
  695.              * The dynamic portion of the hook name, `$id_base`, refers to 
  696.              * the base slug of the setting name, initialized from `$this->id_data['base']`. 
  697.              * 
  698.              * For settings handled as theme_mods or options, see those corresponding 
  699.              * functions for available hooks. 
  700.              * 
  701.              * @since 3.4.0 
  702.              * @since 4.6.0 Added the `$this` setting instance as the second parameter. 
  703.              * 
  704.              * @param mixed                $default The setting default value. Default empty. 
  705.              * @param WP_Customize_Setting $this    The setting instance. 
  706.              */ 
  707.             $value = apply_filters( "customize_value_{$id_base}", $value, $this ); 
  708.         } elseif ( $this->is_multidimensional_aggregated ) { 
  709.             $root_value = self::$aggregated_multidimensionals[ $this->type ][ $id_base ]['root_value']; 
  710.             $value = $this->multidimensional_get( $root_value, $this->id_data['keys'], $this->default ); 
  711.  
  712.             // Ensure that the post value is used if the setting is previewed, since preview filters aren't applying on cached $root_value. 
  713.             if ( $this->is_previewed ) { 
  714.                 $value = $this->post_value( $value ); 
  715.             } 
  716.         } else { 
  717.             $value = $this->get_root_value( $this->default ); 
  718.         } 
  719.         return $value; 
  720.     }

  722.     /** 
  723.      * Sanitize the setting's value for use in JavaScript. 
  724.      * 
  725.      * @since 3.4.0 
  726.      * 
  727.      * @return mixed The requested escaped value. 
  728.      */ 
  729.     public function js_value() { 
  730.  
  731.         /** 
  732.          * Filters a Customize setting value for use in JavaScript. 
  733.          * 
  734.          * The dynamic portion of the hook name, `$this->id`, refers to the setting ID. 
  735.          * 
  736.          * @since 3.4.0 
  737.          * 
  738.          * @param mixed                $value The setting value. 
  739.          * @param WP_Customize_Setting $this  WP_Customize_Setting instance. 
  740.          */ 
  741.         $value = apply_filters( "customize_sanitize_js_{$this->id}", $this->value(), $this ); 
  742.  
  743.         if ( is_string( $value ) ) 
  744.             return html_entity_decode( $value, ENT_QUOTES, 'UTF-8'); 
  745.  
  746.         return $value; 
  747.     }

  775.             return false;

  777.         if ( $this->theme_supports && ! call_user_func_array( 'current_theme_supports', (array) $this->theme_supports ) )

  783.     /** 
  784.      * Multidimensional helper function. 
  785.      * 
  786.      * @since 3.4.0 
  787.      * 
  788.      * @param $root 
  789.      * @param $keys 
  790.      * @param bool $create Default is false. 
  791.      * @return array|void Keys are 'root', 'node', and 'key'. 
  792.      */ 
  793.     final protected function multidimensional( &$root, $keys, $create = false ) { 
  794.         if ( $create && empty( $root ) ) 
  795.             $root = array(); 
  796.  
  797.         if ( ! isset( $root ) || empty( $keys ) ) 
  798.             return; 
  799.  
  800.         $last = array_pop( $keys ); 
  801.         $node = &$root; 
  802.  
  803.         foreach ( $keys as $key ) { 
  804.             if ( $create && ! isset( $node[ $key ] ) ) 
  805.                 $node[ $key ] = array(); 
  806.  
  807.             if ( ! is_array( $node ) || ! isset( $node[ $key ] ) ) 
  808.                 return; 
  809.  
  810.             $node = &$node[ $key ]; 
  811.         } 
  812.  
  813.         if ( $create ) { 
  814.             if ( ! is_array( $node ) ) { 
  815.                 // account for an array overriding a string or object value 
  816.                 $node = array(); 
  817.             } 
  818.             if ( ! isset( $node[ $last ] ) ) { 
  819.                 $node[ $last ] = array(); 
  820.             } 
  821.         } 
  822.  
  823.         if ( ! isset( $node[ $last ] ) ) 
  824.             return; 
  825.  
  826.         return array( 
  827.             'root' => &$root,  
  828.             'node' => &$node,  
  829.             'key' => $last,  
  830.  ); 
  831.     }

  833.     /** 
  834.      * Will attempt to replace a specific value in a multidimensional array. 
  835.      * 
  836.      * @since 3.4.0 
  837.      * 
  838.      * @param $root 
  839.      * @param $keys 
  840.      * @param mixed $value The value to update. 
  841.      * @return mixed 
  842.      */ 
  843.     final protected function multidimensional_replace( $root, $keys, $value ) { 
  844.         if ( ! isset( $value ) ) 
  845.             return $root; 
  846.         elseif ( empty( $keys ) ) // If there are no keys, we're replacing the root. 
  847.             return $value; 
  848.  
  849.         $result = $this->multidimensional( $root, $keys, true ); 
  850.  
  851.         if ( isset( $result ) ) 
  852.             $result['node'][ $result['key'] ] = $value; 
  853.  
  854.         return $root; 
  855.     } 
  856.  
  857.     /** 
  858.      * Will attempt to fetch a specific value from a multidimensional array. 
  859.      * 
  860.      * @since 3.4.0 
  861.      * 
  862.      * @param $root 
  863.      * @param $keys 
  864.      * @param mixed $default A default value which is used as a fallback. Default is null. 
  865.      * @return mixed The requested value or the default value. 
  866.      */ 
  867.     final protected function multidimensional_get( $root, $keys, $default = null ) { 
  868.         if ( empty( $keys ) ) // If there are no keys, test the root. 
  869.             return isset( $root ) ? $root : $default; 
  870.  
  871.         $result = $this->multidimensional( $root, $keys ); 
  872.         return isset( $result ) ? $result['node'][ $result['key'] ] : $default; 
  873.     }

  875.     /** 
  876.      * Will attempt to check if a specific value in a multidimensional array is set. 
  877.      * 
  878.      * @since 3.4.0 
  879.      * 
  880.      * @param $root 
  881.      * @param $keys 
  882.      * @return bool True if value is set, false if not. 
  883.      */ 
  884.     final protected function multidimensional_isset( $root, $keys ) { 
  885.         $result = $this->multidimensional_get( $root, $keys ); 
  886.         return isset( $result ); 
  887.     }
