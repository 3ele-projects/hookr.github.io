  169.         if ( isset( $this->cache[ $folder ] ) ) 
  170.             return $this->cache[ $folder ]; 
  171.  
  172.         if ( stripos($this->method, 'ftp') !== false ) { 
  173.             $constant_overrides = array( 
  174.                 'FTP_BASE' => ABSPATH,  
  175.                 'FTP_CONTENT_DIR' => WP_CONTENT_DIR,  
  176.                 'FTP_PLUGIN_DIR' => WP_PLUGIN_DIR,  
  177.                 'FTP_LANG_DIR' => WP_LANG_DIR 
  178.  ); 
  179.  
  180.             // Direct matches ( folder = CONSTANT/ ) 
  181.             foreach ( $constant_overrides as $constant => $dir ) { 
  182.                 if ( ! defined( $constant ) ) 
  183.                     continue; 
  184.                 if ( $folder === $dir ) 
  185.                     return trailingslashit( constant( $constant ) ); 
  186.             } 
  187.  
  188.             // Prefix Matches ( folder = CONSTANT/subdir ) 
  189.             foreach ( $constant_overrides as $constant => $dir ) { 
  190.                 if ( ! defined( $constant ) ) 
  191.                     continue; 
  192.                 if ( 0 === stripos( $folder, $dir ) ) { // $folder starts with $dir 
  193.                     $potential_folder = preg_replace( '#^' . preg_quote( $dir, '#' ) . '/#i', trailingslashit( constant( $constant ) ), $folder ); 
  194.                     $potential_folder = trailingslashit( $potential_folder ); 
  195.  
  196.                     if ( $this->is_dir( $potential_folder ) ) { 
  197.                         $this->cache[ $folder ] = $potential_folder; 
  198.                         return $potential_folder; 
  199.                     } 
  200.                 } 
  201.             } 
  202.         } elseif ( 'direct' == $this->method ) { 
  203.             $folder = str_replace('\\', '/', $folder); // Windows path sanitisation 
  204.             return trailingslashit($folder); 
  205.         }

  224.      * Locate a folder on the remote filesystem. 
  225.      * 
  226.      * Expects Windows sanitized path. 
  227.      * 
  228.      * @access public 
  229.      * @since 2.7.0 
  230.      * 
  231.      * @param string $folder The folder to locate. 
  232.      * @param string $base   The folder to start searching from. 
  233.      * @param bool   $loop   If the function has recursed, Internal use only. 
  234.      * @return string|false The location of the remote path, false to cease looping. 
  235.      */ 
  236.     public function search_for_folder( $folder, $base = '.', $loop = false ) { 
  237.         if ( empty( $base ) || '.' == $base ) 
  238.             $base = trailingslashit($this->cwd()); 
  239.  
  240.         $folder = untrailingslashit($folder); 
  241.  
  242.         if ( $this->verbose ) { 
  243.             /** translators: 1: folder to locate, 2: folder to start searching from */ 
  244.             printf( "\n" . __( 'Looking for %1$s in %2$s' ) . "<br/>\n", $folder, $base ); 
  245.         } 
  246.  
  247.         $folder_parts = explode('/', $folder); 
  248.         $folder_part_keys = array_keys( $folder_parts ); 
  249.         $last_index = array_pop( $folder_part_keys ); 
  250.         $last_path = $folder_parts[ $last_index ]; 
  251.  
  252.         $files = $this->dirlist( $base ); 
  253.  
  254.         foreach ( $folder_parts as $index => $key ) { 
  255.             if ( $index == $last_index ) 
  256.                 continue; // We want this to be caught by the next code block. 
  257.  
  258.             /** 
  259.              * Working from /home/ to /user/ to /wordpress/ see if that file exists within 
  260.              * the current folder, If it's found, change into it and follow through looking 
  261.              * for it. If it cant find WordPress down that route, it'll continue onto the next 
  262.              * folder level, and see if that matches, and so on. If it reaches the end, and still 
  263.              * cant find it, it'll return false for the entire function. 
  264.              */ 
  265.             if ( isset($files[ $key ]) ) { 
  266.  
  267.                 // Let's try that folder: 
  268.                 $newdir = trailingslashit(path_join($base, $key)); 
  269.                 if ( $this->verbose ) { 
  270.                     /** translators: %s: directory name */ 
  271.                     printf( "\n" . __( 'Changing to %s' ) . "<br/>\n", $newdir ); 
  272.                 } 
  273.  
  274.                 // Only search for the remaining path tokens in the directory, not the full path again. 
  275.                 $newfolder = implode( '/', array_slice( $folder_parts, $index + 1 ) ); 
  276.                 if ( $ret = $this->search_for_folder( $newfolder, $newdir, $loop) ) 
  277.                     return $ret; 
  278.             } 
  279.         } 
  280.  
  281.         // Only check this as a last resort, to prevent locating the incorrect install. 
  282.         // All above procedures will fail quickly if this is the right branch to take. 
  283.         if (isset( $files[ $last_path ] ) ) { 
  284.             if ( $this->verbose ) { 
  285.                 /** translators: %s: directory name */ 
  286.                 printf( "\n" . __( 'Found %s' ) . "<br/>\n", $base . $last_path ); 
  287.             } 
  288.             return trailingslashit($base . $last_path); 
  289.         } 
  290.  
  291.         // Prevent this function from looping again. 
  292.         // No need to proceed if we've just searched in / 
  293.         if ( $loop || '/' == $base ) 
  294.             return false; 
  295.  
  296.         // As an extra last resort, Change back to / if the folder wasn't found. 
  297.         // This comes into effect when the CWD is /home/user/ but WP is at /var/www/.... 
  298.         return $this->search_for_folder( $folder, '/', true ); 
  299.  
  300.     }

  303.      * Return the *nix-style file permissions for a file. 
  304.      * 
  305.      * From the PHP documentation page for fileperms(). 
  306.      * 
  307.      * @link https://secure.php.net/manual/en/function.fileperms.php 
  308.      * 
  309.      * @access public 
  310.      * @since 2.5.0 
  311.      * 
  312.      * @param string $file String filename. 
  313.      * @return string The *nix-style representation of permissions. 
  314.      */ 
  315.     public function gethchmod( $file ) { 
  316.         $perms = intval( $this->getchmod( $file ), 8 ); 
  317.         if (($perms & 0xC000) == 0xC000) // Socket 
  318.             $info = 's'; 
  319.         elseif (($perms & 0xA000) == 0xA000) // Symbolic Link 
  320.             $info = 'l'; 
  321.         elseif (($perms & 0x8000) == 0x8000) // Regular 
  322.             $info = '-'; 
  323.         elseif (($perms & 0x6000) == 0x6000) // Block special 
  324.             $info = 'b'; 
  325.         elseif (($perms & 0x4000) == 0x4000) // Directory 
  326.             $info = 'd'; 
  327.         elseif (($perms & 0x2000) == 0x2000) // Character special 
  328.             $info = 'c'; 
  329.         elseif (($perms & 0x1000) == 0x1000) // FIFO pipe 
  330.             $info = 'p'; 
  331.         else // Unknown 
  332.             $info = 'u'; 
  333.  
  334.         // Owner 
  335.         $info .= (($perms & 0x0100) ? 'r' : '-'); 
  336.         $info .= (($perms & 0x0080) ? 'w' : '-'); 
  337.         $info .= (($perms & 0x0040) ? 
  338.                     (($perms & 0x0800) ? 's' : 'x' ) : 
  339.                     (($perms & 0x0800) ? 'S' : '-')); 
  340.  
  341.         // Group 
  342.         $info .= (($perms & 0x0020) ? 'r' : '-'); 
  343.         $info .= (($perms & 0x0010) ? 'w' : '-'); 
  344.         $info .= (($perms & 0x0008) ? 
  345.                     (($perms & 0x0400) ? 's' : 'x' ) : 
  346.                     (($perms & 0x0400) ? 'S' : '-')); 
  347.  
  348.         // World 
  349.         $info .= (($perms & 0x0004) ? 'r' : '-'); 
  350.         $info .= (($perms & 0x0002) ? 'w' : '-'); 
  351.         $info .= (($perms & 0x0001) ? 
  352.                     (($perms & 0x0200) ? 't' : 'x' ) : 
  353.                     (($perms & 0x0200) ? 'T' : '-')); 
  354.         return $info; 
  355.     }

  384.         $realmode = ''; 
  385.         $legal =  array('', 'w', 'r', 'x', '-'); 
  386.         $attarray = preg_split('//', $mode); 
  387.  
  388.         for ( $i = 0, $c = count( $attarray ); $i < $c; $i++ ) {

  390.                $realmode .= $legal[$key];

  393.
