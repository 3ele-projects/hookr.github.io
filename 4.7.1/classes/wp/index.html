  505.          * Fires once the requested HTTP headers for caching, content type, etc. have been sent. 
  506.          * 
  507.          * @since 2.1.0 
  508.          * 
  509.          * @param WP &$this Current WordPress environment instance (passed by reference). 
  510.          */ 
  511.         do_action_ref_array( 'send_headers', array( &$this ) ); 
  512.     }

  514.     /** 
  515.      * Sets the query string property based off of the query variable property. 
  516.      * 
  517.      * The {@see 'query_string'} filter is deprecated, but still works. Plugins should 
  518.      * use the {@see 'request'} filter instead. 
  519.      * 
  520.      * @since 2.0.0 
  521.      * @access public 
  522.      */ 
  523.     public function build_query_string() { 
  524.         $this->query_string = ''; 
  525.         foreach ( (array) array_keys($this->query_vars) as $wpvar) { 
  526.             if ( '' != $this->query_vars[$wpvar] ) { 
  527.                 $this->query_string .= (strlen($this->query_string) < 1) ? '' : '&'; 
  528.                 if ( !is_scalar($this->query_vars[$wpvar]) ) // Discard non-scalars. 
  529.                     continue; 
  530.                 $this->query_string .= $wpvar . '=' . rawurlencode($this->query_vars[$wpvar]); 
  531.             } 
  532.         } 
  533.  
  534.         if ( has_filter( 'query_string' ) ) {  // Don't bother filtering and parsing if no plugins are hooked in. 
  535.             /** 
  536.              * Filters the query string before parsing. 
  537.              * 
  538.              * @since 1.5.0 
  539.              * @deprecated 2.1.0 Use 'query_vars' or 'request' filters instead. 
  540.              * 
  541.              * @param string $query_string The query string to modify. 
  542.              */ 
  543.             $this->query_string = apply_filters( 'query_string', $this->query_string ); 
  544.             parse_str($this->query_string, $this->query_vars); 
  545.         } 
  546.     } 
  547.  
  548.     /** 
  549.      * Set up the WordPress Globals. 
  550.      * 
  551.      * The query_vars property will be extracted to the GLOBALS. So care should 
  552.      * be taken when naming global variables that might interfere with the 
  553.      * WordPress environment. 
  554.      * 
  555.      * @since 2.0.0 
  556.      * @access public 
  557.      * 
  558.      * @global WP_Query     $wp_query 
  559.      * @global string       $query_string Query string for the loop. 
  560.      * @global array        $posts The found posts. 
  561.      * @global WP_Post|null $post The current post, if available. 
  562.      * @global string       $request The SQL statement for the request. 
  563.      * @global int          $more Only set, if single page or post. 
  564.      * @global int          $single If single page or post. Only set, if single page or post. 
  565.      * @global WP_User      $authordata Only set, if author archive. 
  566.      */ 
  567.     public function register_globals() { 
  568.         global $wp_query; 
  569.  
  570.         // Extract updated query vars back into global namespace. 
  571.         foreach ( (array) $wp_query->query_vars as $key => $value ) { 
  572.             $GLOBALS[ $key ] = $value; 
  573.         } 
  574.  
  575.         $GLOBALS['query_string'] = $this->query_string; 
  576.         $GLOBALS['posts'] = & $wp_query->posts; 
  577.         $GLOBALS['post'] = isset( $wp_query->post ) ? $wp_query->post : null; 
  578.         $GLOBALS['request'] = $wp_query->request; 
  579.  
  580.         if ( $wp_query->is_single() || $wp_query->is_page() ) { 
  581.             $GLOBALS['more'] = 1; 
  582.             $GLOBALS['single'] = 1; 
  583.         } 
  584.  
  585.         if ( $wp_query->is_author() && isset( $wp_query->post ) ) 
  586.             $GLOBALS['authordata'] = get_userdata( $wp_query->post->post_author ); 
  587.     } 
  588.

  590.      * Set up the current user. 
  591.      * 
  592.      * @since 2.0.0 
  593.      * @access public 
  594.      */ 
  595.     public function init() { 
  596.         wp_get_current_user(); 
  597.     }

  599.     /** 
  600.      * Set up the Loop based on the query variables. 
  601.      * 
  602.      * @since 2.0.0 
  603.      * @access public 
  604.      * 
  605.      * @global WP_Query $wp_the_query 
  606.      */ 
  607.     public function query_posts() { 
  608.         global $wp_the_query; 
  609.         $this->build_query_string(); 
  610.         $wp_the_query->query($this->query_vars); 
  611.      }

  613.      /** 
  614.      * Set the Headers for 404, if nothing is found for requested URL. 
  615.      * 
  616.      * Issue a 404 if a request doesn't match any posts and doesn't match 
  617.      * any object (e.g. an existing-but-empty category, tag, author) and a 404 was not already 
  618.      * issued, and if the request was not a search or the homepage. 
  619.      * 
  620.      * Otherwise, issue a 200. 
  621.      * 
  622.      * This sets headers after posts have been queried. handle_404() really means "handle status." 
  623.      * By inspecting the result of querying posts, seemingly successful requests can be switched to 
  624.      * a 404 so that canonical redirection logic can kick in. 
  625.      * 
  626.      * @since 2.0.0 
  627.      * @access public 
  628.      * 
  629.      * @global WP_Query $wp_query 
  630.       */ 
  631.     public function handle_404() { 
  632.         global $wp_query; 
  633.  
  634.         /** 
  635.          * Filters whether to short-circuit default header status handling. 
  636.          * 
  637.          * Returning a non-false value from the filter will short-circuit the handling 
  638.          * and return early. 
  639.          * 
  640.          * @since 4.5.0 
  641.          * 
  642.          * @param bool     $preempt  Whether to short-circuit default header status handling. Default false. 
  643.          * @param WP_Query $wp_query WordPress Query object. 
  644.          */ 
  645.         if ( false !== apply_filters( 'pre_handle_404', false, $wp_query ) ) { 
  646.             return; 
  647.         } 
  648.  
  649.         // If we've already issued a 404, bail. 
  650.         if ( is_404() ) 
  651.             return; 
  652.  
  653.         // Never 404 for the admin, robots, or if we found posts. 
  654.         if ( is_admin() || is_robots() || $wp_query->posts ) { 
  655.  
  656.             $success = true; 
  657.             if ( is_singular() ) { 
  658.                 $p = false; 
  659.  
  660.                 if ( $wp_query->post instanceof WP_Post ) { 
  661.                     $p = clone $wp_query->post; 
  662.                 } 
  663.  
  664.                 // Only set X-Pingback for single posts that allow pings. 
  665.                 if ( $p && pings_open( $p ) ) { 
  666.                     @header( 'X-Pingback: ' . get_bloginfo( 'pingback_url', 'display' ) ); 
  667.                 } 
  668.  
  669.                 // check for paged content that exceeds the max number of pages 
  670.                 $next = '<!--nextpage-->'; 
  671.                 if ( $p && false !== strpos( $p->post_content, $next ) && ! empty( $this->query_vars['page'] ) ) { 
  672.                     $page = trim( $this->query_vars['page'], '/' ); 
  673.                     $success = (int) $page <= ( substr_count( $p->post_content, $next ) + 1 ); 
  674.                 } 
  675.             } 
  676.  
  677.             if ( $success ) { 
  678.                 status_header( 200 ); 
  679.                 return; 
  680.             } 
  681.         } 
  682.  
  683.         // We will 404 for paged queries, as no posts were found. 
  684.         if ( ! is_paged() ) { 
  685.  
  686.             // Don't 404 for authors without posts as long as they matched an author on this site. 
  687.             $author = get_query_var( 'author' ); 
  688.             if ( is_author() && is_numeric( $author ) && $author > 0 && is_user_member_of_blog( $author ) ) { 
  689.                 status_header( 200 ); 
  690.                 return; 
  691.             } 
  692.  
  693.             // Don't 404 for these queries if they matched an object. 
  694.             if ( ( is_tag() || is_category() || is_tax() || is_post_type_archive() ) && get_queried_object() ) { 
  695.                 status_header( 200 ); 
  696.                 return; 
  697.             } 
  698.  
  699.             // Don't 404 for these queries either. 
  700.             if ( is_home() || is_search() || is_feed() ) { 
  701.                 status_header( 200 ); 
  702.                 return; 
  703.             } 
  704.         } 
  705.  
  706.         // Guess it's time to 404. 
  707.         $wp_query->set_404(); 
  708.         status_header( 404 ); 
  709.         nocache_headers(); 
  710.     }

  712.     /** 
  713.      * Sets up all of the variables required by the WordPress environment. 
  714.      * 
  715.      * The action {@see 'wp'} has one parameter that references the WP object. It 
  716.      * allows for accessing the properties and methods to further manipulate the 
  717.      * object. 
  718.      * 
  719.      * @since 2.0.0 
  720.      * @access public 
  721.      * 
  722.      * @param string|array $query_args Passed to parse_request(). 
  723.      */ 
  724.     public function main($query_args = '') { 
  725.         $this->init(); 
  726.         $this->parse_request($query_args); 
  727.         $this->send_headers(); 
  728.         $this->query_posts(); 
  729.         $this->handle_404(); 
  730.         $this->register_globals(); 
  731.  
  732.         /** 
  733.          * Fires once the WordPress environment has been set up. 
  734.          * 
  735.          * @since 2.1.0 
  736.          * 
  737.          * @param WP &$this Current WordPress environment instance (passed by reference). 
  738.          */ 
  739.         do_action_ref_array( 'wp', array( &$this ) ); 
  740.     }
