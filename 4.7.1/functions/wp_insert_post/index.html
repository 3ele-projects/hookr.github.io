wp_insert_post

Insert or update a post.

Description

(int|WP_Error) wp_insert_post( (array) $postarr, (bool) $wp_error = false ); 

If the $postarr parameter has ID set to a value, then post will be updated.

You can set the post date manually, by setting the values for post_date and post_date_gmt keys. You can close the comments or open the comments by setting the value for comment_status key.

Returns (int|WP_Error)

The post ID on success. The value 0 or WP_Error on failure.

Parameters (2)

0. $postarr (array)
An array of elements that make up a post to update or insert.

  316.     $data = wp_unslash( $data ); 
  317.     $where = array( 'ID' => $post_ID ); 
  318.  
  319.     if ( $update ) { 
  320.         /** 
  321.          * Fires immediately before an existing post is updated in the database. 
  322.          * 
  323.          * @since 2.5.0 
  324.          * 
  325.          * @param int   $post_ID Post ID. 
  326.          * @param array $data    Array of unslashed post data. 
  327.          */ 
  328.         do_action( 'pre_post_update', $post_ID, $data ); 
  329.         if ( false === $wpdb->update( $wpdb->posts, $data, $where ) ) { 
  330.             if ( $wp_error ) { 
  331.                 return new WP_Error('db_update_error', __('Could not update post in the database'), $wpdb->last_error); 
  332.             } else { 
  333.                 return 0; 
  334.             } 
  335.         } 
  336.     } else { 
  337.         // If there is a suggested ID, use it if not already present. 
  338.         if ( ! empty( $import_id ) ) { 
  339.             $import_id = (int) $import_id; 
  340.             if ( ! $wpdb->get_var( $wpdb->prepare("SELECT ID FROM $wpdb->posts WHERE ID = %d", $import_id) ) ) { 
  341.                 $data['ID'] = $import_id; 
  342.             } 
  343.         } 
  344.         if ( false === $wpdb->insert( $wpdb->posts, $data ) ) { 
  345.             if ( $wp_error ) { 
  346.                 return new WP_Error('db_insert_error', __('Could not insert post into the database'), $wpdb->last_error); 
  347.             } else { 
  348.                 return 0; 
  349.             } 
  350.         } 
  351.         $post_ID = (int) $wpdb->insert_id; 
  352.  
  353.         // Use the newly generated $post_ID. 
  354.         $where = array( 'ID' => $post_ID ); 
  355.     } 
  356.  
  357.     if ( empty( $data['post_name'] ) && ! in_array( $data['post_status'], array( 'draft', 'pending', 'auto-draft' ) ) ) { 
  358.         $data['post_name'] = wp_unique_post_slug( sanitize_title( $data['post_title'], $post_ID ), $post_ID, $data['post_status'], $post_type, $post_parent ); 
  359.         $wpdb->update( $wpdb->posts, array( 'post_name' => $data['post_name'] ), $where ); 
  360.         clean_post_cache( $post_ID ); 
  361.     } 
  362.  
  363.     if ( is_object_in_taxonomy( $post_type, 'category' ) ) { 
  364.         wp_set_post_categories( $post_ID, $post_category ); 
  365.     } 
  366.  
  367.     if ( isset( $postarr['tags_input'] ) && is_object_in_taxonomy( $post_type, 'post_tag' ) ) { 
  368.         wp_set_post_tags( $post_ID, $postarr['tags_input'] ); 
  369.     } 
  370.  
  371.     // New-style support for all custom taxonomies. 
  372.     if ( ! empty( $postarr['tax_input'] ) ) { 
  373.         foreach ( $postarr['tax_input'] as $taxonomy => $tags ) { 
  374.             $taxonomy_obj = get_taxonomy($taxonomy); 
  375.             if ( ! $taxonomy_obj ) { 
  376.                 /** translators: %s: taxonomy name */ 
  377.                 _doing_it_wrong( __FUNCTION__, sprintf( __( 'Invalid taxonomy: %s.' ), $taxonomy ), '4.4.0' ); 
  378.                 continue; 
  379.             } 
  380.  
  381.             // array = hierarchical, string = non-hierarchical. 
  382.             if ( is_array( $tags ) ) { 
  383.                 $tags = array_filter($tags); 
  384.             } 
  385.             if ( current_user_can( $taxonomy_obj->cap->assign_terms ) ) { 
  386.                 wp_set_post_terms( $post_ID, $tags, $taxonomy ); 
  387.             } 
  388.         } 
  389.     } 
  390.  
  391.     if ( ! empty( $postarr['meta_input'] ) ) { 
  392.         foreach ( $postarr['meta_input'] as $field => $value ) { 
  393.             update_post_meta( $post_ID, $field, $value ); 
  394.         } 
  395.     } 
  396.  
  397.     $current_guid = get_post_field( 'guid', $post_ID ); 
  398.  
  399.     // Set GUID. 
  400.     if ( ! $update && '' == $current_guid ) { 
  401.         $wpdb->update( $wpdb->posts, array( 'guid' => get_permalink( $post_ID ) ), $where ); 
  402.     } 
  403.  
  404.     if ( 'attachment' === $postarr['post_type'] ) { 
  405.         if ( ! empty( $postarr['file'] ) ) { 
  406.             update_attached_file( $post_ID, $postarr['file'] ); 
  407.         } 
  408.  
  409.         if ( ! empty( $postarr['context'] ) ) { 
  410.             add_post_meta( $post_ID, '_wp_attachment_context', $postarr['context'], true ); 
  411.         } 
  412.     } 
  413.  
  414.     // Set or remove featured image. 
  415.     if ( isset( $postarr['_thumbnail_id'] ) ) { 
  416.         $thumbnail_support = current_theme_supports( 'post-thumbnails', $post_type ) && post_type_supports( $post_type, 'thumbnail' ) || 'revision' === $post_type; 
  417.         if ( ! $thumbnail_support && 'attachment' === $post_type && $post_mime_type ) { 
  418.             if ( wp_attachment_is( 'audio', $post_ID ) ) { 
  419.                 $thumbnail_support = post_type_supports( 'attachment:audio', 'thumbnail' ) || current_theme_supports( 'post-thumbnails', 'attachment:audio' ); 
  420.             } elseif ( wp_attachment_is( 'video', $post_ID ) ) { 
  421.                 $thumbnail_support = post_type_supports( 'attachment:video', 'thumbnail' ) || current_theme_supports( 'post-thumbnails', 'attachment:video' ); 
  422.             } 
  423.         } 
  424.  
  425.         if ( $thumbnail_support ) { 
  426.             $thumbnail_id = intval( $postarr['_thumbnail_id'] ); 
  427.             if ( -1 === $thumbnail_id ) { 
  428.                 delete_post_thumbnail( $post_ID ); 
  429.             } else { 
  430.                 set_post_thumbnail( $post_ID, $thumbnail_id ); 
  431.             } 
  432.         } 
  433.     } 
  434.  
  435.     clean_post_cache( $post_ID ); 
  436.  
  437.     $post = get_post( $post_ID ); 
  438.  
  439.     if ( ! empty( $postarr['page_template'] ) ) { 
  440.         $post->page_template = $postarr['page_template']; 
  441.         $page_templates = wp_get_theme()->get_page_templates( $post ); 
  442.         if ( 'default' != $postarr['page_template'] && ! isset( $page_templates[ $postarr['page_template'] ] ) ) { 
  443.             if ( $wp_error ) { 
  444.                 return new WP_Error( 'invalid_page_template', __( 'Invalid page template.' ) ); 
  445.             } 
  446.             update_post_meta( $post_ID, '_wp_page_template', 'default' ); 
  447.         } else { 
  448.             update_post_meta( $post_ID, '_wp_page_template', $postarr['page_template'] ); 
  449.         } 
  450.     } 
  451.  
  452.     if ( 'attachment' !== $postarr['post_type'] ) { 
  453.         wp_transition_post_status( $data['post_status'], $previous_status, $post ); 
  454.     } else { 
  455.         if ( $update ) { 
  456.             /** 
  457.              * Fires once an existing attachment has been updated. 
  458.              * 
  459.              * @since 2.0.0 
  460.              * 
  461.              * @param int $post_ID Attachment ID. 
  462.              */ 
  463.             do_action( 'edit_attachment', $post_ID ); 
  464.             $post_after = get_post( $post_ID ); 
  465.  
  466.             /** 
  467.              * Fires once an existing attachment has been updated. 
  468.              * 
  469.              * @since 4.4.0 
  470.              * 
  471.              * @param int     $post_ID      Post ID. 
  472.              * @param WP_Post $post_after   Post object following the update. 
  473.              * @param WP_Post $post_before  Post object before the update. 
  474.              */ 
  475.             do_action( 'attachment_updated', $post_ID, $post_after, $post_before ); 
  476.         } else { 
  477.  
  478.             /** 
  479.              * Fires once an attachment has been added. 
  480.              * 
  481.              * @since 2.0.0 
  482.              * 
  483.              * @param int $post_ID Attachment ID. 
  484.              */ 
  485.             do_action( 'add_attachment', $post_ID ); 
  486.         } 
  487.  
  488.         return $post_ID; 
  489.     } 
  490.  
  491.     if ( $update ) { 
  492.         /** 
  493.          * Fires once an existing post has been updated. 
  494.          * 
  495.          * @since 1.2.0 
  496.          * 
  497.          * @param int     $post_ID Post ID. 
  498.          * @param WP_Post $post    Post object. 
  499.          */ 
  500.         do_action( 'edit_post', $post_ID, $post ); 
  501.         $post_after = get_post($post_ID); 
  502.  
  503.         /** 
  504.          * Fires once an existing post has been updated. 
  505.          * 
  506.          * @since 3.0.0 
  507.          * 
  508.          * @param int     $post_ID      Post ID. 
  509.          * @param WP_Post $post_after   Post object following the update. 
  510.          * @param WP_Post $post_before  Post object before the update. 
  511.          */ 
  512.         do_action( 'post_updated', $post_ID, $post_after, $post_before); 
  513.     } 
  514.  
  515.     /** 
  516.      * Fires once a post has been saved. 
  517.      * 
  518.      * The dynamic portion of the hook name, `$post->post_type`, refers to 
  519.      * the post type slug. 
  520.      * 
  521.      * @since 3.7.0 
  522.      * 
  523.      * @param int     $post_ID Post ID. 
  524.      * @param WP_Post $post    Post object. 
  525.      * @param bool    $update  Whether this is an existing post being updated or not. 
  526.      */ 
  527.     do_action( "save_post_{$post->post_type}", $post_ID, $post, $update ); 
  528.  
  529.     /** 
  530.      * Fires once a post has been saved. 
  531.      * 
  532.      * @since 1.5.0 
  533.      * 
  534.      * @param int     $post_ID Post ID. 
  535.      * @param WP_Post $post    Post object. 
  536.      * @param bool    $update  Whether this is an existing post being updated or not. 
  537.      */ 
  538.     do_action( 'save_post', $post_ID, $post, $update ); 
  539.  
  540.     /** 
  541.      * Fires once a post has been saved. 
  542.      * 
  543.      * @since 2.0.0 
  544.      * 
  545.      * @param int     $post_ID Post ID. 
  546.      * @param WP_Post $post    Post object. 
  547.      * @param bool    $update  Whether this is an existing post being updated or not. 
  548.      */ 
  549.     do_action( 'wp_insert_post', $post_ID, $post, $update ); 
  550.  
  551.     return $post_ID; 
  552. }
