wp_get_document_title

Returns document title for the current page.

Description

(string) wp_get_document_title(); 

Returns (string)

Tag with the document title.

Usage

  1. if ( !function_exists( 'wp_get_document_title' ) ) { 
  2.     require_once ABSPATH . WPINC . '/general-template.php'; 
  3. } 
  4.  
  5.   
  6. // NOTICE! Understand what this does before running. 
  7. $result = wp_get_document_title(); 
  8.

  1. function wp_get_document_title() { 
  2.  
  3.     /** 
  4.      * Filters the document title before it is generated. 
  5.      * 
  6.      * Passing a non-empty value will short-circuit wp_get_document_title(),  
  7.      * returning that value instead. 
  8.      * 
  9.      * @since 4.4.0 
  10.      * 
  11.      * @param string $title The document title. Default empty string. 
  12.      */ 
  13.     $title = apply_filters( 'pre_get_document_title', '' ); 
  14.     if ( ! empty( $title ) ) { 
  15.         return $title; 
  16.     } 
  17.  
  18.     global $page, $paged; 
  19.  
  20.     $title = array( 
  21.         'title' => '',  
  22.  ); 
  23.  
  24.     // If it's a 404 page, use a "Page not found" title. 
  25.     if ( is_404() ) { 
  26.         $title['title'] = __( 'Page not found' ); 
  27.  
  28.     // If it's a search, use a dynamic search results title. 
  29.     } elseif ( is_search() ) { 
  30.         /** translators: %s: search phrase */ 
  31.         $title['title'] = sprintf( __( 'Search Results for “%s”' ), get_search_query() ); 
  32.  
  33.     // If on the front page, use the site title. 
  34.     } elseif ( is_front_page() ) { 
  35.         $title['title'] = get_bloginfo( 'name', 'display' ); 
  36.  
  37.     // If on a post type archive, use the post type archive title. 
  38.     } elseif ( is_post_type_archive() ) { 
  39.         $title['title'] = post_type_archive_title( '', false ); 
  40.  
  41.     // If on a taxonomy archive, use the term title. 
  42.     } elseif ( is_tax() ) { 
  43.         $title['title'] = single_term_title( '', false ); 
  44.  
  45.     /** 
  46.      * If we're on the blog page that is not the homepage or 
  47.      * a single post of any post type, use the post title. 
  48.      */ 
  49.     } elseif ( is_home() || is_singular() ) { 
  50.         $title['title'] = single_post_title( '', false ); 
  51.  
  52.     // If on a category or tag archive, use the term title. 
  53.     } elseif ( is_category() || is_tag() ) { 
  54.         $title['title'] = single_term_title( '', false ); 
  55.  
  56.     // If on an author archive, use the author's display name. 
  57.     } elseif ( is_author() && $author = get_queried_object() ) { 
  58.         $title['title'] = $author->display_name; 
  59.  
  60.     // If it's a date archive, use the date as the title. 
  61.     } elseif ( is_year() ) { 
  62.         $title['title'] = get_the_date( _x( 'Y', 'yearly archives date format' ) ); 
  63.  
  64.     } elseif ( is_month() ) { 
  65.         $title['title'] = get_the_date( _x( 'F Y', 'monthly archives date format' ) ); 
  66.  
  67.     } elseif ( is_day() ) { 
  68.         $title['title'] = get_the_date(); 
  69.     } 
  70.  
  71.     // Add a page number if necessary. 
  72.     if ( ( $paged >= 2 || $page >= 2 ) && ! is_404() ) { 
  73.         $title['page'] = sprintf( __( 'Page %s' ), max( $paged, $page ) ); 
  74.     } 
  75.  
  76.     // Append the description or site title to give context. 
  77.     if ( is_front_page() ) { 
  78.         $title['tagline'] = get_bloginfo( 'description', 'display' ); 
  79.     } else { 
  80.         $title['site'] = get_bloginfo( 'name', 'display' ); 
  81.     } 
  82.  
  83.     /** 
  84.      * Filters the separator for the document title. 
  85.      * 
  86.      * @since 4.4.0 
  87.      * 
  88.      * @param string $sep Document title separator. Default '-'. 
  89.      */ 
  90.     $sep = apply_filters( 'document_title_separator', '-' ); 
  91.  
  92.     /** 
  93.      * Filters the parts of the document title. 
  94.      * 
  95.      * @since 4.4.0 
  96.      * 
  97.      * @param array $title { 
  98.      *     The document title parts. 
  99.      * 
  100.      *     @type string $title   Title of the viewed page. 
  101.      *     @type string $page    Optional. Page number if paginated. 
  102.      *     @type string $tagline Optional. Site description when on home page. 
  103.      *     @type string $site    Optional. Site title when not on home page. 
  104.      * } 
  105.      */ 
  106.     $title = apply_filters( 'document_title_parts', $title ); 
  107.  
  108.     $title = implode( " $sep ", array_filter( $title ) ); 
  109.     $title = wptexturize( $title ); 
  110.     $title = convert_chars( $title ); 
  111.     $title = esc_html( $title ); 
  112.     $title = capital_P_dangit( $title ); 
  113.  
  114.     return $title; 
  115. }
