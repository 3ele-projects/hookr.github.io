  669.          * children_pages is two dimensional array, eg. 
  670.          * children_pages[10][] contains all sub-pages whose parent is 10. 
  671.          * It only takes O( N ) to arrange this and it takes O( 1 ) for subsequent lookup operations 
  672.          * If searching, ignore hierarchy and treat everything as top level

  674.         if ( empty( $_REQUEST['s'] ) ) {

  686.                 }

  690.                 else 
  691.                     $children_pages[ $page->post_parent ][] = $page; 
  692.             }

  695.         }

  699.         $end = $start + $per_page; 
  700.         $to_display = array(); 
  701.  
  702.         foreach ( $pages as $page ) { 
  703.             if ( $count >= $end ) 
  704.                 break;

  719.                 foreach ( $orphans as $op ) {

  721.                         break; 
  722.  
  723.                     if ( $count >= $start ) { 
  724.                         $to_display[$op->ID] = 0; 
  725.                     } 
  726.  
  727.                     $count++; 
  728.                 } 
  729.             } 
  730.         }

  732.         $ids = array_keys( $to_display ); 
  733.         _prime_post_caches( $ids );

  735.         if ( ! isset( $GLOBALS['post'] ) ) { 
  736.             $GLOBALS['post'] = reset( $ids );

  745.     /** 
  746.      * Given a top level page ID, display the nested hierarchy of sub-pages 
  747.      * together with paging support 
  748.      * 
  749.      * @since 3.1.0 (Standalone function exists since 2.6.0) 
  750.      * @since 4.2.0 Added the `$to_display` parameter. 
  751.      * 
  752.      * @param array $children_pages 
  753.      * @param int $count 
  754.      * @param int $parent 
  755.      * @param int $level 
  756.      * @param int $pagenum 
  757.      * @param int $per_page 
  758.      * @param array $to_display List of pages to be displayed. Passed by reference. 
  759.      */ 
  760.     private function _page_rows( &$children_pages, &$count, $parent, $level, $pagenum, $per_page, &$to_display ) { 
  761.         if ( ! isset( $children_pages[$parent] ) ) 
  762.             return; 
  763.  
  764.         $start = ( $pagenum - 1 ) * $per_page; 
  765.         $end = $start + $per_page; 
  766.  
  767.         foreach ( $children_pages[$parent] as $page ) { 
  768.             if ( $count >= $end ) 
  769.                 break; 
  770.  
  771.             // If the page starts in a subtree, print the parents. 
  772.             if ( $count == $start && $page->post_parent > 0 ) { 
  773.                 $my_parents = array(); 
  774.                 $my_parent = $page->post_parent; 
  775.                 while ( $my_parent ) { 
  776.                     // Get the ID from the list or the attribute if my_parent is an object 
  777.                     $parent_id = $my_parent; 
  778.                     if ( is_object( $my_parent ) ) { 
  779.                         $parent_id = $my_parent->ID; 
  780.                     } 
  781.  
  782.                     $my_parent = get_post( $parent_id ); 
  783.                     $my_parents[] = $my_parent; 
  784.                     if ( !$my_parent->post_parent ) 
  785.                         break; 
  786.                     $my_parent = $my_parent->post_parent; 
  787.                 } 
  788.                 $num_parents = count( $my_parents ); 
  789.                 while ( $my_parent = array_pop( $my_parents ) ) { 
  790.                     $to_display[$my_parent->ID] = $level - $num_parents; 
  791.                     $num_parents--; 
  792.                 } 
  793.             } 
  794.  
  795.             if ( $count >= $start ) { 
  796.                 $to_display[$page->ID] = $level; 
  797.             } 
  798.  
  799.             $count++; 
  800.  
  801.             $this->_page_rows( $children_pages, $count, $page->ID, $level + 1, $pagenum, $per_page, $to_display ); 
  802.         } 
  803.  
  804.         unset( $children_pages[$parent] ); //required in order to keep track of orphans 
  805.     }
