  215.      * Process changed lines to do word-by-word diffs for extra highlighting. 
  216.      * 
  217.      * (TRAC style) sometimes these lines can actually be deleted or added rows. 
  218.      * We do additional processing to figure that out 
  219.      * 
  220.      * @access public 
  221.      * @since 2.6.0 
  222.      * 
  223.      * @param array $orig 
  224.      * @param array $final 
  225.      * @return string 
  226.      */ 
  227.     public function _changed( $orig, $final ) { 
  228.         $r = ''; 
  229.  
  230.         // Does the aforementioned additional processing 
  231.         // *_matches tell what rows are "the same" in orig and final. Those pairs will be diffed to get word changes 
  232.         //    match is numeric: an index in other column 
  233.         //    match is 'X': no match. It is a new row 
  234.         // *_rows are column vectors for the orig column and the final column. 
  235.         //    row >= 0: an indix of the $orig or $final array 
  236.         //    row  < 0: a blank row for that column 
  237.         list($orig_matches, $final_matches, $orig_rows, $final_rows) = $this->interleave_changed_lines( $orig, $final ); 
  238.  
  239.         // These will hold the word changes as determined by an inline diff 
  240.         $orig_diffs = array(); 
  241.         $final_diffs = array(); 
  242.  
  243.         // Compute word diffs for each matched pair using the inline diff 
  244.         foreach ( $orig_matches as $o => $f ) { 
  245.             if ( is_numeric($o) && is_numeric($f) ) { 
  246.                 $text_diff = new Text_Diff( 'auto', array( array($orig[$o]), array($final[$f]) ) ); 
  247.                 $renderer = new $this->inline_diff_renderer; 
  248.                 $diff = $renderer->render( $text_diff ); 
  249.  
  250.                 // If they're too different, don't include any <ins> or <dels> 
  251.                 if ( preg_match_all( '!(<ins>.*?</ins>|<del>.*?</del>)!', $diff, $diff_matches ) ) { 
  252.                     // length of all text between <ins> or <del> 
  253.                     $stripped_matches = strlen(strip_tags( join(' ', $diff_matches[0]) )); 
  254.                     // since we count lengith of text between <ins> or <del> (instead of picking just one),  
  255.                     //    we double the length of chars not in those tags. 
  256.                     $stripped_diff = strlen(strip_tags( $diff )) * 2 - $stripped_matches; 
  257.                     $diff_ratio = $stripped_matches / $stripped_diff; 
  258.                     if ( $diff_ratio > $this->_diff_threshold ) 
  259.                         continue; // Too different. Don't save diffs. 
  260.                 } 
  261.  
  262.                 // Un-inline the diffs by removing del or ins 
  263.                 $orig_diffs[$o] = preg_replace( '|<ins>.*?</ins>|', '', $diff ); 
  264.                 $final_diffs[$f] = preg_replace( '|<del>.*?</del>|', '', $diff ); 
  265.             } 
  266.         } 
  267.  
  268.         foreach ( array_keys($orig_rows) as $row ) { 
  269.             // Both columns have blanks. Ignore them. 
  270.             if ( $orig_rows[$row] < 0 && $final_rows[$row] < 0 ) 
  271.                 continue; 
  272.  
  273.             // If we have a word based diff, use it. Otherwise, use the normal line. 
  274.             if ( isset( $orig_diffs[$orig_rows[$row]] ) ) 
  275.                 $orig_line = $orig_diffs[$orig_rows[$row]]; 
  276.             elseif ( isset( $orig[$orig_rows[$row]] ) ) 
  277.                 $orig_line = htmlspecialchars($orig[$orig_rows[$row]]); 
  278.             else 
  279.                 $orig_line = ''; 
  280.  
  281.             if ( isset( $final_diffs[$final_rows[$row]] ) ) 
  282.                 $final_line = $final_diffs[$final_rows[$row]]; 
  283.             elseif ( isset( $final[$final_rows[$row]] ) ) 
  284.                 $final_line = htmlspecialchars($final[$final_rows[$row]]); 
  285.             else 
  286.                 $final_line = ''; 
  287.  
  288.             if ( $orig_rows[$row] < 0 ) { // Orig is blank. This is really an added row. 
  289.                 $r .= $this->_added( array($final_line), false ); 
  290.             } elseif ( $final_rows[$row] < 0 ) { // Final is blank. This is really a deleted row. 
  291.                 $r .= $this->_deleted( array($orig_line), false ); 
  292.             } else { // A true changed row. 
  293.                 if ( $this->_show_split_view ) { 
  294.                     $r .= '<tr>' . $this->deletedLine( $orig_line ) . $this->emptyLine() . $this->addedLine( $final_line ) . "</tr>\n"; 
  295.                 } else { 
  296.                     $r .= '<tr>' . $this->deletedLine( $orig_line ) . "</tr><tr>" . $this->addedLine( $final_line ) . "</tr>\n"; 
  297.                 } 
  298.             } 
  299.         } 
  300.  
  301.         return $r; 
  302.     }

  304.     /** 
  305.      * Takes changed blocks and matches which rows in orig turned into which rows in final. 
  306.      * 
  307.      * Returns 
  308.      *    *_matches ( which rows match with which ) 
  309.      *    *_rows ( order of rows in each column interleaved with blank rows as 
  310.      *        necessary ) 
  311.      * 
  312.      * @since 2.6.0 
  313.      * 
  314.      * @param array $orig 
  315.      * @param array $final 
  316.      * @return array 
  317.      */ 
  318.     public function interleave_changed_lines( $orig, $final ) { 
  319.  
  320.         // Contains all pairwise string comparisons. Keys are such that this need only be a one dimensional array. 
  321.         $matches = array(); 
  322.         foreach ( array_keys($orig) as $o ) { 
  323.             foreach ( array_keys($final) as $f ) { 
  324.                 $matches["$o, $f"] = $this->compute_string_distance( $orig[$o], $final[$f] ); 
  325.             } 
  326.         } 
  327.         asort($matches); // Order by string distance. 
  328.  
  329.         $orig_matches = array(); 
  330.         $final_matches = array(); 
  331.  
  332.         foreach ( $matches as $keys => $difference ) { 
  333.             list($o, $f) = explode(', ', $keys); 
  334.             $o = (int) $o; 
  335.             $f = (int) $f; 
  336.  
  337.             // Already have better matches for these guys 
  338.             if ( isset($orig_matches[$o]) && isset($final_matches[$f]) ) 
  339.                 continue; 
  340.  
  341.             // First match for these guys. Must be best match 
  342.             if ( !isset($orig_matches[$o]) && !isset($final_matches[$f]) ) { 
  343.                 $orig_matches[$o] = $f; 
  344.                 $final_matches[$f] = $o; 
  345.                 continue; 
  346.             } 
  347.  
  348.             // Best match of this final is already taken?  Must mean this final is a new row. 
  349.             if ( isset($orig_matches[$o]) ) 
  350.                 $final_matches[$f] = 'x'; 
  351.  
  352.             // Best match of this orig is already taken?  Must mean this orig is a deleted row. 
  353.             elseif ( isset($final_matches[$f]) ) 
  354.                 $orig_matches[$o] = 'x'; 
  355.         } 
  356.  
  357.         // We read the text in this order 
  358.         ksort($orig_matches); 
  359.         ksort($final_matches); 
  360.  
  361.         // Stores rows and blanks for each column. 
  362.         $orig_rows = $orig_rows_copy = array_keys($orig_matches); 
  363.         $final_rows = array_keys($final_matches); 
  364.  
  365.         // Interleaves rows with blanks to keep matches aligned. 
  366.         // We may end up with some extraneous blank rows, but we'll just ignore them later. 
  367.         foreach ( $orig_rows_copy as $orig_row ) { 
  368.             $final_pos = array_search($orig_matches[$orig_row], $final_rows, true); 
  369.             $orig_pos = (int) array_search($orig_row, $orig_rows, true); 
  370.  
  371.             if ( false === $final_pos ) { // This orig is paired with a blank final. 
  372.                 array_splice( $final_rows, $orig_pos, 0, -1 ); 
  373.             } elseif ( $final_pos < $orig_pos ) { // This orig's match is up a ways. Pad final with blank rows. 
  374.                 $diff_pos = $final_pos - $orig_pos; 
  375.                 while ( $diff_pos < 0 ) 
  376.                     array_splice( $final_rows, $orig_pos, 0, $diff_pos++ ); 
  377.             } elseif ( $final_pos > $orig_pos ) { // This orig's match is down a ways. Pad orig with blank rows. 
  378.                 $diff_pos = $orig_pos - $final_pos; 
  379.                 while ( $diff_pos < 0 ) 
  380.                     array_splice( $orig_rows, $orig_pos, 0, $diff_pos++ ); 
  381.             } 
  382.         } 
  383.  
  384.         // Pad the ends with blank rows if the columns aren't the same length 
  385.         $diff_count = count($orig_rows) - count($final_rows); 
  386.         if ( $diff_count < 0 ) { 
  387.             while ( $diff_count < 0 ) 
  388.                 array_push($orig_rows, $diff_count++); 
  389.         } elseif ( $diff_count > 0 ) { 
  390.             $diff_count = -1 * $diff_count; 
  391.             while ( $diff_count < 0 ) 
  392.                 array_push($final_rows, $diff_count++); 
  393.         } 
  394.  
  395.         return array($orig_matches, $final_matches, $orig_rows, $final_rows); 
  396.     }

  398.     /** 
  399.      * Computes a number that is intended to reflect the "distance" between two strings. 
  400.      * 
  401.      * @since 2.6.0 
  402.      * 
  403.      * @param string $string1 
  404.      * @param string $string2 
  405.      * @return int 
  406.      */ 
  407.     public function compute_string_distance( $string1, $string2 ) { 
  408.         // Vectors containing character frequency for all chars in each string 
  409.         $chars1 = count_chars($string1); 
  410.         $chars2 = count_chars($string2); 
  411.  
  412.         // L1-norm of difference vector. 
  413.         $difference = array_sum( array_map( array($this, 'difference'), $chars1, $chars2 ) ); 
  414.  
  415.         // $string1 has zero length? Odd. Give huge penalty by not dividing. 
  416.         if ( !$string1 ) 
  417.             return $difference; 
  418.  
  419.         // Return distance per character (of string1). 
  420.         return $difference / strlen($string1); 
  421.     }
