  92.     private function resort_active_iterations( $new_priority = false, $priority_existed = false ) { 
  93.         $new_priorities = array_keys( $this->callbacks ); 
  94.  
  95.         // If there are no remaining hooks, clear out all running iterations. 
  96.         if ( ! $new_priorities ) { 
  97.             foreach ( $this->iterations as $index => $iteration ) { 
  98.                 $this->iterations[ $index ] = $new_priorities; 
  99.             } 
  100.             return; 
  101.         } 
  102.  
  103.         $min = min( $new_priorities ); 
  104.         foreach ( $this->iterations as $index => &$iteration ) { 
  105.             $current = current( $iteration ); 
  106.             // If we're already at the end of this iteration, just leave the array pointer where it is. 
  107.             if ( false === $current ) { 
  108.                 continue; 
  109.             } 
  110.  
  111.             $iteration = $new_priorities; 
  112.  
  113.             if ( $current < $min ) { 
  114.                 array_unshift( $iteration, $current ); 
  115.                 continue; 
  116.             } 
  117.  
  118.             while ( current( $iteration ) < $current ) { 
  119.                 if ( false === next( $iteration ) ) { 
  120.                     break; 
  121.                 } 
  122.             } 
  123.  
  124.             // If we have a new priority that didn't exist, but ::apply_filters() or ::do_action() thinks it's the current priority... 
  125.             if ( $new_priority === $this->current_priority[ $index ] && ! $priority_existed ) { 
  126.                 /** 
  127.                  * ... and the new priority is the same as what $this->iterations thinks is the previous 
  128.                  * priority, we need to move back to it. 
  129.                  */ 
  130.  
  131.                 if ( false === current( $iteration ) ) { 
  132.                     // If we've already moved off the end of the array, go back to the last element. 
  133.                     $prev = end( $iteration ); 
  134.                 } else { 
  135.                     // Otherwise, just go back to the previous element. 
  136.                     $prev = prev( $iteration ); 
  137.                 } 
  138.                 if ( false === $prev ) { 
  139.                     // Start of the array. Reset, and go about our day. 
  140.                     reset( $iteration ); 
  141.                 } elseif ( $new_priority !== $prev ) { 
  142.                     // Previous wasn't the same. Move forward again. 
  143.                     next( $iteration ); 
  144.                 } 
  145.             } 
  146.         } 
  147.         unset( $iteration ); 
  148.     }

  259.     public function apply_filters( $value, $args ) { 
  260.         if ( ! $this->callbacks ) { 
  261.             return $value; 
  262.         } 
  263.  
  264.         $nesting_level = $this->nesting_level++; 
  265.  
  266.         $this->iterations[ $nesting_level ] = array_keys( $this->callbacks ); 
  267.         $num_args = count( $args ); 
  268.  
  269.         do { 
  270.             $this->current_priority[ $nesting_level ] = $priority = current( $this->iterations[ $nesting_level ] ); 
  271.  
  272.             foreach ( $this->callbacks[ $priority ] as $the_ ) { 
  273.                 if( ! $this->doing_action ) { 
  274.                     $args[ 0 ] = $value; 
  275.                 } 
  276.  
  277.                 // Avoid the array_slice if possible. 
  278.                 if ( $the_['accepted_args'] == 0 ) { 
  279.                     $value = call_user_func_array( $the_['function'], array() ); 
  280.                 } elseif ( $the_['accepted_args'] >= $num_args ) { 
  281.                     $value = call_user_func_array( $the_['function'], $args ); 
  282.                 } else { 
  283.                     $value = call_user_func_array( $the_['function'], array_slice( $args, 0, (int)$the_['accepted_args'] ) ); 
  284.                 } 
  285.             } 
  286.         } while ( false !== next( $this->iterations[ $nesting_level ] ) ); 
  287.  
  288.         unset( $this->iterations[ $nesting_level ] ); 
  289.         unset( $this->current_priority[ $nesting_level ] ); 
  290.  
  291.         $this->nesting_level--; 
  292.  
  293.         return $value;
