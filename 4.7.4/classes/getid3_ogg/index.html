  462.     public function ParseOggPageHeader() { 
  463.         // http://xiph.org/ogg/vorbis/doc/framing.html 
  464.         $oggheader['page_start_offset'] = $this->ftell(); // where we started from in the file 
  465.  
  466.         $filedata = $this->fread($this->getid3->fread_buffer_size()); 
  467.         $filedataoffset = 0; 
  468.         while ((substr($filedata, $filedataoffset++, 4) != 'OggS')) { 
  469.             if (($this->ftell() - $oggheader['page_start_offset']) >= $this->getid3->fread_buffer_size()) { 
  470.                 // should be found before here 
  471.                 return false; 
  472.             } 
  473.             if ((($filedataoffset + 28) > strlen($filedata)) || (strlen($filedata) < 28)) { 
  474.                 if ($this->feof() || (($filedata .= $this->fread($this->getid3->fread_buffer_size())) === false)) { 
  475.                     // get some more data, unless eof, in which case fail 
  476.                     return false; 
  477.                 } 
  478.             } 
  479.         } 
  480.         $filedataoffset += strlen('OggS') - 1; // page, delimited by 'OggS' 
  481.  
  482.         $oggheader['stream_structver'] = getid3_lib::LittleEndian2Int(substr($filedata, $filedataoffset, 1)); 
  483.         $filedataoffset += 1; 
  484.         $oggheader['flags_raw'] = getid3_lib::LittleEndian2Int(substr($filedata, $filedataoffset, 1)); 
  485.         $filedataoffset += 1; 
  486.         $oggheader['flags']['fresh'] = (bool) ($oggheader['flags_raw'] & 0x01); // fresh packet 
  487.         $oggheader['flags']['bos'] = (bool) ($oggheader['flags_raw'] & 0x02); // first page of logical bitstream (bos) 
  488.         $oggheader['flags']['eos'] = (bool) ($oggheader['flags_raw'] & 0x04); // last page of logical bitstream (eos) 
  489.  
  490.         $oggheader['pcm_abs_position'] = getid3_lib::LittleEndian2Int(substr($filedata, $filedataoffset, 8)); 
  491.         $filedataoffset += 8; 
  492.         $oggheader['stream_serialno'] = getid3_lib::LittleEndian2Int(substr($filedata, $filedataoffset, 4)); 
  493.         $filedataoffset += 4; 
  494.         $oggheader['page_seqno'] = getid3_lib::LittleEndian2Int(substr($filedata, $filedataoffset, 4)); 
  495.         $filedataoffset += 4; 
  496.         $oggheader['page_checksum'] = getid3_lib::LittleEndian2Int(substr($filedata, $filedataoffset, 4)); 
  497.         $filedataoffset += 4; 
  498.         $oggheader['page_segments'] = getid3_lib::LittleEndian2Int(substr($filedata, $filedataoffset, 1)); 
  499.         $filedataoffset += 1; 
  500.         $oggheader['page_length'] = 0; 
  501.         for ($i = 0; $i < $oggheader['page_segments']; $i++) { 
  502.             $oggheader['segment_table'][$i] = getid3_lib::LittleEndian2Int(substr($filedata, $filedataoffset, 1)); 
  503.             $filedataoffset += 1; 
  504.             $oggheader['page_length'] += $oggheader['segment_table'][$i]; 
  505.         } 
  506.         $oggheader['header_end_offset'] = $oggheader['page_start_offset'] + $filedataoffset; 
  507.         $oggheader['page_end_offset'] = $oggheader['header_end_offset'] + $oggheader['page_length']; 
  508.         $this->fseek($oggheader['header_end_offset']); 
  509.  
  510.         return $oggheader; 
  511.     }

  514.     public function ParseVorbisComments() { 
  515.         $info = &$this->getid3->info; 
  516.  
  517.         $OriginalOffset = $this->ftell(); 
  518.         $commentdataoffset = 0; 
  519.         $VorbisCommentPage = 1; 
  520.  
  521.         switch ($info['audio']['dataformat']) { 
  522.             case 'vorbis': 
  523.             case 'speex': 
  524.             case 'opus': 
  525.                 $CommentStartOffset = $info['ogg']['pageheader'][$VorbisCommentPage]['page_start_offset'];  // Second Ogg page, after header block 
  526.                 $this->fseek($CommentStartOffset); 
  527.                 $commentdataoffset = 27 + $info['ogg']['pageheader'][$VorbisCommentPage]['page_segments']; 
  528.                 $commentdata = $this->fread(self::OggPageSegmentLength($info['ogg']['pageheader'][$VorbisCommentPage], 1) + $commentdataoffset); 
  529.  
  530.                 if ($info['audio']['dataformat'] == 'vorbis') { 
  531.                     $commentdataoffset += (strlen('vorbis') + 1); 
  532.                 } 
  533.                 else if ($info['audio']['dataformat'] == 'opus') { 
  534.                     $commentdataoffset += strlen('OpusTags'); 
  535.                 } 
  536.  
  537.                 break; 
  538.  
  539.             case 'flac': 
  540.                 $CommentStartOffset = $info['flac']['VORBIS_COMMENT']['raw']['offset'] + 4; 
  541.                 $this->fseek($CommentStartOffset); 
  542.                 $commentdata = $this->fread($info['flac']['VORBIS_COMMENT']['raw']['block_length']); 
  543.                 break; 
  544.  
  545.             default: 
  546.                 return false; 
  547.         } 
  548.  
  549.         $VendorSize = getid3_lib::LittleEndian2Int(substr($commentdata, $commentdataoffset, 4)); 
  550.         $commentdataoffset += 4; 
  551.  
  552.         $info['ogg']['vendor'] = substr($commentdata, $commentdataoffset, $VendorSize); 
  553.         $commentdataoffset += $VendorSize; 
  554.  
  555.         $CommentsCount = getid3_lib::LittleEndian2Int(substr($commentdata, $commentdataoffset, 4)); 
  556.         $commentdataoffset += 4; 
  557.         $info['avdataoffset'] = $CommentStartOffset + $commentdataoffset; 
  558.  
  559.         $basicfields = array('TITLE', 'ARTIST', 'ALBUM', 'TRACKNUMBER', 'GENRE', 'DATE', 'DESCRIPTION', 'COMMENT'); 
  560.         $ThisFileInfo_ogg_comments_raw = &$info['ogg']['comments_raw']; 
  561.         for ($i = 0; $i < $CommentsCount; $i++) { 
  562.  
  563.             if ($i >= 10000) { 
  564.                 // https://github.com/owncloud/music/issues/212#issuecomment-43082336 
  565.                 $info['warning'][] = 'Unexpectedly large number ('.$CommentsCount.') of Ogg comments - breaking after reading '.$i.' comments'; 
  566.                 break; 
  567.             } 
  568.  
  569.             $ThisFileInfo_ogg_comments_raw[$i]['dataoffset'] = $CommentStartOffset + $commentdataoffset; 
  570.  
  571.             if ($this->ftell() < ($ThisFileInfo_ogg_comments_raw[$i]['dataoffset'] + 4)) { 
  572.                 if ($oggpageinfo = $this->ParseOggPageHeader()) { 
  573.                     $info['ogg']['pageheader'][$oggpageinfo['page_seqno']] = $oggpageinfo; 
  574.  
  575.                     $VorbisCommentPage++; 
  576.  
  577.                     // First, save what we haven't read yet 
  578.                     $AsYetUnusedData = substr($commentdata, $commentdataoffset); 
  579.  
  580.                     // Then take that data off the end 
  581.                     $commentdata = substr($commentdata, 0, $commentdataoffset); 
  582.  
  583.                     // Add [headerlength] bytes of dummy data for the Ogg Page Header, just to keep absolute offsets correct 
  584.                     $commentdata .= str_repeat("\x00", 27 + $info['ogg']['pageheader'][$oggpageinfo['page_seqno']]['page_segments']); 
  585.                     $commentdataoffset += (27 + $info['ogg']['pageheader'][$oggpageinfo['page_seqno']]['page_segments']); 
  586.  
  587.                     // Finally, stick the unused data back on the end 
  588.                     $commentdata .= $AsYetUnusedData; 
  589.  
  590.                     //$commentdata .= $this->fread($info['ogg']['pageheader'][$oggpageinfo['page_seqno']]['page_length']); 
  591.                     $commentdata .= $this->fread($this->OggPageSegmentLength($info['ogg']['pageheader'][$VorbisCommentPage], 1)); 
  592.                 } 
  593.  
  594.             } 
  595.             $ThisFileInfo_ogg_comments_raw[$i]['size'] = getid3_lib::LittleEndian2Int(substr($commentdata, $commentdataoffset, 4)); 
  596.  
  597.             // replace avdataoffset with position just after the last vorbiscomment 
  598.             $info['avdataoffset'] = $ThisFileInfo_ogg_comments_raw[$i]['dataoffset'] + $ThisFileInfo_ogg_comments_raw[$i]['size'] + 4; 
  599.  
  600.             $commentdataoffset += 4; 
  601.             while ((strlen($commentdata) - $commentdataoffset) < $ThisFileInfo_ogg_comments_raw[$i]['size']) { 
  602.                 if (($ThisFileInfo_ogg_comments_raw[$i]['size'] > $info['avdataend']) || ($ThisFileInfo_ogg_comments_raw[$i]['size'] < 0)) { 
  603.                     $info['warning'][] = 'Invalid Ogg comment size (comment #'.$i.', claims to be '.number_format($ThisFileInfo_ogg_comments_raw[$i]['size']).' bytes) - aborting reading comments'; 
  604.                     break 2; 
  605.                 } 
  606.  
  607.                 $VorbisCommentPage++; 
  608.  
  609.                 $oggpageinfo = $this->ParseOggPageHeader(); 
  610.                 $info['ogg']['pageheader'][$oggpageinfo['page_seqno']] = $oggpageinfo; 
  611.  
  612.                 // First, save what we haven't read yet 
  613.                 $AsYetUnusedData = substr($commentdata, $commentdataoffset); 
  614.  
  615.                 // Then take that data off the end 
  616.                 $commentdata = substr($commentdata, 0, $commentdataoffset); 
  617.  
  618.                 // Add [headerlength] bytes of dummy data for the Ogg Page Header, just to keep absolute offsets correct 
  619.                 $commentdata .= str_repeat("\x00", 27 + $info['ogg']['pageheader'][$oggpageinfo['page_seqno']]['page_segments']); 
  620.                 $commentdataoffset += (27 + $info['ogg']['pageheader'][$oggpageinfo['page_seqno']]['page_segments']); 
  621.  
  622.                 // Finally, stick the unused data back on the end 
  623.                 $commentdata .= $AsYetUnusedData; 
  624.  
  625.                 //$commentdata .= $this->fread($info['ogg']['pageheader'][$oggpageinfo['page_seqno']]['page_length']); 
  626.                 if (!isset($info['ogg']['pageheader'][$VorbisCommentPage])) { 
  627.                     $info['warning'][] = 'undefined Vorbis Comment page "'.$VorbisCommentPage.'" at offset '.$this->ftell(); 
  628.                     break; 
  629.                 } 
  630.                 $readlength = self::OggPageSegmentLength($info['ogg']['pageheader'][$VorbisCommentPage], 1); 
  631.                 if ($readlength <= 0) { 
  632.                     $info['warning'][] = 'invalid length Vorbis Comment page "'.$VorbisCommentPage.'" at offset '.$this->ftell(); 
  633.                     break; 
  634.                 } 
  635.                 $commentdata .= $this->fread($readlength); 
  636.  
  637.                 //$filebaseoffset += $oggpageinfo['header_end_offset'] - $oggpageinfo['page_start_offset']; 
  638.             } 
  639.             $ThisFileInfo_ogg_comments_raw[$i]['offset'] = $commentdataoffset; 
  640.             $commentstring = substr($commentdata, $commentdataoffset, $ThisFileInfo_ogg_comments_raw[$i]['size']); 
  641.             $commentdataoffset += $ThisFileInfo_ogg_comments_raw[$i]['size']; 
  642.  
  643.             if (!$commentstring) { 
  644.  
  645.                 // no comment? 
  646.                 $info['warning'][] = 'Blank Ogg comment ['.$i.']'; 
  647.  
  648.             } elseif (strstr($commentstring, '=')) { 
  649.  
  650.                 $commentexploded = explode('=', $commentstring, 2); 
  651.                 $ThisFileInfo_ogg_comments_raw[$i]['key'] = strtoupper($commentexploded[0]); 
  652.                 $ThisFileInfo_ogg_comments_raw[$i]['value'] = (isset($commentexploded[1]) ? $commentexploded[1] : '');

  657.                     // The unencoded format is that of the FLAC picture block. The fields are stored in big endian order as in FLAC, picture data is stored according to the relevant standard. 
  658.                     // http://flac.sourceforge.net/format.html#metadata_block_picture

  688.                 } else {

  690.                     $info['ogg']['comments'][strtolower($ThisFileInfo_ogg_comments_raw[$i]['key'])][] = $ThisFileInfo_ogg_comments_raw[$i]['value'];

  692.                 } 
  693.  
  694.             } else {

  697.  
  698.             } 
  699.             unset($ThisFileInfo_ogg_comments_raw[$i]); 
  700.         } 
  701.         unset($ThisFileInfo_ogg_comments_raw); 
  702.

  705.         // http://privatewww.essex.ac.uk/~djmrob/replaygain/

  710.                     case 'replaygain_album_gain':

  713.                         break; 
  714.  
  715.                     case 'rg_radio': 
  716.                     case 'replaygain_track_gain': 
  717.                         $info['replay_gain']['track']['adjustment'] = (double) $commentvalue[0];

  719.                         break; 
  720.  
  721.                     case 'replaygain_album_peak': 
  722.                         $info['replay_gain']['album']['peak'] = (double) $commentvalue[0];

  724.                         break; 
  725.  
  726.                     case 'rg_peak': 
  727.                     case 'replaygain_track_peak':

  731.  
  732.                     case 'replaygain_reference_loudness':

  737.                     default:

  741.             }

  774.     public static function get_quality_from_nominal_bitrate($nominal_bitrate) { 
  775.  
  776.         // decrease precision 
  777.         $nominal_bitrate = $nominal_bitrate / 1000; 
  778.  
  779.         if ($nominal_bitrate < 128) { 
  780.             // q-1 to q4 
  781.             $qval = ($nominal_bitrate - 64) / 16; 
  782.         } elseif ($nominal_bitrate < 256) { 
  783.             // q4 to q8 
  784.             $qval = $nominal_bitrate / 32; 
  785.         } elseif ($nominal_bitrate < 320) { 
  786.             // q8 to q9 
  787.             $qval = ($nominal_bitrate + 256) / 64; 
  788.         } else { 
  789.             // q9 to q10
