  591.      * @since 4.4.0 
  592.      * @access public 
  593.      * 
  594.      * @param WP_REST_Response $response Response object. 
  595.      * @param bool             $embed    Whether links should be embedded. 
  596.      * @return WP_REST_Response New response with wrapped data 
  597.      */ 
  598.     public function envelope_response( $response, $embed ) {

  602.             'headers' => $response->get_headers(),  
  603.  );

  611.          * @param WP_REST_Response $response Original response data. 
  612.          */ 
  613.         $envelope = apply_filters( 'rest_envelope_response', $envelope, $response );

  631.     public function register_route( $namespace, $route, $route_args, $override = false ) {

  662.     /** 
  663.      * Retrieves the route map. 
  664.      * 
  665.      * The route map is an associative array with path regexes as the keys. The 
  666.      * value is an indexed array with the callback function/method as the first 
  667.      * item, and a bitmask of HTTP methods as the second item (see the class 
  668.      * constants). 
  669.      * 
  670.      * Each route can be mapped to more than one callback by using an array of 
  671.      * the indexed arrays. This allows mapping e.g. GET requests to one callback 
  672.      * and POST requests to another. 
  673.      * 
  674.      * Note that the path regexes (array keys) must have @ escaped, as this is 
  675.      * used as the delimiter with preg_match() 
  676.      * 
  677.      * @since 4.4.0 
  678.      * @access public 
  679.      * 
  680.      * @return array `'/path/regex' => array( $callback, $bitmask )` or 
  681.      *               `'/path/regex' => array( array( $callback, $bitmask ), ...)`. 
  682.      */ 
  683.     public function get_routes() { 
  684.  
  685.         /** 
  686.          * Filters the array of available endpoints. 
  687.          * 
  688.          * @since 4.4.0 
  689.          * 
  690.          * @param array $endpoints The available endpoints. An array of matching regex patterns, each mapped 
  691.          *                         to an array of callbacks for the endpoint. These take the format 
  692.          *                         `'/path/regex' => array( $callback, $bitmask )` or 
  693.          *                         `'/path/regex' => array( array( $callback, $bitmask ). 
  694.          */ 
  695.         $endpoints = apply_filters( 'rest_endpoints', $this->endpoints ); 
  696.  
  697.         // Normalise the endpoints. 
  698.         $defaults = array( 
  699.             'methods' => '',  
  700.             'accept_json' => false,  
  701.             'accept_raw' => false,  
  702.             'show_in_index' => true,  
  703.             'args' => array(),  
  704.  ); 
  705.  
  706.         foreach ( $endpoints as $route => &$handlers ) { 
  707.  
  708.             if ( isset( $handlers['callback'] ) ) { 
  709.                 // Single endpoint, add one deeper. 
  710.                 $handlers = array( $handlers ); 
  711.             } 
  712.  
  713.             if ( ! isset( $this->route_options[ $route ] ) ) { 
  714.                 $this->route_options[ $route ] = array(); 
  715.             } 
  716.  
  717.             foreach ( $handlers as $key => &$handler ) { 
  718.  
  719.                 if ( ! is_numeric( $key ) ) { 
  720.                     // Route option, move it to the options. 
  721.                     $this->route_options[ $route ][ $key ] = $handler; 
  722.                     unset( $handlers[ $key ] ); 
  723.                     continue; 
  724.                 } 
  725.  
  726.                 $handler = wp_parse_args( $handler, $defaults ); 
  727.  
  728.                 // Allow comma-separated HTTP methods. 
  729.                 if ( is_string( $handler['methods'] ) ) { 
  730.                     $methods = explode( ', ', $handler['methods'] ); 
  731.                 } elseif ( is_array( $handler['methods'] ) ) { 
  732.                     $methods = $handler['methods']; 
  733.                 } else { 
  734.                     $methods = array(); 
  735.                 } 
  736.  
  737.                 $handler['methods'] = array(); 
  738.  
  739.                 foreach ( $methods as $method ) { 
  740.                     $method = strtoupper( trim( $method ) ); 
  741.                     $handler['methods'][ $method ] = true; 
  742.                 } 
  743.             } 
  744.         } 
  745.  
  746.         return $endpoints; 
  747.     }

  778.     /** 
  779.      * Matches the request to a callback and call it. 
  780.      * 
  781.      * @since 4.4.0 
  782.      * @access public 
  783.      * 
  784.      * @param WP_REST_Request $request Request to attempt dispatching. 
  785.      * @return WP_REST_Response Response returned by the callback. 
  786.      */ 
  787.     public function dispatch( $request ) { 
  788.         /** 
  789.          * Filters the pre-calculated result of a REST dispatch request. 
  790.          * 
  791.          * Allow hijacking the request before dispatching by returning a non-empty. The returned value 
  792.          * will be used to serve the request instead. 
  793.          * 
  794.          * @since 4.4.0 
  795.          * 
  796.          * @param mixed           $result  Response to replace the requested version with. Can be anything 
  797.          *                                 a normal endpoint can return, or null to not hijack the request. 
  798.          * @param WP_REST_Server  $this    Server instance. 
  799.          * @param WP_REST_Request $request Request used to generate the response. 
  800.          */ 
  801.         $result = apply_filters( 'rest_pre_dispatch', null, $this, $request ); 
  802.  
  803.         if ( ! empty( $result ) ) { 
  804.             return $result; 
  805.         } 
  806.  
  807.         $method = $request->get_method(); 
  808.         $path = $request->get_route(); 
  809.  
  810.         foreach ( $this->get_routes() as $route => $handlers ) { 
  811.             $match = preg_match( '@^' . $route . '$@i', $path, $args ); 
  812.  
  813.             if ( ! $match ) { 
  814.                 continue; 
  815.             } 
  816.  
  817.             foreach ( $handlers as $handler ) { 
  818.                 $callback = $handler['callback']; 
  819.                 $response = null; 
  820.  
  821.                 // Fallback to GET method if no HEAD method is registered. 
  822.                 $checked_method = $method; 
  823.                 if ( 'HEAD' === $method && empty( $handler['methods']['HEAD'] ) ) { 
  824.                     $checked_method = 'GET'; 
  825.                 } 
  826.                 if ( empty( $handler['methods'][ $checked_method ] ) ) { 
  827.                     continue; 
  828.                 } 
  829.  
  830.                 if ( ! is_callable( $callback ) ) { 
  831.                     $response = new WP_Error( 'rest_invalid_handler', __( 'The handler for the route is invalid' ), array( 'status' => 500 ) ); 
  832.                 } 
  833.  
  834.                 if ( ! is_wp_error( $response ) ) { 
  835.                     // Remove the redundant preg_match argument. 
  836.                     unset( $args[0] ); 
  837.  
  838.                     $request->set_url_params( $args ); 
  839.                     $request->set_attributes( $handler ); 
  840.  
  841.                     $defaults = array(); 
  842.  
  843.                     foreach ( $handler['args'] as $arg => $options ) { 
  844.                         if ( isset( $options['default'] ) ) { 
  845.                             $defaults[ $arg ] = $options['default']; 
  846.                         } 
  847.                     } 
  848.  
  849.                     $request->set_default_params( $defaults ); 
  850.  
  851.                     $check_required = $request->has_valid_params(); 
  852.                     if ( is_wp_error( $check_required ) ) { 
  853.                         $response = $check_required; 
  854.                     } else { 
  855.                         $check_sanitized = $request->sanitize_params(); 
  856.                         if ( is_wp_error( $check_sanitized ) ) { 
  857.                             $response = $check_sanitized; 
  858.                         } 
  859.                     } 
  860.                 } 
  861.  
  862.                 /** 
  863.                  * Filters the response before executing any REST API callbacks. 
  864.                  * 
  865.                  * Allows plugins to perform additional validation after a 
  866.                  * request is initialized and matched to a registered route,  
  867.                  * but before it is executed. 
  868.                  * 
  869.                  * Note that this filter will not be called for requests that 
  870.                  * fail to authenticate or match to a registered route. 
  871.                  * 
  872.                  * @since 4.7.0 
  873.                  * 
  874.                  * @param WP_HTTP_Response $response Result to send to the client. Usually a WP_REST_Response. 
  875.                  * @param WP_REST_Server   $handler  ResponseHandler instance (usually WP_REST_Server). 
  876.                  * @param WP_REST_Request  $request  Request used to generate the response. 
  877.                  */ 
  878.                 $response = apply_filters( 'rest_request_before_callbacks', $response, $handler, $request ); 
  879.  
  880.                 if ( ! is_wp_error( $response ) ) { 
  881.                     // Check permission specified on the route. 
  882.                     if ( ! empty( $handler['permission_callback'] ) ) { 
  883.                         $permission = call_user_func( $handler['permission_callback'], $request ); 
  884.  
  885.                         if ( is_wp_error( $permission ) ) { 
  886.                             $response = $permission; 
  887.                         } elseif ( false === $permission || null === $permission ) { 
  888.                             $response = new WP_Error( 'rest_forbidden', __( 'Sorry, you are not allowed to do that.' ), array( 'status' => 403 ) ); 
  889.                         } 
  890.                     } 
  891.                 } 
  892.  
  893.                 if ( ! is_wp_error( $response ) ) { 
  894.                     /** 
  895.                      * Filters the REST dispatch request result. 
  896.                      * 
  897.                      * Allow plugins to override dispatching the request. 
  898.                      * 
  899.                      * @since 4.4.0 
  900.                      * @since 4.5.0 Added `$route` and `$handler` parameters. 
  901.                      * 
  902.                      * @param bool            $dispatch_result Dispatch result, will be used if not empty. 
  903.                      * @param WP_REST_Request $request         Request used to generate the response. 
  904.                      * @param string          $route           Route matched for the request. 
  905.                      * @param array           $handler         Route handler used for the request. 
  906.                      */ 
  907.                     $dispatch_result = apply_filters( 'rest_dispatch_request', null, $request, $route, $handler ); 
  908.  
  909.                     // Allow plugins to halt the request via this filter. 
  910.                     if ( null !== $dispatch_result ) { 
  911.                         $response = $dispatch_result; 
  912.                     } else { 
  913.                         $response = call_user_func( $callback, $request ); 
  914.                     } 
  915.                 } 
  916.  
  917.                 /** 
  918.                  * Filters the response immediately after executing any REST API 
  919.                  * callbacks. 
  920.                  * 
  921.                  * Allows plugins to perform any needed cleanup, for example,  
  922.                  * to undo changes made during the {@see 'rest_request_before_callbacks'} 
  923.                  * filter. 
  924.                  * 
  925.                  * Note that this filter will not be called for requests that 
  926.                  * fail to authenticate or match to a registered route. 
  927.                  * 
  928.                  * Note that an endpoint's `permission_callback` can still be 
  929.                  * called after this filter - see `rest_send_allow_header()`. 
  930.                  * 
  931.                  * @since 4.7.0 
  932.                  * 
  933.                  * @param WP_HTTP_Response $response Result to send to the client. Usually a WP_REST_Response. 
  934.                  * @param WP_REST_Server   $handler  ResponseHandler instance (usually WP_REST_Server). 
  935.                  * @param WP_REST_Request  $request  Request used to generate the response. 
  936.                  */ 
  937.                 $response = apply_filters( 'rest_request_after_callbacks', $response, $handler, $request ); 
  938.  
  939.                 if ( is_wp_error( $response ) ) { 
  940.                     $response = $this->error_to_response( $response ); 
  941.                 } else { 
  942.                     $response = rest_ensure_response( $response ); 
  943.                 } 
  944.  
  945.                 $response->set_matched_route( $route ); 
  946.                 $response->set_matched_handler( $handler ); 
  947.  
  948.                 return $response; 
  949.             } 
  950.         } 
  951.  
  952.         return $this->error_to_response( new WP_Error( 'rest_no_route', __( 'No route was found matching the URL and request method' ), array( 'status' => 404 ) ) ); 
  953.     }

  955.     /** 
  956.      * Returns if an error occurred during most recent JSON encode/decode.

  958.      * Strings to be translated will be in format like 
  959.      * "Encoding error: Maximum stack depth exceeded". 
  960.      * 
  961.      * @since 4.4.0 
  962.      * @access protected 
  963.      *

  966.     protected function get_json_last_error() {

  1081.     public function get_data_for_routes( $routes, $context = 'view' ) {

  1085.         foreach ( $routes as $route => $callbacks ) {

  1087.             if ( empty( $data ) ) {

  1090.  
  1091.             /** 
  1092.              * Filters the REST endpoint data. 
  1093.              *

  1106.          * acts as a form of self-documentation. 
  1107.          * 
  1108.          * @since 4.4.0

  1116.     /** 
  1117.      * Retrieves publicly-visible data for the route. 
  1118.      * 
  1119.      * @since 4.4.0 
  1120.      * @access public 
  1121.      * 
  1122.      * @param string $route     Route to get data for. 
  1123.      * @param array  $callbacks Callbacks to convert to data. 
  1124.      * @param string $context   Optional. Context for the data. Accepts 'view' or 'help'. Default 'view'. 
  1125.      * @return array|null Data for the route, or null if no publicly-visible data. 
  1126.      */ 
  1127.     public function get_data_for_route( $route, $callbacks, $context = 'view' ) { 
  1128.         $data = array( 
  1129.             'namespace' => '',  
  1130.             'methods' => array(),  
  1131.             'endpoints' => array(),  
  1132.  ); 
  1133.  
  1134.         if ( isset( $this->route_options[ $route ] ) ) { 
  1135.             $options = $this->route_options[ $route ]; 
  1136.  
  1137.             if ( isset( $options['namespace'] ) ) { 
  1138.                 $data['namespace'] = $options['namespace']; 
  1139.             } 
  1140.  
  1141.             if ( isset( $options['schema'] ) && 'help' === $context ) { 
  1142.                 $data['schema'] = call_user_func( $options['schema'] ); 
  1143.             } 
  1144.         } 
  1145.  
  1146.         $route = preg_replace( '#\(\?P<(\w+?)>.*?\)#', '{$1}', $route ); 
  1147.  
  1148.         foreach ( $callbacks as $callback ) { 
  1149.             // Skip to the next route if any callback is hidden. 
  1150.             if ( empty( $callback['show_in_index'] ) ) { 
  1151.                 continue; 
  1152.             } 
  1153.  
  1154.             $data['methods'] = array_merge( $data['methods'], array_keys( $callback['methods'] ) ); 
  1155.             $endpoint_data = array( 
  1156.                 'methods' => array_keys( $callback['methods'] ),  
  1157.  ); 
  1158.
