  136.      * Validate CSS. 
  137.      * 
  138.      * Checks for imbalanced braces, brackets, and comments. 
  139.      * Notifications are rendered when the customizer state is saved. 
  140.      * 
  141.      * @todo There are cases where valid CSS can be incorrectly marked as invalid when strings or comments include balancing characters. To fix, CSS tokenization needs to be used. 
  142.      * 
  143.      * @since 4.7.0 
  144.      * @access public 
  145.      * 
  146.      * @param string $css The input string. 
  147.      * @return true|WP_Error True if the input was validated, otherwise WP_Error.

  149.     public function validate( $css ) { 
  150.         $validity = new WP_Error(); 
  151.  
  152.         if ( preg_match( '#</?\w+#', $css ) ) { 
  153.             $validity->add( 'illegal_markup', __( 'Markup is not allowed in CSS.' ) ); 
  154.         } 
  155.  
  156.         $imbalanced = false; 
  157.  
  158.         // Make sure that there is a closing brace for each opening brace. 
  159.         if ( ! $this->validate_balanced_characters( '{', '}', $css ) ) { 
  160.             $validity->add( 'imbalanced_curly_brackets', __( 'Your curly brackets <code>{}</code> are imbalanced. Make sure there is a closing <code>}</code> for every opening <code>{</code>.' ) ); 
  161.             $imbalanced = true; 
  162.         } 
  163.  
  164.         // Ensure brackets are balanced. 
  165.         if ( ! $this->validate_balanced_characters( '[', ']', $css ) ) { 
  166.             $validity->add( 'imbalanced_braces', __( 'Your brackets <code>[]</code> are imbalanced. Make sure there is a closing <code>]</code> for every opening <code>[</code>.' ) ); 
  167.             $imbalanced = true; 
  168.         } 
  169.  
  170.         // Ensure parentheses are balanced. 
  171.         if ( ! $this->validate_balanced_characters( '(', ')', $css ) ) { 
  172.             $validity->add( 'imbalanced_parentheses', __( 'Your parentheses <code>()</code> are imbalanced. Make sure there is a closing <code>)</code> for every opening <code>(</code>.' ) ); 
  173.             $imbalanced = true; 
  174.         } 
  175.  
  176.         // Ensure double quotes are equal. 
  177.         if ( ! $this->validate_equal_characters( '"', $css ) ) { 
  178.             $validity->add( 'unequal_double_quotes', __( 'Your double quotes <code>"</code> are uneven. Make sure there is a closing <code>"</code> for every opening <code>"</code>.' ) ); 
  179.             $imbalanced = true; 
  180.         } 
  181.  
  182.         /** 
  183.          * Make sure any code comments are closed properly. 
  184.          * 
  185.          * The first check could miss stray an unpaired comment closing figure, so if 
  186.          * The number appears to be balanced, then check for equal numbers 
  187.          * of opening/closing comment figures. 
  188.          * 
  189.          * Although it may initially appear redundant, we use the first method 
  190.          * to give more specific feedback to the user. 
  191.          */ 
  192.         $unclosed_comment_count = $this->validate_count_unclosed_comments( $css ); 
  193.         if ( 0 < $unclosed_comment_count ) { 
  194.             $validity->add( 'unclosed_comment', sprintf( _n( 'There is %s unclosed code comment. Close each comment with <code>*/</code>.', 'There are %s unclosed code comments. Close each comment with <code>*/</code>.', $unclosed_comment_count ), $unclosed_comment_count ) ); 
  195.             $imbalanced = true; 
  196.         } elseif ( ! $this->validate_balanced_characters( '/*', '*/', $css ) ) { 
  197.             $validity->add( 'imbalanced_comments', __( 'There is an extra <code>*/</code>, indicating an end to a comment.  Be sure that there is an opening <code>/*</code> for every closing <code>*/</code>.' ) );

  286.      * 
  287.      * Used during validation. 
  288.      * 
  289.      * @see self::validate() 
  290.      * 
  291.      * @since 4.7.0

  297.     private function validate_count_unclosed_comments( $css ) { 
  298.         $count = 0; 
  299.         $comments = explode( '/*', $css ); 
  300.   
  301.         if ( ! is_array( $comments ) || ( 1 >= count( $comments ) ) ) { 
  302.             return $count; 
  303.         } 
  304.   
  305.         unset( $comments[0] ); // The first item is before the first comment. 
  306.         foreach ( $comments as $comment ) { 
  307.             if ( false === strpos( $comment, '*/' ) ) { 
  308.                 $count++; 
  309.             } 
  310.         } 
  311.         return $count;

  314.     /** 
  315.      * Find "content:" within a string. 
  316.      * 
  317.      * Imbalanced/Unclosed validation errors may be caused 
  318.      * when a character is used in a "content:" declaration. 
  319.      * 
  320.      * This function is used to detect if this is a possible 
  321.      * cause of the validation error, so that if it is,  
  322.      * a notification may be added to the Validation Errors.

  324.      * Example: 
  325.      * .element::before { 
  326.      *   content: "(\""; 
  327.      * } 
  328.      * .element::after { 
  329.      *   content: "\")"; 
  330.      * }

  345.             return true; 
  346.         }
