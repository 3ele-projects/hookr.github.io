  594.     /** 
  595.      * HTTP response parser 
  596.      * 
  597.      * @throws Requests_Exception On missing head/body separator (`requests.no_crlf_separator`) 
  598.      * @throws Requests_Exception On missing head/body separator (`noversion`) 
  599.      * @throws Requests_Exception On missing head/body separator (`toomanyredirects`) 
  600.      * 
  601.      * @param string $headers Full response text including headers and body 
  602.      * @param string $url Original request URL 
  603.      * @param array $req_headers Original $headers array passed to {@link request()}, in case we need to follow redirects 
  604.      * @param array $req_data Original $data array passed to {@link request()}, in case we need to follow redirects 
  605.      * @param array $options Original $options array passed to {@link request()}, in case we need to follow redirects 
  606.      * @return Requests_Response 
  607.      */ 
  608.     protected static function parse_response($headers, $url, $req_headers, $req_data, $options) { 
  609.         $return = new Requests_Response(); 
  610.         if (!$options['blocking']) { 
  611.             return $return; 
  612.         } 
  613.  
  614.         $return->raw = $headers; 
  615.         $return->url = $url; 
  616.  
  617.         if (!$options['filename']) { 
  618.             if (($pos = strpos($headers, "\r\n\r\n")) === false) { 
  619.                 // Crap! 
  620.                 throw new Requests_Exception('Missing header/body separator', 'requests.no_crlf_separator'); 
  621.             } 
  622.  
  623.             $headers = substr($return->raw, 0, $pos); 
  624.             $return->body = substr($return->raw, $pos + strlen("\n\r\n\r")); 
  625.         } 
  626.         else { 
  627.             $return->body = ''; 
  628.         } 
  629.         // Pretend CRLF = LF for compatibility (RFC 2616, section 19.3) 
  630.         $headers = str_replace("\r\n", "\n", $headers); 
  631.         // Unfold headers (replace [CRLF] 1*( SP | HT ) with SP) as per RFC 2616 (section 2.2) 
  632.         $headers = preg_replace('/\n[ \t]/', ' ', $headers); 
  633.         $headers = explode("\n", $headers); 
  634.         preg_match('#^HTTP/(1\.\d)[ \t]+(\d+)#i', array_shift($headers), $matches); 
  635.         if (empty($matches)) { 
  636.             throw new Requests_Exception('Response could not be parsed', 'noversion', $headers); 
  637.         } 
  638.         $return->protocol_version = (float) $matches[1]; 
  639.         $return->status_code = (int) $matches[2]; 
  640.         if ($return->status_code >= 200 && $return->status_code < 300) { 
  641.             $return->success = true; 
  642.         } 
  643.  
  644.         foreach ($headers as $header) { 
  645.             list($key, $value) = explode(':', $header, 2); 
  646.             $value = trim($value); 
  647.             preg_replace('#(\s+)#i', ' ', $value); 
  648.             $return->headers[$key] = $value; 
  649.         } 
  650.         if (isset($return->headers['transfer-encoding'])) { 
  651.             $return->body = self::decode_chunked($return->body); 
  652.             unset($return->headers['transfer-encoding']); 
  653.         } 
  654.         if (isset($return->headers['content-encoding'])) { 
  655.             $return->body = self::decompress($return->body); 
  656.         } 
  657.  
  658.         //fsockopen and cURL compatibility 
  659.         if (isset($return->headers['connection'])) { 
  660.             unset($return->headers['connection']); 
  661.         } 
  662.  
  663.         $options['hooks']->dispatch('requests.before_redirect_check', array(&$return, $req_headers, $req_data, $options)); 
  664.  
  665.         if ($return->is_redirect() && $options['follow_redirects'] === true) { 
  666.             if (isset($return->headers['location']) && $options['redirected'] < $options['redirects']) { 
  667.                 if ($return->status_code === 303) { 
  668.                     $options['type'] = self::GET; 
  669.                 } 
  670.                 $options['redirected']++; 
  671.                 $location = $return->headers['location']; 
  672.                 if (strpos($location, 'http://') !== 0 && strpos($location, 'https://') !== 0) { 
  673.                     // relative redirect, for compatibility make it absolute 
  674.                     $location = Requests_IRI::absolutize($url, $location); 
  675.                     $location = $location->uri; 
  676.                 } 
  677.  
  678.                 $hook_args = array( 
  679.                     &$location,  
  680.                     &$req_headers,  
  681.                     &$req_data,  
  682.                     &$options,  
  683.                     $return 
  684.  ); 
  685.                 $options['hooks']->dispatch('requests.before_redirect', $hook_args); 
  686.                 $redirected = self::request($location, $req_headers, $req_data, $options['type'], $options); 
  687.                 $redirected->history[] = $return; 
  688.                 return $redirected; 
  689.             } 
  690.             elseif ($options['redirected'] >= $options['redirects']) { 
  691.                 throw new Requests_Exception('Too many redirects', 'toomanyredirects', $return); 
  692.             } 
  693.         } 
  694.  
  695.         $return->redirects = $options['redirected']; 
  696.  
  697.         $options['hooks']->dispatch('requests.after_request', array(&$return, $req_headers, $req_data, $options)); 
  698.         return $return; 
  699.     }

  701.     /** 
  702.      * Callback for `transport.internal.parse_response` 
  703.      * 
  704.      * Internal use only. Converts a raw HTTP response to a Requests_Response 
  705.      * while still executing a multiple request. 
  706.      * 
  707.      * @param string $response Full response text including headers and body (will be overwritten with Response instance) 
  708.      * @param array $request Request data as passed into {@see Requests::request_multiple()} 
  709.      * @return null `$response` is either set to a Requests_Response instance, or a Requests_Exception object 
  710.      */ 
  711.     public static function parse_multiple(&$response, $request) { 
  712.         try { 
  713.             $url = $request['url']; 
  714.             $headers = $request['headers']; 
  715.             $data = $request['data']; 
  716.             $options = $request['options']; 
  717.             $response = self::parse_response($response, $url, $headers, $data, $options); 
  718.         }

  724.     /** 
  725.      * Decoded a chunked body as per RFC 2616 
  726.      * 
  727.      * @see https://tools.ietf.org/html/rfc2616#section-3.6.1 
  728.      * @param string $data Chunked body 
  729.      * @return string Decoded body 
  730.      */ 
  731.     protected static function decode_chunked($data) { 
  732.         if (!preg_match('/^([0-9a-f]+)(?:;(?:[\w-]*)(?:=(?:(?:[\w-]*)*|"(?:[^\r\n])*"))?)*\r\n/i', trim($data))) { 
  733.             return $data; 
  734.         } 
  735.  
  736.  
  737.  
  738.         $decoded = ''; 
  739.         $encoded = $data; 
  740.  
  741.         while (true) { 
  742.             $is_chunked = (bool) preg_match('/^([0-9a-f]+)(?:;(?:[\w-]*)(?:=(?:(?:[\w-]*)*|"(?:[^\r\n])*"))?)*\r\n/i', $encoded, $matches); 
  743.             if (!$is_chunked) { 
  744.                 // Looks like it's not chunked after all 
  745.                 return $data; 
  746.             } 
  747.  
  748.             $length = hexdec(trim($matches[1])); 
  749.             if ($length === 0) { 
  750.                 // Ignore trailer headers 
  751.                 return $decoded; 
  752.             } 
  753.  
  754.             $chunk_length = strlen($matches[0]); 
  755.             $decoded .= substr($encoded, $chunk_length, $length); 
  756.             $encoded = substr($encoded, $chunk_length + $length + 2); 
  757.  
  758.             if (trim($encoded) === '0' || empty($encoded)) { 
  759.                 return $decoded; 
  760.             } 
  761.         } 
  762.  
  763.         // We'll never actually get down here 
  764.         // @codeCoverageIgnoreStart 
  765.     } 
  766.     // @codeCoverageIgnoreEnd

  794.     /** 
  795.      * Decompress an encoded body 
  796.      * 
  797.      * Implements gzip, compress and deflate. Guesses which it is by attempting 
  798.      * to decode. 
  799.      * 
  800.      * @param string $data Compressed data in one of the above formats 
  801.      * @return string Decompressed string 
  802.      */ 
  803.     public static function decompress($data) { 
  804.         if (substr($data, 0, 2) !== "\x1f\x8b" && substr($data, 0, 2) !== "\x78\x9c") { 
  805.             // Not actually compressed. Probably cURL ruining this for us. 
  806.             return $data; 
  807.         } 
  808.  
  809.         if (function_exists('gzdecode') && ($decoded = @gzdecode($data)) !== false) { 
  810.             return $decoded; 
  811.         } 
  812.         elseif (function_exists('gzinflate') && ($decoded = @gzinflate($data)) !== false) { 
  813.             return $decoded; 
  814.         } 
  815.         elseif (($decoded = self::compatible_gzinflate($data)) !== false) { 
  816.             return $decoded; 
  817.         } 
  818.         elseif (function_exists('gzuncompress') && ($decoded = @gzuncompress($data)) !== false) { 
  819.             return $decoded; 
  820.         } 
  821.  
  822.         return $data; 
  823.     }

  825.     /** 
  826.      * Decompression of deflated string while staying compatible with the majority of servers. 
  827.      * 
  828.      * Certain Servers will return deflated data with headers which PHP's gzinflate() 
  829.      * function cannot handle out of the box. The following function has been created from 
  830.      * various snippets on the gzinflate() PHP documentation. 
  831.      * 
  832.      * Warning: Magic numbers within. Due to the potential different formats that the compressed 
  833.      * data may be returned in, some "magic offsets" are needed to ensure proper decompression 
  834.      * takes place. For a simple progmatic way to determine the magic offset in use, see: 
  835.      * https://core.trac.wordpress.org/ticket/18273 
  836.      * 
  837.      * @since 2.8.1 
  838.      * @link https://core.trac.wordpress.org/ticket/18273 
  839.      * @link https://secure.php.net/manual/en/function.gzinflate.php#70875 
  840.      * @link https://secure.php.net/manual/en/function.gzinflate.php#77336 
  841.      * 
  842.      * @param string $gzData String to decompress. 
  843.      * @return string|bool False on failure. 
  844.      */ 
  845.     public static function compatible_gzinflate($gzData) { 
  846.         // Compressed data might contain a full zlib header, if so strip it for 
  847.         // gzinflate() 
  848.         if (substr($gzData, 0, 3) == "\x1f\x8b\x08") { 
  849.             $i = 10; 
  850.             $flg = ord(substr($gzData, 3, 1)); 
  851.             if ($flg > 0) { 
  852.                 if ($flg & 4) { 
  853.                     list($xlen) = unpack('v', substr($gzData, $i, 2)); 
  854.                     $i = $i + 2 + $xlen; 
  855.                 } 
  856.                 if ($flg & 8) { 
  857.                     $i = strpos($gzData, "\0", $i) + 1; 
  858.                 } 
  859.                 if ($flg & 16) { 
  860.                     $i = strpos($gzData, "\0", $i) + 1; 
  861.                 } 
  862.                 if ($flg & 2) { 
  863.                     $i = $i + 2; 
  864.                 } 
  865.             } 
  866.             $decompressed = self::compatible_gzinflate(substr($gzData, $i)); 
  867.             if (false !== $decompressed) { 
  868.                 return $decompressed; 
  869.             } 
  870.         } 
  871.  
  872.         // If the data is Huffman Encoded, we must first strip the leading 2 
  873.         // byte Huffman marker for gzinflate() 
  874.         // The response is Huffman coded by many compressors such as 
  875.         // java.util.zip.Deflater, Ruby*s Zlib::Deflate, and .NET's 
  876.         // System.IO.Compression.DeflateStream. 
  877.         // 
  878.         // See https://decompres.blogspot.com/ for a quick explanation of this 
  879.         // data type 
  880.         $huffman_encoded = false; 
  881.  
  882.         // low nibble of first byte should be 0x08 
  883.         list(, $first_nibble) = unpack('h', $gzData); 
  884.  
  885.         // First 2 bytes should be divisible by 0x1F 
  886.         list(, $first_two_bytes) = unpack('n', $gzData); 
  887.  
  888.         if (0x08 == $first_nibble && 0 == ($first_two_bytes % 0x1F)) { 
  889.             $huffman_encoded = true; 
  890.         } 
  891.  
  892.         if ($huffman_encoded) { 
  893.             if (false !== ($decompressed = @gzinflate(substr($gzData, 2)))) { 
  894.                 return $decompressed; 
  895.             } 
  896.         } 
  897.  
  898.         if ("\x50\x4b\x03\x04" == substr($gzData, 0, 4)) { 
  899.             // ZIP file format header 
  900.             // Offset 6: 2 bytes, General-purpose field 
  901.             // Offset 26: 2 bytes, filename length 
  902.             // Offset 28: 2 bytes, optional field length 
  903.             // Offset 30: Filename field, followed by optional field, followed 
  904.             // immediately by data 
  905.             list(, $general_purpose_flag) = unpack('v', substr($gzData, 6, 2)); 
  906.  
  907.             // If the file has been compressed on the fly, 0x08 bit is set of 
  908.             // the general purpose field. We can use this to differentiate 
  909.             // between a compressed document, and a ZIP file 
  910.             $zip_compressed_on_the_fly = (0x08 == (0x08 & $general_purpose_flag)); 
  911.  
  912.             if (!$zip_compressed_on_the_fly) { 
  913.                 // Don't attempt to decode a compressed zip file 
  914.                 return $gzData; 
  915.             } 
  916.  
  917.             // Determine the first byte of data, based on the above ZIP header 
  918.             // offsets: 
  919.             $first_file_start = array_sum(unpack('v2', substr($gzData, 26, 4))); 
  920.             if (false !== ($decompressed = @gzinflate(substr($gzData, 30 + $first_file_start)))) { 
  921.                 return $decompressed; 
  922.             } 
  923.             return false; 
  924.         } 
  925.  
  926.         // Finally fall back to straight gzinflate 
  927.         if (false !== ($decompressed = @gzinflate($gzData))) { 
  928.             return $decompressed; 
  929.         } 
  930.  
  931.         // Fallback for all above failing, not expected, but included for 
  932.         // debugging and preventing regressions and to track stats 
  933.         if (false !== ($decompressed = @gzinflate(substr($gzData, 2)))) { 
  934.             return $decompressed; 
  935.         } 
  936.  
  937.         return false; 
  938.     }

  940.     public static function match_domain($host, $reference) { 
  941.         // Check for a direct match 
  942.         if ($host === $reference) { 
  943.             return true; 
  944.         } 
  945.  
  946.         // Calculate the valid wildcard match if the host is not an IP address 
  947.         // Also validates that the host has 3 parts or more, as per Firefox's 
  948.         // ruleset. 
  949.         $parts = explode('.', $host); 
  950.         if (ip2long($host) === false && count($parts) >= 3) { 
  951.             $parts[0] = '*'; 
  952.             $wildcard = implode('.', $parts); 
  953.             if ($wildcard === $reference) { 
  954.                 return true; 
  955.             } 
  956.         } 
  957.  
  958.         return false; 
  959.     }
