  300.     /** 
  301.      * Verifies the received SSL certificate against its Common Names and subjectAltName fields. 
  302.      * 
  303.      * PHP's SSL verifications only verify that it's a valid Certificate, it doesn't verify if 
  304.      * the certificate is valid for the hostname which was requested. 
  305.      * This function verifies the requested hostname against certificate's subjectAltName field,  
  306.      * if that is empty, or contains no DNS entries, a fallback to the Common Name field is used. 
  307.      * 
  308.      * IP Address support is included if the request is being made to an IP address. 
  309.      * 
  310.      * @since 3.7.0 
  311.      * @static 
  312.      * 
  313.      * @param stream $stream The PHP Stream which the SSL request is being made over 
  314.      * @param string $host The hostname being requested 
  315.      * @return bool If the cerficiate presented in $stream is valid for $host 
  316.      */ 
  317.     public static function verify_ssl_certificate( $stream, $host ) { 
  318.         $context_options = stream_context_get_options( $stream ); 
  319.  
  320.         if ( empty( $context_options['ssl']['peer_certificate'] ) ) 
  321.             return false; 
  322.  
  323.         $cert = openssl_x509_parse( $context_options['ssl']['peer_certificate'] ); 
  324.         if ( ! $cert ) 
  325.             return false; 
  326.  
  327.         /** 
  328.          * If the request is being made to an IP address, we'll validate against IP fields 
  329.          * in the cert (if they exist) 
  330.          */ 
  331.         $host_type = ( WP_Http::is_ip_address( $host ) ? 'ip' : 'dns' ); 
  332.  
  333.         $certificate_hostnames = array(); 
  334.         if ( ! empty( $cert['extensions']['subjectAltName'] ) ) { 
  335.             $match_against = preg_split( '/, \s*/', $cert['extensions']['subjectAltName'] ); 
  336.             foreach ( $match_against as $match ) { 
  337.                 list( $match_type, $match_host ) = explode( ':', $match ); 
  338.                 if ( $host_type == strtolower( trim( $match_type ) ) ) // IP: or DNS: 
  339.                     $certificate_hostnames[] = strtolower( trim( $match_host ) ); 
  340.             } 
  341.         } elseif ( !empty( $cert['subject']['CN'] ) ) { 
  342.             // Only use the CN when the certificate includes no subjectAltName extension. 
  343.             $certificate_hostnames[] = strtolower( $cert['subject']['CN'] ); 
  344.         } 
  345.  
  346.         // Exact hostname/IP matches. 
  347.         if ( in_array( strtolower( $host ), $certificate_hostnames ) ) 
  348.             return true; 
  349.  
  350.         // IP's can't be wildcards, Stop processing. 
  351.         if ( 'ip' == $host_type ) 
  352.             return false; 
  353.  
  354.         // Test to see if the domain is at least 2 deep for wildcard support. 
  355.         if ( substr_count( $host, '.' ) < 2 ) 
  356.             return false; 
  357.  
  358.         // Wildcard subdomains certs (*.example.com) are valid for a.example.com but not a.b.example.com. 
  359.         $wildcard_host = preg_replace( '/^[^.]+\./', '*.', $host ); 
  360.  
  361.         return in_array( strtolower( $wildcard_host ), $certificate_hostnames ); 
  362.     }
