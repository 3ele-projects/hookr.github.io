  484.                 $this->query_limit = $wpdb->prepare("LIMIT %d, %d", $qv['offset'], $qv['number']); 
  485.             } else { 
  486.                 $this->query_limit = $wpdb->prepare( "LIMIT %d, %d", $qv['number'] * ( $qv['paged'] - 1 ), $qv['number'] ); 
  487.             } 
  488.         } 
  489.  
  490.         $search = ''; 
  491.         if ( isset( $qv['search'] ) ) 
  492.             $search = trim( $qv['search'] ); 
  493.  
  494.         if ( $search ) { 
  495.             $leading_wild = ( ltrim($search, '*') != $search ); 
  496.             $trailing_wild = ( rtrim($search, '*') != $search ); 
  497.             if ( $leading_wild && $trailing_wild ) 
  498.                 $wild = 'both'; 
  499.             elseif ( $leading_wild ) 
  500.                 $wild = 'leading'; 
  501.             elseif ( $trailing_wild ) 
  502.                 $wild = 'trailing'; 
  503.             else 
  504.                 $wild = false; 
  505.             if ( $wild ) 
  506.                 $search = trim($search, '*'); 
  507.  
  508.             $search_columns = array(); 
  509.             if ( $qv['search_columns'] ) 
  510.                 $search_columns = array_intersect( $qv['search_columns'], array( 'ID', 'user_login', 'user_email', 'user_url', 'user_nicename' ) ); 
  511.             if ( ! $search_columns ) { 
  512.                 if ( false !== strpos( $search, '@') ) 
  513.                     $search_columns = array('user_email'); 
  514.                 elseif ( is_numeric($search) ) 
  515.                     $search_columns = array('user_login', 'ID'); 
  516.                 elseif ( preg_match('|^https?://|', $search) && ! ( is_multisite() && wp_is_large_network( 'users' ) ) ) 
  517.                     $search_columns = array('user_url'); 
  518.                 else 
  519.                     $search_columns = array('user_login', 'user_url', 'user_email', 'user_nicename', 'display_name'); 
  520.             } 
  521.  
  522.             /** 
  523.              * Filters the columns to search in a WP_User_Query search. 
  524.              * 
  525.              * The default columns depend on the search term, and include 'user_email',  
  526.              * 'user_login', 'ID', 'user_url', 'display_name', and 'user_nicename'. 
  527.              * 
  528.              * @since 3.6.0 
  529.              * 
  530.              * @param array         $search_columns Array of column names to be searched. 
  531.              * @param string        $search         Text being searched. 
  532.              * @param WP_User_Query $this           The current WP_User_Query instance. 
  533.              */ 
  534.             $search_columns = apply_filters( 'user_search_columns', $search_columns, $search, $this ); 
  535.  
  536.             $this->query_where .= $this->get_search_sql( $search, $search_columns, $wild ); 
  537.         } 
  538.  
  539.         if ( ! empty( $include ) ) { 
  540.             // Sanitized earlier. 
  541.             $ids = implode( ', ', $include ); 
  542.             $this->query_where .= " AND $wpdb->users.ID IN ($ids)"; 
  543.         } elseif ( ! empty( $qv['exclude'] ) ) { 
  544.             $ids = implode( ', ', wp_parse_id_list( $qv['exclude'] ) ); 
  545.             $this->query_where .= " AND $wpdb->users.ID NOT IN ($ids)"; 
  546.         } 
  547.  
  548.         // Date queries are allowed for the user_registered field. 
  549.         if ( ! empty( $qv['date_query'] ) && is_array( $qv['date_query'] ) ) { 
  550.             $date_query = new WP_Date_Query( $qv['date_query'], 'user_registered' ); 
  551.             $this->query_where .= $date_query->get_sql(); 
  552.         } 
  553.  
  554.         /** 
  555.          * Fires after the WP_User_Query has been parsed, and before 
  556.          * the query is executed. 
  557.          * 
  558.          * The passed WP_User_Query object contains SQL parts formed 
  559.          * from parsing the given query. 
  560.          * 
  561.          * @since 3.1.0 
  562.          * 
  563.          * @param WP_User_Query $this The current WP_User_Query instance,  
  564.          *                            passed by reference. 
  565.          */ 
  566.         do_action_ref_array( 'pre_user_query', array( &$this ) ); 
  567.     } 
  568.  
  569.     /** 
  570.      * Execute the query, with the current variables. 
  571.      * 
  572.      * @since 3.1.0 
  573.      * 
  574.      * @global wpdb $wpdb WordPress database abstraction object. 
  575.      */ 
  576.     public function query() { 
  577.         global $wpdb; 
  578.  
  579.         $qv =& $this->query_vars; 
  580.  
  581.         $this->request = "SELECT $this->query_fields $this->query_from $this->query_where $this->query_orderby $this->query_limit"; 
  582.  
  583.         if ( is_array( $qv['fields'] ) || 'all' == $qv['fields'] ) { 
  584.             $this->results = $wpdb->get_results( $this->request ); 
  585.         } else { 
  586.             $this->results = $wpdb->get_col( $this->request ); 
  587.         } 
  588.  
  589.         /** 
  590.          * Filters SELECT FOUND_ROWS() query for the current WP_User_Query instance. 
  591.          * 
  592.          * @since 3.2.0 
  593.          * 
  594.          * @global wpdb $wpdb WordPress database abstraction object. 
  595.          * 
  596.          * @param string $sql The SELECT FOUND_ROWS() query for the current WP_User_Query. 
  597.          */ 
  598.         if ( isset( $qv['count_total'] ) && $qv['count_total'] ) 
  599.             $this->total_users = $wpdb->get_var( apply_filters( 'found_users_query', 'SELECT FOUND_ROWS()' ) ); 
  600.  
  601.         if ( !$this->results ) 
  602.             return; 
  603.  
  604.         if ( 'all_with_meta' == $qv['fields'] ) { 
  605.             cache_users( $this->results ); 
  606.  
  607.             $r = array(); 
  608.             foreach ( $this->results as $userid ) 
  609.                 $r[ $userid ] = new WP_User( $userid, '', $qv['blog_id'] ); 
  610.  
  611.             $this->results = $r; 
  612.         } elseif ( 'all' == $qv['fields'] ) { 
  613.             foreach ( $this->results as $key => $user ) { 
  614.                 $this->results[ $key ] = new WP_User( $user, '', $qv['blog_id'] ); 
  615.             } 
  616.         } 
  617.     }

  648.     /** 
  649.      * Used internally to generate an SQL string for searching across multiple columns 
  650.      * 
  651.      * @access protected 
  652.      * @since 3.1.0 
  653.      * 
  654.      * @global wpdb $wpdb WordPress database abstraction object. 
  655.      * 
  656.      * @param string $string 
  657.      * @param array  $cols 
  658.      * @param bool   $wild   Whether to allow wildcard searches. Default is false for Network Admin, true for single site. 
  659.      *                       Single site allows leading and trailing wildcards, Network Admin only trailing. 
  660.      * @return string 
  661.      */ 
  662.     protected function get_search_sql( $string, $cols, $wild = false ) { 
  663.         global $wpdb; 
  664.  
  665.         $searches = array(); 
  666.         $leading_wild = ( 'leading' == $wild || 'both' == $wild ) ? '%' : ''; 
  667.         $trailing_wild = ( 'trailing' == $wild || 'both' == $wild ) ? '%' : ''; 
  668.         $like = $leading_wild . $wpdb->esc_like( $string ) . $trailing_wild; 
  669.  
  670.         foreach ( $cols as $col ) { 
  671.             if ( 'ID' == $col ) { 
  672.                 $searches[] = $wpdb->prepare( "$col = %s", $string ); 
  673.             } else { 
  674.                 $searches[] = $wpdb->prepare( "$col LIKE %s", $like ); 
  675.             } 
  676.         } 
  677.  
  678.         return ' AND (' . implode(' OR ', $searches) . ')'; 
  679.     }

  705.     /** 
  706.      * Parse and sanitize 'orderby' keys passed to the user query. 
  707.      * 
  708.      * @since 4.2.0 
  709.      * @access protected 
  710.      * 
  711.      * @global wpdb $wpdb WordPress database abstraction object. 
  712.      * 
  713.      * @param string $orderby Alias for the field to order by. 
  714.      * @return string Value to used in the ORDER clause, if `$orderby` is valid. 
  715.      */ 
  716.     protected function parse_orderby( $orderby ) { 
  717.         global $wpdb; 
  718.  
  719.         $meta_query_clauses = $this->meta_query->get_clauses(); 
  720.  
  721.         $_orderby = ''; 
  722.         if ( in_array( $orderby, array( 'login', 'nicename', 'email', 'url', 'registered' ) ) ) { 
  723.             $_orderby = 'user_' . $orderby; 
  724.         } elseif ( in_array( $orderby, array( 'user_login', 'user_nicename', 'user_email', 'user_url', 'user_registered' ) ) ) { 
  725.             $_orderby = $orderby; 
  726.         } elseif ( 'name' == $orderby || 'display_name' == $orderby ) { 
  727.             $_orderby = 'display_name'; 
  728.         } elseif ( 'post_count' == $orderby ) { 
  729.             // todo: avoid the JOIN 
  730.             $where = get_posts_by_author_sql( 'post' ); 
  731.             $this->query_from .= " LEFT OUTER JOIN ( 
  732.                 SELECT post_author, COUNT(*) as post_count 
  733.                 FROM $wpdb->posts 
  734.                 $where 
  735.                 GROUP BY post_author 
  736.  ) p ON ({$wpdb->users}.ID = p.post_author) 
  737.             "; 
  738.             $_orderby = 'post_count'; 
  739.         } elseif ( 'ID' == $orderby || 'id' == $orderby ) { 
  740.             $_orderby = 'ID'; 
  741.         } elseif ( 'meta_value' == $orderby || $this->get( 'meta_key' ) == $orderby ) { 
  742.             $_orderby = "$wpdb->usermeta.meta_value"; 
  743.         } elseif ( 'meta_value_num' == $orderby ) { 
  744.             $_orderby = "$wpdb->usermeta.meta_value+0"; 
  745.         } elseif ( 'include' === $orderby && ! empty( $this->query_vars['include'] ) ) { 
  746.             $include = wp_parse_id_list( $this->query_vars['include'] ); 
  747.             $include_sql = implode( ', ', $include ); 
  748.             $_orderby = "FIELD( $wpdb->users.ID, $include_sql )"; 
  749.         } elseif ( 'nicename__in' === $orderby ) { 
  750.             $sanitized_nicename__in = array_map( 'esc_sql', $this->query_vars['nicename__in'] ); 
  751.             $nicename__in = implode( "', '", $sanitized_nicename__in ); 
  752.             $_orderby = "FIELD( user_nicename, '$nicename__in' )"; 
  753.         } elseif ( 'login__in' === $orderby ) { 
  754.             $sanitized_login__in = array_map( 'esc_sql', $this->query_vars['login__in'] ); 
  755.             $login__in = implode( "', '", $sanitized_login__in ); 
  756.             $_orderby = "FIELD( user_login, '$login__in' )"; 
  757.         } elseif ( isset( $meta_query_clauses[ $orderby ] ) ) { 
  758.             $meta_clause = $meta_query_clauses[ $orderby ]; 
  759.             $_orderby = sprintf( "CAST(%s.meta_value AS %s)", esc_sql( $meta_clause['alias'] ), esc_sql( $meta_clause['cast'] ) ); 
  760.         } 
  761.  
  762.         return $_orderby; 
  763.     }

  765.     /** 
  766.      * Parse an 'order' query variable and cast it to ASC or DESC as necessary. 
  767.      * 
  768.      * @since 4.2.0 
  769.      * @access protected 
  770.      * 
  771.      * @param string $order The 'order' query variable. 
  772.      * @return string The sanitized 'order' query variable. 
  773.      */ 
  774.     protected function parse_order( $order ) { 
  775.         if ( ! is_string( $order ) || empty( $order ) ) { 
  776.             return 'DESC'; 
  777.         } 
  778.  
  779.         if ( 'ASC' === strtoupper( $order ) ) { 
  780.             return 'ASC'; 
  781.         } else { 
  782.             return 'DESC'; 
  783.         } 
  784.     }
