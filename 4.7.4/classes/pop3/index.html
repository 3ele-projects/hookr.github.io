  611.     function parse_banner ( $server_text ) { 
  612.         $outside = true; 
  613.         $banner = ""; 
  614.         $length = strlen($server_text); 
  615.         for($count =0; $count < $length; $count++) 
  616.         { 
  617.             $digit = substr($server_text, $count, 1); 
  618.             if(!empty($digit))             { 
  619.                 if( (!$outside) && ($digit != '<') && ($digit != '>') ) 
  620.                 { 
  621.                     $banner .= $digit; 
  622.                 } 
  623.                 if ($digit == '<') 
  624.                 { 
  625.                     $outside = false; 
  626.                 } 
  627.                 if($digit == '>') 
  628.                 { 
  629.                     $outside = true; 
  630.                 } 
  631.             } 
  632.         } 
  633.         $banner = $this->strip_clf($banner);    // Just in case 
  634.         return "<$banner>";
