  17. /** 
  18.  * Registers a REST API route. 
  19.  * 
  20.  * @since 4.4.0 
  21.  * 
  22.  * @global WP_REST_Server $wp_rest_server ResponseHandler instance (usually WP_REST_Server). 
  23.  * 
  24.  * @param string $namespace The first URL segment after core prefix. Should be unique to your package/plugin. 
  25.  * @param string $route     The base URL for route you are adding. 
  26.  * @param array  $args      Optional. Either an array of options for the endpoint, or an array of arrays for 
  27.  *                          multiple methods. Default empty array. 
  28.  * @param bool   $override  Optional. If the route already exists, should we override it? True overrides,  
  29.  *                          false merges (with newer overriding if duplicate keys exist). Default false. 
  30.  * @return bool True on success, false on error. 
  31.  */ 
  32. function register_rest_route( $namespace, $route, $args = array(), $override = false ) { 
  33.   /** @var WP_REST_Server $wp_rest_server */ 
  34.   global $wp_rest_server; 
  35.  
  36.   if ( empty( $namespace ) ) { 
  37.       /** 
  38.        * Non-namespaced routes are not allowed, with the exception of the main 
  39.        * and namespace indexes. If you really need to register a 
  40.        * non-namespaced route, call `WP_REST_Server::register_route` directly. 
  41.        */ 
  42.       _doing_it_wrong( 'register_rest_route', __( 'Routes must be namespaced with plugin or theme name and version.' ), '4.4.0' ); 
  43.       return false; 
  44.   } else if ( empty( $route ) ) { 
  45.       _doing_it_wrong( 'register_rest_route', __( 'Route must be specified.' ), '4.4.0' ); 
  46.       return false; 
  47.   } 
  48.  
  49.   if ( isset( $args['args'] ) ) { 
  50.       $common_args = $args['args']; 
  51.       unset( $args['args'] ); 
  52.   } else { 
  53.       $common_args = array(); 
  54.   } 
  55.  
  56.   if ( isset( $args['callback'] ) ) { 
  57.       // Upgrade a single set to multiple. 
  58.       $args = array( $args ); 
  59.   } 
  60.  
  61.   $defaults = array( 
  62.       'methods' => 'GET',  
  63.       'callback' => null,  
  64.       'args' => array(),  
  65.  ); 
  66.   foreach ( $args as $key => &$arg_group ) { 
  67.       if ( ! is_numeric( $key ) ) { 
  68.           // Route option, skip here. 
  69.           continue; 
  70.       } 
  71.  
  72.       $arg_group = array_merge( $defaults, $arg_group ); 
  73.       $arg_group['args'] = array_merge( $common_args, $arg_group['args'] ); 
  74.   } 
  75.  
  76.   $full_route = '/' . trim( $namespace, '/' ) . '/' . trim( $route, '/' ); 
  77.   $wp_rest_server->register_route( $namespace, $full_route, $args, $override ); 
  78.   return true; 
  79. }

  295. /** 
  296.  * Retrieves the URL to a REST endpoint on a site. 
  297.  * 
  298.  * Note: The returned URL is NOT escaped. 
  299.  * 
  300.  * @since 4.4.0 
  301.  * 
  302.  * @todo Check if this is even necessary 
  303.  * @global WP_Rewrite $wp_rewrite 
  304.  * 
  305.  * @param int    $blog_id Optional. Blog ID. Default of null returns URL for current blog. 
  306.  * @param string $path    Optional. REST route. Default '/'. 
  307.  * @param string $scheme  Optional. Sanitization scheme. Default 'rest'. 
  308.  * @return string Full URL to the endpoint. 
  309.  */ 
  310. function get_rest_url( $blog_id = null, $path = '/', $scheme = 'rest' ) { 
  311.   if ( empty( $path ) ) { 
  312.       $path = '/'; 
  313.   } 
  314.  
  315.   if ( is_multisite() && get_blog_option( $blog_id, 'permalink_structure' ) || get_option( 'permalink_structure' ) ) { 
  316.       global $wp_rewrite; 
  317.  
  318.       if ( $wp_rewrite->using_index_permalinks() ) { 
  319.           $url = get_home_url( $blog_id, $wp_rewrite->index . '/' . rest_get_url_prefix(), $scheme ); 
  320.       } else { 
  321.           $url = get_home_url( $blog_id, rest_get_url_prefix(), $scheme ); 
  322.       } 
  323.  
  324.       $url .= '/' . ltrim( $path, '/' ); 
  325.   } else { 
  326.       $url = trailingslashit( get_home_url( $blog_id, '', $scheme ) ); 
  327.  
  328.       $path = '/' . ltrim( $path, '/' ); 
  329.  
  330.       $url = add_query_arg( 'rest_route', $path, $url ); 
  331.   } 
  332.  
  333.   if ( is_ssl() ) { 
  334.       // If the current host is the same as the REST URL host, force the REST URL scheme to HTTPS. 
  335.       if ( $_SERVER['SERVER_NAME'] === parse_url( get_home_url( $blog_id ), PHP_URL_HOST ) ) { 
  336.           $url = set_url_scheme( $url, 'https' ); 
  337.       } 
  338.   } 
  339.  
  340.   /** 
  341.    * Filters the REST URL. 
  342.    * 
  343.    * Use this filter to adjust the url returned by the get_rest_url() function. 
  344.    * 
  345.    * @since 4.4.0 
  346.    * 
  347.    * @param string $url     REST URL. 
  348.    * @param string $path    REST route. 
  349.    * @param int    $blog_id Blog ID. 
  350.    * @param string $scheme  Sanitization scheme. 
  351.    */ 
  352.   return apply_filters( 'rest_url', $url, $path, $blog_id, $scheme ); 
  353. }

  537. /** 
  538.  * Handles OPTIONS requests for the server. 
  539.  * 
  540.  * This is handled outside of the server code, as it doesn't obey normal route 
  541.  * mapping. 
  542.  * 
  543.  * @since 4.4.0 
  544.  * 
  545.  * @param mixed           $response Current response, either response or `null` to indicate pass-through. 
  546.  * @param WP_REST_Server  $handler  ResponseHandler instance (usually WP_REST_Server). 
  547.  * @param WP_REST_Request $request  The request that was used to make current response. 
  548.  * @return WP_REST_Response Modified response, either response or `null` to indicate pass-through. 
  549.  */ 
  550. function rest_handle_options_request( $response, $handler, $request ) { 
  551.   if ( ! empty( $response ) || $request->get_method() !== 'OPTIONS' ) { 
  552.       return $response; 
  553.   } 
  554.  
  555.   $response = new WP_REST_Response(); 
  556.   $data = array(); 
  557.  
  558.   foreach ( $handler->get_routes() as $route => $endpoints ) { 
  559.       $match = preg_match( '@^' . $route . '$@i', $request->get_route() ); 
  560.  
  561.       if ( ! $match ) { 
  562.           continue; 
  563.       } 
  564.  
  565.       $data = $handler->get_data_for_route( $route, $endpoints, 'help' );

  576.  * 
  577.  * @since 4.4.0 
  578.  * 
  579.  * @param WP_REST_Response $response Current response being served. 
  580.  * @param WP_REST_Server   $server   ResponseHandler instance (usually WP_REST_Server). 
  581.  * @param WP_REST_Request  $request  The request that was used to make current response. 
  582.  * @return WP_REST_Response Response to be served, with "Allow" header if route has allowed methods.

  584. function rest_send_allow_header( $response, $server, $request ) {

  613.   if ( $allowed_methods ) { 
  614.       $response->header( 'Allow', implode( ', ', array_map( 'strtoupper', array_keys( $allowed_methods ) ) ) );

  674. /** 
  675.  * Checks for errors when using cookie-based authentication. 
  676.  * 
  677.  * WordPress' built-in cookie authentication is always active 
  678.  * for logged in users. However, the API has to check nonces 
  679.  * for each request to ensure users are not vulnerable to CSRF. 
  680.  * 
  681.  * @since 4.4.0 
  682.  * 
  683.  * @global mixed          $wp_rest_auth_cookie 
  684.  * @global WP_REST_Server $wp_rest_server      REST server instance. 
  685.  * 
  686.  * @param WP_Error|mixed $result Error from another authentication handler,  
  687.  *                               null if we should handle it, or another value 
  688.  *                               if not. 
  689.  * @return WP_Error|mixed|bool WP_Error if the cookie is invalid, the $result, otherwise true. 
  690.  */ 
  691. function rest_cookie_check_errors( $result ) { 
  692.   if ( ! empty( $result ) ) { 
  693.       return $result; 
  694.   } 
  695.  
  696.   global $wp_rest_auth_cookie, $wp_rest_server; 
  697.  
  698.   /** 
  699.    * Is cookie authentication being used? (If we get an auth 
  700.    * error, but we're still logged in, another authentication 
  701.    * must have been used). 
  702.    */ 
  703.   if ( true !== $wp_rest_auth_cookie && is_user_logged_in() ) { 
  704.       return $result; 
  705.   } 
  706.  
  707.   // Determine if there is a nonce. 
  708.   $nonce = null; 
  709.  
  710.   if ( isset( $_REQUEST['_wpnonce'] ) ) { 
  711.       $nonce = $_REQUEST['_wpnonce']; 
  712.   } elseif ( isset( $_SERVER['HTTP_X_WP_NONCE'] ) ) { 
  713.       $nonce = $_SERVER['HTTP_X_WP_NONCE']; 
  714.   } 
  715.  
  716.   if ( null === $nonce ) { 
  717.       // No nonce at all, so act as if it's an unauthenticated request. 
  718.       wp_set_current_user( 0 ); 
  719.       return true; 
  720.   } 
  721.  
  722.   // Check the nonce. 
  723.   $result = wp_verify_nonce( $nonce, 'wp_rest' ); 
  724.  
  725.   if ( ! $result ) { 
  726.       return new WP_Error( 'rest_cookie_invalid_nonce', __( 'Cookie nonce is invalid' ), array( 'status' => 403 ) ); 
  727.   } 
  728.  
  729.   // Send a refreshed nonce in header. 
  730.   $wp_rest_server->send_header( 'X-WP-Nonce', wp_create_nonce( 'wp_rest' ) ); 
  731.  
  732.   return true; 
  733. }

  758. /** 
  759.  * Parses an RFC3339 time into a Unix timestamp. 
  760.  * 
  761.  * @since 4.4.0 
  762.  * 
  763.  * @param string $date      RFC3339 timestamp. 
  764.  * @param bool   $force_utc Optional. Whether to force UTC timezone instead of using 
  765.  *                          the timestamp's timezone. Default false. 
  766.  * @return int Unix timestamp. 
  767.  */ 
  768. function rest_parse_date( $date, $force_utc = false ) { 
  769.   if ( $force_utc ) { 
  770.       $date = preg_replace( '/[+-]\d+:?\d+$/', '+00:00', $date ); 
  771.   } 
  772.  
  773.   $regex = '#^\d{4}-\d{2}-\d{2}[Tt ]\d{2}:\d{2}:\d{2}(?:\.\d+)?(?:Z|[+-]\d{2}(?::\d{2})?)?$#'; 
  774.  
  775.   if ( ! preg_match( $regex, $date, $matches ) ) { 
  776.       return false; 
  777.   } 
  778.  
  779.   return strtotime( $date ); 
  780. }

  782. /** 
  783.  * Parses a date into both its local and UTC equivalent, in MySQL datetime format. 
  784.  * 
  785.  * @since 4.4.0 
  786.  * 
  787.  * @see rest_parse_date() 
  788.  * 
  789.  * @param string $date   RFC3339 timestamp. 
  790.  * @param bool   $is_utc Whether the provided date should be interpreted as UTC. Default false. 
  791.  * @return array|null Local and UTC datetime strings, in MySQL datetime format (Y-m-d H:i:s),  
  792.  *                    null on failure. 
  793.  */ 
  794. function rest_get_date_with_gmt( $date, $is_utc = false ) { 
  795.   // Whether or not the original date actually has a timezone string 
  796.   // changes the way we need to do timezone conversion.  Store this info 
  797.   // before parsing the date, and use it later. 
  798.   $has_timezone = preg_match( '#(Z|[+-]\d{2}(:\d{2})?)$#', $date ); 
  799.  
  800.   $date = rest_parse_date( $date ); 
  801.  
  802.   if ( empty( $date ) ) { 
  803.       return null; 
  804.   } 
  805.  
  806.   // At this point $date could either be a local date (if we were passed a 
  807.   // *local* date without a timezone offset) or a UTC date (otherwise). 
  808.   // Timezone conversion needs to be handled differently between these two 
  809.   // cases. 
  810.   if ( ! $is_utc && ! $has_timezone ) { 
  811.       $local = date( 'Y-m-d H:i:s', $date ); 
  812.       $utc = get_gmt_from_date( $local ); 
  813.   } else { 
  814.       $utc = date( 'Y-m-d H:i:s', $date ); 
  815.       $local = get_date_from_gmt( $utc ); 
  816.   } 
  817.  
  818.   return array( $local, $utc ); 
  819. }

  1014. /** 
  1015.  * Validate a value based on a schema. 
  1016.  * 
  1017.  * @param mixed  $value The value to validate. 
  1018.  * @param array  $args  Schema array to use for validation. 
  1019.  * @param string $param The parameter name, used in error messages. 
  1020.  * @return true|WP_Error 
  1021.  */ 
  1022. function rest_validate_value_from_schema( $value, $args, $param = '' ) { 
  1023.   if ( 'array' === $args['type'] ) { 
  1024.       if ( ! is_array( $value ) ) { 
  1025.           $value = preg_split( '/[\s, ]+/', $value ); 
  1026.       } 
  1027.       if ( ! wp_is_numeric_array( $value ) ) { 
  1028.           /** translators: 1: parameter, 2: type name */ 
  1029.           return new WP_Error( 'rest_invalid_param', sprintf( __( '%1$s is not of type %2$s.' ), $param, 'array' ) ); 
  1030.       } 
  1031.       foreach ( $value as $index => $v ) { 
  1032.           $is_valid = rest_validate_value_from_schema( $v, $args['items'], $param . '[' . $index . ']' ); 
  1033.           if ( is_wp_error( $is_valid ) ) { 
  1034.               return $is_valid; 
  1035.           } 
  1036.       } 
  1037.   } 
  1038.   if ( ! empty( $args['enum'] ) ) { 
  1039.       if ( ! in_array( $value, $args['enum'], true ) ) { 
  1040.           /** translators: 1: parameter, 2: list of valid values */ 
  1041.           return new WP_Error( 'rest_invalid_param', sprintf( __( '%1$s is not one of %2$s.' ), $param, implode( ', ', $args['enum'] ) ) ); 
  1042.       } 
  1043.   } 
  1044.  
  1045.   if ( in_array( $args['type'], array( 'integer', 'number' ) ) && ! is_numeric( $value ) ) { 
  1046.       /** translators: 1: parameter, 2: type name */ 
  1047.       return new WP_Error( 'rest_invalid_param', sprintf( __( '%1$s is not of type %2$s.' ), $param, $args['type'] ) ); 
  1048.   } 
  1049.  
  1050.   if ( 'integer' === $args['type'] && round( floatval( $value ) ) !== floatval( $value ) ) { 
  1051.       /** translators: 1: parameter, 2: type name */ 
  1052.       return new WP_Error( 'rest_invalid_param', sprintf( __( '%1$s is not of type %2$s.' ), $param, 'integer' ) ); 
  1053.   } 
  1054.  
  1055.   if ( 'boolean' === $args['type'] && ! rest_is_boolean( $value ) ) { 
  1056.       /** translators: 1: parameter, 2: type name */ 
  1057.       return new WP_Error( 'rest_invalid_param', sprintf( __( '%1$s is not of type %2$s.' ), $value, 'boolean' ) ); 
  1058.   } 
  1059.  
  1060.   if ( 'string' === $args['type'] && ! is_string( $value ) ) { 
  1061.       /** translators: 1: parameter, 2: type name */ 
  1062.       return new WP_Error( 'rest_invalid_param', sprintf( __( '%1$s is not of type %2$s.' ), $param, 'string' ) ); 
  1063.   } 
  1064.  
  1065.   if ( isset( $args['format'] ) ) { 
  1066.       switch ( $args['format'] ) { 
  1067.           case 'date-time' : 
  1068.               if ( ! rest_parse_date( $value ) ) { 
  1069.                   return new WP_Error( 'rest_invalid_date', __( 'Invalid date.' ) ); 
  1070.               } 
  1071.               break; 
  1072.  
  1073.           case 'email' : 
  1074.               // is_email() checks for 3 characters (a@b), but 
  1075.               // wp_handle_comment_submission() requires 6 characters (a@b.co) 
  1076.               // 
  1077.               // https://core.trac.wordpress.org/ticket/38506 
  1078.               if ( ! is_email( $value ) || strlen( $value ) < 6 ) { 
  1079.                   return new WP_Error( 'rest_invalid_email', __( 'Invalid email address.' ) ); 
  1080.               } 
  1081.               break; 
  1082.           case 'ip' : 
  1083.               if ( ! rest_is_ip_address( $value ) ) { 
  1084.                   /** translators: %s: IP address */ 
  1085.                   return new WP_Error( 'rest_invalid_param', sprintf( __( '%s is not a valid IP address.' ), $value ) ); 
  1086.               } 
  1087.               break; 
  1088.       } 
  1089.   } 
  1090.  
  1091.   if ( in_array( $args['type'], array( 'number', 'integer' ), true ) && ( isset( $args['minimum'] ) || isset( $args['maximum'] ) ) ) { 
  1092.       if ( isset( $args['minimum'] ) && ! isset( $args['maximum'] ) ) { 
  1093.           if ( ! empty( $args['exclusiveMinimum'] ) && $value <= $args['minimum'] ) { 
  1094.               /** translators: 1: parameter, 2: minimum number */ 
  1095.               return new WP_Error( 'rest_invalid_param', sprintf( __( '%1$s must be greater than %2$d (exclusive)' ), $param, $args['minimum'] ) ); 
  1096.           } elseif ( empty( $args['exclusiveMinimum'] ) && $value < $args['minimum'] ) { 
  1097.               /** translators: 1: parameter, 2: minimum number */ 
  1098.               return new WP_Error( 'rest_invalid_param', sprintf( __( '%1$s must be greater than %2$d (inclusive)' ), $param, $args['minimum'] ) ); 
  1099.           } 
  1100.       } elseif ( isset( $args['maximum'] ) && ! isset( $args['minimum'] ) ) { 
  1101.           if ( ! empty( $args['exclusiveMaximum'] ) && $value >= $args['maximum'] ) { 
  1102.               /** translators: 1: parameter, 2: maximum number */ 
  1103.               return new WP_Error( 'rest_invalid_param', sprintf( __( '%1$s must be less than %2$d (exclusive)' ), $param, $args['maximum'] ) );
