_wp_batch_split_terms

Splits a batch of shared taxonomy terms.

Description

_wp_batch_split_terms(); 

Usage

  1. if ( !function_exists( '_wp_batch_split_terms' ) ) { 
  2.     require_once ABSPATH . WPINC . '/taxonomy.php'; 
  3. } 
  4.  
  5.   
  6. // NOTICE! Understand what this does before running. 
  7. $result = _wp_batch_split_terms(); 
  8.

  2.     global $wpdb; 
  3.  
  4.     $lock_name = 'term_split.lock'; 
  5.  
  6.     // Try to lock. 
  7.     $lock_result = $wpdb->query( $wpdb->prepare( "INSERT IGNORE INTO `$wpdb->options` ( `option_name`, `option_value`, `autoload` ) VALUES (%s, %s, 'no') /* LOCK */", $lock_name, time() ) ); 
  8.  
  9.     if ( ! $lock_result ) { 
  10.         $lock_result = get_option( $lock_name ); 
  11.

  13.         if ( ! $lock_result || ( $lock_result > ( time() - HOUR_IN_SECONDS ) ) ) { 
  14.             wp_schedule_single_event( time() + ( 5 * MINUTE_IN_SECONDS ), 'wp_split_shared_term_batch' ); 
  15.             return; 
  16.         } 
  17.     } 
  18.  
  19.     // Update the lock, as by this point we've definitely got a lock, just need to fire the actions. 
  20.     update_option( $lock_name, time() ); 
  21.

  23.     $shared_terms = $wpdb->get_results( 
  24.         "SELECT tt.term_id, t.*, count(*) as term_tt_count FROM {$wpdb->term_taxonomy} tt 
  25.          LEFT JOIN {$wpdb->terms} t ON t.term_id = tt.term_id 
  26.          GROUP BY t.term_id 
  27.          HAVING term_tt_count > 1 
  28.          LIMIT 10" 
  29.  ); 
  30.  
  31.     // No more terms, we're done here. 
  32.     if ( ! $shared_terms ) { 
  33.         update_option( 'finished_splitting_shared_terms', true ); 
  34.         delete_option( $lock_name ); 
  35.         return; 
  36.     } 
  37.  
  38.     // Shared terms found? We'll need to run this script again. 
  39.     wp_schedule_single_event( time() + ( 2 * MINUTE_IN_SECONDS ), 'wp_split_shared_term_batch' ); 
  40.  
  41.     // Rekey shared term array for faster lookups. 
  42.     $_shared_terms = array(); 
  43.     foreach ( $shared_terms as $shared_term ) { 
  44.         $term_id = intval( $shared_term->term_id ); 
  45.         $_shared_terms[ $term_id ] = $shared_term; 
  46.     } 
  47.     $shared_terms = $_shared_terms; 
  48.  
  49.     // Get term taxonomy data for all shared terms. 
  50.     $shared_term_ids = implode( ', ', array_keys( $shared_terms ) ); 
  51.     $shared_tts = $wpdb->get_results( "SELECT * FROM {$wpdb->term_taxonomy} WHERE `term_id` IN ({$shared_term_ids})" ); 
  52.  
  53.     // Split term data recording is slow, so we do it just once, outside the loop. 
  54.     $split_term_data = get_option( '_split_terms', array() ); 
  55.     $skipped_first_term = $taxonomies = array(); 
  56.     foreach ( $shared_tts as $shared_tt ) { 
  57.         $term_id = intval( $shared_tt->term_id ); 
  58.  
  59.         // Don't split the first tt belonging to a given term_id. 
  60.         if ( ! isset( $skipped_first_term[ $term_id ] ) ) { 
  61.             $skipped_first_term[ $term_id ] = 1; 
  62.             continue; 
  63.         } 
  64.  
  65.         if ( ! isset( $split_term_data[ $term_id ] ) ) { 
  66.             $split_term_data[ $term_id ] = array(); 
  67.         } 
  68.  
  69.         // Keep track of taxonomies whose hierarchies need flushing. 
  70.         if ( ! isset( $taxonomies[ $shared_tt->taxonomy ] ) ) { 
  71.             $taxonomies[ $shared_tt->taxonomy ] = 1; 
  72.         } 
  73.  
  74.         // Split the term. 
  75.         $split_term_data[ $term_id ][ $shared_tt->taxonomy ] = _split_shared_term( $shared_terms[ $term_id ], $shared_tt, false ); 
  76.     } 
  77.  
  78.     // Rebuild the cached hierarchy for each affected taxonomy. 
  79.     foreach ( array_keys( $taxonomies ) as $tax ) { 
  80.         delete_option( "{$tax}_children" ); 
  81.         _get_term_hierarchy( $tax ); 
  82.     } 
  83.  
  84.     update_option( '_split_terms', $split_term_data ); 
  85.  
  86.     delete_option( $lock_name ); 
  87. }
