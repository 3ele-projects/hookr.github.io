install_plugin_install_status

Determine the status we can perform on a plugin.

Description

(array) install_plugin_install_status( (array|object) $api, (constant) $loop = false ); 

Returns (array)

{ Plugin installation status data. @type string $status Status of a plugin. Could be one of 'install', 'update_available', 'latest_installed' or 'newer_installed'. @type string $url Plugin installation URL. @type string $version The most recent version of the plugin. @type string $file Plugin filename relative to the plugins directory. }

Parameters (2)

0. $api (array|object)
Data about the plugin retrieved from the API.
1. $loop — Optional. (constant) => false
Disable further loops. Default false.

  1. if ( !function_exists( 'install_plugin_install_status' ) ) {

  9. $loop = false;

  1. function install_plugin_install_status($api, $loop = false) { 
  2.     // This function is called recursively, $loop prevents further loops. 
  3.     if ( is_array($api) ) 
  4.         $api = (object) $api; 
  5.  
  6.     // Default to a "new" plugin 
  7.     $status = 'install'; 
  8.     $url = false; 
  9.     $update_file = false; 
  10.  
  11.     /** 
  12.      * Check to see if this plugin is known to be installed,  
  13.      * and has an update awaiting it. 
  14.      */ 
  15.     $update_plugins = get_site_transient('update_plugins'); 
  16.     if ( isset( $update_plugins->response ) ) { 
  17.         foreach ( (array)$update_plugins->response as $file => $plugin ) { 
  18.             if ( $plugin->slug === $api->slug ) { 
  19.                 $status = 'update_available'; 
  20.                 $update_file = $file; 
  21.                 $version = $plugin->new_version; 
  22.                 if ( current_user_can('update_plugins') ) 
  23.                     $url = wp_nonce_url(self_admin_url('update.php?action=upgrade-plugin&plugin=' . $update_file), 'upgrade-plugin_' . $update_file); 
  24.                 break; 
  25.             } 
  26.         } 
  27.     } 
  28.  
  29.     if ( 'install' == $status ) { 
  30.         if ( is_dir( WP_PLUGIN_DIR . '/' . $api->slug ) ) { 
  31.             $installed_plugin = get_plugins('/' . $api->slug); 
  32.             if ( empty($installed_plugin) ) { 
  33.                 if ( current_user_can('install_plugins') ) 
  34.                     $url = wp_nonce_url(self_admin_url('update.php?action=install-plugin&plugin=' . $api->slug), 'install-plugin_' . $api->slug); 
  35.             } else { 
  36.                 $key = array_keys( $installed_plugin ); 
  37.                 $key = reset( $key ); //Use the first plugin regardless of the name, Could have issues for multiple-plugins in one directory if they share different version numbers 
  38.                 $update_file = $api->slug . '/' . $key; 
  39.                 if ( version_compare($api->version, $installed_plugin[ $key ]['Version'], '=') ) { 
  40.                     $status = 'latest_installed'; 
  41.                 } elseif ( version_compare($api->version, $installed_plugin[ $key ]['Version'], '<') ) { 
  42.                     $status = 'newer_installed'; 
  43.                     $version = $installed_plugin[ $key ]['Version']; 
  44.                 } else { 
  45.                     //If the above update check failed, Then that probably means that the update checker has out-of-date information, force a refresh 
  46.                     if ( ! $loop ) { 
  47.                         delete_site_transient('update_plugins'); 
  48.                         wp_update_plugins(); 
  49.                         return install_plugin_install_status($api, true); 
  50.                     } 
  51.                 } 
  52.             } 
  53.         } else { 
  54.             // "install" & no directory with that slug 
  55.             if ( current_user_can('install_plugins') ) 
  56.                 $url = wp_nonce_url(self_admin_url('update.php?action=install-plugin&plugin=' . $api->slug), 'install-plugin_' . $api->slug); 
  57.         } 
  58.     } 
  59.     if ( isset($_GET['from']) ) 
  60.         $url .= '&from=' . urlencode( wp_unslash( $_GET['from'] ) ); 
  61.  
  62.     $file = $update_file; 
  63.     return compact( 'status', 'url', 'version', 'file' ); 
  64. }
