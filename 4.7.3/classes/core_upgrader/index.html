  211.      * @access public

  218.     public static function should_update_to_version( $offered_ver ) {

  220.  
  221.         $current_branch = implode( '.', array_slice( preg_split( '/[.-]/', $wp_version ), 0, 2 ) ); // x.y 
  222.         $new_branch = implode( '.', array_slice( preg_split( '/[.-]/', $offered_ver ), 0, 2 ) ); // x.y 
  223.         $current_is_development_version = (bool) strpos( $wp_version, '-' ); 
  224.  
  225.         // Defaults: 
  226.         $upgrade_dev = true; 
  227.         $upgrade_minor = true; 
  228.         $upgrade_major = false; 
  229.  
  230.         // WP_AUTO_UPDATE_CORE = true (all), 'minor', false. 
  231.         if ( defined( 'WP_AUTO_UPDATE_CORE' ) ) { 
  232.             if ( false === WP_AUTO_UPDATE_CORE ) { 
  233.                 // Defaults to turned off, unless a filter allows it 
  234.                 $upgrade_dev = $upgrade_minor = $upgrade_major = false; 
  235.             } elseif ( true === WP_AUTO_UPDATE_CORE ) { 
  236.                 // ALL updates for core 
  237.                 $upgrade_dev = $upgrade_minor = $upgrade_major = true; 
  238.             } elseif ( 'minor' === WP_AUTO_UPDATE_CORE ) { 
  239.                 // Only minor updates for core 
  240.                 $upgrade_dev = $upgrade_major = false; 
  241.                 $upgrade_minor = true; 
  242.             } 
  243.         } 
  244.  
  245.         // 1: If we're already on that version, not much point in updating? 
  246.         if ( $offered_ver == $wp_version ) 
  247.             return false; 
  248.  
  249.         // 2: If we're running a newer version, that's a nope 
  250.         if ( version_compare( $wp_version, $offered_ver, '>' ) ) 
  251.             return false; 
  252.  
  253.         $failure_data = get_site_option( 'auto_core_update_failed' ); 
  254.         if ( $failure_data ) { 
  255.             // If this was a critical update failure, cannot update. 
  256.             if ( ! empty( $failure_data['critical'] ) ) 
  257.                 return false; 
  258.  
  259.             // Don't claim we can update on update-core.php if we have a non-critical failure logged. 
  260.             if ( $wp_version == $failure_data['current'] && false !== strpos( $offered_ver, '.1.next.minor' ) ) 
  261.                 return false; 
  262.  
  263.             // Cannot update if we're retrying the same A to B update that caused a non-critical failure. 
  264.             // Some non-critical failures do allow retries, like download_failed. 
  265.             // 3.7.1 => 3.7.2 resulted in files_not_writable, if we are still on 3.7.1 and still trying to update to 3.7.2. 
  266.             if ( empty( $failure_data['retry'] ) && $wp_version == $failure_data['current'] && $offered_ver == $failure_data['attempted'] ) 
  267.                 return false; 
  268.         } 
  269.  
  270.         // 3: 3.7-alpha-25000 -> 3.7-alpha-25678 -> 3.7-beta1 -> 3.7-beta2 
  271.         if ( $current_is_development_version ) { 
  272.  
  273.             /** 
  274.              * Filters whether to enable automatic core updates for development versions. 
  275.              * 
  276.              * @since 3.7.0 
  277.              * 
  278.              * @param bool $upgrade_dev Whether to enable automatic updates for 
  279.              *                          development versions. 
  280.              */ 
  281.             if ( ! apply_filters( 'allow_dev_auto_core_updates', $upgrade_dev ) ) 
  282.                 return false; 
  283.             // Else fall through to minor + major branches below. 
  284.         } 
  285.  
  286.         // 4: Minor In-branch updates (3.7.0 -> 3.7.1 -> 3.7.2 -> 3.7.4) 
  287.         if ( $current_branch == $new_branch ) { 
  288.  
  289.             /** 
  290.              * Filters whether to enable minor automatic core updates. 
  291.              * 
  292.              * @since 3.7.0 
  293.              * 
  294.              * @param bool $upgrade_minor Whether to enable minor automatic core updates. 
  295.              */ 
  296.             return apply_filters( 'allow_minor_auto_core_updates', $upgrade_minor ); 
  297.         } 
  298.  
  299.         // 5: Major version updates (3.7.0 -> 3.8.0 -> 3.9.1) 
  300.         if ( version_compare( $new_branch, $current_branch, '>' ) ) { 
  301.  
  302.             /** 
  303.              * Filters whether to enable major automatic core updates. 
  304.              * 
  305.              * @since 3.7.0 
  306.              * 
  307.              * @param bool $upgrade_major Whether to enable major automatic core updates. 
  308.              */ 
  309.             return apply_filters( 'allow_major_auto_core_updates', $upgrade_major ); 
  310.         } 
  311.  
  312.         // If we're not sure, we don't want it 
  313.         return false; 
  314.     }
