  479.

  482.             foreach ( $statuses as $status ) {

  484.                     case 'hold' : 
  485.                         $status_clauses[] = "comment_approved = '0'"; 
  486.                         break; 
  487.  
  488.                     case 'approve' : 
  489.                         $status_clauses[] = "comment_approved = '1'";

  491.  
  492.                     case 'all' :

  498.                         $status_clauses[] = $wpdb->prepare( "comment_approved = %s", $status ); 
  499.                         break; 
  500.                 } 
  501.             } 
  502.  
  503.             if ( ! empty( $status_clauses ) ) { 
  504.                 $approved_clauses[] = '( ' . implode( ' OR ', $status_clauses ) . ' )'; 
  505.             } 
  506.         } 
  507.  
  508.         // User IDs or emails whose unapproved comments are included, regardless of $status. 
  509.         if ( ! empty( $this->query_vars['include_unapproved'] ) ) { 
  510.             $include_unapproved = $this->query_vars['include_unapproved']; 
  511.  
  512.             // Accepts arrays or comma-separated strings. 
  513.             if ( ! is_array( $include_unapproved ) ) { 
  514.                 $include_unapproved = preg_split( '/[\s, ]+/', $include_unapproved ); 
  515.             } 
  516.  
  517.             $unapproved_ids = $unapproved_emails = array(); 
  518.             foreach ( $include_unapproved as $unapproved_identifier ) { 
  519.                 // Numeric values are assumed to be user ids. 
  520.                 if ( is_numeric( $unapproved_identifier ) ) { 
  521.                     $approved_clauses[] = $wpdb->prepare( "( user_id = %d AND comment_approved = '0' )", $unapproved_identifier ); 
  522.  
  523.                 // Otherwise we match against email addresses. 
  524.                 } else { 
  525.                     $approved_clauses[] = $wpdb->prepare( "( comment_author_email = %s AND comment_approved = '0' )", $unapproved_identifier ); 
  526.                 } 
  527.             } 
  528.         } 
  529.  
  530.         // Collapse comment_approved clauses into a single OR-separated clause. 
  531.         if ( ! empty( $approved_clauses ) ) { 
  532.             if ( 1 === count( $approved_clauses ) ) { 
  533.                 $this->sql_clauses['where']['approved'] = $approved_clauses[0]; 
  534.             } else { 
  535.                 $this->sql_clauses['where']['approved'] = '( ' . implode( ' OR ', $approved_clauses ) . ' )'; 
  536.             } 
  537.         } 
  538.  
  539.         $order = ( 'ASC' == strtoupper( $this->query_vars['order'] ) ) ? 'ASC' : 'DESC'; 
  540.  
  541.         // Disable ORDER BY with 'none', an empty array, or boolean false. 
  542.         if ( in_array( $this->query_vars['orderby'], array( 'none', array(), false ), true ) ) { 
  543.             $orderby = ''; 
  544.         } elseif ( ! empty( $this->query_vars['orderby'] ) ) { 
  545.             $ordersby = is_array( $this->query_vars['orderby'] ) ? 
  546.                 $this->query_vars['orderby'] : 
  547.                 preg_split( '/[, \s]/', $this->query_vars['orderby'] ); 
  548.  
  549.             $orderby_array = array(); 
  550.             $found_orderby_comment_ID = false; 
  551.             foreach ( $ordersby as $_key => $_value ) { 
  552.                 if ( ! $_value ) { 
  553.                     continue; 
  554.                 } 
  555.  
  556.                 if ( is_int( $_key ) ) { 
  557.                     $_orderby = $_value; 
  558.                     $_order = $order; 
  559.                 } else { 
  560.                     $_orderby = $_key; 
  561.                     $_order = $_value; 
  562.                 } 
  563.  
  564.                 if ( ! $found_orderby_comment_ID && in_array( $_orderby, array( 'comment_ID', 'comment__in' ) ) ) { 
  565.                     $found_orderby_comment_ID = true; 
  566.                 } 
  567.  
  568.                 $parsed = $this->parse_orderby( $_orderby ); 
  569.  
  570.                 if ( ! $parsed ) { 
  571.                     continue; 
  572.                 } 
  573.  
  574.                 if ( 'comment__in' === $_orderby ) { 
  575.                     $orderby_array[] = $parsed; 
  576.                     continue; 
  577.                 } 
  578.  
  579.                 $orderby_array[] = $parsed . ' ' . $this->parse_order( $_order ); 
  580.             } 
  581.  
  582.             // If no valid clauses were found, order by comment_date_gmt. 
  583.             if ( empty( $orderby_array ) ) { 
  584.                 $orderby_array[] = "$wpdb->comments.comment_date_gmt $order"; 
  585.             } 
  586.  
  587.             // To ensure determinate sorting, always include a comment_ID clause. 
  588.             if ( ! $found_orderby_comment_ID ) { 
  589.                 $comment_ID_order = ''; 
  590.  
  591.                 // Inherit order from comment_date or comment_date_gmt, if available. 
  592.                 foreach ( $orderby_array as $orderby_clause ) { 
  593.                     if ( preg_match( '/comment_date(?:_gmt)*\ (ASC|DESC)/', $orderby_clause, $match ) ) { 
  594.                         $comment_ID_order = $match[1]; 
  595.                         break; 
  596.                     } 
  597.                 } 
  598.  
  599.                 // If no date-related order is available, use the date from the first available clause. 
  600.                 if ( ! $comment_ID_order ) { 
  601.                     foreach ( $orderby_array as $orderby_clause ) { 
  602.                         if ( false !== strpos( 'ASC', $orderby_clause ) ) { 
  603.                             $comment_ID_order = 'ASC'; 
  604.                         } else { 
  605.                             $comment_ID_order = 'DESC'; 
  606.                         } 
  607.  
  608.                         break; 
  609.                     } 
  610.                 } 
  611.  
  612.                 // Default to DESC. 
  613.                 if ( ! $comment_ID_order ) { 
  614.                     $comment_ID_order = 'DESC'; 
  615.                 } 
  616.  
  617.                 $orderby_array[] = "$wpdb->comments.comment_ID $comment_ID_order"; 
  618.             } 
  619.  
  620.             $orderby = implode( ', ', $orderby_array ); 
  621.         } else { 
  622.             $orderby = "$wpdb->comments.comment_date_gmt $order"; 
  623.         } 
  624.  
  625.         $number = absint( $this->query_vars['number'] ); 
  626.         $offset = absint( $this->query_vars['offset'] ); 
  627.  
  628.         if ( ! empty( $number ) ) { 
  629.             if ( $offset ) { 
  630.                 $limits = 'LIMIT ' . $offset . ', ' . $number; 
  631.             } else { 
  632.                 $limits = 'LIMIT ' . $number; 
  633.             } 
  634.         } 
  635.  
  636.         if ( $this->query_vars['count'] ) { 
  637.             $fields = 'COUNT(*)'; 
  638.         } else { 
  639.             $fields = "$wpdb->comments.comment_ID"; 
  640.         } 
  641.  
  642.         $post_id = absint( $this->query_vars['post_id'] ); 
  643.         if ( ! empty( $post_id ) ) { 
  644.             $this->sql_clauses['where']['post_id'] = $wpdb->prepare( 'comment_post_ID = %d', $post_id ); 
  645.         } 
  646.  
  647.         // Parse comment IDs for an IN clause. 
  648.         if ( ! empty( $this->query_vars['comment__in'] ) ) { 
  649.             $this->sql_clauses['where']['comment__in'] = "$wpdb->comments.comment_ID IN ( " . implode( ', ', wp_parse_id_list( $this->query_vars['comment__in'] ) ) . ' )'; 
  650.         } 
  651.  
  652.         // Parse comment IDs for a NOT IN clause. 
  653.         if ( ! empty( $this->query_vars['comment__not_in'] ) ) { 
  654.             $this->sql_clauses['where']['comment__not_in'] = "$wpdb->comments.comment_ID NOT IN ( " . implode( ', ', wp_parse_id_list( $this->query_vars['comment__not_in'] ) ) . ' )'; 
  655.         } 
  656.  
  657.         // Parse comment parent IDs for an IN clause. 
  658.         if ( ! empty( $this->query_vars['parent__in'] ) ) { 
  659.             $this->sql_clauses['where']['parent__in'] = 'comment_parent IN ( ' . implode( ', ', wp_parse_id_list( $this->query_vars['parent__in'] ) ) . ' )'; 
  660.         } 
  661.  
  662.         // Parse comment parent IDs for a NOT IN clause. 
  663.         if ( ! empty( $this->query_vars['parent__not_in'] ) ) { 
  664.             $this->sql_clauses['where']['parent__not_in'] = 'comment_parent NOT IN ( ' . implode( ', ', wp_parse_id_list( $this->query_vars['parent__not_in'] ) ) . ' )'; 
  665.         } 
  666.  
  667.         // Parse comment post IDs for an IN clause. 
  668.         if ( ! empty( $this->query_vars['post__in'] ) ) { 
  669.             $this->sql_clauses['where']['post__in'] = 'comment_post_ID IN ( ' . implode( ', ', wp_parse_id_list( $this->query_vars['post__in'] ) ) . ' )'; 
  670.         } 
  671.  
  672.         // Parse comment post IDs for a NOT IN clause. 
  673.         if ( ! empty( $this->query_vars['post__not_in'] ) ) { 
  674.             $this->sql_clauses['where']['post__not_in'] = 'comment_post_ID NOT IN ( ' . implode( ', ', wp_parse_id_list( $this->query_vars['post__not_in'] ) ) . ' )'; 
  675.         } 
  676.  
  677.         if ( '' !== $this->query_vars['author_email'] ) { 
  678.             $this->sql_clauses['where']['author_email'] = $wpdb->prepare( 'comment_author_email = %s', $this->query_vars['author_email'] ); 
  679.         } 
  680.  
  681.         if ( '' !== $this->query_vars['author_url'] ) { 
  682.             $this->sql_clauses['where']['author_url'] = $wpdb->prepare( 'comment_author_url = %s', $this->query_vars['author_url'] ); 
  683.         } 
  684.  
  685.         if ( '' !== $this->query_vars['karma'] ) { 
  686.             $this->sql_clauses['where']['karma'] = $wpdb->prepare( 'comment_karma = %d', $this->query_vars['karma'] ); 
  687.         } 
  688.  
  689.         // Filtering by comment_type: 'type', 'type__in', 'type__not_in'. 
  690.         $raw_types = array( 
  691.             'IN' => array_merge( (array) $this->query_vars['type'], (array) $this->query_vars['type__in'] ),  
  692.             'NOT IN' => (array) $this->query_vars['type__not_in'],  
  693.  ); 
  694.  
  695.         $comment_types = array(); 
  696.         foreach ( $raw_types as $operator => $_raw_types ) { 
  697.             $_raw_types = array_unique( $_raw_types ); 
  698.  
  699.             foreach ( $_raw_types as $type ) { 
  700.                 switch ( $type ) { 
  701.                     // An empty translates to 'all', for backward compatibility 
  702.                     case '': 
  703.                     case 'all' : 
  704.                         break; 
  705.  
  706.                     case 'comment': 
  707.                     case 'comments': 
  708.                         $comment_types[ $operator ][] = "''"; 
  709.                         break; 
  710.  
  711.                     case 'pings': 
  712.                         $comment_types[ $operator ][] = "'pingback'"; 
  713.                         $comment_types[ $operator ][] = "'trackback'"; 
  714.                         break; 
  715.  
  716.                     default: 
  717.                         $comment_types[ $operator ][] = $wpdb->prepare( '%s', $type ); 
  718.                         break; 
  719.                 } 
  720.             } 
  721.  
  722.             if ( ! empty( $comment_types[ $operator ] ) ) { 
  723.                 $types_sql = implode( ', ', $comment_types[ $operator ] ); 
  724.                 $this->sql_clauses['where']['comment_type__' . strtolower( str_replace( ' ', '_', $operator ) ) ] = "comment_type $operator ($types_sql)"; 
  725.             } 
  726.         } 
  727.  
  728.         $parent = $this->query_vars['parent']; 
  729.         if ( $this->query_vars['hierarchical'] && ! $parent ) { 
  730.             $parent = 0; 
  731.         } 
  732.  
  733.         if ( '' !== $parent ) { 
  734.             $this->sql_clauses['where']['parent'] = $wpdb->prepare( 'comment_parent = %d', $parent ); 
  735.         } 
  736.  
  737.         if ( is_array( $this->query_vars['user_id'] ) ) { 
  738.             $this->sql_clauses['where']['user_id'] = 'user_id IN (' . implode( ', ', array_map( 'absint', $this->query_vars['user_id'] ) ) . ')'; 
  739.         } elseif ( '' !== $this->query_vars['user_id'] ) { 
  740.             $this->sql_clauses['where']['user_id'] = $wpdb->prepare( 'user_id = %d', $this->query_vars['user_id'] ); 
  741.         } 
  742.  
  743.         // Falsy search strings are ignored. 
  744.         if ( strlen( $this->query_vars['search'] ) ) { 
  745.             $search_sql = $this->get_search_sql( 
  746.                 $this->query_vars['search'],  
  747.                 array( 'comment_author', 'comment_author_email', 'comment_author_url', 'comment_author_IP', 'comment_content' ) 
  748.  ); 
  749.  
  750.             // Strip leading 'AND'. 
  751.             $this->sql_clauses['where']['search'] = preg_replace( '/^\s*AND\s*/', '', $search_sql ); 
  752.         } 
  753.  
  754.         // If any post-related query vars are passed, join the posts table. 
  755.         $join_posts_table = false; 
  756.         $plucked = wp_array_slice_assoc( $this->query_vars, array( 'post_author', 'post_name', 'post_parent' ) ); 
  757.         $post_fields = array_filter( $plucked ); 
  758.  
  759.         if ( ! empty( $post_fields ) ) { 
  760.             $join_posts_table = true; 
  761.             foreach ( $post_fields as $field_name => $field_value ) { 
  762.                 // $field_value may be an array. 
  763.                 $esses = array_fill( 0, count( (array) $field_value ), '%s' ); 
  764.                 $this->sql_clauses['where'][ $field_name ] = $wpdb->prepare( " {$wpdb->posts}.{$field_name} IN (" . implode( ', ', $esses ) . ')', $field_value ); 
  765.             } 
  766.         } 
  767.  
  768.         // 'post_status' and 'post_type' are handled separately, due to the specialized behavior of 'any'. 
  769.         foreach ( array( 'post_status', 'post_type' ) as $field_name ) { 
  770.             $q_values = array(); 
  771.             if ( ! empty( $this->query_vars[ $field_name ] ) ) { 
  772.                 $q_values = $this->query_vars[ $field_name ]; 
  773.                 if ( ! is_array( $q_values ) ) { 
  774.                     $q_values = explode( ', ', $q_values ); 
  775.                 } 
  776.  
  777.                 // 'any' will cause the query var to be ignored. 
  778.                 if ( in_array( 'any', $q_values, true ) || empty( $q_values ) ) { 
  779.                     continue; 
  780.                 } 
  781.  
  782.                 $join_posts_table = true; 
  783.  
  784.                 $esses = array_fill( 0, count( $q_values ), '%s' ); 
  785.                 $this->sql_clauses['where'][ $field_name ] = $wpdb->prepare( " {$wpdb->posts}.{$field_name} IN (" . implode( ', ', $esses ) . ")", $q_values ); 
  786.             } 
  787.         } 
  788.  
  789.         // Comment author IDs for an IN clause. 
  790.         if ( ! empty( $this->query_vars['author__in'] ) ) { 
  791.             $this->sql_clauses['where']['author__in'] = 'user_id IN ( ' . implode( ', ', wp_parse_id_list( $this->query_vars['author__in'] ) ) . ' )'; 
  792.         } 
  793.  
  794.         // Comment author IDs for a NOT IN clause. 
  795.         if ( ! empty( $this->query_vars['author__not_in'] ) ) { 
  796.             $this->sql_clauses['where']['author__not_in'] = 'user_id NOT IN ( ' . implode( ', ', wp_parse_id_list( $this->query_vars['author__not_in'] ) ) . ' )'; 
  797.         } 
  798.  
  799.         // Post author IDs for an IN clause. 
  800.         if ( ! empty( $this->query_vars['post_author__in'] ) ) { 
  801.             $join_posts_table = true; 
  802.             $this->sql_clauses['where']['post_author__in'] = 'post_author IN ( ' . implode( ', ', wp_parse_id_list( $this->query_vars['post_author__in'] ) ) . ' )'; 
  803.         } 
  804.  
  805.         // Post author IDs for a NOT IN clause. 
  806.         if ( ! empty( $this->query_vars['post_author__not_in'] ) ) { 
  807.             $join_posts_table = true; 
  808.             $this->sql_clauses['where']['post_author__not_in'] = 'post_author NOT IN ( ' . implode( ', ', wp_parse_id_list( $this->query_vars['post_author__not_in'] ) ) . ' )'; 
  809.         } 
  810.  
  811.         $join = ''; 
  812.  
  813.         if ( $join_posts_table ) { 
  814.             $join .= "JOIN $wpdb->posts ON $wpdb->posts.ID = $wpdb->comments.comment_post_ID"; 
  815.         } 
  816.  
  817.         if ( ! empty( $this->meta_query_clauses ) ) { 
  818.             $join .= $this->meta_query_clauses['join']; 
  819.  
  820.             // Strip leading 'AND'. 
  821.             $this->sql_clauses['where']['meta_query'] = preg_replace( '/^\s*AND\s*/', '', $this->meta_query_clauses['where'] ); 
  822.  
  823.             if ( ! $this->query_vars['count'] ) { 
  824.                 $groupby = "{$wpdb->comments}.comment_ID"; 
  825.             } 
  826.         } 
  827.  
  828.         if ( ! empty( $this->query_vars['date_query'] ) && is_array( $this->query_vars['date_query'] ) ) { 
  829.             $this->date_query = new WP_Date_Query( $this->query_vars['date_query'], 'comment_date' ); 
  830.             $this->sql_clauses['where']['date_query'] = preg_replace( '/^\s*AND\s*/', '', $this->date_query->get_sql() ); 
  831.         } 
  832.  
  833.         $where = implode( ' AND ', $this->sql_clauses['where'] ); 
  834.  
  835.         $pieces = array( 'fields', 'join', 'where', 'orderby', 'limits', 'groupby' ); 
  836.         /** 
  837.          * Filters the comment query clauses. 
  838.          * 
  839.          * @since 3.1.0 
  840.          * 
  841.          * @param array            $pieces A compacted array of comment query clauses. 
  842.          * @param WP_Comment_Query &$this  Current instance of WP_Comment_Query, passed by reference. 
  843.          */ 
  844.         $clauses = apply_filters_ref_array( 'comments_clauses', array( compact( $pieces ), &$this ) ); 
  845.  
  846.         $fields = isset( $clauses[ 'fields' ] ) ? $clauses[ 'fields' ] : ''; 
  847.         $join = isset( $clauses[ 'join' ] ) ? $clauses[ 'join' ] : ''; 
  848.         $where = isset( $clauses[ 'where' ] ) ? $clauses[ 'where' ] : ''; 
  849.         $orderby = isset( $clauses[ 'orderby' ] ) ? $clauses[ 'orderby' ] : ''; 
  850.         $limits = isset( $clauses[ 'limits' ] ) ? $clauses[ 'limits' ] : ''; 
  851.         $groupby = isset( $clauses[ 'groupby' ] ) ? $clauses[ 'groupby' ] : ''; 
  852.  
  853.         $this->filtered_where_clause = $where; 
  854.  
  855.         if ( $where ) { 
  856.             $where = 'WHERE ' . $where; 
  857.         } 
  858.  
  859.         if ( $groupby ) { 
  860.             $groupby = 'GROUP BY ' . $groupby; 
  861.         } 
  862.  
  863.         if ( $orderby ) { 
  864.             $orderby = "ORDER BY $orderby"; 
  865.         } 
  866.  
  867.         $found_rows = ''; 
  868.         if ( ! $this->query_vars['no_found_rows'] ) { 
  869.             $found_rows = 'SQL_CALC_FOUND_ROWS'; 
  870.         } 
  871.  
  872.         $this->sql_clauses['select'] = "SELECT $found_rows $fields"; 
  873.         $this->sql_clauses['from'] = "FROM $wpdb->comments $join"; 
  874.         $this->sql_clauses['groupby'] = $groupby; 
  875.         $this->sql_clauses['orderby'] = $orderby; 
  876.         $this->sql_clauses['limits'] = $limits; 
  877.  
  878.         $this->request = "{$this->sql_clauses['select']} {$this->sql_clauses['from']} {$where} {$this->sql_clauses['groupby']} {$this->sql_clauses['orderby']} {$this->sql_clauses['limits']}"; 
  879.  
  880.         if ( $this->query_vars['count'] ) { 
  881.             return intval( $wpdb->get_var( $this->request ) ); 
  882.         } else { 
  883.             $comment_ids = $wpdb->get_col( $this->request ); 
  884.             return array_map( 'intval', $comment_ids ); 
  885.         } 
  886.     } 
  887.  
  888.     /** 
  889.      * Populates found_comments and max_num_pages properties for the current 
  890.      * query if the limit clause was used. 
  891.      * 
  892.      * @since 4.6.0 
  893.      * @access private 
  894.      * 
  895.      * @global wpdb $wpdb WordPress database abstraction object. 
  896.      */ 
  897.     private function set_found_comments() { 
  898.         global $wpdb; 
  899.  
  900.         if ( $this->query_vars['number'] && ! $this->query_vars['no_found_rows'] ) { 
  901.             /** 
  902.              * Filters the query used to retrieve found comment count. 
  903.              * 
  904.              * @since 4.4.0 
  905.              * 
  906.              * @param string           $found_comments_query SQL query. Default 'SELECT FOUND_ROWS()'. 
  907.              * @param WP_Comment_Query $comment_query        The `WP_Comment_Query` instance. 
  908.              */ 
  909.             $found_comments_query = apply_filters( 'found_comments_query', 'SELECT FOUND_ROWS()', $this ); 
  910.  
  911.             $this->found_comments = (int) $wpdb->get_var( $found_comments_query ); 
  912.         } 
  913.     } 
  914.  
  915.     /** 
  916.      * Fetch descendants for located comments. 
  917.      * 
  918.      * Instead of calling `get_children()` separately on each child comment, we do a single set of queries to fetch 
  919.      * the descendant trees for all matched top-level comments. 
  920.      * 
  921.      * @since 4.4.0 
  922.      * 
  923.      * @param array $comments Array of top-level comments whose descendants should be filled in. 
  924.      * @return array 
  925.      */ 
  926.     protected function fill_descendants( $comments ) { 
  927.         global $wpdb; 
  928.  
  929.         $levels = array( 
  930.             0 => wp_list_pluck( $comments, 'comment_ID' ),  
  931.  ); 
  932.  
  933.         /** 
  934.          * The WHERE clause for the descendant query is the same as for the top-level 
  935.          * query, minus the `parent`, `parent__in`, and `parent__not_in` sub-clauses. 
  936.          */ 
  937.         $_where = $this->filtered_where_clause; 
  938.         $exclude_keys = array( 'parent', 'parent__in', 'parent__not_in' ); 
  939.         foreach ( $exclude_keys as $exclude_key ) { 
  940.             if ( isset( $this->sql_clauses['where'][ $exclude_key ] ) ) { 
  941.                 $clause = $this->sql_clauses['where'][ $exclude_key ]; 
  942.  
  943.                 // Strip the clause as well as any adjacent ANDs. 
  944.                 $pattern = '|(?:AND)?\s*' . $clause . '\s*(?:AND)?|'; 
  945.                 $_where_parts = preg_split( $pattern, $_where ); 
  946.  
  947.                 // Remove empties. 
  948.                 $_where_parts = array_filter( array_map( 'trim', $_where_parts ) ); 
  949.  
  950.                 // Reassemble with an AND. 
  951.                 $_where = implode( ' AND ', $_where_parts ); 
  952.             } 
  953.         } 
  954.  
  955.         $key = md5( serialize( wp_array_slice_assoc( $this->query_vars, array_keys( $this->query_var_defaults ) ) ) ); 
  956.         $last_changed = wp_cache_get_last_changed( 'comment' ); 
  957.  
  958.         // Fetch an entire level of the descendant tree at a time. 
  959.         $level = 0; 
  960.         do { 
  961.             // Parent-child relationships may be cached. Only query for those that are not. 
  962.             $child_ids = $uncached_parent_ids = array(); 
  963.             $_parent_ids = $levels[ $level ]; 
  964.             foreach ( $_parent_ids as $parent_id ) { 
  965.                 $cache_key = "get_comment_child_ids:$parent_id:$key:$last_changed"; 
  966.                 $parent_child_ids = wp_cache_get( $cache_key, 'comment' ); 
  967.                 if ( false !== $parent_child_ids ) { 
  968.                     $child_ids = array_merge( $child_ids, $parent_child_ids ); 
  969.                 } else { 
  970.                     $uncached_parent_ids[] = $parent_id; 
  971.                 } 
  972.             } 
  973.  
  974.             if ( $uncached_parent_ids ) { 
  975.                 // Fetch this level of comments. 
  976.                 $parent_query_args = $this->query_vars; 
  977.                 foreach ( $exclude_keys as $exclude_key ) { 
  978.                     $parent_query_args[ $exclude_key ] = ''; 
  979.                 } 
  980.                 $parent_query_args['parent__in'] = $uncached_parent_ids; 
  981.                 $parent_query_args['no_found_rows'] = true; 
  982.                 $parent_query_args['hierarchical'] = false; 
  983.                 $parent_query_args['offset'] = 0; 
  984.                 $parent_query_args['number'] = 0; 
  985.  
  986.                 $level_comments = get_comments( $parent_query_args ); 
  987.  
  988.                 // Cache parent-child relationships. 
  989.                 $parent_map = array_fill_keys( $uncached_parent_ids, array() ); 
  990.                 foreach ( $level_comments as $level_comment ) { 
  991.                     $parent_map[ $level_comment->comment_parent ][] = $level_comment->comment_ID; 
  992.                     $child_ids[] = $level_comment->comment_ID; 
  993.                 } 
  994.  
  995.                 foreach ( $parent_map as $parent_id => $children ) { 
  996.                     $cache_key = "get_comment_child_ids:$parent_id:$key:$last_changed"; 
  997.                     wp_cache_set( $cache_key, $children, 'comment' ); 
  998.                 } 
  999.             } 
  1000.  
  1001.             $level++; 
  1002.             $levels[ $level ] = $child_ids; 
  1003.         } while ( $child_ids ); 
  1004.  
  1005.         // Prime comment caches for non-top-level comments. 
  1006.         $descendant_ids = array(); 
  1007.         for ( $i = 1, $c = count( $levels ); $i < $c; $i++ ) { 
  1008.             $descendant_ids = array_merge( $descendant_ids, $levels[ $i ] ); 
  1009.         } 
  1010.  
  1011.         _prime_comment_caches( $descendant_ids, $this->query_vars['update_comment_meta_cache'] ); 
  1012.  
  1013.         // Assemble a flat array of all comments + descendants. 
  1014.         $all_comments = $comments; 
  1015.         foreach ( $descendant_ids as $descendant_id ) { 
  1016.             $all_comments[] = get_comment( $descendant_id ); 
  1017.         } 
  1018.  
  1019.         // If a threaded representation was requested, build the tree. 
  1020.         if ( 'threaded' === $this->query_vars['hierarchical'] ) { 
  1021.             $threaded_comments = $ref = array(); 
  1022.             foreach ( $all_comments as $k => $c ) { 
  1023.                 $_c = get_comment( $c->comment_ID ); 
  1024.  
  1025.                 // If the comment isn't in the reference array, it goes in the top level of the thread. 
  1026.                 if ( ! isset( $ref[ $c->comment_parent ] ) ) { 
  1027.                     $threaded_comments[ $_c->comment_ID ] = $_c; 
  1028.                     $ref[ $_c->comment_ID ] = $threaded_comments[ $_c->comment_ID ]; 
  1029.  
  1030.                 // Otherwise, set it as a child of its parent. 
  1031.                 } else { 
  1032.  
  1033.                     $ref[ $_c->comment_parent ]->add_child( $_c ); 
  1034.                     $ref[ $_c->comment_ID ] = $ref[ $_c->comment_parent ]->get_child( $_c->comment_ID ); 
  1035.                 } 
  1036.             } 
  1037.  
  1038.             // Set the 'populated_children' flag, to ensure additional database queries aren't run. 
  1039.             foreach ( $ref as $_ref ) { 
  1040.                 $_ref->populated_children( true ); 
  1041.             } 
  1042.  
  1043.             $comments = $threaded_comments; 
  1044.         } else { 
  1045.             $comments = $all_comments; 
  1046.         } 
  1047.  
  1048.         return $comments; 
  1049.     }

  1052.      * Used internally to generate an SQL string for searching across multiple columns 
  1053.      * 
  1054.      * @since 3.1.0 
  1055.      * @access protected 
  1056.      * 
  1057.      * @global wpdb $wpdb WordPress database abstraction object. 
  1058.      * 
  1059.      * @param string $string 
  1060.      * @param array $cols 
  1061.      * @return string 
  1062.      */ 
  1063.     protected function get_search_sql( $string, $cols ) {

  1065.

  1078.      * 
  1079.      * @since 4.2.0 
  1080.      * @access protected 
  1081.      *

  1083.      * 
  1084.      * @param string $orderby Alias for the field to order by.

  1087.     protected function parse_orderby( $orderby ) {

  1089.  
  1090.         $allowed_keys = array( 
  1091.             'comment_agent',  
  1092.             'comment_approved',  
  1093.             'comment_author',  
  1094.             'comment_author_email',  
  1095.             'comment_author_IP',  
  1096.             'comment_author_url',  
  1097.             'comment_content',  
  1098.             'comment_date',  
  1099.             'comment_date_gmt',  
  1100.             'comment_ID',  
  1101.             'comment_karma',  
  1102.             'comment_parent',  
  1103.             'comment_post_ID',  
  1104.             'comment_type',  
  1105.             'user_id',  
  1106.  ); 
  1107.  
  1108.         if ( ! empty( $this->query_vars['meta_key'] ) ) { 
  1109.             $allowed_keys[] = $this->query_vars['meta_key']; 
  1110.             $allowed_keys[] = 'meta_value'; 
  1111.             $allowed_keys[] = 'meta_value_num'; 
  1112.         } 
  1113.  
  1114.         $meta_query_clauses = $this->meta_query->get_clauses(); 
  1115.         if ( $meta_query_clauses ) { 
  1116.             $allowed_keys = array_merge( $allowed_keys, array_keys( $meta_query_clauses ) ); 
  1117.         } 
  1118.

  1120.         if ( $orderby == $this->query_vars['meta_key'] || $orderby == 'meta_value' ) { 
  1121.             $parsed = "$wpdb->commentmeta.meta_value"; 
  1122.         } elseif ( $orderby == 'meta_value_num' ) { 
  1123.             $parsed = "$wpdb->commentmeta.meta_value+0";

  1125.             $comment__in = implode( ', ', array_map( 'absint', $this->query_vars['comment__in'] ) ); 
  1126.             $parsed = "FIELD( {$wpdb->comments}.comment_ID, $comment__in )"; 
  1127.         } elseif ( in_array( $orderby, $allowed_keys ) ) { 
  1128.  
  1129.             if ( isset( $meta_query_clauses[ $orderby ] ) ) { 
  1130.                 $meta_clause = $meta_query_clauses[ $orderby ]; 
  1131.                 $parsed = sprintf( "CAST(%s.meta_value AS %s)", esc_sql( $meta_clause['alias'] ), esc_sql( $meta_clause['cast'] ) ); 
  1132.             } else { 
  1133.                 $parsed = "$wpdb->comments.$orderby"; 
  1134.             } 
  1135.         } 
  1136.

  1138.     }

  1140.     /** 
  1141.      * Parse an 'order' query variable and cast it to ASC or DESC as necessary. 
  1142.      *

  1149.     protected function parse_order( $order ) {
