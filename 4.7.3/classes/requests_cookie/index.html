  125.             // Cookies created manually; cookies created by Requests will set 
  126.             // the domain to the requested domain

  130.         $domain_string = $this->attributes['domain']; 
  131.         if ($domain_string === $string) { 
  132.             // The domain string and the string are identical. 
  133.             return true; 
  134.         } 
  135.  
  136.         // If the cookie is marked as host-only and we don't have an exact 
  137.         // match, reject the cookie 
  138.         if ($this->flags['host-only'] === true) { 
  139.             return false; 
  140.         } 
  141.  
  142.         if (strlen($string) <= strlen($domain_string)) { 
  143.             // For obvious reasons, the string cannot be a suffix if the domain 
  144.             // is shorter than the domain string 
  145.             return false; 
  146.         } 
  147.  
  148.         if (substr($string, -1 * strlen($domain_string)) !== $domain_string) { 
  149.             // The domain string should be a suffix of the string. 
  150.             return false; 
  151.         }

  154.         if (substr($prefix, -1) !== '.') {

  156.             // domain string should be a %x2E (".") character. 
  157.             return false; 
  158.         }

  244.                 if (is_int($value)) { 
  245.                     return $value;

  247.  
  248.                 $expiry_time = strtotime($value);

  252.

  254.  
  255.             case 'max-age':

  358.     /** 
  359.      * Parse a cookie string into a cookie object 
  360.      * 
  361.      * Based on Mozilla's parsing code in Firefox and related projects, which 
  362.      * is an intentional deviation from RFC 2109 and RFC 2616. RFC 6265 
  363.      * specifies some of this handling, but not in a thorough manner. 
  364.      * 
  365.      * @param string Cookie header value (from a Set-Cookie header) 
  366.      * @return Requests_Cookie Parsed cookie object 
  367.      */ 
  368.     public static function parse($string, $name = '', $reference_time = null) { 
  369.         $parts = explode(';', $string); 
  370.         $kvparts = array_shift($parts); 
  371.  
  372.         if (!empty($name)) { 
  373.             $value = $string; 
  374.         } 
  375.         elseif (strpos($kvparts, '=') === false) { 
  376.             // Some sites might only have a value without the equals separator. 
  377.             // Deviate from RFC 6265 and pretend it was actually a blank name 
  378.             // (`=foo`) 
  379.             // 
  380.             // https://bugzilla.mozilla.org/show_bug.cgi?id=169091 
  381.             $name = ''; 
  382.             $value = $kvparts; 
  383.         } 
  384.         else { 
  385.             list($name, $value) = explode('=', $kvparts, 2); 
  386.         } 
  387.         $name = trim($name); 
  388.         $value = trim($value); 
  389.  
  390.         // Attribute key are handled case-insensitively 
  391.         $attributes = new Requests_Utility_CaseInsensitiveDictionary(); 
  392.  
  393.         if (!empty($parts)) { 
  394.             foreach ($parts as $part) { 
  395.                 if (strpos($part, '=') === false) { 
  396.                     $part_key = $part; 
  397.                     $part_value = true; 
  398.                 } 
  399.                 else { 
  400.                     list($part_key, $part_value) = explode('=', $part, 2); 
  401.                     $part_value = trim($part_value); 
  402.                 } 
  403.  
  404.                 $part_key = trim($part_key); 
  405.                 $attributes[$part_key] = $part_value; 
  406.             } 
  407.         } 
  408.  
  409.         return new Requests_Cookie($name, $value, $attributes, array(), $reference_time); 
  410.     }

  412.     /** 
  413.      * Parse all Set-Cookie headers from request headers 
  414.      * 
  415.      * @param Requests_Response_Headers $headers Headers to parse from 
  416.      * @param Requests_IRI|null $origin URI for comparing cookie origins 
  417.      * @param int|null $time Reference time for expiration calculation 
  418.      * @return array 
  419.      */ 
  420.     public static function parse_from_headers(Requests_Response_Headers $headers, Requests_IRI $origin = null, $time = null) { 
  421.         $cookie_headers = $headers->getValues('Set-Cookie'); 
  422.         if (empty($cookie_headers)) { 
  423.             return array(); 
  424.         } 
  425.  
  426.         $cookies = array(); 
  427.         foreach ($cookie_headers as $header) { 
  428.             $parsed = self::parse($header, '', $time); 
  429.  
  430.             // Default domain/path attributes 
  431.             if (empty($parsed->attributes['domain']) && !empty($origin)) { 
  432.                 $parsed->attributes['domain'] = $origin->host; 
  433.                 $parsed->flags['host-only'] = true; 
  434.             } 
  435.             else { 
  436.                 $parsed->flags['host-only'] = false; 
  437.             } 
  438.  
  439.             $path_is_valid = (!empty($parsed->attributes['path']) && $parsed->attributes['path'][0] === '/'); 
  440.             if (!$path_is_valid && !empty($origin)) { 
  441.                 $path = $origin->path; 
  442.  
  443.                 // Default path normalization as per RFC 6265 section 5.1.4 
  444.                 if (substr($path, 0, 1) !== '/') { 
  445.                     // If the uri-path is empty or if the first character of 
  446.                     // the uri-path is not a %x2F ("/") character, output 
  447.                     // %x2F ("/") and skip the remaining steps. 
  448.                     $path = '/'; 
  449.                 } 
  450.                 elseif (substr_count($path, '/') === 1) { 
  451.                     // If the uri-path contains no more than one %x2F ("/") 
  452.                     // character, output %x2F ("/") and skip the remaining 
  453.                     // step. 
  454.                     $path = '/'; 
  455.                 } 
  456.                 else { 
  457.                     // Output the characters of the uri-path from the first 
  458.                     // character up to, but not including, the right-most 
  459.                     // %x2F ("/"). 
  460.                     $path = substr($path, 0, strrpos($path, '/')); 
  461.                 } 
  462.                 $parsed->attributes['path'] = $path; 
  463.             } 
  464.  
  465.             // Reject invalid cookie domains 
  466.             if (!empty($origin) && !$parsed->domain_matches($origin->host)) { 
  467.                 continue; 
  468.             } 
  469.  
  470.             $cookies[$parsed->name] = $parsed; 
  471.         } 
  472.  
  473.         return $cookies; 
  474.     }
