  544.      * @param WP_REST_Request $request Full details about the request. 
  545.      * @return bool|WP_Error True if the request has access to delete the item, otherwise false or WP_Error object. 
  546.      */ 
  547.     public function delete_item_permissions_check( $request ) { 
  548.         $term = $this->get_term( $request['id'] ); 
  549.         if ( is_wp_error( $term ) ) { 
  550.             return $term; 
  551.         } 
  552.  
  553.         if ( ! current_user_can( 'delete_term', $term->term_id ) ) { 
  554.             return new WP_Error( 'rest_cannot_delete', __( 'Sorry, you are not allowed to delete this term.' ), array( 'status' => rest_authorization_required_code() ) ); 
  555.         } 
  556.  
  557.         return true;

  560.     /** 
  561.      * Deletes a single term from a taxonomy. 
  562.      * 
  563.      * @since 4.7.0 
  564.      * @access public 
  565.      * 
  566.      * @param WP_REST_Request $request Full details about the request. 
  567.      * @return WP_REST_Response|WP_Error Response object on success, or WP_Error object on failure. 
  568.      */ 
  569.     public function delete_item( $request ) { 
  570.         $term = $this->get_term( $request['id'] ); 
  571.         if ( is_wp_error( $term ) ) { 
  572.             return $term; 
  573.         } 
  574.  
  575.         $force = isset( $request['force'] ) ? (bool) $request['force'] : false; 
  576.  
  577.         // We don't support trashing for terms. 
  578.         if ( ! $force ) { 
  579.             return new WP_Error( 'rest_trash_not_supported', __( 'Terms do not support trashing. Set force=true to delete.' ), array( 'status' => 501 ) ); 
  580.         } 
  581.  
  582.         $request->set_param( 'context', 'view' ); 
  583.  
  584.         $previous = $this->prepare_item_for_response( $term, $request ); 
  585.  
  586.         $retval = wp_delete_term( $term->term_id, $term->taxonomy ); 
  587.  
  588.         if ( ! $retval ) { 
  589.             return new WP_Error( 'rest_cannot_delete', __( 'The term cannot be deleted.' ), array( 'status' => 500 ) ); 
  590.         } 
  591.  
  592.         $response = new WP_REST_Response(); 
  593.         $response->set_data( array( 'deleted' => true, 'previous' => $previous->get_data() ) ); 
  594.  
  595.         /** 
  596.          * Fires after a single term is deleted via the REST API. 
  597.          * 
  598.          * The dynamic portion of the hook name, `$this->taxonomy`, refers to the taxonomy slug. 
  599.          * 
  600.          * @since 4.7.0 
  601.          * 
  602.          * @param WP_Term          $term     The deleted term. 
  603.          * @param WP_REST_Response $response The response data. 
  604.          * @param WP_REST_Request  $request  The request sent to the API. 
  605.          */ 
  606.         do_action( "rest_delete_{$this->taxonomy}", $term, $response, $request ); 
  607.  
  608.         return $response; 
  609.     }

  619.      */ 
  620.     public function prepare_item_for_database( $request ) { 
  621.         $prepared_term = new stdClass; 
  622.  
  623.         $schema = $this->get_item_schema(); 
  624.         if ( isset( $request['name'] ) && ! empty( $schema['properties']['name'] ) ) { 
  625.             $prepared_term->name = $request['name']; 
  626.         } 
  627.  
  628.         if ( isset( $request['slug'] ) && ! empty( $schema['properties']['slug'] ) ) { 
  629.             $prepared_term->slug = $request['slug']; 
  630.         } 
  631.  
  632.         if ( isset( $request['taxonomy'] ) && ! empty( $schema['properties']['taxonomy'] ) ) { 
  633.             $prepared_term->taxonomy = $request['taxonomy']; 
  634.         } 
  635.  
  636.         if ( isset( $request['description'] ) && ! empty( $schema['properties']['description'] ) ) { 
  637.             $prepared_term->description = $request['description']; 
  638.         } 
  639.  
  640.         if ( isset( $request['parent'] ) && ! empty( $schema['properties']['parent'] ) ) {

  644.             if ( $parent_term ) {

  646.             } 
  647.  
  648.             $prepared_term->parent = $parent_term_id; 
  649.         }

  662.     }

  664.     /** 
  665.      * Prepares a single term output for response. 
  666.      * 
  667.      * @since 4.7.0 
  668.      * @access public 
  669.      * 
  670.      * @param obj             $item    Term object. 
  671.      * @param WP_REST_Request $request Request object. 
  672.      * @return WP_REST_Response $response Response object. 
  673.      */ 
  674.     public function prepare_item_for_response( $item, $request ) { 
  675.  
  676.         $schema = $this->get_item_schema(); 
  677.         $data = array(); 
  678.  
  679.         if ( ! empty( $schema['properties']['id'] ) ) { 
  680.             $data['id'] = (int) $item->term_id; 
  681.         } 
  682.  
  683.         if ( ! empty( $schema['properties']['count'] ) ) { 
  684.             $data['count'] = (int) $item->count; 
  685.         } 
  686.  
  687.         if ( ! empty( $schema['properties']['description'] ) ) { 
  688.             $data['description'] = $item->description; 
  689.         } 
  690.  
  691.         if ( ! empty( $schema['properties']['link'] ) ) { 
  692.             $data['link'] = get_term_link( $item ); 
  693.         } 
  694.  
  695.         if ( ! empty( $schema['properties']['name'] ) ) { 
  696.             $data['name'] = $item->name; 
  697.         } 
  698.  
  699.         if ( ! empty( $schema['properties']['slug'] ) ) { 
  700.             $data['slug'] = $item->slug; 
  701.         } 
  702.  
  703.         if ( ! empty( $schema['properties']['taxonomy'] ) ) { 
  704.             $data['taxonomy'] = $item->taxonomy; 
  705.         } 
  706.  
  707.         if ( ! empty( $schema['properties']['parent'] ) ) { 
  708.             $data['parent'] = (int) $item->parent; 
  709.         } 
  710.  
  711.         if ( ! empty( $schema['properties']['meta'] ) ) { 
  712.             $data['meta'] = $this->meta->get_value( $item->term_id, $request ); 
  713.         } 
  714.  
  715.         $context = ! empty( $request['context'] ) ? $request['context'] : 'view'; 
  716.         $data = $this->add_additional_fields_to_object( $data, $request ); 
  717.         $data = $this->filter_response_by_context( $data, $context ); 
  718.  
  719.         $response = rest_ensure_response( $data ); 
  720.  
  721.         $response->add_links( $this->prepare_links( $item ) ); 
  722.  
  723.         /** 
  724.          * Filters a term item returned from the API. 
  725.          * 
  726.          * The dynamic portion of the hook name, `$this->taxonomy`, refers to the taxonomy slug. 
  727.          * 
  728.          * Allows modification of the term data right before it is returned. 
  729.          * 
  730.          * @since 4.7.0 
  731.          * 
  732.          * @param WP_REST_Response  $response  The response object. 
  733.          * @param object            $item      The original term object. 
  734.          * @param WP_REST_Request   $request   Request used to generate the response. 
  735.          */ 
  736.         return apply_filters( "rest_prepare_{$this->taxonomy}", $response, $item, $request ); 
  737.     }

  739.     /** 
  740.      * Prepares links for the request. 
  741.      * 
  742.      * @since 4.7.0 
  743.      * @access protected 
  744.      * 
  745.      * @param object $term Term object. 
  746.      * @return array Links for the given term. 
  747.      */ 
  748.     protected function prepare_links( $term ) { 
  749.         $base = $this->namespace . '/' . $this->rest_base; 
  750.         $links = array( 
  751.             'self' => array( 
  752.                 'href' => rest_url( trailingslashit( $base ) . $term->term_id ),  
  753.  ),  
  754.             'collection' => array( 
  755.                 'href' => rest_url( $base ),  
  756.  ),  
  757.             'about' => array( 
  758.                 'href' => rest_url( sprintf( 'wp/v2/taxonomies/%s', $this->taxonomy ) ),  
  759.  ),  
  760.  ); 
  761.  
  762.         if ( $term->parent ) { 
  763.             $parent_term = get_term( (int) $term->parent, $term->taxonomy ); 
  764.  
  765.             if ( $parent_term ) { 
  766.                 $links['up'] = array( 
  767.                     'href' => rest_url( trailingslashit( $base ) . $parent_term->term_id ),  
  768.                     'embeddable' => true,  
  769.  ); 
  770.             } 
  771.         } 
  772.  
  773.         $taxonomy_obj = get_taxonomy( $term->taxonomy ); 
  774.  
  775.         if ( empty( $taxonomy_obj->object_type ) ) { 
  776.             return $links; 
  777.         } 
  778.  
  779.         $post_type_links = array(); 
  780.  
  781.         foreach ( $taxonomy_obj->object_type as $type ) { 
  782.             $post_type_object = get_post_type_object( $type ); 
  783.  
  784.             if ( empty( $post_type_object->show_in_rest ) ) { 
  785.                 continue; 
  786.             } 
  787.  
  788.             $rest_base = ! empty( $post_type_object->rest_base ) ? $post_type_object->rest_base : $post_type_object->name; 
  789.             $post_type_links[] = array( 
  790.                 'href' => add_query_arg( $this->rest_base, $term->term_id, rest_url( sprintf( 'wp/v2/%s', $rest_base ) ) ),  
  791.  ); 
  792.         } 
  793.  
  794.         if ( ! empty( $post_type_links ) ) { 
  795.             $links['https://api.w.org/post_type'] = $post_type_links; 
  796.         } 
  797.  
  798.         return $links; 
  799.     }
