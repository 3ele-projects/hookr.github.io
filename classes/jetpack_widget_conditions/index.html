  451.                             break; 
  452.                             case 'month': 
  453.                                 $condition_result = is_month(); 
  454.                             break; 
  455.                             case 'day': 
  456.                                 $condition_result = is_day(); 
  457.                             break; 
  458.                             case 'year': 
  459.                                 $condition_result = is_year(); 
  460.                             break; 
  461.                         } 
  462.                     break; 
  463.                     case 'page': 
  464.                         // Previously hardcoded post type options. 
  465.                         if ( 'post' == $rule['minor'] ) 
  466.                             $rule['minor'] = 'post_type-post'; 
  467.                         else if ( ! $rule['minor'] ) 
  468.                             $rule['minor'] = 'post_type-page'; 
  469.  
  470.                         switch ( $rule['minor'] ) { 
  471.                             case '404': 
  472.                                 $condition_result = is_404(); 
  473.                             break; 
  474.                             case 'search': 
  475.                                 $condition_result = is_search(); 
  476.                             break; 
  477.                             case 'archive': 
  478.                                 $condition_result = is_archive(); 
  479.                             break; 
  480.                             case 'posts': 
  481.                                 $condition_result = $wp_query->is_posts_page; 
  482.                             break; 
  483.                             case 'home': 
  484.                                 $condition_result = is_home(); 
  485.                             break; 
  486.                             case 'front': 
  487.                                 if ( current_theme_supports( 'infinite-scroll' ) ) 
  488.                                     $condition_result = is_front_page(); 
  489.                                 else { 
  490.                                     $condition_result = is_front_page() && !is_paged(); 
  491.                                 } 
  492.                             break; 
  493.                             default: 
  494.                                 if ( substr( $rule['minor'], 0, 10 ) == 'post_type-' ) { 
  495.                                     $condition_result = is_singular( substr( $rule['minor'], 10 ) ); 
  496.                                 } elseif ( substr( $rule['minor'], 0, 18 ) == 'post_type_archive-' ) { 
  497.                                     $condition_result = is_post_type_archive( substr( $rule['minor'], 18 ) ); 
  498.                                 } elseif ( $rule['minor'] == get_option( 'page_for_posts' ) ) { 
  499.                                     // If $rule['minor'] is a page ID which is also the posts page 
  500.                                     $condition_result = $wp_query->is_posts_page; 
  501.                                 } else { 
  502.                                     // $rule['minor'] is a page ID 
  503.                                     $condition_result = is_page() && ( $rule['minor'] == get_the_ID() ); 
  504.  
  505.                                     // Check if $rule['minor'] is parent of page ID 
  506.                                     if ( ! $condition_result && isset( $rule['has_children'] ) && $rule['has_children'] ) 
  507.                                         $condition_result = wp_get_post_parent_id( get_the_ID() ) == $rule['minor']; 
  508.                                 } 
  509.                             break; 
  510.                         } 
  511.                     break; 
  512.                     case 'tag': 
  513.                         // All tag pages. 
  514.                         if( ! $rule['minor'] ) { 
  515.                             if ( is_tag() ) { 
  516.                                 $condition_result = true; 
  517.                             } else if ( is_singular() ) { 
  518.                                 if( in_array( 'post_tag', get_post_taxonomies() ) ) { 
  519.                                     $condition_result = true; 
  520.                                 } 
  521.                             } 
  522.                             break; 
  523.                         } 
  524.  
  525.                         // All pages with the specified tag term. 
  526.                         if ( is_tag( $rule['minor'] ) ) { 
  527.                             $condition_result = true; 
  528.                         } 
  529.                         else if ( is_singular() && has_term( $rule['minor'], 'post_tag' ) ) { 
  530.                             $condition_result = true; 
  531.                         } 
  532.                     break; 
  533.                     case 'category': 
  534.                         // All category pages. 
  535.                         if( ! $rule['minor'] ) { 
  536.                             if ( is_category() ) { 
  537.                                 $condition_result = true; 
  538.                             } 
  539.                             else if ( is_singular() ) { 
  540.                                 if( in_array( 'category', get_post_taxonomies() ) ) { 
  541.                                     $condition_result = true; 
  542.                                 } 
  543.                             } 
  544.                             break; 
  545.                         } 
  546.  
  547.                         // All pages with the specified category term. 
  548.                         if ( is_category( $rule['minor'] ) ) { 
  549.                             $condition_result = true; 
  550.                         } 
  551.                         else if ( is_singular() && has_term( $rule['minor'], 'category' ) ) { 
  552.                             $condition_result = true; 
  553.                         } 
  554.                     break; 
  555.                     case 'loggedin': 
  556.                         $condition_result = is_user_logged_in(); 
  557.                         if ( 'loggedin' !== $rule['minor'] ) { 
  558.                             $condition_result = ! $condition_result; 
  559.                         } 
  560.                     break; 
  561.                     case 'author': 
  562.                         $post = get_post(); 
  563.                         if ( ! $rule['minor'] && is_author() ) 
  564.                             $condition_result = true; 
  565.                         else if ( $rule['minor'] && is_author( $rule['minor'] ) ) 
  566.                             $condition_result = true; 
  567.                         else if ( is_singular() && $rule['minor'] && $rule['minor'] == $post->post_author ) 
  568.                             $condition_result = true; 
  569.                     break; 
  570.                     case 'role': 
  571.                         if( is_user_logged_in() ) { 
  572.                             $current_user = wp_get_current_user(); 
  573.  
  574.                             $user_roles = $current_user->roles; 
  575.  
  576.                             if( in_array( $rule['minor'], $user_roles ) ) { 
  577.                                 $condition_result = true; 
  578.                             } else { 
  579.                                 $condition_result = false; 
  580.                             } 
  581.  
  582.                         } else { 
  583.                             $condition_result = false; 
  584.                         } 
  585.                     break; 
  586.                     case 'post_type': 
  587.                         if ( substr( $rule['minor'], 0, 10 ) == 'post_type-' ) { 
  588.                             $condition_result = is_singular( substr( $rule['minor'], 10 ) ); 
  589.                         } elseif ( substr( $rule['minor'], 0, 18 ) == 'post_type_archive-' ) { 
  590.                             $condition_result = is_post_type_archive( substr( $rule['minor'], 18 ) ); 
  591.                         } 
  592.                     break; 
  593.                     case 'taxonomy': 
  594.                         // All taxonomy pages. 
  595.                         if( ! $rule['minor'] ) { 
  596.                             if ( is_archive() ) { 
  597.                                 if ( is_tag() || is_category() || is_tax() ) { 
  598.                                     $condition_result = true; 
  599.                                 } 
  600.                             } 
  601.                             else if ( is_singular() ) { 
  602.                                 $post_taxonomies = get_post_taxonomies(); 
  603.                                 $condition_result = ! empty( $post_taxonomies ); 
  604.                             } 
  605.                             break; 
  606.                         } 
  607.  
  608.                         // Specified taxonomy page. 
  609.                         $term = explode( '_tax_', $rule['minor'] ); // $term[0] = taxonomy name; $term[1] = term id 
  610.                         if ( isset( $term[0] ) && isset( $term[1] ) ) { 
  611.                             $term[1] = self::maybe_get_split_term( $term[1], $term[0] ); 
  612.                         } 
  613.  
  614.                         // All pages of the specified taxonomy. 
  615.                         if ( ! isset( $term[1] ) || ! $term[1] ) { 
  616.                             if ( is_tax( $term[0] ) ) { 
  617.                                 $condition_result = true; 
  618.                             } 
  619.                             else if ( is_singular() ) { 
  620.                                 if( in_array( $term[0], get_post_taxonomies() ) ) { 
  621.                                     $condition_result = true; 
  622.                                 } 
  623.                             } 
  624.                             break; 
  625.                         } 
  626.  
  627.                         // All pages with the specified taxonomy term. 
  628.                         if ( is_tax( $term[0], $term[1] ) ) { 
  629.                             $condition_result = true; 
  630.                         } 
  631.                         else if ( is_singular() && has_term( $term[1], $term[0] ) ) { 
  632.                             $condition_result = true; 
  633.                         } 
  634.                     break; 
  635.                 } 
  636.  
  637.                 if ( $condition_result || self::$passed_template_redirect ) { 
  638.                     // Some of the conditions will return false when checked before the template_redirect 
  639.                     // action has been called, like is_page(). Only store positive lookup results, which 
  640.                     // won't be false positives, before template_redirect, and everything after. 
  641.                     $condition_result_cache[ $condition_key ] = $condition_result; 
  642.                 } 
  643.             } 
  644.  
  645.             if ( 
  646.                 isset( $instance['conditions']['match_all'] ) 
  647.                 && $instance['conditions']['match_all'] == '1' 
  648.                 && ! $condition_result 
  649.  ) { 
  650.  
  651.                 // In case the match_all flag was set we quit on first failed condition 
  652.                 break; 
  653.             } elseif ( 
  654.                 ( 
  655.                     empty( $instance['conditions']['match_all'] ) 
  656.                     || $instance['conditions']['match_all'] !== '1' 
  657.  ) 
  658.                 && $condition_result 
  659.  ) { 
  660.  
  661.                 // Only quit on first condition if the match_all flag was not set 
  662.                 break; 
  663.             } 
  664.         } 
  665.  
  666.         if ( 
  667.             ( 
  668.                 'show' == $instance['conditions']['action'] 
  669.                 && ! $condition_result 
  670.  ) || ( 
  671.                 'hide' == $instance['conditions']['action'] 
  672.                 && $condition_result 
  673.  ) 
  674.  ) { 
  675.             return false; 
  676.         } 
  677.  
  678.         return $instance; 
  679.     }

  685.     public static function maybe_get_split_term( $old_term_id = '', $taxonomy = '' ) { 
  686.         $term_id = $old_term_id; 
  687.  
  688.         if ( 'tag' == $taxonomy ) { 
  689.             $taxonomy = 'post_tag'; 
  690.         } 
  691.  
  692.         if ( function_exists( 'wp_get_split_term' ) && $new_term_id = wp_get_split_term( $old_term_id, $taxonomy ) ) { 
  693.             $term_id = $new_term_id; 
  694.         } 
  695.  
  696.         return $term_id;

  699.     /** 
  700.      * Upgrade routine to go through all widgets and move the Post Type 
  701.      * setting to its newer location. 
  702.      * 
  703.      * @since 4.7.1 
  704.      * 
  705.      */ 
  706.     static function migrate_post_type_rules() { 
  707.         global $wp_registered_widgets; 
  708.  
  709.         $sidebars_widgets = get_option( 'sidebars_widgets' ); 
  710.  
  711.         // Going through all sidebars and through inactive and orphaned widgets 
  712.         foreach ( $sidebars_widgets as $s => $sidebar ) { 
  713.             if ( ! is_array( $sidebar ) ) { 
  714.                 continue; 
  715.             } 
  716.  
  717.             foreach ( $sidebar as $w => $widget ) { 
  718.                 // $widget is the id of the widget 
  719.                 if ( empty( $wp_registered_widgets[ $widget ] ) ) { 
  720.                     continue; 
  721.                 } 
  722.  
  723.                 $opts = $wp_registered_widgets[ $widget ]; 
  724.                 $instances = get_option( $opts['callback'][0]->option_name ); 
  725.  
  726.                 // Going through each instance of the widget 
  727.                 foreach( $instances as $number => $instance ) { 
  728.                     if ( 
  729.                         ! is_array( $instance ) || 
  730.                         empty( $instance['conditions'] ) || 
  731.                         empty( $instance['conditions']['rules'] ) 
  732.  ) { 
  733.                         continue; 
  734.                     } 
  735.  
  736.                     // Going through all visibility rules 
  737.                     foreach( $instance['conditions']['rules'] as $index => $rule ) { 
  738.  
  739.                         // We only need Post Type rules 
  740.                         if ( 'post_type' !== $rule['major'] ) { 
  741.                             continue; 
  742.                         } 
  743.  
  744.                         $rule_type = false; 
  745.  
  746.                         // Post type or type archive rule 
  747.                         if ( 0 === strpos( $rule['minor'], 'post_type_archive' ) ) { 
  748.                             $rule_type = 'post_type_archive'; 
  749.                         } else if ( 0 === strpos( $rule['minor'], 'post_type' ) ) { 
  750.                             $rule_type = 'post_type'; 
  751.                         } 
  752.  
  753.                         if ( $rule_type ) { 
  754.                             $post_type = substr( $rule['minor'], strlen( $rule_type ) + 1 ); 
  755.                             $rule['minor'] = $rule_type . '-' . $post_type; 
  756.                             $rule['major'] = 'page'; 
  757.  
  758.                             $instances[ $number ]['conditions']['rules'][ $index ] = $rule; 
  759.                         } 
  760.                     } 
  761.                 } 
  762.  
  763.                 update_option( $opts['callback'][0]->option_name, $instances ); 
  764.             } 
  765.         } 
  766.     } 
  767.
