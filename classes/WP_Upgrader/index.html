  249.         global $wp_filesystem; 
  250.  
  251.         $this->skin->feedback('unpack_package'); 
  252.  
  253.         $upgrade_folder = $wp_filesystem->wp_content_dir() . 'upgrade/'; 
  254.  
  255.         //Clean up contents of upgrade directory beforehand. 
  256.         $upgrade_files = $wp_filesystem->dirlist($upgrade_folder); 
  257.         if ( !empty($upgrade_files) ) { 
  258.             foreach ( $upgrade_files as $file ) 
  259.                 $wp_filesystem->delete($upgrade_folder . $file['name'], true); 
  260.         } 
  261.  
  262.         // We need a working directory - Strip off any .tmp or .zip suffixes

  264.

  266.         if ( $wp_filesystem->is_dir($working_dir) ) 
  267.             $wp_filesystem->delete($working_dir, true); 
  268.  
  269.         // Unzip package to working directory 
  270.         $result = unzip_file( $package, $working_dir ); 
  271.  
  272.         // Once extracted, delete the package if required. 
  273.         if ( $delete_package )

  288.      * Clears the directory where this item is going to be installed into. 
  289.      *

  298.     public function clear_destination( $remote_destination ) { 
  299.         global $wp_filesystem; 
  300.  
  301.         if ( ! $wp_filesystem->exists( $remote_destination ) ) { 
  302.             return true; 
  303.         } 
  304.  
  305.         // Check all files are writable before attempting to clear the destination. 
  306.         $unwritable_files = array(); 
  307.  
  308.         $_files = $wp_filesystem->dirlist( $remote_destination, true, true ); 
  309.  
  310.         // Flatten the resulting array, iterate using each as we append to the array during iteration. 
  311.         while ( $f = each( $_files ) ) { 
  312.             $file = $f['value']; 
  313.             $name = $f['key']; 
  314.  
  315.             if ( ! isset( $file['files'] ) ) { 
  316.                 continue; 
  317.             } 
  318.  
  319.             foreach ( $file['files'] as $filename => $details ) { 
  320.                 $_files[ $name . '/' . $filename ] = $details; 
  321.             } 
  322.         } 
  323.  
  324.         // Check writability. 
  325.         foreach ( $_files as $filename => $file_details ) { 
  326.             if ( ! $wp_filesystem->is_writable( $remote_destination . $filename ) ) { 
  327.  
  328.                 // Attempt to alter permissions to allow writes and try again. 
  329.                 $wp_filesystem->chmod( $remote_destination . $filename, ( 'd' == $file_details['type'] ? FS_CHMOD_DIR : FS_CHMOD_FILE ) ); 
  330.                 if ( ! $wp_filesystem->is_writable( $remote_destination . $filename ) ) { 
  331.                     $unwritable_files[] = $filename; 
  332.                 } 
  333.             } 
  334.         } 
  335.  
  336.         if ( ! empty( $unwritable_files ) ) { 
  337.             return new WP_Error( 'files_not_writable', $this->strings['files_not_writable'], implode( ', ', $unwritable_files ) ); 
  338.         } 
  339.  
  340.         if ( ! $wp_filesystem->delete( $remote_destination, true ) ) { 
  341.             return new WP_Error( 'remove_old_failed', $this->strings['remove_old_failed'] ); 
  342.         } 
  343.  
  344.         return true; 
  345.     }

  378.     public function install_package( $args = array() ) { 
  379.         global $wp_filesystem, $wp_theme_directories; 
  380.  
  381.         $defaults = array( 
  382.             'source' => '', // Please always pass this 
  383.             'destination' => '', // and this 
  384.             'clear_destination' => false,  
  385.             'clear_working' => false,  
  386.             'abort_if_destination_exists' => true,  
  387.             'hook_extra' => array() 
  388.  ); 
  389.  
  390.         $args = wp_parse_args($args, $defaults); 
  391.  
  392.         // These were previously extract()'d. 
  393.         $source = $args['source']; 
  394.         $destination = $args['destination']; 
  395.         $clear_destination = $args['clear_destination']; 
  396.  
  397.         @set_time_limit( 300 ); 
  398.  
  399.         if ( empty( $source ) || empty( $destination ) ) { 
  400.             return new WP_Error( 'bad_request', $this->strings['bad_request'] ); 
  401.         } 
  402.         $this->skin->feedback( 'installing_package' ); 
  403.  
  404.         /** 
  405.          * Filters the install response before the installation has started. 
  406.          * 
  407.          * Returning a truthy value, or one that could be evaluated as a WP_Error 
  408.          * will effectively short-circuit the installation, returning that value 
  409.          * instead.

  416.         $res = apply_filters( 'upgrader_pre_install', true, $args['hook_extra'] ); 
  417.  
  418.         if ( is_wp_error( $res ) ) { 
  419.             return $res; 
  420.         } 
  421.  
  422.         //Retain the Original source and destinations 
  423.         $remote_source = $args['source']; 
  424.         $local_destination = $destination; 
  425.  
  426.         $source_files = array_keys( $wp_filesystem->dirlist( $remote_source ) ); 
  427.         $remote_destination = $wp_filesystem->find_folder( $local_destination ); 
  428.  
  429.         //Locate which directory to copy to the new folder, This is based on the actual folder holding the files. 
  430.         if ( 1 == count( $source_files ) && $wp_filesystem->is_dir( trailingslashit( $args['source'] ) . $source_files[0] . '/' ) ) { //Only one folder? Then we want its contents. 
  431.             $source = trailingslashit( $args['source'] ) . trailingslashit( $source_files[0] ); 
  432.         } elseif ( count( $source_files ) == 0 ) { 
  433.             return new WP_Error( 'incompatible_archive_empty', $this->strings['incompatible_archive'], $this->strings['no_files'] ); // There are no files? 
  434.         } else { // It's only a single file, the upgrader will use the folder name of this file as the destination folder. Folder name is based on zip filename. 
  435.             $source = trailingslashit( $args['source'] ); 
  436.         } 
  437.  
  438.         /** 
  439.          * Filters the source file location for the upgrade package. 
  440.          * 
  441.          * @since 2.8.0 
  442.          * @since 4.4.0 The $hook_extra parameter became available. 
  443.          * 
  444.          * @param string      $source        File source location. 
  445.          * @param string      $remote_source Remote file source location. 
  446.          * @param WP_Upgrader $this          WP_Upgrader instance. 
  447.          * @param array       $hook_extra    Extra arguments passed to hooked filters. 
  448.          */ 
  449.         $source = apply_filters( 'upgrader_source_selection', $source, $remote_source, $this, $args['hook_extra'] ); 
  450.  
  451.         if ( is_wp_error( $source ) ) { 
  452.             return $source; 
  453.         } 
  454.  
  455.         // Has the source location changed? If so, we need a new source_files list. 
  456.         if ( $source !== $remote_source ) { 
  457.             $source_files = array_keys( $wp_filesystem->dirlist( $source ) ); 
  458.         } 
  459.  
  460.         /** 
  461.          * Protection against deleting files in any important base directories. 
  462.          * Theme_Upgrader & Plugin_Upgrader also trigger this, as they pass the 
  463.          * destination directory (WP_PLUGIN_DIR / wp-content/themes) intending 
  464.          * to copy the directory into the directory, whilst they pass the source 
  465.          * as the actual files to copy. 
  466.          */ 
  467.         $protected_directories = array( ABSPATH, WP_CONTENT_DIR, WP_PLUGIN_DIR, WP_CONTENT_DIR . '/themes' ); 
  468.  
  469.         if ( is_array( $wp_theme_directories ) ) { 
  470.             $protected_directories = array_merge( $protected_directories, $wp_theme_directories ); 
  471.         } 
  472.  
  473.         if ( in_array( $destination, $protected_directories ) ) { 
  474.             $remote_destination = trailingslashit( $remote_destination ) . trailingslashit( basename( $source ) ); 
  475.             $destination = trailingslashit( $destination ) . trailingslashit( basename( $source ) ); 
  476.         } 
  477.  
  478.         if ( $clear_destination ) { 
  479.             // We're going to clear the destination if there's something there. 
  480.             $this->skin->feedback('remove_old'); 
  481.  
  482.             $removed = $this->clear_destination( $remote_destination ); 
  483.  
  484.             /** 
  485.              * Filters whether the upgrader cleared the destination. 
  486.              * 
  487.              * @since 2.8.0 
  488.              * 
  489.              * @param mixed  $removed            Whether the destination was cleared. true on success, WP_Error on failure 
  490.              * @param string $local_destination  The local package destination. 
  491.              * @param string $remote_destination The remote package destination. 
  492.              * @param array  $hook_extra         Extra arguments passed to hooked filters. 
  493.              */ 
  494.             $removed = apply_filters( 'upgrader_clear_destination', $removed, $local_destination, $remote_destination, $args['hook_extra'] ); 
  495.  
  496.             if ( is_wp_error( $removed ) ) { 
  497.                 return $removed; 
  498.             } 
  499.         } elseif ( $args['abort_if_destination_exists'] && $wp_filesystem->exists($remote_destination) ) { 
  500.             //If we're not clearing the destination folder and something exists there already, Bail. 
  501.             //But first check to see if there are actually any files in the folder. 
  502.             $_files = $wp_filesystem->dirlist($remote_destination); 
  503.             if ( ! empty($_files) ) { 
  504.                 $wp_filesystem->delete($remote_source, true); //Clear out the source files. 
  505.                 return new WP_Error('folder_exists', $this->strings['folder_exists'], $remote_destination ); 
  506.             } 
  507.         } 
  508.  
  509.         //Create destination if needed 
  510.         if ( ! $wp_filesystem->exists( $remote_destination ) ) { 
  511.             if ( ! $wp_filesystem->mkdir( $remote_destination, FS_CHMOD_DIR ) ) { 
  512.                 return new WP_Error( 'mkdir_failed_destination', $this->strings['mkdir_failed'], $remote_destination ); 
  513.             } 
  514.         } 
  515.         // Copy new version of item into place. 
  516.         $result = copy_dir($source, $remote_destination); 
  517.         if ( is_wp_error($result) ) { 
  518.             if ( $args['clear_working'] ) { 
  519.                 $wp_filesystem->delete( $remote_source, true ); 
  520.             } 
  521.             return $result; 
  522.         } 
  523.  
  524.         //Clear the Working folder? 
  525.         if ( $args['clear_working'] ) { 
  526.             $wp_filesystem->delete( $remote_source, true ); 
  527.         } 
  528.  
  529.         $destination_name = basename( str_replace($local_destination, '', $destination) ); 
  530.         if ( '.' == $destination_name ) { 
  531.             $destination_name = ''; 
  532.         } 
  533.  
  534.         $this->result = compact( 'source', 'source_files', 'destination', 'destination_name', 'local_destination', 'remote_destination', 'clear_destination' ); 
  535.  
  536.         /** 
  537.          * Filters the install response after the installation has finished. 
  538.          * 
  539.          * @since 2.8.0 
  540.          * 
  541.          * @param bool  $response   Install response. 
  542.          * @param array $hook_extra Extra arguments passed to hooked filters. 
  543.          * @param array $result     Installation result data. 
  544.          */ 
  545.         $res = apply_filters( 'upgrader_post_install', true, $args['hook_extra'], $this->result ); 
  546.  
  547.         if ( is_wp_error($res) ) { 
  548.             $this->result = $res; 
  549.             return $res; 
  550.         } 
  551.  
  552.         //Bombard the calling function will all the info which we've just used. 
  553.         return $this->result; 
  554.     } 
  555.  
  556.     /** 
  557.      * Run an upgrade/install. 
  558.      * 
  559.      * Attempts to download the package (if it is not a local file), unpack it, and 
  560.      * install it in the destination folder. 
  561.      * 
  562.      * @since 2.8.0 
  563.      * @access public 
  564.      * 
  565.      * @param array $options { 
  566.      *     Array or string of arguments for upgrading/installing a package. 
  567.      * 
  568.      *     @type string $package                     The full path or URI of the package to install. 
  569.      *                                               Default empty. 
  570.      *     @type string $destination                 The full path to the destination folder. 
  571.      *                                               Default empty. 
  572.      *     @type bool   $clear_destination           Whether to delete any files already in the 
  573.      *                                               destination folder. Default false. 
  574.      *     @type bool   $clear_working               Whether to delete the files form the working 
  575.      *                                               directory after copying to the destination. 
  576.      *                                               Default false. 
  577.      *     @type bool   $abort_if_destination_exists Whether to abort the installation if the destination 
  578.      *                                               folder already exists. When true, `$clear_destination` 
  579.      *                                               should be false. Default true. 
  580.      *     @type bool   $is_multi                    Whether this run is one of multiple upgrade/install 
  581.      *                                               actions being performed in bulk. When true, the skin 
  582.      *                                               WP_Upgrader::header() and WP_Upgrader::footer() 
  583.      *                                               aren't called. Default false. 
  584.      *     @type array  $hook_extra                  Extra arguments to pass to the filter hooks called by 
  585.      *                                               WP_Upgrader::run(). 
  586.      * } 
  587.      * @return array|false|WP_error The result from self::install_package() on success, otherwise a WP_Error,  
  588.      *                              or false if unable to connect to the filesystem. 
  589.      */ 
  590.     public function run( $options ) { 
  591.  
  592.         $defaults = array( 
  593.             'package' => '', // Please always pass this. 
  594.             'destination' => '', // And this 
  595.             'clear_destination' => false,  
  596.             'abort_if_destination_exists' => true, // Abort if the Destination directory exists, Pass clear_destination as false please 
  597.             'clear_working' => true,  
  598.             'is_multi' => false,  
  599.             'hook_extra' => array() // Pass any extra $hook_extra args here, this will be passed to any hooked filters. 
  600.  ); 
  601.  
  602.         $options = wp_parse_args( $options, $defaults ); 
  603.  
  604.         /** 
  605.          * Filters the package options before running an update. 
  606.          * 
  607.          * See also {@see 'upgrader_process_complete'}. 
  608.          * 
  609.          * @since 4.3.0 
  610.          * 
  611.          * @param array $options { 
  612.          *     Options used by the upgrader. 
  613.          * 
  614.          *     @type string $package                     Package for update. 
  615.          *     @type string $destination                 Update location. 
  616.          *     @type bool   $clear_destination           Clear the destination resource. 
  617.          *     @type bool   $clear_working               Clear the working resource. 
  618.          *     @type bool   $abort_if_destination_exists Abort if the Destination directory exists. 
  619.          *     @type bool   $is_multi                    Whether the upgrader is running multiple times. 
  620.          *     @type array  $hook_extra { 
  621.          *         Extra hook arguments. 
  622.          * 
  623.          *         @type string $action               Type of action. Default 'update'. 
  624.          *         @type string $type                 Type of update process. Accepts 'plugin', 'theme', or 'core'. 
  625.          *         @type bool   $bulk                 Whether the update process is a bulk update. Default true. 
  626.          *         @type string $plugin               The base plugin path from the plugins directory. 
  627.          *         @type string $theme                The stylesheet or template name of the theme. 
  628.          *         @type string $language_update_type The language pack update type. Accepts 'plugin', 'theme',  
  629.          *                                            or 'core'. 
  630.          *         @type object $language_update      The language pack update offer. 
  631.          *     } 
  632.          * } 
  633.          */ 
  634.         $options = apply_filters( 'upgrader_package_options', $options ); 
  635.  
  636.         if ( ! $options['is_multi'] ) { // call $this->header separately if running multiple times 
  637.             $this->skin->header(); 
  638.         } 
  639.  
  640.         // Connect to the Filesystem first. 
  641.         $res = $this->fs_connect( array( WP_CONTENT_DIR, $options['destination'] ) ); 
  642.         // Mainly for non-connected filesystem. 
  643.         if ( ! $res ) { 
  644.             if ( ! $options['is_multi'] ) { 
  645.                 $this->skin->footer(); 
  646.             } 
  647.             return false; 
  648.         } 
  649.  
  650.         $this->skin->before(); 
  651.  
  652.         if ( is_wp_error($res) ) { 
  653.             $this->skin->error($res); 
  654.             $this->skin->after(); 
  655.             if ( ! $options['is_multi'] ) { 
  656.                 $this->skin->footer(); 
  657.             } 
  658.             return $res; 
  659.         } 
  660.  
  661.         /** 
  662.          * Download the package (Note, This just returns the filename 
  663.          * of the file if the package is a local file) 
  664.          */ 
  665.         $download = $this->download_package( $options['package'] ); 
  666.         if ( is_wp_error($download) ) { 
  667.             $this->skin->error($download); 
  668.             $this->skin->after(); 
  669.             if ( ! $options['is_multi'] ) { 
  670.                 $this->skin->footer(); 
  671.             } 
  672.             return $download; 
  673.         } 
  674.  
  675.         $delete_package = ( $download != $options['package'] ); // Do not delete a "local" file 
  676.  
  677.         // Unzips the file into a temporary directory. 
  678.         $working_dir = $this->unpack_package( $download, $delete_package ); 
  679.         if ( is_wp_error($working_dir) ) { 
  680.             $this->skin->error($working_dir); 
  681.             $this->skin->after(); 
  682.             if ( ! $options['is_multi'] ) { 
  683.                 $this->skin->footer(); 
  684.             } 
  685.             return $working_dir; 
  686.         } 
  687.  
  688.         // With the given options, this installs it to the destination directory. 
  689.         $result = $this->install_package( array( 
  690.             'source' => $working_dir,  
  691.             'destination' => $options['destination'],  
  692.             'clear_destination' => $options['clear_destination'],  
  693.             'abort_if_destination_exists' => $options['abort_if_destination_exists'],  
  694.             'clear_working' => $options['clear_working'],  
  695.             'hook_extra' => $options['hook_extra'] 
  696.  ) ); 
  697.  
  698.         $this->skin->set_result($result); 
  699.         if ( is_wp_error($result) ) { 
  700.             $this->skin->error($result); 
  701.             $this->skin->feedback('process_failed'); 
  702.         } else { 
  703.             // Install succeeded. 
  704.             $this->skin->feedback('process_success'); 
  705.         } 
  706.  
  707.         $this->skin->after(); 
  708.  
  709.         if ( ! $options['is_multi'] ) { 
  710.  
  711.             /** 
  712.              * Fires when the upgrader process is complete. 
  713.              * 
  714.              * See also {@see 'upgrader_package_options'}. 
  715.              * 
  716.              * @since 3.6.0 
  717.              * @since 3.7.0 Added to WP_Upgrader::run(). 
  718.              * @since 4.6.0 `$translations` was added as a possible argument to `$hook_extra`. 
  719.              * 
  720.              * @param WP_Upgrader $this WP_Upgrader instance. In other contexts, $this, might be a 
  721.              *                          Theme_Upgrader, Plugin_Upgrader, Core_Upgrade, or Language_Pack_Upgrader instance. 
  722.              * @param array       $hook_extra { 
  723.              *     Array of bulk item update data. 
  724.              * 
  725.              *     @type string $action       Type of action. Default 'update'. 
  726.              *     @type string $type         Type of update process. Accepts 'plugin', 'theme', 'translation', or 'core'. 
  727.              *     @type bool   $bulk         Whether the update process is a bulk update. Default true. 
  728.              *     @type array  $plugins      Array of the basename paths of the plugins' main files. 
  729.              *     @type array  $themes       The theme slugs. 
  730.              *     @type array  $translations { 
  731.              *         Array of translations update data. 
  732.              * 
  733.              *         @type string $language The locale the translation is for. 
  734.              *         @type string $type     Type of translation. Accepts 'plugin', 'theme', or 'core'. 
  735.              *         @type string $slug     Text domain the translation is for. The slug of a theme/plugin or 
  736.              *                                'default' for core translations. 
  737.              *         @type string $version  The version of a theme, plugin, or core. 
  738.              *     } 
  739.              * } 
  740.              */ 
  741.             do_action( 'upgrader_process_complete', $this, $options['hook_extra'] ); 
  742.  
  743.             $this->skin->footer(); 
  744.         } 
  745.  
  746.         return $result; 
  747.     }

  749.     /** 
  750.      * Toggle maintenance mode for the site. 
  751.      * 
  752.      * Creates/deletes the maintenance file to enable/disable maintenance mode. 
  753.      * 
  754.      * @since 2.8.0 
  755.      * @access public 
  756.      * 
  757.      * @global WP_Filesystem_Base $wp_filesystem Subclass 
  758.      * 
  759.      * @param bool $enable True to enable maintenance mode, false to disable. 
  760.      */ 
  761.     public function maintenance_mode( $enable = false ) { 
  762.         global $wp_filesystem; 
  763.         $file = $wp_filesystem->abspath() . '.maintenance'; 
  764.         if ( $enable ) { 
  765.             $this->skin->feedback('maintenance_start'); 
  766.             // Create maintenance file to signal that we are upgrading 
  767.             $maintenance_string = '<?php $upgrading = ' . time() . '; ?>'; 
  768.             $wp_filesystem->delete($file); 
  769.             $wp_filesystem->put_contents($file, $maintenance_string, FS_CHMOD_FILE); 
  770.         } elseif ( ! $enable && $wp_filesystem->exists( $file ) ) { 
  771.             $this->skin->feedback('maintenance_end'); 
  772.             $wp_filesystem->delete($file); 
  773.         } 
  774.     }

  776.     /** 
  777.       * Creates a lock using WordPress options. 
  778.       * 
  779.       * @since 4.5.0 
  780.       * @access public 
  781.       * @static 
  782.       * 
  783.       * @param string $lock_name       The name of this unique lock. 
  784.       * @param int    $release_timeout Optional. The duration in seconds to respect an existing lock. 
  785.      *                                Default: 1 hour. 
  786.       * @return bool False if a lock couldn't be created or if the lock is still valid. True otherwise. 
  787.       */ 
  788.     public static function create_lock( $lock_name, $release_timeout = null ) { 
  789.         global $wpdb; 
  790.         if ( ! $release_timeout ) { 
  791.             $release_timeout = HOUR_IN_SECONDS; 
  792.         } 
  793.         $lock_option = $lock_name . '.lock'; 
  794.  
  795.         // Try to lock. 
  796.         $lock_result = $wpdb->query( $wpdb->prepare( "INSERT IGNORE INTO `$wpdb->options` ( `option_name`, `option_value`, `autoload` ) VALUES (%s, %s, 'no') /* LOCK */", $lock_option, time() ) ); 
  797.  
  798.         if ( ! $lock_result ) { 
  799.             $lock_result = get_option( $lock_option ); 
  800.  
  801.             // If a lock couldn't be created, and there isn't a lock, bail. 
  802.             if ( ! $lock_result ) { 
  803.                 return false; 
  804.             } 
  805.  
  806.             // Check to see if the lock is still valid. If it is, bail. 
  807.             if ( $lock_result > ( time() - $release_timeout ) ) { 
  808.                 return false; 
  809.             } 
  810.  
  811.             // There must exist an expired lock, clear it and re-gain it. 
  812.             WP_Upgrader::release_lock( $lock_name ); 
  813.  
  814.             return WP_Upgrader::create_lock( $lock_name, $release_timeout ); 
  815.         } 
  816.  
  817.         // Update the lock, as by this point we've definitely got a lock, just need to fire the actions. 
  818.         update_option( $lock_option, time() ); 
  819.  
  820.         return true; 
  821.     }

  823.     /** 
  824.       * Releases an upgrader lock. 
  825.       * 
  826.       * @since 4.5.0 
  827.       * @access public 
  828.       * @static 
  829.      * 
  830.      * @see WP_Upgrader::create_lock() 
  831.       * 
  832.       * @param string $lock_name The name of this unique lock. 
  833.      * @return bool True if the lock was successfully released. False on failure. 
  834.       */ 
  835.     public static function release_lock( $lock_name ) { 
  836.         return delete_option( $lock_name . '.lock' ); 
  837.     }
