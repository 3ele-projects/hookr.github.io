  458.                 if ( strtotime( isset( $event['DTEND'] ) ? $event['DTEND'] : $event['DTSTART'] ) >= $current ) { 
  459.                     $upcoming[] = $event; 
  460.                 } 
  461.             } 
  462.         } 
  463.         return $upcoming; 
  464.     }

  466.     /** 
  467.      * Parse events from an iCalendar feed 
  468.      * 
  469.      * @param string $url (default: '') 
  470.      * @return array | false on failure 
  471.      */ 
  472.     public function parse( $url = '' ) { 
  473.         $cache_group = 'icalendar_reader_parse'; 
  474.         $disable_get_key = 'disable:' . md5( $url ); 
  475.  
  476.         // Check to see if previous attempts have failed 
  477.         if ( false !== wp_cache_get( $disable_get_key, $cache_group ) ) 
  478.             return false; 
  479.  
  480.         // rewrite webcal: URI schem to HTTP 
  481.         $url = preg_replace('/^webcal/', 'http', $url ); 
  482.         // try to fetch 
  483.         $r = wp_remote_get( $url, array( 'timeout' => 3, 'sslverify' => false ) ); 
  484.         if ( 200 !== wp_remote_retrieve_response_code( $r ) ) { 
  485.             // We were unable to fetch any content, so don't try again for another 60 seconds 
  486.             wp_cache_set( $disable_get_key, 1, $cache_group, 60 ); 
  487.             return false; 
  488.         } 
  489.  
  490.         $body = wp_remote_retrieve_body( $r ); 
  491.         if ( empty( $body ) ) 
  492.             return false; 
  493.  
  494.         $body = str_replace( "\r\n", "\n", $body ); 
  495.         $lines = preg_split( "/\n(?=[A-Z])/", $body ); 
  496.  
  497.         if ( empty( $lines ) ) 
  498.             return false; 
  499.  
  500.         if ( false === stristr( $lines[0], 'BEGIN:VCALENDAR' ) ) 
  501.             return false; 
  502.  
  503.         foreach ( $lines as $line ) { 
  504.             $add = $this->key_value_from_string( $line ); 
  505.             if ( ! $add ) { 
  506.                 $this->add_component( $type, false, $line ); 
  507.                 continue; 
  508.             } 
  509.             list( $keyword, $value ) = $add; 
  510.  
  511.             switch ( $keyword ) { 
  512.                 case 'BEGIN': 
  513.                 case 'END': 
  514.                     switch ( $line ) { 
  515.                         case 'BEGIN:VTODO': 
  516.                             $this->todo_count++; 
  517.                             $type = 'VTODO'; 
  518.                             break; 
  519.                         case 'BEGIN:VEVENT': 
  520.                             $this->event_count++; 
  521.                             $type = 'VEVENT'; 
  522.                             break; 
  523.                         case 'BEGIN:VCALENDAR': 
  524.                         case 'BEGIN:DAYLIGHT': 
  525.                         case 'BEGIN:VTIMEZONE': 
  526.                         case 'BEGIN:STANDARD': 
  527.                             $type = $value; 
  528.                             break; 
  529.                         case 'END:VTODO': 
  530.                         case 'END:VEVENT': 
  531.                         case 'END:VCALENDAR': 
  532.                         case 'END:DAYLIGHT': 
  533.                         case 'END:VTIMEZONE': 
  534.                         case 'END:STANDARD': 
  535.                             $type = 'VCALENDAR'; 
  536.                             break; 
  537.                     } 
  538.                     break; 
  539.                 case 'TZID': 
  540.                     if ( 'VTIMEZONE' == $type && ! $this->timezone ) 
  541.                         $this->timezone = $this->timezone_from_string( $value ); 
  542.                     break; 
  543.                 case 'X-WR-TIMEZONE': 
  544.                     if ( ! $this->timezone ) 
  545.                         $this->timezone = $this->timezone_from_string( $value ); 
  546.                     break; 
  547.                 default: 
  548.                     $this->add_component( $type, $keyword, $value ); 
  549.                     break; 
  550.             } 
  551.         } 
  552.  
  553.         // Filter for RECURRENCE-IDs 
  554.         $recurrences = array(); 
  555.         if ( array_key_exists( 'VEVENT', $this->cal ) ) { 
  556.             foreach ( $this->cal['VEVENT'] as $event ) { 
  557.                 if ( isset( $event['RECURRENCE-ID'] ) ) { 
  558.                     $recurrences[] = $event; 
  559.                 } 
  560.             } 
  561.             foreach ( $recurrences as $recurrence ) { 
  562.                 for ( $i = 0; $i < count( $this->cal['VEVENT'] ); $i++ ) { 
  563.                     if ( $this->cal['VEVENT'][ $i ]['UID'] == $recurrence['UID'] && ! isset( $this->cal['VEVENT'][ $i ]['RECURRENCE-ID'] ) ) { 
  564.                         $this->cal['VEVENT'][ $i ]['EXDATE'][] = $recurrence['RECURRENCE-ID']; 
  565.                         break; 
  566.                     } 
  567.                 } 
  568.             } 
  569.         } 
  570.  
  571.         return $this->cal; 
  572.     }

  575.      * Parse key:value from a string 
  576.      * 
  577.      * @param string $text (default: '') 
  578.      * @return array 
  579.      */ 
  580.     public function key_value_from_string( $text = '' ) { 
  581.         preg_match( '/([^:]+)(;[^:]+)?[:]([\w\W]*)/', $text, $matches ); 
  582.  
  583.         if ( 0 == count( $matches ) ) 
  584.             return false; 
  585.  
  586.         return array( $matches[1], $matches[3] ); 
  587.     }

  589.     /** 
  590.      * Convert a timezone name into a timezone object. 
  591.      * 
  592.      * @param string $text Timezone name. Example: America/Chicago 
  593.      * @return object|null A DateTimeZone object if the conversion was successful. 
  594.      */ 
  595.     private function timezone_from_string( $text ) { 
  596.         try { 
  597.             $timezone = new DateTimeZone( $text ); 
  598.         } catch ( Exception $e ) { 
  599.             $blog_timezone = get_option( 'timezone_string' ); 
  600.             if ( ! $blog_timezone ) { 
  601.                 $blog_timezone = 'Etc/UTC'; 
  602.             } 
  603.  
  604.             $timezone = new DateTimeZone( $blog_timezone ); 
  605.         } 
  606.  
  607.         return $timezone; 
  608.     }

  610.     /** 
  611.      * Add a component to the calendar array 
  612.      * 
  613.      * @param string $component (default: '') 
  614.      * @param string $keyword (default: '') 
  615.      * @param string $value (default: '') 
  616.      * @return void 
  617.      */ 
  618.     public function add_component( $component = '', $keyword = '', $value = '' ) { 
  619.         if ( false == $keyword ) { 
  620.             $keyword = $this->last_keyword; 
  621.             switch ( $component ) { 
  622.             case 'VEVENT': 
  623.                 $value = $this->cal[ $component ][ $this->event_count - 1 ][ $keyword ] . $value; 
  624.                 break; 
  625.             case 'VTODO' : 
  626.                 $value = $this->cal[ $component ][ $this->todo_count - 1 ][ $keyword ] . $value; 
  627.                 break; 
  628.             } 
  629.         } 
  630.  
  631.         /** 
  632.          * Some events have a specific timezone set in their start/end date,  
  633.          * and it may or may not be different than the calendar timzeone. 
  634.          * Valid formats include: 
  635.          * DTSTART;TZID=Pacific Standard Time:20141219T180000 
  636.          * DTEND;TZID=Pacific Standard Time:20141219T200000 
  637.          * EXDATE:19960402T010000Z, 19960403T010000Z, 19960404T010000Z 
  638.          * EXDATE;VALUE=DATE:2015050 
  639.          * EXDATE;TZID=America/New_York:20150424T170000 
  640.          * EXDATE;TZID=Pacific Standard Time:20120615T140000, 20120629T140000, 20120706T140000 
  641.          */ 
  642.  
  643.         // Always store EXDATE as an array 
  644.         if ( stristr( $keyword, 'EXDATE' ) ) { 
  645.             $value = explode( ', ', $value ); 
  646.         } 
  647.  
  648.         // Adjust DTSTART, DTEND, and EXDATE according to their TZID if set 
  649.         if ( strpos( $keyword, ';' ) && ( stristr( $keyword, 'DTSTART' ) || stristr( $keyword, 'DTEND' ) || stristr( $keyword, 'EXDATE' ) || stristr( $keyword, 'RECURRENCE-ID' ) ) ) { 
  650.             $keyword = explode( ';', $keyword ); 
  651.  
  652.             $tzid = false; 
  653.             if ( 2 == count( $keyword ) ) { 
  654.                 $tparam = $keyword[1]; 
  655.  
  656.                 if ( strpos( $tparam, "TZID" ) !== false ) { 
  657.                     $tzid = $this->timezone_from_string( str_replace( 'TZID=', '', $tparam ) ); 
  658.                 } 
  659.             } 
  660.  
  661.             // Normalize all times to default UTC 
  662.             if ( $tzid ) { 
  663.                 $adjusted_times = array(); 
  664.                 foreach ( (array) $value as $v ) { 
  665.                     try { 
  666.                         $adjusted_time = new DateTime( $v, $tzid ); 
  667.                         $adjusted_time->setTimeZone( new DateTimeZone( 'UTC' ) ); 
  668.                         $adjusted_times[] = $adjusted_time->format('Ymd\THis'); 
  669.                     } catch ( Exception $e ) { 
  670.                         // Invalid argument to DateTime 
  671.                         return; 
  672.                     } 
  673.                 } 
  674.                 $value = $adjusted_times; 
  675.             } 
  676.  
  677.             // Format for adding to event 
  678.             $keyword = $keyword[0]; 
  679.             if ( 'EXDATE' != $keyword ) { 
  680.                 $value = implode( (array) $value ); 
  681.             } 
  682.         } 
  683.  
  684.         foreach ( (array) $value as $v ) { 
  685.             switch ($component) { 
  686.                 case 'VTODO': 
  687.                     if ( 'EXDATE' == $keyword ) { 
  688.                         $this->cal[ $component ][ $this->todo_count - 1 ][ $keyword ][] = $v; 
  689.                     } else { 
  690.                         $this->cal[ $component ][ $this->todo_count - 1 ][ $keyword ] = $v; 
  691.                     } 
  692.                     break; 
  693.                 case 'VEVENT': 
  694.                     if ( 'EXDATE' == $keyword ) { 
  695.                         $this->cal[ $component ][ $this->event_count - 1 ][ $keyword ][] = $v; 
  696.                     } else { 
  697.                         $this->cal[ $component ][ $this->event_count - 1 ][ $keyword ] = $v; 
  698.                     } 
  699.                     break; 
  700.                 default: 
  701.                     $this->cal[ $component ][ $keyword ] = $v; 
  702.                     break; 
  703.             } 
  704.         } 
  705.         $this->last_keyword = $keyword; 
  706.     }

  708.     /** 
  709.      * Escape strings with wp_kses, allow links 
  710.      * 
  711.      * @param string $string (default: '') 
  712.      * @return string 
  713.      */ 
  714.     public function escape( $string = '' ) { 
  715.         // Unfold content lines per RFC 5545 
  716.         $string = str_replace( "\n\t", '', $string ); 
  717.         $string = str_replace( "\n ", '', $string ); 
  718.  
  719.         $allowed_html = array( 
  720.             'a' => array( 
  721.                 'href' => array(),  
  722.                 'title' => array() 
  723.  ) 
  724.  ); 
  725.  
  726.         $allowed_tags = ''; 
  727.         foreach ( array_keys( $allowed_html ) as $tag ) { 
  728.             $allowed_tags .= "<{$tag}>"; 
  729.         } 
  730.  
  731.         // Running strip_tags() first with allowed tags to get rid of remaining gallery markup, etc 
  732.         // because wp_kses() would only htmlentity'fy that. Then still running wp_kses(), for extra 
  733.         // safety and good measure. 
  734.         return wp_kses( strip_tags( $string, $allowed_tags ), $allowed_html ); 
  735.     }

  807.     public function formatted_date( $event ) { 
  808.  
  809.         $date_format = get_option( 'date_format' ); 
  810.         $time_format = get_option( 'time_format' ); 
  811.         $start = strtotime( $event['DTSTART'] ); 
  812.         $end = isset( $event['DTEND'] ) ? strtotime( $event['DTEND'] ) : false; 
  813.  
  814.         $all_day = ( 8 == strlen( $event['DTSTART'] ) ); 
  815.  
  816.         if ( !$all_day && $this->timezone ) { 
  817.             try { 
  818.                 $start_time = new DateTime( $event['DTSTART'] ); 
  819.                 $timezone_offset = $this->timezone->getOffset( $start_time ); 
  820.                 $start += $timezone_offset; 
  821.  
  822.                 if ( $end ) { 
  823.                     $end += $timezone_offset; 
  824.                 } 
  825.             } catch ( Exception $e ) { 
  826.                 // Invalid argument to DateTime 
  827.             } 
  828.         } 
  829.         $single_day = $end ? ( $end - $start ) <= DAY_IN_SECONDS : true; 
  830.  
  831.         /** Translators: Date and time */ 
  832.         $date_with_time = __( '%1$s at %2$s' , 'jetpack' ); 
  833.         /** Translators: Two dates with a separator */ 
  834.         $two_dates = __( '%1$s – %2$s' , 'jetpack' ); 
  835.  
  836.         // we'll always have the start date. Maybe with time 
  837.         if ( $all_day ) 
  838.             $date = date_i18n( $date_format, $start ); 
  839.         else 
  840.             $date = sprintf( $date_with_time, date_i18n( $date_format, $start ), date_i18n( $time_format, $start ) ); 
  841.  
  842.         // single day, timed 
  843.         if ( $single_day && ! $all_day && false !== $end ) 
  844.             $date = sprintf( $two_dates, $date, date_i18n( $time_format, $end ) ); 
  845.  
  846.         // multi-day 
  847.         if ( ! $single_day ) { 
  848.  
  849.             if ( $all_day ) { 
  850.                 // DTEND for multi-day events represents "until", not "including", so subtract one minute 
  851.                 $end_date = date_i18n( $date_format, $end - 60 ); 
  852.             } else { 
  853.                 $end_date = sprintf( $date_with_time, date_i18n( $date_format, $end ), date_i18n( $time_format, $end ) ); 
  854.             } 
  855.  
  856.             $date = sprintf( $two_dates, $date, $end_date ); 
  857.  
  858.         } 
  859.  
  860.         return $date; 
  861.     }

  863.     protected function sort_by_recent( $list ) { 
  864.         $dates = $sorted_list = array(); 
  865.  
  866.         foreach ( $list as $key => $row ) {

  879.     }
