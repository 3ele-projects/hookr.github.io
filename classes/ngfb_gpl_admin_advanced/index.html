  511.     private function get_taglist_rows( $table_rows, $form, $network, array $opt_preg_include ) { 
  512.         $table_cells = array(); 
  513.         foreach ( $opt_preg_include as $preg ) { 
  514.             foreach ( $form->defaults as $opt => $val ) { 
  515.  
  516.                 if ( strpos( $opt, 'add_' ) !== 0 ) {    // optimize 
  517.                     continue; 
  518.                 } elseif ( isset( $this->taglist_opts[$opt] ) ) {    // check cache for tags already shown 
  519.                     continue; 
  520.                 } elseif ( ! preg_match( $preg, $opt, $match ) ) {    // check option name for a match 
  521.                     continue; 
  522.                 } 
  523.  
  524.                 $highlight = ''; 
  525.                 $this->taglist_opts[$opt] = $val; 
  526.                 switch ( $opt ) { 
  527.                     // disable with a constant instead 
  528.                     case 'add_meta_name_generator': 
  529.                         continue 2; 
  530.                     // highlight important meta tags 
  531.                     case 'add_link_rel_canonical': 
  532.                     case 'add_meta_name_description': 
  533.                         $highlight = ' highlight'; 
  534.                         break; 
  535.                 } 
  536.                 $table_cells[] = '<!-- '.( implode( ' ', $match ) ).' -->'.    // required for sorting 
  537.                     '<td class="checkbox blank">'.$this->get_nocb( $form, $opt ).'</td>'. 
  538.                     '<td class="xshort'.$highlight.'">'.$match[1].'</td>'. 
  539.                     '<td class="taglist'.$highlight.'">'.$match[2].'</td>'. 
  540.                     '<th class="taglist'.$highlight.'">'.$match[3].'</th>'; 
  541.             } 
  542.         } 
  543.         return array_merge( $table_rows, SucomUtil::get_column_rows( $table_cells, 2 ) ); 
  544.     }
