  490.     } 
  491.  
  492.     /** 
  493.      * Add metadata to imported term. 
  494.      * 
  495.      * @since 0.6.2 
  496.      * 
  497.      * @param array $term    Term data from WXR import. 
  498.      * @param int   $term_id ID of the newly created term. 
  499.      */ 
  500.     protected function process_termmeta( $term, $term_id ) { 
  501.         if ( ! isset( $term['termmeta'] ) ) { 
  502.             $term['termmeta'] = array(); 
  503.         } 
  504.  
  505.         /** 
  506.          * Filters the metadata attached to an imported term. 
  507.          * 
  508.          * @since 0.6.2 
  509.          * 
  510.          * @param array $termmeta Array of term meta. 
  511.          * @param int   $term_id  ID of the newly created term. 
  512.          * @param array $term     Term data from the WXR import. 
  513.          */ 
  514.         $term['termmeta'] = apply_filters( 'wp_import_term_meta', $term['termmeta'], $term_id, $term ); 
  515.  
  516.         if ( empty( $term['termmeta'] ) ) { 
  517.             return; 
  518.         } 
  519.  
  520.         foreach ( $term['termmeta'] as $meta ) { 
  521.             /** 
  522.              * Filters the meta key for an imported piece of term meta. 
  523.              * 
  524.              * @since 0.6.2 
  525.              * 
  526.              * @param string $meta_key Meta key. 
  527.              * @param int    $term_id  ID of the newly created term. 
  528.              * @param array  $term     Term data from the WXR import. 
  529.              */ 
  530.             $key = apply_filters( 'import_term_meta_key', $meta['key'], $term_id, $term ); 
  531.             if ( ! $key ) { 
  532.                 continue; 
  533.             } 
  534.  
  535.             // Export gets meta straight from the DB so could have a serialized string 
  536.             $value = maybe_unserialize( $meta['value'] ); 
  537.  
  538.             add_term_meta( $term_id, $key, $value ); 
  539.  
  540.             /** 
  541.              * Fires after term meta is imported. 
  542.              * 
  543.              * @since 0.6.2 
  544.              * 
  545.              * @param int    $term_id ID of the newly created term. 
  546.              * @param string $key     Meta key. 
  547.              * @param mixed  $value   Meta value. 
  548.              */ 
  549.             do_action( 'import_term_meta', $term_id, $key, $value ); 
  550.         } 
  551.     }

  553.     /** 
  554.      * Create new posts based on import information 
  555.      * 
  556.      * Posts marked as having a parent which doesn't exist will become top level items. 
  557.      * Doesn't create a new post if: the post type doesn't exist, the given post ID 
  558.      * is already noted as imported or a post with the same title and date already exists. 
  559.      * Note that new/updated terms, comments and meta are imported for the last of the above. 
  560.      */ 
  561.     function process_posts() { 
  562.         $this->posts = apply_filters( 'wp_import_posts', $this->posts ); 
  563.  
  564.         foreach ( $this->posts as $post ) { 
  565.             $post = apply_filters( 'wp_import_post_data_raw', $post ); 
  566.  
  567.             if ( ! post_type_exists( $post['post_type'] ) ) { 
  568.                 printf( __( 'Failed to import “%s”: Invalid post type %s', 'wordpress-importer' ),  
  569.                     esc_html($post['post_title']), esc_html($post['post_type']) ); 
  570.                 echo '<br />'; 
  571.                 do_action( 'wp_import_post_exists', $post ); 
  572.                 continue; 
  573.             } 
  574.  
  575.             if ( isset( $this->processed_posts[$post['post_id']] ) && ! empty( $post['post_id'] ) ) 
  576.                 continue; 
  577.  
  578.             if ( $post['status'] == 'auto-draft' ) 
  579.                 continue; 
  580.  
  581.             if ( 'nav_menu_item' == $post['post_type'] ) { 
  582.                 $this->process_menu_item( $post ); 
  583.                 continue; 
  584.             } 
  585.  
  586.             $post_type_object = get_post_type_object( $post['post_type'] ); 
  587.  
  588.             $post_exists = post_exists( $post['post_title'], '', $post['post_date'] ); 
  589.  
  590.             /** 
  591.             * Filter ID of the existing post corresponding to post currently importing. 
  592.             * 
  593.             * Return 0 to force the post to be imported. Filter the ID to be something else 
  594.             * to override which existing post is mapped to the imported post. 
  595.             * 
  596.             * @see post_exists() 
  597.             * @since 0.6.2 
  598.             * 
  599.             * @param int   $post_exists  Post ID, or 0 if post did not exist. 
  600.             * @param array $post         The post array to be inserted. 
  601.             */ 
  602.             $post_exists = apply_filters( 'wp_import_existing_post', $post_exists, $post ); 
  603.  
  604.             if ( $post_exists && get_post_type( $post_exists ) == $post['post_type'] ) { 
  605.                 printf( __('%s “%s” already exists.', 'wordpress-importer'), $post_type_object->labels->singular_name, esc_html($post['post_title']) ); 
  606.                 echo '<br />'; 
  607.                 $comment_post_ID = $post_id = $post_exists; 
  608.                 $this->processed_posts[ intval( $post['post_id'] ) ] = intval( $post_exists ); 
  609.             } else { 
  610.                 $post_parent = (int) $post['post_parent']; 
  611.                 if ( $post_parent ) { 
  612.                     // if we already know the parent, map it to the new local ID 
  613.                     if ( isset( $this->processed_posts[$post_parent] ) ) { 
  614.                         $post_parent = $this->processed_posts[$post_parent]; 
  615.                     // otherwise record the parent for later 
  616.                     } else { 
  617.                         $this->post_orphans[intval($post['post_id'])] = $post_parent; 
  618.                         $post_parent = 0; 
  619.                     } 
  620.                 } 
  621.  
  622.                 // map the post author 
  623.                 $author = sanitize_user( $post['post_author'], true ); 
  624.                 if ( isset( $this->author_mapping[$author] ) ) 
  625.                     $author = $this->author_mapping[$author]; 
  626.                 else 
  627.                     $author = (int) get_current_user_id(); 
  628.  
  629.                 $postdata = array( 
  630.                     'import_id' => $post['post_id'], 'post_author' => $author, 'post_date' => $post['post_date'],  
  631.                     'post_date_gmt' => $post['post_date_gmt'], 'post_content' => $post['post_content'],  
  632.                     'post_excerpt' => $post['post_excerpt'], 'post_title' => $post['post_title'],  
  633.                     'post_status' => $post['status'], 'post_name' => $post['post_name'],  
  634.                     'comment_status' => $post['comment_status'], 'ping_status' => $post['ping_status'],  
  635.                     'guid' => $post['guid'], 'post_parent' => $post_parent, 'menu_order' => $post['menu_order'],  
  636.                     'post_type' => $post['post_type'], 'post_password' => $post['post_password'] 
  637.  ); 
  638.  
  639.                 $original_post_ID = $post['post_id']; 
  640.                 $postdata = apply_filters( 'wp_import_post_data_processed', $postdata, $post ); 
  641.  
  642.                 $postdata = wp_slash( $postdata ); 
  643.  
  644.                 if ( 'attachment' == $postdata['post_type'] ) { 
  645.                     $remote_url = ! empty($post['attachment_url']) ? $post['attachment_url'] : $post['guid']; 
  646.  
  647.                     // try to use _wp_attached file for upload folder placement to ensure the same location as the export site 
  648.                     // e.g. location is 2003/05/image.jpg but the attachment post_date is 2010/09, see media_handle_upload() 
  649.                     $postdata['upload_date'] = $post['post_date']; 
  650.                     if ( isset( $post['postmeta'] ) ) { 
  651.                         foreach( $post['postmeta'] as $meta ) { 
  652.                             if ( $meta['key'] == '_wp_attached_file' ) { 
  653.                                 if ( preg_match( '%^[0-9]{4}/[0-9]{2}%', $meta['value'], $matches ) ) 
  654.                                     $postdata['upload_date'] = $matches[0]; 
  655.                                 break; 
  656.                             } 
  657.                         } 
  658.                     } 
  659.  
  660.                     $comment_post_ID = $post_id = $this->process_attachment( $postdata, $remote_url ); 
  661.                 } else { 
  662.                     $comment_post_ID = $post_id = wp_insert_post( $postdata, true ); 
  663.                     do_action( 'wp_import_insert_post', $post_id, $original_post_ID, $postdata, $post ); 
  664.                 } 
  665.  
  666.                 if ( is_wp_error( $post_id ) ) { 
  667.                     printf( __( 'Failed to import %s “%s”', 'wordpress-importer' ),  
  668.                         $post_type_object->labels->singular_name, esc_html($post['post_title']) ); 
  669.                     if ( defined('IMPORT_DEBUG') && IMPORT_DEBUG ) 
  670.                         echo ': ' . $post_id->get_error_message(); 
  671.                     echo '<br />'; 
  672.                     continue; 
  673.                 } 
  674.  
  675.                 if ( $post['is_sticky'] == 1 ) 
  676.                     stick_post( $post_id ); 
  677.             } 
  678.  
  679.             // map pre-import ID to local ID 
  680.             $this->processed_posts[intval($post['post_id'])] = (int) $post_id; 
  681.  
  682.             if ( ! isset( $post['terms'] ) ) 
  683.                 $post['terms'] = array(); 
  684.  
  685.             $post['terms'] = apply_filters( 'wp_import_post_terms', $post['terms'], $post_id, $post ); 
  686.  
  687.             // add categories, tags and other terms 
  688.             if ( ! empty( $post['terms'] ) ) { 
  689.                 $terms_to_set = array(); 
  690.                 foreach ( $post['terms'] as $term ) { 
  691.                     // back compat with WXR 1.0 map 'tag' to 'post_tag' 
  692.                     $taxonomy = ( 'tag' == $term['domain'] ) ? 'post_tag' : $term['domain']; 
  693.                     $term_exists = term_exists( $term['slug'], $taxonomy ); 
  694.                     $term_id = is_array( $term_exists ) ? $term_exists['term_id'] : $term_exists; 
  695.                     if ( ! $term_id ) { 
  696.                         $t = wp_insert_term( $term['name'], $taxonomy, array( 'slug' => $term['slug'] ) ); 
  697.                         if ( ! is_wp_error( $t ) ) { 
  698.                             $term_id = $t['term_id']; 
  699.                             do_action( 'wp_import_insert_term', $t, $term, $post_id, $post ); 
  700.                         } else { 
  701.                             printf( __( 'Failed to import %s %s', 'wordpress-importer' ), esc_html($taxonomy), esc_html($term['name']) ); 
  702.                             if ( defined('IMPORT_DEBUG') && IMPORT_DEBUG ) 
  703.                                 echo ': ' . $t->get_error_message(); 
  704.                             echo '<br />'; 
  705.                             do_action( 'wp_import_insert_term_failed', $t, $term, $post_id, $post ); 
  706.                             continue; 
  707.                         } 
  708.                     } 
  709.                     $terms_to_set[$taxonomy][] = intval( $term_id ); 
  710.                 } 
  711.  
  712.                 foreach ( $terms_to_set as $tax => $ids ) { 
  713.                     $tt_ids = wp_set_post_terms( $post_id, $ids, $tax ); 
  714.                     do_action( 'wp_import_set_post_terms', $tt_ids, $ids, $tax, $post_id, $post ); 
  715.                 } 
  716.                 unset( $post['terms'], $terms_to_set ); 
  717.             } 
  718.  
  719.             if ( ! isset( $post['comments'] ) ) 
  720.                 $post['comments'] = array(); 
  721.  
  722.             $post['comments'] = apply_filters( 'wp_import_post_comments', $post['comments'], $post_id, $post ); 
  723.  
  724.             // add/update comments 
  725.             if ( ! empty( $post['comments'] ) ) { 
  726.                 $num_comments = 0; 
  727.                 $inserted_comments = array(); 
  728.                 foreach ( $post['comments'] as $comment ) { 
  729.                     $comment_id = $comment['comment_id']; 
  730.                     $newcomments[$comment_id]['comment_post_ID'] = $comment_post_ID; 
  731.                     $newcomments[$comment_id]['comment_author'] = $comment['comment_author']; 
  732.                     $newcomments[$comment_id]['comment_author_email'] = $comment['comment_author_email']; 
  733.                     $newcomments[$comment_id]['comment_author_IP'] = $comment['comment_author_IP']; 
  734.                     $newcomments[$comment_id]['comment_author_url'] = $comment['comment_author_url']; 
  735.                     $newcomments[$comment_id]['comment_date'] = $comment['comment_date']; 
  736.                     $newcomments[$comment_id]['comment_date_gmt'] = $comment['comment_date_gmt']; 
  737.                     $newcomments[$comment_id]['comment_content'] = $comment['comment_content']; 
  738.                     $newcomments[$comment_id]['comment_approved'] = $comment['comment_approved']; 
  739.                     $newcomments[$comment_id]['comment_type'] = $comment['comment_type']; 
  740.                     $newcomments[$comment_id]['comment_parent'] = $comment['comment_parent']; 
  741.                     $newcomments[$comment_id]['commentmeta'] = isset( $comment['commentmeta'] ) ? $comment['commentmeta'] : array(); 
  742.                     if ( isset( $this->processed_authors[$comment['comment_user_id']] ) ) 
  743.                         $newcomments[$comment_id]['user_id'] = $this->processed_authors[$comment['comment_user_id']]; 
  744.                 } 
  745.                 ksort( $newcomments ); 
  746.  
  747.                 foreach ( $newcomments as $key => $comment ) { 
  748.                     // if this is a new post we can skip the comment_exists() check 
  749.                     if ( ! $post_exists || ! comment_exists( $comment['comment_author'], $comment['comment_date'] ) ) { 
  750.                         if ( isset( $inserted_comments[$comment['comment_parent']] ) ) 
  751.                             $comment['comment_parent'] = $inserted_comments[$comment['comment_parent']]; 
  752.                         $comment = wp_filter_comment( $comment ); 
  753.                         $inserted_comments[$key] = wp_insert_comment( $comment ); 
  754.                         do_action( 'wp_import_insert_comment', $inserted_comments[$key], $comment, $comment_post_ID, $post ); 
  755.  
  756.                         foreach( $comment['commentmeta'] as $meta ) { 
  757.                             $value = maybe_unserialize( $meta['value'] ); 
  758.                             add_comment_meta( $inserted_comments[$key], $meta['key'], $value ); 
  759.                         } 
  760.  
  761.                         $num_comments++; 
  762.                     } 
  763.                 } 
  764.                 unset( $newcomments, $inserted_comments, $post['comments'] ); 
  765.             } 
  766.  
  767.             if ( ! isset( $post['postmeta'] ) ) 
  768.                 $post['postmeta'] = array(); 
  769.  
  770.             $post['postmeta'] = apply_filters( 'wp_import_post_meta', $post['postmeta'], $post_id, $post ); 
  771.  
  772.             // add/update post meta 
  773.             if ( ! empty( $post['postmeta'] ) ) { 
  774.                 foreach ( $post['postmeta'] as $meta ) { 
  775.                     $key = apply_filters( 'import_post_meta_key', $meta['key'], $post_id, $post ); 
  776.                     $value = false; 
  777.  
  778.                     if ( '_edit_last' == $key ) { 
  779.                         if ( isset( $this->processed_authors[intval($meta['value'])] ) ) 
  780.                             $value = $this->processed_authors[intval($meta['value'])]; 
  781.                         else 
  782.                             $key = false; 
  783.                     } 
  784.  
  785.                     if ( $key ) { 
  786.                         // export gets meta straight from the DB so could have a serialized string 
  787.                         if ( ! $value ) 
  788.                             $value = maybe_unserialize( $meta['value'] ); 
  789.  
  790.                         add_post_meta( $post_id, $key, $value ); 
  791.                         do_action( 'import_post_meta', $post_id, $key, $value ); 
  792.  
  793.                         // if the post has a featured image, take note of this in case of remap 
  794.                         if ( '_thumbnail_id' == $key ) 
  795.                             $this->featured_images[$post_id] = (int) $value; 
  796.                     } 
  797.                 } 
  798.             } 
  799.         } 
  800.  
  801.         unset( $this->posts ); 
  802.     }

  804.     /** 
  805.      * Attempt to create a new menu item from import data 
  806.      * 
  807.      * Fails for draft, orphaned menu items and those without an associated nav_menu 
  808.      * or an invalid nav_menu term. If the post type or term object which the menu item 
  809.      * represents doesn't exist then the menu item will not be imported (waits until the 
  810.      * end of the import to retry again before discarding). 
  811.      * 
  812.      * @param array $item Menu item details from WXR file 
  813.      */ 
  814.     function process_menu_item( $item ) { 
  815.         // skip draft, orphaned menu items 
  816.         if ( 'draft' == $item['status'] ) 
  817.             return; 
  818.  
  819.         $menu_slug = false; 
  820.         if ( isset($item['terms']) ) { 
  821.             // loop through terms, assume first nav_menu term is correct menu 
  822.             foreach ( $item['terms'] as $term ) { 
  823.                 if ( 'nav_menu' == $term['domain'] ) { 
  824.                     $menu_slug = $term['slug']; 
  825.                     break; 
  826.                 } 
  827.             } 
  828.         } 
  829.  
  830.         // no nav_menu term associated with this menu item 
  831.         if ( ! $menu_slug ) { 
  832.             _e( 'Menu item skipped due to missing menu slug', 'wordpress-importer' ); 
  833.             echo '<br />'; 
  834.             return; 
  835.         } 
  836.  
  837.         $menu_id = term_exists( $menu_slug, 'nav_menu' ); 
  838.         if ( ! $menu_id ) { 
  839.             printf( __( 'Menu item skipped due to invalid menu slug: %s', 'wordpress-importer' ), esc_html( $menu_slug ) ); 
  840.             echo '<br />'; 
  841.             return; 
  842.         } else { 
  843.             $menu_id = is_array( $menu_id ) ? $menu_id['term_id'] : $menu_id; 
  844.         } 
  845.  
  846.         foreach ( $item['postmeta'] as $meta ) 
  847.             $$meta['key'] = $meta['value']; 
  848.  
  849.         if ( 'taxonomy' == $_menu_item_type && isset( $this->processed_terms[intval($_menu_item_object_id)] ) ) { 
  850.             $_menu_item_object_id = $this->processed_terms[intval($_menu_item_object_id)]; 
  851.         } else if ( 'post_type' == $_menu_item_type && isset( $this->processed_posts[intval($_menu_item_object_id)] ) ) { 
  852.             $_menu_item_object_id = $this->processed_posts[intval($_menu_item_object_id)]; 
  853.         } else if ( 'custom' != $_menu_item_type ) { 
  854.             // associated object is missing or not imported yet, we'll retry later 
  855.             $this->missing_menu_items[] = $item; 
  856.             return; 
  857.         } 
  858.  
  859.         if ( isset( $this->processed_menu_items[intval($_menu_item_menu_item_parent)] ) ) { 
  860.             $_menu_item_menu_item_parent = $this->processed_menu_items[intval($_menu_item_menu_item_parent)]; 
  861.         } else if ( $_menu_item_menu_item_parent ) { 
  862.             $this->menu_item_orphans[intval($item['post_id'])] = (int) $_menu_item_menu_item_parent; 
  863.             $_menu_item_menu_item_parent = 0; 
  864.         } 
  865.  
  866.         // wp_update_nav_menu_item expects CSS classes as a space separated string 
  867.         $_menu_item_classes = maybe_unserialize( $_menu_item_classes ); 
  868.         if ( is_array( $_menu_item_classes ) ) 
  869.             $_menu_item_classes = implode( ' ', $_menu_item_classes ); 
  870.  
  871.         $args = array( 
  872.             'menu-item-object-id' => $_menu_item_object_id,  
  873.             'menu-item-object' => $_menu_item_object,  
  874.             'menu-item-parent-id' => $_menu_item_menu_item_parent,  
  875.             'menu-item-position' => intval( $item['menu_order'] ),  
  876.             'menu-item-type' => $_menu_item_type,  
  877.             'menu-item-title' => $item['post_title'],  
  878.             'menu-item-url' => $_menu_item_url,  
  879.             'menu-item-description' => $item['post_content'],  
  880.             'menu-item-attr-title' => $item['post_excerpt'],  
  881.             'menu-item-target' => $_menu_item_target,  
  882.             'menu-item-classes' => $_menu_item_classes,  
  883.             'menu-item-xfn' => $_menu_item_xfn,  
  884.             'menu-item-status' => $item['status'] 
  885.  ); 
  886.  
  887.         $id = wp_update_nav_menu_item( $menu_id, 0, $args ); 
  888.         if ( $id && ! is_wp_error( $id ) ) 
  889.             $this->processed_menu_items[intval($item['post_id'])] = (int) $id; 
  890.     } 
  891.  
  892.     /** 
  893.      * If fetching attachments is enabled then attempt to create a new attachment 
  894.      * 
  895.      * @param array $post Attachment post details from WXR 
  896.      * @param string $url URL to fetch attachment from 
  897.      * @return int|WP_Error Post ID on success, WP_Error otherwise 
  898.      */ 
  899.     function process_attachment( $post, $url ) { 
  900.         if ( ! $this->fetch_attachments ) 
  901.             return new WP_Error( 'attachment_processing_error',  
  902.                 __( 'Fetching attachments is not enabled', 'wordpress-importer' ) ); 
  903.  
  904.         // if the URL is absolute, but does not contain address, then upload it assuming base_site_url 
  905.         if ( preg_match( '|^/[\w\W]+$|', $url ) ) 
  906.             $url = rtrim( $this->base_url, '/' ) . $url; 
  907.  
  908.         $upload = $this->fetch_remote_file( $url, $post ); 
  909.         if ( is_wp_error( $upload ) ) 
  910.             return $upload; 
  911.  
  912.         if ( $info = wp_check_filetype( $upload['file'] ) ) 
  913.             $post['post_mime_type'] = $info['type']; 
  914.         else 
  915.             return new WP_Error( 'attachment_processing_error', __('Invalid file type', 'wordpress-importer') ); 
  916.  
  917.         $post['guid'] = $upload['url']; 
  918.  
  919.         // as per wp-admin/includes/upload.php 
  920.         $post_id = wp_insert_attachment( $post, $upload['file'] ); 
  921.         wp_update_attachment_metadata( $post_id, wp_generate_attachment_metadata( $post_id, $upload['file'] ) ); 
  922.  
  923.         // remap resized image URLs, works by stripping the extension and remapping the URL stub. 
  924.         if ( preg_match( '!^image/!', $info['type'] ) ) { 
  925.             $parts = pathinfo( $url ); 
  926.             $name = basename( $parts['basename'], ".{$parts['extension']}" ); // PATHINFO_FILENAME in PHP 5.2 
  927.  
  928.             $parts_new = pathinfo( $upload['url'] ); 
  929.             $name_new = basename( $parts_new['basename'], ".{$parts_new['extension']}" ); 
  930.  
  931.             $this->url_remap[$parts['dirname'] . '/' . $name] = $parts_new['dirname'] . '/' . $name_new; 
  932.         } 
  933.  
  934.         return $post_id; 
  935.     } 
  936.  
  937.     /** 
  938.      * Attempt to download a remote file attachment 
  939.      * 
  940.      * @param string $url URL of item to fetch 
  941.      * @param array $post Attachment details 
  942.      * @return array|WP_Error Local file location details on success, WP_Error otherwise 
  943.      */ 
  944.     function fetch_remote_file( $url, $post ) { 
  945.         // extract the file name and extension from the url 
  946.         $file_name = basename( $url ); 
  947.  
  948.         // get placeholder file in the upload dir with a unique, sanitized filename 
  949.         $upload = wp_upload_bits( $file_name, 0, '', $post['upload_date'] ); 
  950.         if ( $upload['error'] ) 
  951.             return new WP_Error( 'upload_dir_error', $upload['error'] ); 
  952.  
  953.         // fetch the remote url and write it to the placeholder file 
  954.         $headers = wp_get_http( $url, $upload['file'] ); 
  955.  
  956.         // request failed 
  957.         if ( ! $headers ) { 
  958.             @unlink( $upload['file'] ); 
  959.             return new WP_Error( 'import_file_error', __('Remote server did not respond', 'wordpress-importer') ); 
  960.         } 
  961.  
  962.         // make sure the fetch was successful 
  963.         if ( $headers['response'] != '200' ) { 
  964.             @unlink( $upload['file'] ); 
  965.             return new WP_Error( 'import_file_error', sprintf( __('Remote server returned error response %1$d %2$s', 'wordpress-importer'), esc_html($headers['response']), get_status_header_desc($headers['response']) ) ); 
  966.         } 
  967.  
  968.         $filesize = filesize( $upload['file'] ); 
  969.  
  970.         if ( isset( $headers['content-length'] ) && $filesize != $headers['content-length'] ) { 
  971.             @unlink( $upload['file'] ); 
  972.             return new WP_Error( 'import_file_error', __('Remote file is incorrect size', 'wordpress-importer') ); 
  973.         } 
  974.  
  975.         if ( 0 == $filesize ) { 
  976.             @unlink( $upload['file'] ); 
  977.             return new WP_Error( 'import_file_error', __('Zero size file downloaded', 'wordpress-importer') ); 
  978.         } 
  979.  
  980.         $max_size = (int) $this->max_attachment_size(); 
  981.         if ( ! empty( $max_size ) && $filesize > $max_size ) { 
  982.             @unlink( $upload['file'] ); 
  983.             return new WP_Error( 'import_file_error', sprintf(__('Remote file is too large, limit is %s', 'wordpress-importer'), size_format($max_size) ) ); 
  984.         } 
  985.  
  986.         // keep track of the old and new urls so we can substitute them later 
  987.         $this->url_remap[$url] = $upload['url']; 
  988.         $this->url_remap[$post['guid']] = $upload['url']; // r13735, really needed? 
  989.         // keep track of the destination if the remote url is redirected somewhere else 
  990.         if ( isset($headers['x-final-location']) && $headers['x-final-location'] != $url ) 
  991.             $this->url_remap[$headers['x-final-location']] = $upload['url']; 
  992.  
  993.         return $upload; 
  994.     }

  996.     /** 
  997.      * Attempt to associate posts and menu items with previously missing parents 
  998.      * 
  999.      * An imported post's parent may not have been imported when it was first created 
  1000.      * so try again. Similarly for child menu items and menu items which were missing 
  1001.      * the object (e.g. post) they represent in the menu 
  1002.      */ 
  1003.     function backfill_parents() { 
  1004.         global $wpdb; 
  1005.  
  1006.         // find parents for post orphans 
  1007.         foreach ( $this->post_orphans as $child_id => $parent_id ) { 
  1008.             $local_child_id = $local_parent_id = false; 
  1009.             if ( isset( $this->processed_posts[$child_id] ) ) 
  1010.                 $local_child_id = $this->processed_posts[$child_id]; 
  1011.             if ( isset( $this->processed_posts[$parent_id] ) ) 
  1012.                 $local_parent_id = $this->processed_posts[$parent_id]; 
  1013.  
  1014.             if ( $local_child_id && $local_parent_id ) 
  1015.                 $wpdb->update( $wpdb->posts, array( 'post_parent' => $local_parent_id ), array( 'ID' => $local_child_id ), '%d', '%d' ); 
  1016.         } 
  1017.  
  1018.         // all other posts/terms are imported, retry menu items with missing associated object 
  1019.         $missing_menu_items = $this->missing_menu_items; 
  1020.         foreach ( $missing_menu_items as $item ) 
  1021.             $this->process_menu_item( $item ); 
  1022.  
  1023.         // find parents for menu item orphans 
  1024.         foreach ( $this->menu_item_orphans as $child_id => $parent_id ) { 
  1025.             $local_child_id = $local_parent_id = 0; 
  1026.             if ( isset( $this->processed_menu_items[$child_id] ) ) 
  1027.                 $local_child_id = $this->processed_menu_items[$child_id]; 
  1028.             if ( isset( $this->processed_menu_items[$parent_id] ) ) 
  1029.                 $local_parent_id = $this->processed_menu_items[$parent_id]; 
  1030.  
  1031.             if ( $local_child_id && $local_parent_id ) 
  1032.                 update_post_meta( $local_child_id, '_menu_item_menu_item_parent', (int) $local_parent_id ); 
  1033.         } 
  1034.     }
