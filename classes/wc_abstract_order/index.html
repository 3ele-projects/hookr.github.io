  630.     /** 
  631.      * Calculate taxes for all line items and shipping, and store the totals and tax rows. 
  632.      * 
  633.      * Will use the base country unless customer addresses are set. 
  634.      * 
  635.      * @return bool success or fail. 
  636.      */ 
  637.     public function calculate_taxes() { 
  638.         $tax_total = 0; 
  639.         $shipping_tax_total = 0; 
  640.         $taxes = array(); 
  641.         $shipping_taxes = array(); 
  642.         $tax_based_on = get_option( 'woocommerce_tax_based_on' ); 
  643.  
  644.         // If is_vat_exempt is 'yes', or wc_tax_enabled is false, return and do nothing. 
  645.         if ( 'yes' === $this->is_vat_exempt || ! wc_tax_enabled() ) { 
  646.             return false; 
  647.         } 
  648.  
  649.         if ( 'billing' === $tax_based_on ) { 
  650.             $country = $this->billing_country; 
  651.             $state = $this->billing_state; 
  652.             $postcode = $this->billing_postcode; 
  653.             $city = $this->billing_city; 
  654.         } elseif ( 'shipping' === $tax_based_on ) { 
  655.             $country = $this->shipping_country; 
  656.             $state = $this->shipping_state; 
  657.             $postcode = $this->shipping_postcode; 
  658.             $city = $this->shipping_city; 
  659.         } 
  660.  
  661.         // Default to base 
  662.         if ( 'base' === $tax_based_on || empty( $country ) ) { 
  663.             $default = wc_get_base_location(); 
  664.             $country = $default['country']; 
  665.             $state = $default['state']; 
  666.             $postcode = ''; 
  667.             $city = ''; 
  668.         } 
  669.  
  670.         // Get items 
  671.         foreach ( $this->get_items( array( 'line_item', 'fee' ) ) as $item_id => $item ) { 
  672.  
  673.             $product = $this->get_product_from_item( $item ); 
  674.             $line_total = isset( $item['line_total'] ) ? $item['line_total'] : 0; 
  675.             $line_subtotal = isset( $item['line_subtotal'] ) ? $item['line_subtotal'] : 0; 
  676.             $tax_class = $item['tax_class']; 
  677.             $item_tax_status = $product ? $product->get_tax_status() : 'taxable'; 
  678.  
  679.             if ( '0' !== $tax_class && 'taxable' === $item_tax_status ) { 
  680.  
  681.                 $tax_rates = WC_Tax::find_rates( array( 
  682.                     'country' => $country,  
  683.                     'state' => $state,  
  684.                     'postcode' => $postcode,  
  685.                     'city' => $city,  
  686.                     'tax_class' => $tax_class 
  687.  ) ); 
  688.  
  689.                 $line_subtotal_taxes = WC_Tax::calc_tax( $line_subtotal, $tax_rates, false ); 
  690.                 $line_taxes = WC_Tax::calc_tax( $line_total, $tax_rates, false ); 
  691.                 $line_subtotal_tax = max( 0, array_sum( $line_subtotal_taxes ) ); 
  692.                 $line_tax = max( 0, array_sum( $line_taxes ) ); 
  693.                 $tax_total           += $line_tax; 
  694.  
  695.                 wc_update_order_item_meta( $item_id, '_line_subtotal_tax', wc_format_decimal( $line_subtotal_tax ) ); 
  696.                 wc_update_order_item_meta( $item_id, '_line_tax', wc_format_decimal( $line_tax ) ); 
  697.                 wc_update_order_item_meta( $item_id, '_line_tax_data', array( 'total' => $line_taxes, 'subtotal' => $line_subtotal_taxes ) ); 
  698.  
  699.                 // Sum the item taxes 
  700.                 foreach ( array_keys( $taxes + $line_taxes ) as $key ) { 
  701.                     $taxes[ $key ] = ( isset( $line_taxes[ $key ] ) ? $line_taxes[ $key ] : 0 ) + ( isset( $taxes[ $key ] ) ? $taxes[ $key ] : 0 ); 
  702.                 } 
  703.             } 
  704.         } 
  705.  
  706.         // Calc taxes for shipping 
  707.         foreach ( $this->get_shipping_methods() as $item_id => $item ) { 
  708.             $shipping_tax_class = get_option( 'woocommerce_shipping_tax_class' ); 
  709.  
  710.             // Inherit tax class from items 
  711.             if ( '' === $shipping_tax_class ) { 
  712.                 $tax_classes = WC_Tax::get_tax_classes(); 
  713.                 $found_tax_classes = $this->get_items_tax_classes(); 
  714.  
  715.                 foreach ( $tax_classes as $tax_class ) { 
  716.                     $tax_class = sanitize_title( $tax_class ); 
  717.                     if ( in_array( $tax_class, $found_tax_classes ) ) { 
  718.                         $tax_rates = WC_Tax::find_shipping_rates( array( 
  719.                             'country' => $country,  
  720.                             'state' => $state,  
  721.                             'postcode' => $postcode,  
  722.                             'city' => $city,  
  723.                             'tax_class' => $tax_class,  
  724.  ) ); 
  725.                         break; 
  726.                     } 
  727.                 } 
  728.             } else { 
  729.                 $tax_rates = WC_Tax::find_shipping_rates( array( 
  730.                     'country' => $country,  
  731.                     'state' => $state,  
  732.                     'postcode' => $postcode,  
  733.                     'city' => $city,  
  734.                     'tax_class' => 'standard' === $shipping_tax_class ? '' : $shipping_tax_class,  
  735.  ) ); 
  736.             } 
  737.  
  738.             $line_taxes = WC_Tax::calc_tax( $item['cost'], $tax_rates, false ); 
  739.             $line_tax = max( 0, array_sum( $line_taxes ) ); 
  740.             $shipping_tax_total += $line_tax; 
  741.  
  742.             wc_update_order_item_meta( $item_id, '_line_tax', wc_format_decimal( $line_tax ) ); 
  743.             wc_update_order_item_meta( $item_id, '_line_tax_data', array( 'total' => $line_taxes ) ); 
  744.  
  745.             // Sum the item taxes 
  746.             foreach ( array_keys( $shipping_taxes + $line_taxes ) as $key ) { 
  747.                 $shipping_taxes[ $key ] = ( isset( $line_taxes[ $key ] ) ? $line_taxes[ $key ] : 0 ) + ( isset( $shipping_taxes[ $key ] ) ? $shipping_taxes[ $key ] : 0 ); 
  748.             } 
  749.         } 
  750.  
  751.         // Save tax totals 
  752.         $this->set_total( $shipping_tax_total, 'shipping_tax' ); 
  753.         $this->set_total( $tax_total, 'tax' ); 
  754.  
  755.         // Tax rows 
  756.         $this->remove_order_items( 'tax' ); 
  757.  
  758.         // Now merge to keep tax rows 
  759.         foreach ( array_keys( $taxes + $shipping_taxes ) as $tax_rate_id ) { 
  760.             $this->add_tax( $tax_rate_id, isset( $taxes[ $tax_rate_id ] ) ? $taxes[ $tax_rate_id ] : 0, isset( $shipping_taxes[ $tax_rate_id ] ) ? $shipping_taxes[ $tax_rate_id ] : 0 ); 
  761.         } 
  762.  
  763.         return true; 
  764.     }
