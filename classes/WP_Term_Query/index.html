  521.         if ( $args['hide_empty'] && !$hierarchical ) { 
  522.             $this->sql_clauses['where']['count'] = 'tt.count > 0'; 
  523.         } 
  524.  
  525.         $number = $args['number']; 
  526.         $offset = $args['offset']; 
  527.  
  528.         // Don't limit the query results when we have to descend the family tree. 
  529.         if ( $number && ! $hierarchical && ! $child_of && '' === $parent ) { 
  530.             if ( $offset ) { 
  531.                 $limits = 'LIMIT ' . $offset . ', ' . $number; 
  532.             } else { 
  533.                 $limits = 'LIMIT ' . $number; 
  534.             } 
  535.         } else { 
  536.             $limits = ''; 
  537.         } 
  538.  
  539.  
  540.         if ( ! empty( $args['search'] ) ) { 
  541.             $this->sql_clauses['where']['search'] = $this->get_search_sql( $args['search'] ); 
  542.         } 
  543.  
  544.         // Meta query support. 
  545.         $join = ''; 
  546.         $distinct = ''; 
  547.  
  548.         // Reparse meta_query query_vars, in case they were modified in a 'pre_get_terms' callback. 
  549.         $this->meta_query->parse_query_vars( $this->query_vars ); 
  550.         $mq_sql = $this->meta_query->get_sql( 'term', 't', 'term_id' ); 
  551.         $meta_clauses = $this->meta_query->get_clauses(); 
  552.  
  553.         if ( ! empty( $meta_clauses ) ) { 
  554.             $join .= $mq_sql['join']; 
  555.             $this->sql_clauses['where']['meta_query'] = preg_replace( '/^\s*AND\s*/', '', $mq_sql['where'] ); 
  556.             $distinct .= "DISTINCT"; 
  557.  
  558.         } 
  559.  
  560.         $selects = array(); 
  561.         switch ( $args['fields'] ) { 
  562.             case 'all': 
  563.             case 'all_with_object_id' : 
  564.             case 'tt_ids' : 
  565.             case 'slugs' : 
  566.                 $selects = array( 't.*', 'tt.*' ); 
  567.                 if ( 'all_with_object_id' === $args['fields'] && ! empty( $args['object_ids'] ) ) { 
  568.                     $selects[] = 'tr.object_id'; 
  569.                 } 
  570.                 break; 
  571.             case 'ids': 
  572.             case 'id=>parent': 
  573.                 $selects = array( 't.term_id', 'tt.parent', 'tt.count', 'tt.taxonomy' ); 
  574.                 break; 
  575.             case 'names': 
  576.                 $selects = array( 't.term_id', 'tt.parent', 'tt.count', 't.name', 'tt.taxonomy' ); 
  577.                 break; 
  578.             case 'count': 
  579.                 $orderby = ''; 
  580.                 $order = ''; 
  581.                 $selects = array( 'COUNT(*)' ); 
  582.                 break; 
  583.             case 'id=>name': 
  584.                 $selects = array( 't.term_id', 't.name', 'tt.count', 'tt.taxonomy' ); 
  585.                 break; 
  586.             case 'id=>slug': 
  587.                 $selects = array( 't.term_id', 't.slug', 'tt.count', 'tt.taxonomy' ); 
  588.                 break; 
  589.         } 
  590.  
  591.         $_fields = $args['fields']; 
  592.  
  593.         /** 
  594.          * Filters the fields to select in the terms query. 
  595.          * 
  596.          * Field lists modified using this filter will only modify the term fields returned 
  597.          * by the function when the `$fields` parameter set to 'count' or 'all'. In all other 
  598.          * cases, the term fields in the results array will be determined by the `$fields` 
  599.          * parameter alone. 
  600.          * 
  601.          * Use of this filter can result in unpredictable behavior, and is not recommended. 
  602.          * 
  603.          * @since 2.8.0 
  604.          * 
  605.          * @param array $selects    An array of fields to select for the terms query. 
  606.          * @param array $args       An array of term query arguments. 
  607.          * @param array $taxonomies An array of taxonomies. 
  608.          */ 
  609.         $fields = implode( ', ', apply_filters( 'get_terms_fields', $selects, $args, $taxonomies ) ); 
  610.  
  611.         $join .= " INNER JOIN $wpdb->term_taxonomy AS tt ON t.term_id = tt.term_id"; 
  612.  
  613.         if ( ! empty( $this->query_vars['object_ids'] ) ) { 
  614.             $join .= " INNER JOIN {$wpdb->term_relationships} AS tr ON tr.term_taxonomy_id = tt.term_taxonomy_id"; 
  615.         } 
  616.  
  617.         $where = implode( ' AND ', $this->sql_clauses['where'] ); 
  618.  
  619.         /** 
  620.          * Filters the terms query SQL clauses. 
  621.          * 
  622.          * @since 3.1.0 
  623.          * 
  624.          * @param array $pieces     Terms query SQL clauses. 
  625.          * @param array $taxonomies An array of taxonomies. 
  626.          * @param array $args       An array of terms query arguments. 
  627.          */ 
  628.         $clauses = apply_filters( 'terms_clauses', compact( 'fields', 'join', 'where', 'distinct', 'orderby', 'order', 'limits' ), $taxonomies, $args ); 
  629.  
  630.         $fields = isset( $clauses[ 'fields' ] ) ? $clauses[ 'fields' ] : ''; 
  631.         $join = isset( $clauses[ 'join' ] ) ? $clauses[ 'join' ] : ''; 
  632.         $where = isset( $clauses[ 'where' ] ) ? $clauses[ 'where' ] : ''; 
  633.         $distinct = isset( $clauses[ 'distinct' ] ) ? $clauses[ 'distinct' ] : ''; 
  634.         $orderby = isset( $clauses[ 'orderby' ] ) ? $clauses[ 'orderby' ] : ''; 
  635.         $order = isset( $clauses[ 'order' ] ) ? $clauses[ 'order' ] : ''; 
  636.         $limits = isset( $clauses[ 'limits' ] ) ? $clauses[ 'limits' ] : ''; 
  637.  
  638.         if ( $where ) { 
  639.             $where = "WHERE $where"; 
  640.         } 
  641.  
  642.         $this->sql_clauses['select'] = "SELECT $distinct $fields"; 
  643.         $this->sql_clauses['from'] = "FROM $wpdb->terms AS t $join"; 
  644.         $this->sql_clauses['orderby'] = $orderby ? "$orderby $order" : ''; 
  645.         $this->sql_clauses['limits'] = $limits; 
  646.  
  647.         $this->request = "{$this->sql_clauses['select']} {$this->sql_clauses['from']} {$where} {$this->sql_clauses['orderby']} {$this->sql_clauses['limits']}"; 
  648.  
  649.         // $args can be anything. Only use the args defined in defaults to compute the key. 
  650.         $key = md5( serialize( wp_array_slice_assoc( $args, array_keys( $this->query_var_defaults ) ) ) . serialize( $taxonomies ) . $this->request ); 
  651.         $last_changed = wp_cache_get_last_changed( 'terms' ); 
  652.         $cache_key = "get_terms:$key:$last_changed"; 
  653.         $cache = wp_cache_get( $cache_key, 'terms' ); 
  654.         if ( false !== $cache ) { 
  655.             if ( 'all' === $_fields ) { 
  656.                 $cache = array_map( 'get_term', $cache ); 
  657.             } 
  658.  
  659.             $this->terms = $cache; 
  660.             return $this->terms; 
  661.         } 
  662.  
  663.         if ( 'count' == $_fields ) { 
  664.             $count = $wpdb->get_var( $this->request ); 
  665.             wp_cache_set( $cache_key, $count, 'terms' ); 
  666.             return $count; 
  667.         } 
  668.  
  669.         $terms = $wpdb->get_results( $this->request ); 
  670.         if ( 'all' == $_fields || 'all_with_object_id' === $_fields ) { 
  671.             update_term_cache( $terms ); 
  672.         } 
  673.  
  674.         // Prime termmeta cache. 
  675.         if ( $args['update_term_meta_cache'] ) { 
  676.             $term_ids = wp_list_pluck( $terms, 'term_id' ); 
  677.             update_termmeta_cache( $term_ids ); 
  678.         } 
  679.  
  680.         if ( empty( $terms ) ) { 
  681.             wp_cache_add( $cache_key, array(), 'terms', DAY_IN_SECONDS ); 
  682.             return array(); 
  683.         } 
  684.  
  685.         if ( $child_of ) { 
  686.             foreach ( $taxonomies as $_tax ) { 
  687.                 $children = _get_term_hierarchy( $_tax ); 
  688.                 if ( ! empty( $children ) ) { 
  689.                     $terms = _get_term_children( $child_of, $terms, $_tax ); 
  690.                 } 
  691.             } 
  692.         } 
  693.  
  694.         // Update term counts to include children. 
  695.         if ( $args['pad_counts'] && 'all' == $_fields ) { 
  696.             foreach ( $taxonomies as $_tax ) { 
  697.                 _pad_term_counts( $terms, $_tax ); 
  698.             } 
  699.         } 
  700.  
  701.         // Make sure we show empty categories that have children. 
  702.         if ( $hierarchical && $args['hide_empty'] && is_array( $terms ) ) { 
  703.             foreach ( $terms as $k => $term ) { 
  704.                 if ( ! $term->count ) { 
  705.                     $children = get_term_children( $term->term_id, $term->taxonomy ); 
  706.                     if ( is_array( $children ) ) { 
  707.                         foreach ( $children as $child_id ) { 
  708.                             $child = get_term( $child_id, $term->taxonomy ); 
  709.                             if ( $child->count ) { 
  710.                                 continue 2; 
  711.                             } 
  712.                         } 
  713.                     } 
  714.  
  715.                     // It really is empty. 
  716.                     unset( $terms[ $k ] ); 
  717.                 } 
  718.             } 
  719.         } 
  720.  
  721.         /** 
  722.          * When querying for terms connected to objects, we may get 
  723.          * duplicate results. The duplicates should be preserved if 
  724.          * `$fields` is 'all_with_object_id', but should otherwise be 
  725.          * removed. 
  726.          */ 
  727.         if ( ! empty( $args['object_ids'] ) && 'all_with_object_id' != $_fields ) { 
  728.             $_tt_ids = $_terms = array(); 
  729.             foreach ( $terms as $term ) { 
  730.                 if ( isset( $_tt_ids[ $term->term_id ] ) ) { 
  731.                     continue; 
  732.                 } 
  733.  
  734.                 $_tt_ids[ $term->term_id ] = 1; 
  735.                 $_terms[] = $term; 
  736.             } 
  737.  
  738.             $terms = $_terms; 
  739.         } 
  740.  
  741.         $_terms = array(); 
  742.         if ( 'id=>parent' == $_fields ) { 
  743.             foreach ( $terms as $term ) { 
  744.                 $_terms[ $term->term_id ] = $term->parent; 
  745.             } 
  746.         } elseif ( 'ids' == $_fields ) { 
  747.             foreach ( $terms as $term ) { 
  748.                 $_terms[] = (int) $term->term_id; 
  749.             } 
  750.         } elseif ( 'tt_ids' == $_fields ) { 
  751.             foreach ( $terms as $term ) { 
  752.                 $_terms[] = (int) $term->term_taxonomy_id; 
  753.             } 
  754.         } elseif ( 'names' == $_fields ) { 
  755.             foreach ( $terms as $term ) { 
  756.                 $_terms[] = $term->name; 
  757.             } 
  758.         } elseif ( 'slugs' == $_fields ) { 
  759.             foreach ( $terms as $term ) { 
  760.                 $_terms[] = $term->slug; 
  761.             } 
  762.         } elseif ( 'id=>name' == $_fields ) { 
  763.             foreach ( $terms as $term ) { 
  764.                 $_terms[ $term->term_id ] = $term->name; 
  765.             } 
  766.         } elseif ( 'id=>slug' == $_fields ) { 
  767.             foreach ( $terms as $term ) { 
  768.                 $_terms[ $term->term_id ] = $term->slug; 
  769.             } 
  770.         } 
  771.  
  772.         if ( ! empty( $_terms ) ) { 
  773.             $terms = $_terms; 
  774.         } 
  775.  
  776.         // Hierarchical queries are not limited, so 'offset' and 'number' must be handled now. 
  777.         if ( $hierarchical && $number && is_array( $terms ) ) { 
  778.             if ( $offset >= count( $terms ) ) { 
  779.                 $terms = array(); 
  780.             } else { 
  781.                 $terms = array_slice( $terms, $offset, $number, true ); 
  782.             } 
  783.         } 
  784.  
  785.         wp_cache_add( $cache_key, $terms, 'terms', DAY_IN_SECONDS ); 
  786.  
  787.         if ( 'all' === $_fields || 'all_with_object_id' === $_fields ) { 
  788.             $terms = array_map( 'get_term', $terms ); 
  789.         } 
  790.  
  791.         $this->terms = $terms; 
  792.         return $this->terms; 
  793.     } 
  794.  
  795.     /** 
  796.      * Parse and sanitize 'orderby' keys passed to the term query. 
  797.      * 
  798.      * @since 4.6.0 
  799.      * @access protected 
  800.      * 
  801.      * @global wpdb $wpdb WordPress database abstraction object. 
  802.      * 
  803.      * @param string $orderby_raw Alias for the field to order by. 
  804.      * @return string|false Value to used in the ORDER clause. False otherwise. 
  805.      */ 
  806.     protected function parse_orderby( $orderby_raw ) { 
  807.         $_orderby = strtolower( $orderby_raw ); 
  808.         $maybe_orderby_meta = false; 
  809.  
  810.         if ( in_array( $_orderby, array( 'term_id', 'name', 'slug', 'term_group' ), true ) ) { 
  811.             $orderby = "t.$_orderby"; 
  812.         } elseif ( in_array( $_orderby, array( 'count', 'parent', 'taxonomy', 'term_taxonomy_id', 'description' ), true ) ) { 
  813.             $orderby = "tt.$_orderby"; 
  814.         } elseif ( 'term_order' === $_orderby ) { 
  815.             $orderby = 'tr.term_order'; 
  816.         } elseif ( 'include' == $_orderby && ! empty( $this->query_vars['include'] ) ) { 
  817.             $include = implode( ', ', wp_parse_id_list( $this->query_vars['include'] ) ); 
  818.             $orderby = "FIELD( t.term_id, $include )"; 
  819.         } elseif ( 'none' == $_orderby ) { 
  820.             $orderby = ''; 
  821.         } elseif ( empty( $_orderby ) || 'id' == $_orderby || 'term_id' === $_orderby ) { 
  822.             $orderby = 't.term_id'; 
  823.         } else { 
  824.             $orderby = 't.name'; 
  825.  
  826.             // This may be a value of orderby related to meta. 
  827.             $maybe_orderby_meta = true; 
  828.         } 
  829.  
  830.         /** 
  831.          * Filters the ORDERBY clause of the terms query. 
  832.          * 
  833.          * @since 2.8.0 
  834.          * 
  835.          * @param string $orderby    `ORDERBY` clause of the terms query. 
  836.          * @param array  $args       An array of terms query arguments. 
  837.          * @param array  $taxonomies An array of taxonomies. 
  838.          */ 
  839.         $orderby = apply_filters( 'get_terms_orderby', $orderby, $this->query_vars, $this->query_vars['taxonomy'] ); 
  840.  
  841.         // Run after the 'get_terms_orderby' filter for backward compatibility. 
  842.         if ( $maybe_orderby_meta ) { 
  843.             $maybe_orderby_meta = $this->parse_orderby_meta( $_orderby ); 
  844.             if ( $maybe_orderby_meta ) { 
  845.                 $orderby = $maybe_orderby_meta; 
  846.             } 
  847.         } 
  848.  
  849.         return $orderby; 
  850.     } 
  851.  
  852.     /** 
  853.      * Generate the ORDER BY clause for an 'orderby' param that is potentially related to a meta query. 
  854.      * 
  855.      * @since 4.6.0 
  856.      * @access public 
  857.      * 
  858.      * @param string $orderby_raw Raw 'orderby' value passed to WP_Term_Query. 
  859.      * @return string 
  860.      */ 
  861.     protected function parse_orderby_meta( $orderby_raw ) { 
  862.         $orderby = ''; 
  863.  
  864.         // Tell the meta query to generate its SQL, so we have access to table aliases. 
  865.         $this->meta_query->get_sql( 'term', 't', 'term_id' ); 
  866.         $meta_clauses = $this->meta_query->get_clauses(); 
  867.         if ( ! $meta_clauses || ! $orderby_raw ) { 
  868.             return $orderby; 
  869.         } 
  870.  
  871.         $allowed_keys = array(); 
  872.         $primary_meta_key = null; 
  873.         $primary_meta_query = reset( $meta_clauses ); 
  874.         if ( ! empty( $primary_meta_query['key'] ) ) { 
  875.             $primary_meta_key = $primary_meta_query['key']; 
  876.             $allowed_keys[] = $primary_meta_key; 
  877.         } 
  878.         $allowed_keys[] = 'meta_value'; 
  879.         $allowed_keys[] = 'meta_value_num'; 
  880.         $allowed_keys = array_merge( $allowed_keys, array_keys( $meta_clauses ) ); 
  881.  
  882.         if ( ! in_array( $orderby_raw, $allowed_keys, true ) ) { 
  883.             return $orderby; 
  884.         } 
  885.  
  886.         switch( $orderby_raw ) { 
  887.             case $primary_meta_key: 
  888.             case 'meta_value': 
  889.                 if ( ! empty( $primary_meta_query['type'] ) ) { 
  890.                     $orderby = "CAST({$primary_meta_query['alias']}.meta_value AS {$primary_meta_query['cast']})"; 
  891.                 } else { 
  892.                     $orderby = "{$primary_meta_query['alias']}.meta_value"; 
  893.                 } 
  894.                 break; 
  895.  
  896.             case 'meta_value_num': 
  897.                 $orderby = "{$primary_meta_query['alias']}.meta_value+0"; 
  898.                 break; 
  899.  
  900.             default: 
  901.                 if ( array_key_exists( $orderby_raw, $meta_clauses ) ) { 
  902.                     // $orderby corresponds to a meta_query clause. 
  903.                     $meta_clause = $meta_clauses[ $orderby_raw ]; 
  904.                     $orderby = "CAST({$meta_clause['alias']}.meta_value AS {$meta_clause['cast']})"; 
  905.                 } 
  906.                 break; 
  907.         } 
  908.  
  909.         return $orderby; 
  910.     } 
  911.  
  912.     /** 
  913.      * Parse an 'order' query variable and cast it to ASC or DESC as necessary. 
  914.      * 
  915.      * @since 4.6.0 
  916.      * @access protected 
  917.      * 
  918.      * @param string $order The 'order' query variable. 
  919.      * @return string The sanitized 'order' query variable. 
  920.      */ 
  921.     protected function parse_order( $order ) { 
  922.         if ( ! is_string( $order ) || empty( $order ) ) { 
  923.             return 'DESC'; 
  924.         } 
  925.  
  926.         if ( 'ASC' === strtoupper( $order ) ) { 
  927.             return 'ASC'; 
  928.         } else { 
  929.             return 'DESC'; 
  930.         } 
  931.     } 
  932.  
  933.     /** 
  934.      * Used internally to generate a SQL string related to the 'search' parameter. 
  935.      * 
  936.      * @since 4.6.0 
  937.      * @access protected 
  938.      * 
  939.      * @global wpdb $wpdb WordPress database abstraction object. 
  940.      * 
  941.      * @param string $string 
  942.      * @return string 
  943.      */ 
  944.     protected function get_search_sql( $string ) { 
  945.         global $wpdb; 
  946.  
  947.         $like = '%' . $wpdb->esc_like( $string ) . '%'; 
  948.  
  949.         return $wpdb->prepare( '((t.name LIKE %s) OR (t.slug LIKE %s))', $like, $like ); 
  950.     }
