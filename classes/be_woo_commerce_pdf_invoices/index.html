  436.         // Only attach to emails with WC_Order object.

  450.         // Email It In option enabled?

  452.             return $headers; 
  453.         } 
  454.  
  455.         // check if current email type is enabled.

  667.     public function download_invoice_shortcode( $atts ) { 
  668.         if ( ! isset( $atts['order_id'] ) || 0 === intval( $atts['order_id'] ) ) { 
  669.             return; 
  670.         } 
  671.  
  672.         // by default order status should be Processing or Completed. 
  673.         $order = wc_get_order( $atts['order_id'] ); 
  674.         if ( ! $order->is_paid() ) { 
  675.             return; 
  676.         } 
  677.  
  678.         $order_id = BEWPI_WC_Order_Compatibility::get_id( $order ); 
  679.         if ( ! BEWPI_Invoice::exists( $order_id ) ) { 
  680.             return; 
  681.         } 
  682.  
  683.         $url = add_query_arg( array( 
  684.             'bewpi_action' => 'view',  
  685.             'post' => $order_id,  
  686.             'nonce' => wp_create_nonce( 'view' ),  
  687.  ) ); 
  688.  
  689.         $invoice = new BEWPI_Invoice( $order_id ); 
  690.         $tags = array( 
  691.             '{formatted_invoice_number}' => $invoice->get_formatted_number(),  
  692.             '{order_number}' => $order_id,  
  693.             '{formatted_invoice_date}' => $invoice->get_formatted_invoice_date(),  
  694.             '{formatted_order_date}' => $invoice->get_formatted_order_date(),  
  695.  ); 
  696.         // find and replace placeholders. 
  697.         $title = str_replace( array_keys( $tags ), array_values( $tags ), $atts['title'] ); 
  698.         printf( '<a href="%1$s">%2$s</a>', esc_attr( $url ), esc_html( $title ) );
