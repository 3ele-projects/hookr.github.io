  127.              * so that we don't have to serialize the whole object into the queue.

  132.              * @param int The ID of the user who triggered the action 
  133.              */

  135.             wp_suspend_cache_addition( false );

  141.             $upload_size += strlen( $encoded_item ); 
  142.             if ( $upload_size > $this->upload_max_bytes && count( $items_to_send ) > 0 ) { 
  143.                 break; 
  144.             } 
  145.             $items_to_send[ $key ] = $encoded_item; 
  146.             if ( microtime(true) - $start_time > $this->max_dequeue_time ) { 
  147.                 break; 
  148.             } 
  149.         }

  154.     public function do_sync_for_queue( $queue ) { 
  155.

  174.         } 
  175.  
  176.         if ( is_wp_error( $buffer ) ) { 
  177.             return $buffer; 
  178.         } 
  179.  
  180.         $checkout_duration = microtime( true ) - $checkout_start_time; 
  181.  
  182.         list( $items_to_send, $skipped_items_ids, $items, $preprocess_duration ) = $this->get_items_to_send( $buffer, true ); 
  183.  
  184.         /** 
  185.          * Fires when data is ready to send to the server. 
  186.          * Return false or WP_Error to abort the sync (e.g. if there's an error) 
  187.          * The items will be automatically re-sent later 
  188.          * 
  189.          * @since 4.2.0 
  190.          * 
  191.          * @param array $data The action buffer 
  192.          * @param string $codec The codec name used to encode the data 
  193.          * @param double $time The current time 
  194.          * @param string $queue The queue used to send ('sync' or 'full_sync') 
  195.          */ 
  196.         Jetpack_Sync_Settings::set_is_sending( true ); 
  197.         $processed_item_ids = apply_filters( 'jetpack_sync_send_data', $items_to_send, $this->codec->name(), microtime( true ), $queue->id, $checkout_duration, $preprocess_duration ); 
  198.         Jetpack_Sync_Settings::set_is_sending( false ); 
  199.          
  200.         if ( ! $processed_item_ids || is_wp_error( $processed_item_ids ) ) { 
  201.             $checked_in_item_ids = $queue->checkin( $buffer ); 
  202.             if ( is_wp_error( $checked_in_item_ids ) ) { 
  203.                 error_log( 'Error checking in buffer: ' . $checked_in_item_ids->get_error_message() ); 
  204.                 $queue->force_checkin(); 
  205.             } 
  206.             if ( is_wp_error( $processed_item_ids ) ) { 
  207.                 return $processed_item_ids; 
  208.             } 
  209.             // returning a WP_Error is a sign to the caller that we should wait a while 
  210.             // before syncing again 
  211.             return new WP_Error( 'server_error' ); 
  212.         } else { 
  213.             // detect if the last item ID was an error 
  214.             $had_wp_error = is_wp_error( end( $processed_item_ids ) ); 
  215.             if ( $had_wp_error ) { 
  216.                 $wp_error = array_pop( $processed_item_ids ); 
  217.             } 
  218.             // also checkin any items that were skipped 
  219.             if ( count( $skipped_items_ids ) > 0 ) { 
  220.                 $processed_item_ids = array_merge( $processed_item_ids, $skipped_items_ids ); 
  221.             } 
  222.             $processed_items = array_intersect_key( $items, array_flip( $processed_item_ids ) ); 
  223.             /** 
  224.              * Allows us to keep track of all the actions that have been sent. 
  225.              * Allows us to calculate the progress of specific actions. 
  226.              * 
  227.              * @since 4.2.0 
  228.              * 
  229.              * @param array $processed_actions The actions that we send successfully. 
  230.              */

  234.             // before syncing again
