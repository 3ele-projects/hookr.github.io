  696.     private function apply_defaults( $controls ) { 
  697.  
  698.         $defaults = apply_filters( 'layers_customizer_control_defaults' , array() ); 
  699.  
  700.         if( empty( $defaults ) ) return $controls; 
  701.  
  702.         foreach( $controls as $section_key => $control ) { 
  703.  
  704.             foreach( $control as $control_key => $control_data ) { 
  705.                 if( isset( $defaults[ $control_key ] ) ) { 
  706.                     $controls[ $section_key ][ $control_key ][ 'default' ] = $defaults[ $control_key ]; 
  707.                 } 
  708.             } 
  709.         } 
  710.  
  711.         return $controls; 
  712.     }
