  592.     /** 
  593.      * Reformat a number to the way the value of the field is displayed 
  594.      * 
  595.      * @param mixed $value 
  596.      * @param string $name 
  597.      * @param array $options 
  598.      * @param array $pod 
  599.      * @param int $id 
  600.      * 
  601.      * @return string 
  602.      * @since 2.0 
  603.      */ 
  604.     public function format( $value = null, $name = null, $options = null, $pod = null, $id = null ) { 
  605.  
  606.         global $wp_locale; 
  607.  
  608.         if ( null === $value ) { 
  609.             // Don't enforce a default value here 
  610.             return null; 
  611.         } 
  612.  
  613.         if ( '9.999, 99' == pods_v( self::$type . '_format', $options ) ) { 
  614.             $thousands = '.'; 
  615.             $dot = ', '; 
  616.         } 
  617.         elseif ( '9, 999.99' == pods_v( self::$type . '_format', $options ) ) { 
  618.             $thousands = ', '; 
  619.             $dot = '.'; 
  620.         } 
  621.         elseif ( '9\'999.99' == pods_v( self::$type . '_format', $options ) ) { 
  622.             $thousands = '\''; 
  623.             $dot = '.'; 
  624.         } 
  625.         elseif ( '9 999, 99' == pods_v( self::$type . '_format', $options ) ) { 
  626.             $thousands = ' '; 
  627.             $dot = ', '; 
  628.         } 
  629.         elseif ( '9999.99' == pods_v( self::$type . '_format', $options ) ) { 
  630.             $thousands = ''; 
  631.             $dot = '.'; 
  632.         } 
  633.         elseif ( '9999, 99' == pods_v( self::$type . '_format', $options ) ) { 
  634.             $thousands = ''; 
  635.             $dot = ', '; 
  636.         } 
  637.         else { 
  638.             $thousands = $wp_locale->number_format[ 'thousands_sep' ]; 
  639.             $dot = $wp_locale->number_format[ 'decimal_point' ]; 
  640.         } 
  641.  
  642.         $length = (int) pods_v( self::$type . '_max_length', $options, 12, true ); 
  643.  
  644.         if ( $length < 1 || 64 < $length ) { 
  645.             $length = 64; 
  646.         } 
  647.  
  648.         $decimals = (int) pods_v( self::$type . '_decimals', $options, 2 ); 
  649.  
  650.         if ( $decimals < 1 ) { 
  651.             $decimals = 0; 
  652.         } 
  653.         elseif ( 30 < $decimals ) { 
  654.             $decimals = 30; 
  655.         } 
  656.  
  657.         if ( $length < $decimals ) { 
  658.             $decimals = $length; 
  659.         } 
  660.  
  661.         if ( 'i18n' == pods_v( self::$type . '_format', $options ) ) { 
  662.             $value = number_format_i18n( (float) $value, $decimals ); 
  663.         } 
  664.         else { 
  665.             $value = number_format( (float) $value, $decimals, $dot, $thousands ); 
  666.         } 
  667.  
  668.         // Additional output handling for decimals 
  669.         $decimal_handling = pods_v( self::$type . '_decimal_handling', $options, 'none' ) ; 
  670.         if ( 'none' !== $decimal_handling ) { 
  671.             $value_parts = explode( $dot, $value ); 
  672.             if ( 'remove' === $decimal_handling ) { 
  673.                 array_pop( $value_parts ); 
  674.             } elseif ( 'dash' === $decimal_handling ) { 
  675.                 array_pop( $value_parts ); 
  676.                 $value_parts[] = '-'; 
  677.             } 
  678.             $value = implode( $dot, $value_parts ); 
  679.         } 
  680.  
  681.         return $value; 
  682.  
  683.     }
