  629.     if ( $base_nodes->length && ($href = $base_nodes->item(0)->getAttribute("href")) ) {

  631.     } 
  632.  
  633.     // Set the base path of the Stylesheet to that of the file being processed 
  634.     $this->_css->set_protocol($this->_protocol); 
  635.     $this->_css->set_host($this->_base_host); 
  636.     $this->_css->set_base_path($this->_base_path); 
  637.  
  638.     // Get all the stylesheets so that they are processed in document order 
  639.     $xpath = new DOMXPath($this->_xml); 
  640.     $stylesheets = $xpath->query("//*[name() = 'link' or name() = 'style']"); 
  641.  
  642.     foreach($stylesheets as $tag) { 
  643.       switch (strtolower($tag->nodeName)) { 
  644.         // load <link rel="STYLESHEET" ... /> tags 
  645.         case "link": 
  646.           if ( mb_strtolower(stripos($tag->getAttribute("rel"), "stylesheet") !== false) || // may be "appendix stylesheet" 
  647.             mb_strtolower($tag->getAttribute("type")) === "text/css" ) { 
  648.             //Check if the css file is for an accepted media type 
  649.             //media not given then always valid 
  650.             $formedialist = preg_split("/[\s\n, ]/", $tag->getAttribute("media"), -1, PREG_SPLIT_NO_EMPTY); 
  651.             if ( count($formedialist) > 0 ) { 
  652.               $accept = false; 
  653.               foreach ( $formedialist as $type ) { 
  654.                 if ( in_array(mb_strtolower(trim($type)), $acceptedmedia) ) { 
  655.                   $accept = true; 
  656.                   break; 
  657.                 } 
  658.               } 
  659.  
  660.               if (!$accept) { 
  661.                 //found at least one mediatype, but none of the accepted ones 
  662.                 //Skip this css file. 
  663.                 continue; 
  664.               } 
  665.             } 
  666.  
  667.             $url = $tag->getAttribute("href"); 
  668.             $url = build_url($this->_protocol, $this->_base_host, $this->_base_path, $url); 
  669.  
  670.             $this->_css->load_css_file($url, Stylesheet::ORIG_AUTHOR); 
  671.           } 
  672.           break; 
  673.  
  674.         // load <style> tags 
  675.         case "style": 
  676.           // Accept all <style> tags by default (note this is contrary to W3C 
  677.           // HTML 4.0 spec: 
  678.           // http://www.w3.org/TR/REC-html40/present/styles.html#adef-media 
  679.           // which states that the default media type is 'screen' 
  680.           if ( $tag->hasAttributes() && 
  681.             ($media = $tag->getAttribute("media")) && 
  682.             !in_array($media, $acceptedmedia) ) { 
  683.             continue; 
  684.           } 
  685.  
  686.           $css = ""; 
  687.           if ( $tag->hasChildNodes() ) { 
  688.             $child = $tag->firstChild; 
  689.             while ( $child ) { 
  690.               $css .= $child->nodeValue; // Handle <style><!-- blah --></style> 
  691.               $child = $child->nextSibling; 
  692.             } 
  693.           } 
  694.           else { 
  695.             $css = $tag->nodeValue; 
  696.           } 
  697.  
  698.           $this->_css->load_css($css); 
  699.           break; 
  700.       } 
  701.     } 
  702.   }

  772.   /** 
  773.    * Renders the HTML to PDF 
  774.    */ 
  775.   function render() { 
  776.     $this->save_locale(); 
  777.  
  778.     $log_output_file = $this->get_option("log_output_file"); 
  779.     if ( $log_output_file ) { 
  780.       if ( !file_exists($log_output_file) && is_writable(dirname($log_output_file)) ) { 
  781.         touch($log_output_file); 
  782.       } 
  783.  
  784.       $this->_start_time = microtime(true); 
  785.       ob_start(); 
  786.     } 
  787.  
  788.     //enable_mem_profile(); 
  789.  
  790.     $this->_process_html(); 
  791.  
  792.     $this->_css->apply_styles($this->_tree); 
  793.  
  794.     // @page style rules : size, margins 
  795.     $page_styles = $this->_css->get_page_styles(); 
  796.  
  797.     $base_page_style = $page_styles["base"]; 
  798.     unset($page_styles["base"]); 
  799.  
  800.     foreach($page_styles as $_page_style) { 
  801.       $_page_style->inherit($base_page_style); 
  802.     } 
  803.  
  804.     if ( is_array($base_page_style->size) ) { 
  805.       $this->set_paper(array(0, 0, $base_page_style->size[0], $base_page_style->size[1])); 
  806.     } 
  807.  
  808.     $this->_pdf = Canvas_Factory::get_instance($this, $this->_paper_size, $this->_paper_orientation); 
  809.     Font_Metrics::init($this->_pdf); 
  810.  
  811.     if ( $this->get_option("enable_font_subsetting") && $this->_pdf instanceof CPDF_Adapter ) { 
  812.       foreach ($this->_tree->get_frames() as $frame) { 
  813.         $style = $frame->get_style(); 
  814.         $node = $frame->get_node(); 
  815.  
  816.         // Handle text nodes 
  817.         if ( $node->nodeName === "#text" ) { 
  818.           $this->_pdf->register_string_subset($style->font_family, $node->nodeValue); 
  819.           continue; 
  820.         } 
  821.  
  822.         // Handle generated content (list items) 
  823.         if ( $style->display === "list-item" ) { 
  824.           $chars = List_Bullet_Renderer::get_counter_chars($style->list_style_type); 
  825.           $this->_pdf->register_string_subset($style->font_family, $chars); 
  826.           continue; 
  827.         } 
  828.          
  829.         // Handle other generated content (pseudo elements) 
  830.         // FIXME: This only captures the text of the stylesheet declaration,  
  831.         //        not the actual generated content, and forces all possible counter 
  832.         //        values. See notes in issue #750. 
  833.         if ( $frame->get_node()->nodeName == "dompdf_generated" ) { 
  834.           // all possible counter values 
  835.           $chars = List_Bullet_Renderer::get_counter_chars('decimal'); 
  836.           $this->_pdf->register_string_subset($style->font_family, $chars); 
  837.           $chars = List_Bullet_Renderer::get_counter_chars('upper-alpha'); 
  838.           $this->_pdf->register_string_subset($style->font_family, $chars); 
  839.           $chars = List_Bullet_Renderer::get_counter_chars('lower-alpha'); 
  840.           $this->_pdf->register_string_subset($style->font_family, $chars); 
  841.           $chars = List_Bullet_Renderer::get_counter_chars('lower-greek'); 
  842.           $this->_pdf->register_string_subset($style->font_family, $chars); 
  843.           // the text of the stylesheet declaration 
  844.           $this->_pdf->register_string_subset($style->font_family, $style->content); 
  845.           continue; 
  846.         } 
  847.       } 
  848.     } 
  849.  
  850.     $root = null; 
  851.  
  852.     foreach ($this->_tree->get_frames() as $frame) { 
  853.       // Set up the root frame 
  854.       if ( is_null($root) ) { 
  855.         $root = Frame_Factory::decorate_root( $this->_tree->get_root(), $this ); 
  856.         continue; 
  857.       } 
  858.  
  859.       // Create the appropriate decorators, reflowers & positioners. 
  860.       Frame_Factory::decorate_frame($frame, $this, $root); 
  861.     } 
  862.  
  863.     // Add meta information 
  864.     $title = $this->_xml->getElementsByTagName("title"); 
  865.     if ( $title->length ) { 
  866.       $this->_pdf->add_info("Title", trim($title->item(0)->nodeValue)); 
  867.     } 
  868.  
  869.     $metas = $this->_xml->getElementsByTagName("meta"); 
  870.     $labels = array( 
  871.       "author" => "Author",  
  872.       "keywords" => "Keywords",  
  873.       "description" => "Subject",  
  874.  ); 
  875.     foreach($metas as $meta) { 
  876.       $name = mb_strtolower($meta->getAttribute("name")); 
  877.       $value = trim($meta->getAttribute("content")); 
  878.  
  879.       if ( isset($labels[$name]) ) { 
  880.         $this->_pdf->add_info($labels[$name], $value); 
  881.         continue; 
  882.       } 
  883.  
  884.       if ( $name === "dompdf.view" && $this->parse_default_view($value) ) { 
  885.         $this->_pdf->set_default_view($this->_default_view, $this->_default_view_options); 
  886.       } 
  887.     } 
  888.  
  889.     $root->set_containing_block(0, 0, $this->_pdf->get_width(), $this->_pdf->get_height()); 
  890.     $root->set_renderer(new Renderer($this)); 
  891.  
  892.     // This is where the magic happens: 
  893.     $root->reflow(); 
  894.  
  895.     // Clean up cached images 
  896.     Image_Cache::clear(); 
  897.  
  898.     global $_dompdf_warnings, $_dompdf_show_warnings; 
  899.     if ( $_dompdf_show_warnings ) { 
  900.       echo '<b>DOMPDF Warnings</b><br><pre>'; 
  901.       foreach ($_dompdf_warnings as $msg) { 
  902.         echo $msg . "\n"; 
  903.       } 
  904.       echo $this->get_canvas()->get_cpdf()->messages; 
  905.       echo '</pre>'; 
  906.       flush(); 
  907.     } 
  908.  
  909.     $this->restore_locale(); 
  910.   }

  922.    * Writes the output buffer in the log file

  924.    * @return void 
  925.    */ 
  926.   private function write_log() { 
  927.     $log_output_file = $this->get_option("log_output_file"); 
  928.     if ( !$log_output_file || !is_writable($log_output_file) ) { 
  929.       return; 
  930.     } 
  931.  
  932.     $frames = Frame::$ID_COUNTER; 
  933.     $memory = DOMPDF_memory_usage() / 1024;

  935.  
  936.     $out = sprintf( 
  937.       "<span style='color: #000' title='Frames'>%6d</span>". 
  938.         "<span style='color: #009' title='Memory'>%10.2f KB</span>". 
  939.         "<span style='color: #900' title='Time'>%10.2f ms</span>". 
  940.         "<span  title='Quirksmode'>  ". 
  941.         ($this->_quirksmode ? "<span style='color: #d00'> ON</span>" : "<span style='color: #0d0'>OFF</span>"). 
  942.         "</span><br />", $frames, $memory, $time); 
  943.  
  944.     $out .= ob_get_clean(); 
  945.  
  946.     $log_output_file = $this->get_option("log_output_file"); 
  947.     file_put_contents($log_output_file, $out); 
  948.   }
