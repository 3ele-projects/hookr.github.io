  689. /** 
  690. * Get parent posts by path. Currently, this method only supports getting parents of the 'page' 
  691. * post type. The goal of this function is to create a clear path back to home given what would 
  692. * normally be a "ghost" directory. If any page matches the given path, it'll be added. 
  693. * 
  694. * @since 0.6.0 
  695. * @access public 
  696. * @param string $path The path (slug) to search for posts by. 
  697. * @return void 
  698. */ 
  699. function do_path_parents( $path ) { 
  700. /** Trim '/' off $path in case we just got a simple '/' instead of a real path. */ 
  701. $path = trim( $path, '/' ); 
  702. /** If there's no path, return. */ 
  703. if ( empty( $path ) ) 
  704. return; 
  705. /** Get parent post by the path. */ 
  706. $post = get_page_by_path( $path ); 
  707. if ( !empty( $post ) ) { 
  708. $this->do_post_parents( $post->ID ); 
  709. } 
  710. elseif ( is_null( $post ) ) { 
  711. /** Separate post names into separate paths by '/'. */ 
  712. $path = trim( $path, '/' ); 
  713. preg_match_all( "/\/.*?\z/", $path, $matches ); 
  714. /** If matches are found for the path. */ 
  715. if ( isset( $matches ) ) { 
  716. /** Reverse the array of matches to search for posts in the proper order. */ 
  717. $matches = array_reverse( $matches ); 
  718. /** Loop through each of the path matches. */ 
  719. foreach ( $matches as $match ) { 
  720. /** If a match is found. */ 
  721. if ( isset( $match[0] ) ) { 
  722. /** Get the parent post by the given path. */ 
  723. $path = str_replace( $match[0], '', $path ); 
  724. $post = get_page_by_path( trim( $path, '/' ) ); 
  725. /** If a parent post is found, set the $post_id and break out of the loop. */ 
  726. if ( !empty( $post ) && 0 < $post->ID ) { 
  727. $this->do_post_parents( $post->ID ); 
  728. break; 
  729. } 
  730. } 
  731. } 
  732. } 
  733. } 
  734. } 
  735. /** 
  736. * Searches for term parents of hierarchical taxonomies. This function is similar to the WordPress 
  737. * function get_category_parents() but handles any type of taxonomy. 
  738. * 
  739. * @since 0.6.0 
  740. * @param int $term_id ID of the term to get the parents of. 
  741. * @param string $taxonomy Name of the taxonomy for the given term. 
  742. * @return void 
  743. */ 
  744. function do_term_parents( $term_id, $taxonomy ) { 
  745. /** Set up some default arrays. */ 
  746. $parents = array(); 
  747. /** While there is a parent ID, add the parent term link to the $parents array. */ 
  748. while ( $term_id ) { 
  749. /** Get the parent term. */ 
  750. $term = get_term( $term_id, $taxonomy ); 
  751. /** Add the formatted term link to the array of parent terms. */ 
  752. $parents[] = '<a href="' . get_term_link( $term, $taxonomy ) . '" title="' . esc_attr( $term->name ) . '">' . $term->name . '</a>'; 
  753. /** Set the parent term's parent as the parent ID. */ 
  754. $term_id = $term->parent; 
  755. } 
  756. /** If we have parent terms, reverse the array to put them in the proper order for the trail. */ 
  757. if ( !empty( $parents ) ) 
  758. $this->items = array_merge( $this->items, $parents ); 
  759. } 
  760. /** 
  761. * Turns %tag% from permalink structures into usable links for the breadcrumb trail. This feels kind of 
  762. * hackish for now because we're checking for specific %tag% examples and only doing it for the 'post' 
  763. * post type. In the future, maybe it'll handle a wider variety of possibilities, especially for custom post 
  764. * types. 
  765. * 
  766. * @since 0.6.0 
  767. * @access public 
  768. * @param int $post_id ID of the post whose parents we want. 
  769. * @param string $path Path of a potential parent page. 
  770. * @param array $args Mixed arguments for the menu. 
  771. * @return array 
  772. */ 
  773. public function map_rewrite_tags( $post_id, $path ) { 
  774. /** Get the post based on the post ID. */ 
  775. $post = get_post( $post_id ); 
  776. /** If no post is returned, an error is returned, or the post does not have a 'post' post type, return. */ 
  777. if ( empty( $post ) || is_wp_error( $post ) || 'post' !== $post->post_type ) 
  778. return $trail; 
  779. /** Trim '/' from both sides of the $path. */ 
  780. $path = trim( $path, '/' ); 
  781. /** Split the $path into an array of strings. */ 
  782. $matches = explode( '/', $path ); 
  783. /** If matches are found for the path. */ 
  784. if ( is_array( $matches ) ) { 
  785. /** Loop through each of the matches, adding each to the $trail array. */ 
  786. foreach ( $matches as $match ) { 
  787. /** Trim any '/' from the $match. */ 
  788. $tag = trim( $match, '/' ); 
  789. /** If using the %year% tag, add a link to the yearly archive. */ 
  790. if ( '%year%' == $tag ) 
  791. $this->items[] = '<a href="' . get_year_link( get_the_time( 'Y', $post_id ) ) . '">' . sprintf( $this->args['labels']['archive_year'], get_the_time( _x( 'Y', 'yearly archives date format', 'onetone' ) ) ) . '</a>'; 
  792. /** If using the %monthnum% tag, add a link to the monthly archive. */ 
  793. elseif ( '%monthnum%' == $tag ) 
  794. $this->items[] = '<a href="' . get_month_link( get_the_time( 'Y', $post_id ), get_the_time( 'm', $post_id ) ) . '">' . sprintf( $this->args['labels']['archive_month'], get_the_time( _x( 'F', 'monthly archives date format', 'onetone' ) ) ) . '</a>'; 
  795. /** If using the %day% tag, add a link to the daily archive. */ 
  796. elseif ( '%day%' == $tag ) 
  797. $this->items[] = '<a href="' . get_day_link( get_the_time( 'Y', $post_id ), get_the_time( 'm', $post_id ), get_the_time( 'd', $post_id ) ) . '">' . sprintf( $this->args['labels']['archive_day'], get_the_time( _x( 'j', 'daily archives date format', 'onetone' ) ) ) . '</a>'; 
  798. /** If using the %author% tag, add a link to the post author archive. */ 
  799. elseif ( '%author%' == $tag ) 
  800. $this->items[] = '<a href="' . get_author_posts_url( $post->post_author ) . '" title="' . esc_attr( get_the_author_meta( 'display_name', $post->post_author ) ) . '">' . get_the_author_meta( 'display_name', $post->post_author ) . '</a>'; 
  801. /** If using the %category% tag, add a link to the first category archive to match permalinks. */ 
  802. elseif ( '%category%' == $tag ) { 
  803. /** Force override terms in this post type. */ 
  804. $this->args['post_taxonomy'][ $post->post_type ] = false; 
  805. /** Get the post categories. */ 
  806. $terms = get_the_category( $post_id ); 
  807. /** Check that categories were returned. */ 
  808. if ( $terms ) { 
  809. /** Sort the terms by ID and get the first category. */ 
  810. usort( $terms, '_usort_terms_by_ID' ); 
  811. $term = get_term( $terms[0], 'category' ); 
  812. /** If the category has a parent, add the hierarchy to the trail. */ 
  813. if ( 0 < $term->parent ) 
  814. $this->do_term_parents( $term->parent, 'category' ); 
  815. /** Add the category archive link to the trail. */ 
  816. $this->items[] = '<a href="' . get_term_link( $term, 'category' ) . '" title="' . esc_attr( $term->name ) . '">' . $term->name . '</a>'; 
  817. } 
  818. } 
  819. } 
  820. } 
  821. }
