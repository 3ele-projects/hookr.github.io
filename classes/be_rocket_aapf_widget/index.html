  791.     // 99% copy of WordPress' get_pagenum_link. 
  792.     public static function get_pagenum_link($pagenum = 1, $escape = true ) { 
  793.         global $wp_rewrite; 
  794.  
  795.         $pagenum = (int) $pagenum; 
  796.  
  797.         $request = remove_query_arg( 'paged', preg_replace( "~".home_url()."~", "", @$_POST['location'] ) ); 
  798.  
  799.         $home_root = parse_url(home_url()); 
  800.         $home_root = ( isset($home_root['path']) ) ? $home_root['path'] : ''; 
  801.         $home_root = preg_quote( $home_root, '|' ); 
  802.  
  803.         $request = preg_replace('|^'. $home_root . '|i', '', $request); 
  804.         unset( $home_root ); 
  805.         $request = preg_replace('|^/+|', '', $request); 
  806.  
  807.         if ( !$wp_rewrite->using_permalinks() ) { 
  808.             $base = trailingslashit( get_bloginfo( 'url' ) ); 
  809.  
  810.             if ( $pagenum > 1 ) { 
  811.                 $result = add_query_arg( 'paged', $pagenum, $base . $request ); 
  812.             } else { 
  813.                 $result = $base . $request; 
  814.             } 
  815.         } else { 
  816.             $qs_regex = '|\?.*?$|'; 
  817.             preg_match( $qs_regex, $request, $qs_match ); 
  818.  
  819.             if ( !empty( $qs_match[0] ) ) { 
  820.                 $query_string = $qs_match[0]; 
  821.                 $request = preg_replace( $qs_regex, '', $request ); 
  822.             } else { 
  823.                 $query_string = ''; 
  824.             } 
  825.  
  826.             $request = preg_replace( "|$wp_rewrite->pagination_base/\d+/?$|", '', $request); 
  827.             $request = preg_replace( '|^' . preg_quote( $wp_rewrite->index, '|' ) . '|i', '', $request); 
  828.             $request = ltrim($request, '/'); 
  829.  
  830.             $base = trailingslashit( get_bloginfo( 'url' ) ); 
  831.  
  832.             if ( $wp_rewrite->using_index_permalinks() && ( $pagenum > 1 || '' != $request ) ) 
  833.                 $base .= $wp_rewrite->index . '/'; 
  834.  
  835.             if ( $pagenum > 1 ) { 
  836.                 $request = ( ( !empty( $request ) ) ? trailingslashit( $request ) : $request ) . user_trailingslashit( $wp_rewrite->pagination_base . "/" . $pagenum, 'paged' ); 
  837.             } 
  838.  
  839.             $result = $base . $request . $query_string; 
  840.         } 
  841.  
  842.         /** 
  843.          * Filter the page number link for the current request. 
  844.          * 
  845.          * @since 2.5.0 
  846.          * 
  847.          * @param string $result The page number link. 
  848.          */ 
  849.         $result = apply_filters( 'get_pagenum_link', $result ); 
  850.  
  851.         if ( $escape ) 
  852.             return esc_url( $result ); 
  853.         else 
  854.             return esc_url_raw( $result ); 
  855.     }
