Jetpack_Sync_Actions

The role of this class is to hook the Sync subsystem into WordPress - when to listen for actions, when to send, when to perform a full sync, etc.

  230.     /** 
  231.      * Try to send actions until we run out of things to send,  
  232.      * or have to wait more than 15s before sending again,  
  233.      * or we hit a lock or some other sending issue 
  234.      * 
  235.      * @param string $type Sync type. Can be `sync` or `full_sync`. 
  236.      */ 
  237.     static function do_cron_sync_by_type( $type ) { 
  238.         if ( ! self::sync_allowed() || ( 'sync' !== $type && 'full_sync' !== $type ) ) { 
  239.             return; 
  240.         } 
  241.  
  242.         self::initialize_sender(); 
  243.  
  244.         $time_limit = Jetpack_Sync_Settings::get_setting( 'cron_sync_time_limit' ); 
  245.         $start_time = time(); 
  246.  
  247.         do { 
  248.             $next_sync_time = self::$sender->get_next_sync_time( $type ); 
  249.  
  250.             if ( $next_sync_time ) { 
  251.                 $delay = $next_sync_time - time() + 1; 
  252.                 if ( $delay > 15 ) { 
  253.                     break; 
  254.                 } elseif ( $delay > 0 ) { 
  255.                     sleep( $delay ); 
  256.                 } 
  257.             } 
  258.  
  259.             $result = 'full_sync' === $type ? self::$sender->do_full_sync() : self::$sender->do_sync(); 
  260.         } while ( $result && ( $start_time + $time_limit ) > time() ); 
  261.     }

  337.     static function maybe_schedule_sync_cron( $schedule, $hook ) { 
  338.         if ( ! $hook ) { 
  339.             return; 
  340.         } 
  341.         $schedule = self::sanitize_filtered_sync_cron_schedule( $schedule ); 
  342.  
  343.         $start_time = time() + self::get_start_time_offset( $schedule, $hook ); 
  344.         if ( ! wp_next_scheduled( $hook ) ) { 
  345.             // Schedule a job to send pending queue items once a minute 
  346.             wp_schedule_event( $start_time, $schedule, $hook );
