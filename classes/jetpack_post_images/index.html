  537.     /** 
  538.      * Takes an image URL and pixel dimensions then returns a URL for the 
  539.      * resized and cropped image. 
  540.      * 
  541.      * @param  string $src 
  542.      * @param  int    $dimension 
  543.      * @return string            Transformed image URL 
  544.      */ 
  545.     static function fit_image_url( $src, $width, $height ) { 
  546.         $width = (int) $width; 
  547.         $height = (int) $height; 
  548.  
  549.         if ( $width < 1 || $height < 1 ) { 
  550.             return $src; 
  551.         } 
  552.  
  553.         // See if we should bypass WordPress.com SaaS resizing 
  554.         if ( has_filter( 'jetpack_images_fit_image_url_override' ) ) { 
  555.             /** 
  556.              * Filters the image URL used after dimensions are set by Photon. 
  557.              * 
  558.              * @since 3.3.0

  561.              * @param int $width Image width. 
  562.              * @param int $width Image height. 
  563.              */ 
  564.             return apply_filters( 'jetpack_images_fit_image_url_override', $src, $width, $height ); 
  565.         } 
  566.  
  567.         // If WPCOM hosted image use native transformations 
  568.         $img_host = parse_url( $src, PHP_URL_HOST ); 
  569.         if ( '.files.wordpress.com' == substr( $img_host, -20 ) ) { 
  570.             return add_query_arg( array( 'w' => $width, 'h' => $height, 'crop' => 1 ), set_url_scheme( $src ) ); 
  571.         } 
  572.  
  573.         // Use Photon magic 
  574.         if( function_exists( 'jetpack_photon_url' ) ) { 
  575.             return jetpack_photon_url( $src, array( 'resize' => "$width, $height" ) ); 
  576.         } 
  577.          
  578.         // Arg... no way to resize image using WordPress.com infrastructure! 
  579.         return $src;
