  300.     public function gformConfirmation($confirmation, $form, $entry, $ajax) { 
  301.  
  302.         // Return if not the current form. 
  303.         if (RGForms::post('gform_submit') != $form['id']) { 
  304.             return $confirmation; 
  305.         } 
  306.  
  307.         // Get feed mapping form fields to payment request, return if not set as need the feed to actually do anything. 
  308.         $feed = $this->getFeed($form['id']); 
  309.         if (!$feed) { 
  310.             return $confirmation; 
  311.         } 
  312.  
  313.         // -------------------------------------- 
  314.         // Record payment gateway. 
  315.         gform_update_meta($entry['id'], 'payment_gateway', 'gfpaystation'); 
  316.  
  317.         // -------------------------------------- 
  318.         // Get the data posted from the form and build the payment request by creating a Paystation payment request object,  
  319.         // setting it's properties, then calling the processPayment() method. 
  320.         $formData = $this->getFormData($form); 
  321.         $paymentReq = null; 
  322.          
  323.         // If a paystation override id has been specified on the form from dropdown to override etc then create new payment object with that Paystation id 
  324.         // Otherwise default to the normal paystation_id in the main settings. This feature allows money to be paid in to different paystation accounts 
  325.         // based on user selection of something on the form such as branch, region, country etc. 
  326.         if ((isset($formData->PaystationOverrideId)) && ($formData->PaystationOverrideId)) 
  327.         { 
  328.             $paymentReq = new GFPaystationPayment($formData->PaystationOverrideId, $this->options['gatewayId'], $this->options['testMode'], $this->securityHash); 
  329.         } 
  330.         else 
  331.         { 
  332.             $paymentReq = new GFPaystationPayment($this->options['paystationId'], $this->options['gatewayId'], $this->options['testMode'], $this->securityHash); 
  333.         } 
  334.          
  335.         $paymentReq->amount = (int) ($formData->total * 100);       // We multiply by 100 because the amount sent to the gateawy must be an int, so cents not a float. 
  336.         $paymentReq->currency = GFCommon::get_currency();             // Get the currency from gravity forms. 
  337.         $paymentReq->merchantSession = $this->buildSessionId($entry['id']);  // Call function in this class to create the merchantSession id. 
  338.         $paymentReq->merchantReference = $formData->MerchantReference;         // Set other things. 
  339.         $paymentReq->customerDetails = $formData->CustomerDetails; 
  340.         $paymentReq->orderDetails = $formData->OrderDetails; 
  341.          
  342.         // -------------------------------------- 
  343.         // Try making the payment, if successful the redirect the user to the payment URL, if not then catch error. 
  344.         try { 
  345.             $response = $paymentReq->processPayment();  // Call processPayment function in PaystationPayment class, it will do the POST to the paystation gateway. 
  346.  
  347.             // If digitalOrder is populated then there were no issues with the transaction request, so update the status of the 
  348.             // the lead record and then set the confirmation to the url of the payment screen. 
  349.             if ($response->digitalOrder) {    
  350.                 GFFormsModel::update_lead_property($entry['id'], 'payment_status', 'Processing'); 
  351.                 GFFormsModel::update_lead_property($entry['id'], 'transaction_id', $response->transactionId); 
  352.  
  353.                 // NB: GF handles redirect via JavaScript if headers already sent, or AJAX. 
  354.                 $confirmation = array('redirect' => $response->digitalOrder); 
  355.             } 
  356.             else { 
  357.                 // There was an error with the payment initiation, an error message will be set so 
  358.                 // throw it so that the exception code below will update the status to failed 
  359.                 // and cause the user to see the error emssage. 
  360.                 throw new GFPaystationException($response->errorMessage); 
  361.             } 
  362.         } 
  363.

  373.  
  374.         return $confirmation;

  415.     public function processPaystationReturn() {

  418.         $parts = parse_url($_SERVER['REQUEST_URI']); 
  419.         $query = isset($parts['query']) ? $parts['query'] : ''; 
  420.          
  421.         // If have query parts.

  424.             // If the query part contains the paystation return string, then we are probably getting

  426.             if (strpos($query, GFPAYSTATION_RETURN) !== false) { 
  427.              
  428.                 try { 
  429.                     // Create object to deal with the return result passing the query string in to it for parsing. 
  430.                     $resultReq = new GFPaystationReturnResult(stripslashes($query));

  433.                     if ($resultReq->isValid) { 
  434.

  436.                         $leadId = $this->getLeadIdFromMerchantSession($resultReq->merchantSession);

  438.                         $form = GFFormsModel::get_form_meta($lead['form_id']); 
  439.                          
  440.                         // Check to see if the payment was successful. 
  441.                         // The redirect back is not the final say on the matter, the postback is so we don't  
  442.                         // update anything in the database until the postback has been recieved. 
  443.                          
  444.                         // If failed then redirect the user to the failure URL, otherwise let the gravity form normal submit happen  
  445.                         // and take the user to wherever the gravity form is supposed to go on success. 
  446.                         if ($resultReq->errorCode != 0) { 
  447.                              
  448.                             $feed = $this->getFeed($form['id']); 
  449.                              
  450.                             // Redirect to the fail url for the form if there is one then exit so the code below does not 
  451.                             // execute which is the redirect when successful or no fail url specified. 
  452.                             if ($feed->UrlFail) { 
  453.                                 wp_redirect($feed->UrlFail); 
  454.                                 exit; 
  455.                             } 
  456.                         } 
  457.                          
  458.                         // Redirect to Gravity Forms page, passing form and lead IDs. 
  459.                         // Whatever the developer specified to happen after the form was submitted will happen. 
  460.                         $query = "form_id={$lead['form_id']}&lead_id={$lead['id']}"; 
  461.                         $query .= "&hash=" . wp_hash($query); 
  462.                            
  463.                         wp_redirect(add_query_arg(array(GFPAYSTATION_RETURN => base64_encode($query)), $lead['source_url'])); 
  464.                         exit; 
  465.                     } 
  466.                 } 
  467.                  
  468.                 // If there is an exception with this then there is not much more we can do than echo it out. 
  469.                 // We may not have the id of the form so cannot update its status to failed, nor do I know a way 
  470.                 // for us to set something in the url to get wordpress to display the exception to the user in a nice way. 
  471.                 catch (GFPaystationException $e) { 
  472.                     echo nl2br(htmlspecialchars($e->getMessage())); 
  473.                     exit; 
  474.                 } 
  475.             }

  477.                  
  478.                 // In order to help ensure that the postback XML came from Paystation, ensure that the value of the GFPAYSTATION_POSTBACK 
  479.                 // in the querystring is equal to that of the security hash saved in the plugin options. 
  480.                 $pluginOptions = (array) get_option(GFPAYSTATION_PLUGIN_OPTIONS); 
  481.                 parse_str($query, $params); 
  482.                  
  483.                 if ((isset($params[GFPAYSTATION_POSTBACK])) && ($params[GFPAYSTATION_POSTBACK] == $pluginOptions['securityHash'])) 
  484.                 { 
  485.                     try { 
  486.

  488.                         $postback = file_get_contents("php://input"); 
  489.                      
  490.                         // Create object to deal with the postback passing the postback as it will parse and put xml in to properties of the class. 
  491.                         $resultReq = new GFPaystationPostbackResult($postback); 
  492.                           
  493.                         // If got valid paystation postback response. 
  494.                         if ($resultReq->isValid) { 
  495.                              
  496.                             // Load the lead and the form as we need some of the details of these. 
  497.                             $leadId = $this->getLeadIdFromMerchantSession($resultReq->merchantSession); 
  498.                             $lead = GFFormsModel::get_lead($leadId); 
  499.                             $form = GFFormsModel::get_form_meta($lead['form_id']); 
  500.                             $feed = $this->getFeed($form['id']); 
  501.                              
  502.                             // Check to see if the payment was successful 
  503.                             // Because this time it is from the postback we know 100% for sure that the payment was successul or not 
  504.                             // so we must update the status of the lead to sucess or failure and also send any responses. 
  505.                             if ($resultReq->errorCode === 0) { 
  506.                                  
  507.                                 // Success (i.e. error code is 0). Update some details on the lead. 
  508.                                 $lead['payment_status'] = 'Approved'; 
  509.

  512.                                 $nz_timezone = new DateTimeZone('Pacific/Auckland'); 
  513.                                 $nz_datetime = new DateTime($resultReq->transactionTime, $nz_timezone); 
  514.                                 $utc_datetime = gmdate('Y-m-d H:i:s', strtotime($nz_datetime->format('r'))); 
  515.                                  
  516.                                 $lead['payment_date'] = $utc_datetime; 
  517.

  520.                                 $lead['transaction_id'] = $resultReq->transactionId; 
  521.                                 $lead['transaction_type'] = 1; 
  522.                                 GFFormsModel::update_lead($lead); 
  523.                                  
  524.                                 // Trigger hook if one is present. 
  525.                                 do_action('gfpaystation_post_payment', $lead, $form);  
  526.  
  527.                                 // Record entry's unique ID in database. 
  528.                                 gform_update_meta($lead['id'], 'gfpaystation_unique_id', GFFormsModel::get_form_unique_id($form['id']));                         
  529.                                  
  530.                                 // Now we need to send the email notifications etc since it is all confirmed. 
  531.                                 if (!$lead['is_fulfilled']) { 
  532.                                      
  533.                                     if ($feed->DelayPost) { 
  534.                                         GFFormsModel::create_post($form, $lead); 
  535.                                     } 
  536.                                      
  537.                                     if ($feed->DelayNotify || $feed->DelayAutorespond) { 
  538.                                         $this->sendDeferredNotifications($feed, $form, $lead); 
  539.                                     } 
  540.                                      
  541.                                     GFFormsModel::update_lead_property($lead['id'], 'is_fulfilled', true); 
  542.                                 } 
  543.                             } 
  544.                             else { 
  545.                                  
  546.                                 // Failure, error code is not 0 so update the status of the lead to Failed. 
  547.                                 $lead['payment_status'] = 'Failed'; 
  548.                                 GFFormsModel::update_lead($lead); 
  549.                                  
  550.                                 // Also set the error message for the lead so that this can be included with an error message 
  551.                                 // to the user if the developer has not specified a failure URL (which means the form will re-load). 
  552.                                 gform_update_meta($lead['id'], 'gfpaystation_error', $resultReq->errorMessage); 
  553.                                  
  554.                                 // Trigger hook if there is one. 
  555.                                 do_action('gfpaystation_post_payment_fail', $lead, $form); 
  556.                             } 
  557.                              
  558.                             // No redirects are done as the result of this is not seen by the user. 
  559.                             // The home page of the site is displayed so that will satisfy the paystation system that the response was received. 
  560.                         } 
  561.                     } 
  562.                  
  563.                     // Because the output from this code is only ever seen by the paystation system, echoing the error message etc 
  564.                     // will be of not use; because a page will be rendered the paystation system will think that the resaponse was 
  565.                     // successfully received, so in this case if there is an error we case an error 500 which means the paystation 
  566.                     // system will try again a few times until successful and if not then paystation will know there was an issue. 
  567.                     catch (GFPaystationException $e) { 
  568.                          
  569.                         header('HTTP/1.1 500 Internal Server Error', true, 500); 
  570.                         exit; 
  571.                     } 
  572.                 } 
  573.                 else { 
  574.                      
  575.                     // Not valid from paystation so also error 500 (just in case was sent from paystation but security hash no longer matches 
  576.                     // it will be recorded in the paystation system that there was an error). 
  577.                     header('HTTP/1.1 500 Internal Server Error', true, 500); 
  578.                     exit; 
  579.                 } 
  580.             } 
  581.         } 
  582.     }

  585.     /** 
  586.     * send deferred notifications, handling pre- and post-1.7.0 worlds 
  587.     * @param array $feed 
  588.     * @param array $form the form submission data 
  589.     * @param array $lead the form entry 
  590.     */ 
  591.     // ==================================================================================================================================== 
  592.     protected function sendDeferredNotifications($feed, $form, $lead) { 
  593.          
  594.         // Get the version of gravity forms. 
  595.         // NOTE: if there is not a valid license key entered for gravity forms the version information will be empty. 
  596.         $gfversion = GFCommon::get_version_info(); 
  597.  
  598.         if (version_compare($gfversion['version'], '1.7.0', '<')) { 
  599.          
  600.             // pre-1.7.0 notifications 
  601.             //+ the code will also come in to this block if Gravity Forms is not proppery licenced 
  602.             //+ which might be an issue during development and testing if a licence has not been purchased. 
  603.              
  604.             if ($feed->DelayNotify) { 
  605.                 GFCommon::send_admin_notification($form, $lead); 
  606.             } 
  607.             if ($feed->DelayAutorespond) { 
  608.                 GFCommon::send_user_notification($form, $lead); 
  609.             } 
  610.         } 
  611.         else { 
  612.              
  613.             $notifications = GFCommon::get_notifications_to_send("form_submission", $form, $lead); 
  614.              
  615.             foreach ($notifications as $notification) { 
  616.              
  617.                 switch (rgar($notification, 'type')) { 
  618.                     // old "user" notification 
  619.                     case 'user': 
  620.                         if ($feed->DelayAutorespond) { 
  621.                             GFCommon::send_notification($notification, $form, $lead); 
  622.                         } 
  623.                         break; 
  624.  
  625.                     // old "admin" notification 
  626.                     case 'admin': 
  627.                         if ($feed->DelayNotify) { 
  628.                             GFCommon::send_notification($notification, $form, $lead); 
  629.                         } 
  630.                         break; 
  631.  
  632.                     // new since 1.7, add any notification you like 
  633.                     default: 
  634.                         if (trim($notification['to']) == '{admin_email}') { 
  635.                          
  636.                             if ($feed->DelayNotify) { 
  637.                                 GFCommon::send_notification($notification, $form, $lead); 
  638.                             } 
  639.                         } 
  640.                         else { 
  641.                             if ($feed->DelayAutorespond) { 
  642.                                 GFCommon::send_notification($notification, $form, $lead); 
  643.                             } 
  644.                         } 
  645.                         break; 
  646.                 } 
  647.             } 
  648.         } 
  649.     }

  652.     /** 
  653.     * This simply checks if the confirmation page needs to be displayed and if so then does. 
  654.     */ 
  655.     // ==================================================================================================================================== 
  656.     public function processFormConfirmation() { 
  657.          
  658.         // check for redirect to Gravity Forms page with our encoded parameters. 
  659.         if (isset($_GET[GFPAYSTATION_RETURN])) { 
  660.              
  661.             // Decode the encoded form and lead parameters. 
  662.             parse_str(base64_decode($_GET[GFPAYSTATION_RETURN]), $query); 
  663.  
  664.             // Make sure we have a match. 
  665.             if (wp_hash("form_id={$query['form_id']}&lead_id={$query['lead_id']}") == $query['hash']) { 
  666.  
  667.                 // stop WordPress SEO from stripping off our query parameters and redirecting the page 
  668.                 global $wpseo_front; 
  669.                  
  670.                 if (isset($wpseo_front)) { 
  671.                     remove_action('template_redirect', array($wpseo_front, 'clean_permalink'), 1); 
  672.                 } 
  673.                  
  674.                 // Load form and lead data. 
  675.                 $form = GFFormsModel::get_form_meta($query['form_id']); 
  676.                 $lead = GFFormsModel::get_lead($query['lead_id']); 
  677.  
  678.                 // Get confirmation page. 
  679.                 if (!class_exists('GFFormDisplay')) 
  680.                     require_once(GFCommon::get_base_path() . '/form_display.php'); 
  681.                  
  682.                 $confirmation = GFFormDisplay::handle_confirmation($form, $lead, false); 
  683.  
  684.                 // ------------------------------ 
  685.                 // If the confirmation is not an array (which means its just a message), then check to see if the status 
  686.                 // of the payment is failed. If so then change the confirmnation message to an error div with a message that 
  687.                 // the playment failed along with the error message from Paystation which is stored in the meta table for the lead. 
  688.                  
  689.                 // In order for this to work the postback needs to have happened before this code has loaded, which should nornally be the case 
  690.                 // given that paystation waits 10 seconds before redirecting back to this site. 
  691.                 if ((!is_array($confirmation)) && ($lead['payment_status'] == 'Failed')) { 
  692.                     $confirmation = "<div class='gform_wrapper'><div class='validation_error'>Sorry the payment failed : " . gform_get_meta($lead['id'], 'gfpaystation_error') . "</div></div>"; 
  693.                 } 
  694.                  
  695.                 // ------------------------------ 
  696.                 // Preload the GF submission, ready for processing the confirmation message. 
  697.                 GFFormDisplay::$submission[$form['id']] = array( 
  698.                     'is_confirmation' => true,  
  699.                     'confirmation_message' => $confirmation,  
  700.                     'form' => $form,  
  701.                     'lead' => $lead,  
  702.  ); 
  703.                  
  704.                 // ------------------------------ 
  705.                 // If it's a redirection (page or other URL) then do the redirect now. 
  706.                 if (is_array($confirmation) && isset($confirmation['redirect'])) { 
  707.                     header('Location: ' . $confirmation['redirect']); 
  708.                     exit; 
  709.                 } 
  710.             } 
  711.         } 
  712.     }
