  18.         // get the right filename extension 
  19.         $new_filename_paths = pathinfo( $new_filename );

  21.  
  22.         // take out filename from the original file or from the current attachment 
  23.         $original_media = (array) self::get_original_media( $media_id ); 
  24.  
  25.         if ( ! empty( $original_media ) ) { 
  26.             $original_file_parts = pathinfo( $original_media['file'] ); 
  27.             $filename_base = $original_file_parts['filename'];

  55.      * Post dates can change, so we use the original date used in the `guid` 
  56.      * url so edits can remain in the same folder. In the following function

  59.      * For example with a guid of

  314.     /** 
  315.      * Remove specific items from the `revision history` array 
  316.      * depending on the given criteria: array( 
  317.      *   'from' => (int) <from>,  
  318.      *   'to' =>   (int) <to>,  
  319.      * ) 
  320.      *  
  321.      * Also, it removes the file defined in each item. 
  322.      * 
  323.      * @param  number $media_id - media post ID 
  324.      * @param  object $criteria - criteria to remove the items 
  325.      * @param  array [$revision_history] - revision history array 
  326.      * @return array `revision_history` array updated. 
  327.      */ 
  328.     public static function remove_items_from_revision_history( $media_id, $criteria = array(), $revision_history ) { 
  329.         if ( ! isset ( $revision_history ) ) { 
  330.             $revision_history = self::get_revision_history( $media_id ); 
  331.         } 
  332.  
  333.         $from = $criteria['from']; 
  334.         $to = $criteria['to'] ? $criteria['to'] : ( $from + 1 ); 
  335.  
  336.         for ( $i = $from; $i < $to; $i++ ) { 
  337.             $removed_item = array_slice( $revision_history, $from, 1 ); 
  338.             if ( ! $removed_item ) { 
  339.                 break; 
  340.             } 
  341.  
  342.             array_splice( $revision_history, $from, 1 ); 
  343.             self::delete_media_history_file( $media_id, $removed_item[0]->file ); 
  344.         } 
  345.  
  346.         // override all history items 
  347.         delete_post_meta( $media_id, self::$WP_REVISION_HISTORY ); 
  348.         $revision_history = array_reverse( $revision_history ); 
  349.         foreach ( $revision_history as &$item ) { 
  350.             add_post_meta( $media_id, self::$WP_REVISION_HISTORY, $item ); 
  351.         } 
  352.  
  353.         return $revision_history; 
  354.     }

  444.         $has_original_media = self::get_original_media( $media_id ); 
  445.  
  446.         if ( ! $has_original_media ) {

  459.         }

  463.

  465.         $udpated_mime_type = $uploaded_file['type'];

  482.
