  342.     static function is_whitelisted_option( $option ) { 
  343.         foreach ( self::$default_options_whitelist as $whitelisted_option ) { 
  344.             if ( $whitelisted_option[0] === '/' && preg_match( $whitelisted_option, $option ) ) { 
  345.                 return true; 
  346.             } elseif ( $whitelisted_option === $option ) { 
  347.                 return true; 
  348.             } 
  349.         } 
  350.  
  351.         return false; 
  352.     }

  354.     static function get_max_sync_execution_time() { 
  355.         $max_exec_time = intval( ini_get( 'max_execution_time' ) ); 
  356.         if ( 0 === $max_exec_time ) { 
  357.             // 0 actually means "unlimited", but let's not treat it that way 
  358.             $max_exec_time = 60; 
  359.         } 
  360.         return floor( $max_exec_time / 3 );
