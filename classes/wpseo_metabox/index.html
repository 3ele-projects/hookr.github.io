  764.     /** 
  765.      * Save the WP SEO metadata for posts. 
  766.      * 
  767.      * @internal $_POST parameters are validated via sanitize_post_meta() 
  768.      * 
  769.      * @param int $post_id Post ID. 
  770.      * 
  771.      * @return  bool|void   Boolean false if invalid save post request 
  772.      */ 
  773.     function save_postdata( $post_id ) { 
  774.         // Bail if this is a multisite installation and the site has been switched. 
  775.         if ( is_multisite() && ms_is_switched() ) { 
  776.             return false; 
  777.         } 
  778.  
  779.         if ( $post_id === null ) { 
  780.             return false; 
  781.         } 
  782.  
  783.         if ( wp_is_post_revision( $post_id ) ) { 
  784.             $post_id = wp_is_post_revision( $post_id ); 
  785.         } 
  786.  
  787.         /** 
  788.          * Determine we're not accidentally updating a different post. 
  789.          * We can't use filter_input here as the ID isn't available at this point, other than in the $_POST data. 
  790.          */ 
  791.         if ( ! isset( $_POST['ID'] ) || $post_id !== (int) $_POST['ID'] ) { 
  792.             return false; 
  793.         } 
  794.  
  795.         clean_post_cache( $post_id ); 
  796.         $post = get_post( $post_id ); 
  797.  
  798.         if ( ! is_object( $post ) ) { 
  799.             // Non-existent post. 
  800.             return false; 
  801.         } 
  802.  
  803.         do_action( 'wpseo_save_compare_data', $post ); 
  804.  
  805.         $meta_boxes = apply_filters( 'wpseo_save_metaboxes', array() ); 
  806.         $meta_boxes = array_merge( $meta_boxes, $this->get_meta_field_defs( 'general', $post->post_type ), $this->get_meta_field_defs( 'advanced' ) ); 
  807.  
  808.         foreach ( $meta_boxes as $key => $meta_box ) { 
  809.  
  810.             // If analysis is disabled remove that analysis score value from the DB. 
  811.             if ( $this->is_meta_value_disabled( $key ) ) { 
  812.                 self::delete( $key, $post_id ); 
  813.                 continue; 
  814.             } 
  815.  
  816.             $data = null; 
  817.             if ( 'checkbox' === $meta_box['type'] ) { 
  818.                 $data = isset( $_POST[ self::$form_prefix . $key ] ) ? 'on' : 'off'; 
  819.             } 
  820.             else { 
  821.                 if ( isset( $_POST[ self::$form_prefix . $key ] ) ) { 
  822.                     $data = $_POST[ self::$form_prefix . $key ]; 
  823.                 } 
  824.             } 
  825.             if ( isset( $data ) ) { 
  826.                 self::set_value( $key, $data, $post_id ); 
  827.             } 
  828.         } 
  829.  
  830.         do_action( 'wpseo_saved_postdata' ); 
  831.     }
