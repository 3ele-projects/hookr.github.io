  483.      * an existing table alias that is compatible with the query that it

  486.      * An existing alias is compatible if (a) it is a sibling of `$clause` 
  487.      * (ie, it's under the scope of the same relation), and (b) the combination 
  488.      * of operator and relation between the clauses allows for a shared table 
  489.      * join. In the case of WP_Tax_Query, this only applies to 'IN' 
  490.      * clauses that are connected by the relation 'OR'.

  499.     protected function find_compatible_table_alias( $clause, $parent_query ) { 
  500.         $alias = false; 
  501.  
  502.         // Sanity check. Only IN queries use the JOIN syntax . 
  503.         if ( ! isset( $clause['operator'] ) || 'IN' !== $clause['operator'] ) { 
  504.             return $alias; 
  505.         } 
  506.  
  507.         // Since we're only checking IN queries, we're only concerned with OR relations. 
  508.         if ( ! isset( $parent_query['relation'] ) || 'OR' !== $parent_query['relation'] ) { 
  509.             return $alias; 
  510.         }

  512.         $compatible_operators = array( 'IN' ); 
  513.  
  514.         foreach ( $parent_query as $sibling ) {

  516.                 continue; 
  517.             } 
  518.  
  519.             if ( empty( $sibling['alias'] ) || empty( $sibling['operator'] ) ) { 
  520.                 continue; 
  521.             }

  523.             // The sibling must both have compatible operator to share its alias. 
  524.             if ( in_array( strtoupper( $sibling['operator'] ), $compatible_operators ) ) {

  527.             } 
  528.         }

  563.             $children = array(); 
  564.             foreach ( $query['terms'] as $term ) { 
  565.                 $children = array_merge( $children, get_term_children( $term, $query['taxonomy'] ) ); 
  566.                 $children[] = $term; 
  567.             } 
  568.             $query['terms'] = $children; 
  569.         } 
  570.  
  571.         $this->transform_query( $query, 'term_taxonomy_id' ); 
  572.     } 
  573.  
  574.     /** 
  575.      * Transforms a single query, from one field to another. 
  576.      * 
  577.      * @since 3.2.0 
  578.      * 
  579.      * @global wpdb $wpdb The WordPress database abstraction object. 
  580.      * 
  581.      * @param array  $query           The single query. Passed by reference. 
  582.      * @param string $resulting_field The resulting field. Accepts 'slug', 'name', 'term_taxonomy_id',  
  583.      *                                or 'term_id'. Default 'term_id'. 
  584.      */ 
  585.     public function transform_query( &$query, $resulting_field ) { 
  586.         global $wpdb; 
  587.  
  588.         if ( empty( $query['terms'] ) ) 
  589.             return; 
  590.  
  591.         if ( $query['field'] == $resulting_field ) 
  592.             return; 
  593.  
  594.         $resulting_field = sanitize_key( $resulting_field ); 
  595.  
  596.         switch ( $query['field'] ) { 
  597.             case 'slug': 
  598.             case 'name': 
  599.                 foreach ( $query['terms'] as &$term ) { 
  600.                     /** 
  601.                      * 0 is the $term_id parameter. We don't have a term ID yet, but it doesn't 
  602.                      * matter because `sanitize_term_field()` ignores the $term_id param when the 
  603.                      * context is 'db'. 
  604.                      */ 
  605.                     $term = "'" . esc_sql( sanitize_term_field( $query['field'], $term, 0, $query['taxonomy'], 'db' ) ) . "'"; 
  606.                 } 
  607.  
  608.                 $terms = implode( ", ", $query['terms'] ); 
  609.  
  610.                 $terms = $wpdb->get_col( " 
  611.                     SELECT $wpdb->term_taxonomy.$resulting_field 
  612.                     FROM $wpdb->term_taxonomy 
  613.                     INNER JOIN $wpdb->terms USING (term_id) 
  614.                     WHERE taxonomy = '{$query['taxonomy']}' 
  615.                     AND $wpdb->terms.{$query['field']} IN ($terms) 
  616.                 " ); 
  617.                 break; 
  618.             case 'term_taxonomy_id': 
  619.                 $terms = implode( ', ', array_map( 'intval', $query['terms'] ) ); 
  620.                 $terms = $wpdb->get_col( " 
  621.                     SELECT $resulting_field 
  622.                     FROM $wpdb->term_taxonomy 
  623.                     WHERE term_taxonomy_id IN ($terms) 
  624.                 " ); 
  625.                 break; 
  626.             default: 
  627.                 $terms = implode( ', ', array_map( 'intval', $query['terms'] ) ); 
  628.                 $terms = $wpdb->get_col( " 
  629.                     SELECT $resulting_field 
  630.                     FROM $wpdb->term_taxonomy 
  631.                     WHERE taxonomy = '{$query['taxonomy']}' 
  632.                     AND term_id IN ($terms) 
  633.                 " ); 
  634.         } 
  635.  
  636.         if ( 'AND' == $query['operator'] && count( $terms ) < count( $query['terms'] ) ) { 
  637.             $query = new WP_Error( 'inexistent_terms', __( 'Inexistent terms.' ) ); 
  638.             return; 
  639.         } 
  640.  
  641.         $query['terms'] = $terms; 
  642.         $query['field'] = $resulting_field; 
  643.     } 
  644. }
