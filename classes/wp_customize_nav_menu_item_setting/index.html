  562.  
  563.             foreach ( $items as $k => $item ) { 
  564.                 $items[ $k ]->{$args['output_key']} = $i++; 
  565.             } 
  566.         } 
  567.  
  568.         return $items; 
  569.     }

  571.     /**

  576.      * 
  577.      * @return WP_Post With wp_setup_nav_menu_item() applied. 
  578.      */ 
  579.     public function value_as_wp_post_nav_menu_item() { 
  580.         $item = (object) $this->value(); 
  581.         unset( $item->nav_menu_term_id ); 
  582.  
  583.         $item->post_status = $item->status; 
  584.         unset( $item->status ); 
  585.  
  586.         $item->post_type = 'nav_menu_item'; 
  587.         $item->menu_order = $item->position; 
  588.         unset( $item->position ); 
  589.  
  590.         if ( empty( $item->original_title ) ) { 
  591.             $item->original_title = $this->get_original_title( $item ); 
  592.         } 
  593.         if ( empty( $item->title ) && ! empty( $item->original_title ) ) { 
  594.             $item->title = $item->original_title; 
  595.         } 
  596.         if ( $item->title ) { 
  597.             $item->post_title = $item->title; 
  598.         }

  601.         $item->db_id = $this->post_id; 
  602.         $post = new WP_Post( (object) $item );

  605.             $post->post_author = get_current_user_id(); 
  606.         } 
  607.  
  608.         if ( ! isset( $post->type_label ) ) { 
  609.             $post->type_label = $this->get_type_label( $post ); 
  610.         } 
  611.

  613.         if ( 'post_type' === $post->type && ! empty( $post->object_id ) ) { 
  614.             $post->url = get_permalink( $post->object_id ); 
  615.         } elseif ( 'taxonomy' === $post->type && ! empty( $post->object ) && ! empty( $post->object_id ) ) { 
  616.             $post->url = get_term_link( (int) $post->object_id, $post->object ); 
  617.         } elseif ( 'post_type_archive' === $post->type && ! empty( $post->object ) ) { 
  618.             $post->url = get_post_type_archive_link( $post->object ); 
  619.         } 
  620.         if ( is_wp_error( $post->url ) ) { 
  621.             $post->url = ''; 
  622.         } 
  623.  
  624.         /** This filter is documented in wp-includes/nav-menu.php */ 
  625.         $post->attr_title = apply_filters( 'nav_menu_attr_title', $post->attr_title ); 
  626.  
  627.         /** This filter is documented in wp-includes/nav-menu.php */ 
  628.         $post->description = apply_filters( 'nav_menu_description', wp_trim_words( $post->description, 200 ) ); 
  629.  
  630.         /** This filter is documented in wp-includes/nav-menu.php */ 
  631.         $post = apply_filters( 'wp_setup_nav_menu_item', $post ); 
  632.  
  633.         return $post;

  636.     /** 
  637.      * Sanitize an input. 
  638.      * 
  639.      * Note that parent::sanitize() erroneously does wp_unslash() on $value, but 
  640.      * we remove that in this override. 
  641.      * 
  642.      * @since 4.3.0 
  643.      * @access public 
  644.      * 
  645.      * @param array $menu_item_value The value to sanitize. 
  646.      * @return array|false|null Null if an input isn't valid. False if it is marked for deletion. 
  647.      *                          Otherwise the sanitized value. 
  648.      */ 
  649.     public function sanitize( $menu_item_value ) { 
  650.         // Menu is marked for deletion. 
  651.         if ( false === $menu_item_value ) { 
  652.             return $menu_item_value; 
  653.         } 
  654.  
  655.         // Invalid. 
  656.         if ( ! is_array( $menu_item_value ) ) { 
  657.             return null; 
  658.         } 
  659.  
  660.         $default = array( 
  661.             'object_id' => 0,  
  662.             'object' => '',  
  663.             'menu_item_parent' => 0,  
  664.             'position' => 0,  
  665.             'type' => 'custom',  
  666.             'title' => '',  
  667.             'url' => '',  
  668.             'target' => '',  
  669.             'attr_title' => '',  
  670.             'description' => '',  
  671.             'classes' => '',  
  672.             'xfn' => '',  
  673.             'status' => 'publish',  
  674.             'original_title' => '',  
  675.             'nav_menu_term_id' => 0,  
  676.             '_invalid' => false,  
  677.  ); 
  678.         $menu_item_value = array_merge( $default, $menu_item_value ); 
  679.         $menu_item_value = wp_array_slice_assoc( $menu_item_value, array_keys( $default ) ); 
  680.         $menu_item_value['position'] = intval( $menu_item_value['position'] ); 
  681.  
  682.         foreach ( array( 'object_id', 'menu_item_parent', 'nav_menu_term_id' ) as $key ) { 
  683.             // Note we need to allow negative-integer IDs for previewed objects not inserted yet. 
  684.             $menu_item_value[ $key ] = intval( $menu_item_value[ $key ] ); 
  685.         } 
  686.  
  687.         foreach ( array( 'type', 'object', 'target' ) as $key ) { 
  688.             $menu_item_value[ $key ] = sanitize_key( $menu_item_value[ $key ] ); 
  689.         } 
  690.  
  691.         foreach ( array( 'xfn', 'classes' ) as $key ) { 
  692.             $value = $menu_item_value[ $key ]; 
  693.             if ( ! is_array( $value ) ) { 
  694.                 $value = explode( ' ', $value ); 
  695.             } 
  696.             $menu_item_value[ $key ] = implode( ' ', array_map( 'sanitize_html_class', $value ) ); 
  697.         } 
  698.  
  699.         $menu_item_value['original_title'] = sanitize_text_field( $menu_item_value['original_title'] ); 
  700.  
  701.         // Apply the same filters as when calling wp_insert_post(). 
  702.         $menu_item_value['title'] = wp_unslash( apply_filters( 'title_save_pre', wp_slash( $menu_item_value['title'] ) ) ); 
  703.         $menu_item_value['attr_title'] = wp_unslash( apply_filters( 'excerpt_save_pre', wp_slash( $menu_item_value['attr_title'] ) ) ); 
  704.         $menu_item_value['description'] = wp_unslash( apply_filters( 'content_save_pre', wp_slash( $menu_item_value['description'] ) ) ); 
  705.  
  706.         $menu_item_value['url'] = esc_url_raw( $menu_item_value['url'] ); 
  707.         if ( 'publish' !== $menu_item_value['status'] ) { 
  708.             $menu_item_value['status'] = 'draft'; 
  709.         } 
  710.  
  711.         $menu_item_value['_invalid'] = (bool) $menu_item_value['_invalid']; 
  712.  
  713.         /** This filter is documented in wp-includes/class-wp-customize-setting.php */ 
  714.         return apply_filters( "customize_sanitize_{$this->id}", $menu_item_value, $this ); 
  715.     }

  717.     /** 
  718.      * Creates/updates the nav_menu_item post for this setting. 
  719.      * 
  720.      * Any created menu items will have their assigned post IDs exported to the client 
  721.      * via the {@see 'customize_save_response'} filter. Likewise, any errors will be 
  722.      * exported to the client via the customize_save_response() filter. 
  723.      * 
  724.      * To delete a menu, the client can send false as the value. 
  725.      * 
  726.      * @since 4.3.0 
  727.      * @access protected 
  728.      * 
  729.      * @see wp_update_nav_menu_item() 
  730.      * 
  731.      * @param array|false $value The menu item array to update. If false, then the menu item will be deleted 
  732.      *                           entirely. See WP_Customize_Nav_Menu_Item_Setting::$default for what the value 
  733.      *                           should consist of. 
  734.      * @return null|void 
  735.      */ 
  736.     protected function update( $value ) { 
  737.         if ( $this->is_updated ) { 
  738.             return; 
  739.         } 
  740.  
  741.         $this->is_updated = true; 
  742.         $is_placeholder = ( $this->post_id < 0 ); 
  743.         $is_delete = ( false === $value ); 
  744.  
  745.         // Update the cached value. 
  746.         $this->value = $value; 
  747.  
  748.         add_filter( 'customize_save_response', array( $this, 'amend_customize_save_response' ) ); 
  749.  
  750.         if ( $is_delete ) { 
  751.             // If the current setting post is a placeholder, a delete request is a no-op. 
  752.             if ( $is_placeholder ) { 
  753.                 $this->update_status = 'deleted'; 
  754.             } else { 
  755.                 $r = wp_delete_post( $this->post_id, true ); 
  756.  
  757.                 if ( false === $r ) { 
  758.                     $this->update_error = new WP_Error( 'delete_failure' ); 
  759.                     $this->update_status = 'error'; 
  760.                 } else { 
  761.                     $this->update_status = 'deleted'; 
  762.                 } 
  763.                 // @todo send back the IDs for all associated nav menu items deleted, so these settings (and controls) can be removed from Customizer? 
  764.             } 
  765.         } else { 
  766.  
  767.             // Handle saving menu items for menus that are being newly-created. 
  768.             if ( $value['nav_menu_term_id'] < 0 ) { 
  769.                 $nav_menu_setting_id = sprintf( 'nav_menu[%s]', $value['nav_menu_term_id'] ); 
  770.                 $nav_menu_setting = $this->manager->get_setting( $nav_menu_setting_id ); 
  771.  
  772.                 if ( ! $nav_menu_setting || ! ( $nav_menu_setting instanceof WP_Customize_Nav_Menu_Setting ) ) { 
  773.                     $this->update_status = 'error'; 
  774.                     $this->update_error = new WP_Error( 'unexpected_nav_menu_setting' ); 
  775.                     return; 
  776.                 } 
  777.  
  778.                 if ( false === $nav_menu_setting->save() ) { 
  779.                     $this->update_status = 'error'; 
  780.                     $this->update_error = new WP_Error( 'nav_menu_setting_failure' ); 
  781.                     return; 
  782.                 } 
  783.  
  784.                 if ( $nav_menu_setting->previous_term_id !== intval( $value['nav_menu_term_id'] ) ) { 
  785.                     $this->update_status = 'error'; 
  786.                     $this->update_error = new WP_Error( 'unexpected_previous_term_id' ); 
  787.                     return; 
  788.                 } 
  789.  
  790.                 $value['nav_menu_term_id'] = $nav_menu_setting->term_id; 
  791.             } 
  792.  
  793.             // Handle saving a nav menu item that is a child of a nav menu item being newly-created. 
  794.             if ( $value['menu_item_parent'] < 0 ) { 
  795.                 $parent_nav_menu_item_setting_id = sprintf( 'nav_menu_item[%s]', $value['menu_item_parent'] ); 
  796.                 $parent_nav_menu_item_setting = $this->manager->get_setting( $parent_nav_menu_item_setting_id ); 
  797.  
  798.                 if ( ! $parent_nav_menu_item_setting || ! ( $parent_nav_menu_item_setting instanceof WP_Customize_Nav_Menu_Item_Setting ) ) { 
  799.                     $this->update_status = 'error'; 
  800.                     $this->update_error = new WP_Error( 'unexpected_nav_menu_item_setting' ); 
  801.                     return; 
  802.                 } 
  803.  
  804.                 if ( false === $parent_nav_menu_item_setting->save() ) { 
  805.                     $this->update_status = 'error'; 
  806.                     $this->update_error = new WP_Error( 'nav_menu_item_setting_failure' ); 
  807.                     return; 
  808.                 } 
  809.  
  810.                 if ( $parent_nav_menu_item_setting->previous_post_id !== intval( $value['menu_item_parent'] ) ) { 
  811.                     $this->update_status = 'error'; 
  812.                     $this->update_error = new WP_Error( 'unexpected_previous_post_id' ); 
  813.                     return; 
  814.                 } 
  815.  
  816.                 $value['menu_item_parent'] = $parent_nav_menu_item_setting->post_id; 
  817.             } 
  818.  
  819.             // Insert or update menu. 
  820.             $menu_item_data = array( 
  821.                 'menu-item-object-id' => $value['object_id'],  
  822.                 'menu-item-object' => $value['object'],  
  823.                 'menu-item-parent-id' => $value['menu_item_parent'],  
  824.                 'menu-item-position' => $value['position'],  
  825.                 'menu-item-type' => $value['type'],  
  826.                 'menu-item-title' => $value['title'],  
  827.                 'menu-item-url' => $value['url'],  
  828.                 'menu-item-description' => $value['description'],  
  829.                 'menu-item-attr-title' => $value['attr_title'],  
  830.                 'menu-item-target' => $value['target'],  
  831.                 'menu-item-classes' => $value['classes'],  
  832.                 'menu-item-xfn' => $value['xfn'],  
  833.                 'menu-item-status' => $value['status'],  
  834.  ); 
  835.  
  836.             $r = wp_update_nav_menu_item( 
  837.                 $value['nav_menu_term_id'],  
  838.                 $is_placeholder ? 0 : $this->post_id,  
  839.                 wp_slash( $menu_item_data ) 
  840.  ); 
  841.  
  842.             if ( is_wp_error( $r ) ) { 
  843.                 $this->update_status = 'error'; 
  844.                 $this->update_error = $r; 
  845.             } else { 
  846.                 if ( $is_placeholder ) { 
  847.                     $this->previous_post_id = $this->post_id; 
  848.                     $this->post_id = $r; 
  849.                     $this->update_status = 'inserted'; 
  850.                 } else { 
  851.                     $this->update_status = 'updated'; 
  852.                 } 
  853.             } 
  854.         } 
  855.  
  856.     }

  858.     /** 
  859.      * Export data for the JS client. 
  860.      * 
  861.      * @since 4.3.0 
  862.      * @access public 
  863.      * 
  864.      * @see WP_Customize_Nav_Menu_Item_Setting::update() 
  865.      * 
  866.      * @param array $data Additional information passed back to the 'saved' event on `wp.customize`. 
  867.      * @return array Save response data. 
  868.      */ 
  869.     public function amend_customize_save_response( $data ) { 
  870.         if ( ! isset( $data['nav_menu_item_updates'] ) ) { 
  871.             $data['nav_menu_item_updates'] = array(); 
  872.         } 
  873.  
  874.         $data['nav_menu_item_updates'][] = array( 
  875.             'post_id' => $this->post_id,

  881.     }
