  641.         case 'assign': 
  642.             list(, $name, $value) = $prop;

  644.                 $this->set($name, $value);

  666.                         $keywordArgs[$arg[1]] = $this->reduce($arg[2]); 
  667.                     } 
  668.                     break; 
  669.   
  670.                 case "lit": 
  671.                     $orderedArgs[] = $this->reduce($arg[1]); 
  672.                     break; 
  673.                 default: 
  674.                     $this->throwError("Unknown arg type: " . $arg[0]); 
  675.                 } 
  676.             } 
  677.

  679.   
  680.             if ($mixins === null) { 
  681.                 $this->throwError("{$prop[1][0]} is undefined"); 
  682.             }

  686.                     continue; 
  687.                 }

  707.                     if ($suffix !== null && 
  708.                         $subProp[0] == "assign" && 
  709.                         is_string($subProp[1]) && 
  710.                         $subProp[1]{0} != $this->vPrefix) 
  711.                     { 
  712.                         $subProp[2] = array( 
  713.                             'list', ' ',  
  714.                             array($subProp[2], array('keyword', $suffix)) 
  715.  ); 
  716.                     } 
  717.   
  718.                     $this->compileProp($subProp, $mixin, $out); 
  719.                 } 
  720.   
  721.                 $mixin->parent = $oldParent; 
  722.   
  723.                 if ($haveArgs) $this->popEnv(); 
  724.                 if ($haveScope) $this->popEnv(); 
  725.             } 
  726.   
  727.             break; 
  728.         case 'raw': 
  729.             $out->lines[] = $prop[1]; 
  730.             break;

  732.             list(, $name, $value) = $prop; 
  733.             $out->lines[] = "@$name " . $this->compileValue($this->reduce($value)).';'; 
  734.             break; 
  735.         case "comment": 
  736.             $out->lines[] = $prop[1]; 
  737.             break;

  743.                 $this->env->imports = array(); 
  744.             }

  752.             break; 
  753.         case "import_mixin": 
  754.             list(, $importId) = $prop; 
  755.             $import = $this->env->imports[$importId];

  772.     /** 
  773.      * Compiles a primitive value into a CSS property value. 
  774.      * 
  775.      * Values in lessphp are typed by being wrapped in arrays, their format is 
  776.      * typically: 
  777.      * 
  778.      *     array(type, contents [, additional_contents]*) 
  779.      * 
  780.      * The input is expected to be reduced. This function will not work on 
  781.      * things like expressions and variables. 
  782.      */ 
  783.     public function compileValue($value) { 
  784.         switch ($value[0]) { 
  785.         case 'list': 
  786.             // [1] - delimiter 
  787.             // [2] - array of values 
  788.             return implode($value[1], array_map(array($this, 'compileValue'), $value[2])); 
  789.         case 'raw_color': 
  790.             if (!empty($this->formatter->compressColors)) { 
  791.                 return $this->compileValue($this->coerceColor($value)); 
  792.             } 
  793.             return $value[1]; 
  794.         case 'keyword': 
  795.             // [1] - the keyword 
  796.             return $value[1]; 
  797.         case 'number': 
  798.             list(, $num, $unit) = $value; 
  799.             // [1] - the number 
  800.             // [2] - the unit 
  801.             if ($this->numberPrecision !== null) { 
  802.                 $num = round($num, $this->numberPrecision); 
  803.             } 
  804.             return $num . $unit; 
  805.         case 'string': 
  806.             // [1] - contents of string (includes quotes) 
  807.             list(, $delim, $content) = $value; 
  808.             foreach ($content as &$part) { 
  809.                 if (is_array($part)) { 
  810.                     $part = $this->compileValue($part); 
  811.                 } 
  812.             } 
  813.             return $delim . implode($content) . $delim; 
  814.         case 'color': 
  815.             // [1] - red component (either number or a %) 
  816.             // [2] - green component 
  817.             // [3] - blue component 
  818.             // [4] - optional alpha component 
  819.             list(, $r, $g, $b) = $value; 
  820.             $r = round($r); 
  821.             $g = round($g); 
  822.             $b = round($b); 
  823.  
  824.             if (count($value) == 5 && $value[4] != 1) { // rgba 
  825.                 return 'rgba('.$r.', '.$g.', '.$b.', '.$value[4].')'; 
  826.             } 
  827.  
  828.             $h = sprintf("#%02x%02x%02x", $r, $g, $b); 
  829.  
  830.             if (!empty($this->formatter->compressColors)) { 
  831.                 // Converting hex color to short notation (e.g. #003399 to #039) 
  832.                 if ($h[1] === $h[2] && $h[3] === $h[4] && $h[5] === $h[6]) { 
  833.                     $h = '#' . $h[1] . $h[3] . $h[5]; 
  834.                 } 
  835.             } 
  836.  
  837.             return $h; 
  838.  
  839.         case 'function': 
  840.             list(, $name, $args) = $value; 
  841.             return $name.'('.$this->compileValue($args).')'; 
  842.         default: // assumed to be unit 
  843.             $this->throwError("unknown value type: $value[0]"); 
  844.         } 
  845.     }

  1903.     /** 
  1904.      * Execute lessphp on a .less file or a lessphp cache structure 
  1905.      * 
  1906.      * The lessphp cache structure contains information about a specific 
  1907.      * less file having been parsed. It can be used as a hint for future 
  1908.      * calls to determine whether or not a rebuild is required. 
  1909.      * 
  1910.      * The cache structure contains two important keys that may be used 
  1911.      * externally: 
  1912.      * 
  1913.      * compiled: The final compiled CSS 
  1914.      * updated: The time (in seconds) the CSS was last compiled 
  1915.      * 
  1916.      * The cache structure is a plain-ol' PHP associative array and can 
  1917.      * be serialized and unserialized without a hitch. 
  1918.      * 
  1919.      * @param mixed $in Input 
  1920.      * @param bool $force Force rebuild? 
  1921.      * @return array lessphp cache structure 
  1922.      */ 
  1923.     public function cachedCompile($in, $force = false) { 
  1924.         // assume no root 
  1925.         $root = null; 
  1926.  
  1927.         if (is_string($in)) { 
  1928.             $root = $in; 
  1929.         } elseif (is_array($in) and isset($in['root'])) { 
  1930.             if ($force or ! isset($in['files'])) { 
  1931.                 // If we are forcing a recompile or if for some reason the 
  1932.                 // structure does not contain any file information we should 
  1933.                 // specify the root to trigger a rebuild. 
  1934.                 $root = $in['root']; 
  1935.             } elseif (isset($in['files']) and is_array($in['files'])) { 
  1936.                 foreach ($in['files'] as $fname => $ftime ) { 
  1937.                     if (!file_exists($fname) or filemtime($fname) > $ftime) { 
  1938.                         // One of the files we knew about previously has changed 
  1939.                         // so we should look at our incoming root again. 
  1940.                         $root = $in['root']; 
  1941.                         break; 
  1942.                     } 
  1943.                 } 
  1944.             } 
  1945.         } else { 
  1946.             // TODO: Throw an exception? We got neither a string nor something 
  1947.             // that looks like a compatible lessphp cache structure. 
  1948.             return null; 
  1949.         } 
  1950.  
  1951.         if ($root !== null) { 
  1952.             // If we have a root value which means we should rebuild. 
  1953.             $out = array(); 
  1954.             $out['root'] = $root; 
  1955.             $out['compiled'] = $this->compileFile($root); 
  1956.             $out['files'] = $this->allParsedFiles(); 
  1957.             $out['updated'] = time(); 
  1958.             return $out; 
  1959.         } else { 
  1960.             // No changes, pass back the structure 
  1961.             // we were given initially. 
  1962.             return $in; 
  1963.         }
