  565.     static function import_gallery($galleryfolder, $gallery_id=NULL) { 
  566.  
  567.         global $wpdb, $user_ID; 
  568.  
  569.         // get the current user ID 
  570.         wp_get_current_user(); 
  571.  
  572.         $created_msg = ''; 
  573.  
  574.         // remove trailing slash at the end, if somebody use it 
  575.         $galleryfolder = untrailingslashit($galleryfolder); 
  576.  
  577.         $fs = C_Fs::get_instance(); 
  578.         if (is_null($gallery_id)) { 
  579.             $gallerypath = $fs->join_paths($fs->get_document_root('content'), $galleryfolder); 
  580.         } else { 
  581.             $storage = C_Gallery_Storage::get_instance(); 
  582.             $gallerypath = $storage->get_gallery_abspath($gallery_id); 
  583.         } 
  584.  
  585.         if (!is_dir($gallerypath)) { 
  586.             nggGallery::show_error(sprintf(__("Directory <strong>%s</strong> doesn`t exist!", 'nggallery'), esc_html($gallerypath))); 
  587.             return ; 
  588.         } 
  589.  
  590.         // read list of images 
  591.         $new_imageslist = nggAdmin::scandir($gallerypath); 
  592.  
  593.         if (empty($new_imageslist)) { 
  594.             nggGallery::show_message(sprintf(__("Directory <strong>%s</strong> contains no pictures", 'nggallery'), esc_html($gallerypath))); 
  595.             return; 
  596.         } 
  597.  
  598.         // take folder name as gallery name 
  599.         $galleryname = basename($galleryfolder); 
  600.         $galleryname = apply_filters('ngg_gallery_name', $galleryname); 
  601.  
  602.         // check for existing gallery folder 
  603.         if (is_null($gallery_id)) 
  604.             $gallery_id = $wpdb->get_var("SELECT gid FROM $wpdb->nggallery WHERE path = '$galleryfolder' "); 
  605.  
  606.         if (!$gallery_id) { 
  607.             // now add the gallery to the database 
  608.             $gallery_id = nggdb::add_gallery( $galleryname, $galleryfolder, '', 0, 0, $user_ID ); 
  609.             if (!$gallery_id) { 
  610.                 nggGallery::show_error(__('Database error. Could not add gallery!', 'nggallery')); 
  611.                 return; 
  612.             } 
  613.             else { 
  614.                 do_action('ngg_created_new_gallery', $gallery_id); 
  615.             } 
  616.             $created_msg = sprintf( 
  617.                 _n("Gallery <strong>%s</strong> successfully created!", 'Galleries <strong>%s</strong> successfully created!', 1, 'nggallery'),  
  618.                 esc_html($galleryname) 
  619.  ); 
  620.         } 
  621.  
  622.         // Look for existing image list 
  623.         $old_imageslist = $wpdb->get_col("SELECT filename FROM $wpdb->nggpictures WHERE galleryid = '$gallery_id' "); 
  624.  
  625.         // if no images are there, create empty array 
  626.         if ($old_imageslist == NULL) 
  627.             $old_imageslist = array(); 
  628.  
  629.         // check difference 
  630.         $new_images = array_diff($new_imageslist, $old_imageslist); 
  631.  
  632.         // all images must be valid files 
  633.         foreach($new_images as $key => $picture) { 
  634.  
  635.             // filter function to rename/change/modify image before 
  636.             $picture = apply_filters('ngg_pre_add_new_image', $picture, $gallery_id); 
  637.             $new_images[$key] = $picture; 
  638.  
  639.             if (!@getimagesize($gallerypath . '/' . $picture) ) { 
  640.                 unset($new_images[$key]); 
  641.                 @unlink($gallerypath . '/' . $picture); 
  642.             } 
  643.         } 
  644.  
  645.         // add images to database 
  646.         $image_ids = nggAdmin::add_Images($gallery_id, $new_images); 
  647.         do_action('ngg_after_new_images_added', $gallery_id, $image_ids); 
  648.  
  649.         //add the preview image if needed 
  650.         nggAdmin::set_gallery_preview ( $gallery_id ); 
  651.  
  652.         // now create thumbnails 
  653.         nggAdmin::do_ajax_operation( 'create_thumbnail' , $image_ids, __('Create new thumbnails', 'nggallery') ); 
  654.  
  655.         //TODO:Message will not shown, because AJAX routine require more time, message should be passed to AJAX 
  656.         $message = $created_msg . sprintf(_n('%s picture successfully added', '%s pictures successfully added', count($image_ids), 'nggallery'), count($image_ids));

  658.         $message .=  __('Edit gallery', 'nggallery'); 
  659.         $message .= '</a>]'; 
  660.  
  661.         nggGallery::show_message($message); 
  662.  
  663.         return; 
  664.     }
