  508.                         break; 
  509.                     } 
  510.  
  511.                     // Allow to clear the API key field

  517.                     require_once WP_PLUGIN_DIR . '/akismet/class.akismet.php';

  539.                     $updated = false; 
  540.                     break; 
  541.  
  542.                 case 'google_analytics_tracking_id':

  570.         } else {

  576.                 /** Translators: the plural variable is a comma-separated list. Example: dog, cat, bird. */

  580.             }

  647.     public function can_request( $request ) {

  650.         } else {
