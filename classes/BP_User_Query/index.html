  517.             wp_cache_set( 'bp_core_userdata_' . $u->ID, $u, 'bp' ); 
  518.         } 
  519.  
  520.         // We calculate total_users using a standalone query, except 
  521.         // when a whitelist of user_ids is passed to the constructor. 
  522.         // This clause covers the latter situation, and ensures that 
  523.         // pagination works when querying by $user_ids. 
  524.         if ( empty( $this->total_users ) ) { 
  525.             $this->total_users = count( $wp_user_query->results ); 
  526.         } 
  527.  
  528.         // Reindex for easier matching. 
  529.         $r = array(); 
  530.         foreach ( $wp_user_query->results as $u ) { 
  531.             $r[ $u->ID ] = $u; 
  532.         } 
  533.  
  534.         // Match up to the user ids from the main query. 
  535.         foreach ( $this->user_ids as $key => $uid ) { 
  536.             if ( isset( $r[ $uid ] ) ) { 
  537.                 $r[ $uid ]->ID = (int) $uid; 
  538.                 $r[ $uid ]->user_status = (int) $r[ $uid ]->user_status; 
  539.  
  540.                 $this->results[ $uid ] = $r[ $uid ]; 
  541.  
  542.                 // The BP template functions expect an 'id' 
  543.                 // (as opposed to 'ID') property. 
  544.                 $this->results[ $uid ]->id = (int) $uid; 
  545.  
  546.             // Remove user ID from original user_ids property. 
  547.             } else { 
  548.                 unset( $this->user_ids[ $key ] ); 
  549.             } 
  550.         } 
  551.     } 
  552.  
  553.     /** 
  554.      * Fetch the IDs of users to put in the IN clause of the main query. 
  555.      * 
  556.      * By default, returns the value passed to it 
  557.      * ($this->query_vars['include']). Having this abstracted into a 
  558.      * standalone method means that extending classes can override the 
  559.      * logic, parsing together their own user_id limits with the 'include' 
  560.      * ids passed to the class constructor. See {@link BP_Group_Member_Query} 
  561.      * for an example. 
  562.      * 
  563.      * @since 1.8.0 
  564.      * 
  565.      * @param array $include Sanitized array of user IDs, as passed to the 'include' 
  566.      *                       parameter of the class constructor. 
  567.      * @return array The list of users to which the main query should be 
  568.      *               limited. 
  569.      */ 
  570.     public function get_include_ids( $include = array() ) { 
  571.         return $include; 
  572.     } 
  573.  
  574.     /** 
  575.      * Perform a database query to populate any extra metadata we might need. 
  576.      * 
  577.      * Different components will hook into the 'bp_user_query_populate_extras' 
  578.      * action to loop in the things they want. 
  579.      * 
  580.      * @since 1.7.0 
  581.      * 
  582.      * @global WPDB $wpdb Global WordPress database access object. 
  583.      */ 
  584.     public function populate_extras() { 
  585.         global $wpdb; 
  586.  
  587.         // Bail if no users. 
  588.         if ( empty( $this->user_ids ) || empty( $this->results ) ) { 
  589.             return; 
  590.         } 
  591.  
  592.         // Bail if the populate_extras flag is set to false 
  593.         // In the case of the 'popular' sort type, we force 
  594.         // populate_extras to true, because we need the friend counts. 
  595.         if ( 'popular' == $this->query_vars['type'] ) { 
  596.             $this->query_vars['populate_extras'] = 1; 
  597.         } 
  598.  
  599.         if ( ! (bool) $this->query_vars['populate_extras'] ) { 
  600.             return; 
  601.         } 
  602.  
  603.         // Turn user ID's into a query-usable, comma separated value. 
  604.         $user_ids_sql = implode( ', ', wp_parse_id_list( $this->user_ids ) ); 
  605.  
  606.         /** 
  607.          * Allows users to independently populate custom extras. 
  608.          * 
  609.          * Note that anything you add here should query using $user_ids_sql, to 
  610.          * avoid running multiple queries per user in the loop. 
  611.          * 
  612.          * Two BuddyPress components currently do this: 
  613.          * - XProfile: To override display names. 
  614.          * - Friends:  To set whether or not a user is the current users friend. 
  615.          * 
  616.          * @see bp_xprofile_filter_user_query_populate_extras() 
  617.          * @see bp_friends_filter_user_query_populate_extras() 
  618.          * 
  619.          * @since 1.7.0 
  620.          * 
  621.          * @param BP_User_Query $this         Current BP_User_Query instance. 
  622.          * @param string        $user_ids_sql Comma-separated string of user IDs. 
  623.          */ 
  624.         do_action_ref_array( 'bp_user_query_populate_extras', array( $this, $user_ids_sql ) ); 
  625.  
  626.         // Fetch last_active data from the activity table. 
  627.         $last_activities = BP_Core_User::get_last_activity( $this->user_ids ); 
  628.  
  629.         // Set a last_activity value for each user, even if it's empty. 
  630.         foreach ( $this->results as $user_id => $user ) { 
  631.             $user_last_activity = isset( $last_activities[ $user_id ] ) ? $last_activities[ $user_id ]['date_recorded'] : ''; 
  632.             $this->results[ $user_id ]->last_activity = $user_last_activity; 
  633.         } 
  634.  
  635.         // Fetch usermeta data 
  636.         // We want the three following pieces of info from usermeta: 
  637.         // - friend count 
  638.         // - latest update. 
  639.         $total_friend_count_key = bp_get_user_meta_key( 'total_friend_count' ); 
  640.         $bp_latest_update_key = bp_get_user_meta_key( 'bp_latest_update' ); 
  641.  
  642.         // Total_friend_count must be set for each user, even if its 
  643.         // value is 0. 
  644.         foreach ( $this->results as $uindex => $user ) { 
  645.             $this->results[$uindex]->total_friend_count = 0; 
  646.         } 
  647.  
  648.         // Create, prepare, and run the separate usermeta query. 
  649.         $user_metas = $wpdb->get_results( $wpdb->prepare( "SELECT user_id, meta_key, meta_value FROM {$wpdb->usermeta} WHERE meta_key IN (%s, %s) AND user_id IN ({$user_ids_sql})", $total_friend_count_key, $bp_latest_update_key ) ); 
  650.  
  651.         // The $members_template global expects the index key to be different 
  652.         // from the meta_key in some cases, so we rejig things here. 
  653.         foreach ( $user_metas as $user_meta ) { 
  654.             switch ( $user_meta->meta_key ) { 
  655.                 case $total_friend_count_key : 
  656.                     $key = 'total_friend_count'; 
  657.                     break; 
  658.  
  659.                 case $bp_latest_update_key : 
  660.                     $key = 'latest_update'; 
  661.                     break; 
  662.             } 
  663.  
  664.             if ( isset( $this->results[ $user_meta->user_id ] ) ) { 
  665.                 $this->results[ $user_meta->user_id ]->{$key} = $user_meta->meta_value; 
  666.             } 
  667.         } 
  668.  
  669.         // When meta_key or meta_value have been passed to the query,  
  670.         // fetch the resulting values for use in the template functions. 
  671.         if ( ! empty( $this->query_vars['meta_key'] ) ) { 
  672.             $meta_sql = array( 
  673.                 'select' => "SELECT user_id, meta_key, meta_value",  
  674.                 'from' => "FROM $wpdb->usermeta",  
  675.                 'where' => $wpdb->prepare( "WHERE meta_key = %s", $this->query_vars['meta_key'] ) 
  676.  ); 
  677.  
  678.             if ( false !== $this->query_vars['meta_value'] ) { 
  679.                 $meta_sql['where'] .= $wpdb->prepare( " AND meta_value = %s", $this->query_vars['meta_value'] ); 
  680.             } 
  681.  
  682.             $metas = $wpdb->get_results( "{$meta_sql['select']} {$meta_sql['from']} {$meta_sql['where']}" ); 
  683.  
  684.             if ( ! empty( $metas ) ) { 
  685.                 foreach ( $metas as $meta ) { 
  686.                     if ( isset( $this->results[ $meta->user_id ] ) ) { 
  687.                         $this->results[ $meta->user_id ]->meta_key = $meta->meta_key; 
  688.  
  689.                         if ( ! empty( $meta->meta_value ) ) { 
  690.                             $this->results[ $meta->user_id ]->meta_value = $meta->meta_value; 
  691.                         } 
  692.                     } 
  693.                 } 
  694.             } 
  695.         } 
  696.     }

  698.     /** 
  699.      * Get a SQL clause representing member_type include/exclusion. 
  700.      * 
  701.      * @since 2.4.0 
  702.      * 
  703.      * @param string|array $member_types Array or comma-separated list of member types. 
  704.      * @param string       $operator     'IN' or 'NOT IN'. 
  705.      * @return string 
  706.      */ 
  707.     protected function get_sql_clause_for_member_types( $member_types, $operator ) { 
  708.         global $wpdb; 
  709.  
  710.         // Sanitize. 
  711.         if ( 'NOT IN' !== $operator ) { 
  712.             $operator = 'IN'; 
  713.         } 
  714.  
  715.         // Parse and sanitize types. 
  716.         if ( ! is_array( $member_types ) ) { 
  717.             $member_types = preg_split( '/[, \s+]/', $member_types ); 
  718.         } 
  719.  
  720.         $types = array(); 
  721.         foreach ( $member_types as $mt ) { 
  722.             if ( bp_get_member_type_object( $mt ) ) { 
  723.                 $types[] = $mt; 
  724.             } 
  725.         } 
  726.  
  727.         $tax_query = new WP_Tax_Query( array( 
  728.             array( 
  729.                 'taxonomy' => bp_get_member_type_tax_name(),  
  730.                 'field' => 'name',  
  731.                 'operator' => $operator,  
  732.                 'terms' => $types,  
  733.  ),  
  734.  ) ); 
  735.  
  736.         // Switch to the root blog, where member type taxonomies live. 
  737.         $site_id = bp_get_taxonomy_term_site_id( bp_get_member_type_tax_name() ); 
  738.         $switched = false; 
  739.         if ( $site_id !== get_current_blog_id() ) { 
  740.             switch_to_blog( $site_id ); 
  741.             $switched = true; 
  742.         } 
  743.  
  744.         $sql_clauses = $tax_query->get_sql( 'u', $this->uid_name ); 
  745.  
  746.         $clause = ''; 
  747.  
  748.         // The no_results clauses are the same between IN and NOT IN. 
  749.         if ( false !== strpos( $sql_clauses['where'], '0 = 1' ) ) { 
  750.             $clause = $this->no_results['where']; 
  751.  
  752.         // The tax_query clause generated for NOT IN can be used almost as-is. We just trim the leading 'AND'. 
  753.         } elseif ( 'NOT IN' === $operator ) { 
  754.             $clause = preg_replace( '/^\s*AND\s*/', '', $sql_clauses['where'] ); 
  755.  
  756.         // IN clauses must be converted to a subquery. 
  757.         } elseif ( preg_match( '/' . $wpdb->term_relationships . '\.term_taxonomy_id IN \([0-9, ]+\)/', $sql_clauses['where'], $matches ) ) { 
  758.             $clause = "u.{$this->uid_name} IN ( SELECT object_id FROM $wpdb->term_relationships WHERE {$matches[0]} )"; 
  759.         } 
  760.  
  761.         if ( $switched ) { 
  762.             restore_current_blog(); 
  763.         } 
  764.  
  765.         return $clause; 
  766.     } 
  767. }
