  602.     private function _can_enqueue_script( $enqueue_conditions, $form, $is_ajax ) { 
  603.         if ( empty( $enqueue_conditions ) ) { 
  604.             return false; 
  605.         } 
  606.  
  607.         foreach ( $enqueue_conditions as $condition ) { 
  608.             if ( is_callable( $condition ) ) { 
  609.                 return call_user_func( $condition, $form, $is_ajax ); 
  610.             } else { 
  611.                 $query_matches = isset( $condition['query'] ) ? $this->_request_condition_matches( $_GET, $condition['query'] ) : true; 
  612.                 $post_matches = isset( $condition['post'] ) ? $this->_request_condition_matches( $_POST, $condition['query'] ) : true; 
  613.                 $admin_page_matches = isset( $condition['admin_page'] ) ? $this->_page_condition_matches( $condition['admin_page'], rgar( $condition, 'tab' ) ) : true; 
  614.                 $field_type_matches = isset( $condition['field_types'] ) ? $this->_field_condition_matches( $condition['field_types'], $form ) : true; 
  615.  
  616.                 if ( $query_matches && $post_matches && $admin_page_matches && $field_type_matches ) { 
  617.                     return true; 
  618.                 } 
  619.             } 
  620.         } 
  621.  
  622.         return false; 
  623.     }

  625.     private function _request_condition_matches( $request, $query ) { 
  626.         parse_str( $query, $query_array ); 
  627.         foreach ( $query_array as $key => $value ) { 
  628.  
  629.             switch ( $value ) { 
  630.                 case '_notempty_' : 
  631.                     if ( rgempty( $key, $request ) ) { 
  632.                         return false; 
  633.                     } 
  634.                     break; 
  635.                 case '_empty_' :

  637.                         return false;

  642.                         return false;

  645.             } 
  646.         }
