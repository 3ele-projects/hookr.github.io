  214.     /** 
  215.      * Updates information on the "View version x.x details" page with custom data. 
  216.      * 
  217.      * @uses api_request() 
  218.      * 
  219.      * @param mixed   $_data 
  220.      * @param string  $_action 
  221.      * @param object  $_args 
  222.      * @return object $_data 
  223.      */ 
  224.     public function plugins_api_filter( $_data, $_action = '', $_args = null ) { 
  225.  
  226.         if ( $_action != 'plugin_information' ) { 
  227.  
  228.             return $_data; 
  229.  
  230.         } 
  231.  
  232.         if ( ! isset( $_args->slug ) || ( $_args->slug != $this->slug ) ) { 
  233.  
  234.             return $_data; 
  235.  
  236.         } 
  237.  
  238.         $to_send = array( 
  239.             'slug' => $this->slug,  
  240.             'is_ssl' => is_ssl(),  
  241.             'fields' => array( 
  242.                 'banners' => array(),  
  243.                 'reviews' => false 
  244.  ) 
  245.  ); 
  246.  
  247.         $cache_key = 'edd_api_request_' . md5( serialize( $this->slug . $this->api_data['license'] ) ); 
  248.  
  249.         // Get the transient where we store the api request for this plugin for 24 hours 
  250.         $edd_api_request_transient = $this->get_cached_version_info( $cache_key ); 
  251.  
  252.         //If we have no transient-saved value, run the API, set a fresh transient with the API value, and return that value too right now. 
  253.         if ( empty( $edd_api_request_transient ) ) { 
  254.  
  255.             $api_response = $this->api_request( 'plugin_information', $to_send ); 
  256.  
  257.             // Expires in 3 hours 
  258.             $this->set_version_info_cache( $api_response, $cache_key ); 
  259.  
  260.             if ( false !== $api_response ) { 
  261.                 $_data = $api_response; 
  262.             } 
  263.  
  264.         } else { 
  265.             $_data = $edd_api_request_transient; 
  266.         } 
  267.  
  268.         return $_data; 
  269.     }

  286.     /** 
  287.      * Calls the API and, if successfull, returns the object delivered by the API. 
  288.      * 
  289.      * @uses get_bloginfo() 
  290.      * @uses wp_remote_post() 
  291.      * @uses is_wp_error() 
  292.      * 
  293.      * @param string  $_action The requested action. 
  294.      * @param array   $_data   Parameters for the API action. 
  295.      * @return false|object 
  296.      */ 
  297.     private function api_request( $_action, $_data ) { 
  298.  
  299.         global $wp_version; 
  300.  
  301.         $data = array_merge( $this->api_data, $_data ); 
  302.  
  303.         if ( $data['slug'] != $this->slug ) { 
  304.             return; 
  305.         } 
  306.  
  307.         if( $this->api_url == trailingslashit (home_url() ) ) { 
  308.             return false; // Don't allow a plugin to ping itself 
  309.         } 
  310.  
  311.         $api_params = array( 
  312.             'edd_action' => 'get_version',  
  313.             'license' => ! empty( $data['license'] ) ? $data['license'] : '',  
  314.             'item_name' => isset( $data['item_name'] ) ? $data['item_name'] : false,  
  315.             'item_id' => isset( $data['item_id'] ) ? $data['item_id'] : false,  
  316.             'slug' => $data['slug'],  
  317.             'author' => $data['author'],  
  318.             'url' => home_url(),  
  319.             'beta' => ! empty( $data['beta'] ),  
  320.  ); 
  321.  
  322.         $request = wp_remote_post( $this->api_url, array( 'timeout' => 15, 'sslverify' => false, 'body' => $api_params ) ); 
  323.  
  324.         if ( ! is_wp_error( $request ) ) { 
  325.             $request = json_decode( wp_remote_retrieve_body( $request ) ); 
  326.         } 
  327.  
  328.         if ( $request && isset( $request->sections ) ) { 
  329.             $request->sections = maybe_unserialize( $request->sections ); 
  330.         } else { 
  331.             $request = false; 
  332.         } 
  333.  
  334.         if ( $request && isset( $request->banners ) ) { 
  335.             $request->banners = maybe_unserialize( $request->banners ); 
  336.         } 
  337.  
  338.         if( ! empty( $request ) && is_array( $request->sections ) ) { 
  339.             foreach( $request->sections as $key => $section ) { 
  340.                 $request->$key = (array) $section; 
  341.             } 
  342.         } 
  343.  
  344.         return $request; 
  345.     }

  413.     public function get_cached_version_info( $cache_key = '' ) { 
  414. return false; 
  415.         if( empty( $cache_key ) ) {
