  419.     /** 
  420.      * The default value for mulit-input date fields will always be an array in mdy order 
  421.      * this code will alter the order of the values to the date format of the field 
  422.      */ 
  423.     public function get_date_array_by_format( $value ) { 
  424.         $format = empty( $this->dateFormat ) ? 'mdy' : esc_attr( $this->dateFormat ); 
  425.         $position = substr( $format, 0, 3 ); 
  426.         $date = array_combine( array( 'm', 'd', 'y' ), $value );            // takes our numerical array and converts it to an associative array 
  427.         $value = array_merge( array_flip( str_split( $position ) ), $date ); // uses the mdy position as the array keys and creates a new array in the desired order 
  428.  
  429.         return $value; 
  430.     }
