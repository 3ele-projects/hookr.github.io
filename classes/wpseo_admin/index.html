  536.     /** 
  537.      * Filter the stopwords from the slug 
  538.      * 
  539.      * @param string $slug       The current slug, if not empty there will be done nothing. 
  540.      * @param string $post_title The title which will be used in case of an empty slug. 
  541.      * 
  542.      * @return string 
  543.      */ 
  544.     public function filter_stopwords_from_slug( $slug, $post_title ) { 
  545.         // Don't change an existing slug. 
  546.         if ( isset( $slug ) && $slug !== '' ) { 
  547.             return $slug; 
  548.         } 
  549.  
  550.         // When the post title is empty, just return the slug. 
  551.         if ( empty( $post_title ) ) { 
  552.             return $slug; 
  553.         } 
  554.  
  555.         // Don't change the slug if this is a multisite installation and the site has been switched. 
  556.         if ( is_multisite() && ms_is_switched() ) { 
  557.             return $slug; 
  558.         } 
  559.  
  560.         // Don't change slug if the post is a draft, this conflicts with polylang. 
  561.         // Doesn't work with filter_input() since need current value, not originally submitted one. 
  562.         if ( 'draft' === $_POST['post_status'] ) { 
  563.             return $slug; 
  564.         } 
  565.  
  566.         // Lowercase the slug and strip slashes. 
  567.         $new_slug = sanitize_title( stripslashes( $post_title ) ); 
  568.  
  569.         $stop_words = new WPSEO_Admin_Stop_Words(); 
  570.         return $stop_words->remove_in( $new_slug ); 
  571.     }
