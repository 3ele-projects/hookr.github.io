  629.      */ 
  630.     protected function parse_json_params() { 
  631.         if ( $this->parsed_json ) { 
  632.             return true; 
  633.         } 
  634.  
  635.         $this->parsed_json = true; 
  636.  
  637.         // Check that we actually got JSON. 
  638.         $content_type = $this->get_content_type(); 
  639.  
  640.         if ( empty( $content_type ) || 'application/json' !== $content_type['value'] ) { 
  641.             return true;

  643.  
  644.         $body = $this->get_body(); 
  645.         if ( empty( $body ) ) { 
  646.             return true;

  649.         $params = json_decode( $body, true );

  651.         /**

  669.             return new WP_Error( 'rest_invalid_json', __( 'Invalid JSON body passed.' ), $error_data ); 
  670.         } 
  671.  
  672.         $this->params['JSON'] = $params; 
  673.         return true; 
  674.     } 
  675.  
  676.     /** 
  677.      * Parses the request body parameters. 
  678.      * 
  679.      * Parses out URL-encoded bodies for request methods that aren't supported 
  680.      * natively by PHP. In PHP 5.x, only POST has these parsed automatically. 
  681.      * 
  682.      * @since 4.4.0 
  683.      * @access protected 
  684.      */ 
  685.     protected function parse_body_params() { 
  686.         if ( $this->parsed_body ) { 
  687.             return; 
  688.         } 
  689.  
  690.         $this->parsed_body = true; 
  691.  
  692.         /** 
  693.          * Check that we got URL-encoded. Treat a missing content-type as 
  694.          * URL-encoded for maximum compatibility. 
  695.          */ 
  696.         $content_type = $this->get_content_type(); 
  697.  
  698.         if ( ! empty( $content_type ) && 'application/x-www-form-urlencoded' !== $content_type['value'] ) { 
  699.             return; 
  700.         } 
  701.  
  702.         parse_str( $this->get_body(), $params ); 
  703.  
  704.         /** 
  705.          * Amazingly, parse_str follows magic quote rules. Sigh. 
  706.          * 
  707.          * NOTE: Do not refactor to use `wp_unslash`. 
  708.          */ 
  709.         if ( get_magic_quotes_gpc() ) { 
  710.             $params = stripslashes_deep( $params ); 
  711.         } 
  712.  
  713.         /** 
  714.          * Add to the POST parameters stored internally. If a user has already 
  715.          * set these manually (via `set_body_params`), don't override them. 
  716.          */ 
  717.         $this->params['POST'] = array_merge( $params, $this->params['POST'] ); 
  718.     }

  770.     /** 
  771.      * Sanitizes (where possible) the params on the request. 
  772.      * 
  773.      * This is primarily based off the sanitize_callback param on each registered 
  774.      * argument. 
  775.      * 
  776.      * @since 4.4.0 
  777.      * @access public 
  778.      * 
  779.      * @return true|WP_Error True if parameters were sanitized, WP_Error if an error occurred during sanitization. 
  780.      */ 
  781.     public function sanitize_params() { 
  782.         $attributes = $this->get_attributes(); 
  783.  
  784.         // No arguments set, skip sanitizing. 
  785.         if ( empty( $attributes['args'] ) ) { 
  786.             return true; 
  787.         } 
  788.  
  789.         $order = $this->get_parameter_order(); 
  790.  
  791.         $invalid_params = array(); 
  792.  
  793.         foreach ( $order as $type ) { 
  794.             if ( empty( $this->params[ $type ] ) ) { 
  795.                 continue; 
  796.             } 
  797.             foreach ( $this->params[ $type ] as $key => $value ) { 
  798.                 if ( ! isset( $attributes['args'][ $key ] ) ) { 
  799.                     continue; 
  800.                 } 
  801.                 $param_args = $attributes['args'][ $key ]; 
  802.  
  803.                 // If the arg has a type but no sanitize_callback attribute, default to rest_parse_request_arg. 
  804.                 if ( ! array_key_exists( 'sanitize_callback', $param_args ) && ! empty( $param_args['type'] ) ) { 
  805.                     $param_args['sanitize_callback'] = 'rest_parse_request_arg'; 
  806.                 } 
  807.                 // If there's still no sanitize_callback, nothing to do here. 
  808.                 if ( empty( $param_args['sanitize_callback'] ) ) { 
  809.                     continue; 
  810.                 } 
  811.  
  812.                 $sanitized_value = call_user_func( $param_args['sanitize_callback'], $value, $this, $key ); 
  813.  
  814.                 if ( is_wp_error( $sanitized_value ) ) { 
  815.                     $invalid_params[ $key ] = $sanitized_value->get_error_message(); 
  816.                 } else { 
  817.                     $this->params[ $type ][ $key ] = $sanitized_value; 
  818.                 } 
  819.             } 
  820.         } 
  821.  
  822.         if ( $invalid_params ) { 
  823.             return new WP_Error( 'rest_invalid_param', sprintf( __( 'Invalid parameter(s): %s' ), implode( ', ', array_keys( $invalid_params ) ) ), array( 'status' => 400, 'params' => $invalid_params ) ); 
  824.         } 
  825.  
  826.         return true; 
  827.     } 
  828.  
  829.     /** 
  830.      * Checks whether this request is valid according to its attributes. 
  831.      * 
  832.      * @since 4.4.0 
  833.      * @access public 
  834.      * 
  835.      * @return bool|WP_Error True if there are no parameters to validate or if all pass validation,  
  836.      *                       WP_Error if required parameters are missing. 
  837.      */ 
  838.     public function has_valid_params() { 
  839.         // If JSON data was passed, check for errors. 
  840.         $json_error = $this->parse_json_params(); 
  841.         if ( is_wp_error( $json_error ) ) { 
  842.             return $json_error; 
  843.         } 
  844.  
  845.         $attributes = $this->get_attributes(); 
  846.         $required = array(); 
  847.  
  848.         // No arguments set, skip validation. 
  849.         if ( empty( $attributes['args'] ) ) { 
  850.             return true; 
  851.         } 
  852.  
  853.         foreach ( $attributes['args'] as $key => $arg ) { 
  854.  
  855.             $param = $this->get_param( $key ); 
  856.             if ( isset( $arg['required'] ) && true === $arg['required'] && null === $param ) { 
  857.                 $required[] = $key; 
  858.             } 
  859.         } 
  860.  
  861.         if ( ! empty( $required ) ) { 
  862.             return new WP_Error( 'rest_missing_callback_param', sprintf( __( 'Missing parameter(s): %s' ), implode( ', ', $required ) ), array( 'status' => 400, 'params' => $required ) ); 
  863.         } 
  864.  
  865.         /** 
  866.          * Check the validation callbacks for each registered arg. 
  867.          * 
  868.          * This is done after required checking as required checking is cheaper. 
  869.          */ 
  870.         $invalid_params = array(); 
  871.  
  872.         foreach ( $attributes['args'] as $key => $arg ) { 
  873.  
  874.             $param = $this->get_param( $key ); 
  875.  
  876.             if ( null !== $param && ! empty( $arg['validate_callback'] ) ) { 
  877.                 $valid_check = call_user_func( $arg['validate_callback'], $param, $this, $key ); 
  878.  
  879.                 if ( false === $valid_check ) { 
  880.                     $invalid_params[ $key ] = __( 'Invalid parameter.' ); 
  881.                 } 
  882.  
  883.                 if ( is_wp_error( $valid_check ) ) { 
  884.                     $invalid_params[ $key ] = $valid_check->get_error_message(); 
  885.                 } 
  886.             } 
  887.         } 
  888.  
  889.         if ( $invalid_params ) { 
  890.             return new WP_Error( 'rest_invalid_param', sprintf( __( 'Invalid parameter(s): %s' ), implode( ', ', array_keys( $invalid_params ) ) ), array( 'status' => 400, 'params' => $invalid_params ) ); 
  891.         } 
  892.  
  893.         return true; 
  894.  
  895.     }

  897.     /** 
  898.      * Checks if a parameter is set. 
  899.      * 
  900.      * @since 4.4.0 
  901.      * @access public 
  902.      * 
  903.      * @param string $offset Parameter name. 
  904.      * @return bool Whether the parameter is set. 
  905.      */ 
  906.     public function offsetExists( $offset ) {

  911.                 return true; 
  912.             } 
  913.         }

  916.     }

  961.     /** 
  962.      * Retrieves a WP_REST_Request object from a full URL. 
  963.      * 
  964.      * @static 
  965.      * @since 4.5.0 
  966.      * @access public 
  967.      * 
  968.      * @param string $url URL with protocol, domain, path and query args. 
  969.      * @return WP_REST_Request|false WP_REST_Request object on success, false on failure. 
  970.      */ 
  971.     public static function from_url( $url ) { 
  972.         $bits = parse_url( $url ); 
  973.         $query_params = array(); 
  974.  
  975.         if ( ! empty( $bits['query'] ) ) { 
  976.             wp_parse_str( $bits['query'], $query_params ); 
  977.         } 
  978.  
  979.         $api_root = rest_url(); 
  980.         if ( get_option( 'permalink_structure' ) && 0 === strpos( $url, $api_root ) ) { 
  981.             // Pretty permalinks on, and URL is under the API root. 
  982.             $api_url_part = substr( $url, strlen( untrailingslashit( $api_root ) ) ); 
  983.             $route = parse_url( $api_url_part, PHP_URL_PATH ); 
  984.         } elseif ( ! empty( $query_params['rest_route'] ) ) { 
  985.             // ?rest_route=... set directly 
  986.             $route = $query_params['rest_route']; 
  987.             unset( $query_params['rest_route'] ); 
  988.         } 
  989.  
  990.         $request = false; 
  991.         if ( ! empty( $route ) ) { 
  992.             $request = new WP_REST_Request( 'GET', $route ); 
  993.             $request->set_query_params( $query_params ); 
  994.         } 
  995.  
  996.         /** 
  997.          * Filters the request generated from a URL. 
  998.          * 
  999.          * @since 4.5.0 
  1000.          * 
  1001.          * @param WP_REST_Request|false $request Generated request object, or false if URL 
  1002.          *                                       could not be parsed. 
  1003.          * @param string                $url     URL the request was generated from. 
  1004.          */ 
  1005.         return apply_filters( 'rest_request_from_url', $request, $url ); 
  1006.     }
