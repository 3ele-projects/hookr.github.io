  524.     /** 
  525.      * Appends tax queries to an array. 
  526.      * @param array $tax_query 
  527.      * @param bool  $main_query 
  528.      * @return array 
  529.      */ 
  530.     public function get_tax_query( $tax_query = array(), $main_query = false ) { 
  531.         if ( ! is_array( $tax_query ) ) { 
  532.             $tax_query = array( 'relation' => 'AND' ); 
  533.         } 
  534.  
  535.         // Layered nav filters on terms. 
  536.         if ( $main_query && ( $_chosen_attributes = $this->get_layered_nav_chosen_attributes() ) ) { 
  537.             foreach ( $_chosen_attributes as $taxonomy => $data ) { 
  538.                 $tax_query[] = array( 
  539.                     'taxonomy' => $taxonomy,  
  540.                     'field' => 'slug',  
  541.                     'terms' => $data['terms'],  
  542.                     'operator' => 'and' === $data['query_type'] ? 'AND' : 'IN',  
  543.                     'include_children' => false,  
  544.  ); 
  545.             } 
  546.         } 
  547.  
  548.         $product_visibility_terms = wc_get_product_visibility_term_ids(); 
  549.         $product_visibility_not_in = array( is_search() && $main_query ? $product_visibility_terms['exclude-from-search'] : $product_visibility_terms['exclude-from-catalog'] ); 
  550.  
  551.         // Hide out of stock products. 
  552.         if ( 'yes' === get_option( 'woocommerce_hide_out_of_stock_items' ) ) { 
  553.             $product_visibility_not_in[] = $product_visibility_terms['outofstock']; 
  554.         } 
  555.  
  556.         // Filter by rating. 
  557.         if ( isset( $_GET['rating_filter'] ) ) { 
  558.             $rating_filter = array_filter( array_map( 'absint', explode( ', ', $_GET['rating_filter'] ) ) ); 
  559.             $rating_terms = array(); 
  560.             for ( $i = 1; $i <= 5; $i ++ ) { 
  561.                 if ( in_array( $i, $rating_filter ) && isset( $product_visibility_terms[ 'rated-' . $i ] ) ) { 
  562.                     $rating_terms[] = $product_visibility_terms[ 'rated-' . $i ]; 
  563.                 } 
  564.             } 
  565.             if ( ! empty( $rating_terms ) ) { 
  566.                 $tax_query[] = array( 
  567.                     'taxonomy' => 'product_visibility',  
  568.                     'field' => 'term_taxonomy_id',  
  569.                     'terms' => $rating_terms,  
  570.                     'operator' => 'IN',  
  571.                     'rating_filter' => true,  
  572.  ); 
  573.             } 
  574.         } 
  575.  
  576.         if ( ! empty( $product_visibility_not_in ) ) { 
  577.             $tax_query[] = array( 
  578.                 'taxonomy' => 'product_visibility',  
  579.                 'field' => 'term_taxonomy_id',  
  580.                 'terms' => $product_visibility_not_in,  
  581.                 'operator' => 'NOT IN',  
  582.  ); 
  583.         } 
  584.  
  585.         return array_filter( apply_filters( 'woocommerce_product_query_tax_query', $tax_query, $this ) ); 
  586.     }

  660.     /** 
  661.      * Based on WP_Query::parse_search 
  662.      */ 
  663.     public static function get_main_search_query_sql() { 
  664.         global $wp_the_query, $wpdb; 
  665.  
  666.         $args = $wp_the_query->query_vars; 
  667.         $search_terms = isset( $args['search_terms'] ) ? $args['search_terms'] : array(); 
  668.         $sql = array(); 
  669.  
  670.         foreach ( $search_terms as $term ) { 
  671.             // Terms prefixed with '-' should be excluded. 
  672.             $include = '-' !== substr( $term, 0, 1 ); 
  673.  
  674.             if ( $include ) { 
  675.                 $like_op = 'LIKE'; 
  676.                 $andor_op = 'OR'; 
  677.             } else { 
  678.                 $like_op = 'NOT LIKE'; 
  679.                 $andor_op = 'AND'; 
  680.                 $term = substr( $term, 1 ); 
  681.             } 
  682.  
  683.             $like = '%' . $wpdb->esc_like( $term ) . '%'; 
  684.             $sql[] = $wpdb->prepare( "(($wpdb->posts.post_title $like_op %s) $andor_op ($wpdb->posts.post_excerpt $like_op %s) $andor_op ($wpdb->posts.post_content $like_op %s))", $like, $like, $like ); 
  685.         } 
  686.  
  687.         if ( ! empty( $sql ) && ! is_user_logged_in() ) { 
  688.             $sql[] = "($wpdb->posts.post_password = '')"; 
  689.         } 
  690.  
  691.         return implode( ' AND ', $sql ); 
  692.     }

  695.      * Layered Nav Init. 
  696.      */ 
  697.     public static function get_layered_nav_chosen_attributes() { 
  698.         if ( ! is_array( self::$_chosen_attributes ) ) { 
  699.             self::$_chosen_attributes = array(); 
  700.  
  701.             if ( $attribute_taxonomies = wc_get_attribute_taxonomies() ) { 
  702.                 foreach ( $attribute_taxonomies as $tax ) { 
  703.                     $attribute = wc_sanitize_taxonomy_name( $tax->attribute_name ); 
  704.                     $taxonomy = wc_attribute_taxonomy_name( $attribute ); 
  705.                     $filter_terms = ! empty( $_GET[ 'filter_' . $attribute ] ) ? explode( ', ', wc_clean( $_GET[ 'filter_' . $attribute ] ) ) : array(); 
  706.  
  707.                     if ( empty( $filter_terms ) || ! taxonomy_exists( $taxonomy ) ) { 
  708.                         continue; 
  709.                     } 
  710.  
  711.                     $query_type = ! empty( $_GET[ 'query_type_' . $attribute ] ) && in_array( $_GET[ 'query_type_' . $attribute ], array( 'and', 'or' ) ) ? wc_clean( $_GET[ 'query_type_' . $attribute ] ) : ''; 
  712.                     self::$_chosen_attributes[ $taxonomy ]['terms'] = array_map( 'sanitize_title', $filter_terms ); // Ensures correct encoding 
  713.                     self::$_chosen_attributes[ $taxonomy ]['query_type'] = $query_type ? $query_type : apply_filters( 'woocommerce_layered_nav_default_query_type', 'and' );
