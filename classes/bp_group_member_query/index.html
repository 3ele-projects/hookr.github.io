  219.     public function set_orderby( $query ) { 
  220.         $gm_ids = $this->get_group_member_ids(); 
  221.         if ( empty( $gm_ids ) ) { 
  222.             $gm_ids = array( 0 ); 
  223.         } 
  224.  
  225.         // For 'last_joined', 'first_joined', and 'group_activity' 
  226.         // types, we override the default orderby clause of 
  227.         // BP_User_Query. In the case of 'group_activity', we perform 
  228.         // a separate query to get the necessary order. In the case of 
  229.         // 'last_joined' and 'first_joined', we can trust the order of 
  230.         // results from  BP_Group_Member_Query::get_group_members(). 
  231.         // In all other cases, we fall through and let BP_User_Query 
  232.         // do its own (non-group-specific) ordering. 
  233.         if ( in_array( $query->query_vars['type'], array( 'last_joined', 'first_joined', 'group_activity' ) ) ) { 
  234.  
  235.             // Group Activity DESC. 
  236.             if ( 'group_activity' == $query->query_vars['type'] ) { 
  237.                 $gm_ids = $this->get_gm_ids_ordered_by_activity( $query, $gm_ids ); 
  238.             } 
  239.  
  240.             // The first param in the FIELD() clause is the sort column id. 
  241.             $gm_ids = array_merge( array( 'u.id' ), wp_parse_id_list( $gm_ids ) );
