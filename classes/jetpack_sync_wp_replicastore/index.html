  594.     public function checksum_all() { 
  595.         $post_meta_checksum = $this->checksum_histogram( 'post_meta', 1 ); 
  596.         $comment_meta_checksum = $this->checksum_histogram( 'comment_meta', 1 ); 
  597.  
  598.         return array( 
  599.             'posts' => $this->posts_checksum(),  
  600.             'comments' => $this->comments_checksum(),  
  601.             'post_meta'=> reset( $post_meta_checksum ),  
  602.             'comment_meta'=> reset( $comment_meta_checksum ),  
  603.  );

  606.     function checksum_histogram( $object_type, $buckets, $start_id = null, $end_id = null, $columns = null, $strip_non_ascii = true ) { 
  607.         global $wpdb; 
  608.  
  609.         $wpdb->queries = array(); 
  610.  
  611.         switch( $object_type ) { 
  612.             case "posts": 
  613.                 $object_count = $this->post_count( null, $start_id, $end_id ); 
  614.                 $object_table = $wpdb->posts; 
  615.                 $id_field = 'ID'; 
  616.                 $where_sql = Jetpack_Sync_Settings::get_blacklisted_post_types_sql(); 
  617.                 if ( empty( $columns ) ) { 
  618.                     $columns = Jetpack_Sync_Defaults::$default_post_checksum_columns; 
  619.                 } 
  620.                 break; 
  621.             case "post_meta": 
  622.                 $object_table = $wpdb->postmeta; 
  623.                 $where_sql = Jetpack_Sync_Settings::get_whitelisted_post_meta_sql(); 
  624.                 $object_count = $this->meta_count( $object_table, $where_sql, $start_id, $end_id ); 
  625.                 $id_field = 'meta_id'; 
  626.                  
  627.                 if ( empty( $columns ) ) { 
  628.                     $columns = Jetpack_Sync_Defaults::$default_post_meta_checksum_columns; 
  629.                 } 
  630.                 break; 
  631.             case "comments": 
  632.                 $object_count = $this->comment_count( null, $start_id, $end_id ); 
  633.                 $object_table = $wpdb->comments; 
  634.                 $id_field = 'comment_ID'; 
  635.                 $where_sql = Jetpack_Sync_Settings::get_comments_filter_sql(); 
  636.                 if ( empty( $columns ) ) { 
  637.                     $columns = Jetpack_Sync_Defaults::$default_comment_checksum_columns; 
  638.                 } 
  639.                 break; 
  640.             case "comment_meta": 
  641.                 $object_table = $wpdb->commentmeta; 
  642.                 $where_sql = Jetpack_Sync_Settings::get_whitelisted_comment_meta_sql(); 
  643.                 $object_count = $this->meta_count( $object_table, $where_sql, $start_id, $end_id ); 
  644.                 $id_field = 'meta_id'; 
  645.                 if ( empty( $columns ) ) { 
  646.                     $columns = Jetpack_Sync_Defaults::$default_post_meta_checksum_columns; 
  647.                 } 
  648.                 break; 
  649.             default: 
  650.                 return false; 
  651.         } 
  652.  
  653.         $bucket_size = intval( ceil( $object_count / $buckets ) ); 
  654.         $previous_max_id = 0; 
  655.         $histogram = array(); 
  656.  
  657.         $where = '1=1'; 
  658.  
  659.         if ( $start_id ) { 
  660.             $where .= " AND $id_field >= " . intval( $start_id ); 
  661.         } 
  662.  
  663.         if ( $end_id ) { 
  664.             $where .= " AND $id_field <= " . intval( $end_id ); 
  665.         } 
  666.  
  667.         do { 
  668.             list( $first_id, $last_id ) = $wpdb->get_row( 
  669.                 "SELECT MIN($id_field) as min_id, MAX($id_field) as max_id FROM ( SELECT $id_field FROM $object_table WHERE $where AND $id_field > $previous_max_id ORDER BY $id_field ASC LIMIT $bucket_size ) as ids",  
  670.                 ARRAY_N 
  671.  ); 
  672.  
  673.             // get the checksum value 
  674.             $value = $this->table_checksum( $object_table, $columns, $id_field, $where_sql, $first_id, $last_id, $strip_non_ascii ); 
  675.  
  676.             if ( is_wp_error( $value ) ) { 
  677.                 return $value; 
  678.             } 
  679.  
  680.             if ( $first_id === null || $last_id === null ) { 
  681.                 break; 
  682.             } elseif ( $first_id === $last_id ) { 
  683.                 $histogram[ $first_id ] = $value; 
  684.             } else { 
  685.                 $histogram[ "{$first_id}-{$last_id}" ] = $value; 
  686.             } 
  687.  
  688.             $previous_max_id = $last_id; 
  689.         } while ( true ); 
  690.  
  691.         return $histogram; 
  692.     }

  694.     private function table_checksum( $table, $columns, $id_column, $where_sql = '1=1', $min_id = null, $max_id = null, $strip_non_ascii = true ) { 
  695.         global $wpdb; 
  696.  
  697.         // sanitize to just valid MySQL column names 
  698.         $sanitized_columns = preg_grep ( '/^[0-9, a-z, A-Z$_]+$/i', $columns ); 
  699.  
  700.         if ( $strip_non_ascii ) { 
  701.             $columns_sql = implode( ', ', array_map( array( $this, 'strip_non_ascii_sql' ), $sanitized_columns ) ); 
  702.         } else { 
  703.             $columns_sql = implode( ', ', $sanitized_columns ); 
  704.         } 
  705.  
  706.         if ( $min_id !== null ) { 
  707.             $min_id = intval( $min_id ); 
  708.             $where_sql .= " AND $id_column >= $min_id"; 
  709.         } 
  710.  
  711.         if ( $max_id !== null ) { 
  712.             $max_id = intval( $max_id ); 
  713.             $where_sql .= " AND $id_column <= $max_id"; 
  714.         } 
  715.  
  716.         $query = <<<ENDSQL 
  717.             SELECT CONV(BIT_XOR(CRC32(CONCAT({$columns_sql}))), 10, 16) 
  718.                 FROM $table 
  719.                 WHERE $where_sql 
  720. ENDSQL; 
  721.         $result = $wpdb->get_var( $query ); 
  722.  
  723.         if ( $wpdb->last_error ) { 
  724.             return new WP_Error( 'database_error', $wpdb->last_error ); 
  725.         } 
  726.  
  727.         return $result; 
  728.  
  729.     }

  731.     private function meta_count( $table, $where_sql, $min_id, $max_id ) { 
  732.         global $wpdb;

  734.         if ( $min_id != null ) {

  745.     /** 
  746.      * Wraps a column name in SQL which strips non-ASCII chars. 
  747.      * This helps normalize data to avoid checksum differences caused by 
  748.      * badly encoded data in the DB 
  749.      */ 
  750.     function strip_non_ascii_sql( $column_name ) { 
  751.         return "REPLACE( CONVERT( $column_name USING ascii ), '?', '' )";
