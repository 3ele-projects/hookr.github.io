  687.     public function admin_ajax_upload () {

  689.  
  690.         // Sanitize input 
  691.         $params = pods_unslash( (array) $_POST ); 
  692.  
  693.         foreach ( $params as $key => $value ) { 
  694.             if ( 'action' == $key ) 
  695.                 continue; 
  696.  
  697.             unset( $params[ $key ] ); 
  698.  
  699.             $params[ str_replace( '_podsfix_', '', $key ) ] = $value; 
  700.         } 
  701.  
  702.         $params = (object) $params; 
  703.  
  704.         $methods = array( 
  705.             'upload',  
  706.  );

  709.             pods_error( 'Invalid AJAX request', PodsInit::$admin );

  711.             pods_error( 'Invalid AJAX request', PodsInit::$admin );

  713.             pods_error( 'Invalid AJAX request', PodsInit::$admin ); 
  714.  
  715.         // Flash often fails to send cookies with the POST or upload, so we need to pass it in GET or POST instead 
  716.         if ( is_ssl() && empty( $_COOKIE[ SECURE_AUTH_COOKIE ] ) && !empty( $_REQUEST[ 'auth_cookie' ] ) ) 
  717.             $_COOKIE[ SECURE_AUTH_COOKIE ] = $_REQUEST[ 'auth_cookie' ]; 
  718.         elseif ( empty( $_COOKIE[ AUTH_COOKIE ] ) && !empty( $_REQUEST[ 'auth_cookie' ] ) ) 
  719.             $_COOKIE[ AUTH_COOKIE ] = $_REQUEST[ 'auth_cookie' ];

  724.         global $current_user; 
  725.         unset( $current_user );

  727.         /** 
  728.          * Access Checking 
  729.          */

  731.  
  732.         if ( defined( 'PODS_DISABLE_FILE_UPLOAD' ) && true === PODS_DISABLE_FILE_UPLOAD )

  737.             $upload_disabled = true;

  741.         if ( is_user_logged_in() ) 
  742.             $uid = 'user_' . get_current_user_id(); 
  743.  
  744.         $nonce_check = 'pods_upload_' . (int) $params->pod . '_' . $uid . '_' . $params->uri . '_' . (int) $params->field; 
  745.

  747.             pods_error( __( 'Unauthorized request', 'pods' ), PodsInit::$admin );

  751.             'type' => 'file',  
  752.             'options' => array()

  761.             $field = $api->load_field( array( 'id' => (int) $params->field ) ); 
  762.  
  763.             if ( empty( $pod ) || empty( $field ) || $pod[ 'id' ] != $field[ 'pod_id' ] || !isset( $pod[ 'fields' ][ $field[ 'name' ] ] ) ) 
  764.                 pods_error( __( 'Invalid field request', 'pods' ), PodsInit::$admin ); 
  765.  
  766.             if ( !in_array( $field[ 'type' ], PodsForm::file_field_types() ) ) 
  767.                 pods_error( __( 'Invalid field', 'pods' ), PodsInit::$admin );

  769.

  771.

  774.         unset( $params->method );

  799.                 } 
  800.                 elseif ( false !== stripos( $limit_size, 'B' ) )

  802.                 else
