  155. function colorway_import_file($file, $post_id = 0, $import_date = 'file') { 
  156.   set_time_limit(120); 
  157.  
  158.   // Initially, Base it on the -current- time. 
  159.   $time = current_time('mysql', 1); 
  160. //     Next, If it's post to base the upload off: 
  161.  
  162.   $time = gmdate('Y-m-d H:i:s', @filemtime($file)); 
  163.  
  164.  
  165. //     A writable uploads dir will pass this test. Again, there's no point overriding this one. 
  166.   if (!( ( $uploads = wp_upload_dir($time) ) && false === $uploads['error'] )) { 
  167.       return new WP_Error('upload_error', $uploads['error']); 
  168.   } 
  169.  
  170.   $wp_filetype = wp_check_filetype($file, null); 
  171.  
  172.   extract($wp_filetype); 
  173.  
  174.   if ((!$type || !$ext ) && !current_user_can('unfiltered_upload')) { 
  175.       return new WP_Error('wrong_file_type', __('Sorry, this file type is not permitted for security reasons.', 'colorway')); //A WP-core string.. 
  176.   } 
  177.  
  178.   $file_name = str_replace('\\', '/', $file); 
  179.  
  180.   if (preg_match('|^' . preg_quote(str_replace('\\', '/', $uploads['basedir'])) . '(.*)$|i', $file_name, $mat)) { 
  181.       $filename = basename($file); 
  182.       $new_file = $file; 
  183.       $url = $uploads['baseurl'] . $mat[1]; 
  184.       $attachment = get_posts(array('post_type' => 'attachment', 'meta_key' => '_wp_attached_file', 'meta_value' => ltrim($mat[1], '/'))); 
  185.       if (!empty($attachment)) { 
  186.           return new WP_Error('file_exists', __('Sorry, That file already exists in the WordPress media library.', 'colorway')); 
  187.       } 
  188.  
  189.       //Ok, Its in the uploads folder, But NOT in WordPress's media library. 
  190.       if ('file' == $import_date) { 
  191.           $time = @filemtime($file); 
  192.           if (preg_match("|(\d+)/(\d+)|", $mat[1], $datemat)) { //So lets set the date of the import to the date folder its in, IF its in a date folder. 
  193.               $hour = $min = $sec = 0; 
  194.               $day = 1; 
  195.               $year = $datemat[1]; 
  196.               $month = $datemat[2]; 
  197.  
  198.               // If the files datetime is set, and it's in the same region of upload directory, set the minute details to that too, else, override it. 
  199.               if ($time && date('Y-m', $time) == "$year-$month") { 
  200.                   list($hour, $min, $sec, $day) = explode(';', date('H;i;s;j', $time)); 
  201.               } 
  202.               $time = mktime($hour, $min, $sec, $month, $day, $year); 
  203.           } 
  204.           $time = gmdate('Y-m-d H:i:s', $time); 
  205.  
  206.           // A new time has been found! Get the new uploads folder: 
  207.           // A writable uploads dir will pass this test. Again, there's no point overriding this one. 
  208.           if (!( ( $uploads = wp_upload_dir($time) ) && false === $uploads['error'] )) 
  209.               return new WP_Error('upload_error', $uploads['error']); 
  210.           $url = $uploads['baseurl'] . $mat[1]; 
  211.       } 
  212.   } else { 
  213.       $filename = wp_unique_filename($uploads['path'], basename($file)); 
  214.       // copy the file to the uploads dir 
  215.       $new_file = $uploads['path'] . '/' . $filename; 
  216.       if (false === @copy($file, $new_file)) 
  217.           return new WP_Error('upload_error', sprintf(__('The selected file could not be copied to %s.', 'colorway'), $uploads['path'])); 
  218.  
  219.       // Set correct file permissions 
  220.       $stat = stat(dirname($new_file)); 
  221.       $perms = $stat['mode'] & 0000666; 
  222.       @ chmod($new_file, $perms); 
  223.       // Compute the URL 
  224.       $url = $uploads['url'] . '/' . $filename; 
  225.  
  226.       if ('file' == $import_date) 
  227.           $time = gmdate('Y-m-d H:i:s', @filemtime($file)); 
  228.   } 
  229.  
  230.   //Apply upload filters 
  231.   $return = apply_filters('wp_handle_upload', array('file' => $new_file, 'url' => $url, 'type' => $type)); 
  232.   $new_file = $return['file']; 
  233.   $url = $return['url']; 
  234.   $type = $return['type']; 
  235.  
  236.   $title = preg_replace('!\.[^.]+$!', '', basename($file)); 
  237.   $content = ''; 
  238.  
  239.   if ($time) { 
  240.       $post_date_gmt = $time; 
  241.       $post_date = $time; 
  242.   } else { 
  243.       $post_date = current_time('mysql'); 
  244.       $post_date_gmt = current_time('mysql', 1); 
  245.   } 
  246.  
  247.   // Construct the attachment array 
  248.   $attachment = array( 
  249.       'post_mime_type' => $type,  
  250.       'guid' => $url,  
  251.       'post_parent' => $post_id,  
  252.       'post_title' => $title,  
  253.       'post_name' => $title,  
  254.       'post_content' => $content,  
  255.       'post_date' => $post_date,  
  256.       'post_date_gmt' => $post_date_gmt 
  257.  ); 
  258.  
  259.   $attachment = apply_filters('afs-import_details', $attachment, $file, $post_id, $import_date); 
  260.  
  261.   //Win32 fix: 
  262.   $new_file = str_replace(strtolower(str_replace('\\', '/', $uploads['basedir'])), $uploads['basedir'], $new_file); 
  263.  
  264.   // Save the data 
  265.  
  266.   $id = wp_insert_attachment($attachment, $new_file, $post_id); 
  267.   if (!is_wp_error($id)) { 
  268.       $data = wp_generate_attachment_metadata($id, $new_file); 
  269.       wp_update_attachment_metadata($id, $data); 
  270.   } 
  271.   //update_post_meta( $id, '_wp_attached_file', $uploads['subdir'] . '/' . $filename ); 
  272.  
  273.   return $id; 
  274. }
