  93.   private function do_envato_api_call( $endpoint = 'market/total-items.json', $query_string = NULL , $method = 'get', $timeout = 5 ) { 
  94.  
  95.       $default_query_string = 'page_size=100&sort_by=updated&sort_direction=desc'; 
  96.  
  97.       $query_string = ( $query_string ? '?' . $query_string . '&' . $default_query_string : '?' . $default_query_string ); 
  98.  
  99.       // Set the remote URL 
  100.       $remote_url = self::ENVATO_API_URL . $endpoint . $query_string; 
  101.  
  102.       // Set the query transient key 
  103.       $cache_key = 'lmp_' . $query_string; 
  104.  
  105.       // Quick cache dumper 
  106.       $dump_cache = 0; 
  107.       if( 1 == $dump_cache ) delete_transient( $cache_key ); 
  108.  
  109.       // Return a cached version of the query if we have one 
  110.       if( FALSE !== get_transient( $cache_key ) ) { 
  111.           return get_transient( $cache_key ); 
  112.       } 
  113.  
  114.       // Set the Auth token for our query 
  115.       $remote_args = array( 
  116.               'timeout' => $timeout,  
  117.               'headers' => array( 
  118.                   'Authorization' => 'Bearer ' . $this->get_auth_token() 
  119.  ) 
  120.  ); 
  121.  
  122.       // Choose a method 
  123.       if( 'get' == $method ) { 
  124.           $remote_query = wp_remote_get( $remote_url, $remote_args ); 
  125.       } else { 
  126.           $remote_query = wp_remote_post( $remote_url, $remote_args ); 
  127.       } 
  128.  
  129.       if( is_wp_error( $remote_query ) ) { 
  130.  
  131.           // If there's an error, we handle it on the front end so just return it 
  132.           return $remote_query; 
  133.  
  134.       } else if( isset( $remote_query[ 'response' ][ 'code' ] ) && 200 == $remote_query[ 'response' ][ 'code' ] ) { 
  135.  
  136.           // Cache a successful query 
  137.           set_transient( $cache_key , wp_remote_retrieve_body( $remote_query ), 60 ); 
  138.  
  139.           return wp_remote_retrieve_body( $remote_query ); 
  140.       } else { 
  141.  
  142.           // If the response code isn't right, throw an error 
  143.           return new WP_Error( __( 'Error' , 'layerswp' ) , __( 'Something broke and we can\'t load the stream' , 'layerswp' ) ); 
  144.       } 
  145.   }
