  974.     /** 
  975.      * Merge Compontent 
  976.      */ 
  977.     public function merge_component( $defaults, $args ) { 
  978.  
  979.         // Grab the elements and unset them - so we can work with them individually. 
  980.         $defaults_elements = isset( $defaults['elements'] ) ? $defaults['elements'] : array() ; 
  981.         if ( isset( $defaults['elements'] ) ) unset( $defaults['elements'] ); 
  982.  
  983.         $args_elements = isset( $args['elements'] ) ? $args['elements'] : array() ; 
  984.         if ( isset( $args['elements'] ) ) unset( $args['elements'] ); 
  985.  
  986.         // New collection of elements consisting of a specific combo of the $defaults and the $args. 
  987.         $new_elements = array(); 
  988.  
  989.         foreach ( $args_elements as $args_key => $args_value ) { 
  990.  
  991.             if ( is_string( $args_value ) && isset( $defaults_elements[ $args_value ] ) ) { 
  992.  
  993.                 // This case means the caller has specified a custom $args 'elements' config 
  994.                 // but has only passed a ref to the input by it's 'string 'background-image' 
  995.                 // allowing them to reposition the input without redefining all the settings 
  996.                 // the input. 
  997.                 $new_elements[ $args_value ] = $defaults_elements[ $args_value ]; 
  998.  
  999.                 // We've got what we needed from this element so remove it from the reference array. 
  1000.                 if ( isset( $defaults_elements[ $args_key ] ) ) { 
  1001.                     unset( $defaults_elements[ $args_value ] ); 
  1002.                 } 
  1003.             } 
  1004.             else if( isset( $defaults_elements[ $args_key ] ) && is_array( $defaults_elements[ $args_key ] ) && is_array( $args_elements[ $args_key ] ) ) { 
  1005.  
  1006.                 // This case means the caller intends to combine the defaults with new 
  1007.                 // parameters, keeping the existing fields but adding new things to it 
  1008.                 $new_elements[ $args_key ] =  $args_elements[ $args_key ] + $defaults_elements[ $args_key ]; 
  1009.             } 
  1010.             else if ( is_array( $args_value ) ) { 
  1011.  
  1012.                 // This case means the caller has specified a custom $args 'elements' config 
  1013.                 // and has specified their own custom input field config - allowing them to 
  1014.                 // create a new custom field. 
  1015.                 $new_elements[ $args_key ] = $args_value; 
  1016.  
  1017.                 // We've got what we needed from this element so remove it from the reference array. 
  1018.                 if ( isset( $defaults_elements[ $args_key ] ) ) { 
  1019.                     unset( $defaults_elements[ $args_key ] ); 
  1020.                 } 
  1021.             } 
  1022.         } 
  1023.  
  1024.         // This handles merging the important non-elements like 'icon-css' and 'title' 
  1025.         $args = array_merge( $defaults, $args ); 
  1026.  
  1027.         // Either 'replace' or 'merge' the new input - so either show only the ones you have chosen 
  1028.         // or show the ones you have chosen after the defaults of the component. 
  1029.         if ( isset( $args['elements_combine'] ) && 'replace' === $args['elements_combine'] ) { 
  1030.             $args['elements'] = $new_elements; 
  1031.         } 
  1032.         else{ // 'merge' or anything else. 
  1033.             $args['elements'] = array_merge( $defaults_elements, $new_elements ); 
  1034.         } 
  1035.  
  1036.         return $args; 
  1037.     }
