  26. function responsive_mobile_get_breadcrumb_lists() { 
  27.   $responsive_mobile_options = get_option( 'responsive_mobile_theme_options' ); 
  28.   $yoast_options = get_option( 'wpseo_internallinks' ); 
  29.   if ( 1 == $responsive_mobile_options['breadcrumb'] ) { 
  30.       return; 
  31.   } elseif ( function_exists( 'bcn_display' ) ) { 
  32.       bcn_display(); 
  33.   } elseif ( function_exists( 'breadcrumb_trail' ) ) { 
  34.       breadcrumb_trail(); 
  35.   } elseif ( function_exists( 'yoast_breadcrumb' ) && true === $yoast_options['breadcrumbs-enable'] ) { 
  36.       yoast_breadcrumb( '<p id="breadcrumbs">', '</p>' ); 
  37.   } elseif ( ! is_search() ) { 
  38.       responsive_mobile_breadcrumb_lists(); 
  39.   }

  42. /** 
  43.  * Breadcrumb Lists 
  44.  * Allows visitors to quickly navigate back to a previous section or the root page. 
  45.  * 
  46.  * Adopted from Dimox 
  47.  * 
  48.  */ 
  49. if ( !function_exists( 'responsive_mobile_breadcrumb_lists' ) ) { 
  50.  
  51.   function responsive_mobile_breadcrumb_lists() { 
  52.  
  53.       /** === OPTIONS === */ 
  54.       $text['home'] = __( 'Home', 'responsive-mobile' ); // text for the 'Home' link 
  55.       $text['category'] = __( 'Archive for %s', 'responsive-mobile' ); // text for a category page 
  56.       $text['search'] = __( 'Search results for: %s', 'responsive-mobile' ); // text for a search results page 
  57.       $text['tag'] = __( 'Posts tagged %s', 'responsive-mobile' ); // text for a tag page 
  58.       $text['author'] = __( 'View all posts by %s', 'responsive-mobile' ); // text for an author page 
  59.       $text['404'] = __( 'Error 404', 'responsive-mobile' ); // text for the 404 page 
  60.  
  61.       $show['current'] = 1; // 1 - show current post/page title in breadcrumbs, 0 - don't show 
  62.       $show['home'] = 0; // 1 - show breadcrumbs on the homepage, 0 - don't show 
  63.       $show['search'] = 0; // 1 - show breadcrumbs on the search page, 0 - don't show 
  64.  
  65.       $delimiter = ' <span class="chevron">›</span> '; // delimiter between crumbs 
  66.       $before = '<span class="breadcrumb-current">'; // tag before the current crumb 
  67.       $after = '</span>'; // tag after the current crumb 
  68.       /** === END OF OPTIONS === */ 
  69.  
  70.       $home_link = home_url( '/' ); 
  71.       $before_link = '<span class="breadcrumb" typeof="v:Breadcrumb">'; 
  72.       $after_link = '</span>'; 
  73.       $link_att = ' rel="v:url" property="v:title"'; 
  74.       $link = $before_link . '<a' . $link_att . ' href="%1$s">%2$s</a>' . $after_link; 
  75.  
  76.       $post = get_queried_object(); 
  77.       $parent_id = isset( $post->post_parent ) ? $post->post_parent : ''; 
  78.  
  79.       $html_output = ''; 
  80.  
  81.       if ( is_front_page() ) { 
  82.           if ( 1 == $show['home'] ) { 
  83.               $html_output .= '<div class="breadcrumb-list"><a href="' . $home_link . '">' . $text['home'] . '</a></div>'; 
  84.           } 
  85.  
  86.       } else { 
  87.           $html_output .= '<div class="breadcrumb-list" xmlns:v="http://rdf.data-vocabulary.org/#">' . sprintf( $link, $home_link, $text['home'] ) . $delimiter; 
  88.  
  89.           if ( is_home() ) { 
  90.               if ( 1 == $show['current'] ) { 
  91.                   $html_output .= $before . get_the_title( get_option( 'page_for_posts', true ) ) . $after; 
  92.               } 
  93.  
  94.           } elseif ( is_category() ) { 
  95.               $this_cat = get_category( get_query_var( 'cat' ), false ); 
  96.               if ( 0 != $this_cat->parent ) { 
  97.                   $cats = get_category_parents( $this_cat->parent, true, $delimiter ); 
  98.                   $cats = str_replace( '<a', $before_link . '<a' . $link_att, $cats ); 
  99.                   $cats = str_replace( '</a>', '</a>' . $after_link, $cats ); 
  100.                   $html_output .= $cats; 
  101.               } 
  102.               $html_output .= $before . sprintf( $text['category'], single_cat_title( '', false ) ) . $after; 
  103.  
  104.           } elseif ( is_search() ) { 
  105.               if ( 1 == $show['search'] ) { 
  106.                   $html_output .= $before . sprintf( $text['search'], get_search_query() ) . $after; 
  107.               } 
  108.  
  109.           } elseif ( is_day() ) { 
  110.               $html_output .= sprintf( $link, get_year_link( get_the_time( 'Y' ) ), get_the_time( 'Y' ) ) . $delimiter; 
  111.               $html_output .= sprintf( $link, get_month_link( get_the_time( 'Y' ), get_the_time( 'm' ) ), get_the_time( 'F' ) ) . $delimiter; 
  112.               $html_output .= $before . get_the_time( 'd' ) . $after; 
  113.  
  114.           } elseif ( is_month() ) { 
  115.               $html_output .= sprintf( $link, get_year_link( get_the_time( 'Y' ) ), get_the_time( 'Y' ) ) . $delimiter; 
  116.               $html_output .= $before . get_the_time( 'F' ) . $after; 
  117.  
  118.           } elseif ( is_year() ) { 
  119.               $html_output .= $before . get_the_time( 'Y' ) . $after; 
  120.  
  121.           } elseif ( is_single() && !is_attachment() ) { 
  122.               if ( 'post' != get_post_type() ) { 
  123.                   $post_type = get_post_type_object( get_post_type() ); 
  124.                   $archive_link = get_post_type_archive_link( $post_type->name ); 
  125.                   $html_output .= sprintf( $link, $archive_link, $post_type->labels->singular_name ); 
  126.                   if( 1 == $show['current'] ) { 
  127.                       $html_output .= $delimiter . $before . get_the_title() . $after; 
  128.                   } 
  129.               } else { 
  130.                   $cat = get_the_category(); 
  131.                   $cat = $cat[0]; 
  132.                   $cats = get_category_parents( $cat, true, $delimiter ); 
  133.                   if ( 0 == $show['current'] ) { 
  134.                       $cats = preg_replace( "#^(.+)$delimiter$#", "$1", $cats ); 
  135.                   } 
  136.                   $cats = str_replace( '<a', $before_link . '<a' . $link_att, $cats ); 
  137.                   $cats = str_replace( '</a>', '</a>' . $after_link, $cats ); 
  138.                   $html_output .= $cats; 
  139.                   if ( 1 == $show['current'] ) { 
  140.                       $html_output .= $before . get_the_title() . $after; 
  141.                   } 
  142.               } 
  143.  
  144.           } elseif ( !is_single() && !is_page() && !is_404() && 'post' != get_post_type() ) { 
  145.               $post_type = get_post_type_object( get_post_type() ); 
  146.               $html_output .= $before . $post_type->labels->singular_name . $after; 
  147.  
  148.           } elseif ( is_attachment() ) { 
  149.               $parent = get_post( $parent_id ); 
  150.               $cat = get_the_category( $parent->ID ); 
  151.  
  152.               if( isset( $cat[0] ) ) { 
  153.                   $cat = $cat[0]; 
  154.               } 
  155.  
  156.               if ( $cat ) { 
  157.                   $cats = get_category_parents( $cat, true, $delimiter ); 
  158.                   $cats = str_replace( '<a', $before_link . '<a' . $link_att, $cats ); 
  159.                   $cats = str_replace( '</a>', '</a>' . $after_link, $cats ); 
  160.                   $html_output .= $cats; 
  161.               } 
  162.  
  163.               $html_output .= sprintf( $link, get_permalink( $parent ), $parent->post_title ); 
  164.               if ( 1 == $show['current'] ) { 
  165.                   $html_output .= $delimiter . $before . get_the_title() . $after; 
  166.               } 
  167.  
  168.           } elseif ( is_page() && !$parent_id ) { 
  169.               if( 1 == $show['current'] ) { 
  170.                   $html_output .=  $before . get_the_title() . $after; 
  171.               } 
  172.  
  173.           } elseif ( is_page() && $parent_id ) { 
  174.               $breadcrumbs = array(); 
  175.               while( $parent_id ) { 
  176.                   $page_child = get_page( $parent_id ); 
  177.                   $breadcrumbs[] = sprintf( $link, get_permalink( $page_child->ID ), get_the_title( $page_child->ID ) ); 
  178.                   $parent_id = $page_child->post_parent; 
  179.               } 
  180.               $breadcrumbs = array_reverse( $breadcrumbs ); 
  181.               for( $i = 0; $i < count( $breadcrumbs ); $i++ ) { 
  182.                   $html_output .= $breadcrumbs[$i]; 
  183.                   if( $i != count( $breadcrumbs ) - 1 ) { 
  184.                       $html_output .= $delimiter; 
  185.                   } 
  186.               } 
  187.               if( 1 == $show['current'] ) { 
  188.                   $html_output .= $delimiter . $before . get_the_title() . $after; 
  189.               } 
  190.  
  191.           } elseif ( is_tag() ) { 
  192.               $html_output .= $before . sprintf( $text['tag'], single_tag_title( '', false ) ) . $after; 
  193.  
  194.           } elseif ( is_author() ) { 
  195.               $user_id = get_query_var( 'author' ); 
  196.               $userdata = get_the_author_meta( 'display_name', $user_id ); 
  197.               $html_output .= $before . sprintf( $text['author'], $userdata ) . $after; 
  198.  
  199.           } elseif ( is_404() ) { 
  200.               $html_output .= $before . $text['404'] . $after; 
  201.  
  202.           } 
  203.  
  204.           if ( get_query_var( 'paged' ) || get_query_var( 'page' ) ) { 
  205.               $page_num = get_query_var( 'page' ) ? get_query_var( 'page' ) : get_query_var( 'paged' ); 
  206.               $html_output .= $delimiter . sprintf( __( 'Page %s', 'responsive-mobile' ), $page_num ); 
  207.  
  208.           } 
  209.  
  210.           $html_output .= '</div>'; 
  211.  
  212.       } 
  213.  
  214.       echo $html_output; 
  215.  
  216.   } // end responsive_mobile_breadcrumb_lists
