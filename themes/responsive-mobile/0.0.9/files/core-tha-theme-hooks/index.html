  76. /** 
  77.  * Determines, whether the specific hook type is actually supported. 
  78.  * 
  79.  * Plugin developers should always check for the support of a <strong>specific</strong> 
  80.  * hook type before hooking a callback function to a hook of this type. 
  81.  * 
  82.  * Example: 
  83.  * <code> 
  84.  *         if ( current_theme_supports( 'tha_hooks', 'header' ) ) 
  85.  *               add_action( 'tha_head_top', 'prefix_header_top' ); 
  86.  * </code> 
  87.  * 
  88.  * @param bool $bool true 
  89.  * @param array $args The hook type being checked 
  90.  * @param array $registered All registered hook types 
  91.  * 
  92.  * @return bool 
  93.  */ 
  94. function tha_current_theme_supports( $bool, $args, $registered ) { 
  95.   return in_array( $args[0], $registered[0] ) || in_array( 'all', $registered[0] ); 
  96. }
