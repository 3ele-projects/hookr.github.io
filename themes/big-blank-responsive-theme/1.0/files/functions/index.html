  187. function be_save_post($post_id) { 
  188.   if (defined('DOING_AUTOSAVE') && DOING_AUTOSAVE) 
  189.       return; 
  190.  
  191.   if (isset($_POST['_wpnonce_post_layout']) && !wp_verify_nonce($_POST['_wpnonce_post_layout'], 'post_layout_nonce')) 
  192.       return; 
  193.  
  194.   if (isset($_POST['post_type']) && 'page' == $_POST['post_type']) 
  195.       if (!current_user_can('edit_page', $post_id)) 
  196.           return; 
  197.       else 
  198.       if (!current_user_can('edit_post', $post_id)) 
  199.           return; 
  200.  
  201.   $post_layout = isset($_POST['be_post_layout']) ? $_POST['be_post_layout'] : ''; 
  202.   if (array_key_exists($post_layout, be_layouts())) { 
  203.       update_post_meta($post_id, 'be_post_layout', $post_layout); 
  204.   } elseif ($post_layout == 'default_layout') 
  205.       delete_post_meta($post_id, 'be_post_layout'); 
  206. }

  302.   function start_el(&$output, $item, $depth, $args) { 
  303.       if ($depth != 0) 
  304.           return; 
  305.       global $wp_query; 
  306.       $indent = ( $depth ) ? str0_repeat("\t", $depth) : ''; 
  307.       $class_names = join(' ', apply_filters( 
  308.                       'nav_menu_css_class', array_filter( 
  309.                               empty($item->classes) ? 
  310.                                       array() : 
  311.                                       (array) $item->classes), $item)); 
  312.  
  313.       $output .= $indent . '<li class="' . apply_filters('the_title', $item->title, $item->ID) . ' ' . esc_attr($class_names) . '">'; 
  314.  
  315.       $attributes = !empty($item->attr_title) ? ' title="' . esc_attr($item->attr_title) . '"' : ''; 
  316.       $attributes .=!empty($item->target) ? ' target="' . esc_attr($item->target) . '"' : ''; 
  317.       $attributes .=!empty($item->xfn) ? ' rel="' . esc_attr($item->xfn) . '"' : ''; 
  318.       $attributes .=!empty($item->url) ? ' href="' . esc_attr($item->url) . '"' : ''; 
  319.  
  320.       $item_output = $args->before; 
  321.       $item_output .= '<a' . $attributes . '>'; 
  322.       $item_output .= $args->link_before . apply_filters('the_title', $item->title, $item->ID) . $args->link_after; 
  323.       $item_output .= '</a>'; 
  324.       $item_output .= $args->after; 
  325.  
  326.       $output .= apply_filters('walker_nav_menu_start_el', $item_output, $item, $depth, $args); 
  327.   } 
  328.  
  329.   function end_el(&$output, $item, $depth) { 
  330.       if ($depth != 0) 
  331.           return; 
  332.       $output .= "</li>\n"; 
  333.   }
