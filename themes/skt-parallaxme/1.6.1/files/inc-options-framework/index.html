  104. function optionsframework_setdefaults() { 
  105.    
  106.   $optionsframework_settings = get_option( 'optionsframework' ); 
  107.  
  108.   // Gets the unique option id 
  109.   $option_name = $optionsframework_settings['id']; 
  110.    
  111.   /**  
  112.    * Each theme will hopefully have a unique id, and all of its options saved 
  113.    * as a separate option set.  We need to track all of these option sets so 
  114.    * it can be easily deleted if someone wishes to remove the plugin and 
  115.    * its associated data.  No need to clutter the database.   
  116.    * 
  117.    */ 
  118.    
  119.   if ( isset( $optionsframework_settings['knownoptions'] ) ) { 
  120.       $knownoptions =  $optionsframework_settings['knownoptions']; 
  121.       if ( !in_array( $option_name, $knownoptions ) ) { 
  122.           array_push( $knownoptions, $option_name ); 
  123.           $optionsframework_settings['knownoptions'] = $knownoptions; 
  124.           update_option( 'optionsframework', $optionsframework_settings ); 
  125.       } 
  126.   } else { 
  127.       $newoptionname = array( $option_name ); 
  128.       $optionsframework_settings['knownoptions'] = $newoptionname; 
  129.       update_option( 'optionsframework', $optionsframework_settings ); 
  130.   } 
  131.    
  132.   // Gets the default options data from the array in options.php 
  133.   $options =& _optionsframework_options(); 
  134.    
  135.   // If the options haven't been added to the database yet, they are added now 
  136.   $values = of_get_default_values(); 
  137.    
  138.   if ( isset( $values ) ) { 
  139.       add_option( $option_name, $values ); // Add option with default settings 
  140.   }

  275. /** 
  276.  * Validate Options. 
  277.  * 
  278.  * This runs after the submit/reset button has been clicked and 
  279.  * validates the inputs. 
  280.  * 
  281.  * @uses $_POST['reset'] to restore default options 
  282.  */ 
  283. function optionsframework_validate( $input ) { 
  284.  
  285.   /** 
  286.    * Restore Defaults. 
  287.    * 
  288.    * In the event that the user clicked the "Restore Defaults" 
  289.    * button, the options defined in the theme's options.php 
  290.    * file will be added to the option for the active theme. 
  291.    */ 
  292.  
  293.   if ( isset( $_POST['reset'] ) ) { 
  294.       add_settings_error( 'options-framework', 'restore_defaults', __( 'Default options restored.', 'options_framework_theme' ), 'updated fade' ); 
  295.       return of_get_default_values(); 
  296.   } 
  297.    
  298.   /** 
  299.    * Update Settings 
  300.    * 
  301.    * This used to check for $_POST['update'], but has been updated 
  302.    * to be compatible with the theme customizer introduced in WordPress 3.4 
  303.    */ 
  304.     
  305.   $clean = array(); 
  306.   $options =& _optionsframework_options(); 
  307.   foreach ( $options as $option ) { 
  308.  
  309.       if ( ! isset( $option['id'] ) ) { 
  310.           continue; 
  311.       } 
  312.  
  313.       if ( ! isset( $option['type'] ) ) { 
  314.           continue; 
  315.       } 
  316.  
  317.       $id = preg_replace( '/[^a-zA-Z0-9._\-]/', '', strtolower( $option['id'] ) ); 
  318.  
  319.       // Set checkbox to false if it wasn't sent in the $_POST 
  320.       if ( 'checkbox' == $option['type'] && ! isset( $input[$id] ) ) { 
  321.           $input[$id] = false; 
  322.       } 
  323.  
  324.       // Set each item in the multicheck to false if it wasn't sent in the $_POST 
  325.       if ( 'multicheck' == $option['type'] && ! isset( $input[$id] ) ) { 
  326.           foreach ( $option['options'] as $key => $value ) { 
  327.               $input[$id][$key] = false; 
  328.           } 
  329.       } 
  330.  
  331.       // For a value to be submitted to database it must pass through a sanitization filter 
  332.       if ( has_filter( 'of_sanitize_' . $option['type'] ) ) { 
  333.           $clean[$id] = apply_filters( 'of_sanitize_' . $option['type'], $input[$id], $option ); 
  334.       } 
  335.   } 
  336.    
  337.   // Hook to run after validation 
  338.   do_action( 'optionsframework_after_validate', $clean ); 
  339.    
  340.   return $clean; 
  341. }

  353. /** 
  354.  * Format Configuration Array.

  356.  * Get an array of all default values as set in 
  357.  * options.php. The 'id', 'std' and 'type' keys need 
  358.  * to be defined in the configuration array. In the 
  359.  * event that these keys are not present the option 
  360.  * will not be included in this function's output. 
  361.  * 
  362.  * @return    array     Rey-keyed options configuration array. 
  363.  * 
  364.  * @access    private 
  365.  */ 
  366.   
  367. function of_get_default_values() { 
  368.   $output = array(); 
  369.   $config =& _optionsframework_options(); 
  370.   foreach ( (array) $config as $option ) { 
  371.       if ( ! isset( $option['id'] ) ) { 
  372.           continue; 
  373.       } 
  374.       if ( ! isset( $option['std'] ) ) { 
  375.           continue; 
  376.       } 
  377.       if ( ! isset( $option['type'] ) ) { 
  378.           continue; 
  379.       } 
  380.       if ( has_filter( 'of_sanitize_' . $option['type'] ) ) { 
  381.           $output[$option['id']] = apply_filters( 'of_sanitize_' . $option['type'], $option['std'], $option ); 
  382.       } 
  383.   } 
  384.   return $output;
