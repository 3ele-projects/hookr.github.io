  357. /** 
  358.  * Is a given string a color formatted in hexidecimal notation? 
  359.  * 
  360.  * @param    string    Color in hexidecimal notation. "#" may or may not be prepended to the string. 
  361.  * @return   bool 
  362.  * 
  363.  */ 
  364.  
  365. function of_validate_hex( $hex ) { 
  366.   $hex = trim( $hex ); 
  367.   /** Strip recognized prefixes. */ 
  368.   if ( 0 === strpos( $hex, '#' ) ) { 
  369.       $hex = substr( $hex, 1 ); 
  370.   } 
  371.   elseif ( 0 === strpos( $hex, '%23' ) ) { 
  372.       $hex = substr( $hex, 3 ); 
  373.   } 
  374.   /** Regex match. */ 
  375.   if ( 0 === preg_match( '/^[0-9a-fA-F]{6}$/', $hex ) ) { 
  376.       return false; 
  377.   } 
  378.   else { 
  379.       return true; 
  380.   } 
  381. }
