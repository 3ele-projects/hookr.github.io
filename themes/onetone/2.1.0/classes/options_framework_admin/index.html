  229.         /**

  235.  
  236.         $clean = array(); 
  237.         $options = & Options_Framework::_optionsframework_options(); 
  238.          
  239.         foreach ( $options as $option ) { 
  240.  
  241.             if ( ! isset( $option['id'] ) ) { 
  242.                 continue; 
  243.             } 
  244.  
  245.             if ( ! isset( $option['type'] ) ) { 
  246.                 continue; 
  247.             } 
  248.  
  249.             $id = preg_replace( '/[^a-zA-Z0-9._\-]/', '', strtolower( $option['id'] ) ); 
  250.

  266.  
  267.  
  268.             // For a value to be submitted to database it must pass through a sanitization filter 
  269.             if ( has_filter( 'of_sanitize_' . $option['type'] ) ) { 
  270.                 $clean[$id] = apply_filters( 'of_sanitize_' . $option['type'], $input[$id], $option );

  272.         }
