  694.  
  695.                             $style .=

  697.                                 .$data['property'].':'.$rgb_color.'}';

  751.     /** 
  752.      * Convert Hexa to RGB 
  753.      * 
  754.      * @return void 
  755.      **/ 
  756.     function onetone_hex2RGB( $hex ) { 
  757.         preg_match( "/^#{0, 1}([0-9a-f]{1, 6})$/i", $hex, $match ); 
  758.          
  759.         if ( ! isset( $match[1] ) ) { 
  760.             return false; 
  761.         } 
  762.  
  763.         if ( strlen( $match[1] ) == 6 ) { 
  764.             list( $r, $g, $b ) = array( $hex[0] . $hex[1], $hex[2] . $hex[3], $hex[4] . $hex[5] ); 
  765.         } elseif ( strlen( $match[1] ) == 3 ) { 
  766.             list( $r, $g, $b ) = array( $hex[0] . $hex[0], $hex[1] . $hex[1], $hex[2] . $hex[2] ); 
  767.         } else if ( strlen( $match[1] ) == 2 ) { 
  768.             list( $r, $g, $b ) = array( $hex[0] . $hex[1], $hex[0] . $hex[1], $hex[0] . $hex[1] ); 
  769.         } else if ( strlen( $match[1] ) == 1 ) { 
  770.             list( $r, $g, $b ) = array( $hex . $hex, $hex . $hex, $hex . $hex ); 
  771.         } else { 
  772.             return false; 
  773.         } 
  774.  
  775.         $color = array(); 
  776.         $color['r'] = hexdec( $r ); 
  777.         $color['g'] = hexdec( $g ); 
  778.         $color['b'] = hexdec( $b ); 
  779.  
  780.         return $color; 
  781.     }
