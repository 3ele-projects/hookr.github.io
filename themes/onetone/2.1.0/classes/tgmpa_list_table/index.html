  617.     /** 
  618.      * Defines the bulk actions for handling registered plugins. 
  619.      * 
  620.      * @since 2.2.0 
  621.      * 
  622.      * @return array $actions The bulk actions for the plugin install table. 
  623.      */ 
  624.     public function get_bulk_actions() { 
  625.  
  626.         $actions = array(); 
  627.  
  628.         if ( 'update' !== $this->view_context && 'activate' !== $this->view_context ) { 
  629.             if ( current_user_can( 'install_plugins' ) ) { 
  630.                 $actions['tgmpa-bulk-install'] = __( 'Install', 'onetone' ); 
  631.             } 
  632.         } 
  633.  
  634.         if ( 'install' !== $this->view_context ) { 
  635.             if ( current_user_can( 'update_plugins' ) ) { 
  636.                 $actions['tgmpa-bulk-update'] = __( 'Update', 'onetone' ); 
  637.             } 
  638.             if ( current_user_can( 'activate_plugins' ) ) { 
  639.                 $actions['tgmpa-bulk-activate'] = __( 'Activate', 'onetone' ); 
  640.             } 
  641.         } 
  642.  
  643.         return $actions; 
  644.     }

  646.     /** 
  647.      * Processes bulk installation and activation actions. 
  648.      * 
  649.      * The bulk installation process looks for the $_POST information and passes that 
  650.      * through if a user has to use WP_Filesystem to enter their credentials. 
  651.      * 
  652.      * @since 2.2.0 
  653.      */ 
  654.     public function process_bulk_actions() { 
  655.         // Bulk installation process. 
  656.         if ( 'tgmpa-bulk-install' === $this->current_action() || 'tgmpa-bulk-update' === $this->current_action() ) { 
  657.  
  658.             check_admin_referer( 'bulk-' . $this->_args['plural'] ); 
  659.  
  660.             $install_type = 'install'; 
  661.             if ( 'tgmpa-bulk-update' === $this->current_action() ) { 
  662.                 $install_type = 'update'; 
  663.             } 
  664.  
  665.             $plugins_to_install = array(); 
  666.  
  667.             // Did user actually select any plugins to install/update ? 
  668.             if ( empty( $_POST['plugin'] ) ) { 
  669.                 if ( 'install' === $install_type ) { 
  670.                     $message = __( 'No plugins were selected to be installed. No action taken.', 'onetone' ); 
  671.                 } else { 
  672.                     $message = __( 'No plugins were selected to be updated. No action taken.', 'onetone' ); 
  673.                 } 
  674.  
  675.                 echo '<div id="message" class="error"><p>', esc_html( $message ), '</p></div>'; 
  676.  
  677.                 return false; 
  678.             } 
  679.  
  680.             if ( is_array( $_POST['plugin'] ) ) { 
  681.                 $plugins_to_install = (array) $_POST['plugin']; 
  682.             } elseif ( is_string( $_POST['plugin'] ) ) { 
  683.                 // Received via Filesystem page - un-flatten array (WP bug #19643). 
  684.                 $plugins_to_install = explode( ', ', $_POST['plugin'] ); 
  685.             } 
  686.  
  687.             // Sanitize the received input. 
  688.             $plugins_to_install = array_map( 'urldecode', $plugins_to_install ); 
  689.             $plugins_to_install = array_map( array( $this->tgmpa, 'sanitize_key' ), $plugins_to_install ); 
  690.  
  691.             // Validate the received input. 
  692.             foreach ( $plugins_to_install as $key => $slug ) { 
  693.                 // Check if the plugin was registered with TGMPA and remove if not. 
  694.                 if ( ! isset( $this->tgmpa->plugins[ $slug ] ) ) { 
  695.                     unset( $plugins_to_install[ $key ] ); 
  696.                     continue; 
  697.                 } 
  698.  
  699.                 // For install: make sure this is a plugin we *can* install and not one already installed. 
  700.                 if ( 'install' === $install_type && true === $this->tgmpa->is_plugin_installed( $slug ) ) { 
  701.                     unset( $plugins_to_install[ $key ] ); 
  702.                 } 
  703.  
  704.                 // For updates: make sure this is a plugin we *can* update (update available and WP version ok). 
  705.                 if ( 'update' === $install_type && false === $this->tgmpa->is_plugin_updatetable( $slug ) ) { 
  706.                     unset( $plugins_to_install[ $key ] ); 
  707.                 } 
  708.             } 
  709.  
  710.             // No need to proceed further if we have no plugins to handle. 
  711.             if ( empty( $plugins_to_install ) ) { 
  712.                 if ( 'install' === $install_type ) { 
  713.                     $message = __( 'No plugins are available to be installed at this time.', 'onetone' ); 
  714.                 } else { 
  715.                     $message = __( 'No plugins are available to be updated at this time.', 'onetone' ); 
  716.                 } 
  717.  
  718.                 echo '<div id="message" class="error"><p>', esc_html( $message ), '</p></div>'; 
  719.  
  720.                 return false; 
  721.             } 
  722.  
  723.             // Pass all necessary information if WP_Filesystem is needed. 
  724.             $url = wp_nonce_url( 
  725.                 $this->tgmpa->get_tgmpa_url(),  
  726.                 'bulk-' . $this->_args['plural'] 
  727.  ); 
  728.  
  729.             // Give validated data back to $_POST which is the only place the filesystem looks for extra fields. 
  730.             $_POST['plugin'] = implode( ', ', $plugins_to_install ); // Work around for WP bug #19643. 
  731.  
  732.             $method = ''; // Leave blank so WP_Filesystem can populate it as necessary. 
  733.             $fields = array_keys( $_POST ); // Extra fields to pass to WP_Filesystem. 
  734.  
  735.             if ( false === ( $creds = request_filesystem_credentials( esc_url_raw( $url ), $method, false, false, $fields ) ) ) { 
  736.                 return true; // Stop the normal page form from displaying, credential request form will be shown. 
  737.             } 
  738.  
  739.             // Now we have some credentials, setup WP_Filesystem. 
  740.             if ( ! WP_Filesystem( $creds ) ) { 
  741.                 // Our credentials were no good, ask the user for them again. 
  742.                 request_filesystem_credentials( esc_url_raw( $url ), $method, true, false, $fields ); 
  743.  
  744.                 return true; 
  745.             } 
  746.  
  747.             /** If we arrive here, we have the filesystem */ 
  748.  
  749.             // Store all information in arrays since we are processing a bulk installation. 
  750.             $names = array(); 
  751.             $sources = array(); // Needed for installs. 
  752.             $file_paths = array(); // Needed for upgrades. 
  753.             $to_inject = array(); // Information to inject into the update_plugins transient. 
  754.  
  755.             // Prepare the data for validated plugins for the install/upgrade. 
  756.             foreach ( $plugins_to_install as $slug ) { 
  757.                 $name = $this->tgmpa->plugins[ $slug ]['name']; 
  758.                 $source = $this->tgmpa->get_download_url( $slug ); 
  759.  
  760.                 if ( ! empty( $name ) && ! empty( $source ) ) { 
  761.                     $names[] = $name; 
  762.  
  763.                     switch ( $install_type ) { 
  764.  
  765.                         case 'install': 
  766.                             $sources[] = $source; 
  767.                             break; 
  768.  
  769.                         case 'update': 
  770.                             $file_paths[] = $this->tgmpa->plugins[ $slug ]['file_path']; 
  771.                             $to_inject[ $slug ]           = $this->tgmpa->plugins[ $slug ]; 
  772.                             $to_inject[ $slug ]['source'] = $source; 
  773.                             break; 
  774.                     } 
  775.                 } 
  776.             } 
  777.             unset( $slug, $name, $source ); 
  778.  
  779.             // Create a new instance of TGMPA_Bulk_Installer. 
  780.             $installer = new TGMPA_Bulk_Installer( 
  781.                 new TGMPA_Bulk_Installer_Skin( 
  782.                     array( 
  783.                         'url' => esc_url_raw( $this->tgmpa->get_tgmpa_url() ),  
  784.                         'nonce' => 'bulk-' . $this->_args['plural'],  
  785.                         'names' => $names,  
  786.                         'install_type' => $install_type,  
  787.  ) 
  788.  ) 
  789.  ); 
  790.  
  791.             // Wrap the install process with the appropriate HTML. 
  792.             echo '<div class="tgmpa">',  
  793.                 '<h2 style="font-size: 23px; font-weight: 400; line-height: 29px; margin: 0; padding: 9px 15px 4px 0;">', esc_html( get_admin_page_title() ), '</h2> 
  794.                 <div class="update-php" style="width: 100%; height: 98%; min-height: 850px; padding-top: 1px;">'; 
  795.  
  796.             // Process the bulk installation submissions. 
  797.             add_filter( 'upgrader_source_selection', array( $this->tgmpa, 'maybe_adjust_source_dir' ), 1, 3 ); 
  798.  
  799.             if ( 'tgmpa-bulk-update' === $this->current_action() ) { 
  800.                 // Inject our info into the update transient. 
  801.                 $this->tgmpa->inject_update_info( $to_inject ); 
  802.  
  803.                 $installer->bulk_upgrade( $file_paths ); 
  804.             } else { 
  805.                 $installer->bulk_install( $sources ); 
  806.             } 
  807.  
  808.             remove_filter( 'upgrader_source_selection', array( $this->tgmpa, 'maybe_adjust_source_dir' ), 1 ); 
  809.  
  810.             echo '</div></div>'; 
  811.  
  812.             return true; 
  813.         } 
  814.  
  815.         // Bulk activation process. 
  816.         if ( 'tgmpa-bulk-activate' === $this->current_action() ) { 
  817.             check_admin_referer( 'bulk-' . $this->_args['plural'] ); 
  818.  
  819.             // Did user actually select any plugins to activate ? 
  820.             if ( empty( $_POST['plugin'] ) ) { 
  821.                 echo '<div id="message" class="error"><p>', esc_html__( 'No plugins were selected to be activated. No action taken.', 'onetone' ), '</p></div>'; 
  822.  
  823.                 return false; 
  824.             } 
  825.  
  826.             // Grab plugin data from $_POST. 
  827.             $plugins = array(); 
  828.             if ( isset( $_POST['plugin'] ) ) { 
  829.                 $plugins = array_map( 'urldecode', (array) $_POST['plugin'] ); 
  830.                 $plugins = array_map( array( $this->tgmpa, 'sanitize_key' ), $plugins ); 
  831.             } 
  832.  
  833.             $plugins_to_activate = array(); 
  834.             $plugin_names = array(); 
  835.  
  836.             // Grab the file paths for the selected & inactive plugins from the registration array. 
  837.             foreach ( $plugins as $slug ) { 
  838.                 if ( $this->tgmpa->can_plugin_activate( $slug ) ) { 
  839.                     $plugins_to_activate[] = $this->tgmpa->plugins[ $slug ]['file_path']; 
  840.                     $plugin_names[] = $this->tgmpa->plugins[ $slug ]['name']; 
  841.                 } 
  842.             } 
  843.             unset( $slug ); 
  844.  
  845.             // Return early if there are no plugins to activate. 
  846.             if ( empty( $plugins_to_activate ) ) { 
  847.                 echo '<div id="message" class="error"><p>', esc_html__( 'No plugins are available to be activated at this time.', 'onetone' ), '</p></div>'; 
  848.  
  849.                 return false; 
  850.             } 
  851.  
  852.             // Now we are good to go - let's start activating plugins. 
  853.             $activate = activate_plugins( $plugins_to_activate ); 
  854.  
  855.             if ( is_wp_error( $activate ) ) { 
  856.                 echo '<div id="message" class="error"><p>', wp_kses_post( $activate->get_error_message() ), '</p></div>'; 
  857.             } else { 
  858.                 $count = count( $plugin_names ); // Count so we can use _n function. 
  859.                 $plugin_names = array_map( array( 'TGMPA_Utils', 'wrap_in_strong' ), $plugin_names ); 
  860.                 $last_plugin = array_pop( $plugin_names ); // Pop off last name to prep for readability. 
  861.                 $imploded = empty( $plugin_names ) ? $last_plugin : ( implode( ', ', $plugin_names ) . ' ' . esc_html_x( 'and', 'plugin A *and* plugin B', 'onetone' ) . ' ' . $last_plugin ); 
  862.  
  863.                 printf( // WPCS: xss ok. 
  864.                     '<div id="message" class="updated"><p>%1$s %2$s.</p></div>',  
  865.                     esc_html( _n( 'The following plugin was activated successfully:', 'The following plugins were activated successfully:', $count, 'onetone' ) ),  
  866.                     $imploded 
  867.  ); 
  868.  
  869.                 // Update recently activated plugins option. 
  870.                 $recent = (array) get_option( 'recently_activated' ); 
  871.                 foreach ( $plugins_to_activate as $plugin => $time ) { 
  872.                     if ( isset( $recent[ $plugin ] ) ) { 
  873.                         unset( $recent[ $plugin ] ); 
  874.                     } 
  875.                 } 
  876.                 update_option( 'recently_activated', $recent ); 
  877.             } 
  878.  
  879.             unset( $_POST ); // Reset the $_POST variable in case user wants to perform one action after another. 
  880.  
  881.             return true; 
  882.         } 
  883.  
  884.         return false; 
  885.     }

  888.      * Prepares all of our information to be outputted into a usable table. 
  889.      *

  892.     public function prepare_items() { 
  893.         $columns = $this->get_columns(); // Get all necessary column information. 
  894.         $hidden = array(); // No columns to hide, but we must set as an array. 
  895.         $sortable = array(); // No reason to make sortable columns. 
  896.         $primary = $this->get_primary_column_name(); // Column which has the row actions. 
  897.         $this->_column_headers = array( $columns, $hidden, $sortable, $primary ); // Get all necessary column headers. 
  898.  
  899.         // Process our bulk activations here. 
  900.         if ( 'tgmpa-bulk-activate' === $this->current_action() ) { 
  901.             $this->process_bulk_actions(); 
  902.         } 
  903.  
  904.         // Store all of our plugin data into $items array so WP_List_Table can use it. 
  905.         $this->items = apply_filters( 'tgmpa_table_data_items', $this->_gather_plugin_data() ); 
  906.     }
