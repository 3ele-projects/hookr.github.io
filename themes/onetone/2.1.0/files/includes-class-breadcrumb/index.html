  487. function onetone_breadcrumbs_get_parents( $post_id = '', $path = '' ) { 
  488.  
  489.   /** Set up an empty trail array. */ 
  490.   $trail = array(); 
  491.  
  492.   /** If neither a post ID nor path set, return an empty array. */ 
  493.   if ( empty( $post_id ) && empty( $path ) ) 
  494.       return $trail; 
  495.  
  496.   /** If the post ID is empty, use the path to get the ID. */ 
  497.   if ( empty( $post_id ) ) { 
  498.  
  499.       /** Get parent post by the path. */ 
  500.       $parent_page = get_page_by_path( $path ); 
  501.  
  502.  
  503.       if( empty( $parent_page ) ) 
  504.               // search on page name (single word) 
  505.           $parent_page = get_page_by_title ( $path ); 
  506.  
  507.       if( empty( $parent_page ) ) 
  508.           // search on page title (multiple words) 
  509.           $parent_page = get_page_by_title ( str_replace( array('-', '_'), ' ', $path ) ); 
  510.  
  511.       /** End Modification */ 
  512.  
  513.       /** If a parent post is found, set the $post_id variable to it. */ 
  514.       if ( !empty( $parent_page ) ) 
  515.           $post_id = $parent_page->ID; 
  516.   } 
  517.  
  518.   /** If a post ID and path is set, search for a post by the given path. */ 
  519.   if ( $post_id == 0 && !empty( $path ) ) { 
  520.  
  521.       /** Separate post names into separate paths by '/'. */ 
  522.       $path = trim( $path, '/' ); 
  523.       preg_match_all( "/\/.*?\z/", $path, $matches ); 
  524.  
  525.       /** If matches are found for the path. */ 
  526.       if ( isset( $matches ) ) { 
  527.  
  528.           /** Reverse the array of matches to search for posts in the proper order. */ 
  529.           $matches = array_reverse( $matches ); 
  530.  
  531.           /** Loop through each of the path matches. */ 
  532.           foreach ( $matches as $match ) { 
  533.  
  534.               /** If a match is found. */ 
  535.               if ( isset( $match[0] ) ) { 
  536.  
  537.                   /** Get the parent post by the given path. */ 
  538.                   $path = str_replace( $match[0], '', $path ); 
  539.                   $parent_page = get_page_by_path( trim( $path, '/' ) ); 
  540.  
  541.                   /** If a parent post is found, set the $post_id and break out of the loop. */ 
  542.                   if ( !empty( $parent_page ) && $parent_page->ID > 0 ) { 
  543.                       $post_id = $parent_page->ID; 
  544.                       break; 
  545.                   } 
  546.               } 
  547.           } 
  548.       } 
  549.   } 
  550.  
  551.   /** While there's a post ID, add the post link to the $parents array. */ 
  552.   while ( $post_id ) { 
  553.  
  554.       /** Get the post by ID. */ 
  555.       $page = get_page( $post_id ); 
  556.  
  557.       /** Add the formatted post link to the array of parents. */ 
  558.       $parents[] = '<a href="' . get_permalink( $post_id ) . '" title="' . esc_attr( get_the_title( $post_id ) ) . '">' . get_the_title( $post_id ) . '</a>'; 
  559.  
  560.       /** Set the parent post's parent to the post ID. */ 
  561.       $post_id = $page->post_parent; 
  562.   } 
  563.  
  564.   /** If we have parent posts, reverse the array to put them in the proper order for the trail. */ 
  565.   if ( isset( $parents ) ) 
  566.       $trail = array_reverse( $parents ); 
  567.  
  568.   /** Return the trail of parent posts. */ 
  569.   return $trail; 
  570.  
  571. } // End onetone_breadcrumbs_get_parents()

  574. function onetone_breadcrumbs_get_term_parents( $parent_id = '', $taxonomy = '' ) { 
  575.  
  576.   /** Set up some default arrays. */ 
  577.   $trail = array(); 
  578.   $parents = array(); 
  579.  
  580.   /** If no term parent ID or taxonomy is given, return an empty array. */ 
  581.   if ( empty( $parent_id ) || empty( $taxonomy ) ) 
  582.       return $trail; 
  583.  
  584.   /** While there is a parent ID, add the parent term link to the $parents array. */ 
  585.   while ( $parent_id ) { 
  586.  
  587.       /** Get the parent term. */ 
  588.       $parent = get_term( $parent_id, $taxonomy ); 
  589.  
  590.       /** Add the formatted term link to the array of parent terms. */ 
  591.       $parents[] = '<a href="' . get_term_link( $parent, $taxonomy ) . '" title="' . esc_attr( $parent->name ) . '">' . $parent->name . '</a>'; 
  592.  
  593.       /** Set the parent term's parent as the parent ID. */ 
  594.       $parent_id = $parent->parent; 
  595.   } 
  596.  
  597.   /** If we have parent terms, reverse the array to put them in the proper order for the trail. */ 
  598.   if ( !empty( $parents ) ) 
  599.       $trail = array_reverse( $parents ); 
  600.  
  601.   /** Return the trail of parent terms. */ 
  602.   return $trail; 
  603.  
  604. } // End onetone_breadcrumbs_get_term_parents()
