  62. /** 
  63. * Sets up the breadcrumb trail. 
  64. *

  66. * @access public 
  67. * @param array $args The arguments for how to build the breadcrumb trail. 
  68. * @return void 
  69. */ 
  70. public function __construct( $args = array() ) { 
  71.   /** Remove the bbPress breadcrumbs. */ 
  72.   add_filter( 'bbp_get_breadcrumb', '__return_false' ); 
  73.   $defaults = array( 
  74.   'container' => 'div',  
  75.   'separator' => '/',  
  76.   'before' => '',  
  77.   'after' => '',  
  78.   'show_on_front' => true,  
  79.   'network' => false,  
  80.   //'show_edit_link' => false,  
  81.   'show_title' => true,  
  82.   'show_browse' => true,  
  83.   'echo' => true,  
  84.   /** Post taxonomy (examples follow). */ 
  85.   'post_taxonomy' => array( 
  86.     'portfolio' => 'portfolio-category',  
  87.   // 'book' => 'genre',  
  88.  ),  
  89.   /** Labels for text used (see Breadcrumb_Trail::default_labels). */ 
  90.   'labels' => array() 
  91.  ); 
  92.   $this->args = apply_filters( 'breadcrumb_trail_args', wp_parse_args( $args, $defaults ) ); 
  93.   /** Merge the user-added labels with the defaults. */ 
  94.   $this->args['labels'] = wp_parse_args( $this->args['labels'], $this->default_labels() );

  97. /** 
  98. * Formats and outputs the breadcrumb trail. 
  99. * 
  100. * @since 0.6.0 
  101. * @access public 
  102. * @return string 
  103. */ 
  104. public function trail() { 
  105.   $breadcrumb = ''; 
  106.   /** Connect the breadcrumb trail if there are items in the trail. */ 
  107.   if ( !empty( $this->items ) && is_array( $this->items ) ) { 
  108.   /** Make sure we have a unique array of items. */ 
  109.   $this->items = array_unique( $this->items ); 
  110.   /** Open the breadcrumb trail containers. */ 
  111.   $breadcrumb = "\n\t\t" . '<' . tag_escape( $this->args['container'] ) . ' class="breadcrumb-nav breadcrumbs" itemprop="breadcrumb">'; 
  112.   /** If $before was set, wrap it in a container. */ 
  113.   $breadcrumb .= ( !empty( $this->args['before'] ) ? "\n\t\t\t" . '<span class="trail-before">' . $this->args['before'] . '</span> ' . "\n\t\t\t" : '' ); 
  114.   /** Add 'browse' label if it should be shown. */ 
  115.   if ( true === $this->args['show_browse'] ) 
  116.   $breadcrumb .= "\n\t\t\t" . '<span class="trail-browse">' . $this->args['labels']['browse'] . '</span> '; 
  117.   /** Adds the 'trail-begin' class around first item if there's more than one item. */ 
  118.   if ( 1 < count( $this->items ) ) 
  119.   array_unshift( $this->items, '<span class="trail-begin">' . array_shift( $this->items ) . '</span>' ); 
  120.   /** Adds the 'trail-end' class around last item. */ 
  121.   array_push( $this->items, '<span class="trail-end">' . array_pop( $this->items ) . '</span>' ); 
  122.   /** Format the separator. */ 
  123.   $separator = ( !empty( $this->args['separator'] ) ? '<span class="sep">' . $this->args['separator'] . '</span>' : '<span class="sep">/</span>' ); 
  124.   /** Join the individual trail items into a single string. */ 
  125.   $breadcrumb .= join( "\n\t\t\t {$separator} ", $this->items ); 
  126.   /** If $after was set, wrap it in a container. */ 
  127.   $breadcrumb .= ( !empty( $this->args['after'] ) ? "\n\t\t\t" . ' <span class="trail-after">' . $this->args['after'] . '</span>' : '' ); 
  128.   /** Close the breadcrumb trail containers. */ 
  129.   $breadcrumb .= "\n\t\t" . '</' . tag_escape( $this->args['container'] ) . '>'; 
  130.   } 
  131.   /** Allow developers to filter the breadcrumb trail HTML. */ 
  132.   $breadcrumb = apply_filters( 'breadcrumb_trail', $breadcrumb, $this->args ); 
  133.   if ( true === $this->args['echo'] ) 
  134.   echo $breadcrumb; 
  135.   else 
  136.   return $breadcrumb; 
  137. }

  168. /** 
  169. * Runs through the various WordPress conditional tags to check the current page being viewed. Once 
  170. * a condition is met, a specific method is launched to add items to the $items array. 
  171. * 
  172. * @since 0.6.0 
  173. * @access public 
  174. * @return void 
  175. */ 
  176. public function do_trail_items() { 
  177. /** If viewing the front page. */ 
  178. if ( is_front_page() ) { 
  179. $this->do_front_page_items(); 
  180. } 
  181. /** If not viewing the front page. */ 
  182. else { 
  183. /** Add the network and site home links. */ 
  184. $this->do_network_home_link(); 
  185. $this->do_site_home_link(); 
  186. /** If viewing the home/blog page. */ 
  187. if ( is_home() ) { 
  188. $this->do_posts_page_items(); 
  189. } 
  190. /** If viewing a single post. */ 
  191. elseif ( is_singular() ) { 
  192. $this->do_singular_items(); 
  193. } 
  194. /** If viewing an archive page. */ 
  195. elseif ( is_archive() ) { 
  196. if ( is_post_type_archive() ) 
  197. $this->do_post_type_archive_items(); 
  198. elseif ( is_category() || is_tag() || is_tax() ) 
  199. $this->do_term_archive_items(); 
  200. elseif ( is_author() ) 
  201. $this->do_user_archive_items(); 
  202. elseif ( get_query_var( 'minute' ) && get_query_var( 'hour' ) ) 
  203. $this->do_minute_hour_archive_items(); 
  204. elseif ( get_query_var( 'minute' ) ) 
  205. $this->do_minute_archive_items(); 
  206. elseif ( get_query_var( 'hour' ) ) 
  207. $this->do_hour_archive_items(); 
  208. elseif ( is_day() ) 
  209. $this->do_day_archive_items(); 
  210. elseif ( get_query_var( 'w' ) ) 
  211. $this->do_week_archive_items(); 
  212. elseif ( is_month() ) 
  213. $this->do_month_archive_items(); 
  214. elseif ( is_year() ) 
  215. $this->do_year_archive_items(); 
  216. else 
  217. $this->do_default_archive_items(); 
  218. } 
  219. /** If viewing a search results page. */ 
  220. elseif ( is_search() ) { 
  221. $this->do_search_items(); 
  222. } 
  223. /** If viewing the 404 page. */ 
  224. elseif ( is_404() ) { 
  225. $this->do_404_items(); 
  226. } 
  227. } 
  228. /** Add paged items if they exist. */ 
  229. $this->do_paged_items(); 
  230. /** Allow developers to overwrite the items for the breadcrumb trail. */ 
  231. $this->items = apply_filters( 'breadcrumb_trail_items', $this->items, $this->args ); 
  232. } 
  233. /** 
  234. * Gets front items based on $wp_rewrite->front. 
  235. * 
  236. * @since 0.6.0 
  237. * @access public 
  238. * @return void 
  239. */ 
  240. public function do_rewrite_front_items() { 
  241. global $wp_rewrite; 
  242. if ( $wp_rewrite->front ) 
  243. $this->do_path_parents( $wp_rewrite->front ); 
  244. } 
  245. /** 
  246. * Adds the page/paged number to the items array. 
  247. * 
  248. * @since 0.6.0 
  249. * @access public 
  250. * @return void 
  251. */ 
  252. public function do_paged_items() { 
  253. /** If viewing a paged singular post. */ 
  254. if ( is_singular() && 1 < get_query_var( 'page' ) && true === $this->args['show_title'] ) 
  255. $this->items[] = sprintf( $this->args['labels']['paged'], number_format_i18n( absint( get_query_var( 'page' ) ) ) ); 
  256. /** If viewing a paged archive-type page. */ 
  257. elseif ( is_paged() && true === $this->args['show_title'] ) 
  258. $this->items[] = sprintf( $this->args['labels']['paged'], number_format_i18n( absint( get_query_var( 'paged' ) ) ) ); 
  259. } 
  260. /** 
  261. * Adds the network (all sites) home page link to the items array. 
  262. * 
  263. * @since 0.6.0 
  264. * @access public 
  265. * @return void 
  266. */ 
  267. public function do_network_home_link() { 
  268. if ( is_multisite() && !is_main_site() && true === $this->args['network'] ) 
  269. $this->items[] = '<a href="' . network_home_url() . '" title="' . esc_attr( $this->args['labels']['home'] ) . '" rel="home">' . $this->args['labels']['home'] . '</a>'; 
  270. } 
  271. /** 
  272. * Adds the current site's home page link to the items array. 
  273. * 
  274. * @since 0.6.0 
  275. * @access public 
  276. * @return void 
  277. */ 
  278. public function do_site_home_link() { 
  279. $label = ( is_multisite() && !is_main_site() && true === $this->args['network'] ) ? get_bloginfo( 'name' ) : $this->args['labels']['home']; 
  280. $rel = ( is_multisite() && !is_main_site() && true === $this->args['network'] ) ? '' : ' rel="home"'; 
  281. $this->items[] = '<a href="' . home_url() . '" title="' . esc_attr( get_bloginfo( 'name' ) ) . '"' . $rel .'>' . $label . '</a>'; 
  282. } 
  283. /** 
  284. * Adds items for the front page to the items array. 
  285. * 
  286. * @since 0.6.0 
  287. * @access public 
  288. * @return void 
  289. */ 
  290. public function do_front_page_items() { 
  291. /** Only show front items if the 'show_on_front' argument is set to 'true'. */ 
  292. if ( true === $this->args['show_on_front'] || is_paged() || ( is_singular() && 1 < get_query_var( 'page' ) ) ) { 
  293. /** If on a paged view, add the home link items. */ 
  294. if ( is_paged() ) { 
  295. $this->do_network_home_link(); 
  296. $this->do_site_home_link(); 
  297. } 
  298. /** If on the main front page, add the network home link item and the home item. */ 
  299. else { 
  300. $this->do_network_home_link(); 
  301. if ( true === $this->args['show_title'] ) 
  302. $this->items[] = ( is_multisite() && true === $this->args['network'] ) ? get_bloginfo( 'name' ) : $this->args['labels']['home']; 
  303. } 
  304. } 
  305. } 
  306. /** 
  307. * Adds items for the posts page (i.e., is_home()) to the items array. 
  308. * 
  309. * @since 0.6.0 
  310. * @access public 
  311. * @return void 
  312. */ 
  313. public function do_posts_page_items() { 
  314. /** Get the post ID and post. */ 
  315. $post_id = get_queried_object_id(); 
  316. $post = get_page( $post_id ); 
  317. /** If the post has parents, add them to the trail. */ 
  318. if ( 0 < $post->post_parent ) 
  319. $this->do_post_parents( $post->post_parent ); 
  320. /** Get the page title. */ 
  321. $title = get_the_title( $post_id ); 
  322. /** Add the posts page item. */ 
  323. if ( is_paged() ) 
  324. $this->items[] = '<a href="' . get_permalink( $post_id ) . '" title="' . esc_attr( $title ) . '">' . $title . '</a>'; 
  325. elseif ( $title && true === $this->args['show_title'] ) 
  326. $this->items[] = $title; 
  327. } 
  328. /** 
  329. * Adds singular post items to the items array. 
  330. * 
  331. * @since 0.6.0 
  332. * @access public 
  333. * @return void 
  334. */ 
  335. public function do_singular_items() { 
  336.    
  337.  $show_categories = onetone_option('breadcrumb_show_categories', 'yes'); 
  338.  $show_post_type_archive = onetone_option('breadcrumb_show_post_type_archive', 'yes'); 
  339. /** Get the queried post. */ 
  340. $post = get_queried_object(); 
  341. $post_id = get_queried_object_id(); 
  342.  
  343. if( $show_categories == 'yes') { 
  344. /** If the post has a parent, follow the parent trail. */ 
  345. if ( 0 < $post->post_parent ) 
  346. $this->do_post_parents( $post->post_parent ); 
  347. /** If the post doesn't have a parent, get its hierarchy based off the post type. */ 
  348. else 
  349. $this->do_post_hierarchy( $post_id ); 
  350. } 
  351. if( $show_post_type_archive == 'yes') { 
  352. /** Display terms for specific post type taxonomy if requested. */ 
  353. $this->do_post_terms( $post_id ); 
  354. } 
  355. /** End with the post title. */ 
  356. if ( $post_title = single_post_title( '', false ) ) { 
  357. if ( 1 < get_query_var( 'page' ) ) 
  358. $this->items[] = '<a href="' . get_permalink( $post_id ) . '" title="' . esc_attr( $post_title ) . '">' . $post_title . '</a>'; 
  359. elseif ( true === $this->args['show_title'] ) 
  360. $this->items[] = $post_title; 
  361. } 
  362. } 
  363. /** 
  364. * Adds a specific post's parents to the items array. 
  365. * 
  366. * @since 0.6.0 
  367. * @access public 
  368. * @param int $post_id The ID of the post to get the parents of. 
  369. * @return void 
  370. */ 
  371. public function do_post_parents( $post_id ) { 
  372. $parents = array(); 
  373. while ( $post_id ) { 
  374. /** Get the post by ID. */ 
  375. $post = get_post( $post_id ); 
  376. /** Add the formatted post link to the array of parents. */ 
  377. $parents[] = '<a href="' . get_permalink( $post_id ) . '" title="' . esc_attr( get_the_title( $post_id ) ) . '">' . get_the_title( $post_id ) . '</a>'; 
  378. /** If there's no longer a post parent, brea out of the loop. */ 
  379. if ( 0 >= $post->post_parent ) 
  380. break; 
  381. /** Change the post ID to the parent post to continue looping. */ 
  382. $post_id = $post->post_parent; 
  383. } 
  384. /** Get the post hierarchy based off the final parent post. */ 
  385. $this->do_post_hierarchy( $post_id ); 
  386. /** Merge the parent items into the items array. */ 
  387. $this->items = array_merge( $this->items, array_reverse( $parents ) ); 
  388. } 
  389. /** 
  390. * Adds a post's terms from a specific taxonomy to the items array. 
  391. * 
  392. * @since 0.6.0 
  393. * @access public 
  394. * @param int $post_id The ID of the post to get the terms for. 
  395. * @return void 
  396. */ 
  397. public function do_post_terms( $post_id ) { 
  398. /** Get the post type. */ 
  399. $post_type = get_post_type( $post_id ); 
  400. /** Add the terms of the taxonomy for this post. */ 
  401. if ( !empty( $this->args['post_taxonomy'][ $post_type ] ) ) 
  402. $this->items[] = get_the_term_list( $post_id, $this->args['post_taxonomy'][ $post_type ], '', ', ', '' ); 
  403. } 
  404. /** 
  405. * Adds a specific post's hierarchy to the items array. The hierarchy is determined by post type's 
  406. * rewrite arguments and whether it has an archive page. 
  407. * 
  408. * @since 0.6.0 
  409. * @access public 
  410. * @param int $post_id The ID of the post to get the hierarchy for. 
  411. * @return void 
  412. */ 
  413. public function do_post_hierarchy( $post_id ) { 
  414. /** Get the post type. */ 
  415. $post_type = get_post_type( $post_id ); 
  416. $post_type_object = get_post_type_object( $post_type ); 
  417. /** If this is the 'post' post type, get the rewrite front items and map the rewrite tags. */ 
  418. if ( 'post' === $post_type ) { 
  419. /** Add $wp_rewrite->front to the trail. */ 
  420. $this->do_rewrite_front_items(); 
  421. /** Map the rewrite tags. */ 
  422. $this->map_rewrite_tags( $post_id, get_option( 'permalink_structure' ) ); 
  423. } 
  424. /** If the post type has rewrite rules. */ 
  425. elseif ( false !== $post_type_object->rewrite ) { 
  426. /** If 'with_front' is true, add $wp_rewrite->front to the trail. */ 
  427. if ( $post_type_object->rewrite['with_front'] ) 
  428. $this->do_rewrite_front_items(); 
  429. /** If there's a path, check for parents. */ 
  430. if ( !empty( $post_type_object->rewrite['slug'] ) ) 
  431. $this->do_path_parents( $post_type_object->rewrite['slug'] ); 
  432. } 
  433. /** If there's an archive page, add it to the trail. */ 
  434. if ( !empty( $post_type_object->has_archive ) ) { 
  435. /** Add support for a non-standard label of 'archive_title' (special use case). */ 
  436. $label = !empty( $post_type_object->labels->archive_title ) ? $post_type_object->labels->archive_title : $post_type_object->labels->name; 
  437. $this->items[] = '<a href="' . get_post_type_archive_link( $post_type ) . '">' . $label . '</a>'; 
  438. } 
  439. } 
  440. /** 
  441. * Gets post types by slug. This is needed because the get_post_types() function doesn't exactly 
  442. * match the 'has_archive' argument when it's set as a string instead of a boolean. 
  443. * 
  444. * @since 0.6.0 
  445. * @access public 
  446. * @param int $slug The post type archive slug to search for. 
  447. * @return void 
  448. */ 
  449. public function get_post_types_by_slug( $slug ) { 
  450. $return = array(); 
  451. $post_types = get_post_types( array(), 'objects' ); 
  452. foreach ( $post_types as $type ) { 
  453. if ( $slug === $type->has_archive || ( true === $type->has_archive && $slug === $type->rewrite['slug'] ) ) 
  454. $return[] = $type; 
  455. } 
  456. return $return; 
  457. } 
  458. /** 
  459. * Adds the items to the trail items array for taxonomy term archives. 
  460. * 
  461. * @since 0.6.0 
  462. * @access public 
  463. * @global object $wp_rewrite 
  464. * @return void 
  465. */ 
  466. public function do_term_archive_items() { 
  467. global $wp_rewrite; 
  468. /** Get some taxonomy and term variables. */ 
  469. $term = get_queried_object(); 
  470. $taxonomy = get_taxonomy( $term->taxonomy ); 
  471. /** If there are rewrite rules for the taxonomy. */ 
  472. if ( false !== $taxonomy->rewrite ) { 
  473. /** If 'with_front' is true, dd $wp_rewrite->front to the trail. */ 
  474. if ( $taxonomy->rewrite['with_front'] && $wp_rewrite->front ) 
  475. $this->do_rewrite_front_items(); 
  476. /** Get parent pages by path if they exist. */ 
  477. $this->do_path_parents( $taxonomy->rewrite['slug'] ); 
  478. /** Add post type archive if its 'has_archive' matches the taxonomy rewrite 'slug'. */ 
  479. if ( $taxonomy->rewrite['slug'] ) { 
  480. $slug = trim( $taxonomy->rewrite['slug'], '/' ); 
  481. /** 
  482. * Deals with the situation if the slug has a '/' between multiple strings. For 
  483. * example, "movies/genres" where "movies" is the post type archive. 
  484. */ 
  485. $matches = explode( '/', $slug ); 
  486. /** If matches are found for the path. */ 
  487. if ( isset( $matches ) ) { 
  488. /** Reverse the array of matches to search for posts in the proper order. */ 
  489. $matches = array_reverse( $matches );

  733. /** 
  734. * Get parent posts by path. Currently, this method only supports getting parents of the 'page' 
  735. * post type. The goal of this function is to create a clear path back to home given what would 
  736. * normally be a "ghost" directory. If any page matches the given path, it'll be added. 
  737. * 
  738. * @since 0.6.0 
  739. * @access public 
  740. * @param string $path The path (slug) to search for posts by. 
  741. * @return void 
  742. */ 
  743. function do_path_parents( $path ) { 
  744. /** Trim '/' off $path in case we just got a simple '/' instead of a real path. */ 
  745. $path = trim( $path, '/' ); 
  746. /** If there's no path, return. */ 
  747. if ( empty( $path ) ) 
  748. return; 
  749. /** Get parent post by the path. */ 
  750. $post = get_page_by_path( $path ); 
  751. if ( !empty( $post ) ) { 
  752. $this->do_post_parents( $post->ID ); 
  753. } 
  754. elseif ( is_null( $post ) ) { 
  755. /** Separate post names into separate paths by '/'. */ 
  756. $path = trim( $path, '/' ); 
  757. preg_match_all( "/\/.*?\z/", $path, $matches ); 
  758. /** If matches are found for the path. */ 
  759. if ( isset( $matches ) ) { 
  760. /** Reverse the array of matches to search for posts in the proper order. */ 
  761. $matches = array_reverse( $matches ); 
  762. /** Loop through each of the path matches. */ 
  763. foreach ( $matches as $match ) { 
  764. /** If a match is found. */ 
  765. if ( isset( $match[0] ) ) { 
  766. /** Get the parent post by the given path. */ 
  767. $path = str_replace( $match[0], '', $path ); 
  768. $post = get_page_by_path( trim( $path, '/' ) ); 
  769. /** If a parent post is found, set the $post_id and break out of the loop. */ 
  770. if ( !empty( $post ) && 0 < $post->ID ) { 
  771. $this->do_post_parents( $post->ID ); 
  772. break; 
  773. } 
  774. } 
  775. } 
  776. } 
  777. } 
  778. }

  804. /** 
  805. * Turns %tag% from permalink structures into usable links for the breadcrumb trail. This feels kind of 
  806. * hackish for now because we're checking for specific %tag% examples and only doing it for the 'post' 
  807. * post type. In the future, maybe it'll handle a wider variety of possibilities, especially for custom post 
  808. * types. 
  809. * 
  810. * @since 0.6.0 
  811. * @access public 
  812. * @param int $post_id ID of the post whose parents we want. 
  813. * @param string $path Path of a potential parent page.

  817. public function map_rewrite_tags( $post_id, $path ) { 
  818. /** Get the post based on the post ID. */ 
  819. $post = get_post( $post_id ); 
  820. /** If no post is returned, an error is returned, or the post does not have a 'post' post type, return. */ 
  821. if ( empty( $post ) || is_wp_error( $post ) || 'post' !== $post->post_type ) 
  822. return $trail; 
  823. /** Trim '/' from both sides of the $path. */ 
  824. $path = trim( $path, '/' ); 
  825. /** Split the $path into an array of strings. */ 
  826. $matches = explode( '/', $path ); 
  827. /** If matches are found for the path. */ 
  828. if ( is_array( $matches ) ) { 
  829. /** Loop through each of the matches, adding each to the $trail array. */ 
  830. foreach ( $matches as $match ) {

  832. $tag = trim( $match, '/' ); 
  833. /** If using the %year% tag, add a link to the yearly archive. */ 
  834. if ( '%year%' == $tag )

  836. /** If using the %monthnum% tag, add a link to the monthly archive. */ 
  837. elseif ( '%monthnum%' == $tag ) 
  838. $this->items[] = '<a href="' . get_month_link( get_the_time( 'Y', $post_id ), get_the_time( 'm', $post_id ) ) . '">' . sprintf( $this->args['labels']['archive_month'], get_the_time( _x( 'F', 'monthly archives date format', 'onetone' ) ) ) . '</a>'; 
  839. /** If using the %day% tag, add a link to the daily archive. */ 
  840. elseif ( '%day%' == $tag ) 
  841. $this->items[] = '<a href="' . get_day_link( get_the_time( 'Y', $post_id ), get_the_time( 'm', $post_id ), get_the_time( 'd', $post_id ) ) . '">' . sprintf( $this->args['labels']['archive_day'], get_the_time( _x( 'j', 'daily archives date format', 'onetone' ) ) ) . '</a>'; 
  842. /** If using the %author% tag, add a link to the post author archive. */ 
  843. elseif ( '%author%' == $tag ) 
  844. $this->items[] = '<a href="' . get_author_posts_url( $post->post_author ) . '" title="' . esc_attr( get_the_author_meta( 'display_name', $post->post_author ) ) . '">' . get_the_author_meta( 'display_name', $post->post_author ) . '</a>'; 
  845. /** If using the %category% tag, add a link to the first category archive to match permalinks. */ 
  846. elseif ( '%category%' == $tag ) {

  851. /** Check that categories were returned. */

  860. $this->items[] = '<a href="' . get_term_link( $term, 'category' ) . '" title="' . esc_attr( $term->name ) . '">' . $term->name . '</a>'; 
  861. } 
  862. } 
  863. } 
  864. } 
  865. }
