  507.                 $plugins_to_activate[] = explode( ', ', $plugin ); 
  508.  
  509.             foreach ( $plugins_to_activate as $i => $array ) { 
  510.                 if ( ! preg_match( '|.php$|', $array[0] ) ) // Plugins that haven't been installed yet won't have the correct file path 
  511.                     unset( $plugins_to_activate[$i] ); 
  512.             } 
  513.  
  514.             /** Return early if there are no plugins to activate */ 
  515.             if ( empty( $plugins_to_activate ) ) 
  516.                 return; 
  517.  
  518.             $plugins = array(); 
  519.             $plugin_names = array(); 
  520.  
  521.             foreach ( $plugins_to_activate as $plugin_string ) { 
  522.                 $plugins[] = $plugin_string[0]; 
  523.                 $plugin_names[] = $plugin_string[2]; 
  524.             } 
  525.  
  526.             $count = count( $plugin_names ); // Count so we can use _n function 
  527.             $last_plugin = array_pop( $plugin_names ); // Pop off last name to prep for readability 
  528.             $imploded = empty( $plugin_names ) ? '<strong>' . $last_plugin . '</strong>' : '<strong>' . ( implode( ', ', $plugin_names ) . '</strong> and <strong>' . $last_plugin . '</strong>.' ); 
  529.  
  530.             /** Now we are good to go - let's start activating plugins */ 
  531.             $activate = activate_plugins( $plugins ); 
  532.  
  533.             if ( is_wp_error( $activate ) ) 
  534.                 echo '<div id="message" class="error"><p>' . $activate->get_error_message() . '</p></div>'; 
  535.             else 
  536.                 printf( '<div id="message" class="updated"><p>%1$s %2$s</p></div>', _n( 'The following plugin was activated successfully:', 'The following plugins were activated successfully:', $count, TGM_Plugin_Activation::$instance->domain ), $imploded ); 
  537.  
  538.              /** Update recently activated plugins option */ 
  539.             $recent = (array) get_option( 'recently_activated' ); 
  540.  
  541.             foreach ( $plugins as $plugin => $time ) 
  542.                 if ( isset( $recent[$plugin] ) ) 
  543.                     unset( $recent[$plugin] ); 
  544.  
  545.             update_option( 'recently_activated', $recent ); 
  546.  
  547.             unset( $_POST ); // Reset the $_POST variable in case user wants to perform one action after another 
  548.         } 
  549.     } 
  550.
