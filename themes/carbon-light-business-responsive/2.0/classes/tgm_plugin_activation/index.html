  502.             $plugin_to_activate = $plugin['slug'] . '/' . $plugin_file[0]; // Match plugin slug with appropriate plugin file 
  503.             $activate = activate_plugin( $plugin_to_activate ); // Activate the plugin 
  504.  
  505.             if ( is_wp_error( $activate ) ) { 
  506.                 echo '<div id="message" class="error"><p>' . $activate->get_error_message() . '</p></div>'; 
  507.                 echo '<p><a href="' . add_query_arg( 'page', $this->menu, admin_url( $this->parent_url_slug ) ) . '" title="' . esc_attr( $this->strings['return'] ) . '" target="_parent">' . __( $this->strings['return'], $this->domain ) . '</a></p>'; 
  508.                 return true; // End it here if there is an error with activation 
  509.             } 
  510.             else { 
  511.                 /** Make sure message doesn't display again if bulk activation is performed immediately after a single activation */ 
  512.                 if ( ! isset( $_POST[sanitize_key( 'action' )] ) ) { 
  513.                     $msg = sprintf( __( 'The following plugin was activated successfully: %s.', $this->domain ), '<strong>' . $plugin['name'] . '</strong>' ); 
  514.                     echo '<div id="message" class="updated"><p>' . $msg . '</p></div>'; 
  515.                 } 
  516.             } 
  517.         } 
  518.  
  519.         return false; 
  520.  
  521.     }

  523.     /** 
  524.      * Echoes required plugin notice. 
  525.      * 
  526.      * Outputs a message telling users that a specific plugin is required for 
  527.      * their theme. If appropriate, it includes a link to the form page where 
  528.      * users can install and activate the plugin. 
  529.      * 
  530.      * @since 1.0.0 
  531.      * 
  532.      * @global object $current_screen 
  533.      * @return null Returns early if we're on the Install page 
  534.      */ 
  535.     public function notices() { 
  536.  
  537.         global $current_screen; 
  538.  
  539.         /** Remove nag on the install page */ 
  540.         if ( $this->is_tgmpa_page() ) 
  541.             return; 
  542.  
  543.         $installed_plugins = get_plugins(); // Retrieve a list of all the plugins 
  544.         $this->populate_file_path(); 
  545.  
  546.         $message = array(); // Store the messages in an array to be outputted after plugins have looped through 
  547.         $install_link = false; // Set to false, change to true in loop if conditions exist, used for action link 'install' 
  548.         $install_link_count = 0; // Used to determine plurality of install action link text 
  549.         $activate_link = false; // Set to false, change to true in loop if conditions exist, used for action link 'activate' 
  550.         $activate_link_count = 0; // Used to determine plurality of activate action link text 
  551.  
  552.         foreach ( $this->plugins as $plugin ) { 
  553.             /** If the plugin is installed and active, check for minimum version argument before moving forward */ 
  554.             if ( is_plugin_active( $plugin['file_path'] ) ) { 
  555.                 /** A minimum version has been specified */ 
  556.                 if ( isset( $plugin['version'] ) ) { 
  557.                     if ( isset( $installed_plugins[$plugin['file_path']]['Version'] ) ) { 
  558.                         /** If the current version is less than the minimum required version, we display a message */ 
  559.                         if ( version_compare( $installed_plugins[$plugin['file_path']]['Version'], $plugin['version'], '<' ) ) { 
  560.                             if ( current_user_can( 'install_plugins' ) ) 
  561.                                 $message['notice_ask_to_update'][] = $plugin['name']; 
  562.                             else 
  563.                                 $message['notice_cannot_update'][] = $plugin['name']; 
  564.                         } 
  565.                     } 
  566.                     /** Can't find the plugin, so iterate to the next condition */ 
  567.                     else { 
  568.                         continue; 
  569.                     } 
  570.                 } 
  571.                 /** No minimum version specified, so iterate over the plugin */ 
  572.                 else { 
  573.                     continue; 
  574.                 } 
  575.             } 
  576.  
  577.             /** Not installed */ 
  578.             if ( ! isset( $installed_plugins[$plugin['file_path']] ) ) { 
  579.                 $install_link = true; // We need to display the 'install' action link 
  580.                 $install_link_count++; // Increment the install link count 
  581.                 if ( current_user_can( 'install_plugins' ) ) { 
  582.                     if ( $plugin['required'] ) 
  583.                         $message['notice_can_install_required'][] = $plugin['name']; 
  584.                     /** This plugin is only recommended */ 
  585.                     else 
  586.                         $message['notice_can_install_recommended'][] = $plugin['name']; 
  587.                 } 
  588.                 /** Need higher privileges to install the plugin */ 
  589.                 else { 
  590.                     $message['notice_cannot_install'][] = $plugin['name']; 
  591.                 } 
  592.             } 
  593.             /** Installed but not active */ 
  594.             elseif ( is_plugin_inactive( $plugin['file_path'] ) ) { 
  595.                 $activate_link = true; // We need to display the 'activate' action link 
  596.                 $activate_link_count++; // Increment the activate link count 
  597.                 if ( current_user_can( 'activate_plugins' ) ) { 
  598.                     if ( ( isset( $plugin['required'] ) ) && ( $plugin['required'] ) ) 
  599.                         $message['notice_can_activate_required'][] = $plugin['name']; 
  600.                     /** This plugin is only recommended */ 
  601.                     else { 
  602.                         $message['notice_can_activate_recommended'][] = $plugin['name']; 
  603.                     } 
  604.                 } 
  605.                 /** Need higher privileges to activate the plugin */ 
  606.                 else { 
  607.                     $message['notice_cannot_activate'][] = $plugin['name']; 
  608.                 } 
  609.             } 
  610.         } 
  611.  
  612.         /** Only process the nag messages if the user has not dismissed them already */ 
  613.         if ( ! get_user_meta( get_current_user_id(), 'tgmpa_dismissed_notice', true ) ) { 
  614.             /** If we have notices to display, we move forward */ 
  615.             if ( ! empty( $message ) ) { 
  616.                 krsort( $message ); // Sort messages 
  617.                 $rendered = ''; // Display all nag messages as strings 
  618.  
  619.                 /** Grab all plugin names */ 
  620.                 foreach ( $message as $type => $plugin_groups ) { 
  621.                     $linked_plugin_groups = array(); 
  622.  
  623.                     /** Count number of plugins in each message group to calculate singular/plural message */ 
  624.                     $count = count( $plugin_groups ); 
  625.  
  626.                     /** Loop through the plugin names to make the ones pulled from the .org repo linked */ 
  627.                     foreach ( $plugin_groups as $plugin_group_single_name ) { 
  628.                         $external_url = $this->_get_plugin_data_from_name( $plugin_group_single_name, 'external_url' ); 
  629.                         $source = $this->_get_plugin_data_from_name( $plugin_group_single_name, 'source' ); 
  630.  
  631.                         if ( $external_url && preg_match( '|^http(s)?://|', $external_url ) ) { 
  632.                             $linked_plugin_groups[] = '<a href="' . esc_url( $external_url ) . '" title="' . $plugin_group_single_name . '" target="_blank">' . $plugin_group_single_name . '</a>'; 
  633.                         } 
  634.                         elseif ( ! $source || preg_match( '|^http://wordpress.org/extend/plugins/|', $source ) ) { 
  635.                             $url = add_query_arg( 
  636.                                 array( 
  637.                                     'tab' => 'plugin-information',  
  638.                                     'plugin' => $this->_get_plugin_data_from_name( $plugin_group_single_name ),  
  639.                                     'TB_iframe' => 'true',  
  640.                                     'width' => '640',  
  641.                                     'height' => '500',  
  642.  ),  
  643.                                 admin_url( 'plugin-install.php' ) 
  644.  ); 
  645.  
  646.                             $linked_plugin_groups[] = '<a href="' . esc_url( $url ) . '" class="thickbox" title="' . $plugin_group_single_name . '">' . $plugin_group_single_name . '</a>'; 
  647.                         } 
  648.                         else { 
  649.                             $linked_plugin_groups[] = $plugin_group_single_name; // No hyperlink 
  650.                         } 
  651.  
  652.                         if ( isset( $linked_plugin_groups ) && (array) $linked_plugin_groups ) 
  653.                             $plugin_groups = $linked_plugin_groups; 
  654.                     } 
  655.  
  656.                     $last_plugin = array_pop( $plugin_groups ); // Pop off last name to prep for readability 
  657.                     $imploded = empty( $plugin_groups ) ? '<em>' . $last_plugin . '</em>' : '<em>' . ( implode( ', ', $plugin_groups ) . '</em> and <em>' . $last_plugin . '</em>' ); 
  658.  
  659.                     $rendered .= '<p>' . sprintf( translate_nooped_plural( $this->strings[$type], $count, $this->domain ), $imploded, $count ) . '</p>'; // All messages now stored 
  660.                 } 
  661.  
  662.                 /** Setup variables to determine if action links are needed */ 
  663.                 $show_install_link = $install_link ? '<a href="' . add_query_arg( 'page', $this->menu, admin_url( $this->parent_url_slug ) ) . '">' . translate_nooped_plural( $this->strings['install_link'], $install_link_count, $this->domain ) . '</a>' : ''; 
  664.                 $show_activate_link = $activate_link ? '<a href="' . admin_url( 'plugins.php' ) . '">' . translate_nooped_plural( $this->strings['activate_link'], $activate_link_count, $this->domain ) . '</a>'  : ''; 
  665.  
  666.                 /** Define all of the action links */ 
  667.                 $action_links = apply_filters( 
  668.                     'tgmpa_notice_action_links',  
  669.                     array( 
  670.                         'install' => ( current_user_can( 'install_plugins' ) ) ? $show_install_link : '',  
  671.                         'activate' => ( current_user_can( 'activate_plugins' ) ) ? $show_activate_link : '',  
  672.                         'dismiss' => '<a class="dismiss-notice" href="' . add_query_arg( 'tgmpa-dismiss', 'dismiss_admin_notices' ) . '" target="_parent">' . __( 'Dismiss this notice', $this->domain ) . '</a>',  
  673.  ) 
  674.  ); 
  675.  
  676.                 $action_links = array_filter( $action_links ); // Remove any empty array items 
  677.                 if ( $action_links ) 
  678.                     $rendered .= '<p>' . implode( ' | ', $action_links ) . '</p>'; 
  679.  
  680.                 /** Register the nag messages and prepare them to be processed */ 
  681.                    if ( isset( $this->strings['nag_type'] ) ) 
  682.                     add_settings_error( 'tgmpa', 'tgmpa', $rendered, sanitize_html_class( strtolower( $this->strings['nag_type'] ), 'updated' ) ); 
  683.                 else 
  684.                     add_settings_error( 'tgmpa', 'tgmpa', $rendered, 'updated' ); 
  685.             } 
  686.         } 
  687.  
  688.         /** Admin options pages already output settings_errors, so this is to avoid duplication */ 
  689.         if ( 'options-general' !== $current_screen->parent_base ) 
  690.             settings_errors( 'tgmpa' ); 
  691.  
  692.     }

  781.     /** 
  782.      * Helper function to extract the file path of the plugin file from the 
  783.      * plugin slug, if the plugin is installed. 
  784.      * 
  785.      * @since 2.0.0 
  786.      * 
  787.      * @param string $slug Plugin slug (typically folder name) as provided by the developer 
  788.      * @return string Either file path for plugin if installed, or just the plugin slug 
  789.      */ 
  790.     protected function _get_plugin_basename_from_slug( $slug ) { 
  791.  
  792.         $keys = array_keys( get_plugins() ); 
  793.  
  794.         foreach ( $keys as $key ) { 
  795.             if ( preg_match( '|^' . $slug .'|', $key ) ) 
  796.                 return $key; 
  797.         } 
  798.  
  799.         return $slug;

  804.      * Retrieve plugin data, given the plugin name. 
  805.      * 
  806.      * Loops through the registered plugins looking for $name. If it finds it,  
  807.      * it returns the $data from that plugin. Otherwise, returns false. 
  808.      * 
  809.      * @since 2.1.0 
  810.      * 
  811.      * @param string $name Name of the plugin, as it was registered 
  812.      * @param string $data Optional. Array key of plugin data to return. Default is slug 
  813.      * @return string|boolean Plugin slug if found, false otherwise. 
  814.      */ 
  815.     protected function _get_plugin_data_from_name( $name, $data = 'slug' ) { 
  816.  
  817.         foreach ( $this->plugins as $plugin => $values ) { 
  818.             if ( $name == $values['name'] && isset( $values[$data] ) ) 
  819.                 return $values[$data]; 
  820.         } 
  821.  
  822.         return false; 
  823.

  866.     /** 
  867.      * Forces plugin activation if the parameter 'force_activation' is 
  868.      * set to true. 
  869.      * 
  870.      * This allows theme authors to specify certain plugins that must be 
  871.      * active at all times while using the current theme. 
  872.      * 
  873.      * Please take special care when using this parameter as it has the 
  874.      * potential to be harmful if not used correctly. Setting this parameter 
  875.      * to true will not allow the specified plugin to be deactivated unless 
  876.      * the user switches themes. 
  877.      * 
  878.      * @since 2.2.0 
  879.      */ 
  880.     public function force_activation() { 
  881.  
  882.         /** Set file_path parameter for any installed plugins */ 
  883.         $this->populate_file_path(); 
  884.          
  885.         $installed_plugins = get_plugins(); 
  886.  
  887.         foreach ( $this->plugins as $plugin ) { 
  888.             /** Oops, plugin isn't there so iterate to next condition */ 
  889.             if ( isset( $plugin['force_activation'] ) && $plugin['force_activation'] && ! isset( $installed_plugins[$plugin['file_path']] ) ) 
  890.                 continue; 
  891.             /** There we go, activate the plugin */ 
  892.             elseif ( isset( $plugin['force_activation'] ) && $plugin['force_activation'] && is_plugin_inactive( $plugin['file_path'] ) ) 
  893.                 activate_plugin( $plugin['file_path'] ); 
  894.         }

  903.      * deactived upon switching from the current theme to another. 
  904.      * 
  905.      * Please take special care when using this parameter as it has the 
  906.      * potential to be harmful if not used correctly. 
  907.      * 
  908.      * @since 2.2.0 
  909.      */ 
  910.     public function force_deactivation() {

  913.         $this->populate_file_path(); 
  914.  
  915.         foreach ( $this->plugins as $plugin ) {

  917.             if ( isset( $plugin['force_deactivation'] ) && $plugin['force_deactivation'] && is_plugin_active( $plugin['file_path'] ) ) 
  918.                 deactivate_plugins( $plugin['file_path'] ); 
  919.         } 
  920.  
  921.     }
