  132.   function start_el(&$output, $item, $depth, $args) { 
  133.       global $wp_query; 
  134.       $indent = ( $depth ) ? str_repeat( "\t", $depth ) : ''; 
  135.  
  136.       $class_names = $value = ''; 
  137.  
  138.       $classes = empty( $item->classes ) ? array() : (array) $item->classes; 
  139.  
  140.       $class_names = join( ' ', apply_filters( 'nav_menu_css_class', array_filter( $classes ), $item ) ); 
  141.       $class_names = ' class="' . esc_attr( $class_names ) . '"'; 
  142.  
  143.       $output .= $indent . '<li id="menu-item-'. $item->ID . '"' . $value . $class_names .'>'; 
  144.  
  145.       $attributes = ! empty( $item->attr_title ) ? ' title="' . esc_attr( $item->attr_title ) .'"' : ''; 
  146.       $attributes .= ! empty( $item->target ) ? ' target="' . esc_attr( $item->target ) .'"' : ''; 
  147.       $attributes .= ! empty( $item->xfn ) ? ' rel="' . esc_attr( $item->xfn ) .'"' : ''; 
  148.       $attributes .= ! empty( $item->url ) ? ' href="' . esc_attr( $item->url ) .'"' : ''; 
  149.  
  150.       $item_output = $args->before; 
  151.       $item_output .= '<a'. $attributes .'>'; 
  152.       $item_output .= $args->link_before . apply_filters( 'the_title', $item->title, $item->ID ) . $args->link_after; 
  153.          $item_output .= '<br /><span class="subtitle">' . $item->description . '</span>'; 
  154.       $item_output .= '</a>'; 
  155.       $item_output .= $args->after; 
  156.  
  157.       $output .= apply_filters( 'walker_nav_menu_start_el', $item_output, $item, $depth, $args ); 
  158.   }

  288. /** When the post is saved, saves our custom data */ 
  289. function otw_save_postdata( $post_id ) { 
  290.  
  291. // First we need to check if the current user is authorised to do this action. 
  292. if ( isset($_REQUEST['post_type']) && $_REQUEST['post_type'] != 'page') { 
  293.   if ( ! current_user_can( 'edit_page', $post_id ) ) 
  294.       return; 
  295. } else { 
  296.   if ( ! current_user_can( 'edit_post', $post_id ) ) 
  297.       return; 
  298. } 
  299.  
  300.   global $post; 
  301.     if (defined('DOING_AUTOSAVE') && DOING_AUTOSAVE) { 
  302.       return; 
  303.     }; 
  304.  
  305. // Secondly we need to check if the user intended to change this value. 
  306. if ( ! isset( $_POST['otw_noncename'] ) || ! wp_verify_nonce( $_POST['otw_noncename'], plugin_basename( __FILE__ ) ) ) 
  307.     return; 
  308.  
  309.   update_post_meta($post->ID, "otw_head_title_setting", $_POST["otw_head_title_setting"]); 
  310.  
  311. $post_ID = $_POST['post_ID']; 
  312. //sanitize user input 
  313. $mydata = sanitize_text_field( $_POST['otw_page_sub_title'] ); 
  314.  
  315. add_post_meta($post_ID, '_otw_meta_value_key', $mydata, true) or 
  316. update_post_meta($post_ID, '_otw_meta_value_key', $mydata); 
  317.  
  318. }

  439. function otw_breadcrumb() { 
  440.  
  441.   /** === OPTIONS === */ 
  442.   $text['home'] = __('Home', 'otw-carbon-light'); // text for the 'Home' link 
  443.   $text['category'] = __('Archive for %s', 'otw-carbon-light'); // text for a category page 
  444.   $text['search'] = __('Search results for: %s', 'otw-carbon-light'); // text for a search results page 
  445.   $text['tag'] = __('Posts tagged %s', 'otw-carbon-light'); // text for a tag page 
  446.   $text['author'] = __('View all posts by %s', 'otw-carbon-light'); // text for an author page 
  447.   $text['404'] = __('Error 404', 'otw-carbon-light'); // text for the 404 page 
  448.  
  449.   $showCurrent = 1; // 1 - show current post/page title in breadcrumbs, 0 - don't show 
  450.   $showOnHome = 0; // 1 - show breadcrumbs on the homepage, 0 - don't show 
  451.   $delimiter = ' » '; // delimiter between crumbs 
  452.   $before = '<span class="current">'; // tag before the current crumb 
  453.   $after = '</span>'; // tag after the current crumb 
  454.   /** === END OF OPTIONS === */ 
  455.  
  456.   global $post; 
  457.   $homeLink = get_home_url('url') . '/'; 
  458.   $linkBefore = '<span typeof="v:Breadcrumb">'; 
  459.   $linkAfter = '</span>'; 
  460.   $linkAttr = ' rel="v:url" property="v:title"'; 
  461.   $link = $linkBefore . '<a' . $linkAttr . ' href="%1$s">%2$s</a>' . $linkAfter; 
  462.  
  463.   if (is_home() || is_front_page()) { 
  464.  
  465.       if ($showOnHome == 1) echo '<div id="breadcrumb"><a href="' . $homeLink . '">' . $text['home'] . '</a></div>'; 
  466.  
  467.   } else { 
  468.  
  469.       echo '<div id="breadcrumb">' . sprintf($link, $homeLink, $text['home']) . $delimiter; 
  470.  
  471.       if ( is_category() ) { 
  472.           $thisCat = get_category(get_query_var('cat'), false); 
  473.           if ($thisCat->parent != 0) { 
  474.               $cats = get_category_parents($thisCat->parent, TRUE, $delimiter); 
  475.               $cats = str_replace('<a', $linkBefore . '<a' . $linkAttr, $cats); 
  476.               $cats = str_replace('</a>', '</a>' . $linkAfter, $cats); 
  477.               echo $cats; 
  478.           } 
  479.           echo $before . sprintf($text['category'], single_cat_title('', false)) . $after; 
  480.  
  481.       } elseif ( is_search() ) { 
  482.           echo $before . sprintf($text['search'], get_search_query()) . $after;
