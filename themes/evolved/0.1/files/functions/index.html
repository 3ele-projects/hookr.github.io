  445. function evolved_formating($content) { 
  446.   $new_content = ''; 
  447.   $pattern_full = '{(\[raw\].*?\[/raw\])}is'; 
  448.   $pattern_contents = '{\[raw\](.*?)\[/raw\]}is'; 
  449.   $pieces = preg_split($pattern_full, $content, -1, PREG_SPLIT_DELIM_CAPTURE); 
  450.  
  451.   foreach ($pieces as $piece) { 
  452.       if (preg_match($pattern_contents, $piece, $matches)) { 
  453.           $new_content .= $matches[1]; 
  454.       } else { 
  455.           $new_content .= wptexturize(wpautop($piece)); 
  456.       } 
  457.   } 
  458.  
  459.   return $new_content; 
  460. }
