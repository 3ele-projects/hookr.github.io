  390.       function save_post( $post_id ) 
  391.       { 
  392.           global $post_type; 
  393.           $post_type_object = get_post_type_object( $post_type ); 
  394.  
  395.           // Check whether: 
  396.           // - the post is autosaved 
  397.           // - the post is a revision 
  398.           // - current post type is supported 
  399.           // - user has proper capability 
  400.           if ( 
  401.               ( defined( 'DOING_AUTOSAVE' ) && DOING_AUTOSAVE ) 
  402.               || ( ! isset( $_POST['post_ID'] ) || $post_id != $_POST['post_ID'] ) 
  403.               || ( ! in_array( $post_type, $this->meta_box['pages'] ) ) 
  404.               || ( ! current_user_can( $post_type_object->cap->edit_post, $post_id ) ) 
  405.  )

  408.           } 
  409.  
  410.           // Verify nonce 
  411.           check_admin_referer( "rwmb-save-{$this->meta_box['id']}", "nonce_{$this->meta_box['id']}" ); 
  412.  
  413.           foreach ( $this->fields as $field ) 
  414.           { 
  415.               $name = $field['id']; 
  416.               $old = get_post_meta( $post_id, $name, ! $field['multiple'] ); 
  417.               $new = isset( $_POST[ $name ] ) ? $_POST[ $name ] : ( $field['multiple'] ? array() : '' ); 
  418.  
  419.               // Allow field class change the value

  423.               // 1st filter applies to all fields with the same type

  425.               $new = apply_filters( "rwmb_{$field['type']}_value", $new, $field, $old ); 
  426.               $new = apply_filters( "rwmb_{$name}_value", $new, $field, $old ); 
  427.  
  428.               // Call defined method to save meta value, if there's no methods, call common one 
  429.               self::do_field_class_actions( $field, 'save', $new, $old, $post_id ); 
  430.           }

  527.       /**

  533.        *

  536.       static function apply_field_class_filters( $field, $method_name, $value ) 
  537.       { 
  538.           $args = array_slice( func_get_args(), 2 ); 
  539.           $args[] = $field;

  541.           // Call:     field class method 
  542.           // Fallback: RW_Meta_Box method

  545.           { 
  546.               $value = call_user_func_array( array( $class, $method_name ), $args ); 
  547.           } 
  548.           elseif ( method_exists( __CLASS__, $method_name ) ) 
  549.           { 
  550.               $value = call_user_func_array( array( __CLASS__, $method_name ), $args ); 
  551.           } 
  552.

  554.       }
