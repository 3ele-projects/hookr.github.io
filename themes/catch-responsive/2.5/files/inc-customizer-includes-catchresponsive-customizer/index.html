  241. /** 
  242.  * Returns list of color keys of array with default values for each color scheme as index 
  243.  * 
  244.  * @since Catch Base 2.1 
  245.  */ 
  246. function catchresponsive_color_list() { 
  247.   // Get default color scheme values 
  248.   $default = catchresponsive_get_default_theme_options(); 
  249.   // Get default dark color scheme valies 
  250.   $default_dark = catchresponsive_default_dark_color_options(); 
  251.  
  252.   $catchresponsive_color_list['background_color']['light'] = $default['background_color']; 
  253.   $catchresponsive_color_list['background_color']['dark'] = $default_dark['background_color']; 
  254.  
  255.   $catchresponsive_color_list['header_textcolor']['light'] = $default['header_textcolor']; 
  256.   $catchresponsive_color_list['header_textcolor']['dark'] = $default_dark['header_textcolor']; 
  257.  
  258.   return $catchresponsive_color_list;
