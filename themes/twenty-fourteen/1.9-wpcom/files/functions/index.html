  307.  * Print the attached image with a link to the next attached image.

  309.  * @since Twenty Fourteen 1.0 
  310.  */ 
  311. function twentyfourteen_the_attached_image() { 
  312.   $post = get_post(); 
  313.   /** 
  314.    * Filter the default Twenty Fourteen attachment size. 
  315.    * 
  316.    * @since Twenty Fourteen 1.0 
  317.    * 
  318.    * @param array $dimensions { 
  319.    *     An array of height and width dimensions. 
  320.    * 
  321.    *     @type int $height Height of the image in pixels. Default 810. 
  322.    *     @type int $width  Width of the image in pixels. Default 810. 
  323.    * } 
  324.    */ 
  325.   $attachment_size = apply_filters( 'twentyfourteen_attachment_size', array( 810, 810 ) ); 
  326.   $next_attachment_url = wp_get_attachment_url(); 
  327.  
  328.   if ( $post->post_parent ) : // Only look for attachments if this attachment has a parent object. 
  329.   /** 
  330.    * Grab the IDs of all the image attachments in a gallery so we can get the URL 
  331.    * of the next adjacent image in a gallery, or the first image (if we're 
  332.    * looking at the last image in a gallery), or, in a gallery of one, just the 
  333.    * link to that image file. 
  334.    */ 
  335.   $attachment_ids = get_posts( array( 
  336.       'post_parent' => $post->post_parent,  
  337.       'fields' => 'ids',  
  338.       'numberposts' => -1,  
  339.       'post_status' => 'inherit',  
  340.       'post_type' => 'attachment',  
  341.       'post_mime_type' => 'image',  
  342.       'order' => 'ASC',  
  343.       'orderby' => 'menu_order ID',  
  344.  ) ); 
  345.  
  346.   // If there is more than 1 attachment in a gallery... 
  347.   if ( count( $attachment_ids ) > 1 ) { 
  348.       foreach ( $attachment_ids as $idx => $attachment_id ) { 
  349.           if ( $attachment_id == $post->ID ) { 
  350.               $next_id = $attachment_ids[ ( $idx + 1 ) % count( $attachment_ids ) ]; 
  351.               break; 
  352.           } 
  353.       } 
  354.  
  355.       // get the URL of the next image attachment... 
  356.       if ( $next_id ) { 
  357.           $next_attachment_url = get_attachment_link( $next_id ); 
  358.       } 
  359.  
  360.       // or get the URL of the first image attachment. 
  361.       else { 
  362.           $next_attachment_url = get_attachment_link( reset( $attachment_ids ) ); 
  363.       } 
  364.   } 
  365.   endif; // end post->post_parent check 
  366.  
  367.   printf( '<a href="%1$s" rel="attachment">%2$s</a>',  
  368.       esc_url( $next_attachment_url ),  
  369.       wp_get_attachment_image( $post->ID, $attachment_size ) 
  370.  ); 
  371. } 
  372. endif;
