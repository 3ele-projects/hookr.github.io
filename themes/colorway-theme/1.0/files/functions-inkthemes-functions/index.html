  360. function pagination($pages = '', $range = 2) 
  361. {   
  362.    $showitems = ($range * 2)+1;   
  363.    global $paged; 
  364.    if(empty($paged)) $paged = 1; 
  365.    if($pages == '') 
  366.    { 
  367.        global $wp_query; 
  368.        $pages = $wp_query->max_num_pages; 
  369.        if(!$pages) 
  370.        { 
  371.            $pages = 1; 
  372.        } 
  373.    }    
  374.    if(1 != $pages) 
  375.    { 
  376.        echo "<ul class='paging'>"; 
  377.        if($paged > 2 && $paged > $range+1 && $showitems < $pages) echo "<li><a href='".get_pagenum_link(1)."'>«</a></li>"; 
  378.        if($paged > 1 && $showitems < $pages) echo "<li><a href='".get_pagenum_link($paged - 1)."'>‹</a></li>"; 
  379.        for ($i=1; $i <= $pages; $i++) 
  380.        { 
  381.            if (1 != $pages &&( !($i >= $paged+$range+1 || $i <= $paged-$range-1) || $pages <= $showitems )) 
  382.            { 
  383.                echo ($paged == $i)? "<li><a href='".get_pagenum_link($i)."' class='current' >".$i."</a></li>":"<li><a href='".get_pagenum_link($i)."' class='inactive' >".$i."</a></li>"; 
  384.            } 
  385.        } 
  386.        if ($paged < $pages && $showitems < $pages) echo "<li><a href='".get_pagenum_link($paged + 1)."'>›</a></li>";   
  387.        if ($paged < $pages-1 &&  $paged+$range-1 < $pages && $showitems < $pages) echo "<li><a href='".get_pagenum_link($pages)."'>»</a></li>"; 
  388.        echo "</ul>\n"; 
  389.    }
