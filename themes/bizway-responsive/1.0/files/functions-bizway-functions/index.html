  309. function bizway_pagination($pages = '', $range = 2) { 
  310.   $showitems = ($range * 2) + 1; 
  311.   global $paged; 
  312.   if (empty($paged)) 
  313.       $paged = 1; 
  314.   if ($pages == '') { 
  315.       global $wp_query; 
  316.       $pages = $wp_query->max_num_pages; 
  317.       if (!$pages) { 
  318.           $pages = 1; 
  319.       } 
  320.   } 
  321.   if (1 != $pages) { 
  322.       echo "<ul class='paging'>"; 
  323.       if ($paged > 2 && $paged > $range + 1 && $showitems < $pages) 
  324.           echo "<li><a href='" . get_pagenum_link(1) . "'>«</a></li>"; 
  325.       if ($paged > 1 && $showitems < $pages) 
  326.           echo "<li><a href='" . get_pagenum_link($paged - 1) . "'>‹</a></li>"; 
  327.       for ($i = 1; $i <= $pages; $i++) { 
  328.           if (1 != $pages && (!($i >= $paged + $range + 1 || $i <= $paged - $range - 1) || $pages <= $showitems )) { 
  329.               echo ($paged == $i) ? "<li><a href='" . get_pagenum_link($i) . "' class='current' >" . $i . "</a></li>" : "<li><a href='" . get_pagenum_link($i) . "' class='inactive' >" . $i . "</a></li>"; 
  330.           } 
  331.       } 
  332.       if ($paged < $pages && $showitems < $pages) 
  333.           echo "<li><a href='" . get_pagenum_link($paged + 1) . "'>›</a></li>"; 
  334.       if ($paged < $pages - 1 && $paged + $range - 1 < $pages && $showitems < $pages) 
  335.           echo "<li><a href='" . get_pagenum_link($pages) . "'>»</a></li>"; 
  336.       echo "</ul>\n"; 
  337.   } 
  338. }

  429. function bizway_custom_trim_excerpt($length) { 
  430.   global $post; 
  431.   $explicit_excerpt = $post->post_excerpt; 
  432.    
  433.   if ( '' == $explicit_excerpt ) { 
  434.       $text = get_the_content(''); 
  435.       $text = apply_filters('the_content', $text); 
  436.       $text = str_replace(']]>', ']]>', $text); 
  437.   } 
  438.   else { 
  439.       $text = apply_filters( 'the_content', $explicit_excerpt ); 
  440.   } 
  441.    
  442.   $text = strip_shortcodes( $text ); // optional 
  443.   $text = strip_tags($text); 
  444.   $excerpt_length = $length; 
  445.   $words = explode(' ', $text, $excerpt_length + 1); 
  446.   if ( count( $words ) > $excerpt_length ) { 
  447.       array_pop($words); 
  448.       array_push($words, '[…]'); 
  449.       $text = implode(' ', $words); 
  450.       $text = apply_filters('the_excerpt', $text); 
  451.   } 
  452.   return $text; 
  453. }
