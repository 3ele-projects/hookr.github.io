  475. function advantage_theme_options_validate( $input ) { 
  476.   $theme_options = advantage_theme_options_array(); 
  477.   foreach ( $theme_options as $theme_option ) { 
  478.       switch ( $theme_option['type'] ) { 
  479.           case 'checkbox': 
  480.               if ( ! isset( $input[ $theme_option['name'] ] ) ) 
  481.                   $input[$theme_option['name']] = null; 
  482.               $input[ $theme_option['name'] ] = ( $input[ $theme_option['name'] ] == 1 ? 1 : 0 ); 
  483.               break; 
  484.           case 'text': 
  485.           case 'textarea': 
  486.               $input[ $theme_option['name'] ] = wp_kses_stripslashes( $input[ $theme_option['name'] ] ); 
  487.               break; 
  488.           case 'number':     
  489.               $input[ $theme_option['name'] ] = intval( $input[ $theme_option['name'] ] );     
  490.               break;                 
  491.           case 'url':     
  492.               $input[ $theme_option['name'] ] = esc_url_raw( $input[$theme_option['name'] ] );     
  493.               break; 
  494.           case 'color': 
  495.               $input[ $theme_option['name'] ] = sanitize_text_field( $input[ $theme_option['name'] ] );     
  496.               break; 
  497.       } 
  498.   } 
  499.   if ( $input['respbp'] < 960 ) 
  500.       $input['respbp'] = 960; 
  501.   $input['advantage_scheme_css'] = advantage_scheme_css( $input ); 
  502.   //Update Scheme Options 
  503.   $options = advantage_get_options(); 
  504.   if ( $input['colorscheme'] != $options['colorscheme'] ) { 
  505.       $scheme = $theme_options['colorscheme']['values'][ $input['colorscheme'] ]; 
  506.       $input['schemecss'] = $scheme['css'];  
  507.       foreach ( $scheme['options'] as $scheme_options ) 
  508.           $input[ $scheme_options['name'] ] = $scheme_options['value']; 
  509.   } 
  510.   return $input; 
  511. }

  513. function advantage_scheme_css($input) { 
  514.   $advantage_fonts = advantage_fonts_array(); 
  515.   $css = ''; 
  516.   if ( 1080 != $input['gridwidth'] ) { 
  517.       $css .= '.row, .contain-to-grid .top-bar {max-width: ' . $input['gridwidth'] . 'px; }' . "\n";     
  518.   } 
  519.   if ( 1 == $input['sidebarresp'] && 3 != $input['sidebarpos'] ) { 
  520.       if ( ( $input['content'] + $input['sidebar1'] + $input['sidebar2']) <= 12 ) { 
  521.           $pct = 8.547008547 * ( $input['sidebar1'] + $input['sidebar2'] ) - 2.764102564; 
  522.           $css .= '@media only screen and (min-width: 768px) '; 
  523.           $css .= 'and (max-width: ' . $input['respbp'] . 'px) {' . "\n"; 
  524.           $css .= '#sidebar_one, #sidebar_two { width: ' . $pct; 
  525.           $css .= '%; float: left; }}' . "\n"; 
  526.       } 
  527.   } 
  528.   if ( 1 == $input['sidebarresp'] ) { 
  529.       $pct = 48.71794872; 
  530.       $css .= '@media only screen and (min-width: 481px) '; 
  531.       $css .= 'and (max-width: 767px) {' . "\n"; 
  532.       $css .= '#sidebar_one, #sidebar_two { width: ' . $pct; 
  533.       $css .= '%;  float: left; } #sidebar_two { margin-left: 2.564102564%; } }' . "\n"; 
  534.   } 
  535.   if ( $input['bodyfont'] > 0 ) 
  536.       $css .= 'body {font-family:' . $advantage_fonts[ $input['bodyfont'] ]['family'] . ';}' . "\n"; 
  537.   if ( $input['headingfont'] > 0 ) 
  538.       $css .= 'h1, h2, h3, h4, h5, h6 {font-family:' . $advantage_fonts[$input['headingfont']]['family'] . ';}' . "\n"; 
  539.   if ( $input['entrytitlefont'] > 0 ) 
  540.       $css .= '.entry-title {font-family:' . $advantage_fonts[$input['entrytitlefont']]['family'] . ';}' . "\n"; 
  541.   if ( $input['sitetitlefont'] > 0 ) 
  542.       $css .= '#site-title {font-family:' . $advantage_fonts[$input['sitetitlefont']]['family'] . ';}' . "\n"; 
  543.   if ( $input['sitedescfont'] > 0 ) 
  544.       $css .= '#site-description {font-family:' . $advantage_fonts[$input['sitedescfont']]['family'] . ';}' . "\n"; 
  545.   if ( $input['widgettitlefont'] > 0 ) 
  546.       $css .= '.widget-title {font-family:' . $advantage_fonts[$input['widgettitlefont']]['family'] . ';}' . "\n"; 
  547.   if ( $input['sidebarfont'] > 0 ) 
  548.       $css .= '.blog-widgets {font-family:' . $advantage_fonts[$input['sidebarfont']]['family'] . ';}' . "\n"; 
  549.   if ( $input['footerfont'] > 0 ) 
  550.       $css .= '#footer {font-family:' . $advantage_fonts[$input['footerfont']]['family'] . ';}' . "\n"; 
  551.   if ( $input['mainmenufont'] > 0 ) 
  552.       $css .= '#topbar {font-family:' . $advantage_fonts[$input['mainmenufont']]['family'] . ';}' . "\n"; 
  553. //Background 
  554.   if ( ! empty( $input['headerbg'] ) ) 
  555.       $css .= '.custom-background .site-header, .site-header {background:' .  $input['headerbg'] . ';}' . "\n"; 
  556.   if ( ! empty( $input['titlebarbg'] ) ) 
  557.       $css .= '.custom-background .titlebar, .titlebar {background:' .  $input['titlebarbg'] . ';}' . "\n"; 
  558.   if ( ! empty( $input['contentbg'] ) ) 
  559.       $css .= '.custom-background .content-area, .content-area {background:' .  $input['contentbg'] . ';}' . "\n"; 
  560.   if ( ! empty( $input['footerbg'] ) ) 
  561.       $css .= '.custom-background #footer, #footer {background:' .  $input['footerbg'] . ';}' . "\n"; 
  562.   return apply_filters( 'advantage_scheme_css', $css ); 
  563. }

  565. function xinwp_option_display( $option_name ) { 
  566.   global $advantage_options, $advantage_theme_options, $advantage_fonts; 
  567.   $theme_option = $advantage_theme_options[ $option_name ]; 
  568.   $options = $advantage_options; 
  569.   if ( $theme_option['type'] != 'hidden' && empty( $theme_option['fieldonly'] ) ) { 
  570.       if ( isset( $theme_option['label'] ) ) { 
  571.           echo '<div class="grid_3 alpha">';     
  572.           echo '<p><b>' . $theme_option['label'] . '</b></p></div>';         
  573.       } 
  574.       echo '<div class="grid_9"><p>';         
  575.   } 
  576.   switch ( $theme_option['type'] ) { 
  577.       case 'radio': 
  578.           $values = $theme_option['values']; 
  579.           foreach ( $values as $value ) { 
  580.               printf( '<input id="advantage_theme_options[%1$s]_%2$s" name="advantage_theme_options[%1$s]" type="radio" value="%2$s" %3$s />',  
  581.                   $theme_option['name'],  
  582.                    $value['key'],  
  583.                    checked( $value['key'], $options[$theme_option['name']], false ) ); 
  584.               printf( '<label class="description" for="advantage_theme_options[%1$s]_%2$s">%3$s</label>',  
  585.                   $theme_option['name'],  
  586.                   $value['key'],  
  587.                   esc_attr( $value['label'] )    ); 
  588.           } 
  589.           break; 
  590.       case 'checkbox': 
  591.           printf( '<input id="advantage_theme_options[%1$s]" name="advantage_theme_options[%1$s]" type="checkbox" value="1" %2$s />',  
  592.                   $theme_option['name'],  
  593.                    checked( '1', $options[$theme_option['name']], false ) ); 
  594.               printf( '<label class="description" for="advantage_theme_options[%1$s]">%2$s</label>',  
  595.                   $theme_option['name'],  
  596.                   esc_attr( $theme_option['desc'] )    ); 
  597.           break;                 
  598.       case 'url': 
  599.       case 'text': 
  600.           printf( '<input name="advantage_theme_options[%s]" type="text" value="%s" size="80" />',  
  601.                   $theme_option['name'],  
  602.                    esc_attr( $options[$theme_option['name']] ) ); 
  603.           break; 
  604.       case 'color': 
  605.           printf( '<input name="advantage_theme_options[%1$s]" type="text" value="%2$s" class="advantage-color-field" />',  
  606.                   $theme_option['name'],  
  607.                    esc_attr( $options[$theme_option['name']] ) ); 
  608.           break; 
  609.       case 'textarea': 
  610.           printf( '<textarea name="advantage_theme_options[%s]" cols="120" rows="%s">%s</textarea>',  
  611.                   $theme_option['name'],  
  612.                   $theme_option['row'],  
  613.                    esc_textarea( $options[ $theme_option['name'] ] ) ); 
  614.           break; 
  615.       case 'number': 
  616.           if ( ! empty( $theme_option['fieldonly'] ) && ! empty( $theme_option['label'] ) ) 
  617.               printf( '<label class="description">%s</label>', esc_attr( $theme_option['label'] ) ); 
  618.           printf( '<input name="advantage_theme_options[%s]" type="text" value="%s" size="4" />',  
  619.                   $theme_option['name'],  
  620.                    esc_attr( $options[ $theme_option['name'] ] ) ); 
  621.           if ( ! empty( $theme_option['desc'] ) ) 
  622.               printf( '<label class="description">%s</label>', esc_attr( $theme_option['desc'] ) ); 
  623.           echo '    '; 
  624.           break; 
  625.       case 'select': 
  626.           printf( '<select name="advantage_theme_options[%s]" >',  
  627.               $theme_option['name'] ); 
  628.           foreach ( $theme_option['values'] as $value ) { 
  629.               printf ('<option value="%1$s" %2$s>%3$s</option>',  
  630.                   $value['key'],  
  631.                   selected( $options[$theme_option['name']], $value['key'], false ),  
  632.                   $value['label'] ); 
  633.           } 
  634.           echo '</select>'; 
  635.           break; 
  636.       case 'font': 
  637.           printf( '<select style="font-family:%2$s;font-size:14px;" name="advantage_theme_options[%1$s]" >',  
  638.               $theme_option['name'],  
  639.               $advantage_fonts[ $options[ $theme_option['name'] ] ]['family'] ); 
  640.           $old_font_type = ''; 
  641.           foreach ( $theme_option['values'] as $value ) { 
  642.               if ( $value['type'] != $old_font_type ) { 
  643.                   if ( $old_font_type != '' ) 
  644.                       echo '</optgroup>'; 
  645.                   printf( '<optgroup label="%1$s">', $value['type'] );                     
  646.               } 
  647.               printf( '<option style="font-family: %4$s;%5$s" value="%1$s" %2$s>%3$s</option>',  
  648.                   $value['key'],  
  649.                   selected( $options[ $theme_option['name'] ], $value['key'], false ),  
  650.                   $value['label'],  
  651.                   $value['family'],  
  652.                   ( empty( $value['url'] ) ? '' : 'color:blue;' ) ); 
  653.               $old_font_type = $value['type'];     
  654.           } 
  655.           echo '</optgroup>'; 
  656.           echo '</select>'; 
  657.           printf( '  <span style="font-family:%2$s;font-size:16px;%3$s">%1$s</span>',

  664.               $theme_option['name'] );

  672.                   $option->term_id,  
  673.                   selected( $selected_category, $option->term_id, false ),  
  674.                   $option->name ); 
  675.           } 
  676.           echo '</select>'; 
  677.           break; 
  678.       case 'hidden': 
  679.           printf( '<input id="%1$s" name="advantage_theme_options[%1$s]" type="hidden" value="%2$s" />',  
  680.                   $theme_option['name'],  
  681.                    esc_attr( $options[ $theme_option['name'] ] ) ); 
  682.           break; 
  683.       default: 
  684.           echo __( 'Not Availavle Yet', 'advantage' );             
  685.   } 
  686.   if ( $theme_option['type'] != 'hidden' && empty( $theme_option['fieldonly'] ) ) { 
  687.       echo '</p>'; 
  688.       if ( ! empty( $theme_option['helptext'] ) ) 
  689.           printf( '<p><label class="helptext">%s</label></p>', $theme_option['helptext']); 
  690.       echo '</div><div class="clear"></div>';     
  691.   } 
  692. }
