  511. function advantage_scripts_method() {     
  512.   global $advantage_options, $advantage_fonts; 
  513.    
  514.   $theme_uri = get_template_directory_uri(); 
  515.   // Check if the fonts are webfont, if yes, load the font. 
  516.   $advantage_fonts = advantage_fonts_array(); 
  517.   $font_elements = array( 
  518.           'bodyfont', 'headingfont', 'entrytitlefont',  
  519.           'sitetitlefont', 'sitedescfont', 'mainmenufont',  
  520.           'sidebarfont', 'widgettitlefont', 'footerfont' 
  521.  ); 
  522.   $fonts = array(); 
  523.   foreach ( $font_elements as $element ) { 
  524.       if ( $advantage_options[$element] > 0 
  525.               && ! in_array( $advantage_options[ $element ], $fonts) ) 
  526.           $fonts[] = $advantage_options[ $element ];         
  527.   } 
  528.  
  529.   foreach ( $fonts as $font ) { 
  530.       if ( ! empty( $advantage_fonts[ $font ]['url'] ) ) 
  531.           wp_enqueue_style( str_replace(' ', '', $advantage_fonts[ $font ]['label']), $advantage_fonts[ $font ]['url'], false, 1.0 ); 
  532.   } 
  533.    
  534.   wp_enqueue_style('bootstrap', $theme_uri . '/css/bootstrap.min.css', null, '2.3.1'); 
  535.   wp_enqueue_style('fontawesome', $theme_uri . '/css/font-awesome.min.css', array( 'bootstrap' ), '3.0.2' ); 
  536.   wp_enqueue_style('advantage', $theme_uri . '/dev/advantage.css', array( 'bootstrap', 'fontawesome' ), ADVANTAGE_VERSION); 
  537.   $child_pre = array( 'advantage' ); 
  538.   // Load Scheme CSS 
  539.   if ( ! empty( $advantage_options['schemecss'] ) ) { 
  540.       wp_enqueue_style('advantage-scheme', $advantage_options['schemecss'], $child_pre, ADVANTAGE_VERSION ); 
  541.       $child_pre[] = 'advantage-scheme';         
  542.   } 
  543.   //Load child theme's style.css 
  544.   if ( $theme_uri != get_stylesheet_directory_uri() ) 
  545.       wp_enqueue_style('advantage-child', get_stylesheet_uri(), $child_pre, ADVANTAGE_VERSION ); 
  546.  
  547.   // Page Template specific 
  548.   if ( is_page_template( 'pages/portfolio.php' ) ) { 
  549.       wp_enqueue_script( 'infinite-scroll' , $theme_uri . '/js/jquery.infinitescroll.min.js', array( 'jquery-masonry' ), '2.0', true );     
  550.       $advantage_dep[] = 'infinite-scroll'; 
  551.   }         
  552.   //Scripts 
  553.   if ( is_singular() && get_option( 'thread_comments' ) ) 
  554.       wp_enqueue_script( 'comment-reply' );     
  555.   wp_enqueue_script( 'modernizr' , $theme_uri . '/js/modernizr.custom.js', null ); 
  556.   wp_enqueue_script( 'bootstrap' , $theme_uri . '/js/bootstrap.min.js', array( 'jquery'), '2.3.1', true ); 
  557.   wp_enqueue_script( 'advantage' , $theme_uri . '/js/advantage.js', array( 'bootstrap'), ADVANTAGE_VERSION, true ); 
  558. } 
  559. if ( ! is_admin() ) 
  560.   add_action( 'wp_enqueue_scripts', 'advantage_scripts_method' );  
  561.

  564. require( get_template_directory() . '/inc/lib-general.php' ); 
  565. require( get_template_directory() . '/inc/lib-functions.php' ); 
  566. require( get_template_directory() . '/inc/lib-content.php' ); 
  567. require( get_template_directory() . '/inc/widgets.php' ); 
  568. require( get_template_directory() . '/scheme/scheme.php' ); 
  569. if ( is_admin() ) { 
  570.   require( get_template_directory() . '/inc/lib-admin.php' );
