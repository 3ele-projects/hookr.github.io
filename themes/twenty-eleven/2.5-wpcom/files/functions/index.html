  512.  * post permalink is used as a fallback. 
  513.  * 
  514.  * @since Twenty Eleven 1.0 
  515.  * 
  516.  * @uses get_url_in_content() to get the first URL from the post content. 
  517.  * 
  518.  * @return string The first link. 
  519.  */ 
  520. function twentyeleven_get_first_url() { 
  521.   $content = get_the_content(); 
  522.   $has_url = function_exists( 'get_url_in_content' ) ? get_url_in_content( $content ) : false; 
  523.  
  524.   if ( ! $has_url ) 
  525.       $has_url = twentyeleven_url_grabber(); 
  526.  
  527.   /** This filter is documented in wp-includes/link-template.php */ 
  528.   return ( $has_url ) ? $has_url : apply_filters( 'the_permalink', get_permalink() ); 
  529. }

  531. /** 
  532.  * Return the URL for the first link found in the post content. 
  533.  * 
  534.  * @since Twenty Eleven 1.0 
  535.  * 
  536.  * @return string|bool URL or false when no link is present. 
  537.  */ 
  538. function twentyeleven_url_grabber() { 
  539.   if ( ! preg_match( '/<a\s[^>]*?href=[\'"](.+?)[\'"]/is', get_the_content(), $matches ) ) 
  540.       return false; 
  541.  
  542.   return esc_url_raw( $matches[1] ); 
  543. }

  697. /** 
  698.  * Retrieve the IDs for images in a gallery. 
  699.  * 
  700.  * @uses get_post_galleries() First, if available. Falls back to shortcode parsing,  
  701.  *                            then as last option uses a get_posts() call. 
  702.  * 
  703.  * @since Twenty Eleven 1.6 
  704.  * 
  705.  * @return array List of image IDs from the post gallery. 
  706.  */ 
  707. function twentyeleven_get_gallery_images() { 
  708.   $images = array(); 
  709.  
  710.   if ( function_exists( 'get_post_galleries' ) ) { 
  711.       $galleries = get_post_galleries( get_the_ID(), false ); 
  712.       if ( isset( $galleries[0]['ids'] ) ) 
  713.           $images = explode( ', ', $galleries[0]['ids'] ); 
  714.   } else { 
  715.       $pattern = get_shortcode_regex(); 
  716.       preg_match( "/$pattern/s", get_the_content(), $match ); 
  717.       $atts = shortcode_parse_atts( $match[3] ); 
  718.       if ( isset( $atts['ids'] ) ) 
  719.           $images = explode( ', ', $atts['ids'] ); 
  720.   } 
  721.  
  722.   if ( ! $images ) { 
  723.       $images = get_posts( array( 
  724.           'fields' => 'ids',  
  725.           'numberposts' => 999,  
  726.           'order' => 'ASC',  
  727.           'orderby' => 'menu_order',  
  728.           'post_mime_type' => 'image',  
  729.           'post_parent' => get_the_ID(),  
  730.           'post_type' => 'attachment',  
  731.  ) ); 
  732.   } 
  733.  
  734.   return $images; 
  735. }
