  822.     protected function reduce($value, $inExp = false) { 
  823.         list($type) = $value; 
  824.         switch ($type) { 
  825.             case "exp": 
  826.                 list(, $op, $left, $right, $inParens) = $value; 
  827.                 $opName = isset(self::$operatorNames[$op]) ? self::$operatorNames[$op] : $op; 
  828.  
  829.                 $inExp = $inExp || $this->shouldEval($left) || $this->shouldEval($right); 
  830.  
  831.                 $left = $this->reduce($left, true); 
  832.                 $right = $this->reduce($right, true); 
  833.  
  834.                 // only do division in special cases 
  835.                 if ($opName == "div" && !$inParens && !$inExp) { 
  836.                     if ($left[0] != "color" && $right[0] != "color") { 
  837.                         return $this->expToString($value); 
  838.                     } 
  839.                 } 
  840.  
  841.                 $left = $this->coerceForExpression($left); 
  842.                 $right = $this->coerceForExpression($right); 
  843.  
  844.                 $ltype = $left[0]; 
  845.                 $rtype = $right[0]; 
  846.  
  847.                 // this tries: 
  848.                 // 1. op_[op name]_[left type]_[right type] 
  849.                 // 2. op_[left type]_[right type] (passing the op as first arg 
  850.                 // 3. op_[op name] 
  851.                 $fn = "op_${opName}_${ltype}_${rtype}"; 
  852.                 if (is_callable(array($this, $fn)) || 
  853.                     (($fn = "op_${ltype}_${rtype}") && 
  854.                         is_callable(array($this, $fn)) && 
  855.                         $passOp = true) || 
  856.                     (($fn = "op_${opName}") && 
  857.                         is_callable(array($this, $fn)) && 
  858.                         $genOp = true)) 
  859.                 { 
  860.                     $unitChange = false; 
  861.                     if (!isset($genOp) && 
  862.                         $left[0] == "number" && $right[0] == "number") 
  863.                     { 
  864.                         if ($opName == "mod" && $right[2] != "") { 
  865.                             $this->throwError("Cannot modulo by a number with units: $right[1]$right[2]."); 
  866.                         } 
  867.  
  868.                         $unitChange = true; 
  869.                         $emptyUnit = $left[2] == "" || $right[2] == ""; 
  870.                         $targetUnit = "" != $left[2] ? $left[2] : $right[2]; 
  871.  
  872.                         if ($opName != "mul") { 
  873.                             $left[2] = "" != $left[2] ? $left[2] : $targetUnit; 
  874.                             $right[2] = "" != $right[2] ? $right[2] : $targetUnit; 
  875.                         } 
  876.  
  877.                         if ($opName != "mod") { 
  878.                             $left = $this->normalizeNumber($left); 
  879.                             $right = $this->normalizeNumber($right); 
  880.                         } 
  881.  
  882.                         if ($opName == "div" && !$emptyUnit && $left[2] == $right[2]) { 
  883.                             $targetUnit = ""; 
  884.                         } 
  885.  
  886.                         if ($opName == "mul") { 
  887.                             $left[2] = "" != $left[2] ? $left[2] : $right[2]; 
  888.                             $right[2] = "" != $right[2] ? $right[2] : $left[2]; 
  889.                         } elseif ($opName == "div" && $left[2] == $right[2]) { 
  890.                             $left[2] = ""; 
  891.                             $right[2] = ""; 
  892.                         } 
  893.                     } 
  894.  
  895.                     $shouldEval = $inParens || $inExp; 
  896.                     if (isset($passOp)) { 
  897.                         $out = $this->$fn($op, $left, $right, $shouldEval); 
  898.                     } else { 
  899.                         $out = $this->$fn($left, $right, $shouldEval); 
  900.                     } 
  901.  
  902.                     if (isset($out)) { 
  903.                         if ($unitChange && $out[0] == "number") { 
  904.                             $out = $this->coerceUnit($out, $targetUnit); 
  905.                         } 
  906.                         return $out; 
  907.                     } 
  908.                 } 
  909.  
  910.                 return $this->expToString($value); 
  911.             case "unary": 
  912.                 list(, $op, $exp, $inParens) = $value; 
  913.                 $inExp = $inExp || $this->shouldEval($exp); 
  914.  
  915.                 $exp = $this->reduce($exp); 
  916.                 if ($exp[0] == "number") { 
  917.                     switch ($op) { 
  918.                     case "+": 
  919.                         return $exp; 
  920.                     case "-": 
  921.                         $exp[1] *= -1; 
  922.                         return $exp; 
  923.                     } 
  924.                 } 
  925.  
  926.                 if ($op == "not") { 
  927.                     if ($inExp || $inParens) { 
  928.                         if ($exp == self::$false) { 
  929.                             return self::$true; 
  930.                         } else { 
  931.                             return self::$false; 
  932.                         } 
  933.                     } else { 
  934.                         $op = $op . " "; 
  935.                     } 
  936.                 } 
  937.  
  938.                 return array("string", "", array($op, $exp)); 
  939.             case "var": 
  940.                 list(, $name) = $value; 
  941.                 return $this->reduce($this->get($name)); 
  942.             case "list": 
  943.                 foreach ($value[2] as &$item) { 
  944.                     $item = $this->reduce($item); 
  945.                 } 
  946.                 return $value; 
  947.             case "string": 
  948.                 foreach ($value[2] as &$item) { 
  949.                     if (is_array($item)) { 
  950.                         $item = $this->reduce($item); 
  951.                     } 
  952.                 } 
  953.                 return $value; 
  954.             case "interpolate": 
  955.                 $value[1] = $this->reduce($value[1]); 
  956.                 return $value; 
  957.             case "fncall": 
  958.                 list(, $name, $argValues) = $value; 
  959.  
  960.                 // user defined function? 
  961.                 $func = $this->get(self::$namespaces["function"] . $name, false); 
  962.                 if ($func) { 
  963.                     $this->pushEnv(); 
  964.  
  965.                     // set the args 
  966.                     if (isset($func->args)) { 
  967.                         $this->applyArguments($func->args, $argValues); 
  968.                     } 
  969.  
  970.                     // throw away lines and children 
  971.                     $tmp = (object)array( 
  972.                         "lines" => array(),  
  973.                         "children" => array() 
  974.  ); 
  975.                     $ret = $this->compileChildren($func->children, $tmp); 
  976.                     $this->popEnv(); 
  977.  
  978.                     return !isset($ret) ? self::$defaultValue : $ret; 
  979.                 } 
  980.  
  981.                 // built in function 
  982.                 if ($this->callBuiltin($name, $argValues, $returnValue)) { 
  983.                     return $returnValue; 
  984.                 } 
  985.  
  986.                 // need to flatten the arguments into a list 
  987.                 $listArgs = array(); 
  988.                 foreach ((array)$argValues as $arg) { 
  989.                     if (empty($arg[0])) { 
  990.                         $listArgs[] = $this->reduce($arg[1]); 
  991.                     } 
  992.                 } 
  993.                 return array("function", $name, array("list", ", ", $listArgs)); 
  994.             default: 
  995.                 return $value; 
  996.         } 
  997.     }

  1176.     /** 
  1177.      * Compiles a primitive value into a CSS property value. 
  1178.      * 
  1179.      * Values in scssphp are typed by being wrapped in arrays, their format is 
  1180.      * typically: 
  1181.      * 
  1182.      *     array(type, contents [, additional_contents]*) 
  1183.      * 
  1184.      * The input is expected to be reduced. This function will not work on 
  1185.      * things like expressions and variables. 
  1186.      * 
  1187.      * @param array $value 
  1188.      */ 
  1189.     protected function compileValue($value) { 
  1190.         $value = $this->reduce($value); 
  1191.  
  1192.         list($type) = $value; 
  1193.         switch ($type) { 
  1194.         case "keyword": 
  1195.             return $value[1]; 
  1196.         case "color": 
  1197.             // [1] - red component (either number for a %) 
  1198.             // [2] - green component 
  1199.             // [3] - blue component 
  1200.             // [4] - optional alpha component 
  1201.             list(, $r, $g, $b) = $value; 
  1202.  
  1203.             $r = round($r); 
  1204.             $g = round($g); 
  1205.             $b = round($b); 
  1206.  
  1207.             if (count($value) == 5 && $value[4] != 1) { // rgba 
  1208.                 return 'rgba('.$r.', '.$g.', '.$b.', '.$value[4].')'; 
  1209.             } 
  1210.  
  1211.             $h = sprintf("#%02x%02x%02x", $r, $g, $b);
