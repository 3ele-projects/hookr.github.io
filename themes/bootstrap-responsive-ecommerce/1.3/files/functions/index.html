  102. /** 
  103.  * Filter callback to add image sizes to Media Uploader 
  104.  * 
  105.  * WP 3.3 beta adds a new filter 'image_size_names_choose' to 
  106.  * the list of image sizes which are displayed in the Media Uploader 
  107.  * after an image has been uploaded. 
  108.  * 
  109.  * See image_size_input_fields() in wp-admin/includes/media.php 
  110.  *  
  111.  * 
  112.  * @uses get_intermediate_image_sizes() 
  113.  * 
  114.  * @param $sizes, array of default image sizes (associative array) 
  115.  * @return $new_sizes, array of all image sizes (associative array) 
  116.  */ 
  117. function sgr_display_image_size_names_muploader( $sizes ) { 
  118.    
  119.   $new_sizes = array(); 
  120.    
  121.   $added_sizes = get_intermediate_image_sizes(); 
  122.    
  123.   // $added_sizes is an indexed array, therefore need to convert it 
  124.   // to associative array, using $value for $key and $value 
  125.   foreach( $added_sizes as $key => $value) { 
  126.       $new_sizes[$value] = $value; 
  127.   } 
  128.    
  129.   // This preserves the labels in $sizes, and merges the two arrays 
  130.   $new_sizes = array_merge( $new_sizes, $sizes ); 
  131.    
  132.   return $new_sizes; 
  133. } 
  134. add_filter('image_size_names_choose', 'sgr_display_image_size_names_muploader', 11, 1);

  190.    * 
  191.    * To disable in a child theme, use wp_dequeue_style()

  193.    *     wp_dequeue_style( 'twentytwelve-fonts' );

  197.

  208.           $subsets .= ', cyrillic, cyrillic-ext'; 
  209.       elseif ( 'greek' == $subset ) 
  210.           $subsets .= ', greek, greek-ext'; 
  211.       elseif ( 'vietnamese' == $subset ) 
  212.           $subsets .= ', vietnamese';

  215.       $query_args = array( 
  216.           'family' => 'Open+Sans:400italic, 700italic, 400, 600, 700',  
  217.           'subset' => $subsets,  
  218.  ); 
  219.       wp_enqueue_style( 'twentytwelve-fonts', add_query_arg( $query_args, "$protocol://fonts.googleapis.com/css" ), array(), null );

  241. /** 
  242.  * Creates a nicely formatted and more specific title element text 
  243.  * for output in head of document, based on current view. 
  244.  * 
  245.  * @since Twenty Twelve 1.0 
  246.  * 
  247.  * @param string $title Default title text for current view. 
  248.  * @param string $sep Optional separator. 
  249.  * @return string Filtered title. 
  250.  */ 
  251. function twentytwelve_wp_title( $title, $sep ) { 
  252.   global $paged, $page; 
  253.  
  254.   if ( is_feed() ) 
  255.       return $title; 
  256.  
  257.   // Add the site name. 
  258.   $title .= get_bloginfo( 'name' ); 
  259.  
  260.   // Add the site description for the home/front page. 
  261.   $site_description = get_bloginfo( 'description', 'display' ); 
  262.   if ( $site_description && ( is_home() || is_front_page() ) ) 
  263.       $title = "$title $sep $site_description"; 
  264.  
  265.   // Add a page number if necessary. 
  266.   if ( $paged >= 2 || $page >= 2 ) 
  267.       $title = "$title $sep " . sprintf( __( 'Page %s', 'tcp-bootstrap-ecommerce' ), max( $paged, $page ) ); 
  268.  
  269.   return $title; 
  270. } 
  271. add_filter( 'wp_title', 'twentytwelve_wp_title', 10, 2 );
