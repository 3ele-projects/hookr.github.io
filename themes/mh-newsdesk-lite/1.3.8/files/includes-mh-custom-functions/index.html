  115. if (!function_exists('mh_newsdesk_lite_trim_excerpt')) { 
  116.   function mh_newsdesk_lite_trim_excerpt($text = '') { 
  117.       $raw_excerpt = $text; 
  118.       if ('' == $text) { 
  119.           $mh_newsdesk_lite_lite_options = mh_newsdesk_lite_theme_options(); 
  120.           $text = get_the_content(''); 
  121.           $text = strip_shortcodes($text); 
  122.           $text = apply_filters('the_content', $text); 
  123.           $text = str_replace(']]>', ']]>', $text); 
  124.           $excerpt_length = apply_filters('excerpt_length', $mh_newsdesk_lite_lite_options['excerpt_length']); 
  125.           $excerpt_more = apply_filters('excerpt_more', '...'); 
  126.           $text = wp_trim_words($text, $excerpt_length, $excerpt_more); 
  127.       } 
  128.       return apply_filters('wp_trim_excerpt', $text, $raw_excerpt); 
  129.   } 
  130. } 
  131. remove_filter('get_the_excerpt', 'wp_trim_excerpt'); 
  132. add_filter('get_the_excerpt', 'mh_newsdesk_lite_trim_excerpt');

  165.   function mh_newsdesk_lite_postnav() { 
  166.       global $post; 
  167.       $parent_post = get_post($post->post_parent); 
  168.       $attachment = is_attachment(); 
  169.       $previous = ($attachment) ? $parent_post : get_adjacent_post(false, '', true); 
  170.       $next = get_adjacent_post(false, '', false); 
  171.  
  172.       if (!$next && !$previous) 
  173.       return; 
  174.  
  175.       if ($attachment) { 
  176.           $attachments = get_children(array('post_type' => 'attachment', 'post_mime_type' => 'image', 'post_parent' => $parent_post->ID)); 
  177.           $count = count($attachments); 
  178.       } 
  179.       echo '<nav class="post-nav-wrap" role="navigation">' . "\n"; 
  180.           echo '<ul class="post-nav clearfix">' . "\n"; 
  181.               echo '<li class="post-nav-prev">' . "\n"; 
  182.                   if ($attachment) { 
  183.                       if ($count == 1) { 
  184.                           $permalink = get_permalink($parent_post); 
  185.                           echo '<a href="' . $permalink . '"><i class="fa fa-chevron-left"></i>' . __('Back to post', 'mh-newsdesk-lite') . '</a>'; 
  186.                       } else { 
  187.                           previous_image_link('%link', '<i class="fa fa-chevron-left"></i>' . __('Previous image', 'mh-newsdesk-lite')); 
  188.                       } 
  189.                   } else { 
  190.                       previous_post_link('%link', '<i class="fa fa-chevron-left"></i>' . __('Previous post', 'mh-newsdesk-lite')); 
  191.                   } 
  192.               echo '</li>' . "\n"; 
  193.               echo '<li class="post-nav-next">' . "\n"; 
  194.                   if ($attachment) { 
  195.                       next_image_link('%link', __('Next image', 'mh-newsdesk-lite') . '<i class="fa fa-chevron-right"></i>'); 
  196.                   } else { 
  197.                       next_post_link('%link', __('Next post', 'mh-newsdesk-lite') . '<i class="fa fa-chevron-right"></i>'); 
  198.                   }
