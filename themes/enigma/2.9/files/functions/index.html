  290.   function weblizar_breadcrumbs() { 
  291.   $delimiter = ''; 
  292.   $home = __('Home', 'enigma' ); // text for the 'Home' link 
  293.   $before = '<li>'; // tag before the current crumb 
  294.   $after = '</li>'; // tag after the current crumb 
  295.   echo '<ul class="breadcrumb">'; 
  296.   global $post; 
  297.   $homeLink = home_url(); 
  298.   echo '<li><a href="' . $homeLink . '">' . $home . '</a></li>' . $delimiter . ' '; 
  299.   if (is_category()) { 
  300.       global $wp_query; 
  301.       $cat_obj = $wp_query->get_queried_object(); 
  302.       $thisCat = $cat_obj->term_id; 
  303.       $thisCat = get_category($thisCat); 
  304.       $parentCat = get_category($thisCat->parent); 
  305.       if ($thisCat->parent != 0) 
  306.           echo(get_category_parents($parentCat, TRUE, ' ' . $delimiter . ' ')); 
  307.       echo $before . ' _e("Archive by category", "enigma") "' . single_cat_title('', false) . '"' . $after; 
  308.   } elseif (is_day()) { 
  309.       echo '<li><a href="' . get_year_link(get_the_time('Y')) . '">' . get_the_time('Y') . '</a></li> ' . $delimiter . ' '; 
  310.       echo '<li><a href="' . get_month_link(get_the_time('Y'), get_the_time('m')) . '">' . get_the_time('F') . '</a></li> ' . $delimiter . ' '; 
  311.       echo $before . get_the_time('d') . $after; 
  312.   } elseif (is_month()) { 
  313.       echo '<li><a href="' . get_year_link(get_the_time('Y')) . '">' . get_the_time('Y') . '</a></li> ' . $delimiter . ' '; 
  314.       echo $before . get_the_time('F') . $after; 
  315.   } elseif (is_year()) { 
  316.       echo $before . get_the_time('Y') . $after; 
  317.   } elseif (is_single() && !is_attachment()) { 
  318.       if (get_post_type() != 'post') { 
  319.           $post_type = get_post_type_object(get_post_type()); 
  320.           $slug = $post_type->rewrite; 
  321.           echo '<li><a href="' . $homeLink . '/' . $slug['slug'] . '/">' . $post_type->labels->singular_name . '</a></li> ' . $delimiter . ' '; 
  322.           echo $before . get_the_title() . $after; 
  323.       } else { 
  324.           $cat = get_the_category(); 
  325.           $cat = $cat[0]; 
  326.           //echo get_category_parents($cat, TRUE, ' ' . $delimiter . ' '); 
  327.           echo $before . get_the_title() . $after; 
  328.       } 
  329.        
  330.   } elseif (!is_single() && !is_page() && get_post_type() != 'post') { 
  331.       $post_type = get_post_type_object(get_post_type()); 
  332.       echo $before . $post_type->labels->singular_name . $after; 
  333.   } elseif (is_attachment()) { 
  334.       $parent = get_post($post->post_parent); 
  335.       $cat = get_the_category($parent->ID); 
  336.       //$cat = $cat[0]; 
  337.      // echo get_category_parents($cat, TRUE, ' ' . $delimiter . ' '); 
  338.       echo '<li><a href="' . get_permalink($parent) . '">' . $parent->post_title . '</a></li> ' . $delimiter . ' '; 
  339.       echo $before . get_the_title() . $after; 
  340.   } elseif (is_page() && !$post->post_parent) { 
  341.       echo $before . get_the_title() . $after; 
  342.   } elseif (is_page() && $post->post_parent) { 
  343.       $parent_id = $post->post_parent; 
  344.       $breadcrumbs = array(); 
  345.       while ($parent_id) { 
  346.           $page = get_page($parent_id); 
  347.           $breadcrumbs[] = '<li><a href="' . get_permalink($page->ID) . '">' . get_the_title($page->ID) . '</a></li>'; 
  348.           $parent_id = $page->post_parent; 
  349.       } 
  350.       $breadcrumbs = array_reverse($breadcrumbs); 
  351.       foreach ($breadcrumbs as $crumb) 
  352.           echo $crumb . ' ' . $delimiter . ' '; 
  353.       echo $before . get_the_title() . $after; 
  354.   } elseif (is_search()) { 
  355.       echo $before . _e("Search results for", "enigma")  . get_search_query() . '"' . $after; 
  356.  
  357.   } elseif (is_tag()) {         
  358.       echo $before . _e('Tag', 'enigma') . single_tag_title('', false) . $after; 
  359.   } elseif (is_author()) { 
  360.       global $author; 
  361.       $userdata = get_userdata($author); 
  362.       echo $before . _e("Articles posted by", "enigma") . $userdata->display_name . $after; 
  363.   } elseif (is_404()) { 
  364.       echo $before . _e("Error 404", "enigma") . $after; 
  365.   } 
  366.    
  367.   echo '</ul>'; 
  368.   }

  371.   //PAGINATION 
  372.       function weblizar_pagination($pages = '', $range = 2) 
  373. {   
  374.    $showitems = ($range * 2)+1;   
  375.  
  376.    global $paged; 
  377.    if(empty($paged)) $paged = 1; 
  378.  
  379.    if($pages == '') 
  380.    { 
  381.        global $wp_query; 
  382.        $pages = $wp_query->max_num_pages; 
  383.        if(!$pages) 
  384.        { 
  385.            $pages = 1; 
  386.        } 
  387.    }    
  388.  
  389.    if(1 != $pages) 
  390.    { 
  391.        echo "<div class='enigma_blog_pagination'><div class='enigma_blog_pagi'>"; 
  392.        if($paged > 2 && $paged > $range+1 && $showitems < $pages) echo "<a href='".get_pagenum_link(1)."'>«</a>"; 
  393.        if($paged > 1 && $showitems < $pages) echo "<a href='".get_pagenum_link($paged - 1)."'>‹</a>"; 
  394.  
  395.        for ($i=1; $i <= $pages; $i++) 
  396.        { 
  397.            if (1 != $pages &&( !($i >= $paged+$range+1 || $i <= $paged-$range-1) || $pages <= $showitems )) 
  398.            { 
  399.               echo ($paged == $i)? "<a class='active'>".$i."</a>":"<a href='".get_pagenum_link($i)."'>".$i."</a>"; 
  400.            } 
  401.        } 
  402.  
  403.        if ($paged < $pages && $showitems < $pages) echo "<a href='".get_pagenum_link($paged + 1)."'>›</a>";   
  404.        if ($paged < $pages-1 &&  $paged+$range-1 < $pages && $showitems < $pages) echo "<a href='".get_pagenum_link($pages)."'>»</a>"; 
  405.        echo "</div></div>"; 
  406.    } 
  407. }
