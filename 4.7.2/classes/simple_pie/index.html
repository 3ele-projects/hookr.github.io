  945.                         return false; 
  946.                     } 
  947.  
  948.                     if (isset($headers)) 
  949.                     {

  953.  
  954.                     // Cache the file if caching is enabled 
  955.                     if ($cache && !$cache->save($this))

  961.             } 
  962.         } 
  963.  
  964.         if (isset($parser)) 
  965.         { 
  966.             // We have an error, just set SimplePie_Misc::error to it and quit

  968.         } 
  969.         else 
  970.         { 
  971.             $this->error = 'The data could not be converted to UTF-8. You MUST have either the iconv or mbstring extension installed. Upgrading to PHP 5.x (which includes iconv) is highly recommended.'; 
  972.         } 
  973.  
  974.         $this->registry->call('Misc', 'error', array($this->error, E_USER_NOTICE, __FILE__, __LINE__)); 
  975.  
  976.         return false; 
  977.     }

  979.     /** 
  980.      * Fetch the data via SimplePie_File 
  981.      * 
  982.      * If the data is already cached, attempt to fetch it from there instead 
  983.      * @param SimplePie_Cache|false $cache Cache handler, or false to not load from the cache 
  984.      * @return array|true Returns true if the data was loaded from the cache, or an array of HTTP headers and sniffed type 
  985.      */

  987.     { 
  988.         // If it's enabled, use the cache 
  989.         if ($cache) 
  990.         { 
  991.             // Load the Cache 
  992.             $this->data = $cache->load(); 
  993.             if (!empty($this->data)) 
  994.             { 
  995.                 // If the cache is for an outdated build of SimplePie 
  996.                 if (!isset($this->data['build']) || $this->data['build'] !== SIMPLEPIE_BUILD) 
  997.                 { 
  998.                     $cache->unlink(); 
  999.                     $this->data = array(); 
  1000.                 } 
  1001.                 // If we've hit a collision just rerun it with caching disabled 
  1002.                 elseif (isset($this->data['url']) && $this->data['url'] !== $this->feed_url) 
  1003.                 { 
  1004.                     $cache = false; 
  1005.                     $this->data = array(); 
  1006.                 } 
  1007.                 // If we've got a non feed_url stored (if the page isn't actually a feed, or is a redirect) use that URL. 
  1008.                 elseif (isset($this->data['feed_url'])) 
  1009.                 { 
  1010.                     // If the autodiscovery cache is still valid use it. 
  1011.                     if ($cache->mtime() + $this->autodiscovery_cache_duration > time()) 
  1012.                     { 
  1013.                         // Do not need to do feed autodiscovery yet. 
  1014.                         if ($this->data['feed_url'] !== $this->data['url']) 
  1015.                         { 
  1016.                             $this->set_feed_url($this->data['feed_url']); 
  1017.                             return $this->init(); 
  1018.                         } 
  1019.  
  1020.                         $cache->unlink(); 
  1021.                         $this->data = array(); 
  1022.                     } 
  1023.                 } 
  1024.                 // Check if the cache has been updated 
  1025.                 elseif ($cache->mtime() + $this->cache_duration < time()) 
  1026.                 {

  1028.                     if (isset($this->data['headers']['last-modified']) || isset($this->data['headers']['etag'])) 
  1029.                     {

  1118.                 } 
  1119.                 if ($cache)
