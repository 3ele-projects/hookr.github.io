  521.         // Now, sideload it in. 
  522.         $file_data = array( 
  523.             'error' => null,  
  524.             'tmp_name' => $tmpfname,  
  525.             'name' => $filename,  
  526.             'type' => $type,  
  527.  ); 
  528.  
  529.         $overrides = array( 
  530.             'test_form' => false,  
  531.  ); 
  532.  
  533.         $sideloaded = wp_handle_sideload( $file_data, $overrides ); 
  534.  
  535.         if ( isset( $sideloaded['error'] ) ) { 
  536.             @unlink( $tmpfname ); 
  537.  
  538.             return new WP_Error( 'rest_upload_sideload_error', $sideloaded['error'], array( 'status' => 500 ) ); 
  539.         } 
  540.  
  541.         return $sideloaded; 
  542.     }

  544.     /** 
  545.      * Parses filename from a Content-Disposition header value. 
  546.      * 
  547.      * As per RFC6266: 
  548.      * 
  549.      *     content-disposition = "Content-Disposition" ":" 
  550.      *                            disposition-type *( ";" disposition-parm ) 
  551.      * 
  552.      *     disposition-type = "inline" | "attachment" | disp-ext-type 
  553.      *                         ; case-insensitive 
  554.      *     disp-ext-type = token 
  555.      * 
  556.      *     disposition-parm = filename-parm | disp-ext-parm 
  557.      * 
  558.      *     filename-parm = "filename" "=" value 
  559.      *                         | "filename*" "=" ext-value 
  560.      * 
  561.      *     disp-ext-parm = token "=" value 
  562.      *                         | ext-token "=" ext-value 
  563.      *     ext-token = <the characters in token, followed by "*"> 
  564.      * 
  565.      * @since 4.7.0 
  566.      * @access public 
  567.      * 
  568.      * @link http://tools.ietf.org/html/rfc2388 
  569.      * @link http://tools.ietf.org/html/rfc6266 
  570.      * 
  571.      * @param string[] $disposition_header List of Content-Disposition header values. 
  572.      * @return string|null Filename if available, or null if not found. 
  573.      */ 
  574.     public static function get_filename_from_disposition( $disposition_header ) { 
  575.         // Get the filename. 
  576.         $filename = null; 
  577.  
  578.         foreach ( $disposition_header as $value ) { 
  579.             $value = trim( $value ); 
  580.  
  581.             if ( strpos( $value, ';' ) === false ) { 
  582.                 continue; 
  583.             } 
  584.  
  585.             list( $type, $attr_parts ) = explode( ';', $value, 2 ); 
  586.  
  587.             $attr_parts = explode( ';', $attr_parts ); 
  588.             $attributes = array(); 
  589.  
  590.             foreach ( $attr_parts as $part ) { 
  591.                 if ( strpos( $part, '=' ) === false ) { 
  592.                     continue; 
  593.                 } 
  594.  
  595.                 list( $key, $value ) = explode( '=', $part, 2 ); 
  596.  
  597.                 $attributes[ trim( $key ) ] = trim( $value ); 
  598.             } 
  599.  
  600.             if ( empty( $attributes['filename'] ) ) { 
  601.                 continue; 
  602.             } 
  603.  
  604.             $filename = trim( $attributes['filename'] ); 
  605.  
  606.             // Unquote quoted filename, but after trimming. 
  607.             if ( substr( $filename, 0, 1 ) === '"' && substr( $filename, -1, 1 ) === '"' ) { 
  608.                 $filename = substr( $filename, 1, -1 ); 
  609.             } 
  610.         } 
  611.  
  612.         return $filename; 
  613.     }

  664.     /** 
  665.      * Handles an upload via multipart/form-data ($_FILES). 
  666.      * 
  667.      * @since 4.7.0 
  668.      * @access protected 
  669.      * 
  670.      * @param array $files   Data from the `$_FILES` superglobal. 
  671.      * @param array $headers HTTP headers from the request. 
  672.      * @return array|WP_Error Data from wp_handle_upload(). 
  673.      */ 
  674.     protected function upload_from_file( $files, $headers ) { 
  675.         if ( empty( $files ) ) { 
  676.             return new WP_Error( 'rest_upload_no_data', __( 'No data supplied.' ), array( 'status' => 400 ) ); 
  677.         } 
  678.  
  679.         // Verify hash, if given. 
  680.         if ( ! empty( $headers['content_md5'] ) ) { 
  681.             $content_md5 = array_shift( $headers['content_md5'] ); 
  682.             $expected = trim( $content_md5 ); 
  683.             $actual = md5_file( $files['file']['tmp_name'] ); 
  684.  
  685.             if ( $expected !== $actual ) { 
  686.                 return new WP_Error( 'rest_upload_hash_mismatch', __( 'Content hash did not match expected.' ), array( 'status' => 412 ) ); 
  687.             } 
  688.         } 
  689.  
  690.         // Pass off to WP to handle the actual upload. 
  691.         $overrides = array( 
  692.             'test_form' => false,  
  693.  ); 
  694.  
  695.         // Bypasses is_uploaded_file() when running unit tests. 
  696.         if ( defined( 'DIR_TESTDATA' ) && DIR_TESTDATA ) { 
  697.             $overrides['action'] = 'wp_handle_mock_upload'; 
  698.         } 
  699.  
  700.         /** Include admin functions to get access to wp_handle_upload() */ 
  701.         require_once ABSPATH . 'wp-admin/includes/admin.php'; 
  702.  
  703.         $file = wp_handle_upload( $files['file'], $overrides ); 
  704.  
  705.         if ( isset( $file['error'] ) ) { 
  706.             return new WP_Error( 'rest_upload_unknown_error', $file['error'], array( 'status' => 500 ) ); 
  707.         } 
  708.  
  709.         return $file; 
  710.     }
