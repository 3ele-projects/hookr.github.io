  224.                     $top_level_elements[] = $e;

  298.                 $start = $total_top - $end; 
  299.                 $end = $total_top - $oldstart;

  301.  
  302.             $empty_array = array(); 
  303.             foreach ( $elements as $e ) { 
  304.                 $count++;
