  171.      * Recursive-friendly query sanitizer. 
  172.      * 
  173.      * Ensures that each query-level clause has a 'relation' key, and that 
  174.      * each first-order clause contains all the necessary keys from 
  175.      * `$defaults`. 
  176.      * 
  177.      * @since 4.1.0

  185.     public function sanitize_query( $queries, $parent_query = null ) { 
  186.         $cleaned_query = array(); 
  187.  
  188.         $defaults = array( 
  189.             'column' => 'post_date',  
  190.             'compare' => '=',  
  191.             'relation' => 'AND',  
  192.  ); 
  193.  
  194.         // Numeric keys should always have array values. 
  195.         foreach ( $queries as $qkey => $qvalue ) { 
  196.             if ( is_numeric( $qkey ) && ! is_array( $qvalue ) ) { 
  197.                 unset( $queries[ $qkey ] ); 
  198.             } 
  199.         } 
  200.  
  201.         // Each query should have a value for each default key. Inherit from the parent when possible. 
  202.         foreach ( $defaults as $dkey => $dvalue ) { 
  203.             if ( isset( $queries[ $dkey ] ) ) { 
  204.                 continue; 
  205.             } 
  206.  
  207.             if ( isset( $parent_query[ $dkey ] ) ) { 
  208.                 $queries[ $dkey ] = $parent_query[ $dkey ]; 
  209.             } else { 
  210.                 $queries[ $dkey ] = $dvalue; 
  211.             } 
  212.         } 
  213.  
  214.         // Validate the dates passed in the query. 
  215.         if ( $this->is_first_order_clause( $queries ) ) { 
  216.             $this->validate_date_values( $queries ); 
  217.         } 
  218.  
  219.         foreach ( $queries as $key => $q ) { 
  220.             if ( ! is_array( $q ) || in_array( $key, $this->time_keys, true ) ) { 
  221.                 // This is a first-order query. Trust the values and sanitize when building SQL. 
  222.                 $cleaned_query[ $key ] = $q; 
  223.             } else { 
  224.                 // Any array without a time key is another query, so we recurse. 
  225.                 $cleaned_query[] = $this->sanitize_query( $q, $queries ); 
  226.             } 
  227.         } 
  228.  
  229.         return $cleaned_query; 
  230.     }

  571.     protected function get_sql_clauses() {

  581.     /** 
  582.      * Generate SQL clauses for a single query array. 
  583.      * 
  584.      * If nested subqueries are found, this method recurses the tree to 
  585.      * produce the properly nested SQL. 
  586.      * 
  587.      * @since 4.1.0 
  588.      * @access protected 
  589.      * 
  590.      * @param array $query Query to parse. 
  591.      * @param int   $depth Optional. Number of tree levels deep we currently are. 
  592.      *                     Used to calculate indentation. Default 0. 
  593.      * @return array { 
  594.      *     Array containing JOIN and WHERE SQL clauses to append to a single query array. 
  595.      * 
  596.      *     @type string $join  SQL fragment to append to the main JOIN clause. 
  597.      *     @type string $where SQL fragment to append to the main WHERE clause. 
  598.      * } 
  599.      */ 
  600.     protected function get_sql_for_query( $query, $depth = 0 ) { 
  601.         $sql_chunks = array( 
  602.             'join' => array(),  
  603.             'where' => array(),  
  604.  ); 
  605.  
  606.         $sql = array( 
  607.             'join' => '',  
  608.             'where' => '',  
  609.  ); 
  610.  
  611.         $indent = ''; 
  612.         for ( $i = 0; $i < $depth; $i++ ) { 
  613.             $indent .= "  "; 
  614.         } 
  615.  
  616.         foreach ( $query as $key => $clause ) { 
  617.             if ( 'relation' === $key ) { 
  618.                 $relation = $query['relation']; 
  619.             } elseif ( is_array( $clause ) ) { 
  620.  
  621.                 // This is a first-order clause. 
  622.                 if ( $this->is_first_order_clause( $clause ) ) { 
  623.                     $clause_sql = $this->get_sql_for_clause( $clause, $query ); 
  624.  
  625.                     $where_count = count( $clause_sql['where'] ); 
  626.                     if ( ! $where_count ) { 
  627.                         $sql_chunks['where'][] = ''; 
  628.                     } elseif ( 1 === $where_count ) { 
  629.                         $sql_chunks['where'][] = $clause_sql['where'][0]; 
  630.                     } else { 
  631.                         $sql_chunks['where'][] = '( ' . implode( ' AND ', $clause_sql['where'] ) . ' )'; 
  632.                     } 
  633.  
  634.                     $sql_chunks['join'] = array_merge( $sql_chunks['join'], $clause_sql['join'] ); 
  635.                 // This is a subquery, so we recurse. 
  636.                 } else { 
  637.                     $clause_sql = $this->get_sql_for_query( $clause, $depth + 1 ); 
  638.  
  639.                     $sql_chunks['where'][] = $clause_sql['where']; 
  640.                     $sql_chunks['join'][] = $clause_sql['join']; 
  641.                 } 
  642.             } 
  643.         } 
  644.  
  645.         // Filter to remove empties. 
  646.         $sql_chunks['join'] = array_filter( $sql_chunks['join'] ); 
  647.         $sql_chunks['where'] = array_filter( $sql_chunks['where'] ); 
  648.  
  649.         if ( empty( $relation ) ) { 
  650.             $relation = 'AND'; 
  651.         } 
  652.  
  653.         // Filter duplicate JOIN clauses and combine into a single string. 
  654.         if ( ! empty( $sql_chunks['join'] ) ) { 
  655.             $sql['join'] = implode( ' ', array_unique( $sql_chunks['join'] ) ); 
  656.         } 
  657.  
  658.         // Generate a single WHERE clause with proper brackets and indentation. 
  659.         if ( ! empty( $sql_chunks['where'] ) ) { 
  660.             $sql['where'] = '( ' . "\n  " . $indent . implode( ' ' . "\n  " . $indent . $relation . ' ' . "\n  " . $indent, $sql_chunks['where'] ) . "\n" . $indent . ')'; 
  661.         } 
  662.  
  663.         return $sql; 
  664.     }

  671.      * 
  672.      * @since  3.7.0 
  673.      * @access protected

  676.      * @return array { 
  677.      *     Array containing JOIN and WHERE SQL clauses to append to the main query. 
  678.      *

  687.     /** 
  688.      * Turns a first-order date query into SQL for a WHERE clause. 
  689.      * 
  690.      * @since  4.1.0 
  691.      * @access protected 
  692.      * 
  693.      * @param  array $query        Date query clause. 
  694.      * @param  array $parent_query Parent query of the current date query. 
  695.      * @return array { 
  696.      *     Array containing JOIN and WHERE SQL clauses to append to the main query. 
  697.      * 
  698.      *     @type string $join  SQL fragment to append to the main JOIN clause. 
  699.      *     @type string $where SQL fragment to append to the main WHERE clause. 
  700.      * } 
  701.      */ 
  702.     protected function get_sql_for_clause( $query, $parent_query ) { 
  703.         global $wpdb; 
  704.  
  705.         // The sub-parts of a $where part. 
  706.         $where_parts = array(); 
  707.  
  708.         $column = ( ! empty( $query['column'] ) ) ? esc_sql( $query['column'] ) : $this->column; 
  709.  
  710.         $column = $this->validate_column( $column ); 
  711.  
  712.         $compare = $this->get_compare( $query ); 
  713.  
  714.         $inclusive = ! empty( $query['inclusive'] ); 
  715.  
  716.         // Assign greater- and less-than values. 
  717.         $lt = '<'; 
  718.         $gt = '>'; 
  719.  
  720.         if ( $inclusive ) { 
  721.             $lt .= '='; 
  722.             $gt .= '='; 
  723.         } 
  724.  
  725.         // Range queries. 
  726.         if ( ! empty( $query['after'] ) ) { 
  727.             $where_parts[] = $wpdb->prepare( "$column $gt %s", $this->build_mysql_datetime( $query['after'], ! $inclusive ) ); 
  728.         } 
  729.         if ( ! empty( $query['before'] ) ) { 
  730.             $where_parts[] = $wpdb->prepare( "$column $lt %s", $this->build_mysql_datetime( $query['before'], $inclusive ) ); 
  731.         } 
  732.         // Specific value queries. 
  733.  
  734.         if ( isset( $query['year'] ) && $value = $this->build_value( $compare, $query['year'] ) ) 
  735.             $where_parts[] = "YEAR( $column ) $compare $value"; 
  736.  
  737.         if ( isset( $query['month'] ) && $value = $this->build_value( $compare, $query['month'] ) ) { 
  738.             $where_parts[] = "MONTH( $column ) $compare $value"; 
  739.         } elseif ( isset( $query['monthnum'] ) && $value = $this->build_value( $compare, $query['monthnum'] ) ) { 
  740.             $where_parts[] = "MONTH( $column ) $compare $value"; 
  741.         } 
  742.         if ( isset( $query['week'] ) && false !== ( $value = $this->build_value( $compare, $query['week'] ) ) ) { 
  743.             $where_parts[] = _wp_mysql_week( $column ) . " $compare $value"; 
  744.         } elseif ( isset( $query['w'] ) && false !== ( $value = $this->build_value( $compare, $query['w'] ) ) ) { 
  745.             $where_parts[] = _wp_mysql_week( $column ) . " $compare $value"; 
  746.         } 
  747.         if ( isset( $query['dayofyear'] ) && $value = $this->build_value( $compare, $query['dayofyear'] ) ) 
  748.             $where_parts[] = "DAYOFYEAR( $column ) $compare $value"; 
  749.  
  750.         if ( isset( $query['day'] ) && $value = $this->build_value( $compare, $query['day'] ) ) 
  751.             $where_parts[] = "DAYOFMONTH( $column ) $compare $value"; 
  752.  
  753.         if ( isset( $query['dayofweek'] ) && $value = $this->build_value( $compare, $query['dayofweek'] ) ) 
  754.             $where_parts[] = "DAYOFWEEK( $column ) $compare $value"; 
  755.  
  756.         if ( isset( $query['dayofweek_iso'] ) && $value = $this->build_value( $compare, $query['dayofweek_iso'] ) ) 
  757.             $where_parts[] = "WEEKDAY( $column ) + 1 $compare $value"; 
  758.  
  759.         if ( isset( $query['hour'] ) || isset( $query['minute'] ) || isset( $query['second'] ) ) { 
  760.             // Avoid notices. 
  761.             foreach ( array( 'hour', 'minute', 'second' ) as $unit ) { 
  762.                 if ( ! isset( $query[ $unit ] ) ) { 
  763.                     $query[ $unit ] = null; 
  764.                 } 
  765.             } 
  766.  
  767.             if ( $time_query = $this->build_time_query( $column, $compare, $query['hour'], $query['minute'], $query['second'] ) ) { 
  768.                 $where_parts[] = $time_query; 
  769.             } 
  770.         } 
  771.  
  772.         /** 
  773.          * Return an array of 'join' and 'where' for compatibility 
  774.          * with other query classes. 
  775.          */ 
  776.         return array( 
  777.             'where' => $where_parts,  
  778.             'join' => array(),  
  779.  ); 
  780.     }

  782.     /** 
  783.      * Builds and validates a value string based on the comparison operator. 
  784.      * 
  785.      * @since 3.7.0 
  786.      * @access public 
  787.      * 
  788.      * @param string $compare The compare operator to use 
  789.      * @param string|array $value The value 
  790.      * @return string|false|int The value to be used in SQL or false on error. 
  791.      */ 
  792.     public function build_value( $compare, $value ) { 
  793.         if ( ! isset( $value ) ) 
  794.             return false; 
  795.  
  796.         switch ( $compare ) { 
  797.             case 'IN': 
  798.             case 'NOT IN': 
  799.                 $value = (array) $value; 
  800.  
  801.                 // Remove non-numeric values. 
  802.                 $value = array_filter( $value, 'is_numeric' ); 
  803.  
  804.                 if ( empty( $value ) ) { 
  805.                     return false; 
  806.                 } 
  807.  
  808.                 return '(' . implode( ', ', array_map( 'intval', $value ) ) . ')'; 
  809.  
  810.             case 'BETWEEN': 
  811.             case 'NOT BETWEEN': 
  812.                 if ( ! is_array( $value ) || 2 != count( $value ) ) { 
  813.                     $value = array( $value, $value ); 
  814.                 } else { 
  815.                     $value = array_values( $value ); 
  816.                 } 
  817.  
  818.                 // If either value is non-numeric, bail. 
  819.                 foreach ( $value as $v ) { 
  820.                     if ( ! is_numeric( $v ) ) { 
  821.                         return false; 
  822.                     } 
  823.                 } 
  824.  
  825.                 $value = array_map( 'intval', $value ); 
  826.  
  827.                 return $value[0] . ' AND ' . $value[1]; 
  828.  
  829.             default; 
  830.                 if ( ! is_numeric( $value ) ) { 
  831.                     return false; 
  832.                 } 
  833.  
  834.                 return (int) $value; 
  835.         } 
  836.     }

  838.     /** 
  839.      * Builds a MySQL format date/time based on some query parameters. 
  840.      * 
  841.      * You can pass an array of values (year, month, etc.) with missing parameter values being defaulted to 
  842.      * either the maximum or minimum values (controlled by the $default_to parameter). Alternatively you can 
  843.      * pass a string that will be run through strtotime(). 
  844.      * 
  845.      * @since 3.7.0 
  846.      * @access public 
  847.      * 
  848.      * @param string|array $datetime       An array of parameters or a strotime() string 
  849.      * @param bool         $default_to_max Whether to round up incomplete dates. Supported by values 
  850.      *                                     of $datetime that are arrays, or string values that are a 
  851.      *                                     subset of MySQL date format ('Y', 'Y-m', 'Y-m-d', 'Y-m-d H:i'). 
  852.      *                                     Default: false. 
  853.      * @return string|false A MySQL format date/time or false on failure 
  854.      */ 
  855.     public function build_mysql_datetime( $datetime, $default_to_max = false ) { 
  856.         $now = current_time( 'timestamp' ); 
  857.  
  858.         if ( ! is_array( $datetime ) ) { 
  859.  
  860.             /** 
  861.              * Try to parse some common date formats, so we can detect 
  862.              * the level of precision and support the 'inclusive' parameter. 
  863.              */ 
  864.             if ( preg_match( '/^(\d{4})$/', $datetime, $matches ) ) { 
  865.                 // Y 
  866.                 $datetime = array( 
  867.                     'year' => intval( $matches[1] ),  
  868.  ); 
  869.  
  870.             } elseif ( preg_match( '/^(\d{4})\-(\d{2})$/', $datetime, $matches ) ) { 
  871.                 // Y-m 
  872.                 $datetime = array( 
  873.                     'year' => intval( $matches[1] ),  
  874.                     'month' => intval( $matches[2] ),  
  875.  ); 
  876.  
  877.             } elseif ( preg_match( '/^(\d{4})\-(\d{2})\-(\d{2})$/', $datetime, $matches ) ) { 
  878.                 // Y-m-d 
  879.                 $datetime = array( 
  880.                     'year' => intval( $matches[1] ),  
  881.                     'month' => intval( $matches[2] ),  
  882.                     'day' => intval( $matches[3] ),  
  883.  ); 
  884.  
  885.             } elseif ( preg_match( '/^(\d{4})\-(\d{2})\-(\d{2}) (\d{2}):(\d{2})$/', $datetime, $matches ) ) { 
  886.                 // Y-m-d H:i 
  887.                 $datetime = array( 
  888.                     'year' => intval( $matches[1] ),  
  889.                     'month' => intval( $matches[2] ),  
  890.                     'day' => intval( $matches[3] ),  
  891.                     'hour' => intval( $matches[4] ),  
  892.                     'minute' => intval( $matches[5] ),  
  893.  ); 
  894.             } 
  895.  
  896.             // If no match is found, we don't support default_to_max. 
  897.             if ( ! is_array( $datetime ) ) { 
  898.                 // @todo Timezone issues here possibly 
  899.                 return gmdate( 'Y-m-d H:i:s', strtotime( $datetime, $now ) ); 
  900.             } 
  901.         } 
  902.  
  903.         $datetime = array_map( 'absint', $datetime ); 
  904.  
  905.         if ( ! isset( $datetime['year'] ) ) 
  906.             $datetime['year'] = gmdate( 'Y', $now ); 
  907.  
  908.         if ( ! isset( $datetime['month'] ) ) 
  909.             $datetime['month'] = ( $default_to_max ) ? 12 : 1; 
  910.  
  911.         if ( ! isset( $datetime['day'] ) ) 
  912.             $datetime['day'] = ( $default_to_max ) ? (int) date( 't', mktime( 0, 0, 0, $datetime['month'], 1, $datetime['year'] ) ) : 1; 
  913.  
  914.         if ( ! isset( $datetime['hour'] ) ) 
  915.             $datetime['hour'] = ( $default_to_max ) ? 23 : 0; 
  916.  
  917.         if ( ! isset( $datetime['minute'] ) ) 
  918.             $datetime['minute'] = ( $default_to_max ) ? 59 : 0; 
  919.  
  920.         if ( ! isset( $datetime['second'] ) ) 
  921.             $datetime['second'] = ( $default_to_max ) ? 59 : 0; 
  922.  
  923.         return sprintf( '%04d-%02d-%02d %02d:%02d:%02d', $datetime['year'], $datetime['month'], $datetime['day'], $datetime['hour'], $datetime['minute'], $datetime['second'] ); 
  924.     }

  926.     /** 
  927.      * Builds a query string for comparing time values (hour, minute, second). 
  928.      * 
  929.      * If just hour, minute, or second is set than a normal comparison will be done. 
  930.      * However if multiple values are passed, a pseudo-decimal time will be created 
  931.      * in order to be able to accurately compare against. 
  932.      * 
  933.      * @since 3.7.0 
  934.      * @access public 
  935.      * 
  936.      * @param string $column The column to query against. Needs to be pre-validated! 
  937.      * @param string $compare The comparison operator. Needs to be pre-validated! 
  938.      * @param int|null $hour Optional. An hour value (0-23). 
  939.      * @param int|null $minute Optional. A minute value (0-59). 
  940.      * @param int|null $second Optional. A second value (0-59). 
  941.      * @return string|false A query part or false on failure. 
  942.      */ 
  943.     public function build_time_query( $column, $compare, $hour = null, $minute = null, $second = null ) { 
  944.         global $wpdb; 
  945.  
  946.         // Have to have at least one 
  947.         if ( ! isset( $hour ) && ! isset( $minute ) && ! isset( $second ) ) 
  948.             return false; 
  949.  
  950.         // Complex combined queries aren't supported for multi-value queries 
  951.         if ( in_array( $compare, array( 'IN', 'NOT IN', 'BETWEEN', 'NOT BETWEEN' ) ) ) { 
  952.             $return = array(); 
  953.  
  954.             if ( isset( $hour ) && false !== ( $value = $this->build_value( $compare, $hour ) ) ) 
  955.                 $return[] = "HOUR( $column ) $compare $value"; 
  956.  
  957.             if ( isset( $minute ) && false !== ( $value = $this->build_value( $compare, $minute ) ) ) 
  958.                 $return[] = "MINUTE( $column ) $compare $value"; 
  959.  
  960.             if ( isset( $second ) && false !== ( $value = $this->build_value( $compare, $second ) ) ) 
  961.                 $return[] = "SECOND( $column ) $compare $value"; 
  962.  
  963.             return implode( ' AND ', $return ); 
  964.         } 
  965.  
  966.         // Cases where just one unit is set 
  967.         if ( isset( $hour ) && ! isset( $minute ) && ! isset( $second ) && false !== ( $value = $this->build_value( $compare, $hour ) ) ) { 
  968.             return "HOUR( $column ) $compare $value"; 
  969.         } elseif ( ! isset( $hour ) && isset( $minute ) && ! isset( $second ) && false !== ( $value = $this->build_value( $compare, $minute ) ) ) { 
  970.             return "MINUTE( $column ) $compare $value"; 
  971.         } elseif ( ! isset( $hour ) && ! isset( $minute ) && isset( $second ) && false !== ( $value = $this->build_value( $compare, $second ) ) ) { 
  972.             return "SECOND( $column ) $compare $value"; 
  973.         } 
  974.  
  975.         // Single units were already handled. Since hour & second isn't allowed, minute must to be set. 
  976.         if ( ! isset( $minute ) ) 
  977.             return false; 
  978.  
  979.         $format = $time = ''; 
  980.  
  981.         // Hour 
  982.         if ( null !== $hour ) { 
  983.             $format .= '%H.'; 
  984.             $time   .= sprintf( '%02d', $hour ) . '.'; 
  985.         } else { 
  986.             $format .= '0.'; 
  987.             $time   .= '0.'; 
  988.         } 
  989.  
  990.         // Minute 
  991.         $format .= '%i'; 
  992.         $time   .= sprintf( '%02d', $minute ); 
  993.  
  994.         if ( isset( $second ) ) { 
  995.             $format .= '%s'; 
  996.             $time   .= sprintf( '%02d', $second ); 
  997.         } 
  998.  
  999.         return $wpdb->prepare( "DATE_FORMAT( $column, %s ) $compare %f", $format, $time ); 
  1000.     }
