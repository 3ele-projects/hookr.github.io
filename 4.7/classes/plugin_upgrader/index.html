  316.     public function check_package($source) { 
  317.         global $wp_filesystem; 
  318.  
  319.         if ( is_wp_error($source) ) 
  320.             return $source; 
  321.  
  322.         $working_directory = str_replace( $wp_filesystem->wp_content_dir(), trailingslashit(WP_CONTENT_DIR), $source); 
  323.         if ( ! is_dir($working_directory) ) // Sanity check, if the above fails, let's not prevent installation. 
  324.             return $source; 
  325.  
  326.         // Check the folder contains at least 1 valid plugin. 
  327.         $plugins_found = false; 
  328.         $files = glob( $working_directory . '*.php' ); 
  329.         if ( $files ) { 
  330.             foreach ( $files as $file ) { 
  331.                 $info = get_plugin_data( $file, false, false ); 
  332.                 if ( ! empty( $info['Name'] ) ) { 
  333.                     $plugins_found = true; 
  334.                     break; 
  335.                 } 
  336.             } 
  337.         } 
  338.  
  339.         if ( ! $plugins_found ) 
  340.             return new WP_Error( 'incompatible_archive_no_plugins', $this->strings['incompatible_archive'], __( 'No valid plugins were found.' ) ); 
  341.  
  342.         return $source; 
  343.     }
