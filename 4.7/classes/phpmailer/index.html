  831.             } 
  832.             return false; 
  833.         }

  945.     /**

  947.      * @param string $address The email address to check 
  948.      * @param string $patternselect A selector for the validation pattern to use : 
  949.      * * `auto` Pick best pattern automatically; 
  950.      * * `pcre8` Use the squiloople.com pattern, requires PCRE > 8.0, PHP >= 5.3.2, 5.2.14; 
  951.      * * `pcre` Use old PCRE implementation; 
  952.      * * `php` Use PHP built-in FILTER_VALIDATE_EMAIL; 
  953.      * * `html5` Use the pattern given by the HTML5 spec for 'email' type form input elements. 
  954.      * * `noregex` Don't use a regex: super fast, really dumb.

  957.      * @access public 
  958.      */ 
  959.     public static function validateAddress($address, $patternselect = 'auto')

  962.         if (strpos($address, "\n") !== false or strpos($address, "\r") !== false) { 
  963.             return false; 
  964.         } 
  965.         if (!$patternselect or $patternselect == 'auto') { 
  966.             //Check this constant first so it works when extension_loaded() is disabled by safe mode

  969.                 //This pattern can get stuck in a recursive loop in PCRE <= 8.0.2 
  970.                 if (version_compare(PCRE_VERSION, '8.0.3') >= 0) { 
  971.                     $patternselect = 'pcre8'; 
  972.                 } else { 
  973.                     $patternselect = 'pcre'; 
  974.                 } 
  975.             } elseif (function_exists('extension_loaded') and extension_loaded('pcre')) { 
  976.                 //Fall back to older PCRE 
  977.                 $patternselect = 'pcre'; 
  978.             } else { 
  979.                 //Filter_var appeared in PHP 5.2.0 and does not require the PCRE extension 
  980.                 if (version_compare(PHP_VERSION, '5.2.0') >= 0) { 
  981.                     $patternselect = 'php'; 
  982.                 } else { 
  983.                     $patternselect = 'noregex'; 
  984.                 } 
  985.             } 
  986.         } 
  987.         switch ($patternselect) { 
  988.             case 'pcre8': 
  989.                 /** 
  990.                  * Uses the same RFC5322 regex on which FILTER_VALIDATE_EMAIL is based, but allows dotless domains. 
  991.                  * @link http://squiloople.com/2009/12/20/email-address-validation/ 
  992.                  * @copyright 2009-2010 Michael Rushton 
  993.                  * Feel free to use and redistribute this code. But please keep this copyright notice.

  1000.                     '(?2)")(?>(?1)\.(?1)(?4))*(?1)@(?!(?1)[a-z0-9-]{64, })(?1)(?>([a-z0-9](?>[a-z0-9-]*[a-z0-9])?)' . 
  1001.                     '(?>(?1)\.(?!(?1)[a-z0-9-]{64, })(?1)(?5)) {0, 126}|\[(?:(?>IPv6:(?>([a-f0-9]{1, 4})(?>:(?6)) {7}' . 
  1002.                     '|(?!(?:.*[a-f0-9][:\]]) {8, })((?6)(?>:(?6)) {0, 6})?::(?7)?))|(?>(?>IPv6:(?>(?6)(?>:(?6)) {5}:' .

  1004.                     '|[1-9]?[0-9])(?>\.(?9)) {3}))\])(?1)$/isD',  
  1005.                     $address 
  1006.  ); 
  1007.             case 'pcre': 
  1008.                 //An older regex that doesn't need a recent PCRE 
  1009.                 return (boolean)preg_match( 
  1010.                     '/^(?!(?>"?(?>\\\[ -~]|[^"])"?) {255, })(?!(?>"?(?>\\\[ -~]|[^"])"?) {65, }@)(?>' .

  1058.      * This function silently returns unmodified address if: 
  1059.      * - No conversion is necessary (i.e. domain name is not an IDN, or is already in ASCII form) 
  1060.      * - Conversion to punycode is impossible (e.g. required PHP functions are not available)

  1064.      * @return string The encoded address in ASCII form 
  1065.      */ 
  1066.     public function punyencodeAddress($address) 
  1067.     { 
  1068.         // Verify we have required functions, CharSet, and at-sign. 
  1069.         if ($this->idnSupported() and 
  1070.             !empty($this->CharSet) and

  1077.                     idn_to_ascii($domain, 0, INTL_IDNA_VARIANT_UTS46) : 
  1078.                     idn_to_ascii($domain)) !== false) { 
  1079.                     return substr($address, 0, $pos) . $punycode; 
  1080.                 } 
  1081.             } 
  1082.         } 
  1083.         return $address; 
  1084.     }

  1765.     /** 
  1766.      * Find the last character boundary prior to $maxLength in a utf-8 
  1767.      * quoted-printable encoded string. 
  1768.      * Original written by Colin Brown. 
  1769.      * @access public 
  1770.      * @param string $encodedText utf-8 QP text 
  1771.      * @param integer $maxLength Find the last character boundary prior to this length 
  1772.      * @return integer 
  1773.      */ 
  1774.     public function utf8CharBoundary($encodedText, $maxLength) 
  1775.     { 
  1776.         $foundSplitPos = false; 
  1777.         $lookBack = 3; 
  1778.         while (!$foundSplitPos) {

  1780.             $encodedCharPos = strpos($lastChunk, '='); 
  1781.             if (false !== $encodedCharPos) { 
  1782.                 // Found start of encoded character byte within $lookBack block. 
  1783.                 // Check the encoded byte value (the 2 chars after the '=') 
  1784.                 $hex = substr($encodedText, $maxLength - $lookBack + $encodedCharPos + 1, 2); 
  1785.                 $dec = hexdec($hex); 
  1786.                 if ($dec < 128) { 
  1787.                     // Single byte character. 
  1788.                     // If the encoded char was found at pos 0, it will fit 
  1789.                     // otherwise reduce maxLength to start of the encoded char 
  1790.                     if ($encodedCharPos > 0) { 
  1791.                         $maxLength = $maxLength - ($lookBack - $encodedCharPos); 
  1792.                     } 
  1793.                     $foundSplitPos = true; 
  1794.                 } elseif ($dec >= 192) { 
  1795.                     // First byte of a multi byte character 
  1796.                     // Reduce maxLength to split at start of character 
  1797.                     $maxLength = $maxLength - ($lookBack - $encodedCharPos); 
  1798.                     $foundSplitPos = true; 
  1799.                 } elseif ($dec < 192) { 
  1800.                     // Middle byte of a multi byte character, look further back 
  1801.                     $lookBack += 3; 
  1802.                 } 
  1803.             } else { 
  1804.                 // No encoded character found 
  1805.                 $foundSplitPos = true; 
  1806.             } 
  1807.         } 
  1808.         return $maxLength; 
  1809.     }
