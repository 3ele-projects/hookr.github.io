  1210.     public function filter_wp_nav_menu_args( $args ) { 
  1211.         /** 
  1212.          * The following conditions determine whether or not this instance of 
  1213.          * wp_nav_menu() can use selective refreshed. A wp_nav_menu() can be 
  1214.          * selective refreshed if... 
  1215.          */

  1217.             // ...if wp_nav_menu() is directly echoing out the menu (and thus isn't manipulating the string after generated),  
  1218.             ! empty( $args['echo'] ) 
  1219.             && 
  1220.             // ...and if the fallback_cb can be serialized to JSON, since it will be included in the placement context data,  
  1221.             ( empty( $args['fallback_cb'] ) || is_string( $args['fallback_cb'] ) ) 
  1222.             && 
  1223.             // ...and if the walker can also be serialized to JSON, since it will be included in the placement context data as well,  
  1224.             ( empty( $args['walker'] ) || is_string( $args['walker'] ) ) 
  1225.             // ...and if it has a theme location assigned or an assigned menu to display,  
  1226.             && ( 
  1227.                 ! empty( $args['theme_location'] ) 
  1228.                 || 
  1229.                 ( ! empty( $args['menu'] ) && ( is_numeric( $args['menu'] ) || is_object( $args['menu'] ) ) ) 
  1230.  ) 
  1231.             && 
  1232.             // ...and if the nav menu would be rendered with a wrapper container element (upon which to attach data-* attributes). 
  1233.             ( 
  1234.                 ! empty( $args['container'] ) 
  1235.                 || 
  1236.                 ( isset( $args['items_wrap'] ) && '<' === substr( $args['items_wrap'], 0, 1 ) ) 
  1237.  )

  1241.         $exported_args = $args; 
  1242.  
  1243.         // Empty out args which may not be JSON-serializable. 
  1244.         if ( ! $can_partial_refresh ) {

  1248.  
  1249.         /**

  1253.         if ( ! empty( $exported_args['menu'] ) && is_object( $exported_args['menu'] ) ) {
