  122.

  125.  
  126.               // Then loop and output. 
  127.               echo '<ul class="cmb2-'. str_replace( '_', '-', $this->field->type() ) .'">'; 
  128.               foreach ( $this->field->value as $val ) { 
  129.                   $this->value = $val; 
  130.                   echo '<li>', $this->_display(), '</li>'; 
  131.                   ; 
  132.               } 
  133.               echo '</ul>';

  136.       } else {

  367.   /**

  371.   protected function _display() {

  375.  
  376.       $this->value = esc_url_raw( $this->value ); 
  377.  
  378.       $field_type = new CMB2_Type_File_Base( new CMB2_Types( $this->field ) ); 
  379.  
  380.       $id = $this->field->get_field_clone( array( 
  381.           'id' => $field_type->_id() . '_id',

  386.  
  387.   protected function file_output( $url_value, $id, CMB2_Type_File_Base $field_type ) { 
  388.       // If there is no ID saved yet, try to get it from the url 
  389.       if ( $url_value && ! $id ) { 
  390.           $id = CMB2_Utils::image_id_from_url( esc_url_raw( $url_value ) ); 
  391.       } 
  392.  
  393.       if ( $field_type->is_valid_img_ext( $url_value ) ) { 
  394.           $img_size = $this->field->args( 'preview_size' ); 
  395.  
  396.           if ( $id ) { 
  397.               $image = wp_get_attachment_image( $id, $img_size, null, array( 'class' => 'cmb-image-display' ) ); 
  398.           } else { 
  399.               $size = is_array( $img_size ) ? $img_size[0] : 200; 
  400.               $image = '<img class="cmb-image-display" style="max-width: ' . absint( $size ) . 'px; width: 100%; height: auto;" src="' . $url_value . '" alt="" />'; 
  401.           } 
  402.  
  403.           echo $image; 
  404.  
  405.       } else { 
  406.  
  407.           printf( '<div class="file-status"><span>%1$s <strong><a href="%2$s">%3$s</a></strong></span></div>',  
  408.               esc_html( $field_type->_text( 'file_text', esc_html__( 'File:', 'cmb2' ) ) ),  
  409.               $url_value,  
  410.               CMB2_Utils::get_file_name_from_path( $url_value ) 
  411.  ); 
  412.  
  413.       } 
  414.   } 
  415. }
