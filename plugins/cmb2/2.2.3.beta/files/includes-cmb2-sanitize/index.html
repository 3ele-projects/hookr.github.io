  55.  
  56.       /** 
  57.        * This exists for back-compatibility, but validation 
  58.        * is not what happens here. 
  59.        * @deprecated See documentation for "cmb2_sanitize_{$this->type()}". 
  60.        */

  62.           $override_value = apply_filters_deprecated( "cmb2_validate_{$this->field->type()}", array( null, $this->value, $this->field->object_id, $this->field->args(), $this ), '2.0.0', "cmb2_sanitize_{$this->field->type()}" ); 
  63.       } else { 
  64.           $override_value = apply_filters( "cmb2_validate_{$this->field->type()}", null, $this->value, $this->field->object_id, $this->field->args(), $this ); 
  65.       } 
  66.  
  67.       if ( null !== $override_value ) { 
  68.           return $override_value; 
  69.       } 
  70.  
  71.       $sanitized_value = ''; 
  72.       switch ( $this->field->type() ) { 
  73.           case 'wysiwyg': 
  74.           case 'textarea_small': 
  75.           case 'oembed': 
  76.               $sanitized_value = $this->textarea(); 
  77.               break; 
  78.           case 'taxonomy_select': 
  79.           case 'taxonomy_radio': 
  80.           case 'taxonomy_radio_inline': 
  81.           case 'taxonomy_multicheck': 
  82.           case 'taxonomy_multicheck_inline': 
  83.               if ( $this->field->args( 'taxonomy' ) ) { 
  84.                   wp_set_object_terms( $this->field->object_id, $this->value, $this->field->args( 'taxonomy' ) ); 
  85.               } else { 
  86.                   CMB2_Utils::log_if_debug( __METHOD__, __LINE__, "{$this->field->type()} {$this->field->_id()} is missing the 'taxonomy' parameter." ); 
  87.               } 
  88.               break; 
  89.           case 'multicheck': 
  90.           case 'multicheck_inline': 
  91.           case 'file_list': 
  92.           case 'group': 
  93.               // no filtering 
  94.               $sanitized_value = $this->value; 
  95.               break; 
  96.           default: 
  97.               // Handle repeatable fields array 
  98.               // We'll fallback to 'sanitize_text_field' 
  99.               $sanitized_value = is_array( $this->value ) ? array_map( 'sanitize_text_field', $this->value ) : sanitize_text_field( $this->value ); 
  100.               break; 
  101.       } 
  102.  
  103.       return $this->_is_empty_array( $sanitized_value ) ? '' : $sanitized_value; 
  104.   }

  176.       if ( ! $this->value ) {

  179.  
  180.       global $wp_locale;

  215.   public function text_datetime_timestamp( $repeat = false ) { 
  216.  
  217.       $test = is_array( $this->value ) ? array_filter( $this->value ) : ''; 
  218.       if ( empty( $test ) ) { 
  219.           return ''; 
  220.       } 
  221.  
  222.       $repeat_value = $this->_check_repeat( __FUNCTION__, $repeat ); 
  223.       if ( false !== $repeat_value ) { 
  224.           return $repeat_value; 
  225.       } 
  226.  
  227.       if ( isset( $this->value['date'], $this->value['time'] ) ) { 
  228.           $this->value = $this->field->get_timestamp_from_value( $this->value['date'] . ' ' . $this->value['time'] ); 
  229.       } 
  230.  
  231.       if ( $tz_offset = $this->field->field_timezone_offset() ) {

  238.   /** 
  239.    * Datetime to timestamp with timezone 
  240.    * @since  1.0.1 
  241.    * @return string       Timestring 
  242.    */ 
  243.   public function text_datetime_timestamp_timezone( $repeat = false ) { 
  244.       static $utc_values = array(); 
  245.  
  246.       $test = is_array( $this->value ) ? array_filter( $this->value ) : ''; 
  247.       if ( empty( $test ) ) { 
  248.           return ''; 
  249.       } 
  250.  
  251.       $utc_key = $this->field->_id() . '_utc'; 
  252.  
  253.       $repeat_value = $this->_check_repeat( __FUNCTION__, $repeat ); 
  254.       if ( false !== $repeat_value ) { 
  255.           if ( ! empty( $utc_values[ $utc_key ] ) ) { 
  256.               $this->_save_utc_value( $utc_key, $utc_values[ $utc_key ] ); 
  257.               unset( $utc_values[ $utc_key ] ); 
  258.           } 
  259.  
  260.           return $repeat_value; 
  261.       } 
  262.  
  263.       $tzstring = null; 
  264.  
  265.       if ( is_array( $this->value ) && array_key_exists( 'timezone', $this->value ) ) { 
  266.           $tzstring = $this->value['timezone']; 
  267.       } 
  268.  
  269.       if ( empty( $tzstring ) ) { 
  270.           $tzstring = CMB2_Utils::timezone_string(); 
  271.       } 
  272.  
  273.       $offset = CMB2_Utils::timezone_offset( $tzstring ); 
  274.  
  275.       if ( 'UTC' === substr( $tzstring, 0, 3 ) ) { 
  276.           $tzstring = timezone_name_from_abbr( '', $offset, 0 ); 
  277.           /** 
  278.            * timezone_name_from_abbr() returns false if not found based on offset. 
  279.            * Since there are currently some invalid timezones in wp_timezone_dropdown(),  
  280.            * fallback to an offset of 0 (UTC+0) 
  281.            * https://core.trac.wordpress.org/ticket/29205 
  282.            */ 
  283.           $tzstring = false !== $tzstring ? $tzstring : timezone_name_from_abbr( '', 0, 0 ); 
  284.       } 
  285.  
  286.       $full_format = $this->field->args['date_format'] . ' ' . $this->field->args['time_format']; 
  287.       $full_date = $this->value['date'] . ' ' . $this->value['time']; 
  288.  
  289.       try { 
  290.  
  291.           $datetime = date_create_from_format( $full_format, $full_date ); 
  292.  
  293.           if ( ! is_object( $datetime ) ) { 
  294.               $this->value = $utc_stamp = ''; 
  295.           } else { 
  296.               $timestamp = $datetime->setTimezone( new DateTimeZone( $tzstring ) )->getTimestamp(); 
  297.               $utc_stamp = $timestamp - $offset; 
  298.               $this->value = serialize( $datetime ); 
  299.           } 
  300.  
  301.           if ( $this->field->group ) { 
  302.               $this->value = array( 
  303.                   'supporting_field_value' => $utc_stamp,  
  304.                   'supporting_field_id' => $utc_key,  
  305.                   'value' => $this->value,  
  306.  ); 
  307.           } else { 
  308.               // Save the utc timestamp supporting field 
  309.               if ( $repeat ) { 
  310.                   $utc_values[ $utc_key ][] = $utc_stamp; 
  311.               } else { 
  312.                   $this->_save_utc_value( $utc_key, $utc_stamp ); 
  313.               } 
  314.           } 
  315.  
  316.       } catch ( Exception $e ) { 
  317.           $this->value = ''; 
  318.           CMB2_Utils::log_if_debug( __METHOD__, __LINE__, $e->getMessage() ); 
  319.       } 
  320.  
  321.       return $this->value; 
  322.   }

  426.   /** 
  427.    * If repeating, loop through and re-apply sanitization method 
  428.    * @since  1.1.0

  430.    * @param  bool   $repeat Whether repeating or not 
  431.    * @return mixed          Sanitized value

  433.   public function _check_repeat( $method, $repeat ) { 
  434.       if ( $repeat || ! $this->field->args( 'repeatable' ) ) { 
  435.           return false; 
  436.       } 
  437.  
  438.       $values_array = $this->value; 
  439.  
  440.       $new_value = array(); 
  441.       foreach ( $values_array as $iterator => $this->value ) { 
  442.           if ( $this->value ) { 
  443.               $val = $this->$method( true ); 
  444.               if ( ! empty( $val ) ) { 
  445.                   $new_value[] = $val; 
  446.               } 
  447.           } 
  448.       } 
  449.  
  450.       $this->value = $new_value; 
  451.  
  452.       return empty( $this->value ) ? null : $this->value; 
  453.   }
