  556.      * Determines if the current object is able to be saved 
  557.      * @since  2.0.9 
  558.      * @param  string  $type Current post_type or comment_type 
  559.      * @return bool          Whether object can be saved 
  560.      */ 
  561.     public function can_save( $type = '' ) { 
  562.         return apply_filters( 'cmb2_can_save', ( 
  563.             $this->cmb->prop( 'save_fields' ) 
  564.             // check nonce 
  565.             && isset( $_POST[ $this->cmb->nonce() ] ) 
  566.             && wp_verify_nonce( $_POST[ $this->cmb->nonce() ], $this->cmb->nonce() ) 
  567.             // check if autosave 
  568.             && ! ( defined( 'DOING_AUTOSAVE' ) && DOING_AUTOSAVE ) 
  569.             // get the metabox types & compare it to this type 
  570.             && ( $type && in_array( $type, $this->cmb->prop( 'object_types' ) ) ) 
  571.             // Don't do updates during a switch-to-blog instance. 
  572.             && ! ( is_multisite() && ms_is_switched() ) 
  573.  ) ); 
  574.     }

  576.     /** 
  577.      * Determine if taxonomy of term being modified is cmb2-editable. 
  578.      * @since  2.2.0 
  579.      * @param  string $taxonomy Taxonomy of term being modified. 
  580.      * @return bool             Whether taxonomy is editable. 
  581.      */ 
  582.     public function taxonomy_can_save( $taxonomy ) { 
  583.         if ( empty( $this->taxonomies ) || ! in_array( $taxonomy, $this->taxonomies ) ) { 
  584.             return false; 
  585.         } 
  586.  
  587.         $taxonomy_object = get_taxonomy( $taxonomy ); 
  588.         // Can the user edit this term? 
  589.         if ( ! isset( $taxonomy_object->cap ) || ! current_user_can( $taxonomy_object->cap->edit_terms ) ) { 
  590.             return false; 
  591.         } 
  592.  
  593.         return true; 
  594.     }

  606.             && in_array( $pagenow, array( 'edit.php', 'users.php', 'edit-comments.php', 'edit-tags.php' ), 1 ) 
  607.  ) { 
  608.             self::enqueue_cmb_css( 'cmb2-display-styles' ); 
  609.         }
