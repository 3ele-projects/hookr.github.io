  666.     public function save_group_field( $field_group ) {

  668.  
  669.         if ( ! isset( $this->data_to_save[ $base_id ] ) ) { 
  670.             return; 
  671.         }

  675.         $group_vals = $field_group->sanitization_cb( $this->data_to_save[ $base_id ] );

  677.  
  678.         $field_group->index = 0; 
  679.         $field_group->data_to_save = $this->data_to_save;

  685.  
  686.             foreach ( (array) $group_vals as $field_group->index => $post_vals ) { 
  687.  
  688.                 // Get value 
  689.                 $new_val = isset( $group_vals[ $field_group->index ][ $sub_id ] ) 
  690.                     ? $group_vals[ $field_group->index ][ $sub_id ] 
  691.                     : false; 
  692.  
  693.                 // Sanitize 
  694.                 $new_val = $field->sanitization_cb( $new_val ); 
  695.  
  696.                 if ( is_array( $new_val ) && $field->args( 'has_supporting_data' ) ) { 
  697.                     if ( $field->args( 'repeatable' ) ) { 
  698.                         $_new_val = array(); 
  699.                         foreach ( $new_val as $group_index => $grouped_data ) { 
  700.                             // Add the supporting data to the $saved array stack 
  701.                             $saved[ $field_group->index ][ $grouped_data['supporting_field_id'] ][] = $grouped_data['supporting_field_value']; 
  702.                             // Reset var to the actual value 
  703.                             $_new_val[ $group_index ] = $grouped_data['value']; 
  704.                         } 
  705.                         $new_val = $_new_val; 
  706.                     } else { 
  707.                         // Add the supporting data to the $saved array stack 
  708.                         $saved[ $field_group->index ][ $new_val['supporting_field_id'] ] = $new_val['supporting_field_value']; 
  709.                         // Reset var to the actual value 
  710.                         $new_val = $new_val['value']; 
  711.                     } 
  712.                 }

  714.                 // Get old value 
  715.                 $old_val = is_array( $old ) && isset( $old[ $field_group->index ][ $sub_id ] ) 
  716.                     ? $old[ $field_group->index ][ $sub_id ] 
  717.                     : false; 
  718.  
  719.                 $is_updated = ( ! CMB2_Utils::isempty( $new_val ) && $new_val !== $old_val ); 
  720.                 $is_removed = ( CMB2_Utils::isempty( $new_val ) && ! CMB2_Utils::isempty( $old_val ) ); 
  721.  
  722.                 // Compare values and add to `$updated` array 
  723.                 if ( $is_updated || $is_removed ) { 
  724.                     $this->updated[] = $base_id . '::' . $field_group->index . '::' . $sub_id; 
  725.                 }

  728.                 $saved[ $field_group->index ][ $sub_id ] = $new_val; 
  729.  
  730.             }

  732.             $saved[ $field_group->index ] = CMB2_Utils::filter_empty( $saved[ $field_group->index ] ); 
  733.         }

  736.  
  737.         return $field_group->update_data( $saved, true ); 
  738.     }
