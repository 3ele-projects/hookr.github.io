  342.   /** 
  343.    * Runs recursivly through the opts array, sanitizing and merging in updates from the $input array 
  344.    *  
  345.    * @param array $opts good, clean array 
  346.    * @param array $input unsanitzed input array, not trusted at all 
  347.    * @todo This function should probably get a filter thrown within it to be more extensible 
  348.    */ 
  349.   protected function opts_update_loop(&$opts, $input) 
  350.   { 
  351.       //Loop through all of the existing options (avoids random setting injection) 
  352.       foreach($opts as $option => $value) 
  353.       { 
  354.           //If we have an array, dive into another recursive loop 
  355.           if(isset($input[$option]) && is_array($value)) 
  356.           { 
  357.               $this->opts_update_loop($opts[$option], $input[$option]); 
  358.           } 
  359.           //We must check for unset settings, but booleans are ok to be unset 
  360.           else if(isset($input[$option]) || $option[0] == 'b') 
  361.           { 
  362.               switch($option[0]) 
  363.               { 
  364.                   //Handle the boolean options 
  365.                   case 'b': 
  366.                       $opts[$option] = isset($input[$option]); 
  367.                       break; 
  368.                   //Handle the integer options 
  369.                   case 'i': 
  370.                       $opts[$option] = (int) $input[$option]; 
  371.                       break; 
  372.                   //Handle the absolute integer options 
  373.                   case 'a': 
  374.                       $opts[$option] = absint($input[$option]); 
  375.                       break; 
  376.                   //Handle the floating point options 
  377.                   case 'f': 
  378.                       $opts[$option] = (float) $input[$option]; 
  379.                       break; 
  380.                   //Handle the HTML options 
  381.                   case 'h': 
  382.                       //May be better to use wp_kses here 
  383.                       $opts[$option] = wp_kses(stripslashes($input[$option]), $this->allowed_html); 
  384.                       break; 
  385.                   //Handle the HTML options that must not be null 
  386.                   case 'H': 
  387.                       if(isset($input[$option])) 
  388.                       { 
  389.                           $opts[$option] = wp_kses(stripslashes($input[$option]), $this->allowed_html); 
  390.                       } 
  391.                       break; 
  392.                   //Handle the text options that must not be null 
  393.                   case 'S': 
  394.                       if(isset($input[$option])) 
  395.                       { 
  396.                           $opts[$option] = esc_html($input[$option]); 
  397.                       } 
  398.                       break; 
  399.                   //Deal with strings that can be null 
  400.                   case 's': 
  401.                       $opts[$option] = esc_html($input[$option]); 
  402.                       break; 
  403.                   //By default we have nothing to do, allows for internal settings 
  404.                   default: 
  405.                       break; 
  406.               } 
  407.           } 
  408.       } 
  409.   } 
  410.   /** 
  411.    * A better version of parse_args, will recrusivly follow arrays

  418.   { 
  419.       if(is_object($args))

  421.           $r = get_object_vars($args);

  437.   /** 
  438.    * An alternate version of array_merge_recursive, less flexible 
  439.    * still recursive, ~2x faster than the more flexible version 
  440.    *  
  441.    * @param array $arg1 first array 
  442.    * @param array $arg2 second array to merge into $arg1 
  443.    * @return array 
  444.    */ 
  445.   static function array_merge_recursive($arg1, $arg2) 
  446.   { 
  447.       foreach($arg2 as $key => $value) 
  448.       { 
  449.           if(array_key_exists($key, $arg1) && is_array($value)) 
  450.           { 
  451.               $arg1[$key] = mtekk_adminKit::array_merge_recursive($arg1[$key], $value); 
  452.           } 
  453.           else 
  454.           { 
  455.               $arg1[$key] = $value; 
  456.           } 
  457.       } 
  458.       return $arg1; 
  459.   }

  473.   function opts_update() 
  474.   { 
  475.       //Do some security related thigns as we are not using the normal WP settings API 
  476.       $this->security();

  478.       check_admin_referer($this->unique_prefix . '_options-options');

  482.       //Update our backup options 
  483.       $this->update_option($this->unique_prefix . '_options_bk', $this->opt); 
  484.       //Grab our incomming array (the data is dirty)

  486.       //Run the update loop 
  487.       $this->opts_update_loop($this->opt, $input); 
  488.       //Commit the option changes 
  489.       $this->update_option($this->unique_prefix . '_options', $this->opt); 
  490.       //Check if known settings match attempted save 
  491.       if(count(array_diff_key($input, $this->opt)) == 0) 
  492.       { 
  493.           //Let the user know everything went ok 
  494.           $this->message['updated fade'][] = __('Settings successfully saved.', $this->identifier) . $this->admin_anchor('undo', __('Undo the options save.', $this->identifier), __('Undo', $this->identifier)); 
  495.       } 
  496.       else 
  497.       { 
  498.           //Let the user know the following were not saved 
  499.           $this->message['updated fade'][] = __('Some settings were not saved.', $this->identifier) . $this->admin_anchor('undo', __('Undo the options save.', $this->identifier), __('Undo', $this->identifier)); 
  500.           $temp = __('The following settings were not saved:', $this->identifier);

  502.           { 
  503.               $temp .= '<br />' . $setting; 
  504.           } 
  505.           $this->message['updated fade'][] = $temp . '<br />' . sprintf(__('Please include this message in your %sbug report%s.', $this->identifier), '<a title="' . sprintf(__('Go to the %s support post for your version.', $this->identifier), $this->short_name) . '" href="' . $this->support_url . $this->version . '/#respond">', '</a>'); 
  506.       }

  513.   { 
  514.       //Do a nonce check, prevent malicious link/form problems  
  515.       check_admin_referer($this->unique_prefix . '_admin_import_export'); 
  516.       //Update our internal settings 
  517.       $this->opt = $this->get_option($this->unique_prefix . '_options'); 
  518.       //Create a DOM document
