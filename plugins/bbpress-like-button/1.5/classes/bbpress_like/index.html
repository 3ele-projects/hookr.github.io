  249.     function like_this() {

  251.          
  252.         extract($_POST); 
  253.          
  254.         //Check if previously liked 
  255.         $liked = self::get_like($post_id, $user_id); 
  256.         if($liked) { 
  257.             _e('This post was already liked by this user', $this->l10n); 
  258.         }else{ 
  259.             //Insert like 
  260.             $result = add_post_meta($post_id, 'bbpl_like', $user_id); 
  261.             if(!$result) { _e('An error ocurred: ', $this->l10n).$wpdb->print_error(); }else{ _e('Liked successfully', $this->l10n);} 
  262.             //Buddypress show on activity stream 
  263.             if (get_option('bbpl_buddypress', 0) && (Bool)$this->settings['settingsbbpl_bbpl_general_buddypress_activity_stream']) { 
  264.                 //Get post type 
  265.                 $post_type = get_post_type($post_id); 
  266.                 $user_info = get_userdata($user_id); 
  267.                 switch ($post_type) { 
  268.                     case 'topic': 
  269.                         //Get the data 
  270.                         $reply_info = get_post_ancestors($post_id); 
  271.                         $forum_id = reset($reply_info); 
  272.                         $topic_id = $post_id; 
  273.                         break; 
  274.                     case 'reply': 
  275.                     default: 
  276.                         //Get the data 
  277.                         $reply_info = get_post_ancestors($post_id); 
  278.                         $forum_id = end($reply_info); 
  279.                         $topic_id = reset($reply_info); 
  280.                         break; 
  281.                 } 
  282.  
  283.                 //Placeholders replacement 
  284.                 $placeholders = array( 
  285.                     'USERNAME' => '<a href="'.bbp_get_user_profile_url($user_id).'" title="'.$user_info->user_nicename.'">'.$user_info->user_nicename .'</a>',  
  286.                     'TOPIC' => '<a href="'.trailingslashit(get_post_permalink($topic_id)).'#post-'.$post_id.'" title="'.get_the_title($topic_id).'">'.get_the_title($topic_id).'</a>',  
  287.                     'FORUM' => '<a href="'.get_post_permalink($forum_id).'" title="'.get_the_title($forum_id).'">'.get_the_title($forum_id).'</a>' 
  288.  ); 
  289.                 $action = $this->settings['settingsbbpl_bbpl_labels_label_like_button_buddypress']; 
  290.                 foreach($placeholders as $key => $value) { 
  291.                     $action = str_replace('%%'.strtoupper($key).'%%', $value, $action); 
  292.                 } 
  293.  
  294.                 $args = array( 
  295.                     'action' => $action,  
  296.                     'component' => 'bbpl',  
  297.                     'type' => 'like',  
  298.                     'user_id' => $user_id,  
  299.                     'recorded_time' => date( 'Y-m-d H:i:s'),  
  300.  ); 
  301.                 $activity_id = bp_activity_add( $args ); 
  302.             } 
  303.         } 
  304.         die; 
  305.     }
