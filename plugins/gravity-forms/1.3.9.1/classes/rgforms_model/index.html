  630.     private static function get_field_display($form, $field, $field_values) { 
  631.  
  632.         //use section's logic if one exists 
  633.         $logic = $field["conditionalLogic"]; 
  634.  
  635.         //if this field does not have any conditional logic associated with it, it won't be hidden 
  636.         if(empty($logic)) 
  637.             return "show"; 
  638.  
  639.         $match_count = 0; 
  640.         foreach($logic["rules"] as $rule) { 
  641.             $source_field = RGFormsModel::get_field($form, $rule["fieldId"]); 
  642.             $field_value = self::get_field_value($source_field, $field_values); 
  643.             $is_value_match = is_array($field_value) ? in_array($rule["value"], $field_value) : $field_value == $rule["value"]; 
  644.  
  645.             if( ($rule["operator"] == "is" && $is_value_match ) || ($rule["operator"] == "isnot" && !$is_value_match) ) 
  646.                 $match_count++; 
  647.         } 
  648.  
  649.         $do_action = ($logic["logicType"] == "all" && $match_count == sizeof($logic["rules"]) ) || ($logic["logicType"] == "any"  && $match_count > 0); 
  650.         $is_hidden = ($do_action && $logic["actionType"] == "hide") || (!$do_action && $logic["actionType"] == "show"); 
  651.  
  652.         return $is_hidden ? "hide" : "show";

  834.     private static function prepare_value($form_id, $field, $value, $input_name) { 
  835.         switch(self::get_input_type($field)) 
  836.         { 
  837.             case "post_category" : 
  838.                 $cat = get_category($value); 
  839.                 $value = $cat->name; 
  840.             break; 
  841.  
  842.             case "post_title" : 
  843.             case "post_content" : 
  844.             case "post_excerpt" : 
  845.             case "post_tags" : 
  846.             case "post_custom_fields" : 
  847.             case "post_image": 
  848.                 $value = stripslashes($value); 
  849.             break; 
  850.  
  851.             case "phone" : 
  852.                 if($field["phoneFormat"] == "standard" && preg_match('/^\D?(\d{3})\D?\D?(\d{3})\D?(\d{4})$/', $value, $matches)) 
  853.                     $value = sprintf("(%s)%s-%s", $matches[1], $matches[2], $matches[3]); 
  854.             break; 
  855.  
  856.             case "time": 
  857.  
  858.                 if(!is_array($value) && !empty($value)) { 
  859.                     preg_match('/^(\d*):(\d*) ?(.*)$/', $value, $matches); 
  860.                     $value = array(); 
  861.                     $value[0] = $matches[1]; 
  862.                     $value[1] = $matches[2]; 
  863.                     $value[2] = $matches[3]; 
  864.                 } 
  865.  
  866.                 $hour = empty($value[0]) ? "0" : strip_tags($value[0]); 
  867.                 $minute = empty($value[1]) ? "0" : strip_tags($value[1]); 
  868.                 $ampm = strip_tags($value[2]); 
  869.  
  870.                 if(!(empty($hour) && empty($minute))) 
  871.                     $value = sprintf("%02d:%02d %s", $hour, $minute, $ampm); 
  872.                 else 
  873.                     $value = ""; 
  874.  
  875.             break; 
  876.  
  877.             case "date" : 
  878.                 $format = empty($field["dateFormat"]) ? "mdy" : $field["dateFormat"]; 
  879.                 $date_info = GFCommon::parse_date($value, $format); 
  880.                 if(!empty($date_info)) 
  881.                     $value = sprintf("%d-%02d-%02d", $date_info["year"], $date_info["month"], $date_info["day"]); 
  882.  
  883.             break; 
  884.  
  885.             case "fileupload" : 
  886.                 $value = self::upload_file($form_id, $_FILES[$input_name]); 
  887.             break; 
  888.  
  889.             case "number" : 
  890.                 $value = GFCommon::clean_number($value); 
  891.             break; 
  892.  
  893.             default: 
  894.                 $value = strip_tags(stripslashes($value)); 
  895.             break; 
  896.         } 
  897.         return $value; 
  898.     }
