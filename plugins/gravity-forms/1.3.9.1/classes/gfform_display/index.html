  469.                         if(!empty($field["allowedExtensions"]) && !empty($info["basename"]) && !in_array($extension, $allowedExtensions)) { 
  470.                             $field["failed_validation"] = true; 
  471.                             $field["validation_message"] = empty($field["errorMessage"]) ? sprintf(__("The uploaded file type is not allowed. Must be one of the following: %s", "gravityforms"), strtolower($field["allowedExtensions"]) )  : $field["errorMessage"]; 
  472.                             $is_valid = false; 
  473.                         } 
  474.                     break; 
  475.                 } 
  476.             } 
  477.         } 
  478.         return $is_valid; 
  479.     }

  531.     private static function get_conditional_logic($form) { 
  532.         $logics = ""; 
  533.         $dependents = ""; 
  534.         $fields_with_logic = array(); 
  535.         foreach($form["fields"] as $field) { 
  536.  
  537.             //use section's logic if one exists 
  538.             $section = RGFormsModel::get_section($form, $field["id"]); 
  539.             $section_logic = !empty($section) ? $section["conditionalLogic"] : null; 
  540.  
  541.             $logic = $field["conditionalLogic"]; 
  542.  
  543.             if(!empty($logic)) { 
  544.                 $field_section_logic = array("field" => $logic, "section" => $section_logic); 
  545.                 $logics .= $field["id"] . ": " . GFCommon::json_encode($field_section_logic) . ", "; 
  546.                 $fields_with_logic[] = $field["id"]; 
  547.  
  548.                 $peers = $field["type"] == "section" ? GFCommon::get_section_fields($form, $field["id"]) : array($field); 
  549.                 $peer_ids = array(); 
  550.  
  551.                 foreach ($peers as $peer) 
  552.                     $peer_ids[] = $peer["id"]; 
  553.  
  554.                 $dependents .= $field["id"] . ": " . GFCommon::json_encode($peer_ids) . ", "; 
  555.             } 
  556.         } 
  557.  
  558.         if(!empty($logics)) 
  559.             $logics = substr($logics, 0, strlen($logics) - 1); //removing last comma; 
  560.  
  561.         if(!empty($dependents)) 
  562.             $dependents = substr($dependents, 0, strlen($dependents) - 1); //removing last comma; 
  563.  
  564.  
  565.         $str = "<script type='text/javascript'> 
  566.         jQuery(document).ready(function() { 
  567.             gf_apply_rules({$form['id']}, " . GFCommon::json_encode($fields_with_logic) . "); 
  568.             jQuery('#gform_wrapper_{$form['id']}').show(); 
  569.         }); 
  570.         if(!window['gf_form_conditional_logic']) 
  571.             window['gf_form_conditional_logic'] = new Array(); 
  572.  
  573.         window['gf_form_conditional_logic'][{$form['id']}] = {'logic' : {" . $logics . "}, 'dependents' : {" . $dependents . "}};</script>"; 
  574.  
  575.         return $str; 
  576.     }

  589.     private static function get_conditional_logic_fields($form, $fieldId) { 
  590.         $fields = array(); 
  591.         foreach($form["fields"] as $field) { 
  592.             if(!empty($field["conditionalLogic"])) { 
  593.                 foreach($field["conditionalLogic"]["rules"] as $rule) { 
  594.                     if($rule["fieldId"] == $fieldId) { 
  595.                         $fields[] = $field["id"]; 
  596.  
  597.                         //if field is a section, add all fields in the section that have conditional logic (to support nesting) 
  598.                         if($field["type"] == "section") { 
  599.                             $section_fields = GFCommon::get_section_fields($form, $field["id"]); 
  600.                             foreach($section_fields as $section_field) 
  601.                                 if(!empty($section_field["conditionalLogic"])) 
  602.                                     $fields[] = $section_field["id"]; 
  603.                         } 
  604.                         break; 
  605.                     } 
  606.                 } 
  607.             } 
  608.         } 
  609.         return $fields; 
  610.     }

  612.         public static function get_field($field, $value="", $force_frontend_label = false, $form=null, $field_values=null) { 
  613.         if(!IS_ADMIN && $field["adminOnly"]) 
  614.         { 
  615.             if($field["allowsPrepopulate"]) 
  616.                 $field["type"] = "adminonly_hidden"; 
  617.             else 
  618.                 return; 
  619.         } 
  620.  
  621.         $id = $field["id"]; 
  622.         $type = $field["type"]; 
  623.         $error_class = $field["failed_validation"] ? "gfield_error" : ""; 
  624.         $custom_class = $field["cssClass"]; 
  625.         $admin_only_class =  $field["adminOnly"] ? "field_admin_only" : ""; 
  626.         $selectable_class = IS_ADMIN ? "selectable" : ""; 
  627.  
  628.         $section_class = $field["type"] == "section" ? "gsection" : ""; 
  629.         $css_class = "$selectable_class gfield $error_class $section_class $admin_only_class $custom_class"; 
  630.         $css_class = trim($css_class); 
  631.  
  632.         $style = !empty($form) && !IS_ADMIN && RGFormsModel::is_field_hidden($form, $field, $field_values) ? "style='display:none;'" : ""; 
  633.  
  634.         $field_id = IS_ADMIN || empty($form) ? "field_$id" : "field_" . $form["id"] . "_$id"; 
  635.  
  636.         return "<li id='$field_id' class='$css_class' $style>" . self::get_field_content($field, $value, $force_frontend_label, $form == null ? 0 : $form["id"]) . "</li>"; 
  637.     }

  639.     public static function get_field_content($field, $value="", $force_frontend_label = false, $form_id=0) { 
  640.         $id = $field["id"]; 
  641.         $size = $field["size"]; 
  642.         $validation_message = ($field["failed_validation"] && !empty($field["validation_message"])) ? sprintf("<div class='gfield_description validation_message'>%s</div>", $field["validation_message"]) : ""; 
  643.  
  644.         $delete_field_link = "<a class='field_delete_icon' id='gfield_delete_$id' title='" . __("click to delete this field", "gravityforms") . "' href='javascript:void(0);' onclick='StartDeleteField(this);'>" . __("Delete", "gravityforms") . "</a>"; 
  645.         $delete_field_link = apply_filters("gform_delete_field_link", $delete_field_link); 
  646.  
  647.         $admin_buttons = IS_ADMIN ? $delete_field_link . " <a class='field_edit_icon edit_icon_collapsed' href='javascript:void(0);' title='" . __("click to edit this field", "gravityforms") . "'>" . __("Edit", "gravityforms") . "</a>" : ""; 
  648.  
  649.         $field_label = $force_frontend_label ? $field["label"] : GFCommon::get_label($field); 
  650.         $field_id = IS_ADMIN || $form_id == 0 ? "input_$id" : "input_" . $form_id . "_$id"; 
  651.  
  652.         switch(RGFormsModel::get_input_type($field)) { 
  653.             case "section" : 
  654.                 $description = self::get_description($field["description"], "gsection_description"); 
  655.                 $field_content = sprintf("%s<h2 class='gsection_title'>%s</h2>%s", $admin_buttons, esc_html($field_label), $description); 
  656.             break; 
  657.  
  658.             case "adminonly_hidden": 
  659.             case "hidden" : 
  660.                 $field_content = !IS_ADMIN ? "{FIELD}" : $field_content = sprintf("%s<label class='gfield_label' for='%s'>%s</label>{FIELD}", $admin_buttons, $field_id, esc_html($field_label)); 
  661.             break; 
  662.             default : 
  663.                 $description = self::get_description($field["description"], "gfield_description"); 
  664.                 $field_content = sprintf("%s<label class='gfield_label' for='%s'>%s<span class='gfield_required'>%s</span></label>{FIELD}%s%s", $admin_buttons, $field_id, esc_html($field_label), $field["isRequired"] ? "*" : "", $description, $validation_message); 
  665.             break; 
  666.         } 
  667.  
  668.         if(empty($value)) 
  669.             $value = IS_ADMIN ? $field["defaultValue"] : GFCommon::replace_variables_prepopulate($field["defaultValue"]); 
  670.  
  671.         $field_content = str_replace("{FIELD}", GFCommon::get_field_input($field, $value, 0, $form_id), $field_content); 
  672.  
  673.         return $field_content; 
  674.     }
