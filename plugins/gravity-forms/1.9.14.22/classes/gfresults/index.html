  693.     public function get_results_data( $form, $fields, $search_criteria = array(), $state_array = array(), $max_execution_time = 15 /** seconds */ ) { 
  694.         // todo: add hooks to modify $max_execution_time and $page_size? 
  695.  
  696.         $page_size = 150; 
  697.  
  698.         $time_start = microtime( true ); 
  699.  
  700.         $form_id = $form['id'];

  705.  
  706.  
  707.         if ( $state_array ) {

  711.  
  712.             unset( $data['offset'] );

  718.                 $field_type = GFFormsModel::get_input_type( $field ); 
  719.                 if ( false === isset( $field->choices ) ) { 
  720.                     $field_data[ $field->id ] = 0; 
  721.                     continue; 
  722.                 } 
  723.                 $choices = $field->choices; 
  724.  
  725.                 if ( $field_type == 'likert' && rgar( $field, 'gsurveyLikertEnableMultipleRows' ) ) { 
  726.                     foreach ( $field->gsurveyLikertRows as $row ) { 
  727.                         foreach ( $choices as $choice ) { 
  728.                             $field_data[ $field->id ][ $row['value'] ][ $choice['value'] ] = 0; 
  729.                         } 
  730.                         if ( rgar( $field, 'gsurveyLikertEnableScoring' ) ) { 
  731.                             $field_data[ $field->id ][ $row['value'] ]['row_score_sum'] = 0; 
  732.                         } 
  733.                     } 
  734.                 } else { 
  735.                     if ( ! empty( $choices ) && is_array( $choices ) ) { 
  736.                         foreach ( $choices as $choice ) { 
  737.                             $field_data[ $field->id ][ $choice['value'] ] = 0; 
  738.                         } 
  739.                     } else { 
  740.                         $field_data[ $field->id ] = 0; 
  741.                     } 
  742.                 } 
  743.                 if ( $field_type == 'likert' && rgar( $field, 'gsurveyLikertEnableScoring' ) ) { 
  744.                     $field_data[ $field->id ]['sum_of_scores'] = 0; 
  745.                 } 
  746.             } 
  747.         } 
  748.  
  749.         $count_search_leads = GFAPI::count_entries( $form_id, $search_criteria ); 
  750.         $data['entry_count'] = $count_search_leads; 
  751.  
  752.         $entries_left = $count_search_leads - $offset; 
  753.  
  754.         while ( $entries_left > 0 ) { 
  755.  
  756.             $paging = array( 
  757.                 'offset' => $offset,  
  758.                 'page_size' => $page_size,  
  759.  ); 
  760.  
  761.             $search_leads_time_start = microtime( true ); 
  762.             $leads = GFFormsModel::search_leads( $form_id, $search_criteria, null, $paging ); 
  763.             $search_leads_time_end = microtime( true ); 
  764.             $search_leads_time = $search_leads_time_end - $search_leads_time_start; 
  765.  
  766.             $leads_in_search = count( $leads ); 
  767.  
  768.             $entry_count += $leads_in_search; 
  769.             $leads_processed = 0; 
  770.             foreach ( $leads as $lead ) { 
  771.  
  772.                 $lead_time_start = microtime( true ); 
  773.                 foreach ( $fields as $field ) { 
  774.                     $field_type = GFFormsModel::get_input_type( $field ); 
  775.                     $field_id = $field->id; 
  776.                     $value = RGFormsModel::get_lead_field_value( $lead, $field ); 
  777.  
  778.                     if ( $field_type == 'likert' && rgar( $field, 'gsurveyLikertEnableMultipleRows' ) ) { 
  779.  
  780.                         if ( empty( $value ) ) { 
  781.                             continue; 
  782.                         } 
  783.                         foreach ( $value as $value_vector ) { 
  784.                             if ( empty( $value_vector ) ) { 
  785.                                 continue; 
  786.                             } 
  787.                             list( $row_val, $col_val ) = explode( ':', $value_vector, 2 ); 
  788.                             if ( isset( $field_data[ $field->id ][ $row_val ] ) && isset( $field_data[ $field->id ][ $row_val ][ $col_val ] ) ) { 
  789.                                 $field_data[ $field->id ][ $row_val ][ $col_val ] ++; 
  790.                                 if ( $field->gsurveyLikertEnableScoring ) { 
  791.                                     $field_data[ $field->id ][ $row_val ]['row_score_sum'] += $this->get_likert_row_score( $row_val, $field, $lead ); 
  792.                                 } 
  793.                             } 
  794.                         } 
  795.                     } elseif ( $field_type == 'rank' ) { 
  796.                         $score = count( rgar( $field, 'choices' ) ); 
  797.                         $values = explode( ', ', $value ); 
  798.                         foreach ( $values as $ranked_value ) { 
  799.                             $field_data[ $field->id ][ $ranked_value ] += $score; 
  800.                             $score --; 
  801.                         } 
  802.                     } else { 
  803.  
  804.                         if ( empty( $field->choices ) ) { 
  805.                             if ( false === empty( $value ) ) { 
  806.                                 $field_data[ $field_id ] ++; 
  807.                             } 
  808.                             continue; 
  809.                         } 
  810.                         $choices = $field->choices; 
  811.  
  812.                         foreach ( $choices as $choice ) { 
  813.                             $choice_is_selected = false; 
  814.                             if ( is_array( $value ) ) { 
  815.                                 $choice_value = rgar( $choice, 'value' ); 
  816.                                 if ( in_array( $choice_value, $value ) ) { 
  817.                                     $choice_is_selected = true; 
  818.                                 } 
  819.                             } else { 
  820.                                 if ( RGFormsModel::choice_value_match( $field, $choice, $value ) ) { 
  821.                                     $choice_is_selected = true; 
  822.                                 } 
  823.                             } 
  824.                             if ( $choice_is_selected ) { 
  825.                                 $field_data[ $field_id ][ $choice['value'] ] ++; 
  826.                             } 
  827.                         } 
  828.                     } 
  829.                     if ( $field_type == 'likert' && rgar( $field, 'gsurveyLikertEnableScoring' ) ) { 
  830.                         $field_data[ $field->id ]['sum_of_scores'] += $this->get_likert_score( $field, $lead ); 
  831.                     } 
  832.                 } 
  833.                 $leads_processed ++; 
  834.                 $lead_time_end = microtime( true ); 
  835.                 $total_execution_time = $lead_time_end - $search_leads_time_start; 
  836.                 $lead_execution_time = $lead_time_end - $lead_time_start; 
  837.                 if ( $total_execution_time + $lead_execution_time > $max_execution_time ) { 
  838.                     break; 
  839.                 } 
  840.             } 
  841.             $data['field_data'] = $field_data; 
  842.             if ( isset( $this->_callbacks['calculation'] ) ) { 
  843.                 $data = call_user_func( $this->_callbacks['calculation'], $data, $form, $fields, $leads ); 
  844.                 $field_data = $data['field_data']; 
  845.             } 
  846.             $offset += $leads_processed; 
  847.             $entries_left -= $leads_processed; 
  848.  
  849.             $time_end = microtime( true ); 
  850.             $execution_time = ( $time_end - $time_start ); 
  851.  
  852.             if ( $entries_left > 0 && $execution_time + $search_leads_time > $max_execution_time ) { 
  853.                 $data['status'] = 'incomplete'; 
  854.                 $data['offset'] = $offset; 
  855.                 $progress = $data['entry_count'] > 0 ? round( $data['offset'] / $data['entry_count'] * 100 ) : 0; 
  856.                 $data['progress'] = $progress; 
  857.                 break; 
  858.             } 
  859.  
  860.             if ( $entries_left <= 0 ) { 
  861.                 $data['status'] = 'complete'; 
  862.             } 
  863.         } 
  864.  
  865.         $data['timestamp'] = time(); 
  866.  
  867.         return $data; 
  868.     }
