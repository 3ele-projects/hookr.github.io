  474.         //query db to look for two formats that the feed name could have been auto-generated with 
  475.         //format from migration to add-on framework: 'Feed ' . $counter 
  476.         //new auto-generated format when adding new feed: $short_title . ' Feed ' . $counter

  484.                 $pattern = '/(^Feed|^' . $this->_short_title . ' Feed)\s\d+/'; 
  485.                 preg_match( $pattern, $name, $matches ); 
  486.                 if ( $matches ) { 
  487.                     //number should be characters at the end after a space 
  488.                     $last_space = strrpos( $matches[0], ' ' ); 
  489.                     $digit = substr( $matches[0], $last_space );

  493.                     }
