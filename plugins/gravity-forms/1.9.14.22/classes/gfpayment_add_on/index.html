  621.     protected function get_order_data( $feed, $form, $entry ) {

  636.             $quantity = $product['quantity'] ? $product['quantity'] : 1;

  638.  
  639.             $options = array();

  641.                 foreach ( $product['options'] as $option ) { 
  642.                     $options[] = $option['option_name']; 
  643.                     $product_price += $option['price']; 
  644.                 } 
  645.             }

  647.             $is_trial_or_setup_fee = false; 
  648.  
  649.             if ( ! empty( $trial_field ) && $trial_field == $field_id ) { 
  650.  
  651.                 $trial_amount = $product_price * $quantity; 
  652.                 $is_trial_or_setup_fee = true; 
  653.

  671.  
  672.             $description = '';

  681.                     'description' => $description,  
  682.                     'quantity' => $quantity,  
  683.                     'unit_price' => GFCommon::to_number( $product_price, $entry['currency'] ),  
  684.                     'options' => rgar( $product, 'options' ) 
  685.  );

  691.                     'quantity' => $quantity,

  693.                     'options' => rgar( $product, 'options' ) 
  694.  );

  699.             $trial_amount = rgar( $feed['meta'], 'trial_amount' ) ? GFCommon::to_number( rgar( $feed['meta'], 'trial_amount' ), $entry['currency'] ) : 0; 
  700.         }

  712.         } 
  713.  
  714.         return array( 
  715.             'payment_amount' => $amount,  
  716.             'setup_fee' => $fee_amount,  
  717.             'trial' => $trial_amount,

  786.     /**

  795.      *     'transaction_type' => '',  
  796.      *     'payment_status' => '',  
  797.      *     'note' => '' 
  798.      * ); 
  799.      * 
  800.      * @param array $action 
  801.      * 
  802.      * @return mixed 
  803.      */ 
  804.     private function process_callback_action( $action ) { 
  805.         $this->log_debug( __METHOD__ . '(): Processing callback action.' ); 
  806.         $action = wp_parse_args( 
  807.             $action, array( 
  808.                 'type' => false,  
  809.                 'amount' => false,  
  810.                 'transaction_type' => false,  
  811.                 'transaction_id' => false,  
  812.                 'subscription_id' => false,  
  813.                 'entry_id' => false,  
  814.                 'payment_status' => false,  
  815.                 'note' => false,  
  816.  ) 
  817.  ); 
  818.  
  819.         $result = false; 
  820.  
  821.         if ( rgar( $action, 'id' ) && $this->is_duplicate_callback( $action['id'] ) ) { 
  822.             return new WP_Error( 'duplicate', sprintf( esc_html__( 'This webhook has already been processed (Event Id: %s)', 'gravityforms' ), $action['id'] ) ); 
  823.         } 
  824.  
  825.         $entry = GFAPI::get_entry( $action['entry_id'] ); 
  826.         if ( ! $entry || is_wp_error( $entry ) ) { 
  827.             return $result; 
  828.         } 
  829.  
  830.         //$action = do_action('gform_action_pre_payment_callback', $action, $entry); 
  831.         do_action( 'gform_action_pre_payment_callback', $action, $entry ); 
  832.         if ( has_filter( 'gform_action_pre_payment_callback' ) ) { 
  833.             $this->log_debug( __METHOD__ . '(): Executing functions hooked to gform_action_pre_payment_callback.' ); 
  834.         } 
  835.  
  836.         switch ( $action['type'] ) { 
  837.             case 'complete_payment': 
  838.                 $result = $this->complete_payment( $entry, $action ); 
  839.                 break; 
  840.             case 'refund_payment': 
  841.                 $result = $this->refund_payment( $entry, $action ); 
  842.                 break; 
  843.             case 'fail_payment': 
  844.                 $result = $this->fail_payment( $entry, $action ); 
  845.                 break; 
  846.             case 'add_pending_payment': 
  847.                 $result = $this->add_pending_payment( $entry, $action ); 
  848.                 break; 
  849.             case 'void_authorization': 
  850.                 $result = $this->void_authorization( $entry, $action ); 
  851.                 break; 
  852.             case 'create_subscription': 
  853.                 $result = $this->start_subscription( $entry, $action );

  862.                 break; 
  863.             case 'add_subscription_payment': 
  864.                 $result = $this->add_subscription_payment( $entry, $action );

  866.             case 'fail_subscription_payment': 
  867.                 $result = $this->fail_subscription_payment( $entry, $action );

  869.             default: 
  870.                 // handle custom events 
  871.                 if ( is_callable( array( $this, rgar( $action, 'callback' ) ) ) ) { 
  872.                     $result = call_user_func_array( array( $this, $action['callback'] ), array( $entry, $action ) ); 
  873.                 } 
  874.                 break; 
  875.         } 
  876.  
  877.         if ( rgar( $action, 'id' ) && $result ) {

  879.         } 
  880.  
  881.         /**

  883.          * 
  884.          * @param array $entry The Entry Object

  904.          * - create_subscription
