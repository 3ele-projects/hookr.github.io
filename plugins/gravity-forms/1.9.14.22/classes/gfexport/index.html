  559.     private static function get_field_row_count( $form, $exported_field_ids, $entry_count ) { 
  560.         $list_fields = GFAPI::get_fields_by_type( $form, array( 'list' ), true ); 
  561.  
  562.         //only getting fields that have been exported 
  563.         $field_ids = ''; 
  564.         foreach ( $list_fields as $field ) { 
  565.             if ( in_array( $field->id, $exported_field_ids ) && $field->enableColumns ) { 
  566.                 $field_ids .= $field->id . ', '; 
  567.             } 
  568.         } 
  569.  
  570.         if ( empty( $field_ids ) ) { 
  571.             return array(); 
  572.         } 
  573.  
  574.         $field_ids = substr( $field_ids, 0, strlen( $field_ids ) - 1 ); 
  575.  
  576.         $page_size = 200; 
  577.         $offset = 0; 
  578.  
  579.         $row_counts = array(); 
  580.         global $wpdb; 
  581.  
  582.         $go_to_next_page = true; 
  583.  
  584.         while ( $go_to_next_page ) { 
  585.             $sql = "SELECT d.field_number as field_id, ifnull(l.value, d.value) as value 
  586.                     FROM {$wpdb->prefix}rg_lead_detail d 
  587.                     LEFT OUTER JOIN {$wpdb->prefix}rg_lead_detail_long l ON d.id = l.lead_detail_id 
  588.                     WHERE d.form_id={$form['id']} AND cast(d.field_number as decimal) IN ({$field_ids}) 
  589.                     LIMIT {$offset}, {$page_size}"; 
  590.  
  591.             $results = $wpdb->get_results( $sql, ARRAY_A ); 
  592.  
  593.             foreach ( $results as $result ) { 
  594.                 $list = unserialize( $result['value'] ); 
  595.                 $current_row_count = isset( $row_counts[ $result['field_id'] ] ) ? intval( $row_counts[ $result['field_id'] ] ) : 0; 
  596.  
  597.                 if ( is_array( $list ) && count( $list ) > $current_row_count ) { 
  598.                     $row_counts[ $result['field_id'] ] = count( $list ); 
  599.                 } 
  600.             } 
  601.  
  602.             $offset += $page_size; 
  603.  
  604.             $go_to_next_page = count( $results ) == $page_size; 
  605.         } 
  606.  
  607.         return $row_counts; 
  608.     }

  623.     public static function start_export( $form ) { 
  624.  
  625.         $form_id = $form['id']; 
  626.         $fields = $_POST['export_field']; 
  627.  
  628.         $start_date = empty( $_POST['export_date_start'] ) ? '' : self::get_gmt_date( $_POST['export_date_start'] . ' 00:00:00' ); 
  629.         $end_date = empty( $_POST['export_date_end'] ) ? '' : self::get_gmt_date( $_POST['export_date_end'] . ' 23:59:59' ); 
  630.  
  631.         $search_criteria['status'] = 'active'; 
  632.         $search_criteria['field_filters'] = GFCommon::get_field_filters_from_post( $form ); 
  633.         if ( ! empty( $start_date ) ) { 
  634.             $search_criteria['start_date'] = $start_date; 
  635.         } 
  636.  
  637.         if ( ! empty( $end_date ) ) { 
  638.             $search_criteria['end_date'] = $end_date; 
  639.         } 
  640.  
  641.         $sorting = array( 'key' => 'date_created', 'direction' => 'DESC', 'type' => 'info' ); 
  642.  
  643.         GFCommon::log_debug( "GFExport::start_export(): Start date: {$start_date}" ); 
  644.         GFCommon::log_debug( "GFExport::start_export(): End date: {$end_date}" ); 
  645.  
  646.         $form = self::add_default_export_fields( $form ); 
  647.  
  648.         $entry_count = GFAPI::count_entries( $form_id, $search_criteria ); 
  649.  
  650.         $page_size = 100; 
  651.         $offset = 0; 
  652.  
  653.         //Adding BOM marker for UTF-8 
  654.         $lines = chr( 239 ) . chr( 187 ) . chr( 191 ); 
  655.  
  656.         // set the separater 
  657.         $separator = gf_apply_filters( array( 'gform_export_separator', $form_id ), ', ', $form_id ); 
  658.  
  659.         $field_rows = self::get_field_row_count( $form, $fields, $entry_count ); 
  660.  
  661.         //writing header 
  662.         $headers = array(); 
  663.         foreach ( $fields as $field_id ) { 
  664.             $field = RGFormsModel::get_field( $form, $field_id ); 
  665.             $label = gf_apply_filters( array( 'gform_entries_field_header_pre_export', $form_id, $field_id ), GFCommon::get_label( $field, $field_id ), $form, $field ); 
  666.             $value = str_replace( '"', '""', $label ); 
  667.  
  668.             GFCommon::log_debug( "GFExport::start_export(): Header for field ID {$field_id}: {$value}" ); 
  669.  
  670.             if ( strpos( $value, '=' ) === 0 ) { 
  671.                 // Prevent Excel formulas 
  672.                 $value = "'" . $value; 
  673.             } 
  674.  
  675.             $headers[ $field_id ] = $value; 
  676.  
  677.             $subrow_count = isset( $field_rows[ $field_id ] ) ? intval( $field_rows[ $field_id ] ) : 0; 
  678.             if ( $subrow_count == 0 ) { 
  679.                 $lines .= '"' . $value . '"' . $separator; 
  680.             } else { 
  681.                 for ( $i = 1; $i <= $subrow_count; $i ++ ) { 
  682.                     $lines .= '"' . $value . ' ' . $i . '"' . $separator; 
  683.                 } 
  684.             } 
  685.  
  686.             GFCommon::log_debug( "GFExport::start_export(): Lines: {$lines}" ); 
  687.         } 
  688.         $lines = substr( $lines, 0, strlen( $lines ) - 1 ) . "\n"; 
  689.  
  690.         //paging through results for memory issues 
  691.         while ( $entry_count > 0 ) { 
  692.  
  693.             $paging = array( 
  694.                 'offset' => $offset,  
  695.                 'page_size' => $page_size,  
  696.  ); 
  697.             $leads = GFAPI::get_entries( $form_id, $search_criteria, $sorting, $paging ); 
  698.  
  699.             $leads = gf_apply_filters( array( 'gform_leads_before_export', $form_id ), $leads, $form, $paging ); 
  700.  
  701.             foreach ( $leads as $lead ) { 
  702.                 foreach ( $fields as $field_id ) { 
  703.                     switch ( $field_id ) { 
  704.                         case 'date_created' : 
  705.                             $lead_gmt_time = mysql2date( 'G', $lead['date_created'] ); 
  706.                             $lead_local_time = GFCommon::get_local_timestamp( $lead_gmt_time ); 
  707.                             $value = date_i18n( 'Y-m-d H:i:s', $lead_local_time, true ); 
  708.                             break; 
  709.                         default : 
  710.                             $field = RGFormsModel::get_field( $form, $field_id ); 
  711.  
  712.                             $value = is_object( $field ) ? $field->get_value_export( $lead, $field_id, false, true ) : rgar( $lead, $field_id ); 
  713.                             $value = apply_filters( 'gform_export_field_value', $value, $form_id, $field_id, $lead ); 
  714.  
  715.                             GFCommon::log_debug( "GFExport::start_export(): Value for field ID {$field_id}: {$value}" ); 
  716.                             break; 
  717.                     } 
  718.  
  719.                     if ( isset( $field_rows[ $field_id ] ) ) { 
  720.                         $list = empty( $value ) ? array() : unserialize( $value ); 
  721.  
  722.                         foreach ( $list as $row ) { 
  723.                             $row_values = array_values( $row ); 
  724.                             $row_str = implode( '|', $row_values ); 
  725.  
  726.                             if ( strpos( $row_str, '=' ) === 0 ) { 
  727.                                 // Prevent Excel formulas 
  728.                                 $row_str = "'" . $row_str; 
  729.                             } 
  730.  
  731.                             $lines .= '"' . str_replace( '"', '""', $row_str ) . '"' . $separator; 
  732.                         } 
  733.  
  734.                         //filling missing subrow columns (if any) 
  735.                         $missing_count = intval( $field_rows[ $field_id ] ) - count( $list ); 
  736.                         for ( $i = 0; $i < $missing_count; $i ++ ) { 
  737.                             $lines .= '""' . $separator; 
  738.                         } 
  739.                     } else { 
  740.                         $value = maybe_unserialize( $value ); 
  741.                         if ( is_array( $value ) ) { 
  742.                             $value = implode( '|', $value ); 
  743.                         } 
  744.  
  745.                         if ( strpos( $value, '=' ) === 0 ) { 
  746.                             // Prevent Excel formulas 
  747.                             $value = "'" . $value; 
  748.                         } 
  749.  
  750.                         $lines .= '"' . str_replace( '"', '""', $value ) . '"' . $separator; 
  751.                     } 
  752.                 } 
  753.                 $lines = substr( $lines, 0, strlen( $lines ) - 1 ); 
  754.  
  755.                 GFCommon::log_debug( "GFExport::start_export(): Lines: {$lines}" ); 
  756.  
  757.                 $lines .= "\n"; 
  758.             } 
  759.  
  760.             $offset += $page_size; 
  761.             $entry_count -= $page_size; 
  762.  
  763.             if ( ! seems_utf8( $lines ) ) { 
  764.                 $lines = utf8_encode( $lines ); 
  765.             } 
  766.  
  767.             $lines = apply_filters( 'gform_export_lines', $lines ); 
  768.  
  769.             echo $lines; 
  770.  
  771.             $lines = ''; 
  772.         } 
  773.  
  774.         /** 
  775.          * Fires after exporting all the entries in form 
  776.          * 
  777.          * @param array $form The Form object to get the entries from 
  778.          * @param string $start_date The start date for when the export of entries should take place 
  779.          * @param string $end_date The end date for when the export of entries should stop 
  780.          * @param array $fields The specified fields where the entries should be exported from 
  781.          */ 
  782.         do_action( 'gform_post_export_entries', $form, $start_date, $end_date, $fields ); 
  783.  
  784.     }
