  592.     public static function save_new_form() { 
  593.  
  594.         if ( ! check_admin_referer( 'gf_save_new_form', 'gf_save_new_form' ) ) { 
  595.             die( json_encode( array( 'error' => __( 'There was an issue creating your form.', 'gravityforms' ) ) ) ); 
  596.         } 
  597.  
  598.         GFFormsModel::ensure_tables_exist(); 
  599.  
  600.         require_once( GFCommon::get_base_path() . '/form_detail.php' ); 
  601.  
  602.         $form_json = rgpost( 'form' ); 
  603.  
  604.         $form = json_decode( $form_json, true ); 
  605.  
  606.         if ( empty( $form['title'] ) ) { 
  607.             $result = array( 'error' => __( 'Please enter a form title.', 'gravityforms' ) ); 
  608.             die( json_encode( $result ) ); 
  609.         } 
  610.  
  611.         $result = GFFormDetail::save_form_info( 0, $form_json ); 
  612.  
  613.         switch ( rgar( $result, 'status' ) ) { 
  614.             case 'invalid_json': 
  615.                 $result['error'] = __( 'There was an issue creating your form.', 'gravityforms' ); 
  616.                 die( json_encode( $result ) ); 
  617.  
  618.             case 'duplicate_title': 
  619.                 $result['error'] = __( 'Please enter a unique form title.', 'gravityforms' ); 
  620.                 die( json_encode( $result ) ); 
  621.  
  622.             default: 
  623.                 $form_id = absint( $result['status'] ); 
  624.                 die( json_encode( array( 'redirect' => admin_url( "admin.php?page=gf_edit_forms&id={$form_id}" ) ) ) ); 
  625.         } 
  626.     }
