  602.     /** 
  603.      * Modified _out() method so not each call will add a newline to the output. 
  604.      */ 
  605.     protected function _straightOut($s) 
  606.     { 
  607.         if (!is_subclass_of($this, 'TCPDF')) { 
  608.             if ($this->state == 2) { 
  609.                 $this->pages[$this->page] .= $s; 
  610.             } else { 
  611.                 $this->buffer .= $s; 
  612.             } 
  613.  
  614.         } else { 
  615.             if ($this->state == 2) { 
  616.                 if ($this->inxobj) { 
  617.                     // we are inside an XObject template 
  618.                     $this->xobjects[$this->xobjid]['outdata'] .= $s; 
  619.                 } else if ((!$this->InFooter) AND isset($this->footerlen[$this->page]) AND ($this->footerlen[$this->page] > 0)) { 
  620.                     // puts data before page footer 
  621.                     $pagebuff = $this->getPageBuffer($this->page); 
  622.                     $page = substr($pagebuff, 0, -$this->footerlen[$this->page]); 
  623.                     $footer = substr($pagebuff, -$this->footerlen[$this->page]);

  625.                     // update footer position 
  626.                     $this->footerpos[$this->page] += strlen($s); 
  627.                 } else { 
  628.                     // set page data 
  629.                     $this->setPageBuffer($this->page, $s, true); 
  630.                 } 
  631.             } else if ($this->state > 0) { 
  632.                 // set general data 
  633.                 $this->setBuffer($s); 
  634.             } 
  635.         }
