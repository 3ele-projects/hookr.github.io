  206.   /** 
  207.    * Determines if we should process the PDF at this stage 
  208.    * Fires just before the main WP_Query is executed (we don't need it) 
  209.    * 
  210.    * @since 4.0 
  211.    * 
  212.    * @return void 
  213.    */ 
  214.   public function process_pdf_endpoint() { 
  215.  
  216.       /** exit early if all the required URL parameters aren't met */ 
  217.       if ( empty( $GLOBALS['wp']->query_vars['gpdf'] ) || empty( $GLOBALS['wp']->query_vars['pid'] ) || empty( $GLOBALS['wp']->query_vars['lid'] ) ) { 
  218.           return null; 
  219.       } 
  220.  
  221.       $pid = $GLOBALS['wp']->query_vars['pid']; 
  222.       $lid = (int) $GLOBALS['wp']->query_vars['lid']; 
  223.       $action = ( ( isset( $GLOBALS['wp']->query_vars['action'] ) ) && $GLOBALS['wp']->query_vars['action'] == 'download' ) ? 'download' : 'view'; 
  224.  
  225.       $this->log->addNotice( 'Processing PDF endpoint.', [ 
  226.           'pid' => $pid,  
  227.           'lid' => $lid,  
  228.           'action' => $action,  
  229.  ] ); 
  230.  
  231.       /**  Send to our model to handle validation / authentication */ 
  232.       $results = $this->model->process_pdf( $pid, $lid, $action ); 
  233.  
  234.       /** if error, display to user */ 
  235.       if ( is_wp_error( $results ) ) { 
  236.           $this->pdf_error( $results ); 
  237.       } 
  238.   } 
  239.  
  240.   /** 
  241.    * Determines if we should process the legacy PDF endpoint at this stage (the one with $_GET variables) 
  242.    * Fires just before the main WP_Query is executed (we don't need it) 
  243.    * 
  244.    * @since 4.0 
  245.    * 
  246.    * @return void 
  247.    */ 
  248.   public function process_legacy_pdf_endpoint() { 
  249.  
  250.       /** exit early if all our required parameters aren't met */ 
  251.       if ( empty( $_GET['gf_pdf'] ) || empty( $_GET['fid'] ) || empty( $_GET['lid'] ) || empty( $_GET['template'] ) ) { 
  252.           return null; 
  253.       } 
  254.  
  255.       $config = [ 
  256.           'lid' => $_GET['lid'],  
  257.           'fid' => (int) $_GET['fid'],  
  258.           'aid' => ( isset( $_GET['aid'] ) ) ? (int) $_GET['aid'] : false,  
  259.           'template' => substr( $_GET['template'], 0, -4 ), /** strip .php from the template name */ 
  260.           'action' => ( isset( $_GET['download'] ) ) ? 'download' : 'view',  
  261.  ]; 
  262.  
  263.       $this->log->addNotice( 'Processing Legacy PDF endpoint.', [ 
  264.           'config' => $config,  
  265.  ] ); 
  266.  
  267.       /** Attempt to find a valid config */ 
  268.       $pid = $this->model->get_legacy_config( $config ); 
  269.  
  270.       if ( is_wp_error( $pid ) ) { 
  271.           return $this->pdf_error( $pid ); 
  272.       } 
  273.  
  274.       /** Store our ids in the WP query_vars object */ 
  275.       $GLOBALS['wp']->query_vars['gpdf'] = 1; 
  276.       $GLOBALS['wp']->query_vars['pid'] = $pid; 
  277.       $GLOBALS['wp']->query_vars['lid'] = $config['lid']; 
  278.  
  279.       /** Send to our model to handle validation / authentication */ 
  280.       $results = $this->model->process_pdf( $pid, $config['lid'], $config['action'] ); 
  281.  
  282.       /** if error, display to user */ 
  283.       if ( is_wp_error( $results ) ) { 
  284.           $this->pdf_error( $results ); 
  285.       } 
  286.   } 
  287.  
  288.   /** 
  289.    * Output PDF error to user 
  290.    * 
  291.    * @param  Object $error The WP_Error object 
  292.    * 
  293.    * @return void 
  294.    * 
  295.    * @since 4.0 
  296.    */ 
  297.   private function pdf_error( $error ) { 
  298.  
  299.       $this->log->addError( 'PDF Generation Error.', [ 
  300.           'WP_Error_Message' => $error->get_error_message(),  
  301.           'WP_Error_Code' => $error->get_error_code(),  
  302.  ] ); 
  303.  
  304.       /** only display detailed error to admins */ 
  305.       $whitelist_errors = [ 'timeout_expired', 'access_denied' ]; 
  306.       if ( $this->gform->has_capability( 'gravityforms_view_settings' ) || in_array( $error->get_error_code(), $whitelist_errors ) ) { 
  307.           wp_die( $error->get_error_message() ); 
  308.       } else { 
  309.           wp_die( esc_html__( 'There was a problem generating your PDF', 'gravity-forms-pdf-extended' ) ); 
  310.       } 
  311.   }
