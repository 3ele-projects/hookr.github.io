  546.                 $testID = get_page_by_title( $post_title, ARRAY_A, 'product' ); 
  547.                 if ($testID) { 
  548.                     $this->logger->log( __( 'Product found by title', 'woocommerce-csvimport' ) ); 
  549.                     $id = $testID[ 'ID' ]; 
  550.                     $this->new = FALSE; 
  551.                 } else { 
  552.                     $this->logger->log( __( 'Product not found', 'woocommerce-csvimport' ) ); 
  553.                 } 
  554.             } 
  555.         } 
  556.  
  557.         //check for if we need to merge the product 
  558.  
  559.         if ($id && $woocsv_import->get_merge_products() == 1) { 
  560.             $this->merge_product( $id ); 
  561.         } 
  562.         //fill in the product body 
  563.         foreach ($this->body as $key => $value) { 
  564.             if (in_array( $key, $this->header )) { 
  565.                 $key_body = array_search( $key, $this->header ); 
  566.                 if (isset ($this->raw_data[ $key_body ])) { 
  567.                     $this->body[ $key ] = $this->raw_data[ $key_body ]; 
  568.                 } 
  569.             } 
  570.         } 
  571.  
  572.         // get the author 
  573.         if (isset($this->body[ 'post_author' ]) && in_array( 'post_author', $this->header )) { 
  574.  
  575.             $user = get_user_by( ($woocsv_import->get_match_author_by()) ? $woocsv_import->get_match_author_by() : 'login',  
  576.                 $this->body[ 'post_author' ] ); 
  577.             if ($user) { 
  578.                 $this->body[ 'post_author' ] = $user->ID; 
  579.                 $this->logger->log( __( 'user found', 'woocommerce-csvimport' ) ); 
  580.             } else { 
  581.                 $this->body[ 'post_author' ] = ''; 
  582.                 $this->logger->log( __( 'user not found', 'woocommerce-csvimport' ) ); 
  583.             } 
  584.         } 
  585.  
  586.         //fill in the ID if the product already exists 
  587.         if ($id) { 
  588.             $this->body[ 'ID' ] = $id; 
  589.         } 
  590.  
  591.         //fill in the meta data 
  592.         // @ since 3.0.5 
  593.         // trim meta values to loose spaces 
  594.         foreach ($this->meta as $key => $value) { 
  595.             if (in_array( substr( $key, 1 ), $this->header )) { 
  596.                 $this->meta[ $key ] = trim( $this->raw_data[ array_search( substr( $key, 1 ), $this->header ) ] ); 
  597.             } 
  598.         } 
  599.  
  600.         // @ since 3.0.5 
  601.         // if the product is new add total_sales to show it in the front end 
  602.         // some themes needed total_sales for popularity sorting 
  603.         if (!empty ($this->body[ 'ID' ])) { 
  604.             $this->meta[ 'total_sales' ] = 0; 
  605.         } 
  606.  
  607.         //check if there are tags 
  608.         if (in_array( 'tags', $this->header )) { 
  609.             $key = array_search( 'tags', $this->header ); 
  610.             $this->tags = $this->raw_data[ $key ]; 
  611.         } 
  612.  
  613.         //check if there is a shipping 
  614.         if (in_array( 'shipping_class', $this->header )) { 
  615.             $key = array_search( 'shipping_class', $this->header ); 
  616.             $this->shipping_class = trim( $this->raw_data[ $key ] ); 
  617.         } 
  618.  
  619.         //check if there are categories 
  620.         if (in_array( 'category', $this->header )) { 
  621.             foreach ($this->header as $key => $value) { 
  622.                 if ($value == 'category') { 
  623.                     $this->categories[] = $this->raw_data[ $key ]; 
  624.                 } 
  625.             } 
  626.         } 
  627.  
  628.         /** change_stock */ 
  629.         if (in_array( 'change_stock', $this->header )) { 
  630.             $key = array_search( 'change_stock', $this->header ); 
  631.             $change_stock = $this->raw_data[ $key ]; 
  632.  
  633.             //get the stock 
  634.             $stock = get_post_meta( $this->body[ 'ID' ], '_stock', TRUE ); 
  635.  
  636.             //if the stock is empty set it to 0 
  637.             if (!$stock) { 
  638.                 $stock = 0; 
  639.             } 
  640.  
  641.             //calculate the new stock level 
  642.             $new_stock = $stock + $change_stock; 
  643.  
  644.             //set new stock in the meta 
  645.             $this->meta[ '_stock' ] = $new_stock; 
  646.  
  647.             //set log 
  648.             $this->logger->log( sprintf( __( 'Change stock modus: stock changed from %s to %s', 'woocommerce-csvimport' ), $stock, $new_stock ) ); 
  649.         } 
  650.  
  651.         //check if there is a featured image 
  652.         if (in_array( 'featured_image', $this->header )) { 
  653.             $key = array_search('featured_image', $this->header); 
  654.             $this->featured_image = $this->raw_data[ $key ]; 
  655.         } 
  656.  
  657.         //check if there is a product gallery 
  658.         if (in_array( 'product_gallery', $this->header )) { 
  659.             $key = array_search( 'product_gallery', $this->header ); 
  660.             $this->product_gallery = $this->raw_data[ $key ]; 
  661.         } 
  662.  
  663.         $this->logger->log(__($this->featured_image, 'woocommerce-csvimport')); 
  664.  
  665.         do_action( 'woocsv_product_after_fill_in_data' ); 
  666.  
  667.     }

  669.     public function normalizePrice ($num) 
  670.     { 
  671.  
  672.         if (!$num) { 
  673.             return false; 
  674.         } 
  675.  
  676.         $dotPos = strrpos( $num, '.' ); 
  677.         $commaPos = strrpos( $num, ', ' ); 
  678.         $sep = (($dotPos > $commaPos) && $dotPos) ? $dotPos : ((($commaPos > $dotPos) && $commaPos) ? $commaPos : FALSE); 
  679.  
  680.         if (!$sep) { 
  681.             return floatval( preg_replace( "/[^0-9]/", "", $num ) ); 
  682.         } 
  683.  
  684.         return floatval( preg_replace( "/[^0-9]/", "", substr( $num, 0, $sep ) ) . '.' . preg_replace( "/[^0-9]/", "",  
  685.                 substr( $num, $sep + 1, strlen( $num ) ) ) ); 
  686.     }
