  96. /** 
  97.  * Retrieve coordinates for an address 
  98.  * 
  99.  * Coordinates are cached using transients and a hash of the address 
  100.  * 
  101.  * @access      private 
  102.  * @since       1.0 
  103.  * @return      void 
  104. */ 
  105. function pw_map_get_coordinates( $address, $force_refresh = false ) { 
  106.  
  107.   $address_hash = md5( $address ); 
  108.  
  109.   $coordinates = get_transient( $address_hash ); 
  110.  
  111.   if ( $force_refresh || $coordinates === false ) { 
  112.  
  113.       $args = apply_filters( 'pw_map_query_args', array( 'address' => urlencode( $address ), 'sensor' => 'false' ) ); 
  114.       $url = add_query_arg( $args, 'http://maps.googleapis.com/maps/api/geocode/json' ); 
  115.       $response = wp_remote_get( $url ); 
  116.  
  117.       if( is_wp_error( $response ) ) 
  118.           return; 
  119.  
  120.       $data = wp_remote_retrieve_body( $response ); 
  121.  
  122.       if( is_wp_error( $data ) ) 
  123.           return; 
  124.  
  125.       if ( $response['response']['code'] == 200 ) { 
  126.  
  127.           $data = json_decode( $data ); 
  128.  
  129.           if ( $data->status === 'OK' ) { 
  130.  
  131.               $coordinates = $data->results[0]->geometry->location; 
  132.  
  133.               $cache_value['lat'] = $coordinates->lat; 
  134.               $cache_value['lng'] = $coordinates->lng; 
  135.               $cache_value['address'] = (string) $data->results[0]->formatted_address; 
  136.  
  137.               // cache coordinates for 3 months 
  138.               set_transient($address_hash, $cache_value, 3600*24*30*3); 
  139.               $data = $cache_value; 
  140.  
  141.           } elseif ( $data->status === 'ZERO_RESULTS' ) { 
  142.               return __( 'No location found for the entered address.', 'simple-google-maps-short-code' ); 
  143.           } elseif( $data->status === 'INVALID_REQUEST' ) { 
  144.               return __( 'Invalid request. Did you enter an address?', 'simple-google-maps-short-code' ); 
  145.           } else { 
  146.               return __( 'Something went wrong while retrieving your map, please ensure you have entered the short code correctly.', 'simple-google-maps-short-code' ); 
  147.           } 
  148.  
  149.       } else { 
  150.           return __( 'Unable to contact Google API service.', 'simple-google-maps-short-code' ); 
  151.       } 
  152.  
  153.   } else { 
  154.      // return cached results 
  155.      $data = $coordinates; 
  156.   } 
  157.  
  158.   return $data; 
  159. }
