  328.     function prepare_export( $subscribers ) {

  333.         $exportcsv = _x('User Email, User Type, User Name, Confirm Date, IP', 'Comma Separated Column Header names for CSV Export' , 'subscribe2'); 
  334.         $all_cats = $this->all_cats(false, 'ID'); 
  335.  
  336.         foreach ($all_cats as $cat) { 
  337.             $exportcsv .= ", " . html_entity_decode($cat->cat_name, ENT_QUOTES); 
  338.             $cat_ids[] = $cat->term_id; 
  339.         } 
  340.         $exportcsv .= "\r\n"; 
  341.  
  342.         if ( !function_exists('get_userdata') ) { 
  343.             require_once(ABSPATH . WPINC . '/pluggable.php'); 
  344.         } 
  345.  
  346.         foreach ( $subscribers as $subscriber ) { 
  347.             if ( $this->is_registered($subscriber) ) { 
  348.                 $user_ID = $this->get_user_id( $subscriber ); 
  349.                 $user_info = get_userdata( $user_ID ); 
  350.  
  351.                 $cats = explode(', ', get_user_meta($user_ID, $this->get_usermeta_keyname('s2_subscribed'), true)); 
  352.                 $subscribed_cats = ''; 
  353.                 foreach ( $cat_ids as $cat ) { 
  354.                     (in_array($cat, $cats)) ? $subscribed_cats .= ", Yes" : $subscribed_cats .= ", No"; 
  355.                 } 
  356.  
  357.                 $exportcsv .= $subscriber . ', '; 
  358.                 $exportcsv .= __('Registered User', 'subscribe2'); 
  359.                 $exportcsv .= ', ' . $user_info->display_name; 
  360.                 $exportcsv .= ', , ' . $subscribed_cats . "\r\n"; 
  361.             } else { 
  362.                 if ( $this->is_public($subscriber) === '1' ) { 
  363.                     $exportcsv .= $subscriber . ', ' . __('Confirmed Public Subscriber', 'subscribe2') . ', , ' . $this->signup_date($subscriber) . ', ' . $this->signup_ip($subscriber) . "\r\n"; 
  364.                 } elseif ( $this->is_public($subscriber) === '0' ) { 
  365.                     $exportcsv .= $subscriber . ', ' . __('Unconfirmed Public Subscriber', 'subscribe2') . ', , ' . $this->signup_date($subscriber) . ', ' . $this->signup_ip($subscriber) . "\r\n";

  818.         $useremails = implode(", ", array_map(array($this, 'prepare_in_data'), $useremails)); 
  819.         $ids = $wpdb->get_col("SELECT ID FROM $wpdb->users WHERE user_email IN ($useremails)"); 
  820.         $ids = implode(', ', array_map(array($this, 'prepare_in_data'), $ids)); 
  821.         $sql = "UPDATE $wpdb->usermeta SET meta_value='{$format}' WHERE meta_key='" . $this->get_usermeta_keyname('s2_format') . "' AND user_id IN ($ids)";

  823.     } // end format_change()

  825.     /** 
  826.     Handles bulk update to digest preferences 
  827.     */ 
  828.     function digest_change($emails, $digest) { 
  829.         if ( empty($digest) ) { return; } 
  830.  
  831.         global $wpdb; 
  832.         $useremails = explode(", \r\n", $emails); 
  833.         $useremails = implode(", ", array_map(array($this, 'prepare_in_data'), $useremails)); 
  834.  
  835.         $sql = "SELECT ID FROM $wpdb->users WHERE user_email IN ($useremails)"; 
  836.         $user_IDs = $wpdb->get_col($sql); 
  837.  
  838.         if ( $digest == 'digest' ) { 
  839.             $exclude = explode(', ', $this->subscribe2_options['exclude']); 
  840.             if ( !empty($exclude) ) { 
  841.                 $all_cats = $this->all_cats(true, 'ID'); 
  842.             } else { 
  843.                 $all_cats = $this->all_cats(false, 'ID'); 
  844.             } 
  845.  
  846.             $cats_string = ''; 
  847.             foreach ( $all_cats as $cat ) { 
  848.                 ('' == $cats_string) ? $cats_string = "$cat->term_id" : $cats_string .= ", $cat->term_id"; 
  849.             } 
  850.  
  851.             foreach ( $user_IDs as $user_ID ) { 
  852.                 foreach ( $all_cats as $cat ) { 
  853.                     update_user_meta($user_ID, $this->get_usermeta_keyname('s2_cat') . $cat->term_id, $cat->term_id); 
  854.                 } 
  855.                 update_user_meta($user_ID, $this->get_usermeta_keyname('s2_subscribed'), $cats_string); 
  856.             } 
  857.         } elseif ( $digest == '-1' ) { 
  858.             foreach ( $user_IDs as $user_ID ) { 
  859.                 $cats = explode(', ', get_user_meta($user_ID, $this->get_usermeta_keyname('s2_subscribed'), true)); 
  860.                 foreach ( $cats as $id ) { 
  861.                     delete_user_meta($user_ID, $this->get_usermeta_keyname('s2_cat') . $id); 
  862.                 } 
  863.                 delete_user_meta($user_ID, $this->get_usermeta_keyname('s2_subscribed')); 
  864.             } 
  865.         } 
  866.     } // end digest_change()

  869.     /** 
  870.     Autosubscribe registered users to newly created categories 
  871.     if registered user has selected this option 
  872.     */ 
  873.     function new_category($new_category='') { 
  874.         if ( 'no' == $this->subscribe2_options['show_autosub'] ) { return; } 
  875.  
  876.         global $wpdb; 
  877.         if ( $this->subscribe2_options['email_freq'] != 'never' ) { 
  878.             // if we are doing digests add new categories to users who are currently opted in 
  879.             $sql = $wpdb->prepare("SELECT DISTINCT user_id FROM $wpdb->usermeta WHERE meta_key=%s AND meta_value<>''", $this->get_usermeta_keyname('s2_subscribed')); 
  880.             $user_IDs = $wpdb->get_col($sql); 
  881.             foreach ( $user_IDs as $user_ID ) { 
  882.                 $old_cats = get_user_meta($user_ID, $this->get_usermeta_keyname('s2_subscribed'), true); 
  883.                 $old_cats = explode(', ', $old_cats); 
  884.                 $newcats = array_merge($old_cats, (array)$new_category); 
  885.                 update_user_meta($user_ID, $this->get_usermeta_keyname('s2_cat') . $new_category, $new_category); 
  886.                 update_user_meta($user_ID, $this->get_usermeta_keyname('s2_subscribed'), implode(', ', $newcats)); 
  887.             } 
  888.             return; 
  889.         } 
  890.  
  891.         if ( 'yes' == $this->subscribe2_options['show_autosub'] ) { 
  892.             if ( $this->s2_mu ) { 
  893.                 $sql = $wpdb->prepare("SELECT DISTINCT a.user_id FROM $wpdb->usermeta AS a INNER JOIN $wpdb->usermeta AS b WHERE a.user_id = b.user_id AND a.meta_key=%s AND a.meta_value='yes' AND b.meta_key=%s", $this->get_usermeta_keyname('s2_autosub'), $this->get_usermeta_keyname('s2_subscribed')); 
  894.             } else { 
  895.                 $sql = $wpdb->prepare("SELECT DISTINCT user_id FROM $wpdb->usermeta WHERE $wpdb->usermeta.meta_key=%s AND $wpdb->usermeta.meta_value='yes'", $this->get_usermeta_keyname('s2_autosub')); 
  896.             } 
  897.             $user_IDs = $wpdb->get_col($sql); 
  898.             if ( '' == $user_IDs ) { return; } 
  899.  
  900.             foreach ( $user_IDs as $user_ID ) { 
  901.                 $old_cats = get_user_meta($user_ID, $this->get_usermeta_keyname('s2_subscribed'), true); 
  902.                 if ( empty($old_cats) ) { 
  903.                     $newcats = (array)$new_category; 
  904.                 } else { 
  905.                     $old_cats = explode(', ', $old_cats); 
  906.                     $newcats = array_merge($old_cats, (array)$new_category); 
  907.                 } 
  908.                 // add subscription to these cat IDs 
  909.                 update_user_meta($user_ID, $this->get_usermeta_keyname('s2_cat') . $new_category, $new_category); 
  910.                 update_user_meta($user_ID, $this->get_usermeta_keyname('s2_subscribed'), implode(', ', $newcats)); 
  911.             } 
  912.         } elseif ( 'exclude' == $this->subscribe2_options['show_autosub'] ) { 
  913.             $excluded_cats = explode(', ', $this->subscribe2_options['exclude']); 
  914.             $excluded_cats[] = $new_category; 
  915.             $this->subscribe2_options['exclude'] = implode(', ', $excluded_cats); 
  916.             update_option('subscribe2_options', $this->subscribe2_options); 
  917.         } 
  918.     } // end new_category()

  920.     /** 
  921.     Automatically delete subscriptions to a category when it is deleted 
  922.     */ 
  923.     function delete_category($deleted_category='') { 
  924.         global $wpdb; 
  925.  
  926.         if ( $this->s2_mu ) { 
  927.             $sql = $wpdb->prepare("SELECT DISTINCT a.user_id FROM $wpdb->usermeta AS a INNER JOIN $wpdb->usermeta AS b WHERE a.user_id = b.user_id AND a.meta_key=%s AND b.meta_key=%s", $this->get_usermeta_keyname('s2_cat') . $deleted_category, $this->get_usermeta_keyname('s2_subscribed')); 
  928.         } else { 
  929.             $sql = $wpdb->prepare("SELECT DISTINCT user_id FROM $wpdb->usermeta WHERE meta_key=%s", $this->get_usermeta_keyname('s2_cat') . $deleted_category); 
  930.         } 
  931.         $user_IDs = $wpdb->get_col($sql); 
  932.         if ( '' == $user_IDs ) { return; } 
  933.  
  934.         foreach ( $user_IDs as $user_ID ) { 
  935.             $old_cats = explode(', ', get_user_meta($user_ID, $this->get_usermeta_keyname('s2_subscribed'), true)); 
  936.             if ( !is_array($old_cats) ) { 
  937.                 $old_cats = array($old_cats); 
  938.             } 
  939.             // add subscription to these cat IDs 
  940.             delete_user_meta($user_ID, $this->get_usermeta_keyname('s2_cat') . $deleted_category); 
  941.             $remain = array_diff($old_cats, (array)$deleted_category); 
  942.             update_user_meta($user_ID, $this->get_usermeta_keyname('s2_subscribed'), implode(', ', $remain)); 
  943.         } 
  944.     } // end delete_category()
