  129.               if( strtolower( get_class( $gateway ) ) == $current_section ) { 
  130.                   $current_gateway = $gateway->id; 
  131.                   break; 
  132.               }

  140.            */ 
  141.           $cc_gateways = array( 
  142.               'CardSave' // WooCommerce CardSave Redirect Gateway (http://www.cardsave.net/) 
  143.  ); 
  144.  
  145.           if( in_array( $current_gateway, $cc_gateways ) ) { 
  146.               $html = $this->manage_form_fields_for_cc_gateways( $current_gateway );

  153.  
  154.           wc_enqueue_js( "$( '.form-table:last' ).after( '" . $html . "' );" ); 
  155.       }

  385.

  387.           //Get the current gateway 
  388.           if ( isset( $current_gateway ) && isset( $available_gateways[ $current_gateway ] ) ) { 
  389.               $current_gateway = $available_gateways[ $current_gateway ]; 
  390.           } elseif ( isset( $available_gateways[ get_option( 'woocommerce_default_gateway' ) ] ) ) { 
  391.               $current_gateway = $available_gateways[ get_option( 'woocommerce_default_gateway' ) ]; 
  392.           } else { 
  393.               $current_gateway = current( $available_gateways ); 
  394.           } 
  395.       } 
  396.  
  397.       $this->current_gateway = $current_gateway; //Note: this is an object 
  398.       $extra_charge_max_cart_value = get_option( $this->get_option_id( $current_gateway->id, 'max_cart_value' ) ); 
  399.  
  400.       //Add charges to cart totals 
  401.       if( !empty( $current_gateway ) && ( empty( $extra_charge_max_cart_value ) || $extra_charge_max_cart_value >= $subtotal ) ) { 
  402.  
  403.           /** 
  404.            * These payment gateways use credit/debit cards. 
  405.            * We handle the fee differetly based on the type of card the user is going to use. 
  406.            * 
  407.            * @see http://www.cardsave.net/ 
  408.            */ 
  409.           $cc_gateways = array( 
  410.               'CardSave' // WooCommerce CardSave Redirect Gateway (http://www.cardsave.net/) 
  411.  ); 
  412.  
  413.           if( in_array( $current_gateway->id, $cc_gateways ) ) { 
  414.  
  415.               if( isset( $_REQUEST['post_data'] ) ) { 
  416.                   parse_str( $_REQUEST['post_data'] ); 
  417.               } 
  418.  
  419.               $card_type = isset( $chosen_method_card_type ) ? $chosen_method_card_type : 'cc'; 
  420.  
  421.               $extra_charge_amount = get_option( $this->get_option_id( $current_gateway->id, 'amount_' .  $card_type ) ); 
  422.               $extra_charge_type = get_option( $this->get_option_id( $current_gateway->id, 'type_' .  $card_type ) ); 
  423.               $calc_taxes = get_option( $this->get_option_id( $current_gateway->id, 'calc_taxes_' .  $card_type ) ); 
  424.           } else { 
  425.               $extra_charge_amount = get_option( $this->get_option_id( $current_gateway->id, 'amount' ) ); 
  426.               $extra_charge_type = get_option( $this->get_option_id( $current_gateway->id, 'type' ) ); 
  427.               $calc_taxes = get_option( $this->get_option_id( $current_gateway->id, 'calc_taxes' ) ); 
  428.           } 
  429.  
  430.           $decimal_sep = wp_specialchars_decode( stripslashes( get_option( 'woocommerce_price_decimal_sep' ) ), ENT_QUOTES ); 
  431.           $thousands_sep = wp_specialchars_decode( stripslashes( get_option( 'woocommerce_price_thousand_sep' ) ), ENT_QUOTES ); 
  432.  
  433.           if( $extra_charge_type == 'percentage' ) { 
  434.               $extra_charge_amount = number_format( $subtotal * $extra_charge_amount / 100 , 2, $decimal_sep, $thousands_sep ); 
  435.           } 
  436.  
  437.           $taxable = false; 
  438.           $taxes = 0; 
  439.           if( $calc_taxes != 'no-tax' ) { 
  440.               $taxable = true; 
  441.               $tax = new WC_Tax(); 
  442.               $base_rate = $tax->get_shop_base_rate(); 
  443.               $taxrates = array_shift( $base_rate ); 
  444.               $taxrate = floatval( $taxrates['rate'] ) / 100; 
  445.               if( $calc_taxes == 'tax-incl' ) { 
  446.                   $taxes = $extra_charge_amount - ( $extra_charge_amount / ( 1 + $taxrate ) ); 
  447.                   $extra_charge_amount -= $taxes; 
  448.               } else { 
  449.                   $taxes = $extra_charge_amount * $taxrate; 
  450.               } 
  451.           } 
  452.  
  453.           $extra_charge_amount = apply_filters( 'woocommerce_wc_pf_' . $current_gateway->id . '_amount' , $extra_charge_amount , $subtotal , $current_gateway ); 
  454.           $do_apply = $extra_charge_amount != 0; 
  455.           $do_apply = apply_filters( 'woocommerce_wc_pf_apply' , $do_apply , $extra_charge_amount , $subtotal , $current_gateway ); 
  456.           $do_apply = apply_filters( 'woocommerce_wc_pf_apply_for_' . $current_gateway->id , $do_apply , $extra_charge_amount , $subtotal , $current_gateway ); 
  457.  
  458.           if ( $do_apply ) { 
  459.               $extra_charge_amount = number_format( $extra_charge_amount, 2, $decimal_sep, $thousands_sep ); 
  460.               $cart->add_fee( __( 'Payment method fee', 'wc_pf' ), $extra_charge_amount, $taxable ); 
  461.           } 
  462.  
  463.           $this->current_extra_charge_amount = $extra_charge_amount; 
  464.           $this->current_extra_charge_type = $extra_charge_type; 
  465.       }
