  61. function acf_sync_import_json_field_groups($directory) { 
  62.     
  63.    $dir = new DirectoryIterator( $directory ); 
  64.    
  65.   foreach( $dir as $file ) { 
  66.        
  67.       if ( !$file->isDot() && $file->getExtension() == 'json' ) { 
  68.   
  69.           $json = json_decode( file_get_contents( $file->getPathname() ), true ); 
  70.   
  71.           // What follows is basically a copy of import() in ACF admin/settings-export.php 
  72.   
  73.           // if importing an auto-json, wrap field group in array 
  74.           if( isset($json['key']) ) { 
  75.                
  76.               $json = array( $json ); 
  77.                
  78.           } 
  79.            
  80.           // vars 
  81.           $added = array(); 
  82.           $ignored = array(); 
  83.           $ref = array(); 
  84.           $order = array(); 
  85.            
  86.           foreach( $json as $field_group ) { 
  87.                
  88.               // remove fields 
  89.               $fields = acf_extract_var($field_group, 'fields'); 
  90.             
  91.                
  92.               // format fields 
  93.               $fields = acf_prepare_fields_for_import( $fields ); 
  94.   
  95.                
  96.               // save field group 
  97.               $field_group = acf_update_field_group( $field_group ); 
  98.                
  99.                
  100.               // add to ref 
  101.               $ref[ $field_group['key'] ] = $field_group['ID']; 
  102.                
  103.                
  104.               // add to order 
  105.               $order[ $field_group['ID'] ] = 0; 
  106.                
  107.                
  108.               // add fields 
  109.               foreach( $fields as $field ) { 
  110.                    
  111.                   // add parent 
  112.                   if( empty($field['parent']) ) { 
  113.                        
  114.                       $field['parent'] = $field_group['ID']; 
  115.                        
  116.                   } elseif( isset($ref[ $field['parent'] ]) ) { 
  117.                        
  118.                       $field['parent'] = $ref[ $field['parent'] ]; 
  119.                            
  120.                   } 
  121.                    
  122.                   // add field menu_order 
  123.                   if( !isset($order[ $field['parent'] ]) ) { 
  124.                        
  125.                       $order[ $field['parent'] ] = 0; 
  126.                        
  127.                   } 
  128.                    
  129.                   $field['menu_order'] = $order[ $field['parent'] ]; 
  130.                   $order[ $field['parent'] ]++; 
  131.                    
  132.                   // save field 
  133.                   $field = acf_update_field( $field ); 
  134.                    
  135.               } 
  136.                
  137.           } 
  138.       } 
  139.   } 
  140.   
  141. }

  143. function acf_sync_get_fields_list() { 
  144.   $json_dirs = acf_get_setting('load_json'); 
  145.  
  146.   $json_files = array(); 
  147.  
  148.   foreach ( $json_dirs as $dir ) { 
  149.       if (file_exists($dir)) { 
  150.           $files = new DirectoryIterator( $dir ); 
  151.           foreach( $files as $file ) { 
  152.               if ( !$file->isDot() && $file->getExtension() == 'json' ) { 
  153.                   $path = $file->getPathname(); 
  154.                   $path_uri = strstr($path, '/wp-content'); 
  155.  
  156.                   $contents = file_get_contents($path); 
  157.                   $json = json_decode($contents, true); 
  158.                   $title = $json['title']; 
  159.  
  160.                   $json_files[] = array('title'=>$title, 'path'=>$path_uri); 
  161.               } 
  162.           } 
  163.       } 
  164.   } 
  165.   return $json_files; 
  166. }
