  311.   /* Get new version message, also include error connect 
  312.   /**-----------------------------------------------------------------------------------*/ 
  313.   public function get_version_message() { 
  314.       $version_message = ''; 
  315.  
  316.       //Getting version number 
  317.       $version_transient = get_transient( $this->version_transient ); 
  318.       if ( false !== $version_transient ) { 
  319.           $transient_timeout = '_transient_timeout_' . $this->version_transient; 
  320.           $timeout = get_option( $transient_timeout, false ); 
  321.           if ( false === $timeout ) { 
  322.               $version_message = __( 'You should check now to see if have any new version is available', 'woocommerce-email-inquiry-cart-options' ); 
  323.           } elseif ( 'cannot_connect_api' == $version_transient ) { 
  324.               $version_message = sprintf( __( 'Connection Failure! Please try again. If this issue persists please create a support request on the plugin <a href="%s" target="_blank">a3rev support forum</a>.', 'woocommerce-email-inquiry-cart-options' ), $this->support_url ); 
  325.           } else { 
  326.               $version_info = explode( '||', $version_transient ); 
  327.               if ( FALSE !== stristr( $version_transient, '||' ) 
  328.                   && is_array( $version_info ) 
  329.                   && isset( $version_info[1] ) && $version_info[1] == 'valid' 
  330.                   && version_compare( get_option('a3rev_wc_email_inquiry_version') , $version_info[0], '<' ) ) { 
  331.  
  332.                       $version_message = sprintf( __( 'There is a new version <span class="a3rev-ui-new-plugin-version">%s</span> available, <a href="%s" target="_blank">update now</a> or download direct from <a href="%s" target="_blank">My Account</a> on a3rev.com', 'woocommerce-email-inquiry-cart-options' ),  
  333.                           $version_info[0],  
  334.                           wp_nonce_url( self_admin_url( 'update.php?action=upgrade-plugin&plugin=' . WC_EMAIL_INQUIRY_NAME ), 'upgrade-plugin_' . WC_EMAIL_INQUIRY_NAME ),  
  335.                           'https://a3rev.com/my-account/downloads/' 
  336.  ); 
  337.               } 
  338.           } 
  339.  
  340.       } else { 
  341.           $version_message = __( 'You should check now to see if have any new version is available', 'woocommerce-email-inquiry-cart-options' ); 
  342.       } 
  343.  
  344.       return $version_message; 
  345.   }
