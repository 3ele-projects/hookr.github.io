  149.       function check_jambopay_response() { 
  150.           global $woocommerce; 
  151.           //************************ CHECK IF VALUES HAVE BEEN SET ***************** 
  152.       if (isset($_POST['JP_PASSWORD'])&&isset($_POST['JP_MERCHANT_ORDERID'])) { 
  153.  
  154.           $JP_TRANID = $_POST['JP_TRANID']; 
  155.           $JP_MERCHANT_ORDERID = $_POST['JP_MERCHANT_ORDERID']; 
  156.           $JP_ITEM_NAME = $_POST['JP_ITEM_NAME']; 
  157.           $JP_AMOUNT = $_POST['JP_AMOUNT']; 
  158.           $JP_CURRENCY = $_POST['JP_CURRENCY']; 
  159.           $JP_TIMESTAMP = $_POST['JP_TIMESTAMP']; 
  160.           $JP_PASSWORD = $_POST['JP_PASSWORD']; 
  161.           $JP_CHANNEL = $_POST['JP_CHANNEL']; 
  162.  
  163. //$sharedkey, IS ONLY SHARED BETWEEN THE MERCHANT AND JAMBOPAY. THE KEY SHOULD BE SECRET ******************** 
  164.  
  165.                       $order_id=$_POST['JP_MERCHANT_ORDERID']; 
  166.                       $order = new WC_Order((int)$order_id); 
  167.  
  168. //Make sure you get the key from JamboPay Support team 
  169.           $sharedkey = $this->working_key ; 
  170.  
  171.           $str = $JP_MERCHANT_ORDERID . $JP_AMOUNT . $JP_CURRENCY . $sharedkey . $JP_TIMESTAMP; 
  172.  
  173.  
  174. //**************** VERIFY ************************* 
  175.           if (md5(utf8_encode($str)) == $JP_PASSWORD) { 
  176.  
  177.               //VALID 
  178.               //if valid, mark order as paid 
  179.                       $order->add_order_note( __( 'Jambopay txID:'. $JP_TRANID.' Channel:'.$JP_CHANNEL, 'woocommerce' ) ); 
  180.                       $order->payment_complete(); 
  181.                       $woocommerce->cart->empty_cart(); 
  182.                        
  183.  
  184.           } else{ 
  185.            
  186.                   $order->add_order_note( __( 'Jambopay payment failed txID:'.$JP_TRANID, 'woocommerce' ) ); 
  187.                   $order->update_status( 'failed', sprintf( __( 'Payment %s via IPN.', 'woocommerce' ), strtolower( $posted['payment_status'] ) ) ); 
  188.       } 
  189.             
  190.                   wp_safe_redirect(add_query_arg('key', $order->order_key, add_query_arg('order', $order_id, get_permalink(woocommerce_get_page_id('thanks'))))); 
  191.                     
  192.           } 
  193.               }

  257.       // get all pages 
  258.       function get_pages($title = false, $indent = true) { 
  259.           $wp_pages = get_pages('sort_column=menu_order'); 
  260.           $page_list = array(); 
  261.           if ($title) $page_list[] = $title; 
  262.           foreach ($wp_pages as $page) { 
  263.               $prefix = ''; 
  264.               // show indented child pages? 
  265.               if ($indent) { 
  266.                   $has_parent = $page->post_parent; 
  267.                   while($has_parent) { 
  268.                       $prefix .=  ' - '; 
  269.                       $next_page = get_page($has_parent); 
  270.                       $has_parent = $next_page->post_parent; 
  271.                   } 
  272.               } 
  273.               // add to page list array array 
  274.               $page_list[$page->ID] = $prefix . $page->post_title; 
  275.           } 
  276.           return $page_list; 
  277.       }
