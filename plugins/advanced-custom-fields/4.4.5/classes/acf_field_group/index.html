  842.     /** 
  843.     *  save_post 
  844.     * 
  845.     *  @description: Saves the field / location / option data for a field group 
  846.     *  @since 1.0.0 
  847.     *  @created: 23/06/12 
  848.     */ 
  849.      
  850.     function save_post($post_id) 
  851.     { 
  852.         // do not save if this is an auto save routine 
  853.         if( defined('DOING_AUTOSAVE') && DOING_AUTOSAVE ) 
  854.         { 
  855.             return $post_id; 
  856.         } 
  857.          
  858.          
  859.         // verify nonce 
  860.         if( !isset($_POST['acf_nonce']) || !wp_verify_nonce($_POST['acf_nonce'], 'field_group') ) 
  861.         { 
  862.             return $post_id; 
  863.         } 
  864.          
  865.          
  866.         // only save once! WordPress save's a revision as well. 
  867.         if( wp_is_post_revision($post_id) ) 
  868.         { 
  869.             return $post_id; 
  870.         } 
  871.          
  872.          
  873.         /** 
  874.         *  save fields 
  875.         */ 
  876.          
  877.         // vars 
  878.         $dont_delete = array(); 
  879.          
  880.         if( isset($_POST['fields']) && is_array($_POST['fields']) ) 
  881.         { 
  882.             $i = -1; 
  883.  
  884.  
  885.             // remove clone field 
  886.             unset( $_POST['fields']['field_clone'] ); 
  887.              
  888.              
  889.  
  890.             // loop through and save fields 
  891.             foreach( $_POST['fields'] as $key => $field ) 
  892.             { 
  893.                 $i++; 
  894.                  
  895.                  
  896.                 // order + key 
  897.                 $field['order_no'] = $i; 
  898.                 $field['key'] = $key; 
  899.                  
  900.                  
  901.                 // save 
  902.                 do_action('acf/update_field', $field, $post_id ); 
  903.                  
  904.                  
  905.                 // add to dont delete array 
  906.                 $dont_delete[] = $field['key']; 
  907.             } 
  908.         } 
  909.         unset( $_POST['fields'] ); 
  910.          
  911.          
  912.         // delete all other field 
  913.         $keys = get_post_custom_keys($post_id); 
  914.         foreach( $keys as $key ) 
  915.         { 
  916.             if( strpos($key, 'field_') !== false && !in_array($key, $dont_delete) ) 
  917.             { 
  918.                 // this is a field, and it wasn't found in the dont_delete array 
  919.                 do_action('acf/delete_field', $post_id, $key); 
  920.             } 
  921.         } 
  922.          
  923.          
  924.         /** 
  925.         *  save location rules 
  926.         */ 
  927.          
  928.         if( isset($_POST['location']) && is_array($_POST['location']) ) 
  929.         { 
  930.             delete_post_meta( $post_id, 'rule' ); 
  931.              
  932.              
  933.             // clean array keys 
  934.             $_POST['location'] = array_values( $_POST['location'] ); 
  935.             foreach( $_POST['location'] as $group_id => $group ) 
  936.             { 
  937.                 if( is_array($group) ) 
  938.                 { 
  939.                     // clean array keys 
  940.                     $group = array_values( $group ); 
  941.                     foreach( $group as $rule_id => $rule ) 
  942.                     { 
  943.                         $rule['order_no'] = $rule_id; 
  944.                         $rule['group_no'] = $group_id; 
  945.                          
  946.  
  947.                         add_post_meta( $post_id, 'rule', $rule ); 
  948.                     } 
  949.                 } 
  950.             } 
  951.              
  952.             unset( $_POST['location'] ); 
  953.         } 
  954.          
  955.          
  956.         /** 
  957.         *  save options 
  958.         */

  960.         if( isset($_POST['options']) && is_array($_POST['options']) ) 
  961.         { 
  962.             update_post_meta($post_id, 'position', $_POST['options']['position']); 
  963.             update_post_meta($post_id, 'layout', $_POST['options']['layout']); 
  964.             update_post_meta($post_id, 'hide_on_screen', $_POST['options']['hide_on_screen']); 
  965.         } 
  966.  
  967.          
  968.         unset( $_POST['options'] ); 
  969.      
  970.          
  971.     }
