  20.    *

  33.

  36.  
  37.   /**

  39.    * 
  40.    * @since bbPress (r3389)

  48.   /** 
  49.    * Displays the output, the replies list 
  50.    * 
  51.    * @since bbPress (r2653) 
  52.    * 
  53.    * @param mixed $args 
  54.    * @param array $instance 
  55.    * @uses apply_filters() Calls 'bbp_reply_widget_title' with the title 
  56.    * @uses bbp_get_reply_author_link() To get the reply author link 
  57.    * @uses bbp_get_reply_id() To get the reply id 
  58.    * @uses bbp_get_reply_url() To get the reply url 
  59.    * @uses bbp_get_reply_excerpt() To get the reply excerpt 
  60.    * @uses bbp_get_reply_topic_title() To get the reply topic title 
  61.    * @uses get_the_date() To get the date of the reply 
  62.    * @uses get_the_time() To get the time of the reply 
  63.    */ 
  64.   public function widget( $args, $instance ) { 
  65.  
  66.       // Get widget settings 
  67.       $settings = $this->parse_settings( $instance ); 
  68.  
  69.       // Typical WordPress filter 
  70.       $settings['title'] = apply_filters( 'widget_title', $settings['title'], $instance, $this->id_base ); 
  71.  
  72.       // bbPress filter 
  73.       $settings['title'] = apply_filters( 'bbp_replies_widget_title', $settings['title'], $instance, $this->id_base ); 
  74.  
  75.       // Note: private and hidden forums will be excluded via the 
  76.       // bbp_pre_get_posts_normalize_forum_visibility action and function. 
  77.       $widget_query = new WP_Query( array( 
  78.           'post_type' => bbp_get_reply_post_type(),  
  79.           'post_status' => array( bbp_get_public_status_id(), bbp_get_closed_status_id() ),  
  80.           'posts_per_page' => (int) $settings['max_shown'],  
  81.           'ignore_sticky_posts' => true,  
  82.           'no_found_rows' => true,  
  83.  ) ); 
  84.  
  85.       // Bail if no replies 
  86.       if ( ! $widget_query->have_posts() ) { 
  87.           return; 
  88.       } 
  89.  
  90.       echo $args['before_widget']; 
  91.  
  92.       if ( !empty( $settings['title'] ) ) { 
  93.           echo $args['before_title'] . $settings['title'] . $args['after_title']; 
  94.       } ?> 
  95.  
  96.       <ul> 
  97.  
  98.           <?php while ( $widget_query->have_posts() ) : $widget_query->the_post(); ?> 
  99.  
  100.               <li> 
  101.  
  102.                   <?php 
  103.  
  104.                   // Verify the reply ID 
  105.                   $reply_id = bbp_get_reply_id( $widget_query->post->ID ); 
  106.                   $reply_link = '<a class="bbp-reply-topic-title" href="' . esc_url( bbp_get_reply_url( $reply_id ) ) . '" title="' . esc_attr( bbp_get_reply_excerpt( $reply_id, 50 ) ) . '">' . bbp_get_reply_topic_title( $reply_id ) . '</a>'; 
  107.  
  108.                   // Only query user if showing them 
  109.                   if ( ! empty( $settings['show_user'] ) ) : 
  110.                       $author_link = bbp_get_reply_author_link( array( 'post_id' => $reply_id, 'type' => 'both', 'size' => 14 ) ); 
  111.                   else : 
  112.                       $author_link = false; 
  113.                   endif; 
  114.  
  115.                   // Reply author, link, and timestamp 
  116.                   if ( ! empty( $settings['show_date'] ) && !empty( $author_link ) ) : 
  117.  
  118.                       // translators: 1: reply author, 2: reply link, 3: reply timestamp 
  119.                       $date_format = get_option( 'date_format' ); 
  120.                       $time_format = get_option( 'time_format' ); 
  121.                       $date= date( "{$date_format}", get_the_time( 'U' ) ); 
  122.                       $time=date( "{$time_format}", get_the_time( 'U' ) ); 
  123.                       $active_time = sprintf( _x( '%1$s at %2$s', 'date at time', 'bbp-last-post' ), $date, $time );   
  124.                       printf( _x( '%1$s on %2$s %3$s', 'widgets', 'bbp-last-post' ), $author_link, $reply_link, '<div>' .  $active_time  . '</div>' ); 
  125.  
  126.                   // Reply link and timestamp 
  127.                   elseif ( ! empty( $settings['show_date'] ) ) : 
  128.  
  129.                       // translators: 1: reply link, 2: reply timestamp 
  130.                       $date_format = get_option( 'date_format' ); 
  131.                       $time_format = get_option( 'time_format' ); 
  132.                       $date= date( "{$date_format}", get_the_time( 'U' ) ); 
  133.                       $time=date( "{$time_format}", get_the_time( 'U' ) ); 
  134.                       $active_time = sprintf( _x( '%1$s at %2$s', 'date at time', 'bbp-last-post' ), $date, $time );   
  135.                       printf( _x( '%1$s %2$s', 'widgets', 'bbp-last-post' ), $reply_link, '<div>' . $active_time  . '</div>' ); 
  136.  
  137.                   // Reply author and title 
  138.                   elseif ( !empty( $author_link ) ) : 
  139.  
  140.                       // translators: 1: reply author, 2: reply link 
  141.                       printf( _x( '%1$s on %2$s', 'widgets', 'bbp-last-post' ), $author_link, $reply_link ); 
  142.  
  143.                   // Only the reply title 
  144.                   else : 
  145.  
  146.                       // translators: 1: reply link 
  147.                       printf( _x( '%1$s', 'widgets', 'bbp-last-post' ), $reply_link ); 
  148.  
  149.                   endif; 
  150.  
  151.                   ?> 
  152.  
  153.               </li> 
  154.  
  155.           <?php endwhile; ?> 
  156.  
  157.       </ul> 
  158.  
  159.       <?php echo $args['after_widget']; 
  160.  
  161.       // Reset the $post global 
  162.       wp_reset_postdata(); 
  163.   }

  165.   /** 
  166.    * Update the reply widget options 
  167.    *

  175.       $instance['title'] = strip_tags( $new_instance['title'] ); 
  176.       $instance['show_date'] = (bool) $new_instance['show_date'];

  178.       $instance['max_shown'] = (int) $new_instance['max_shown']; 
  179.  
  180.       return $instance; 
  181.   }
