  120. function w3tc_class_autoload( $class ) { 
  121.   $base = null; 
  122.  
  123.   // some php pass classes with slash 
  124.   if ( substr( $class, 0, 1 ) == "\\" ) 
  125.       $class = substr( $class, 1 ); 
  126.  
  127.   if ( substr( $class, 0, 5 ) == 'HTTP_' || substr( $class, 0, 7 ) == 'Minify_' ) { 
  128.       $base = W3TC_LIB_DIR . DIRECTORY_SEPARATOR . 'Minify' . DIRECTORY_SEPARATOR; 
  129.   } elseif ( substr( $class, 0, 8 ) == 'Minify0_' ) { 
  130.       $base = W3TC_LIB_DIR . DIRECTORY_SEPARATOR . 'Minify' . DIRECTORY_SEPARATOR; 
  131.       $class = substr( $class, 8 ); 
  132.   } elseif ( substr( $class, 0, 7 ) == 'Google_' && 
  133.       ( !defined( 'W3TC_GOOGLE_LIBRARY' ) || W3TC_GOOGLE_LIBRARY ) ) { 
  134.       // Google library 
  135.       $classPath = explode( '_', $class ); 
  136.       if ( count( $classPath ) > 3 ) { 
  137.           // Maximum class file path depth in this project is 3. 
  138.           $classPath = array_slice( $classPath, 0, 3 ); 
  139.       } 
  140.  
  141.       $filePath = W3TC_LIB_DIR . DIRECTORY_SEPARATOR . 
  142.           implode( '/', $classPath ) . '.php'; 
  143.       require $filePath; 
  144.       return; 
  145.   } 
  146.  
  147.   if ( !is_null( $base ) ) { 
  148.       $file = $base . strtr( $class, "\\_",  
  149.           DIRECTORY_SEPARATOR . DIRECTORY_SEPARATOR ) . '.php'; 
  150.       if ( is_readable( $file ) ) 
  151.           require_once $file; 
  152.   } else if ( substr( $class, 0, 5 ) == 'W3TC\\' ) { 
  153.           $filename = W3TC_DIR . DIRECTORY_SEPARATOR . substr( $class, 5 ) . '.php'; 
  154.  
  155.           if ( defined( 'WP_DEBUG' ) && WP_DEBUG ) { 
  156.               if ( !file_exists( $filename ) ) { 
  157.                   debug_print_backtrace(); 
  158.               } 
  159.           } 
  160.  
  161.           require $filename; 
  162.       } 
  163. }
