  263.   private function test_rewrite( $url ) {

  265.       $result = get_site_transient( $key ); 
  266.  
  267.       if ( $result != 'ok' ) { 
  268.           $response = Util_Http::get( $url ); 
  269.  
  270.           $is_ok = ( !is_wp_error( $response ) && 
  271.               $response['response']['code'] == 200 && 
  272.               trim( $response['body'] ) == 'Minify OK' ); 
  273.  
  274.           if ( $is_ok ) 
  275.               $result = 'ok'; 
  276.           else 
  277.               $result = is_wp_error( $response ) ? 
  278.                   $response->get_error_message() : 
  279.                   implode( ' ', $response['response'] ); 
  280.  
  281.           set_site_transient( $key, $result, 30 ); 
  282.       } 
  283.  
  284.       return $result;

  362.   /** 
  363.    * Generates rules 
  364.    * 
  365.    * @param Config  $config 
  366.    * @return string 
  367.    */ 
  368.   function rules_core_generate_apache( $config ) { 
  369.       $cache_dir = Util_Rule::filename_to_uri( W3TC_CACHE_MINIFY_DIR ); 
  370.       $site_uri = rtrim( network_site_url( '', 'relative' ), '/' ) . '/'; 
  371.  
  372.       $engine = $config->get_string( 'minify.engine' ); 
  373.       $browsercache = $config->get_boolean( 'browsercache.enabled' ); 
  374.       $compression = ( $browsercache && 
  375.           $config->get_boolean( 'browsercache.cssjs.compression' ) && 
  376.           !defined( 'W3TC_PAGECACHE_OUTPUT_COMPRESSION_OFF' ) ); 
  377.  
  378.       $rules = ''; 
  379.       $rules .= W3TC_MARKER_BEGIN_MINIFY_CORE . "\n"; 
  380.       $rules .= "<IfModule mod_rewrite.c>\n"; 
  381.       $rules .= "    RewriteEngine On\n"; 
  382.       $rules .= "    RewriteBase " . $cache_dir . "/\n"; 
  383.  
  384.       if ( $engine == 'file' ) { 
  385.           if ( $compression ) { 
  386.               $rules .= "    RewriteCond %{HTTP:Accept-Encoding} gzip\n"; 
  387.               $rules .= "    RewriteRule .* - [E=APPEND_EXT:.gzip]\n"; 
  388.           } 
  389.  
  390.           $rules .= "    RewriteCond %{REQUEST_FILENAME}%{ENV:APPEND_EXT} -" . ( $config->get_boolean( 'minify.file.nfs' ) ? 'F' : 'f' ) . "\n"; 
  391.           $rules .= "    RewriteRule (.*) $1%{ENV:APPEND_EXT} [L]\n"; 
  392.       } 
  393.       $rules .= "    RewriteRule ^(.+\\.(css|js))$ ${site_uri}index.php [L]\n"; 
  394.  
  395.       $rules .= "</IfModule>\n"; 
  396.       $rules .= W3TC_MARKER_END_MINIFY_CORE . "\n"; 
  397.  
  398.       return $rules; 
  399.   }

  407.   function rules_core_generate_nginx( $config ) { 
  408.       $cache_dir = Util_Rule::filename_to_uri( W3TC_CACHE_MINIFY_DIR ); 
  409.       $first_regex_var = '$1'; 
  410.  
  411.       // for subdir - need to count subdir in url 
  412.       if ( Util_Environment::is_wpmu() && !Util_Environment::is_wpmu_subdomain() ) { 
  413.           // take into accont case when whole subdir wpmu is installed in subdir 
  414.           $home_uri = network_home_url( '', 'relative' ); 
  415.           if ( substr( $cache_dir, 0, strlen( $home_uri ) ) == $home_uri ) 
  416.               $cache_dir = $home_uri . '([a-z0-9]+/)?' . 
  417.                   substr( $cache_dir, strlen( $home_uri ) ); 
  418.           else 
  419.               $cache_dir = '(/[a-z0-9]+)?' . $cache_dir; 
  420.  
  421.           $first_regex_var = '$2'; 
  422.       } 
  423.  
  424.       $minify_uri = rtrim( network_site_url( '', 'relative' ), '/' ) . '/'; 
  425.  
  426.       $engine = $config->get_string( 'minify.engine' ); 
  427.       $browsercache = $config->get_boolean( 'browsercache.enabled' ); 
  428.       $compression = ( $browsercache && 
  429.           $config->get_boolean( 'browsercache.cssjs.compression' ) && 
  430.           !defined( 'W3TC_PAGECACHE_OUTPUT_COMPRESSION_OFF' ) ); 
  431.  
  432.       $rules = ''; 
  433.       $rules .= W3TC_MARKER_BEGIN_MINIFY_CORE . "\n"; 
  434.  
  435.       if ( $engine == 'file' ) { 
  436.           $rules .= "set \$w3tc_enc \"\";\n"; 
  437.  
  438.           if ( $compression ) { 
  439.               $rules .= "if (\$http_accept_encoding ~ gzip) {\n"; 
  440.               $rules .= "    set \$w3tc_enc .gzip;\n"; 
  441.               $rules .= "}\n"; 
  442.           } 
  443.  
  444.           $rules .= "if (-f \$request_filename\$w3tc_enc) {\n"; 
  445.           $rules .= "    rewrite (.*) $1\$w3tc_enc break;\n"; 
  446.           $rules .= "}\n"; 
  447.       } 
  448.       $rules .= "rewrite ^$cache_dir/ ${minify_uri}index.php last;\n"; 
  449.       $rules .= W3TC_MARKER_END_MINIFY_CORE . "\n"; 
  450.  
  451.       return $rules; 
  452.   }

  530.       $compression = ( $browsercache && $config->get_boolean( 'browsercache.cssjs.compression' ) ); 
  531.       $expires = ( $browsercache && $config->get_boolean( 'browsercache.cssjs.expires' ) );

  538.       $rules = '';

  596.  
  597.                   if ( $expires ) { 
  598.                       $rules .= "    Header append Cache-Control \"public\"\n";

  604.               case 'cache_validation':

  607.                   break; 
  608.  
  609.               case 'cache_noproxy': 
  610.                   $rules .= "    Header set Pragma \"public\"\n"; 
  611.                   $rules .= "    Header set Cache-Control \"private, must-revalidate\"\n"; 
  612.                   break; 
  613.  
  614.               case 'cache_maxage': 
  615.                   $rules .= "    Header set Pragma \"public\"\n"; 
  616.  
  617.                   if ( $expires ) { 
  618.                       $rules .= "    Header append Cache-Control \"public, must-revalidate, proxy-revalidate\"\n"; 
  619.                   } else {

  629.           } 
  630.  
  631.           $rules .= "</IfModule>\n"; 
  632.       } 
  633.  
  634.       $rules .= W3TC_MARKER_END_MINIFY_CACHE . "\n";
