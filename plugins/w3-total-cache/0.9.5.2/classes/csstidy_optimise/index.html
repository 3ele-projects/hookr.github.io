  608.      * @see merge_bg() 
  609.      * @todo full CSS 3 compliance 
  610.      */ 
  611.     function dissolve_short_bg($str_value) 
  612.     { 
  613.         $background_prop_default =& $GLOBALS['csstidy']['background_prop_default']; 
  614.         $repeat = array('repeat', 'repeat-x', 'repeat-y', 'no-repeat', 'space'); 
  615.         $attachment = array('scroll', 'fixed', 'local'); 
  616.         $clip = array('border', 'padding'); 
  617.         $origin = array('border', 'padding', 'content'); 
  618.         $pos = array('top', 'center', 'bottom', 'left', 'right'); 
  619.         $important = ''; 
  620.         $return = array('background-image' => NULL, 'background-size' => NULL, 'background-repeat' => NULL, 'background-position' => NULL, 'background-attachment'=>NULL, 'background-clip' => NULL, 'background-origin' => NULL, 'background-color' => NULL); 
  621.  
  622.         if(csstidy::is_important($str_value)) 
  623.         { 
  624.             $important = ' !important'; 
  625.             $str_value = csstidy::gvw_important($str_value); 
  626.         } 
  627.  
  628.         $str_value = csstidy_optimise::explode_ws(', ', $str_value); 
  629.         for($i = 0; $i < count($str_value); $i++) 
  630.         { 
  631.             $have['clip'] = FALSE; $have['pos'] = FALSE; 
  632.             $have['color'] = FALSE; $have['bg'] = FALSE; 
  633.  
  634.             $str_value[$i] = csstidy_optimise::explode_ws(' ', trim($str_value[$i])); 
  635.  
  636.             for($j = 0; $j < count($str_value[$i]); $j++) 
  637.             { 
  638.                 if($have['bg'] === FALSE && (substr($str_value[$i][$j], 0, 4) == 'url(' || $str_value[$i][$j] === 'none')) 
  639.                 { 
  640.                     $return['background-image'] .= $str_value[$i][$j].', '; 
  641.                     $have['bg'] = TRUE; 
  642.                 } 
  643.                 elseif(in_array($str_value[$i][$j], $repeat, TRUE)) 
  644.                 { 
  645.                     $return['background-repeat'] .= $str_value[$i][$j].', '; 
  646.                 } 
  647.                 elseif(in_array($str_value[$i][$j], $attachment, TRUE)) 
  648.                 { 
  649.                     $return['background-attachment'] .= $str_value[$i][$j].', '; 
  650.                 } 
  651.                 elseif(in_array($str_value[$i][$j], $clip, TRUE) && !$have['clip']) 
  652.                 { 
  653.                     $return['background-clip'] .= $str_value[$i][$j].', '; 
  654.                     $have['clip'] = TRUE; 
  655.                 } 
  656.                 elseif(in_array($str_value[$i][$j], $origin, TRUE)) 
  657.                 { 
  658.                     $return['background-origin'] .= $str_value[$i][$j].', '; 
  659.                 } 
  660.                 elseif($str_value[$i][$j]{0} == '(') 
  661.                 { 
  662.                     $return['background-size'] .= substr($str_value[$i][$j], 1, -1).', '; 
  663.                 } 
  664.                 elseif(in_array($str_value[$i][$j], $pos, TRUE) || is_numeric($str_value[$i][$j]{0}) || $str_value[$i][$j]{0} === NULL) 
  665.                 { 
  666.                     $return['background-position'] .= $str_value[$i][$j]; 
  667.                     if(!$have['pos']) $return['background-position'] .= ' '; else $return['background-position'].= ', '; 
  668.                     $have['pos'] = TRUE; 
  669.                 } 
  670.                 elseif(!$have['color']) 
  671.                 { 
  672.                     $return['background-color'] .= $str_value[$i][$j].', '; 
  673.                     $have['color'] = TRUE; 
  674.                 } 
  675.             } 
  676.         } 
  677.  
  678.         foreach($background_prop_default as $bg_prop => $default_value) 
  679.         { 
  680.             if($return[$bg_prop] !== NULL) 
  681.             { 
  682.                 $return[$bg_prop] = substr($return[$bg_prop], 0, -1).$important; 
  683.             } 
  684.             else $return[$bg_prop] = $default_value.$important; 
  685.         } 
  686.         return $return; 
  687.     } 
  688.  
  689.     /** 
  690.      * Merges all background properties 
  691.      * @param array $input_css 
  692.      * @return array 
  693.      * @version 1.0 
  694.      * @see dissolve_short_bg() 
  695.      * @todo full CSS 3 compliance 
  696.      */ 
  697.     function merge_bg($input_css) 
  698.     { 
  699.         $background_prop_default =& $GLOBALS['csstidy']['background_prop_default']; 
  700.         // Max number of background images. CSS3 not yet fully implemented 
  701.         $number_of_values = @max(count(csstidy_optimise::explode_ws(', ', $input_css['background-image'])), count(csstidy_optimise::explode_ws(', ', $input_css['background-color'])), 1); 
  702.         // Array with background images to check if BG image exists 
  703.         $bg_img_array = @csstidy_optimise::explode_ws(', ', csstidy::gvw_important($input_css['background-image'])); 
  704.         $new_bg_value = ''; 
  705.         $important = ''; 
  706.  
  707.         for($i = 0; $i < $number_of_values; $i++) 
  708.         { 
  709.             foreach($background_prop_default as $bg_property => $default_value) 
  710.             { 
  711.                 // Skip if property does not exist 
  712.                 if(!isset($input_css[$bg_property])) 
  713.                 { 
  714.                     continue; 
  715.                 } 
  716.  
  717.                 $cur_value = $input_css[$bg_property]; 
  718.  
  719.                 // Skip some properties if there is no background image 
  720.                 if((!isset($bg_img_array[$i]) || $bg_img_array[$i] === 'none') 
  721.                     && ($bg_property === 'background-size' || $bg_property === 'background-position' 
  722.                     || $bg_property === 'background-attachment' || $bg_property === 'background-repeat')) 
  723.                 { 
  724.                     continue; 
  725.                 } 
  726.  
  727.                 // Remove !important 
  728.                 if(csstidy::is_important($cur_value)) 
  729.                 { 
  730.                     $important = ' !important'; 
  731.                     $cur_value = csstidy::gvw_important($cur_value); 
  732.                 } 
  733.  
  734.                 // Do not add default values 
  735.                 if($cur_value === $default_value) 
  736.                 { 
  737.                     continue; 
  738.                 } 
  739.  
  740.                 $temp = csstidy_optimise::explode_ws(', ', $cur_value); 
  741.  
  742.                 if(isset($temp[$i])) 
  743.                 { 
  744.                     if($bg_property == 'background-size') 
  745.                     { 
  746.                         $new_bg_value .= '('.$temp[$i].') '; 
  747.                     } 
  748.                     else 
  749.                     { 
  750.                         $new_bg_value .= $temp[$i].' '; 
  751.                     } 
  752.                 } 
  753.             } 
  754.  
  755.             $new_bg_value = trim($new_bg_value); 
  756.             if($i != $number_of_values-1) $new_bg_value .= ', '; 
  757.         } 
  758.  
  759.         // Delete all background-properties 
  760.         foreach($background_prop_default as $bg_property => $default_value) 
  761.         { 
  762.             unset($input_css[$bg_property]); 
  763.         } 
  764.  
  765.         // Add new background property 
  766.         if($new_bg_value !== '') $input_css['background'] = $new_bg_value.$important; 
  767.  
  768.         return $input_css; 
  769.     } 
  770. }
