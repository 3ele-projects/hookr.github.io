  825.     public function migrate_pod ( $params ) { 
  826.         /** 
  827.          * @var $wpdb WPDB 
  828.          */ 
  829.         global $wpdb; 
  830.  
  831.         if ( !isset( $params->pod ) ) 
  832.             return pods_error( __( 'Invalid Pod.', 'pods' ) ); 
  833.  
  834.         $pod = pods_sanitize( pods_clean_name( $params->pod ) ); 
  835.  
  836.         if ( !in_array( "{$wpdb->prefix}pod_tbl_{$pod}", $this->tables ) ) 
  837.             return pods_error( __( 'Table not found, items cannot be migrated', 'pods' ) ); 
  838.  
  839.         if ( !in_array( "{$wpdb->prefix}pods_{$pod}", $this->tables ) ) 
  840.             return pods_error( __( 'New table not found, items cannot be migrated', 'pods' ) ); 
  841.  
  842.         if ( !in_array( "{$wpdb->prefix}pod_types", $this->tables ) ) 
  843.             return pods_error( __( 'Pod Types table not found, items cannot be migrated', 'pods' ) ); 
  844.  
  845.         if ( !in_array( "{$wpdb->prefix}pod", $this->tables ) ) 
  846.             return pods_error( __( 'Pod table not found, items cannot be migrated', 'pods' ) ); 
  847.  
  848.         if ( true === $this->check_progress( __FUNCTION__, $pod ) ) 
  849.             return '1'; 
  850.  
  851.         $pod_data = $this->api->load_pod( array( 'name' => $pod ), false ); 
  852.  
  853.         if ( empty( $pod_data ) ) 
  854.             return pods_error( sprintf( __( 'Pod <strong>%s</strong> not found, items cannot be migrated', 'pods' ), $pod ) ); 
  855.  
  856.         $columns = array(); 
  857.         $old_columns = array(); 
  858.  
  859.         foreach ( $pod_data[ 'fields' ] as $field ) { 
  860.             if ( !in_array( $field[ 'name' ], array( 'created', 'modified', 'author' ) ) && !in_array( $field[ 'type' ], array( 'file', 'pick' ) ) ) { 
  861.                 $columns[] = pods_sanitize( $field[ 'name' ] ); 
  862.                 $old_columns[] = pods_var( '_pods_1x_field_name', $field[ 'options' ], $field[ 'name' ], null, false ); 
  863.             } 
  864.         } 
  865.  
  866.         $into = '`id`'; 
  867.         $select = '`t`.`id`'; 
  868.  
  869.         if ( !empty( $columns ) ) { 
  870.             $into .= ', `' . implode( '`, `', $columns ) . '`'; 
  871.             $select .= ', `t`.`' . implode( '`, `t`.`', $old_columns ) . '`'; 
  872.         } 
  873.  
  874.         // Copy content from the old table into the new 
  875.         $sql = " 
  876.             REPLACE INTO `@wp_pods_{$pod}` 
  877.                 ( {$into} ) 
  878.                 ( SELECT {$select} 
  879.                   FROM `@wp_pod_tbl_{$pod}` AS `t` ) 
  880.         "; 
  881.  
  882.         pods_query( $sql ); 
  883.  
  884.         // Copy index data from the old index table into the new individual table 
  885.         $sql = " 
  886.             UPDATE `@wp_pods_{$pod}` AS `t` 
  887.             LEFT JOIN `@wp_pod_types` AS `x` ON `x`.`name` = '{$pod}' 
  888.             LEFT JOIN `@wp_pod` AS `p` ON `p`.`datatype` = `x`.`id` AND `p`.`tbl_row_id` = `t`.`id` 
  889.             SET `t`.`created` = `p`.`created`, `t`.`modified` = `p`.`modified` 
  890.             WHERE `x`.`id` IS NOT NULL AND `p`.`id` IS NOT NULL 
  891.         "; 
  892.  
  893.         pods_query( $sql ); 
  894.  
  895.         // Copy name data from the old index table into the new individual table (if name empty in indiv table) 
  896.         $sql = " 
  897.             UPDATE `@wp_pods_{$pod}` AS `t` 
  898.             LEFT JOIN `@wp_pod_types` AS `x` ON `x`.`name` = '{$pod}' 
  899.             LEFT JOIN `@wp_pod` AS `p` ON `p`.`datatype` = `x`.`id` AND `p`.`tbl_row_id` = `t`.`id` 
  900.             SET `t`.`name` = `p`.`name` 
  901.             WHERE ( `t`.`name` IS NULL OR `t`.`name` = '' ) AND `x`.`id` IS NOT NULL AND `p`.`id` IS NOT NULL 
  902.         "; 
  903.  
  904.         pods_query( $sql ); 
  905.  
  906.         $this->update_progress( __FUNCTION__, true, $pod );
