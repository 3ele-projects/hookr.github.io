  557.             $_view = false; 
  558.  
  559.             foreach ( $_views as $_view_check ) { 
  560.                 $_view = self::locate_template( $_view_check ); 
  561.  
  562.                 if ( !empty( $_view ) ) { 
  563.                     break; 
  564.                 } 
  565.             } 
  566.  
  567.             return $_view; 
  568.         } 
  569.  
  570.         // Keep it safe 
  571.         $_view = trim( str_replace( array( '../', '\\' ), array( '', '/' ), (string) $_view ) ); 
  572.  
  573.         if ( empty( $_view ) ) { 
  574.             return false; 
  575.         } 
  576.  
  577.         $_real_view = realpath( $_view ); 
  578.  
  579.         if ( empty( $_real_view ) ) { 
  580.             $_real_view = $_view; 
  581.         } 
  582.  
  583.         $located = false; 
  584.  
  585.         // Is the view's file somewhere within the plugin directory tree? 
  586.         // Note: we explicitly whitelist PODS_DIR for the case of symlinks (see issue #2945) 
  587.         if ( false !== strpos( $_real_view, realpath( WP_PLUGIN_DIR ) ) || 
  588.              false !== strpos( $_real_view, realpath( WPMU_PLUGIN_DIR ) ) || 
  589.              false !== strpos( $_real_view, PODS_DIR ) 
  590.  ) { 
  591.             if ( file_exists( $_view ) ) { 
  592.                 $located = $_view; 
  593.             } 
  594.             else { 
  595.                 $located = apply_filters( 'pods_view_locate_template', $located, $_view ); 
  596.             } 
  597.  
  598.         } else { // The view's file is outside the plugin directory 
  599.             $_real_view = trim( $_real_view, '/' ); 
  600.  
  601.             if ( empty( $_real_view ) ) { 
  602.                 return false; 
  603.             } 
  604.  
  605.             // Allow views in the theme or child theme 
  606.             if ( file_exists( realpath( get_stylesheet_directory() . '/' . $_real_view ) ) ) { 
  607.                 $located = realpath( get_stylesheet_directory() . '/' . $_real_view ); 
  608.             } 
  609.             elseif ( file_exists( realpath( get_template_directory() . '/' . $_real_view ) ) ) { 
  610.                 $located = realpath( get_template_directory() . '/' . $_real_view ); 
  611.             } 
  612.         } 
  613.  
  614.         return $located; 
  615.

  618.     /** 
  619.      * Advanced $expires handling 
  620.      * 
  621.      * @param array|bool|int $expires 
  622.      * @param string $cache_mode 
  623.      * 
  624.      * @return bool|int 
  625.      * 
  626.      * @since 3.0 
  627.      * @static 
  628.      */ 
  629.     public static function expires( $expires, $cache_mode = 'cache' ) { 
  630.  
  631.         // Different $expires if user is anonymous or logged in or specific capability 
  632.         if ( is_array( $expires ) ) { 
  633.             if ( ( isset( $expires[ 'anonymous' ] ) || isset( $expires[ 'user_with_access' ] ) ) && isset( $expires[ 'user' ] ) ) { 
  634.                 if ( isset( $expires[ 'user_with_access' ] ) ) { 
  635.                     $expires = array( 
  636.                         pods_var_raw( 'anonymous', $expires, false ),  
  637.                         pods_var_raw( 'user', $expires, false ),  
  638.                         pods_var_raw( 'user_with_access', $expires, false ),  
  639.                         pods_var_raw( 'capability', $expires, null, null, true ) 
  640.  ); 
  641.                 } 
  642.                 elseif ( isset( $expires[ 'anonymous' ] ) ) { 
  643.                     $expires = array( 
  644.                         pods_var_raw( 'anonymous', $expires, false ),  
  645.                         pods_var_raw( 'user', $expires, false ),  
  646.                         pods_var_raw( 'capability', $expires, null, null, true ) 
  647.  ); 
  648.                 } 
  649.             } 
  650.             else { 
  651.                 $expires = array_values( $expires ); 
  652.             } 
  653.  
  654.             if ( 4 == count( $expires ) ) { 
  655.                 if ( !is_user_logged_in() ) { 
  656.                     $expires = pods_var_raw( 0, $expires, false ); 
  657.                 } 
  658.                 else { 
  659.                     $user_no_access = pods_var_raw( 1, $expires, false ); 
  660.                     $user_with_access = pods_var_raw( 2, $expires, false ); 
  661.                     $capability = pods_var_raw( 3, $expires, null, null, true ); 
  662.  
  663.                     $expires = pods_var_user( $user_no_access, $user_with_access, $capability ); 
  664.                 } 
  665.             } 
  666.             else { 
  667.                 $anon = pods_var_raw( 0, $expires, false ); 
  668.                 $user = pods_var_raw( 1, $expires, false ); 
  669.                 $capability = pods_var_raw( 2, $expires, null, null, true ); 
  670.  
  671.                 $expires = pods_var_user( $anon, $user, $capability ); 
  672.             } 
  673.         } 
  674.  
  675.         if ( 'none' == $cache_mode ) { 
  676.             $expires = false; 
  677.         } 
  678.         elseif ( false !== $expires ) { 
  679.             $expires = (int) $expires; 
  680.  
  681.             if ( $expires < 1 ) { 
  682.                 $expires = 0; 
  683.             } 
  684.         } 
  685.  
  686.         return $expires; 
  687.  
  688.     }
