  191. function pods_attachment_import ( $url, $post_parent = null, $featured = false ) { 
  192.   $filename = explode( '?', $url ); 
  193.   $filename = $filename[ 0 ]; 
  194.  
  195.   $filename = explode( '#', $filename ); 
  196.   $filename = $filename[ 0 ]; 
  197.  
  198.   $filename = substr( $filename, ( strrpos( $filename, '/' ) ) + 1 ); 
  199.  
  200.   $title = substr( $filename, 0, ( strrpos( $filename, '.' ) ) ); 
  201.  
  202.   if ( !( ( $uploads = wp_upload_dir( current_time( 'mysql' ) ) ) && false === $uploads[ 'error' ] ) ) 
  203.       return 0; 
  204.  
  205.   $filename = wp_unique_filename( $uploads[ 'path' ], $filename ); 
  206.   $new_file = $uploads[ 'path' ] . '/' . $filename; 
  207.  
  208.   $file_data = @file_get_contents( $url ); 
  209.  
  210.   if ( !$file_data ) 
  211.       return 0; 
  212.  
  213.   file_put_contents( $new_file, $file_data ); 
  214.  
  215.   $stat = stat( dirname( $new_file ) ); 
  216.   $perms = $stat[ 'mode' ] & 0000666; 
  217.   @chmod( $new_file, $perms ); 
  218.  
  219.   $wp_filetype = wp_check_filetype( $filename ); 
  220.  
  221.   if ( !$wp_filetype[ 'type' ] || !$wp_filetype[ 'ext' ] ) 
  222.       return 0; 
  223.  
  224.   $attachment = array( 
  225.       'post_mime_type' => $wp_filetype[ 'type' ],  
  226.       'guid' => $uploads[ 'url' ] . '/' . $filename,  
  227.       'post_parent' => null,  
  228.       'post_title' => $title,  
  229.       'post_content' => '',  
  230.  ); 
  231.  
  232.   $attachment_id = wp_insert_attachment( $attachment, $new_file, $post_parent ); 
  233.  
  234.   if ( is_wp_error( $attachment_id ) ) 
  235.       return 0; 
  236.  
  237.   require_once( ABSPATH . 'wp-admin/includes/image.php' ); 
  238.  
  239.   wp_update_attachment_metadata( $attachment_id, $meta_data = wp_generate_attachment_metadata( $attachment_id, $new_file ) ); 
  240.  
  241.   if ( 0 < $post_parent && $featured ) 
  242.       update_post_meta( $post_parent, '_thumbnail_id', $attachment_id ); 
  243.  
  244.   return $attachment_id; 
  245. }

  257. function pods_image_resize ( $attachment_id, $size ) { 
  258.   $size_data = array(); 
  259.  
  260.   // Basic image size string 
  261.   if ( !is_array( $size ) ) { 
  262.       global $wp_image_sizes; 
  263.  
  264.       // Registered image size 
  265.       if ( isset( $wp_image_sizes[ $size ] ) && !empty( $wp_image_sizes[ $size ] ) ) 
  266.           $size_data = $wp_image_sizes[ $size ]; 
  267.       // Custom on-the-fly image size 
  268.       elseif ( preg_match( '/[0-9]+x[0-9]+/', $size ) || preg_match( '/[0-9]+x[0-9]+x[0-1]/', $size ) ) { 
  269.           $size = explode( 'x', $size ); 
  270.  
  271.           $size_data = array( 
  272.               'width' => (int) $size[ 0 ],  
  273.               'height' => (int) $size[ 1 ],  
  274.               'crop' => (int) ( isset( $size[ 2 ] ) ? $size[ 2 ] : 1 ),  
  275.  ); 
  276.  
  277.           $size = $size_data[ 'width' ] . 'x' . $size_data[ 'height' ]; 
  278.       } 
  279.   } 
  280.   // Image size array 
  281.   elseif ( 2 <= count( $size ) ) { 
  282.       if ( isset( $size[ 'width' ] ) ) 
  283.           $size_data = $size; 
  284.       else { 
  285.           $size_data = array( 
  286.               'width' => (int) $size[ 0 ],  
  287.               'height' => (int) $size[ 1 ],  
  288.               'crop' => (int) ( isset( $size[ 2 ] ) ? $size[ 2 ] : 1 ),  
  289.  ); 
  290.       } 
  291.  
  292.       $size = $size_data[ 'width' ] . 'x' . $size_data[ 'height' ]; 
  293.   } 
  294.  
  295.   if ( empty( $size_data ) ) 
  296.       return false; 
  297.  
  298.   require_once( ABSPATH . 'wp-admin/includes/image.php' ); 
  299.  
  300.   $attachment = get_post( $attachment_id ); 
  301.   $file = get_attached_file( $attachment_id ); 
  302.  
  303.   if ( $file && file_exists( $file ) ) { 
  304.        $metadata = wp_get_attachment_metadata( $attachment_id ); 
  305.  
  306.       if ( !empty( $metadata ) && preg_match( '!^image/!', get_post_mime_type( $attachment ) ) && file_is_displayable_image( $file ) ) { 
  307.           $editor = wp_get_image_editor( $file ); 
  308.  
  309.           if ( !is_wp_error( $editor ) ) { 
  310.               $metadata[ 'sizes' ] = array_merge( $metadata[ 'sizes' ], $editor->multi_resize( array( $size => $size_data ) ) ); 
  311.  
  312.               wp_update_attachment_metadata( $attachment_id, $metadata ); 
  313.  
  314.               return true; 
  315.           } 
  316.       } 
  317.   } 
  318.  
  319.   return false; 
  320. }
