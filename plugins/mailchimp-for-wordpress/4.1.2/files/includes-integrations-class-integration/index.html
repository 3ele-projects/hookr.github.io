  343.   /** 
  344.    * Makes a subscription request 
  345.    *

  347.    * @param int $related_object_id 
  348.    * 
  349.    * @return boolean 
  350.    */ 
  351.   protected function subscribe( array $data, $related_object_id = 0 ) { 
  352.  
  353.       $integration = $this; 
  354.       $slug = $this->slug; 
  355.       $mailchimp = new MC4WP_MailChimp(); 
  356.       $log = $this->get_log(); 
  357.       $request= $this->get_request(); 
  358.       $list_ids = $this->get_lists(); 
  359.  
  360.       /** @var MC4WP_MailChimp_Subscriber $subscriber */ 
  361.       $subscriber = null; 
  362.       $result = false; 
  363.  
  364.       // validate lists 
  365.       if( empty( $list_ids ) ) { 
  366.           $log->warning( sprintf( '%s > No MailChimp lists were selected', $this->name ) ); 
  367.           return false; 
  368.       } 
  369.  
  370.       /** 
  371.        * Filters data for integration requests. 
  372.        * 
  373.        * @param array $data 
  374.        */ 
  375.       $data = apply_filters( 'mc4wp_integration_data', $data ); 
  376.  
  377.       /** 
  378.        * Filters data for a specific integration request. 
  379.        * 
  380.        * The dynamic portion of the hook, `$slug`, refers to the integration slug. 
  381.        * 
  382.        * @param array $data 
  383.        * @param int $related_object_id 
  384.        */ 
  385.       $data = apply_filters( "mc4wp_integration_{$slug}_data", $data, $related_object_id ); 
  386.  
  387.       /** 
  388.        * @ignore 
  389.        * @deprecated 4.0 
  390.        */ 
  391.       $data = apply_filters( 'mc4wp_merge_vars', $data ); 
  392.  
  393.       /** 
  394.        * @deprecated 4.0 
  395.        * @ignore 
  396.        */ 
  397.       $data = apply_filters( 'mc4wp_integration_merge_vars', $data, $integration ); 
  398.  
  399.       /** 
  400.        * @deprecated 4.0 
  401.        * @ignore 
  402.        */ 
  403.       $data = apply_filters( "mc4wp_integration_{$slug}_merge_vars", $data, $integration ); 
  404.  
  405.       $email_type = mc4wp_get_email_type(); 
  406.  
  407.       $mapper = new MC4WP_List_Data_Mapper( $data, $list_ids ); 
  408.  
  409.       /** @var MC4WP_MailChimp_Subscriber[] $map */ 
  410.       $map = $mapper->map(); 
  411.  
  412.       foreach( $map as $list_id => $subscriber ) { 
  413.           $subscriber->status = $this->options['double_optin'] ? 'pending' : 'subscribed'; 
  414.           $subscriber->email_type = $email_type; 
  415.           $subscriber->ip_signup = $request->get_client_ip(); 
  416.  
  417.           /** @ignore (documented elsewhere) */ 
  418.           $subscriber = apply_filters( 'mc4wp_subscriber_data', $subscriber ); 
  419.  
  420.           /** 
  421.            * Filters subscriber data before it is sent to MailChimp. Only fires for integration requests. 
  422.            * 
  423.            * @param MC4WP_MailChimp_Subscriber $subscriber 
  424.            */ 
  425.           $subscriber = apply_filters( 'mc4wp_integration_subscriber_data', $subscriber ); 
  426.  
  427.           /** 
  428.            * Filters subscriber data before it is sent to MailChimp. Only fires for integration requests. 
  429.            * 
  430.            * The dynamic portion of the hook, `$slug`, refers to the integration slug. 
  431.            * 
  432.            * @param MC4WP_MailChimp_Subscriber $subscriber 
  433.            * @param int $related_object_id 
  434.            */ 
  435.           $subscriber = apply_filters( "mc4wp_integration_{$slug}_subscriber_data", $subscriber, $related_object_id ); 
  436.  
  437.           $result = $mailchimp->list_subscribe( $list_id, $subscriber->email_address, $subscriber->to_array(), $this->options['update_existing'], $this->options['replace_interests'] ); 
  438.       } 
  439.  
  440.       // if result failed, show error message 
  441.       if( ! $result ) { 
  442.  
  443.           // log error 
  444.           if( $mailchimp->get_error_code() == 214 ) { 
  445.               $log->warning( sprintf( "%s > %s is already subscribed to the selected list(s)", $this->name, $subscriber->email_address ) ); 
  446.           } else { 
  447.               $log->error( sprintf( '%s > MailChimp API Error: %s', $this->name, $mailchimp->get_error_message() ) ); 
  448.           } 
  449.  
  450.           // bail 
  451.           return false; 
  452.       } 
  453.  
  454.       $log->info( sprintf( '%s > Successfully subscribed %s', $this->name, $subscriber->email_address ) ); 
  455.  
  456.       /** 
  457.        * Runs right after someone is subscribed using an integration 
  458.        * 
  459.        * @since 3.0 
  460.        * 
  461.        * @param MC4WP_Integration $integration 
  462.        * @param string $email_address 
  463.        * @param array $merge_vars 
  464.        * @param MC4WP_MailChimp_Subscriber[] $subscriber_data 
  465.        * @param int $related_object_id

  467.       do_action( 'mc4wp_integration_subscribed', $integration, $subscriber->email_address, $subscriber->merge_fields, $map, $related_object_id );

  469.       return $result; 
  470.   }
