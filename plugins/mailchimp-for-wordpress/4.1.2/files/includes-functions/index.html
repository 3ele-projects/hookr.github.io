  194. function _mc4wp_update_groupings_data($data = array()) 
  195. { 
  196.  
  197.   // data still has old "GROUPINGS" key? 
  198.   if (empty($data['GROUPINGS'])) { 
  199.       return $data; 
  200.   } 
  201.  
  202.   // prepare new key 
  203.   if (!isset($data['INTERESTS'])) { 
  204.       $data['INTERESTS'] = array(); 
  205.   } 
  206.  
  207.   $map = get_option('mc4wp_groupings_map', array()); 
  208.  
  209.   foreach ($data['GROUPINGS'] as $grouping_id => $groups) { 
  210.  
  211.       // for compatibility with expanded grouping arrays 
  212.       $grouping_key = $grouping_id; 
  213.       if (is_array($groups) && isset($groups['id']) && isset($groups['groups'])) { 
  214.           $grouping_id = $groups['id']; 
  215.           $groups = $groups['groups']; 
  216.       } 
  217.  
  218.       // do we have transfer data for this grouping id? 
  219.       if (!isset($map[$grouping_id])) { 
  220.           continue; 
  221.       } 
  222.  
  223.       // if we get a string, explode on delimiter(s) 
  224.       if (is_string($groups)) { 
  225.           // for BC with 3.x: explode on comma's 
  226.           $groups = join('|', explode(', ', $groups)); 
  227.  
  228.           // explode on current delimiter 
  229.           $groups = explode('|', $groups); 
  230.       } 
  231.  
  232.       // loop through groups and find interest ID 
  233.       $migrated = 0; 
  234.       foreach ($groups as $key => $group_name_or_id) { 
  235.  
  236.           // do we know the new interest ID? 
  237.           if (empty($map[$grouping_id]['groups'][$group_name_or_id])) { 
  238.               continue; 
  239.           } 
  240.  
  241.           $interest_id = $map[$grouping_id]['groups'][$group_name_or_id]; 
  242.  
  243.           // add to interests data 
  244.           if (!in_array($interest_id, $data['INTERESTS'])) { 
  245.               $migrated++; 
  246.               $data['INTERESTS'][] = $interest_id; 
  247.           } 
  248.       } 
  249.  
  250.       // remove old grouping ID if we migrated all groups. 
  251.       if ($migrated === count($groups)) { 
  252.           unset($data['GROUPINGS'][$grouping_key]); 
  253.       } 
  254.   } 
  255.  
  256.   // if everything went well, this is now empty & moved to new INTERESTS key. 
  257.   if (empty($data['GROUPINGS'])) { 
  258.       unset($data['GROUPINGS']); 
  259.   } 
  260.  
  261.   // is this empty? just unset it then. 
  262.   if (empty($data['INTERESTS'])) { 
  263.       unset($data['INTERESTS']); 
  264.   } 
  265.  
  266.   return $data; 
  267. }
