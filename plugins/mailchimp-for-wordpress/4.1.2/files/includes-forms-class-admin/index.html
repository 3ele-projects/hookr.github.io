  224.   public function sanitize_form_data( $data ) { 
  225.  
  226.       $raw_data = $data;

  228.       // strip <form> tags from content

  230.  
  231.       // sanitize text fields 
  232.       $data['settings']['redirect'] = sanitize_text_field( $data['settings']['redirect'] ); 
  233.  
  234.       // strip tags from messages

  236.           $data['messages'][$key] = strip_tags( $message, '<strong><b><br><a><script><u><em><i><span><img>' ); 
  237.       } 
  238.  
  239.       // make sure lists is an array 
  240.       if( ! isset( $data['settings']['lists'] ) ) { 
  241.           $data['settings']['lists'] = array(); 
  242.       } 
  243.  
  244.       $data['settings']['lists'] = array_filter( (array) $data['settings']['lists'] ); 
  245.  
  246.       /** 
  247.        * Filters the form data just before it is saved. 
  248.        * 
  249.        * @param array $data Sanitized array of form data. 
  250.        * @param array $raw_data Raw array of form data.

  252.        * @since 3.0.8

  255.       $data = (array) apply_filters( 'mc4wp_form_sanitized_data', $data, $raw_data );
