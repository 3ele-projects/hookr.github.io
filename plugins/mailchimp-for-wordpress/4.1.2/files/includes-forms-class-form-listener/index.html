  27.   public function listen( MC4WP_Request $request ) {

  29.       $form_id = $request->post->get( '_mc4wp_form_id' ); 
  30.       if( empty( $form_id ) ) { 
  31.           return false; 
  32.       } 
  33.  
  34.       // get form instance

  53.           call_user_func( array( $this, $method ), $form, $request ); 
  54.       } else { 
  55.           $this->get_log()->info( sprintf( "Form %d > Submitted with errors: %s", $form->ID, join( ', ', $form->errors ) ) ); 
  56.       } 
  57.  
  58.       $this->respond( $form ); 
  59.  
  60.       return true; 
  61.   }

  67.    * @param MC4WP_Request $request

  69.   public function process_subscribe_form( MC4WP_Form $form, MC4WP_Request $request ) { 
  70.       $result = false; 
  71.       $mailchimp = new MC4WP_MailChimp(); 
  72.       $email_type = $form->get_email_type(); 
  73.       $data = $form->get_data(); 
  74.       $client_ip = $request->get_client_ip(); 
  75.  
  76.       /** @var MC4WP_MailChimp_Subscriber $subscriber */ 
  77.       $subscriber = null; 
  78.  
  79.       /** 
  80.        * @ignore 
  81.        * @deprecated 4.0 
  82.        */ 
  83.       $data = apply_filters( 'mc4wp_merge_vars', $data ); 
  84.        
  85.       /** 
  86.        * @ignore 
  87.        * @deprecated 4.0 
  88.        */ 
  89.       $data = (array) apply_filters( 'mc4wp_form_merge_vars', $data, $form ); 
  90.  
  91.       // create a map of all lists with list-specific data 
  92.       $mapper = new MC4WP_List_Data_Mapper( $data, $form->get_lists() ); 
  93.  
  94.       /** @var MC4WP_MailChimp_Subscriber[] $map */ 
  95.       $map = $mapper->map(); 
  96.  
  97.       // loop through lists 
  98.       foreach( $map as $list_id => $subscriber ) { 
  99.  
  100.           $subscriber->status = $form->settings['double_optin'] ? 'pending' : 'subscribed'; 
  101.           $subscriber->email_type = $email_type; 
  102.           $subscriber->ip_signup = $client_ip; 
  103.  
  104.           /** 
  105.            * Filters subscriber data before it is sent to MailChimp. Fires for both form & integration requests. 
  106.            * 
  107.            * @param MC4WP_MailChimp_Subscriber $subscriber 
  108.            */ 
  109.           $subscriber = apply_filters( 'mc4wp_subscriber_data', $subscriber ); 
  110.  
  111.           /** 
  112.            * Filters subscriber data before it is sent to MailChimp. Only fires for form requests. 
  113.            * 
  114.            * @param MC4WP_MailChimp_Subscriber $subscriber 
  115.            */ 
  116.           $subscriber = apply_filters( 'mc4wp_form_subscriber_data', $subscriber ); 
  117.  
  118.           // send a subscribe request to MailChimp for each list 
  119.           $result = $mailchimp->list_subscribe( $list_id, $subscriber->email_address, $subscriber->to_array(), $form->settings['update_existing'], $form->settings['replace_interests'] ); 
  120.       } 
  121.  
  122.       $log = $this->get_log(); 
  123.  
  124.       // do stuff on failure 
  125.       if( ! is_object( $result ) || empty( $result->id ) ) { 
  126.  
  127.           $error_code = $mailchimp->get_error_code(); 
  128.           $error_message = $mailchimp->get_error_message(); 
  129.            
  130.           if( $mailchimp->get_error_code() == 214 ) { 
  131.               $form->add_error('already_subscribed'); 
  132.               $log->warning( sprintf( "Form %d > %s is already subscribed to the selected list(s)", $form->ID, $data['EMAIL'] ) ); 
  133.           } else { 
  134.               $form->add_error('error'); 
  135.               $log->error( sprintf( 'Form %d > MailChimp API error: %s %s', $form->ID, $error_code, $error_message ) ); 
  136.  
  137.               /** 
  138.                * Fire action hook so API errors can be hooked into. 
  139.                * 
  140.                * @param MC4WP_Form $form 
  141.                * @param string $error_message 
  142.                */ 
  143.               do_action( 'mc4wp_form_api_error', $form, $error_message ); 
  144.           } 
  145.  
  146.           // bail 
  147.           return; 
  148.       } 
  149.  
  150.       // Success! Did we update or newly subscribe? 
  151.       if( $result->status === 'subscribed' && $result->was_already_on_list ) { 
  152.           $form->add_message('updated'); 
  153.       } else { 
  154.           $form->add_message('subscribed'); 
  155.       } 
  156.  
  157.       $log->info( sprintf( "Form %d > Successfully subscribed %s", $form->ID, $data['EMAIL'] ) ); 
  158.  
  159.       /** 
  160.        * Fires right after a form was used to subscribe. 
  161.        * 
  162.        * @since 3.0 
  163.        * 
  164.        * @param MC4WP_Form $form Instance of the submitted form 
  165.        * @param string $email 
  166.        * @param array $data 
  167.        * @param MC4WP_MailChimp_Subscriber[] $subscriber 
  168.        */ 
  169.       do_action( 'mc4wp_form_subscribed', $form, $subscriber->email_address, $data, $map ); 
  170.   }

  180.       $result = null;

  182.  
  183.       // unsubscribe from each list

  188.       if( ! $result ) {

  194.       } 
  195.  
  196.       // Success! Unsubscribed. 
  197.       $form->add_message('unsubscribed');

  199.  
  200.  
  201.       /**

  203.        *

  215.  
  216.       $success = ! $form->has_errors(); 
  217.  
  218.       if( $success ) { 
  219.  
  220.           /** 
  221.            * Fires right after a form is submitted without any errors (success). 
  222.            *

  231.           /** 
  232.            * Fires right after a form is submitted with errors. 
  233.            * 
  234.            * @since 3.0

  236.            * @param MC4WP_Form $form The submitted form instance.

  238.           do_action( 'mc4wp_form_error', $form ); 
  239.  
  240.           // fire a dedicated event for each error 
  241.           foreach( $form->errors as $error ) { 
  242.  
  243.               /** 
  244.                * Fires right after a form was submitted with errors. 
  245.                * 
  246.                * The dynamic portion of the hook, `$error`, refers to the error that occurred. 
  247.                * 
  248.                * Default errors give us the following possible hooks: 
  249.                * 
  250.                * - mc4wp_form_error_error                     General errors 
  251.                * - mc4wp_form_error_spam 
  252.                * - mc4wp_form_error_invalid_email             Invalid email address

  257.                * @since 3.0

  259.                * @param   MC4WP_Form     $form        The form instance of the submitted form.

  261.               do_action( 'mc4wp_form_error_' . $error, $form ); 
  262.           } 
  263.  
  264.       } 
  265.  
  266.       /** 
  267.        * Fires right before responding to the form request. 
  268.        * 
  269.        * @since 3.0

  272.        */ 
  273.       do_action( 'mc4wp_form_respond', $form ); 
  274.  
  275.       // do stuff on success (non-AJAX only) 
  276.       if( $success && ( ! defined( 'DOING_AJAX' ) || ! DOING_AJAX ) ) { 
  277.  
  278.           // do we want to redirect? 
  279.           $redirect_url = $form->get_redirect_url(); 
  280.           if ( ! empty( $redirect_url ) ) { 
  281.               wp_redirect( $redirect_url ); 
  282.               exit; 
  283.           } 
  284.       } 
  285.   }
