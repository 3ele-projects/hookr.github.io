  543.     private function parse_response( $response ) { 
  544.

  546.             throw new Exception( 'Error connecting to MailChimp. ' . $response->get_error_message(), (int) $response->get_error_code() ); 
  547.         } 
  548.  
  549.         // decode response body 
  550.         $body = wp_remote_retrieve_body( $response ); 
  551.         $data = json_decode( $body ); 
  552.         if( ! is_null( $data ) ) { 
  553.             return $data; 
  554.         } 
  555.  
  556.         $code = (int) wp_remote_retrieve_response_code( $response ); 
  557.         $message = wp_remote_retrieve_response_message( $response ); 
  558.  
  559.         if( $code !== 200 ) { 
  560.             $message = sprintf( 'The MailChimp API server returned the following response: <em>%s %s</em>.', $code, $message ); 
  561.  
  562.             // check for Akamai firewall response 
  563.             if( $code === 403 ) { 
  564.                 preg_match('/Reference (.*)/i', $body, $matches ); 
  565.  
  566.                 if( ! empty( $matches[1] ) ) { 
  567.                     $message .= '</strong><br /><br />' . sprintf( 'This usually means that your server is blacklisted by MailChimp\'s firewall. Please contact MailChimp support with the following reference number: %s </strong>', $matches[1] ); 
  568.                 } 
  569.             } 
  570.         } 
  571.  
  572.         throw new Exception( $message, $code );
