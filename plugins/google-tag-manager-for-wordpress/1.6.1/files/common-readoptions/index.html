  188. function gtm4wp_reload_options() { 
  189.   global $gtm4wp_defaultoptions; 
  190.    
  191.   $storedoptions = (array) get_option( GTM4WP_OPTIONS ); 
  192.   if ( ! is_array( $gtm4wp_defaultoptions ) ) { 
  193.       $gtm4wp_defaultoptions = array(); 
  194.   } 
  195.  
  196.   // update WooCommerce settings from 0.7.x to 0.8 
  197.   if ( isset( $storedoptions[ GTM4WP_OPTION_INTEGRATE_WOOCOMMERCE ] ) && ! isset( $storedoptions[ GTM4WP_OPTION_INTEGRATE_WCTRACKCLASSICEC ] ) ) { 
  198.       $storedoptions[ GTM4WP_OPTION_INTEGRATE_WCTRACKCLASSICEC ]  = $storedoptions[ GTM4WP_OPTION_INTEGRATE_WOOCOMMERCE ]; 
  199.       $storedoptions[ GTM4WP_OPTION_INTEGRATE_WCTRACKENHANCEDEC ] = false; 
  200.       $storedoptions[ GTM4WP_OPTION_INTEGRATE_WCREMARKETING ]     = $storedoptions[ GTM4WP_OPTION_INTEGRATE_WOOCOMMERCE ]; 
  201.  
  202.       unset( $storedoptions[ GTM4WP_OPTION_INTEGRATE_WOOCOMMERCE ] ); 
  203.   } 
  204.    
  205.   return array_merge( $gtm4wp_defaultoptions, $storedoptions ); 
  206. }

  208. function gtp4wp_debug_file( $debug_data ) {
