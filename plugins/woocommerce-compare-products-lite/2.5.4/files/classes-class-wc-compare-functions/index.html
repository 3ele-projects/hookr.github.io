  131.   public static function get_variation_name($variation_id) { 
  132.       $mypost = get_post($variation_id); 
  133.       $product_name = ''; 
  134.       if ($mypost != NULL) { 
  135.           if ($mypost->post_type == 'product_variation') { 
  136.               $attributes = (array) maybe_unserialize(get_post_meta($mypost->post_parent, '_product_attributes', true)); 
  137.               $my_variation = new WC_Product_Variation($variation_id, $mypost->post_parent); 
  138.  
  139.               $variation_data = $my_variation->get_variation_attributes(); 
  140.               $variation_name = ''; 
  141.               if (is_array($attributes) && count($attributes) > 0) { 
  142.                   foreach ($attributes as $attribute) { 
  143.                       if ( !$attribute['is_variation'] ) continue; 
  144.                       $taxonomy = 'attribute_' . sanitize_title($attribute['name']); 
  145.                       if (isset($variation_data[$taxonomy])) { 
  146.                           if (taxonomy_exists(sanitize_title($attribute['name']))) { 
  147.                               $term = get_term_by('slug', $variation_data[$taxonomy], sanitize_title($attribute['name'])); 
  148.                               if (!is_wp_error($term) && isset($term->name) && $term->name != '') { 
  149.                                   $value = $term->name; 
  150.                                   $variation_name .= ' '.$value; 
  151.                               } 
  152.                           }else { 
  153.                               $variation_name .= ' '. wc_clean( $variation_data[$taxonomy] ); 
  154.                           } 
  155.                       } 
  156.  
  157.                   } 
  158.               } 
  159.  
  160.               $product_name = get_the_title($mypost->post_parent).' -'.$variation_name; 
  161.           }else { 
  162.               $product_name = get_the_title($variation_id); 
  163.           } 
  164.       } 
  165.  
  166.       return $product_name; 
  167.   }
