  558.             foreach ( $ftc as $cache ) { 
  559.                 if ( ! empty( $cache ) ) { 
  560.                     $ftc_is_empty = FALSE; 
  561.                 } 
  562.             } 
  563.  
  564.         } 
  565.  
  566.         if ( $ftc === FALSE || $ftc_is_empty === TRUE ) { 
  567.             //$this->error( 3, 'A feed term cache has not recently been created, nothing to clear' ); 
  568.             return FALSE; 
  569.         } 
  570.  
  571.         // Return false if given segment doesn't exist 
  572.         if ( $segment !== NULL && ! array_key_exists($segment, $ftc) ) { 
  573.             $this->error( 2, 'Given segment [' . $segment . '] does not exist in feed term cache:' . __LINE__ ); 
  574.             return FALSE; 
  575.         } 
  576.  
  577.         // Return false if data for given identifier doesn't exist 
  578.         if ( $segment !== NULL && array_key_exists($segment, $ftc) && ! isset($ftc[$segment][$term]) ) { 
  579.             $this->error( 3, 'Given term [' . $term . '] does not exist in feed term cache:' . __LINE__); 
  580.             return FALSE; 
  581.         } 
  582.  
  583.         // Clear the term from the feed term cache 
  584.         // If $segment is NULL, clear all instances of $term 
  585.         if ( $segment === NULL ) { 
  586.             foreach ( $ftc as $s => $t_data ) { 
  587.                 foreach ( $t_data as $t => $data ) { 
  588.                     if ( $t === $term ) { 
  589.                         unset($ftc[$s][$term]); 
  590.                     } 
  591.                 } 
  592.             } 
  593.  
  594.         } else { 
  595.             unset($ftc[$segment][$term]); 
  596.  
  597.         } 
  598.  
  599.         // Push the updated feed term cache 
  600.         $ftc_updated = set_transient( $this->plugin_name . '_ftc', $ftc, 3600 * 24 ); 
  601.         if ( $ftc_updated === TRUE ) { 
  602.             return TRUE; 
  603.         } else { 
  604.             $this->error( 2, __METHOD__ . ', ftc transient update was unsuccessful' ); 
  605.             return FALSE; 
  606.         } 
  607.  
  608.     }

  611.     /** 
  612.     * Batch clear feed cache 
  613.     * 
  614.     * @access public 
  615.     * @return string A line of text describing the outcome of the operation 
  616.     * @since 1.2.0 
  617.     * 
  618.     * @param string $segment The segment of the cache you wish to clear 
  619.     */ 
  620.     public function batch_clear_cache( $segment = 'all' ) { 
  621.         if ( is_array($segment) && isset($segment['segment']) ) { 
  622.             $segment = $segment['segment']; 
  623.         } 
  624.  
  625.         // Check that a cache segment has been passed 
  626.         if ( $segment === '0' ) { 
  627.             return 'no_segment_chosen'; 
  628.         } 
  629.  
  630.         // Check for existing feed term cache 
  631.         $ftc = get_transient( $this->plugin_name . '_ftc' ); 
  632.         $no_cache_records = FALSE; 
  633.  
  634.         // Check that the transient did exist 
  635.         if ( $ftc === FALSE ) { 
  636.             $no_cache_records = TRUE; 
  637.         } 
  638.  
  639.         /** Though the transient may exist, it may be empty so we 
  640.            check through each segment to see if it contains data */ 
  641.         if ( $ftc !== FALSE ) { 
  642.             $active_segments = 0; 
  643.  
  644.             foreach ( $ftc as $seg ) { 
  645.                 if ( is_array($seg) && count($seg) > 0 ) { 
  646.                     $active_segments++; 
  647.                 } 
  648.             } 
  649.  
  650.             if ( $active_segments === 0 ) { 
  651.                 $no_cache_records = TRUE; 
  652.             } 
  653.         } 
  654.  
  655.         if ( $no_cache_records === TRUE ) { 
  656.             return 'no_cache_data_to_clear'; 
  657.         } 
  658.  
  659.         // Holds info on each term cache clear. Useful for debugging 
  660.         $feedback = array(); 
  661.  
  662.         switch ( $segment ) { 
  663.             case 'user_timeline': 
  664.                 $clearance_list = $ftc['user_timeline']; 
  665.             break; 
  666.  
  667.             case 'search': 
  668.                 $clearance_list = $ftc['search']; 
  669.             break; 
  670.  
  671.             case 'all': 
  672.                 $clearance_list = NULL; 
  673.             break; 
  674.  
  675.             default: 
  676.                 $clearance_list = FALSE; 
  677.             break; 
  678.         } 
  679.  
  680.         $cache_cleared = FALSE; 
  681.  
  682.         if ( $clearance_list !== FALSE && is_array($clearance_list) ) { 
  683.  
  684.             // Check that there's data to clear and clear if so 
  685.             if ( is_array($clearance_list) && count($clearance_list) > 0 ) { 
  686.                 foreach ( $clearance_list as $term => $term_data ) { 
  687.                     $feedback[ $term ] = $this->clear_cache_output( $term, $segment ); 
  688.                     $cache_cleared = TRUE; 
  689.                 } 
  690.  
  691.             } else { 
  692.                 $cache_cleared = FALSE; 
  693.             } 
  694.  
  695.             if ( $cache_cleared ) { 
  696.                 return ucfirst($segment) . '_cache_cleared'; 
  697.             } else { 
  698.                 return 'no_' . $segment . '_cache_data_to_clear'; 
  699.             } 
  700.  
  701.         } elseif ( $clearance_list === NULL ) { 
  702.             foreach ( $ftc as $s => $t_data ) { 
  703.                 foreach ( $t_data as $t => $data ) { 
  704.                     $feedback[ $t ] = $this->clear_cache_output( $t, $s ); 
  705.                     $cache_cleared = TRUE; 
  706.                 } 
  707.             } 
  708.  
  709.             if ( $cache_cleared ) { 
  710.                 return 'cache_cleared'; 
  711.             } else { 
  712.                 return 'cache_not_cleared'; 
  713.             } 
  714.  
  715.         } else { 
  716.             return 'unknown_error'; 
  717.         } 
  718.     }
