  91. function wpcf7_number_validation_filter( $result, $tag ) { 
  92.   $tag = new WPCF7_FormTag( $tag ); 
  93.  
  94.   $name = $tag->name; 
  95.  
  96.   $value = isset( $_POST[$name] ) 
  97.       ? trim( strtr( (string) $_POST[$name], "\n", " " ) ) 
  98.       : ''; 
  99.  
  100.   $min = $tag->get_option( 'min', 'signed_int', true ); 
  101.   $max = $tag->get_option( 'max', 'signed_int', true ); 
  102.  
  103.   if ( $tag->is_required() && '' == $value ) { 
  104.       $result->invalidate( $tag, wpcf7_get_message( 'invalid_required' ) ); 
  105.   } elseif ( '' != $value && ! wpcf7_is_number( $value ) ) { 
  106.       $result->invalidate( $tag, wpcf7_get_message( 'invalid_number' ) ); 
  107.   } elseif ( '' != $value && '' != $min && (float) $value < (float) $min ) { 
  108.       $result->invalidate( $tag, wpcf7_get_message( 'number_too_small' ) ); 
  109.   } elseif ( '' != $value && '' != $max && (float) $max < (float) $value ) { 
  110.       $result->invalidate( $tag, wpcf7_get_message( 'number_too_large' ) ); 
  111.   } 
  112.  
  113.   return $result; 
  114. }
