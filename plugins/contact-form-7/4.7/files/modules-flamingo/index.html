  9. function wpcf7_flamingo_submit( $contactform, $result ) { 
  10.   if ( ! class_exists( 'Flamingo_Contact' ) 
  11.   || ! class_exists( 'Flamingo_Inbound_Message' ) ) { 
  12.       return; 
  13.   } 
  14.  
  15.   if ( $contactform->in_demo_mode() 
  16.   || $contactform->is_true( 'do_not_store' ) ) { 
  17.       return; 
  18.   } 
  19.  
  20.   $cases = (array) apply_filters( 'wpcf7_flamingo_submit_if',  
  21.       array( 'spam', 'mail_sent', 'mail_failed' ) ); 
  22.  
  23.   if ( empty( $result['status'] ) 
  24.   || ! in_array( $result['status'], $cases ) ) { 
  25.       return; 
  26.   } 
  27.  
  28.   $submission = WPCF7_Submission::get_instance(); 
  29.  
  30.   if ( ! $submission || ! $posted_data = $submission->get_posted_data() ) { 
  31.       return; 
  32.   } 
  33.  
  34.   $fields_senseless = $contactform->scan_form_tags( 
  35.       array( 'type' => array( 'captchar', 'quiz', 'acceptance' ) ) ); 
  36.  
  37.   $exclude_names = array(); 
  38.  
  39.   foreach ( $fields_senseless as $tag ) { 
  40.       $exclude_names[] = $tag['name']; 
  41.   } 
  42.  
  43.   $exclude_names[] = 'g-recaptcha-response'; 
  44.  
  45.   foreach ( $posted_data as $key => $value ) { 
  46.       if ( '_' == substr( $key, 0, 1 ) || in_array( $key, $exclude_names ) ) { 
  47.           unset( $posted_data[$key] ); 
  48.       } 
  49.   } 
  50.  
  51.   $email = wpcf7_flamingo_get_value( 'email', $contactform ); 
  52.   $name = wpcf7_flamingo_get_value( 'name', $contactform ); 
  53.   $subject = wpcf7_flamingo_get_value( 'subject', $contactform ); 
  54.  
  55.   $meta = array(); 
  56.  
  57.   $special_mail_tags = array( 'remote_ip', 'user_agent', 'url',  
  58.       'date', 'time', 'post_id', 'post_name', 'post_title', 'post_url',  
  59.       'post_author', 'post_author_email' ); 
  60.  
  61.   foreach ( $special_mail_tags as $smt ) { 
  62.       $meta[$smt] = apply_filters( 'wpcf7_special_mail_tags',  
  63.           '', '_' . $smt, false ); 
  64.   } 
  65.  
  66.   $akismet = isset( $submission->akismet ) ? (array) $submission->akismet : null; 
  67.  
  68.   if ( 'mail_sent' == $result['status'] ) { 
  69.       $flamingo_contact = Flamingo_Contact::add( array( 
  70.           'email' => $email,  
  71.           'name' => $name ) ); 
  72.   } 
  73.  
  74.   $channel_id = wpcf7_flamingo_add_channel( 
  75.       $contactform->name(), $contactform->title() ); 
  76.  
  77.   if ( $channel_id ) { 
  78.       $channel = get_term( $channel_id,  
  79.           Flamingo_Inbound_Message::channel_taxonomy ); 
  80.  
  81.       if ( ! $channel || is_wp_error( $channel ) ) { 
  82.           $channel = 'contact-form-7'; 
  83.       } else { 
  84.           $channel = $channel->slug; 
  85.       } 
  86.   } else { 
  87.       $channel = 'contact-form-7'; 
  88.   } 
  89.  
  90.   $args = array( 
  91.       'channel' => $channel,  
  92.       'subject' => $subject,  
  93.       'from' => trim( sprintf( '%s <%s>', $name, $email ) ),  
  94.       'from_name' => $name,  
  95.       'from_email' => $email,  
  96.       'fields' => $posted_data,  
  97.       'meta' => $meta,  
  98.       'akismet' => $akismet,  
  99.       'spam' => ( 'spam' == $result['status'] ) ); 
  100.  
  101.   $flamingo_inbound = Flamingo_Inbound_Message::add( $args ); 
  102.  
  103.   $result += array( 
  104.       'flamingo_contact_id' => 
  105.           empty( $flamingo_contact ) ? 0 : absint( $flamingo_contact->id ),  
  106.       'flamingo_inbound_id' => 
  107.           empty( $flamingo_inbound ) ? 0 : absint( $flamingo_inbound->id ) ); 
  108.  
  109.   do_action( 'wpcf7_after_flamingo', $result ); 
  110. }

  112. function wpcf7_flamingo_get_value( $field, $contactform ) { 
  113.   if ( empty( $field ) || empty( $contactform ) ) { 
  114.       return false; 
  115.   } 
  116.  
  117.   $value = ''; 
  118.  
  119.   if ( in_array( $field, array( 'email', 'name', 'subject' ) ) ) { 
  120.       $templates = $contactform->additional_setting( 'flamingo_' . $field ); 
  121.  
  122.       if ( empty( $templates[0] ) ) { 
  123.           $template = sprintf( '[your-%s]', $field ); 
  124.       } else { 
  125.           $template = trim( wpcf7_strip_quote( $templates[0] ) ); 
  126.       } 
  127.  
  128.       $value = wpcf7_mail_replace_tags( $template ); 
  129.   } 
  130.  
  131.   $value = apply_filters( 'wpcf7_flamingo_get_value', $value,  
  132.       $field, $contactform ); 
  133.  
  134.   return $value; 
  135. }

  137. function wpcf7_flamingo_add_channel( $slug, $name = '' ) { 
  138.   if ( ! class_exists( 'Flamingo_Inbound_Message' ) ) { 
  139.       return false; 
  140.   } 
  141.  
  142.   $parent = term_exists( 'contact-form-7',  
  143.       Flamingo_Inbound_Message::channel_taxonomy ); 
  144.  
  145.   if ( ! $parent ) { 
  146.       $parent = wp_insert_term( __( 'Contact Form 7', 'contact-form-7' ),  
  147.           Flamingo_Inbound_Message::channel_taxonomy,  
  148.           array( 'slug' => 'contact-form-7' ) ); 
  149.  
  150.       if ( is_wp_error( $parent ) ) { 
  151.           return false; 
  152.       } 
  153.   } 
  154.  
  155.   $parent = (int) $parent['term_id']; 
  156.  
  157.   if ( ! is_taxonomy_hierarchical( Flamingo_Inbound_Message::channel_taxonomy ) ) { 
  158.       // backward compat for Flamingo 1.0.4 and lower 
  159.       return $parent; 
  160.   } 
  161.  
  162.   if ( empty( $name ) ) { 
  163.       $name = $slug; 
  164.   } 
  165.  
  166.   $channel = term_exists( $slug,  
  167.       Flamingo_Inbound_Message::channel_taxonomy,  
  168.       $parent ); 
  169.  
  170.   if ( ! $channel ) { 
  171.       $channel = wp_insert_term( $name,  
  172.           Flamingo_Inbound_Message::channel_taxonomy,  
  173.           array( 'slug' => $slug, 'parent' => $parent ) ); 
  174.  
  175.       if ( is_wp_error( $channel ) ) { 
  176.           return false; 
  177.       } 
  178.   } 
  179.  
  180.   return (int) $channel['term_id']; 
  181. }

  193.   } 
  194.

  196.       return $output; 
  197.   } 
  198.

  200.       $contact_form->name(), $contact_form->title() ); 
  201.

  204.           array( 'channel_id' => $channel_id ) );
