  527.               } 
  528.           } 
  529.  
  530.           # Loop through all the (not set) stats and attempt to add them to their correct stat 
  531.           foreach ($notset_stats AS $stat) { 
  532.               # If the stat has a "partner" 
  533.               if ($new_stats[$stat[1]] != NULL) { 
  534.                   # Add the pageviews to the stat 
  535.                   $new_stats[$stat[1]][2] = $new_stats[$stat[1]][2] + $stat[2]; 
  536.               } else { 
  537.                   # Stat goes to the ether since we couldn't find a partner (if anyone reads this and has a suggestion to improve, let me know) 
  538.               } 
  539.           } 
  540.  
  541.           # Renew new_stats back to stats 
  542.           $stats = $new_stats; 
  543.  
  544.           # Sort the stats array, since adding the (not set) items may have changed the order 
  545.           usort($stats, array( 
  546.               $this,  
  547.               'statSort' 
  548.  )); 
  549.  
  550.           # Since we can no longer rely on the API as a limiter, we need to keep track of this ourselves 
  551.           $stat_count = 0; 
  552.  
  553.           # Loop through each stat for display 
  554.           foreach ($stats AS $stat) { 
  555.               $return .= '<li><a href="' . esc_url($stat[1]) . '">' . esc_html($stat[0]) . '</a> - ' . number_format($stat[2]) . ' ' . __('Views', 'google-analyticator') . '</li>'; 
  556.  
  557.               # Increase the stat counter 
  558.               $stat_count++; 
  559.  
  560.               # Stop at 5 
  561.               if ($stat_count >= 5) 
  562.                   break; 
  563.           } 
  564.  
  565.           # Finish the list 
  566.           $return .= '</ol>'; 
  567.       } 
  568.       echo $return;  
  569.   }

  637.    * Convert second to a time format 
  638.    **/ 
  639.   function sec2Time($time) 
  640.   { 
  641.       # Check if numeric 
  642.       if (is_numeric($time)) { 
  643.           $value = array( 
  644.               "years" => '00',  
  645.               "days" => '00',  
  646.               "hours" => '00',  
  647.               "minutes" => '00',  
  648.               "seconds" => '00' 
  649.  ); 
  650.           if ($time >= 31556926) { 
  651.               $value["years"] = floor($time / 31556926); 
  652.               $time = ($time % 31556926); 
  653.           } 
  654.           if ($time >= 86400) { 
  655.               $value["days"] = floor($time / 86400); 
  656.               $time = ($time % 86400); 
  657.           } 
  658.           if ($time >= 3600) { 
  659.               $value["hours"] = str_pad(floor($time / 3600), 2, 0, STR_PAD_LEFT); 
  660.               $time = ($time % 3600); 
  661.           } 
  662.           if ($time >= 60) { 
  663.               $value["minutes"] = str_pad(floor($time / 60), 2, 0, STR_PAD_LEFT); 
  664.               $time = ($time % 60); 
  665.           } 
  666.           $value["seconds"] = str_pad(floor($time), 2, 0, STR_PAD_LEFT); 
  667.            
  668.           # Get the hour:minute:second version 
  669.           return $value['hours'] . ':' . $value['minutes'] . ':' . $value['seconds']; 
  670.       } else { 
  671.           return false; 
  672.       } 
  673.   }
