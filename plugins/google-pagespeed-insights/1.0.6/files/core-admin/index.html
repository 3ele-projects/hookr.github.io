  445.   function prepare_items($ignored_query = false, $type = '', $per_page = 25) { 
  446.       global $wpdb; //This is used only if making any database queries 
  447.        
  448.       // Get options 
  449.       $gpi_options = $this->getOptions(); 
  450.  
  451.       // Setup Columns 
  452.       $columns = $this->gpi_get_columns($ignored_query, $gpi_options['strategy']); 
  453.       $hidden = array(); 
  454.       $sortable = $this->get_sortable_columns($ignored_query); 
  455.       $this->_column_headers = array($columns, $hidden, $sortable);        
  456.        
  457.       // Figure out which page types we are checking (set in options) 
  458.       $typestocheck = $this->getTypesToCheck($type); 
  459.  
  460.       // Not Null check for Report List scores 
  461.       switch($gpi_options['strategy']) { 
  462.  
  463.           case 'both': 
  464.               $nullcheck = 'desktop_score IS NOT NULL AND mobile_score IS NOT NULL'; 
  465.               break; 
  466.  
  467.           case 'mobile': 
  468.               $nullcheck = 'mobile_score IS NOT NULL'; 
  469.               break; 
  470.  
  471.           case 'desktop': 
  472.               $nullcheck = 'desktop_score IS NOT NULL'; 
  473.               break; 
  474.  
  475.       } 
  476.  
  477.       // Get our Data 
  478.       if(!is_null($typestocheck) && !$ignored_query) { 
  479.           $gpi_page_stats = $wpdb->prefix . 'gpi_page_stats'; 
  480.           if(isset($_GET['orderby'])) $orderby = $_GET['orderby']; 
  481.           if(isset($_GET['order'])) $order = $_GET['order']; 
  482.  
  483.           if(isset($orderby)) { 
  484.  
  485.               $data = $wpdb->get_results( 
  486.                           $wpdb->prepare( 
  487.                               "SELECT ID, URL, desktop_score, mobile_score, desktop_last_modified, mobile_last_modified, type 
  488.                               FROM $gpi_page_stats 
  489.                               WHERE ($typestocheck[0]) 
  490.                               AND $nullcheck 
  491.                               ORDER BY $orderby $order",  
  492.                               $typestocheck[1] 
  493.  ),  
  494.                           ARRAY_A  
  495.  ); 
  496.  
  497.           } else { 
  498.  
  499.               $data = $wpdb->get_results( 
  500.                           $wpdb->prepare( 
  501.                               "SELECT ID, URL, desktop_score, mobile_score, desktop_last_modified, mobile_last_modified, type 
  502.                               FROM $gpi_page_stats 
  503.                               WHERE ($typestocheck[0]) 
  504.                               AND $nullcheck",  
  505.                               $typestocheck[1] 
  506.  ),  
  507.                           ARRAY_A  
  508.  ); 
  509.  
  510.           } 
  511.            
  512.       } elseif($ignored_query) { 
  513.           $gpi_page_blacklist = $wpdb->prefix . 'gpi_page_blacklist'; 
  514.           if(isset($_GET['orderby'])) $orderby = $_GET['orderby']; 
  515.           if(isset($_GET['order'])) $order = $_GET['order']; 
  516.           if(isset($orderby)) { 
  517.  
  518.               $data = $wpdb->get_results(  
  519.                           "SELECT ID, URL, type 
  520.                           FROM $gpi_page_blacklist 
  521.                           ORDER BY $orderby $order",  
  522.                           ARRAY_A  
  523.  ); 
  524.  
  525.           } else { 
  526.  
  527.               $data = $wpdb->get_results( 
  528.                           "SELECT ID, URL, type 
  529.                           FROM $gpi_page_blacklist 
  530.                           ORDER BY ID DESC",  
  531.                           ARRAY_A 
  532.  ); 
  533.  
  534.           } 
  535.  
  536.       } else { 
  537.           $data = array(); 
  538.       } 
  539.     
  540.       // Figure out which page of results we are on 
  541.       $current_page = $this->get_pagenum(); 
  542.        
  543.       // Total number of results 
  544.       $total_items = count($data); 
  545.        
  546.       // Slice up our data for Pagination 
  547.       $data = array_slice($data, (($current_page-1)*$per_page), $per_page); 
  548.        
  549.       // Return sorted data to be used 
  550.       $this->items = $data; 
  551.        
  552.       // Register pagination 
  553.       $this->set_pagination_args( array( 
  554.           'total_items' => $total_items,  
  555.           'per_page' => $per_page,  
  556.           'total_pages' => ceil($total_items/$per_page) 
  557.  ) ); 
  558.   } 
  559.    
  560. }
