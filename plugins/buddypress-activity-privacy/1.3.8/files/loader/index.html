  75. function bp_activity_privacy_check_config() { 
  76.   global $bp; 
  77.  
  78.   $config = array( 
  79.       'blog_status' => false,  
  80.       'network_active' => false,  
  81.       'network_status' => true  
  82.  ); 
  83.   if ( get_current_blog_id() == bp_get_root_blog_id() ) { 
  84.       $config['blog_status'] = true; 
  85.   } 
  86.    
  87.   $network_plugins = get_site_option( 'active_sitewide_plugins', array() ); 
  88.  
  89.   // No Network plugins 
  90.   if ( empty( $network_plugins ) ) 
  91.  
  92.   // Looking for BuddyPress and bp-activity plugin 
  93.   $check[] = $bp->basename; 
  94.   $check[] = BP_ACTIVITY_PRIVACY_PLUGIN_BASENAME; 
  95.  
  96.   // Are they active on the network ? 
  97.   $network_active = array_diff( $check, array_keys( $network_plugins ) ); 
  98.    
  99.   // If result is 1, your plugin is network activated 
  100.   // and not BuddyPress or vice & versa. Config is not ok 
  101.   if ( count( $network_active ) == 1 ) 
  102.       $config['network_status'] = false; 
  103.  
  104.   // We need to know if the plugin is network activated to choose the right 
  105.   // notice ( admin or network_admin ) to display the warning message. 
  106.   $config['network_active'] = isset( $network_plugins[ BP_ACTIVITY_PRIVACY_PLUGIN_BASENAME ] ); 
  107.  
  108.   // if BuddyPress config is different than bp-activity plugin 
  109.   if ( !$config['blog_status'] || !$config['network_status'] ) { 
  110.  
  111.       $warnings = array(); 
  112.       if ( !bp_core_do_network_admin() && !$config['blog_status'] ) { 
  113.           $warnings[] = __( 'Buddypress Activity Privacy requires to be activated on the blog where BuddyPress is activated.', 'bp-activity-privacy' ); 
  114.       } 
  115.  
  116.       if ( bp_core_do_network_admin() && !$config['network_status'] ) { 
  117.           $warnings[] = __( 'Buddypress Activity Privacy and BuddyPress need to share the same network configuration.', 'bp-activity-privacy' ); 
  118.       } 
  119.  
  120.       if ( ! empty( $warnings ) ) : 
  121.       ?> 
  122.       <div id="message" class="error"> 
  123.           <?php foreach ( $warnings as $warning ) : ?> 
  124.               <p><?php echo esc_html( $warning ) ; ?></p> 
  125.           <?php endforeach ; ?> 
  126.       </div> 
  127.       <?php 
  128.       endif; 
  129.  
  130.       // Display a warning message in network admin or admin 
  131.       add_action( $config['network_active'] ? 'network_admin_notices' : 'admin_notices', $warning ); 
  132.        
  133.       return false; 
  134.   }  
  135.   return true; 
  136. }
