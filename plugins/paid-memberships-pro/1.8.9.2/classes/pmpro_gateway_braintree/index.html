  653.     function subscribe(&$order) 
  654.     { 
  655.         //create a code for the order 
  656.         if(empty($order->code)) 
  657.             $order->code = $order->getRandomCode(); 
  658.          
  659.         //set up customer 
  660.         $this->getCustomer($order); 
  661.         if(empty($this->customer) || !empty($order->error)) 
  662.             return false;    //error retrieving customer 
  663.                      
  664.         //figure out the amounts 
  665.         $amount = $order->PaymentAmount; 
  666.         $amount_tax = $order->getTaxForPrice($amount);             
  667.         $amount = round((float)$amount + (float)$amount_tax, 2); 
  668.  
  669.         /** 
  670.             There are two parts to the trial. Part 1 is simply the delay until the first payment 
  671.             since we are doing the first payment as a separate transaction. 
  672.             The second part is the actual "trial" set by the admin. 
  673.              
  674.             Stripe only supports Year or Month for billing periods, but we account for Days and Weeks just in case. 
  675.         */ 
  676.         //figure out the trial length (first payment handled by initial charge)             
  677.         if($order->BillingPeriod == "Year") 
  678.             $trial_period_days = $order->BillingFrequency * 365;    //annual 
  679.         elseif($order->BillingPeriod == "Day") 
  680.             $trial_period_days = $order->BillingFrequency * 1;        //daily 
  681.         elseif($order->BillingPeriod == "Week") 
  682.             $trial_period_days = $order->BillingFrequency * 7;        //weekly 
  683.         else 
  684.             $trial_period_days = $order->BillingFrequency * 30;    //assume monthly 
  685.              
  686.         //convert to a profile start date 
  687.         $order->ProfileStartDate = date("Y-m-d", strtotime("+ " . $trial_period_days . " Day", current_time("timestamp"))) . "T0:0:0"; 
  688.          
  689.         //filter the start date 
  690.         $order->ProfileStartDate = apply_filters("pmpro_profile_start_date", $order->ProfileStartDate, $order);             
  691.  
  692.         //convert back to days 
  693.         $trial_period_days = ceil(abs(strtotime(date("Y-m-d")) - strtotime($order->ProfileStartDate, current_time("timestamp"))) / 86400); 
  694.  
  695.         //now add the actual trial set by the site 
  696.         if(!empty($order->TrialBillingCycles))                         
  697.         { 
  698.             $trialOccurrences = (int)$order->TrialBillingCycles; 
  699.             if($order->BillingPeriod == "Year") 
  700.                 $trial_period_days = $trial_period_days + (365 * $order->BillingFrequency * $trialOccurrences);    //annual 
  701.             elseif($order->BillingPeriod == "Day") 
  702.                 $trial_period_days = $trial_period_days + (1 * $order->BillingFrequency * $trialOccurrences);        //daily 
  703.             elseif($order->BillingPeriod == "Week") 
  704.                 $trial_period_days = $trial_period_days + (7 * $order->BillingFrequency * $trialOccurrences);    //weekly 
  705.             else 
  706.                 $trial_period_days = $trial_period_days + (30 * $order->BillingFrequency * $trialOccurrences);    //assume monthly                 
  707.         }                     
  708.                                  
  709.         //subscribe to the plan 
  710.         try 
  711.         {                 
  712.             $details = array( 
  713.               'paymentMethodToken' => $this->customer->creditCards[0]->token,  
  714.               'planId' => 'pmpro_' . $order->membership_id,  
  715.               'price' => $amount                   
  716.  ); 
  717.              
  718.             if(!empty($trial_period_days)) 
  719.             { 
  720.                 $details['trialPeriod'] = true; 
  721.                 $details['trialDuration'] = $trial_period_days; 
  722.                 $details['trialDurationUnit'] = "day"; 
  723.             } 
  724.              
  725.             if(!empty($order->TotalBillingCycles)) 
  726.                 $details['numberOfBillingCycles'] = $order->TotalBillingCycles; 
  727.              
  728.             $result = Braintree_Subscription::create($details);                                 
  729.         } 
  730.         catch (Exception $e) 
  731.         {                 
  732.             $order->error = __("Error subscribing customer to plan with Braintree:", "pmpro") . " " . $e->getMessage(); 
  733.             //return error 
  734.             $order->shorterror = $order->error; 
  735.             return false; 
  736.         } 
  737.          
  738.         if($result->success) 
  739.         {             
  740.             //if we got this far, we're all good                         
  741.             $order->status = "success";         
  742.             $order->subscription_transaction_id = $result->subscription->id; 
  743.             return true; 
  744.         } 
  745.         else 
  746.         { 
  747.             $order->error = __("Failed to subscribe with Braintree:", "pmpro") . " " . $result->message; 
  748.             $order->shorterror = $result->message; 
  749.             return false; 
  750.         }     
  751.     }
