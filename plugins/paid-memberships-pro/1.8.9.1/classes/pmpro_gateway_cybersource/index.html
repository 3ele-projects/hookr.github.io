  562.  
  563.         /** 
  564.             There are two parts to the trial. Part 1 is simply the delay until the first payment 
  565.             since we are doing the first payment as a separate transaction. 
  566.             The second part is the actual "trial" set by the admin.                                 
  567.         */ 
  568.         //figure out the trial length (first payment handled by initial charge)             
  569.         if($order->BillingPeriod == "Year") 
  570.             $trial_period_days = $order->BillingFrequency * 365;    //annual 
  571.         elseif($order->BillingPeriod == "Day") 
  572.             $trial_period_days = $order->BillingFrequency * 1;        //daily 
  573.         elseif($order->BillingPeriod == "Week") 
  574.             $trial_period_days = $order->BillingFrequency * 7;        //weekly 
  575.         else 
  576.             $trial_period_days = $order->BillingFrequency * 30;    //assume monthly 
  577.              
  578.         //convert to a profile start date 
  579.         $order->ProfileStartDate = date("Y-m-d", strtotime("+ " . $trial_period_days . " Day", current_time("timestamp"))) . "T0:0:0"; 
  580.          
  581.         //filter the start date 
  582.         $order->ProfileStartDate = apply_filters("pmpro_profile_start_date", $order->ProfileStartDate, $order);             
  583.  
  584.         //convert back to days 
  585.         $trial_period_days = ceil(abs(strtotime(date("Y-m-d"), current_time('timestamp')) - strtotime($order->ProfileStartDate, current_time("timestamp"))) / 86400); 
  586.  
  587.         //now add the actual trial set by the site 
  588.         if(!empty($order->TrialBillingCycles))                         
  589.         { 
  590.             $trialOccurrences = (int)$order->TrialBillingCycles; 
  591.             if($order->BillingPeriod == "Year") 
  592.                 $trial_period_days = $trial_period_days + (365 * $order->BillingFrequency * $trialOccurrences);    //annual 
  593.             elseif($order->BillingPeriod == "Day") 
  594.                 $trial_period_days = $trial_period_days + (1 * $order->BillingFrequency * $trialOccurrences);        //daily 
  595.             elseif($order->BillingPeriod == "Week") 
  596.                 $trial_period_days = $trial_period_days + (7 * $order->BillingFrequency * $trialOccurrences);    //weekly 
  597.             else 
  598.                 $trial_period_days = $trial_period_days + (30 * $order->BillingFrequency * $trialOccurrences);    //assume monthly                 
  599.         } 
  600.          
  601.         //convert back into a date 
  602.         $profile_start_date = date("Ymd", strtotime("+ " . $trial_period_days . " Days")); 
  603.          
  604.         //figure out the frequency 
  605.         if($order->BillingPeriod == "Year") 
  606.         { 
  607.             $frequency = "annually";    //ignoring BillingFrequency set on level.                 
  608.         } 
  609.         elseif($order->BillingPeriod == "Month") 
  610.         { 
  611.             if($order->BillingFrequency == 6) 
  612.                 $frequency = "semi annually"; 
  613.             elseif($order->BillingFrequency == 3) 
  614.                 $frequency = "quarterly"; 
  615.             else 
  616.                 $frequency = "monthly"; 
  617.         } 
  618.         elseif($order->BillingPeriod == "Week") 
  619.         { 
  620.             if($order->BillingFrequency == 4) 
  621.                 $frequency = "quad-weekly"; 
  622.             elseif($order->BillingFrequency == 2) 
  623.                 $frequency = "bi-weekly"; 
  624.             else 
  625.                 $frequency = "weekly"; 
  626.         } 
  627.         elseif($order->BillingPeriod == "Day") 
  628.         { 
  629.             if($order->BillingFrequency == 365) 
  630.                 $frequency = "annually"; 
  631.             elseif($order->BillingFrequency == 182) 
  632.                 $frequency = "semi annually"; 
  633.             elseif($order->BillingFrequency == 183) 
  634.                 $frequency = "semi annually"; 
  635.             elseif($order->BillingFrequency == 90) 
  636.                 $frequency = "quaterly"; 
  637.             elseif($order->BillingFrequency == 30) 
  638.                 $frequency = "monthly"; 
  639.             elseif($order->BillingFrequency == 15) 
  640.                 $frequency = "semi-monthly"; 
  641.             elseif($order->BillingFrequency == 28) 
  642.                 $frequency = "quad-weekly"; 
  643.             elseif($order->BillingFrequency == 14) 
  644.                 $frequency = "bi-weekly"; 
  645.             elseif($order->BillingFrequency == 7) 
  646.                 $frequency = "weekly";                 
  647.         }             
  648.          
  649.         //set subscription info for API 
  650.         $subscription = new stdClass(); 
  651.         $subscription->title = $order->membership_level->name; 
  652.         $subscription->paymentMethod = "credit card"; 
  653.         $request->subscription = $subscription; 
  654.          
  655.         //recurring info             
  656.         $recurringSubscriptionInfo = new stdClass(); 
  657.         $recurringSubscriptionInfo->amount = number_format($amount, 2);             
  658.         $recurringSubscriptionInfo->startDate = $profile_start_date; 
  659.         $recurringSubscriptionInfo->frequency = $frequency; 
  660.         if(!empty($order->TotalBillingCycles)) 
  661.             $recurringSubscriptionInfo->numberOfPayments = $order->TotalBillingCycles;                 
  662.         $request->recurringSubscriptionInfo = $recurringSubscriptionInfo; 
  663.          
  664.         //combine address             
  665.         $address = $order->Address1; 
  666.         if(!empty($order->Address2)) 
  667.             $address .= "\n" . $order->Address2; 
  668.          
  669.         //bill to 
  670.         $billTo = new stdClass(); 
  671.         $billTo->firstName = $order->FirstName; 
  672.         $billTo->lastName = $order->LastName; 
  673.         $billTo->street1 = $address; 
  674.         $billTo->city = $order->billing->city; 
  675.         $billTo->state = $order->billing->state; 
  676.         $billTo->postalCode = $order->billing->zip; 
  677.         $billTo->country = $order->billing->country; 
  678.         $billTo->email = $order->Email; 
  679.         $billTo->ipAddress = $_SERVER['REMOTE_ADDR']; 
  680.         $request->billTo = $billTo; 
  681.          
  682.         //card 
  683.         $card = new stdClass(); 
  684.         $card->cardType = $this->getCardType($order->cardtype); 
  685.         $card->accountNumber = $order->accountnumber; 
  686.         $card->expirationMonth = $order->expirationmonth; 
  687.         $card->expirationYear = $order->expirationyear; 
  688.         $card->cvNumber = $order->CVV2; 
  689.         $request->card = $card; 
  690.  
  691.         //currency 
  692.         $purchaseTotals = new stdClass(); 
  693.         $purchaseTotals->currency = $pmpro_currency; 
  694.         $request->purchaseTotals = $purchaseTotals;             
  695.          
  696.         $soapClient = new CyberSourceSoapClient($wsdl_url, array("merchantID"=>pmpro_getOption("cybersource_merchantid"), "transactionKey"=>pmpro_getOption("cybersource_securitykey"))); 
  697.         $reply = $soapClient->runTransaction($request); 
  698.                      
  699.         if($reply->reasonCode == "100") 
  700.         { 
  701.             //success 
  702.             $order->subscription_transaction_id = $reply->requestID; 
  703.             $order->status = "success";                             
  704.             return true; 
  705.         } 
  706.         else 
  707.         { 
  708.             //error 
  709.             $order->status = "error"; 
  710.             $order->errorcode = $reply->reasonCode; 
  711.             $order->error = $this->getErrorFromCode($reply->reasonCode); 
  712.             $order->shorterror = $this->getErrorFromCode($reply->reasonCode); 
  713.             return false; 
  714.         }                         
  715.     }
