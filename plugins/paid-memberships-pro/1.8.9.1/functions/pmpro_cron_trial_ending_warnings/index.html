  1. function pmpro_cron_trial_ending_warnings() 
  2. { 
  3.     global $wpdb; 
  4.  
  5.     //make sure we only run once a day 
  6.     $today = date("Y-m-d 00:00:00", current_time("timestamp")); 
  7.  
  8.     $pmpro_email_days_before_trial_end = apply_filters("pmpro_email_days_before_trial_end", 7); 
  9.  
  10.     //look for memberships with trials ending soon (but we haven't emailed them within a week) 
  11.     $sqlQuery = " 
  12.     SELECT 
  13.         mu.user_id, mu.membership_id, mu.startdate, mu.cycle_period, mu.trial_limit FROM $wpdb->pmpro_memberships_users mu LEFT JOIN $wpdb->usermeta um ON um.user_id = mu.user_id AND um.meta_key = 'pmpro_trial_ending_notice' 
  14.     WHERE 
  15.         mu.status = 'active' AND mu.trial_limit IS NOT NULL AND mu.trial_limit > 0 AND 
  16.         ( 
  17.             (cycle_period = 'Day' AND DATE_ADD(mu.startdate, INTERVAL mu.trial_limit Day) <= DATE_ADD('" . $today . "', INTERVAL " . $pmpro_email_days_before_trial_end . " Day)) OR 
  18.             (cycle_period = 'Week' AND DATE_ADD(mu.startdate, INTERVAL mu.trial_limit Week) <= DATE_ADD('" . $today . "', INTERVAL " . $pmpro_email_days_before_trial_end . " Day)) OR 
  19.             (cycle_period = 'Month' AND DATE_ADD(mu.startdate, INTERVAL mu.trial_limit Month) <= DATE_ADD('" . $today . "', INTERVAL " . $pmpro_email_days_before_trial_end . " Day)) OR 
  20.             (cycle_period = 'Year' AND DATE_ADD(mu.startdate, INTERVAL mu.trial_limit Year) <= DATE_ADD('" . $today . "', INTERVAL " . $pmpro_email_days_before_trial_end . " Day)) 
  21.  ) 
  22.  
  23.         AND (um.meta_value IS NULL OR um.meta_value = '' OR DATE_ADD(um.meta_value, INTERVAL " . $pmpro_email_days_before_trial_end . " Day) <= '" . $today . "') 
  24.     ORDER BY mu.startdate"; 
  25.  
  26.     $trial_ending_soon = $wpdb->get_results($sqlQuery); 
  27.  
  28.     foreach($trial_ending_soon as $e) 
  29.     { 
  30.         $send_email = apply_filters("pmpro_send_trial_ending_email", true, $e->user_id); 
  31.         if($send_email) 
  32.         { 
  33.             //send an email 
  34.             $pmproemail = new PMProEmail(); 
  35.             $euser = get_userdata($e->user_id); 
  36.             $pmproemail->sendTrialEndingEmail($euser); 
  37.  
  38.             if(current_user_can('manage_options')) 
  39.                 printf(__("Trial ending email sent to %s. ", "pmpro"), $euser->user_email); 
  40.             else 
  41.                 echo ". "; 
  42.         } 
  43.  
  44.         //update user meta so we don't email them again 
  45.         update_user_meta($e->user_id, "pmpro_trial_ending_notice", $today); 
  46.     } 
  47. }
