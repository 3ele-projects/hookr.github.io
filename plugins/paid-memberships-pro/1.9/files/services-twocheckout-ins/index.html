  226.   function pmpro_twocheckoutValidate() { 
  227.       $params = array(); 
  228.       foreach ( $_REQUEST as $k => $v ) 
  229.           $params[$k] = $v; 
  230.  
  231.       //2Checkout uses an order number of 1 in the hash for demo orders for some reason 
  232.       if(!empty($params['demo']) && $params['demo'] == 'Y') 
  233.           $params['order_number'] = 1; 
  234.  
  235.       //is this a return call or notification 
  236.       if(empty($params['message_type'])) 
  237.           $check = Twocheckout_Return::check( $params, pmpro_getOption( 'twocheckout_secretword' ), 'array' ); 
  238.       else 
  239.           $check = Twocheckout_Notification::check( $params, pmpro_getOption( 'twocheckout_secretword' ), 'array' ); 
  240.  
  241.       if( empty ( $check ) ) 
  242.           $r = false;    //HTTP failure 
  243.       else if( empty ( $check['response_code'] ) ) 
  244.           $r = false;    //Invalid response 
  245.       else 
  246.           $r = $check['response_code']; 
  247.  
  248.       /** 
  249.        * Filter if an twocheckout request is valid or not. 
  250.        * 
  251.        * @since 1.8.6.3 
  252.        * 
  253.        * @param bool $r true or false if the request is valid 
  254.        * @param mixed $check remote post object from request to Twocheckout 
  255.        */ 
  256.       $r = apply_filters('pmpro_twocheckout_validate', $r, $check); 
  257.  
  258.       return $check['response_code'] === 'Success'; 
  259.   }
