  477.    
  478.   //build query 
  479.   global $wpdb; 
  480.  
  481.   $sqlQuery = "SELECT COUNT(DISTINCT user_id) FROM $wpdb->pmpro_memberships_users WHERE startdate >= '" . $startdate . "' "; 
  482.  
  483.   //restrict by level 
  484.   if(!empty($levels) && $levels != 'all') 
  485.       $sqlQuery .= "AND membership_id IN(" . $levels . ") "; 
  486.    
  487.   $signups = $wpdb->get_var($sqlQuery); 
  488.    
  489.   //save in cache 
  490.   if(!empty($cache) && !empty($cache[$period])) 
  491.       $cache[$period][$levels] = $signups; 
  492.   elseif(!empty($cache)) 
  493.       $cache[$period] = array($levels => $signups); 
  494.   else 
  495.       $cache = array($period => array($levels => $signups)); 
  496.    
  497.   set_transient("pmpro_report_memberships_signups", $cache, 3600*24);

  499.   return $signups;

  503. /** 
  504.  * get cancellations by status 
  505.  * 
  506.  * @param string $period - Either a string description ('today', 'this month', 'this year') 
  507.  * @param array(int)|string $levels - Either an array of level IDs or the string 'all' 
  508.  * @param array(string) $status - Array of statuses to fetch data for 
  509.  * @return null|int - The # of cancellations for the period specified 
  510.  */ 
  511. function pmpro_getCancellations($period = null, $levels = 'all', $status = array('inactive', 'expired', 'cancelled', 'admin_cancelled') ) 
  512. { 
  513.   //make sure status is an array 
  514.   if(!is_array($status)) 
  515.       $status = array($status); 
  516.  
  517.   //check for a transient 
  518.   $cache = get_transient( 'pmpro_report_memberships_cancellations' ); 
  519.   $hash = md5($period . $levels . implode(', ', $status)); 
  520.   if( ! empty( $cache ) && ! empty( $cache[$hash] ) ) 
  521.       return $cache[$hash]; 
  522.  
  523.   //figure out start date 
  524.   $now = current_time('timestamp'); 
  525.   $year = date_i18n("Y", $now); 
  526.  
  527.   if( $period == 'today' ) 
  528.   { 
  529.       $startdate = date_i18n('Y-m-d', $now) . " 00:00:00"; 
  530.       $enddate = "'" . date_i18n('Y-m-d', $now) . " 23:59:59'"; 
  531.   } 
  532.   elseif( $period == 'this month') 
  533.   { 
  534.       $startdate = date_i18n( 'Y-m', $now ) . '-01 00:00:00'; 
  535.       $enddate = "CONCAT(LAST_DAY('" . date_i18n( 'Y-m', $now ) . '-01' ."'), ' 23:59:59')"; 
  536.   } 
  537.   elseif( $period == 'this year') 
  538.   { 
  539.       $startdate = date_i18n( 'Y', $now ) . '-01-01 00:00:00'; 
  540.       $enddate = "'" . date_i18n( 'Y', $now ) . "-12-31 23:59:59'"; 
  541.   } 
  542.   else 
  543.   { 
  544.       //all time 
  545.       $startdate = '1970-01-01';    //all time (no point in using a value prior to the start of the UNIX epoch) 
  546.       $enddate = "'".strval(intval($year)+1) . "-01-01'"; 
  547.   } 
  548.        
  549.   /** 
  550.       build query. 
  551.       cancellations are marked in the memberships users table with status 'inactive', 'expired', 'cancelled', 'cancelled_admin' 
  552.       we try to ignore cancellations when the user gets a new level with 24 hours (probably an upgrade or downgrade) 
  553.   */ 
  554.   global $wpdb; 
  555.  
  556.       $sqlQuery = " 
  557.   SELECT COUNT( DISTINCT mu1.user_id ) 
  558.   FROM {$wpdb->pmpro_memberships_users} AS mu1         
  559.   WHERE mu1.status IN('" . implode("', '", $status) . "') 
  560.       AND mu1.enddate >= '" . $startdate . "' 
  561.       AND mu1.enddate <= " . $enddate . " 
  562.   "; 
  563.   
  564.   //restrict by level 
  565.   if(!empty($levels) && $levels != 'all') { 
  566.  
  567.       // the levels provided wasn't in array form 
  568.       if ( ! is_array($levels) ) { 
  569.  
  570.           $levels = array($levels); 
  571.       } 
  572.  
  573.       $sqlQuery .= "AND mu1.membership_id IN(" . implode(", ", $levels) . ") "; 
  574.   } 
  575.    
  576.   /** 
  577.    * Filter query to get cancellation numbers in signups vs cancellations detailed report. 
  578.    * 
  579.    * @since 1.8.8 
  580.    * 
  581.    * @param string $sqlQuery The current SQL 
  582.    * @param string $period Period for report. today, this month, this year, empty string for all time. 
  583.    * @param array(int) $levels Level IDs to include in report. 
  584.    * @param array(string) $status Statuses to include as cancelled. 
  585.    */ 
  586.   $sqlQuery = apply_filters('pmpro_reports_get_cancellations_sql', $sqlQuery, $period, $levels, $status); 
  587.    
  588.   $cancellations = $wpdb->get_var($sqlQuery); 
  589.    
  590.   //save in cache 
  591.   if(!empty($cache) && !empty($cache[$hash])) 
  592.       $cache[$hash] = $cancellations; 
  593.   elseif(!empty($cache)) 
  594.       $cache[$hash] = $cancellations; 
  595.   else 
  596.       $cache = array($hash => $cancellations); 
  597.    
  598.   set_transient("pmpro_report_memberships_cancellations", $cache, 3600*24); 
  599.    
  600.   return $cancellations; 
  601. }

  603. //get MRR 
  604. function pmpro_getMRR($period, $levels = 'all') 
  605. { 
  606.   //check for a transient 
  607.   //$cache = get_transient("pmpro_report_mrr"); 
  608.   if(!empty($cache) && !empty($cache[$period]) && !empty($cache[$period][$levels])) 
  609.       return $cache[$period][$levels];     
  610.        
  611.   //a sale is an order with status NOT IN refunded, review, token, error 
  612.   if($period == "this month") 
  613.       $startdate = date_i18n("Y-m") . "-01"; 
  614.   elseif($period == "this year") 
  615.       $startdate = date_i18n("Y") . "-01-01"; 
  616.   else 
  617.       $startdate = ""; 
  618.    
  619.   $gateway_environment = pmpro_getOption("gateway_environment"); 
  620.    
  621.   //build query 
  622.   global $wpdb; 
  623.   // Get total revenue 
  624.   $sqlQuery = "SELECT SUM(total) FROM $wpdb->pmpro_membership_orders WHERE status NOT IN('refunded', 'review', 'token', 'error') AND timestamp >= '" . $startdate . "' AND gateway_environment = '" . esc_sql($gateway_environment) . "' "; 
  625.  
  626.   //restrict by level 
  627.   if(!empty($levels) && $levels != 'all') { 
  628.       $sqlQuery .= "AND membership_id IN(" . $levels . ") ";     
  629.   } 
  630.    
  631.   $revenue = $wpdb->get_var($sqlQuery); 
  632.    
  633.   //when was the first order 
  634.   $first_order_timestamp = $wpdb->get_var("SELECT UNIX_TIMESTAMP(`timestamp`) FROM $wpdb->pmpro_membership_orders WHERE `timestamp` IS NOT NULL AND `timestamp` >  '0000-00-00 00:00:00' ORDER BY `timestamp` LIMIT 1"); 
  635.    
  636.   //if we don't have a timestamp, we can't do this 
  637.   if(empty($first_order_timestamp)) 
  638.       return false; 
  639.        
  640.   //how many months ago was the first order 
  641.   $months = $wpdb->get_var("SELECT PERIOD_DIFF('" . date_i18n("Ym") . "', '" . date_i18n("Ym", $first_order_timestamp) . "')"); 
  642.    
  643.   /** this works in PHP 5.3+ without using MySQL to get the diff 
  644.   $date1 = new DateTime(date_i18n("Y-m-d", $first_order_timestamp)); 
  645.   $date2 = new DateTime(date_i18n("Y-m-d")); 
  646.   $interval = $date1->diff($date2); 
  647.   $years = intval($interval->format('%y')); 
  648.   $months = $years*12 + intval($interval->format('%m')); 
  649.   */ 
  650.    
  651.   if($months > 0) 
  652.       $mrr = $revenue / $months; 
  653.   else 
  654.       $mrr = 0; 
  655.        
  656.   //save in cache 
  657.   if(!empty($cache) && !empty($cache[$period])) 
  658.       $cache[$period][$levels] = $mrr; 
  659.   elseif(!empty($cache)) 
  660.       $cache[$period] = array($levels => $mrr); 
  661.   else 
  662.       $cache = array($period => array($levels => $mrr)); 
  663.    
  664.   set_transient("pmpro_report_mrr", $cache, 3600*24); 
  665.    
  666.   return $mrr; 
  667. }

  669. //get Cancellation Rate 
  670. function pmpro_getCancellationRate($period, $levels = 'all', $status = NULL) 
  671. {     
  672.   //make sure status is an array 
  673.   if(!is_array($status)) 
  674.       $status = array($status); 
  675.  
  676.   //check for a transient 
  677.   $cache = get_transient("pmpro_report_cancellation_rate"); 
  678.   $hash = md5($period . $levels . implode('', $status)); 
  679.   if(!empty($cache) && !empty($cache[$hash])) 
  680.       return $cache[$hash]; 
  681.    
  682.   $signups = pmpro_getSignups($period, $levels); 
  683.   $cancellations = pmpro_getCancellations($period, $levels, $status); 
  684.    
  685.   if(empty($signups)) 
  686.       return false; 
  687.    
  688.   $rate = number_format(($cancellations / $signups)*100, 2); 
  689.    
  690.   //save in cache 
  691.   if(!empty($cache) && !empty($cache[$period])) 
  692.       $cache[$period][$levels] = $rate; 
  693.   elseif(!empty($cache)) 
  694.       $cache[$period] = array($levels => $rate); 
  695.   else 
  696.       $cache = array($period => array($levels => $rate)); 
  697.    
  698.   set_transient("pmpro_report_cancellation_rate", $cache, 3600*24); 
  699.  
  700.   return $rate; 
  701. }

  703. //get LTV 
  704. function pmpro_getLTV($period, $levels = 'all', $mrr = NULL, $signups = NULL, $cancellation_rate = NULL) 
  705. {     
  706.   if(empty($mrr)) 
  707.       $mrr = pmpro_getMRR($period, $levels); 
  708.   if(empty($signups)) 
  709.       $signups = pmpro_getSignups($period, $levels); 
  710.   if(empty($cancellation_rate)) 
  711.       $cancellation_rate = pmpro_getCancellationRate($period, $levels); 
  712.    
  713.   //average monthly spend 
  714.   if(empty($signups)) 
  715.       return false; 
  716.    
  717.   if($signups > 0) 
  718.       $ams = $mrr / $signups; 
  719.   else 
  720.       $ams = 0; 
  721.        
  722.   if($cancellation_rate > 0) 
  723.       $ltv = $ams * (1/$cancellation_rate); 
  724.   else 
  725.       $ltv = $ams; 
  726.  
  727.   return $ltv; 
  728. }
