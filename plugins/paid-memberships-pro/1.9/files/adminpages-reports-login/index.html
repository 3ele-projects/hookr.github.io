  248. function pmpro_report_login_wp_visits() 
  249. { 
  250.   //don't track admin 
  251.   if(is_admin()) 
  252.       return; 
  253.    
  254.   //only track logged in users 
  255.   if(!is_user_logged_in()) 
  256.       return; 
  257.    
  258.   //check for cookie 
  259.   if(!empty($_COOKIE['pmpro_visit'])) 
  260.       return; 
  261.    
  262.   $now = current_time('timestamp'); 
  263.    
  264.   //set cookie, then track 
  265.   setcookie("pmpro_visit", "1", NULL, COOKIEPATH, COOKIE_DOMAIN, false);     
  266.    
  267.   global $current_user; 
  268.   //track for user 
  269.   if(!empty($current_user->ID)) 
  270.   {         
  271.       $visits = $current_user->pmpro_visits;         
  272.       if(empty($visits)) 
  273.           $visits = array("last"=>"N/A", "thisdate"=>NULL, "month"=>0, "thismonth"=>NULL, "alltime"=>0); 
  274.            
  275.       //track logins for user 
  276.       $visits['last'] = date_i18n(get_option("date_format")); 
  277.       $visits['alltime']++; 
  278.       $thismonth = date_i18n("n", $now); 
  279.       if($thismonth == $visits['thismonth']) 
  280.           $visits['month']++; 
  281.       else 
  282.       { 
  283.           $visits['month'] = 1; 
  284.           $visits['thismonth'] = $thismonth; 
  285.       } 
  286.        
  287.       //update user data 
  288.       update_user_meta($current_user->ID, "pmpro_visits", $visits); 
  289.   } 
  290.        
  291.   //track for all 
  292.   $visits = get_option("pmpro_visits");     
  293.   if(empty($visits)) 
  294.       $visits = array("today"=>0, "thisdate"=>NULL, "month"=>0, "thismonth"=> NULL, "alltime"=>0); 
  295.    
  296.   $visits['alltime']++; 
  297.   $thisdate = date_i18n("Y-d-m", $now); 
  298.   if($thisdate == $visits['thisdate']) 
  299.       $visits['today']++; 
  300.   else 
  301.   { 
  302.       $visits['today'] = 1; 
  303.       $visits['thisdate'] = $thisdate; 
  304.   } 
  305.   if($thismonth == $visits['thismonth']) 
  306.       $visits['month']++; 
  307.   else 
  308.   { 
  309.       $visits['month'] = 1; 
  310.       $visits['thismonth'] = $thismonth; 
  311.   } 
  312.        
  313.   update_option("pmpro_visits", $visits);         
  314. }
