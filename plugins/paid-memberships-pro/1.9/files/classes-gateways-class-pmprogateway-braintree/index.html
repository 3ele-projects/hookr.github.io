  611.       /**

  615.           If not, it checks for a user_id on the order and searches for a customer id in the user meta. 
  616.           If a customer id is found, it checks for a customer through the Braintree API. 
  617.           If a customer is found and there is an AccountNumber on the order passed, it will update the customer. 
  618.           If no customer is found and there is an AccountNumber on the order passed, it will create a customer.

  778.       function subscribe(&$order) 
  779.       { 
  780.           if ( ! self::$is_loaded ) { 
  781.               $order->error = __("Payment error: Please contact the webmaster (braintree-load-error)", 'paid-memberships-pro'); 
  782.               return false; 
  783.           } 
  784.  
  785.           //create a code for the order 
  786.           if(empty($order->code)) 
  787.               $order->code = $order->getRandomCode(); 
  788.            
  789.           //set up customer 
  790.           $this->getCustomer($order); 
  791.           if(empty($this->customer) || !empty($order->error)) 
  792.               return false;    //error retrieving customer 
  793.                        
  794.           //figure out the amounts 
  795.           $amount = $order->PaymentAmount; 
  796.           $amount_tax = $order->getTaxForPrice($amount);             
  797.           $amount = round((float)$amount + (float)$amount_tax, 2); 
  798.  
  799.           /** 
  800.               There are two parts to the trial. Part 1 is simply the delay until the first payment 
  801.               since we are doing the first payment as a separate transaction. 
  802.               The second part is the actual "trial" set by the admin. 
  803.                
  804.               Braintree only supports Year or Month for billing periods, but we account for Days and Weeks just in case. 
  805.           */ 
  806.           //figure out the trial length (first payment handled by initial charge)             
  807.           if($order->BillingPeriod == "Year") 
  808.               $trial_period_days = $order->BillingFrequency * 365;    //annual 
  809.           elseif($order->BillingPeriod == "Day") 
  810.               $trial_period_days = $order->BillingFrequency * 1;        //daily 
  811.           elseif($order->BillingPeriod == "Week") 
  812.               $trial_period_days = $order->BillingFrequency * 7;        //weekly 
  813.           else 
  814.               $trial_period_days = $order->BillingFrequency * 30;    //assume monthly 
  815.                
  816.           //convert to a profile start date 
  817.           $order->ProfileStartDate = date_i18n("Y-m-d", strtotime("+ " . $trial_period_days . " Day", current_time("timestamp"))) . "T0:0:0"; 
  818.            
  819.           //filter the start date 
  820.           $order->ProfileStartDate = apply_filters("pmpro_profile_start_date", $order->ProfileStartDate, $order);             
  821.  
  822.           //convert back to days 
  823.           $trial_period_days = ceil(abs(strtotime(date_i18n("Y-m-d")) - strtotime($order->ProfileStartDate, current_time("timestamp"))) / 86400); 
  824.  
  825.           //now add the actual trial set by the site 
  826.           if(!empty($order->TrialBillingCycles))                         
  827.           { 
  828.               $trialOccurrences = (int)$order->TrialBillingCycles; 
  829.               if($order->BillingPeriod == "Year") 
  830.                   $trial_period_days = $trial_period_days + (365 * $order->BillingFrequency * $trialOccurrences);    //annual 
  831.               elseif($order->BillingPeriod == "Day") 
  832.                   $trial_period_days = $trial_period_days + (1 * $order->BillingFrequency * $trialOccurrences);        //daily 
  833.               elseif($order->BillingPeriod == "Week") 
  834.                   $trial_period_days = $trial_period_days + (7 * $order->BillingFrequency * $trialOccurrences);    //weekly 
  835.               else 
  836.                   $trial_period_days = $trial_period_days + (30 * $order->BillingFrequency * $trialOccurrences);    //assume monthly                 
  837.           }                     
  838.                                    
  839.           //subscribe to the plan 
  840.           try 
  841.           {                 
  842.               $details = array( 
  843.                 'paymentMethodToken' => $this->customer->creditCards[0]->token,  
  844.                 'planId' => 'pmpro_' . $order->membership_id,  
  845.                 'price' => $amount                   
  846.  ); 
  847.                
  848.               if(!empty($trial_period_days)) 
  849.               { 
  850.                   $details['trialPeriod'] = true; 
  851.                   $details['trialDuration'] = $trial_period_days; 
  852.                   $details['trialDurationUnit'] = "day"; 
  853.               } 
  854.                
  855.               if(!empty($order->TotalBillingCycles)) 
  856.                   $details['numberOfBillingCycles'] = $order->TotalBillingCycles; 
  857.                
  858.               $result = Braintree_Subscription::create($details);                                 
  859.           } 
  860.           catch (Exception $e) 
  861.           {                 
  862.               $order->error = __("Error subscribing customer to plan with Braintree:", 'paid-memberships-pro' ) . " " . $e->getMessage() . " (" . get_class($e) . ")"; 
  863.               //return error 
  864.               $order->shorterror = $order->error; 
  865.               return false; 
  866.           } 
  867.            
  868.           if($result->success) 
  869.           {             
  870.               //if we got this far, we're all good                         
  871.               $order->status = "success";         
  872.               $order->subscription_transaction_id = $result->subscription->id; 
  873.               return true; 
  874.           } 
  875.           else 
  876.           { 
  877.               $order->error = __("Failed to subscribe with Braintree:", 'paid-memberships-pro' ) . " " . $result->message; 
  878.               $order->shorterror = $result->message; 
  879.               return false; 
  880.           }     
  881.       }
