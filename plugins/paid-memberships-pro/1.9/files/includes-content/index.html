  265. function pmpro_membership_content_filter($content, $skipcheck = false) 
  266. {     
  267.   global $post, $current_user; 
  268.  
  269.   if(!$skipcheck) 
  270.   { 
  271.       $hasaccess = pmpro_has_membership_access(NULL, NULL, true); 
  272.       if(is_array($hasaccess)) 
  273.       { 
  274.           //returned an array to give us the membership level values 
  275.           $post_membership_levels_ids = $hasaccess[1]; 
  276.           $post_membership_levels_names = $hasaccess[2]; 
  277.           $hasaccess = $hasaccess[0]; 
  278.       } 
  279.   } 
  280.  
  281.   if($hasaccess) 
  282.   { 
  283.       //all good, return content 
  284.       return $content; 
  285.   } 
  286.   else 
  287.   { 
  288.       //if show excerpts is set, return just the excerpt 
  289.       if(pmpro_getOption("showexcerpts")) 
  290.       {             
  291.           //show excerpt 
  292.           global $post; 
  293.           if($post->post_excerpt) 
  294.           {                                 
  295.               //defined exerpt 
  296.               $content = wpautop($post->post_excerpt); 
  297.           } 
  298.           elseif(strpos($content, "<span id=\"more-" . $post->ID . "\"></span>") !== false) 
  299.           {                 
  300.               //more tag 
  301.               $pos = strpos($content, "<span id=\"more-" . $post->ID . "\"></span>"); 
  302.               $content = wpautop(substr($content, 0, $pos)); 
  303.           } 
  304.           elseif(strpos($content, 'class="more-link">') !== false) 
  305.           { 
  306.               //more link 
  307.               $content = preg_replace("/\<a.*class\=\"more\-link\".*\>.*\<\/a\>/", "", $content); 
  308.           } 
  309.           else 
  310.           { 
  311.               //auto generated excerpt. pulled from wp_trim_excerpt 
  312.               $content = strip_shortcodes( $content ); 
  313.               $content = str_replace(']]>', ']]>', $content); 
  314.               $content = strip_tags($content); 
  315.               $excerpt_length = apply_filters('excerpt_length', 55); 
  316.               $words = preg_split("/[\n\r\t ]+/", $content, $excerpt_length + 1, PREG_SPLIT_NO_EMPTY); 
  317.               if ( count($words) > $excerpt_length ) { 
  318.                   array_pop($words); 
  319.                   $content = implode(' ', $words); 
  320.                   $content = $content . "... "; 
  321.               } else { 
  322.                   $content = implode(' ', $words) . "... "; 
  323.               } 
  324.  
  325.               $content = wpautop($content); 
  326.           } 
  327.       } 
  328.       else 
  329.       { 
  330.           //else hide everything 
  331.           $content = ""; 
  332.       } 
  333.  
  334.       if(empty($post_membership_levels_ids)) 
  335.           $post_membership_levels_ids = array(); 
  336.  
  337.       if(empty($post_membership_levels_names)) 
  338.           $post_membership_levels_names = array(); 
  339.  
  340.       //hide levels which don't allow signups by default 
  341.       if(!apply_filters("pmpro_membership_content_filter_disallowed_levels", false, $post_membership_levels_ids, $post_membership_levels_names)) 
  342.       { 
  343.           foreach($post_membership_levels_ids as $key=>$id) 
  344.           { 
  345.               //does this level allow registrations? 
  346.               $level_obj = pmpro_getLevel($id); 
  347.               if(!$level_obj->allow_signups) 
  348.               { 
  349.                   unset($post_membership_levels_ids[$key]); 
  350.                   unset($post_membership_levels_names[$key]); 
  351.               } 
  352.           } 
  353.       } 
  354.  
  355.       $pmpro_content_message_pre = '<div class="pmpro_content_message">'; 
  356.       $pmpro_content_message_post = '</div>'; 
  357.  
  358.       $sr_search = array("!!levels!!", "!!referrer!!"); 
  359.       $sr_replace = array(pmpro_implodeToEnglish($post_membership_levels_names), urlencode(site_url($_SERVER['REQUEST_URI']))); 
  360.  
  361.       //get the correct message to show at the bottom 
  362.       if(is_feed()) 
  363.       { 
  364.           $newcontent = apply_filters("pmpro_rss_text_filter", stripslashes(pmpro_getOption("rsstext"))); 
  365.           $content .= $pmpro_content_message_pre . str_replace($sr_search, $sr_replace, $newcontent) . $pmpro_content_message_post; 
  366.       } 
  367.       elseif($current_user->ID) 
  368.       { 
  369.           //not a member
