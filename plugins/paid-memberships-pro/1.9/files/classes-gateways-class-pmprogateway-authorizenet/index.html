  989.       function send_request_via_fsockopen($host, $path, $content) 
  990.       { 
  991.           $posturl = "ssl://" . $host; 
  992.           $header = "Host: $host\r\n"; 
  993.           $header .= "User-Agent: PHP Script\r\n"; 
  994.           $header .= "Content-Type: text/xml\r\n"; 
  995.           $header .= "Content-Length: ".strlen($content)."\r\n"; 
  996.           $header .= "Connection: close\r\n\r\n"; 
  997.           $fp = fsockopen($posturl, 443, $errno, $errstr, 30); 
  998.           if (!$fp) 
  999.           { 
  1000.               $response = false; 
  1001.           } 
  1002.           else 
  1003.           { 
  1004.               error_reporting(E_ERROR); 
  1005.               fputs($fp, "POST $path  HTTP/1.1\r\n"); 
  1006.               fputs($fp, $header.$content); 
  1007.               fwrite($fp, $out); 
  1008.               $response = ""; 
  1009.               while (!feof($fp)) 
  1010.               { 
  1011.                   $response = $response . fgets($fp, 128); 
  1012.               } 
  1013.               fclose($fp); 
  1014.               error_reporting(E_ALL ^ E_NOTICE); 
  1015.           } 
  1016.           return $response; 
  1017.       }
