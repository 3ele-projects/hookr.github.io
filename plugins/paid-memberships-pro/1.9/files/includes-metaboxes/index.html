  49. function pmpro_page_save($post_id)

  52.  
  53.   if(empty($post_id))

  57.       return $post_id; 
  58.   } 
  59.  
  60.   // verify if this is an auto save routine. If it is our form has not been submitted, so we dont want 
  61.   // to do anything 
  62.   if ( defined('DOING_AUTOSAVE') && DOING_AUTOSAVE ) 
  63.       return $post_id; 
  64.  
  65.   // Check permissions

  67.   {

  69.           return $post_id; 
  70.   }

  74.           return $post_id; 
  75.   } 
  76.  
  77.   // OK, we're authenticated: we need to find and save the data     
  78.   if(isset($_POST['pmpro_noncename'])) 
  79.   { 
  80.       if(!empty($_POST['page_levels'])) 
  81.           $mydata = $_POST['page_levels']; 
  82.       else 
  83.           $mydata = NULL; 
  84.    
  85.       //remove all memberships for this page 
  86.       $wpdb->query("DELETE FROM {$wpdb->pmpro_memberships_pages} WHERE page_id = '$post_id'"); 
  87.

  89.       if(is_array($mydata)) 
  90.       {

  92.               $wpdb->query("INSERT INTO {$wpdb->pmpro_memberships_pages} (membership_id, page_id) VALUES('" . intval($level) . "', '" . intval($post_id) . "')"); 
  93.       } 
  94.    
  95.       return $mydata;
