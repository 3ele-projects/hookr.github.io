  257.       function sendToTwocheckout(&$order) 
  258.       {                         
  259.           global $pmpro_currency;             
  260.            
  261.           $tco_args = array( 
  262.               'sid' => pmpro_getOption("twocheckout_accountnumber"),  
  263.               'mode' => '2CO', // will always be 2CO according to docs (@see https://www.2checkout.com/documentation/checkout/parameter-sets/pass-through-products/) 
  264.               'li_0_type' => 'product',  
  265.               'li_0_name' => substr($order->membership_level->name . " at " . get_bloginfo("name"), 0, 127),  
  266.               'li_0_quantity' => 1,  
  267.               'li_0_tangible' => 'N',  
  268.               'li_0_product_id' => $order->code,  
  269.               'merchant_order_id' => $order->code,  
  270.               'currency_code' => $pmpro_currency,  
  271.               'pay_method' => 'CC',  
  272.               'purchase_step' => 'billing-information',  
  273.               'x_receipt_link_url' => admin_url("admin-ajax.php") . "?action=twocheckout-ins" //pmpro_url("confirmation", "?level=" . $order->membership_level->id) 
  274.  ); 
  275.  
  276.           //taxes on initial amount 
  277.           $initial_payment = $order->InitialPayment; 
  278.           $initial_payment_tax = $order->getTaxForPrice($initial_payment); 
  279.           $initial_payment = round((float)$initial_payment + (float)$initial_payment_tax, 2); 
  280.  
  281.           //taxes on the amount (NOT CURRENTLY USED) 
  282.           $amount = $order->PaymentAmount; 
  283.           $amount_tax = $order->getTaxForPrice($amount);             
  284.           $amount = round((float)$amount + (float)$amount_tax, 2);     
  285.            
  286.           // Recurring membership             
  287.           if( pmpro_isLevelRecurring( $order->membership_level ) ) { 
  288.               $tco_args['li_0_startup_fee'] = number_format($initial_payment - $amount, 2, ".", "");        //negative amount for lower initial payments 
  289.               $recurring_payment = number_format($order->membership_level->billing_amount, 2, ".", ""); 
  290.               $recurring_payment_tax = number_format($order->getTaxForPrice($recurring_payment), 2, ".", ""); 
  291.               $recurring_payment = number_format(round((float)$recurring_payment + (float)$recurring_payment_tax, 2), 2, ".", ""); 
  292.               $tco_args['li_0_price'] = number_format($recurring_payment, 2, ".", ""); 
  293.  
  294.               $tco_args['li_0_recurrence'] = ( $order->BillingFrequency == 1 ) ? $order->BillingFrequency . ' ' . $order->BillingPeriod : $order->BillingFrequency . ' ' . $order->BillingPeriod; 
  295.  
  296.               if( property_exists( $order, 'TotalBillingCycles' ) ) 
  297.                   $tco_args['li_0_duration'] = ($order->BillingFrequency * $order->TotalBillingCycles ) . ' ' . $order->BillingPeriod; 
  298.               else 
  299.                   $tco_args['li_0_duration'] = 'Forever'; 
  300.           } 
  301.           // Non-recurring membership 
  302.           else { 
  303.               $tco_args['li_0_price'] = number_format($initial_payment, 2, ".", ""); 
  304.           } 
  305.  
  306.           // Demo mode? 
  307.           if(empty($order->gateway_environment)) 
  308.               $gateway_environment = pmpro_getOption("gateway_environment"); 
  309.           else 
  310.               $gateway_environment = $order->gateway_environment; 
  311.           if("sandbox" === $gateway_environment || "beta-sandbox" === $gateway_environment) 
  312.           { 
  313.               Twocheckout::sandbox(true); 
  314.               $tco_args['demo'] = 'Y'; 
  315.           } 
  316.           else 
  317.               Twocheckout::sandbox(false); 
  318.            
  319.           // Trial? 
  320.           //li_#_startup_fee    Any start up fees for the product or service. Can be negative to provide discounted first installment pricing, but cannot equal or surpass the product price. 
  321.           if(!empty($order->TrialBillingPeriod)) { 
  322.               $trial_amount = $order->TrialAmount; 
  323.               $trial_tax = $order->getTaxForPrice($trial_amount); 
  324.               $trial_amount = pmpro_formatPrice(round((float)$trial_amount + (float)$trial_tax, 2), false, false); 
  325.               $tco_args['li_0_startup_fee'] = $trial_amount; // Negative trial amount 
  326.           }                 
  327.            
  328.           $ptpStr = ''; 
  329.           foreach( $tco_args as $key => $value ) { 
  330.               reset( $tco_args ); // Used to verify whether or not we're on the first argument 
  331.               $ptpStr .= ( $key == key($tco_args) ) ? '?' . $key . '=' . urlencode( $value ) : '&' . $key . '=' . urlencode( $value ); 
  332.           } 
  333.  
  334.           //anything modders might add 
  335.           $additional_parameters = apply_filters( 'pmpro_twocheckout_return_url_parameters', array() );                                     
  336.           if( ! empty( $additional_parameters ) ) 
  337.               foreach( $additional_parameters as $key => $value )                 
  338.                   $ptpStr .= "&" . urlencode($key) . "=" . urlencode($value); 
  339.  
  340.           $ptpStr = apply_filters( 'pmpro_twocheckout_ptpstr', $ptpStr, $order ); 
  341.                        
  342.           ///useful for debugging 
  343.           ///echo str_replace("&", "&<br />", $ptpStr); 
  344.           ///exit; 
  345.            
  346.           //figure out gateway environment and URL to use             
  347.           if($gateway_environment == "live") 
  348.                   $host = "www.2checkout.com";         
  349.               else 
  350.                   $host = "sandbox.2checkout.com";     
  351.           $tco_url = 'https://' . $host . '/checkout/purchase' . $ptpStr; 
  352.            
  353.           //redirect to 2checkout 
  354.           wp_redirect( $tco_url ); 
  355.           exit; 
  356.       }

  358.       function cancel(&$order) {

  360.           $order->updateStatus("cancelled"); 
  361.  
  362.           //require a payment transaction id

  365.  
  366.           //build api params 
  367.           $params = array(); 
  368.           $params['sale_id'] = $order->payment_transaction_id; 
  369.            
  370.           // Demo mode? 
  371.           if(empty($order->gateway_environment)) 
  372.               $gateway_environment = pmpro_getOption("gateway_environment"); 
  373.           else 
  374.               $gateway_environment = $order->gateway_environment; 
  375.            
  376.           if("sandbox" === $gateway_environment || "beta-sandbox" === $gateway_environment) 
  377.           { 
  378.               Twocheckout::sandbox(true); 
  379.               $params['demo'] = 'Y'; 
  380.           } 
  381.           else 
  382.               Twocheckout::sandbox(false); 
  383.  
  384.           $result = Twocheckout_Sale::stop( $params ); // Stop the recurring billing 
  385.  
  386.           // Successfully cancelled 
  387.           if (isset($result['response_code']) && $result['response_code'] === 'OK') { 
  388.               $order->updateStatus("cancelled");     
  389.               return true; 
  390.           } 
  391.           // Failed 
  392.           else { 
  393.               $order->status = "error"; 
  394.               $order->errorcode = $result->getCode(); 
  395.               $order->error = $result->getMessage(); 
  396.                                
  397.               return false;
