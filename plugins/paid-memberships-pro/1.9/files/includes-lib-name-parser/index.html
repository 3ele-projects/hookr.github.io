  8. // split full names into the following parts: 
  9. // - prefix / salutation  (Mr., Mrs., etc) 
  10. // - given name / first name 
  11. // - middle initials 
  12. // - surname / last name 
  13. // - suffix (II, Phd, Jr, etc) 
  14. if(!function_exists("pnp_split_full_name")) 
  15. { 
  16.   function pnp_split_full_name($full_name) { 
  17.       if(empty($full_name)) 
  18.           return ""; 
  19.            
  20.       $fname = $lname = $initials = NULL; 
  21.        
  22.       $full_name = trim($full_name); 
  23.       // split into words 
  24.       $unfiltered_name_parts = explode(" ", $full_name); 
  25.       $name_parts = array(); 
  26.       // completely ignore any words in parentheses 
  27.       foreach ($unfiltered_name_parts as $word) { 
  28.           if (!empty($word) && $word{0} != "(") 
  29.               $name_parts[] = $word; 
  30.       } 
  31.       $num_words = sizeof($name_parts); 
  32.  
  33.       // is the first word a title? (Mr. Mrs, etc) 
  34.       $salutation = pnp_is_salutation($name_parts[0]); 
  35.       $suffix = pnp_is_suffix($name_parts[sizeof($name_parts)-1]); 
  36.  
  37.       // set the range for the middle part of the name (trim prefixes & suffixes) 
  38.       $start = ($salutation) ? 1 : 0; 
  39.       $end = ($suffix) ? $num_words-1 : $num_words; 
  40.  
  41.       // concat the first name 
  42.       for ($i=$start; $i < $end-1; $i++) { 
  43.           $word = $name_parts[$i]; 
  44.           // move on to parsing the last name if we find an indicator of a compound last name (Von, Van, etc) 
  45.           // we use $i != $start to allow for rare cases where an indicator is actually the first name (like "Von Fabella") 
  46.           if (pnp_is_compound_lname($word) && $i != $start) 
  47.               break; 
  48.           // is it a middle initial or part of their first name? 
  49.           // if we start off with an initial, we'll call it the first name 
  50.           if (pnp_is_initial($word)) { 
  51.               // is the initial the first word?   
  52.               if ($i == $start) { 
  53.                   // if so, do a look-ahead to see if they go by their middle name 
  54.                   // for ex: "R. Jason Smith" => "Jason Smith" & "R." is stored as an initial 
  55.                   // but "R. J. Smith" => "R. Smith" and "J." is stored as an initial 
  56.                   if (pnp_is_initial($name_parts[$i+1])) 
  57.                       $fname .= " ".strtoupper($word); 
  58.                   else 
  59.                       $initials .= " ".strtoupper($word); 
  60.               // otherwise, just go ahead and save the initial 
  61.               } else { 
  62.                   $initials .= " ".strtoupper($word); 
  63.               } 
  64.           } else { 
  65.               $fname .= " ".pnp_fix_case($word); 
  66.           }   
  67.       } 
  68.  
  69.       // check that we have more than 1 word in our string 
  70.       if ($end-$start > 1) { 
  71.           // concat the last name 
  72.           for ($i; $i < $end; $i++) { 
  73.               $lname .= " ".pnp_fix_case($name_parts[$i]); 
  74.           } 
  75.       } else { 
  76.           // otherwise, single word strings are assumed to be first names 
  77.           $fname = pnp_fix_case($name_parts[$i]); 
  78.       } 
  79.  
  80.       // return the various parts in an array 
  81.       $name['salutation'] = $salutation; 
  82.       $name['fname'] = trim($fname); 
  83.       $name['initials'] = trim($initials); 
  84.       $name['lname'] = trim($lname); 
  85.       $name['suffix'] = $suffix; 
  86.       return $name; 
  87.   }

  89.   // detect and format standard salutations 
  90.   // I'm only considering english honorifics for now & not words like 
  91.   function pnp_is_salutation($word) { 
  92.       // ignore periods 
  93.       $word = str_replace('.', '', strtolower($word)); 
  94.       // returns normalized values 
  95.       if ($word == "mr" || $word == "master" || $word == "mister") 
  96.           return "Mr."; 
  97.       else if ($word == "mrs") 
  98.           return "Mrs."; 
  99.       else if ($word == "miss" || $word == "ms") 
  100.           return "Ms."; 
  101.       else if ($word == "dr") 
  102.           return "Dr."; 
  103.       else if ($word == "rev") 
  104.           return "Rev."; 
  105.       else if ($word == "fr") 
  106.           return "Fr."; 
  107.       else 
  108.           return false; 
  109.   } 
  110.  
  111.   //  detect and format common suffixes 
  112.   function pnp_is_suffix($word) { 
  113.       // ignore periods 
  114.       $word = str_replace('.', '', $word); 
  115.       // these are some common suffixes - what am I missing? 
  116.       $suffix_array = array('I', 'II', 'III', 'IV', 'V', 'Senior', 'Junior', 'Jr', 'Sr', 'PhD', 'APR', 'RPh', 'PE', 'MD', 'MA', 'DMD', 'CME'); 
  117.       foreach ($suffix_array as $suffix) { 
  118.           if (strtolower($suffix) == strtolower($word)) 
  119.               return $suffix; 
  120.       } 
  121.       return false; 
  122.   } 
  123.  
  124.   // detect compound last names like "Von Fange" 
  125.   function pnp_is_compound_lname($word) { 
  126.       $word = strtolower($word); 
  127.       // these are some common prefixes that identify a compound last names - what am I missing? 
  128.       $words = array('vere', 'von', 'van', 'de', 'del', 'della', 'di', 'da', 'pietro', 'vanden', 'du', 'st.', 'st', 'la', 'ter'); 
  129.       return array_search($word, $words); 
  130.   } 
  131.  
  132.   // single letter, possibly followed by a period 
  133.   function pnp_is_initial($word) { 
  134.       return ((strlen($word) == 1) || (strlen($word) == 2 && $word{1} == ".")); 
  135.   } 
  136.  
  137.   // detect mixed case words like "McDonald" 
  138.   // returns false if the string is all one case 
  139.   function pnp_is_camel_case($word) { 
  140.       if (preg_match("|[A-Z]+|s", $word) && preg_match("|[a-z]+|s", $word)) 
  141.           return true; 
  142.       return false; 
  143.   } 
  144.  
  145.   // ucfirst words split by dashes or periods 
  146.   // ucfirst all upper/lower strings, but leave camelcase words alone 
  147.   function pnp_fix_case($word) { 
  148.       // uppercase words split by dashes, like "Kimura-Fay" 
  149.       $word = pnp_safe_ucfirst("-", $word); 
  150.       // uppercase words split by periods, like "J.P." 
  151.       $word = pnp_safe_ucfirst(".", $word); 
  152.       return $word; 
  153.   } 
  154.  
  155.   // helper function for fix_case 
  156.   function pnp_safe_ucfirst($seperator, $word) { 
  157.       // uppercase words split by the seperator (ex. dashes or periods) 
  158.       $parts = explode($seperator, $word); 
  159.       foreach ($parts as $word) { 
  160.           $words[] = (pnp_is_camel_case($word)) ? $word : ucfirst(strtolower($word)); 
  161.       } 
  162.       return implode($seperator, $words); 
  163.   }
