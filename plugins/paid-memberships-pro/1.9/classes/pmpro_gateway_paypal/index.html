  631.     /** 
  632.      * PAYPAL Function 
  633.      * Send HTTP POST Request 
  634.      * 
  635.      * @param    string    The API method name 
  636.      * @param    string    The POST Message fields in &name=value pair format 
  637.      * @return    array    Parsed HTTP Response body 
  638.      */ 
  639.     function PPHttpPost($methodName_, $nvpStr_) { 
  640.         global $gateway_environment; 
  641.         $environment = $gateway_environment; 
  642.  
  643.         $API_UserName = pmpro_getOption("apiusername"); 
  644.         $API_Password = pmpro_getOption("apipassword"); 
  645.         $API_Signature = pmpro_getOption("apisignature"); 
  646.         $API_Endpoint = "https://api-3t.paypal.com/nvp"; 
  647.         if("sandbox" === $environment || "beta-sandbox" === $environment) { 
  648.             $API_Endpoint = "https://api-3t.$environment.paypal.com/nvp"; 
  649.         } 
  650.  
  651.         $version = urlencode('72.0'); 
  652.          
  653.         // NVPRequest for submitting to server 
  654.         $nvpreq = "METHOD=" . urlencode($methodName_) . "&VERSION=" . urlencode($version) . "&PWD=" . urlencode($API_Password) . "&USER=" . urlencode($API_UserName) . "&SIGNATURE=" . urlencode($API_Signature) . "&BUTTONSOURCE=" . urlencode(PAYPAL_BN_CODE) . $nvpStr_; 
  655.  
  656.         //post to PayPal 
  657.         $response = wp_remote_post( $API_Endpoint, array( 
  658.                 'timeout' => 60,  
  659.                 'sslverify' => FALSE,  
  660.                 'httpversion' => '1.1',  
  661.                 'body' => $nvpreq 
  662.  ) 
  663.  ); 
  664.  
  665.         if ( is_wp_error( $response ) ) { 
  666.            $error_message = $response->get_error_message(); 
  667.            die( "methodName_ failed: $error_message" ); 
  668.         } else { 
  669.             //extract the response details 
  670.             $httpParsedResponseAr = array(); 
  671.             parse_str(wp_remote_retrieve_body($response), $httpParsedResponseAr); 
  672.  
  673.             //check for valid response 
  674.             if((0 == sizeof($httpParsedResponseAr)) || !array_key_exists('ACK', $httpParsedResponseAr)) { 
  675.                 exit("Invalid HTTP Response for POST request($nvpreq) to $API_Endpoint."); 
  676.             } 
  677.         } 
  678.  
  679.         return $httpParsedResponseAr; 
  680.     }
