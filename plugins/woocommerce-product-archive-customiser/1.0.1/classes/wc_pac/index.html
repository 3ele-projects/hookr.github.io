  440.     /** 
  441.      * Sanitizes choices (selects / radios) 
  442.      * Checks that the input matches one of the available choices 
  443.      * 
  444.      * @param array $input the available choices. 
  445.      * @param array $setting the setting object. 
  446.      */ 
  447.     public function wc_pac_sanitize_choices( $input, $setting ) { 
  448.         // Ensure input is a slug. 
  449.         $input = sanitize_key( $input ); 
  450.  
  451.         // Get list of choices from the control associated with the setting. 
  452.         $choices = $setting->manager->get_control( $setting->id )->choices; 
  453.  
  454.         // If the input is a valid key, return it; otherwise, return the default. 
  455.         return ( array_key_exists( $input, $choices ) ? $input : $setting->default ); 
  456.     }
