  331.       function twocheckout_response() { 
  332.        
  333.           /** 
  334.            * source code: https://github.com/craigchristenson/woocommerce-2checkout-api 
  335.            * Thanks to: https://github.com/craigchristenson 
  336.            */ 
  337.           global $woocommerce; 
  338.            
  339.           @ob_clean(); 
  340.            
  341.           $wc_order_id = $_REQUEST['merchant_order_id']; 
  342.            
  343.           if ($this -> demo == 'yes' || (isset($_REQUEST['demo']) && $_REQUEST['demo'] == 'Y') ) { 
  344.               $compare_string = $this->secret_word . $this->seller_id . "1" . $_REQUEST['total']; 
  345.           }else{ 
  346.               $compare_string = $this->secret_word . $this->seller_id . $_REQUEST['order_number'] . $_REQUEST['total']; 
  347.           } 
  348.            
  349.           $compare_hash1 = strtoupper(md5($compare_string)); 
  350.            
  351.  
  352.           $compare_hash2 = $_REQUEST['key']; 
  353.           if ($compare_hash1 != $compare_hash2) { 
  354.               wp_die( "2Checkout Hash Mismatch... check your secret word." ); 
  355.           } else { 
  356.               $wc_order = new WC_Order( absint( $wc_order_id ) ); 
  357.               // Mark order complete 
  358.               $wc_order->payment_complete(); 
  359.               // Empty cart and clear session 
  360.               $woocommerce->cart->empty_cart(); 
  361.               wp_redirect( $this->get_return_url( $wc_order ) ); 
  362.               exit; 
  363.           } 
  364.       }
