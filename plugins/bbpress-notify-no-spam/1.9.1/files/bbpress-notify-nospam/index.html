  240.   function notify_new_topic( $topic_id = 0, $forum_id = 0 ) 
  241.   { 
  242.       global $wpdb; 
  243.  
  244.       $status = get_post_status( $topic_id );  
  245.  
  246.       if (   in_array( $status, (array) apply_filters( 'bbpnns_post_status_blacklist', array( 'spam' ), $status, $forum_id, $topic_id, $reply_id=false ) ) ||  
  247.            ! in_array( $status, (array) apply_filters( 'bbpnns_post_status_whitelist', array( 'publish' ), $status, $forum_id, $topic_id, $reply_id=false ) ) ) 
  248.           return -1; 
  249.  
  250.       if ( 0 === $forum_id ) 
  251.           $forum_id = bbp_get_topic_forum_id( $topic_id ); 
  252.  
  253.       if ( true === apply_filters( 'bbpnns_skip_topic_notification', false, $forum_id, $topic_id ) ) 
  254.           return -3; 
  255.  
  256.       $recipients = $this->get_recipients( $forum_id, 'topic' ); 
  257.  
  258.       /** 
  259.        * Allow topic recipients munging 
  260.        * @since 1.6.5 
  261.        */ 
  262.       $recipients = apply_filters( 'bbpress_topic_notify_recipients', $recipients, $topic_id, $forum_id ); 
  263.  
  264.       if ( empty( $recipients ) ) 
  265.           return -2; 
  266.  
  267.       list( $email_subject, $email_body ) = $this->_build_email( 'topic', $topic_id ); 
  268.  
  269.       return $this->send_notification( $recipients, $email_subject, $email_body ); 
  270.   }

  359.   private function _build_email( $type, $post_id ) 
  360.   { 
  361.       $email_subject = get_option( "bbpress_notify_new{$type}_email_subject" );

  365.       $excerpt_size = apply_filters( 'bpnns_excerpt_size', 100 ); 
  366.        
  367.       // Replace shortcodes 
  368.       if ( 'topic' === $type ) 
  369.       { 
  370.           $content = bbp_get_topic_content( $post_id ); 
  371.           $title = html_entity_decode( strip_tags( bbp_get_topic_title( $post_id ) ), ENT_NOQUOTES, 'UTF-8' );

  373.           $author = bbp_get_topic_author( $post_id ); 
  374.           $url = apply_filters( 'bbpnns_topic_url', bbp_get_topic_permalink( $post_id ), $post_id, $title ); 
  375.           $forum = html_entity_decode( strip_tags( get_the_title( bbp_get_topic_forum_id( $post_id ) ) ), ENT_NOQUOTES, 'UTF-8' ); 
  376.       } 
  377.       elseif ( 'reply' === $type ) 
  378.       { 
  379.           $content = bbp_get_reply_content( $post_id ); 
  380.           $title = html_entity_decode( strip_tags( bbp_get_reply_title( $post_id ) ), ENT_NOQUOTES, 'UTF-8' );

  401.       $email_subject = str_replace( "[$type-author]", $author, $email_subject ); 
  402.       $email_subject = str_replace( "[$type-url]", $url, $email_subject ); 
  403.       $email_subject = str_replace( "[$type-replyurl]", $topic_reply, $email_subject ); 
  404.       $email_subject = str_replace( "[$type-forum]", $forum, $email_subject ); 
  405.        
  406.       $email_body = str_replace( '[blogname]', $blogname, $email_body ); 
  407.       $email_body = str_replace( "[$type-title]", $title, $email_body ); 
  408.       $email_body = str_replace( "[$type-content]", $content, $email_body ); 
  409.       $email_body = str_replace( "[$type-excerpt]", $excerpt, $email_body ); 
  410.       $email_body = str_replace( "[$type-author]", $author, $email_body ); 
  411.       $email_body = str_replace( "[$type-url]", $url, $email_body ); 
  412.       $email_body = str_replace( "[$type-replyurl]", $topic_reply, $email_body ); 
  413.       $email_body = str_replace( "[$type-forum]", $forum, $email_body ); 
  414.        
  415.       /** 
  416.        * Allow subject and body modifications 
  417.        * @since 1.6.6 
  418.        */ 
  419.       $email_subject = apply_filters( 'bbpnns_filter_email_subject_in_build', $email_subject ); 
  420.       $email_body = apply_filters( 'bbpnns_filter_email_body_in_build', $email_body ); 
  421.        
  422.       return array( $email_subject, $email_body ); 
  423.   }

  428.   public function send_notification( $recipients, $subject, $body ) 
  429.   { 
  430.       $headers = array( sprintf( "From: %s <%s>", get_option( 'blogname' ), get_bloginfo( 'admin_email' ) ) ); 
  431.       $headers = apply_filters( 'bbpnns_extra_headers', $headers, $recipients, $subject, $body ); 
  432.  
  433.       // Allow Management of recipients list 
  434.       $recipients = apply_filters( 'bbpnns_filter_recipients_before_send', $recipients ); 
  435.  
  436.       foreach ( (array) $recipients as $recipient_id => $user_info ) 
  437.       { 
  438.           /** 
  439.            * Allow per user subject and body modifications 
  440.            * @since 1.6.4  
  441.            */  
  442.           // This is a workaround for cases where UTF-8 characters were blocking the message. 
  443.           $enc = mb_internal_encoding(); 
  444.            
  445.           $email = ( $recipient_id == -1 ) ? get_bloginfo( 'admin_email' ) : (string) $user_info->user_email ; 
  446.           $email = apply_filters( 'bbpnns_skip_notification', $email ); // Allow user to be skipped for some reason 
  447.  
  448.           if ( ! empty( $email ) && false === apply_filters( 'bbpnns_dry_run', false ) ) 
  449.           { 
  450.               /** 
  451.                * Allow per user subject and body modifications 
  452.                * @since 1.6.4 
  453.                */ 
  454.               $filtered_body = apply_filters( 'bbpnns_filter_email_body_for_user', $body, $user_info ); 
  455.               $filtered_subject = apply_filters( 'bbpnns_filter_email_subject_for_user', $subject, $user_info ); 
  456.                
  457.               /** 
  458.                * Enable UTF-8 characters in subject line 
  459.                * @since 1.9 
  460.                */ 
  461.               $preferences = apply_filters( 'bbpnns_subject_enc_preferences',  
  462.                       array( 'input-charset' => $enc, 'output-charset' => "UTF-8", 'scheme' => 'Q' ) 
  463.  ); 
  464.                    
  465.               $filtered_subject = iconv_mime_encode( 'Subject', $filtered_subject, $preferences ); 
  466.               $filtered_subject = substr( $filtered_subject, strlen( 'Subject:' ) ); 
  467.                
  468.               /** 
  469.                * User headers, if any 
  470.                */ 
  471.               $recipient_headers = apply_filters( 'bbpnns_extra_headers_recipient', $headers, $user_info, $filtered_subject, $filtered_body ); 
  472.                
  473.               do_action( 'bbpnns_before_wp_mail', $user_info, $filtered_subject, $filtered_body, $recipient_headers ); 
  474.                
  475.               // Turn on nl2br for wpMandrill 
  476.               add_filter( 'mandrill_nl2br', array( $this, 'handle_mandrill_nl2br' ), 10, 2 ); 
  477.                
  478.               if ( ! wp_mail( $email, $filtered_subject, $filtered_body, $recipient_headers ) ) 
  479.               { 
  480.                   do_action( 'bbpnns_email_failed_single_user', $user_info, $filtered_subject, $filtered_body, $recipient_headers ); 
  481.                   do_action( 'bbpnns_after_wp_mail', $user_info, $filtered_subject, $filtered_body, $recipient_headers ); 
  482.                    
  483.                   // Turn off nl2br for wpMandrill 
  484.                   remove_filter( 'mandrill_nl2br', array( $this, 'handle_mandrill_nl2br' ), 10 ); 
  485.                   continue; 
  486.               } 
  487.                
  488.               do_action( 'bbpnns_after_wp_mail', $user_info, $filtered_subject, $filtered_body, $recipient_headers ); 
  489.                
  490.               // Turn off nl2br for wpMandrill 
  491.               remove_filter( 'mandrill_nl2br', array( $this, 'handle_mandrill_nl2br' ), 10 ); 
  492.                
  493.               do_action( 'bbpnns_after_email_sent_single_user', $user_info, $filtered_subject, $filtered_body ); 
  494.           } 
  495.       } 
  496.        
  497.       do_action( 'bbpnns_after_email_sent_all_users', $recipients, $filtered_subject, $filtered_body ); 
  498.        
  499.       if ( true === apply_filters( 'bbpnns_dry_run', false ) ) 
  500.           return array( $recipients, $body ); 
  501.        
  502.       return true; 
  503.   }

  799.   /** 
  800.    * Sends notifications when user saves/publishes a post. Note that the send notification checkbox must be ticked. 
  801.    * @param int $post_id 
  802.    * @param object $post 
  803.    * @return array 
  804.    */ 
  805.   function notify_on_save( $post_id, $post ) 
  806.   { 
  807.       if ( empty( $_POST ) ) return; 
  808.  
  809.       if ( $this->bbpress_topic_post_type !== $post->post_type && $this->bbpress_reply_post_type !== $post->post_type ) return; 
  810.        
  811.       if ( ! current_user_can( 'manage_options' ) && ! current_user_can( 'edit_post', $post_id ) ) return; 
  812.        
  813.       if ( wp_is_post_revision( $post_id ) ) return; 
  814.        
  815.       if ( ! isset( $_POST['bbpress_notify_send_notification'] ) || ! $_POST['bbpress_notify_send_notification'] ) return; 
  816.  
  817.       $type = ( $post->post_type === $this->bbpress_topic_post_type ) ? 'topic' : 'reply'; 
  818.       if ( ! isset( $_POST["bbpress_send_{$type}_notification_nonce"] ) || 
  819.           ! wp_verify_nonce( $_POST["bbpress_send_{$type}_notification_nonce"], "bbpress_send_{$type}_notification_nonce" ) ) 
  820.       { 
  821.           return; 
  822.       } 
  823.  
  824.       // Still here, so we can notify 
  825.       if ( $post->post_type === $this->bbpress_topic_post_type ) 
  826.       { 
  827.           return $this->notify_new_topic( $post_id ); 
  828.       } 
  829.       else  
  830.       { 
  831.           return $this->notify_new_reply( $post_id ); 
  832.       } 
  833.   }
