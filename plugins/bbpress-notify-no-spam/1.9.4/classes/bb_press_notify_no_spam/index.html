  467.                      * Enable UTF-8 characters in subject line 
  468.                      * @since 1.9 
  469.                      */ 
  470.                     $filtered_subject = iconv_mime_encode( 'Subject', $filtered_subject, $preferences ); 
  471.                     $filtered_subject = substr( $filtered_subject, strlen( 'Subject:' ) ); 
  472.                 } 
  473.                  
  474.                 /** 
  475.                  * User headers, if any 
  476.                  */ 
  477.                 $recipient_headers = apply_filters( 'bbpnns_extra_headers_recipient', $headers, $user_info, $filtered_subject, $filtered_body ); 
  478.                  
  479.                 do_action( 'bbpnns_before_wp_mail', $user_info, $filtered_subject, $filtered_body, $recipient_headers ); 
  480.                  
  481.                 // Turn on nl2br for wpMandrill 
  482.                 add_filter( 'mandrill_nl2br', array( $this, 'handle_mandrill_nl2br' ), 10, 2 ); 
  483.                  
  484.                 if ( ! wp_mail( $email, $filtered_subject, $filtered_body, $recipient_headers ) ) 
  485.                 { 
  486.                     do_action( 'bbpnns_email_failed_single_user', $user_info, $filtered_subject, $filtered_body, $recipient_headers ); 
  487.                     do_action( 'bbpnns_after_wp_mail', $user_info, $filtered_subject, $filtered_body, $recipient_headers ); 
  488.                      
  489.                     // Turn off nl2br for wpMandrill 
  490.                     remove_filter( 'mandrill_nl2br', array( $this, 'handle_mandrill_nl2br' ), 10 ); 
  491.                     continue; 
  492.                 } 
  493.                  
  494.                 do_action( 'bbpnns_after_wp_mail', $user_info, $filtered_subject, $filtered_body, $recipient_headers ); 
  495.                  
  496.                 // Turn off nl2br for wpMandrill 
  497.                 remove_filter( 'mandrill_nl2br', array( $this, 'handle_mandrill_nl2br' ), 10 ); 
  498.                  
  499.                 do_action( 'bbpnns_after_email_sent_single_user', $user_info, $filtered_subject, $filtered_body ); 
  500.             } 
  501.         } 
  502.          
  503.         do_action( 'bbpnns_after_email_sent_all_users', $recipients, $filtered_subject, $filtered_body ); 
  504.          
  505.         if ( true === apply_filters( 'bbpnns_dry_run', false ) ) 
  506.             return array( $recipients, $body ); 
  507.          
  508.         return true; 
  509.     }

  820.     /** 
  821.      * Sends notifications when user saves/publishes a post. Note that the send notification checkbox must be ticked. 
  822.      * @param int $post_id 
  823.      * @param object $post 
  824.      * @return array 
  825.      */ 
  826.     function notify_on_save( $post_id, $post ) 
  827.     { 
  828.         if ( empty( $_POST ) ) return; 
  829.  
  830.         if ( $this->bbpress_topic_post_type !== $post->post_type && $this->bbpress_reply_post_type !== $post->post_type ) return; 
  831.          
  832.         if ( ! current_user_can( 'manage_options' ) && ! current_user_can( 'edit_post', $post_id ) ) return; 
  833.          
  834.         if ( wp_is_post_revision( $post_id ) ) return; 
  835.          
  836.         if ( ! isset( $_POST['bbpress_notify_send_notification'] ) || ! $_POST['bbpress_notify_send_notification'] ) return; 
  837.  
  838.         $type = ( $post->post_type === $this->bbpress_topic_post_type ) ? 'topic' : 'reply'; 
  839.         if ( ! isset( $_POST["bbpress_send_{$type}_notification_nonce"] ) || 
  840.             ! wp_verify_nonce( $_POST["bbpress_send_{$type}_notification_nonce"], "bbpress_send_{$type}_notification_nonce" ) ) 
  841.         { 
  842.             return; 
  843.         } 
  844.  
  845.         // Still here, so we can notify 
  846.         if ( $post->post_type === $this->bbpress_topic_post_type ) 
  847.         { 
  848.             return $this->notify_new_topic( $post_id ); 
  849.         } 
  850.         else  
  851.         { 
  852.             return $this->notify_new_reply( $post_id ); 
  853.         } 
  854.     }
