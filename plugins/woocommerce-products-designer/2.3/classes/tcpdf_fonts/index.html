  694.                     }

  1742.  * Based on: http://www.faqs.org/rfcs/rfc3629.html 
  1743.  * <pre> 
  1744.  *    Char. number range  |        UTF-8 octet sequence 
  1745.  *       (hexadecimal)    |              (binary) 
  1746.  *    --------------------+----------------------------------------------- 
  1747.  *    0000 0000-0000 007F | 0xxxxxxx 
  1748.  *    0000 0080-0000 07FF | 110xxxxx 10xxxxxx 
  1749.  *    0000 0800-0000 FFFF | 1110xxxx 10xxxxxx 10xxxxxx 
  1750.  *    0001 0000-0010 FFFF | 11110xxx 10xxxxxx 10xxxxxx 10xxxxxx 
  1751.  *    --------------------------------------------------------------------- 
  1752.  * 
  1753.  *   ABFN notation: 
  1754.  *   --------------------------------------------------------------------- 
  1755.  *   UTF8-octets = *( UTF8-char ) 
  1756.  *   UTF8-char = UTF8-1 / UTF8-2 / UTF8-3 / UTF8-4 
  1757.  *   UTF8-1 = %x00-7F 
  1758.  *   UTF8-2 = %xC2-DF UTF8-tail 
  1759.  * 
  1760.  *   UTF8-3 = %xE0 %xA0-BF UTF8-tail / %xE1-EC 2( UTF8-tail ) / 
  1761.  *                 %xED %x80-9F UTF8-tail / %xEE-EF 2( UTF8-tail ) 
  1762.  *   UTF8-4 = %xF0 %x90-BF 2( UTF8-tail ) / %xF1-F3 3( UTF8-tail ) / 
  1763.  *                 %xF4 %x80-8F 2( UTF8-tail ) 
  1764.  *   UTF8-tail = %x80-BF 
  1765.  *   --------------------------------------------------------------------- 
  1766.  * </pre> 
  1767.  * @param $uch (string) character string to process. 
  1768.  * @return integer Unicode value 
  1769.  * @author Nicola Asuni 
  1770.  * @public static 
  1771.  */ 
  1772. public static function getUniord($uch) { 
  1773.     if (function_exists('mb_convert_encoding')) { 
  1774.         list(, $char) = @unpack('N', mb_convert_encoding($uch, 'UCS-4BE', 'UTF-8')); 
  1775.         if ($char >= 0) { 
  1776.             return $char; 
  1777.         } 
  1778.     } 
  1779.     $bytes = array(); // array containing single character byte sequences 
  1780.     $countbytes = 0; 
  1781.     $numbytes = 1; // number of octetc needed to represent the UTF-8 character 
  1782.     $length = strlen($uch); 
  1783.     for ($i = 0; $i < $length; ++$i) { 
  1784.         $char = ord($uch[$i]); // get one string character at time 
  1785.         if ($countbytes == 0) { // get starting octect 
  1786.             if ($char <= 0x7F) { 
  1787.                 return $char; // use the character "as is" because is ASCII 
  1788.             } elseif (($char >> 0x05) == 0x06) { // 2 bytes character (0x06 = 110 BIN) 
  1789.                 $bytes[] = ($char - 0xC0) << 0x06; 
  1790.                 ++$countbytes; 
  1791.                 $numbytes = 2; 
  1792.             } elseif (($char >> 0x04) == 0x0E) { // 3 bytes character (0x0E = 1110 BIN) 
  1793.                 $bytes[] = ($char - 0xE0) << 0x0C; 
  1794.                 ++$countbytes; 
  1795.                 $numbytes = 3; 
  1796.             } elseif (($char >> 0x03) == 0x1E) { // 4 bytes character (0x1E = 11110 BIN) 
  1797.                 $bytes[] = ($char - 0xF0) << 0x12; 
  1798.                 ++$countbytes; 
  1799.                 $numbytes = 4; 
  1800.             } else { 
  1801.                 // use replacement character for other invalid sequences 
  1802.                 return 0xFFFD; 
  1803.             } 
  1804.         } elseif (($char >> 0x06) == 0x02) { // bytes 2, 3 and 4 must start with 0x02 = 10 BIN 
  1805.             $bytes[] = $char - 0x80; 
  1806.             ++$countbytes; 
  1807.             if ($countbytes == $numbytes) { 
  1808.                 // compose UTF-8 bytes to a single unicode value 
  1809.                 $char = $bytes[0]; 
  1810.                 for ($j = 1; $j < $numbytes; ++$j) { 
  1811.                     $char += ($bytes[$j] << (($numbytes - $j - 1) * 0x06)); 
  1812.                 } 
  1813.                 if ((($char >= 0xD800) AND ($char <= 0xDFFF)) OR ($char >= 0x10FFFF)) { 
  1814.                     // The definition of UTF-8 prohibits encoding character numbers between 
  1815.                     // U+D800 and U+DFFF, which are reserved for use with the UTF-16 
  1816.                     // encoding form (as surrogate pairs) and do not directly represent 
  1817.                     // characters. 
  1818.                     return 0xFFFD; // use replacement character 
  1819.                 } else { 
  1820.                     return $char; 
  1821.                 } 
  1822.             } 
  1823.         } else { 
  1824.             // use replacement character for other invalid sequences 
  1825.             return 0xFFFD; 
  1826.         } 
  1827.     } 
  1828.     return 0xFFFD; 
  1829. }

  872. /**

  894. /**

  905.     $offset = 0; // offset position of the font data 
  906.     if (TCPDF_STATIC::_getULONG($font, $offset) != 0x10000) {

  915.     $offset += 6;

  924.         $table[$tag]['checkSum'] = TCPDF_STATIC::_getULONG($font, $offset); 
  925.         $offset += 4; 
  926.         $table[$tag]['offset'] = TCPDF_STATIC::_getULONG($font, $offset); 
  927.         $offset += 4; 
  928.         $table[$tag]['length'] = TCPDF_STATIC::_getULONG($font, $offset); 
  929.         $offset += 4; 
  930.     } 
  931.     // check magicNumber 
  932.     $offset = $table['head']['offset'] + 12;

  934.         // magicNumber must be 0x5F0F3CF5 
  935.         return $font; 
  936.     }

  938.     // get offset mode (indexToLocFormat : 0 = short, 1 = long)

  940.     $short_offset = (TCPDF_STATIC::_getSHORT($font, $offset) == 0); 
  941.     $offset += 2; 
  942.     // get the offsets to the locations of the glyphs in the font, relative to the beginning of the glyphData table

  946.         // short version 
  947.         $tot_num_glyphs = floor($table['loca']['length'] / 2); // numGlyphs + 1 
  948.         for ($i = 0; $i < $tot_num_glyphs; ++$i) { 
  949.             $indexToLoc[$i] = TCPDF_STATIC::_getUSHORT($font, $offset) * 2; 
  950.             $offset += 2;

  954.         $tot_num_glyphs = ($table['loca']['length'] / 4); // numGlyphs + 1 
  955.         for ($i = 0; $i < $tot_num_glyphs; ++$i) { 
  956.             $indexToLoc[$i] = TCPDF_STATIC::_getULONG($font, $offset); 
  957.             $offset += 4; 
  958.         } 
  959.     }

  978.         $format = TCPDF_STATIC::_getUSHORT($font, $offset); 
  979.         $offset += 2;

  981.             case 0: { // Format 0: Byte encoding table 
  982.                 $offset += 4; // skip length and version/language 
  983.                 for ($c = 0; $c < 256; ++$c) { 
  984.                     if (isset($subsetchars[$c])) {

  1019.                     $numGlyphIndexArray += $subHeaders[$k]['entryCount'];

  1027.                     if ($k == 0) { 
  1028.                         // one byte code 
  1029.                         $c = $i; 
  1030.                         if (isset($subsetchars[$c])) { 
  1031.                             $g = $glyphIndexArray[0]; 
  1032.                             $subsetglyphs[$g] = true; 
  1033.                         } 
  1034.                     } else { 
  1035.                         // two bytes code 
  1036.                         $start_byte = $subHeaders[$k]['firstCode']; 
  1037.                         $end_byte = $start_byte + $subHeaders[$k]['entryCount']; 
  1038.                         for ($j = $start_byte; $j < $end_byte; ++$j) { 
  1039.                             // combine high and low bytes 
  1040.                             $c = (($i << 8) + $j); 
  1041.                             if (isset($subsetchars[$c])) { 
  1042.                                 $idRangeOffset = ($subHeaders[$k]['idRangeOffset'] + $j - $subHeaders[$k]['firstCode']);

  1045.                                     $g = 0; 
  1046.                                 } 
  1047.                                 $subsetglyphs[$g] = true; 
  1048.                             } 
  1049.                         } 
  1050.                     } 
  1051.                 } 
  1052.                 break; 
  1053.             } 
  1054.             case 4: { // Format 4: Segment mapping to delta values 
  1055.                 $length = TCPDF_STATIC::_getUSHORT($font, $offset); 
  1056.                 $offset += 2; 
  1057.                 $offset += 2; // skip version/language 
  1058.                 $segCount = floor(TCPDF_STATIC::_getUSHORT($font, $offset) / 2); 
  1059.                 $offset += 2; 
  1060.                 $offset += 6; // skip searchRange, entrySelector, rangeShift 
  1061.                 $endCount = array(); // array of end character codes for each segment 
  1062.                 for ($k = 0; $k < $segCount; ++$k) { 
  1063.                     $endCount[$k] = TCPDF_STATIC::_getUSHORT($font, $offset); 
  1064.                     $offset += 2; 
  1065.                 } 
  1066.                 $offset += 2; // skip reservedPad 
  1067.                 $startCount = array(); // array of start character codes for each segment 
  1068.                 for ($k = 0; $k < $segCount; ++$k) { 
  1069.                     $startCount[$k] = TCPDF_STATIC::_getUSHORT($font, $offset); 
  1070.                     $offset += 2; 
  1071.                 } 
  1072.                 $idDelta = array(); // delta for all character codes in segment 
  1073.                 for ($k = 0; $k < $segCount; ++$k) { 
  1074.                     $idDelta[$k] = TCPDF_STATIC::_getUSHORT($font, $offset); 
  1075.                     $offset += 2; 
  1076.                 }

  1492. /** 
  1493.  * Converts array of UTF-8 characters to UTF16-BE string.<br> 
  1494.  * Based on: http://www.faqs.org/rfcs/rfc2781.html 
  1495.  * <pre> 
  1496.  *   Encoding UTF-16: 
  1497.  * 
  1498.  *   Encoding of a single character from an ISO 10646 character value to 
  1499.  *    UTF-16 proceeds as follows. Let U be the character number, no greater 
  1500.  *    than 0x10FFFF. 
  1501.  * 
  1502.  *    1) If U < 0x10000, encode U as a 16-bit unsigned integer and 
  1503.  *       terminate. 
  1504.  * 
  1505.  *    2) Let U' = U - 0x10000. Because U is less than or equal to 0x10FFFF,  
  1506.  *       U' must be less than or equal to 0xFFFFF. That is, U' can be 
  1507.  *       represented in 20 bits. 
  1508.  * 
  1509.  *    3) Initialize two 16-bit unsigned integers, W1 and W2, to 0xD800 and 
  1510.  *       0xDC00, respectively. These integers each have 10 bits free to 
  1511.  *       encode the character value, for a total of 20 bits. 
  1512.  * 
  1513.  *    4) Assign the 10 high-order bits of the 20-bit U' to the 10 low-order 
  1514.  *       bits of W1 and the 10 low-order bits of U' to the 10 low-order 
  1515.  *       bits of W2. Terminate. 
  1516.  * 
  1517.  *    Graphically, steps 2 through 4 look like: 
  1518.  *    U' = yyyyyyyyyyxxxxxxxxxx 
  1519.  *    W1 = 110110yyyyyyyyyy 
  1520.  *    W2 = 110111xxxxxxxxxx 
  1521.  * </pre>

  1739. /** 
  1740.  * Converts UTF-8 character to integer value.<br> 
  1741.  * Invalid byte sequences will be replaced with 0xFFFD (replacement character)<br> 
  1742.  * Based on: http://www.faqs.org/rfcs/rfc3629.html 
  1743.  * <pre> 
  1744.  *    Char. number range  |        UTF-8 octet sequence 
  1745.  *       (hexadecimal)    |              (binary) 
  1746.  *    --------------------+----------------------------------------------- 
  1747.  *    0000 0000-0000 007F | 0xxxxxxx 
  1748.  *    0000 0080-0000 07FF | 110xxxxx 10xxxxxx 
  1749.  *    0000 0800-0000 FFFF | 1110xxxx 10xxxxxx 10xxxxxx 
  1750.  *    0001 0000-0010 FFFF | 11110xxx 10xxxxxx 10xxxxxx 10xxxxxx 
  1751.  *    --------------------------------------------------------------------- 
  1752.  * 
  1753.  *   ABFN notation: 
  1754.  *   --------------------------------------------------------------------- 
  1755.  *   UTF8-octets = *( UTF8-char ) 
  1756.  *   UTF8-char = UTF8-1 / UTF8-2 / UTF8-3 / UTF8-4 
  1757.  *   UTF8-1 = %x00-7F 
  1758.  *   UTF8-2 = %xC2-DF UTF8-tail 
  1759.  * 
  1760.  *   UTF8-3 = %xE0 %xA0-BF UTF8-tail / %xE1-EC 2( UTF8-tail ) / 
  1761.  *                 %xED %x80-9F UTF8-tail / %xEE-EF 2( UTF8-tail ) 
  1762.  *   UTF8-4 = %xF0 %x90-BF 2( UTF8-tail ) / %xF1-F3 3( UTF8-tail ) / 
  1763.  *                 %xF4 %x80-8F 2( UTF8-tail ) 
  1764.  *   UTF8-tail = %x80-BF 
  1765.  *   --------------------------------------------------------------------- 
  1766.  * </pre> 
  1767.  * @param $uch (string) character string to process. 
  1768.  * @return integer Unicode value 
  1769.  * @author Nicola Asuni 
  1770.  * @public static 
  1771.  */ 
  1772. public static function getUniord($uch) { 
  1773.     if (function_exists('mb_convert_encoding')) { 
  1774.         list(, $char) = @unpack('N', mb_convert_encoding($uch, 'UCS-4BE', 'UTF-8')); 
  1775.         if ($char >= 0) { 
  1776.             return $char; 
  1777.         } 
  1778.     } 
  1779.     $bytes = array(); // array containing single character byte sequences 
  1780.     $countbytes = 0; 
  1781.     $numbytes = 1; // number of octetc needed to represent the UTF-8 character 
  1782.     $length = strlen($uch); 
  1783.     for ($i = 0; $i < $length; ++$i) { 
  1784.         $char = ord($uch[$i]); // get one string character at time 
  1785.         if ($countbytes == 0) { // get starting octect 
  1786.             if ($char <= 0x7F) { 
  1787.                 return $char; // use the character "as is" because is ASCII 
  1788.             } elseif (($char >> 0x05) == 0x06) { // 2 bytes character (0x06 = 110 BIN) 
  1789.                 $bytes[] = ($char - 0xC0) << 0x06; 
  1790.                 ++$countbytes; 
  1791.                 $numbytes = 2; 
  1792.             } elseif (($char >> 0x04) == 0x0E) { // 3 bytes character (0x0E = 1110 BIN) 
  1793.                 $bytes[] = ($char - 0xE0) << 0x0C; 
  1794.                 ++$countbytes; 
  1795.                 $numbytes = 3; 
  1796.             } elseif (($char >> 0x03) == 0x1E) { // 4 bytes character (0x1E = 11110 BIN) 
  1797.                 $bytes[] = ($char - 0xF0) << 0x12; 
  1798.                 ++$countbytes; 
  1799.                 $numbytes = 4; 
  1800.             } else { 
  1801.                 // use replacement character for other invalid sequences 
  1802.                 return 0xFFFD; 
  1803.             } 
  1804.         } elseif (($char >> 0x06) == 0x02) { // bytes 2, 3 and 4 must start with 0x02 = 10 BIN 
  1805.             $bytes[] = $char - 0x80; 
  1806.             ++$countbytes; 
  1807.             if ($countbytes == $numbytes) { 
  1808.                 // compose UTF-8 bytes to a single unicode value 
  1809.                 $char = $bytes[0]; 
  1810.                 for ($j = 1; $j < $numbytes; ++$j) { 
  1811.                     $char += ($bytes[$j] << (($numbytes - $j - 1) * 0x06)); 
  1812.                 } 
  1813.                 if ((($char >= 0xD800) AND ($char <= 0xDFFF)) OR ($char >= 0x10FFFF)) { 
  1814.                     // The definition of UTF-8 prohibits encoding character numbers between 
  1815.                     // U+D800 and U+DFFF, which are reserved for use with the UTF-16 
  1816.                     // encoding form (as surrogate pairs) and do not directly represent 
  1817.                     // characters. 
  1818.                     return 0xFFFD; // use replacement character 
  1819.                 } else { 
  1820.                     return $char; 
  1821.                 } 
  1822.             } 
  1823.         } else { 
  1824.             // use replacement character for other invalid sequences 
  1825.             return 0xFFFD; 
  1826.         } 
  1827.     } 
  1828.     return 0xFFFD; 
  1829. }

  1925.  * @param $isunicode (boolean) True if the document is in Unicode mode, false otherwise. 
  1926.  * @param $currentfont (array) Reference to current font array. 
  1927.  * @return array of unicode chars 
  1928.  * @author Nicola Asuni 
  1929.  * @since 2.3.000 (2008-03-06) 
  1930.  * @public static 
  1931.  */ 
  1932. public static function utf8Bidi($ta, $str='', $forcertl=false, $isunicode=true, &$currentfont) { 
  1933.     // paragraph embedding level 
  1934.     $pel = 0; 
  1935.     // max level 
  1936.     $maxlevel = 0; 
  1937.     if (TCPDF_STATIC::empty_string($str)) { 
  1938.         // create string from array 
  1939.         $str = self::UTF8ArrSubString($ta, '', '', $isunicode); 
  1940.     } 
  1941.     // check if string contains arabic text 
  1942.     if (preg_match(TCPDF_FONT_DATA::$uni_RE_PATTERN_ARABIC, $str)) { 
  1943.         $arabic = true; 
  1944.     } else { 
  1945.         $arabic = false; 
  1946.     } 
  1947.     // check if string contains RTL text 
  1948.     if (!($forcertl OR $arabic OR preg_match(TCPDF_FONT_DATA::$uni_RE_PATTERN_RTL, $str))) { 
  1949.         return $ta; 
  1950.     } 
  1951.  
  1952.     // get number of chars 
  1953.     $numchars = count($ta); 
  1954.  
  1955.     if ($forcertl == 'R') { 
  1956.         $pel = 1; 
  1957.     } elseif ($forcertl == 'L') { 
  1958.         $pel = 0; 
  1959.     } else { 
  1960.         // P2. In each paragraph, find the first character of type L, AL, or R. 
  1961.         // P3. If a character is found in P2 and it is of type AL or R, then set the paragraph embedding level to one; otherwise, set it to zero. 
  1962.         for ($i=0; $i < $numchars; ++$i) { 
  1963.             $type = TCPDF_FONT_DATA::$uni_type[$ta[$i]]; 
  1964.             if ($type == 'L') { 
  1965.                 $pel = 0; 
  1966.                 break; 
  1967.             } elseif (($type == 'AL') OR ($type == 'R')) { 
  1968.                 $pel = 1; 
  1969.                 break; 
  1970.             } 
  1971.         } 
  1972.     } 
  1973.  
  1974.     // Current Embedding Level 
  1975.     $cel = $pel; 
  1976.     // directional override status 
  1977.     $dos = 'N'; 
  1978.     $remember = array(); 
  1979.     // start-of-level-run 
  1980.     $sor = $pel % 2 ? 'R' : 'L'; 
  1981.     $eor = $sor; 
  1982.  
  1983.     // Array of characters data 
  1984.     $chardata = Array(); 
  1985.  
  1986.     // X1. Begin by setting the current embedding level to the paragraph embedding level. Set the directional override status to neutral. Process each character iteratively, applying rules X2 through X9. Only embedding levels from 0 to 61 are valid in this phase. 
  1987.     // In the resolution of levels in rules I1 and I2, the maximum embedding level of 62 can be reached. 
  1988.     for ($i=0; $i < $numchars; ++$i) { 
  1989.         if ($ta[$i] == TCPDF_FONT_DATA::$uni_RLE) { 
  1990.             // X2. With each RLE, compute the least greater odd embedding level. 
  1991.             //    a. If this new level would be valid, then this embedding code is valid. Remember (push) the current embedding level and override status. Reset the current level to this new level, and reset the override status to neutral. 
  1992.             //    b. If the new level would not be valid, then this code is invalid. Do not change the current level or override status. 
  1993.             $next_level = $cel + ($cel % 2) + 1; 
  1994.             if ($next_level < 62) { 
  1995.                 $remember[] = array('num' => TCPDF_FONT_DATA::$uni_RLE, 'cel' => $cel, 'dos' => $dos); 
  1996.                 $cel = $next_level; 
  1997.                 $dos = 'N'; 
  1998.                 $sor = $eor; 
  1999.                 $eor = $cel % 2 ? 'R' : 'L'; 
  2000.             } 
  2001.         } elseif ($ta[$i] == TCPDF_FONT_DATA::$uni_LRE) { 
  2002.             // X3. With each LRE, compute the least greater even embedding level. 
  2003.             //    a. If this new level would be valid, then this embedding code is valid. Remember (push) the current embedding level and override status. Reset the current level to this new level, and reset the override status to neutral. 
  2004.             //    b. If the new level would not be valid, then this code is invalid. Do not change the current level or override status. 
  2005.             $next_level = $cel + 2 - ($cel % 2); 
  2006.             if ( $next_level < 62 ) { 
  2007.                 $remember[] = array('num' => TCPDF_FONT_DATA::$uni_LRE, 'cel' => $cel, 'dos' => $dos); 
  2008.                 $cel = $next_level; 
  2009.                 $dos = 'N'; 
  2010.                 $sor = $eor; 
  2011.                 $eor = $cel % 2 ? 'R' : 'L'; 
  2012.             } 
  2013.         } elseif ($ta[$i] == TCPDF_FONT_DATA::$uni_RLO) { 
  2014.             // X4. With each RLO, compute the least greater odd embedding level. 
  2015.             //    a. If this new level would be valid, then this embedding code is valid. Remember (push) the current embedding level and override status. Reset the current level to this new level, and reset the override status to right-to-left. 
  2016.             //    b. If the new level would not be valid, then this code is invalid. Do not change the current level or override status. 
  2017.             $next_level = $cel + ($cel % 2) + 1; 
  2018.             if ($next_level < 62) { 
  2019.                 $remember[] = array('num' => TCPDF_FONT_DATA::$uni_RLO, 'cel' => $cel, 'dos' => $dos); 
  2020.                 $cel = $next_level; 
  2021.                 $dos = 'R'; 
  2022.                 $sor = $eor; 
  2023.                 $eor = $cel % 2 ? 'R' : 'L'; 
  2024.             } 
  2025.         } elseif ($ta[$i] == TCPDF_FONT_DATA::$uni_LRO) { 
  2026.             // X5. With each LRO, compute the least greater even embedding level. 
  2027.             //    a. If this new level would be valid, then this embedding code is valid. Remember (push) the current embedding level and override status. Reset the current level to this new level, and reset the override status to left-to-right.
