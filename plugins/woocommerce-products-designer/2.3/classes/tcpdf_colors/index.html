  224. /** 
  225.  * Returns an array (RGB or CMYK) from an html color name, or a six-digit (i.e. #3FE5AA), or three-digit (i.e. #7FF) hexadecimal color, or a javascript color array, or javascript color name. 
  226.  * @param $hcolor (string) HTML color. 
  227.  * @param $spotc (array) Reference to an array of spot colors. 
  228.  * @param $defcol (array) Color to return in case of error. 
  229.  * @return array RGB or CMYK color, or false in case of error. 
  230.  * @public static 
  231.  */ 
  232. public static function convertHTMLColorToDec($hcolor, &$spotc, $defcol=array('R'=>128, 'G'=>128, 'B'=>128)) { 
  233.     $color = preg_replace('/[\s]*/', '', $hcolor); // remove extra spaces 
  234.     $color = strtolower($color); 
  235.     // check for javascript color array syntax 
  236.     if (strpos($color, '[') !== false) { 
  237.         if (preg_match('/[\[][\"\'](t|g|rgb|cmyk)[\"\'][\, ]?([0-9\.]*)[\, ]?([0-9\.]*)[\, ]?([0-9\.]*)[\, ]?([0-9\.]*)[\]]/', $color, $m) > 0) { 
  238.             $returncolor = array(); 
  239.             switch ($m[1]) { 
  240.                 case 'cmyk': { 
  241.                     // RGB 
  242.                     $returncolor['C'] = max(0, min(100, (floatval($m[2]) * 100))); 
  243.                     $returncolor['M'] = max(0, min(100, (floatval($m[3]) * 100))); 
  244.                     $returncolor['Y'] = max(0, min(100, (floatval($m[4]) * 100))); 
  245.                     $returncolor['K'] = max(0, min(100, (floatval($m[5]) * 100))); 
  246.                     break; 
  247.                 } 
  248.                 case 'rgb': { 
  249.                     // RGB 
  250.                     $returncolor['R'] = max(0, min(255, (floatval($m[2]) * 255))); 
  251.                     $returncolor['G'] = max(0, min(255, (floatval($m[3]) * 255))); 
  252.                     $returncolor['B'] = max(0, min(255, (floatval($m[4]) * 255))); 
  253.                     break; 
  254.                 } 
  255.                 case 'g': { 
  256.                     // grayscale 
  257.                     $returncolor['G'] = max(0, min(255, (floatval($m[2]) * 255))); 
  258.                     break; 
  259.                 } 
  260.                 case 't': 
  261.                 default: { 
  262.                     // transparent (empty array) 
  263.                     break; 
  264.                 } 
  265.             } 
  266.             return $returncolor; 
  267.         } 
  268.     } elseif ((substr($color, 0, 4) != 'cmyk') AND (substr($color, 0, 3) != 'rgb') AND (($dotpos = strpos($color, '.')) !== false)) { 
  269.         // remove class parent (i.e.: color.red) 
  270.         $color = substr($color, ($dotpos + 1)); 
  271.         if ($color == 'transparent') { 
  272.             // transparent (empty array) 
  273.             return array(); 
  274.         } 
  275.     } 
  276.     if (strlen($color) == 0) { 
  277.         return $defcol; 
  278.     } 
  279.     // RGB ARRAY 
  280.     if (substr($color, 0, 3) == 'rgb') { 
  281.         $codes = substr($color, 4); 
  282.         $codes = str_replace(')', '', $codes); 
  283.         $returncolor = explode(', ', $codes); 
  284.         foreach ($returncolor as $key => $val) { 
  285.             if (strpos($val, '%') > 0) { 
  286.                 // percentage 
  287.                 $returncolor[$key] = (255 * intval($val) / 100); 
  288.             } else { 
  289.                 $returncolor[$key] = intval($val); 
  290.             } 
  291.             // normalize value 
  292.             $returncolor[$key] = max(0, min(255, $returncolor[$key])); 
  293.         } 
  294.         return $returncolor; 
  295.     } 
  296.     // CMYK ARRAY 
  297.     if (substr($color, 0, 4) == 'cmyk') { 
  298.         $codes = substr($color, 5); 
  299.         $codes = str_replace(')', '', $codes); 
  300.         $returncolor = explode(', ', $codes); 
  301.         foreach ($returncolor as $key => $val) { 
  302.             if (strpos($val, '%') !== false) { 
  303.                 // percentage 
  304.                 $returncolor[$key] = (100 * intval($val) / 100); 
  305.             } else { 
  306.                 $returncolor[$key] = intval($val); 
  307.             } 
  308.             // normalize value 
  309.             $returncolor[$key] = max(0, min(100, $returncolor[$key])); 
  310.         } 
  311.         return $returncolor; 
  312.     } 
  313.     if ($color[0] != '#') { 
  314.         // COLOR NAME 
  315.         if (isset(self::$webcolor[$color])) { 
  316.             // web color 
  317.             $color_code = self::$webcolor[$color]; 
  318.         } else { 
  319.             // spot color 
  320.             $returncolor = self::getSpotColor($color, $spotc); 
  321.             if ($returncolor === false) { 
  322.                 $returncolor = $defcol; 
  323.             } 
  324.             return $returncolor; 
  325.         } 
  326.     } else { 
  327.         $color_code = substr($color, 1); 
  328.     } 
  329.     // HEXADECIMAL REPRESENTATION 
  330.     switch (strlen($color_code)) { 
  331.         case 3: { 
  332.             // 3-digit RGB hexadecimal representation 
  333.             $r = substr($color_code, 0, 1); 
  334.             $g = substr($color_code, 1, 1); 
  335.             $b = substr($color_code, 2, 1); 
  336.             $returncolor = array(); 
  337.             $returncolor['R'] = max(0, min(255, hexdec($r.$r))); 
  338.             $returncolor['G'] = max(0, min(255, hexdec($g.$g))); 
  339.             $returncolor['B'] = max(0, min(255, hexdec($b.$b))); 
  340.             break; 
  341.         } 
  342.         case 6: { 
  343.             // 6-digit RGB hexadecimal representation 
  344.             $returncolor = array(); 
  345.             $returncolor['R'] = max(0, min(255, hexdec(substr($color_code, 0, 2)))); 
  346.             $returncolor['G'] = max(0, min(255, hexdec(substr($color_code, 2, 2)))); 
  347.             $returncolor['B'] = max(0, min(255, hexdec(substr($color_code, 4, 2)))); 
  348.             break; 
  349.         } 
  350.         case 8: { 
  351.             // 8-digit CMYK hexadecimal representation 
  352.             $returncolor = array(); 
  353.             $returncolor['C'] = max(0, min(100, round(hexdec(substr($color_code, 0, 2)) / 2.55))); 
  354.             $returncolor['M'] = max(0, min(100, round(hexdec(substr($color_code, 2, 2)) / 2.55))); 
  355.             $returncolor['Y'] = max(0, min(100, round(hexdec(substr($color_code, 4, 2)) / 2.55))); 
  356.             $returncolor['K'] = max(0, min(100, round(hexdec(substr($color_code, 6, 2)) / 2.55))); 
  357.             break; 
  358.         } 
  359.         default: { 
  360.             $returncolor = $defcol; 
  361.             break; 
  362.         } 
  363.     } 
  364.     return $returncolor; 
  365. }
