  327.  */ 
  328. public static function decodeFilterRunLengthDecode($data) { 
  329.     // intialize string to return 
  330.     $decoded = ''; 
  331.     // data length 
  332.     $data_length = strlen($data); 
  333.     $i = 0; 
  334.     while($i < $data_length) { 
  335.         // get current byte value 
  336.         $byte = ord($data{$i}); 
  337.         if ($byte == 128) { 
  338.             // a length value of 128 denote EOD 
  339.             break; 
  340.         } elseif ($byte < 128) { 
  341.             // if the length byte is in the range 0 to 127 
  342.             // the following length + 1 (1 to 128) bytes shall be copied literally during decompression 
  343.             $decoded .= substr($data, ($i + 1), ($byte + 1)); 
  344.             // move to next block 
  345.             $i += ($byte + 2); 
  346.         } else { 
  347.             // if length is in the range 129 to 255,  
  348.             // the following single byte shall be copied 257 - length (2 to 128) times during decompression 
  349.             $decoded .= str_repeat($data{($i + 1)}, (257 - $byte)); 
  350.             // move to next block 
  351.             $i += 2; 
  352.         } 
  353.     }

  159.     if ($eod !== false) { 
  160.         // remove EOD and extra data (if any) 
  161.         $data = substr($data, 0, $eod); 
  162.     } 
  163.     // data length 
  164.     $data_length = strlen($data); 
  165.     // check for invalid characters 
  166.     if (preg_match('/[^\x21-\x75, \x74]/', $data) > 0) { 
  167.         self::Error('decodeFilterASCII85Decode: invalid code'); 
  168.     } 
  169.     // z sequence 
  170.     $zseq = chr(0).chr(0).chr(0).chr(0); 
  171.     // position inside a group of 4 bytes (0-3) 
  172.     $group_pos = 0; 
  173.     $tuple = 0; 
  174.     $pow85 = array((85*85*85*85), (85*85*85), (85*85), 85, 1); 
  175.     $last_pos = ($data_length - 1); 
  176.     // for each byte 
  177.     for ($i = 0; $i < $data_length; ++$i) { 
  178.         // get char value 
  179.         $char = ord($data[$i]); 
  180.         if ($char == 122) { // 'z' 
  181.             if ($group_pos == 0) { 
  182.                 $decoded .= $zseq; 
  183.             } else { 
  184.                 self::Error('decodeFilterASCII85Decode: invalid code'); 
  185.             } 
  186.         } else { 
  187.             // the value represented by a group of 5 characters should never be greater than 2^32 - 1 
  188.             $tuple += (($char - 33) * $pow85[$group_pos]); 
  189.             if ($group_pos == 4) { 
  190.                 $decoded .= chr($tuple >> 24).chr($tuple >> 16).chr($tuple >> 8).chr($tuple); 
  191.                 $tuple = 0; 
  192.                 $group_pos = 0; 
  193.             } else { 
  194.                 ++$group_pos; 
  195.             } 
  196.         } 
  197.     } 
  198.     if ($group_pos > 1) { 
  199.         $tuple += $pow85[($group_pos - 1)]; 
  200.     } 
  201.     // last tuple (if any) 
  202.     switch ($group_pos) { 
  203.         case 4: { 
  204.             $decoded .= chr($tuple >> 24).chr($tuple >> 16).chr($tuple >> 8); 
  205.             break; 
  206.         } 
  207.         case 3: { 
  208.             $decoded .= chr($tuple >> 24).chr($tuple >> 16); 
  209.             break; 
  210.         } 
  211.         case 2: { 
  212.             $decoded .= chr($tuple >> 24); 
  213.             break; 
  214.         } 
  215.         case 1: { 
  216.             self::Error('decodeFilterASCII85Decode: invalid code'); 
  217.             break; 
  218.         } 
  219.     } 
  220.     return $decoded; 
  221. }

  223. /** 
  224.  * LZWDecode 
  225.  * Decompresses data encoded using the LZW (Lempel-Ziv-Welch) adaptive compression method, reproducing the original text or binary data. 
  226.  * @param $data (string) Data to decode. 
  227.  * @return Decoded data string. 
  228.  * @since 3.000 (2011-05-23) 
  229.  * @public static 
  230.  */ 
  231. public static function decodeFilterLZWDecode($data) { 
  232.     // intialize string to return 
  233.     $decoded = ''; 
  234.     // data length 
  235.     $data_length = strlen($data); 
  236.     // convert string to binary string 
  237.     $bitstring = ''; 
  238.     for ($i = 0; $i < $data_length; ++$i) { 
  239.         $bitstring .= sprintf('%08b', ord($data{$i})); 
  240.     } 
  241.     // get the number of bits 
  242.     $data_length = strlen($bitstring); 
  243.     // initialize code length in bits 
  244.     $bitlen = 9; 
  245.     // initialize dictionary index 
  246.     $dix = 258; 
  247.     // initialize the dictionary (with the first 256 entries). 
  248.     $dictionary = array(); 
  249.     for ($i = 0; $i < 256; ++$i) { 
  250.         $dictionary[$i] = chr($i); 
  251.     } 
  252.     // previous val 
  253.     $prev_index = 0; 
  254.     // while we encounter EOD marker (257), read code_length bits 
  255.     while (($data_length > 0) AND (($index = bindec(substr($bitstring, 0, $bitlen))) != 257)) { 
  256.         // remove read bits from string 
  257.         $bitstring = substr($bitstring, $bitlen); 
  258.         // update number of bits 
  259.         $data_length -= $bitlen; 
  260.         if ($index == 256) { // clear-table marker 
  261.             // reset code length in bits 
  262.             $bitlen = 9; 
  263.             // reset dictionary index 
  264.             $dix = 258; 
  265.             $prev_index = 256; 
  266.             // reset the dictionary (with the first 256 entries). 
  267.             $dictionary = array(); 
  268.             for ($i = 0; $i < 256; ++$i) { 
  269.                 $dictionary[$i] = chr($i); 
  270.             } 
  271.         } elseif ($prev_index == 256) { 
  272.             // first entry 
  273.             $decoded .= $dictionary[$index]; 
  274.             $prev_index = $index; 
  275.         } else { 
  276.             // check if index exist in the dictionary 
  277.             if ($index < $dix) { 
  278.                 // index exist on dictionary 
  279.                 $decoded .= $dictionary[$index]; 
  280.                 $dic_val = $dictionary[$prev_index].$dictionary[$index][0]; 
  281.                 // store current index 
  282.                 $prev_index = $index; 
  283.             } else { 
  284.                 // index do not exist on dictionary 
  285.                 $dic_val = $dictionary[$prev_index].$dictionary[$prev_index][0]; 
  286.                 $decoded .= $dic_val; 
  287.             } 
  288.             // update dictionary 
  289.             $dictionary[$dix] = $dic_val; 
  290.             ++$dix; 
  291.             // change bit length by case 
  292.             if ($dix == 2047) { 
  293.                 $bitlen = 12; 
  294.             } elseif ($dix == 1023) { 
  295.                 $bitlen = 11; 
  296.             } elseif ($dix == 511) { 
  297.                 $bitlen = 10; 
  298.             } 
  299.         } 
  300.     } 
  301.     return $decoded; 
  302. }

  321. /** 
  322.  * RunLengthDecode 
  323.  * Decompresses data encoded using a byte-oriented run-length encoding algorithm. 
  324.  * @param $data (string) Data to decode. 
  325.  * @since 3.000 (2011-05-23) 
  326.  * @public static 
  327.  */ 
  328. public static function decodeFilterRunLengthDecode($data) { 
  329.     // intialize string to return 
  330.     $decoded = ''; 
  331.     // data length 
  332.     $data_length = strlen($data); 
  333.     $i = 0; 
  334.     while($i < $data_length) { 
  335.         // get current byte value 
  336.         $byte = ord($data{$i}); 
  337.         if ($byte == 128) { 
  338.             // a length value of 128 denote EOD 
  339.             break; 
  340.         } elseif ($byte < 128) { 
  341.             // if the length byte is in the range 0 to 127 
  342.             // the following length + 1 (1 to 128) bytes shall be copied literally during decompression 
  343.             $decoded .= substr($data, ($i + 1), ($byte + 1)); 
  344.             // move to next block 
  345.             $i += ($byte + 2); 
  346.         } else { 
  347.             // if length is in the range 129 to 255,  
  348.             // the following single byte shall be copied 257 - length (2 to 128) times during decompression 
  349.             $decoded .= str_repeat($data{($i + 1)}, (257 - $byte)); 
  350.             // move to next block 
  351.             $i += 2; 
  352.         } 
  353.     } 
  354.     return $decoded; 
  355. }
