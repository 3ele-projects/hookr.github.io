  93.   public static function process_product_simple_countries_prices( $post_id, $i = false, $variable = '' ) {                 
  94.        
  95.       foreach ( WCPBC()->get_regions() as $key => $value ) { 
  96.            
  97.           $key_prefix = '_' . $key . $variable; 
  98.           $key_regular_price = $key_prefix . '_regular_price'; 
  99.           $key_sale_price = $key_prefix . '_sale_price';             
  100.           $key_price = $key_prefix . '_price';                         
  101.           $key_price_method = $key_prefix . '_price_method'; 
  102.            
  103.  
  104.           if ( $i === false ) { 
  105.                
  106.               $_regular_price = $_POST[$key_regular_price]; 
  107.               $_sale_price = $_POST[$key_sale_price]; 
  108.               $_price_method = $_POST[$key_price_method];     
  109.  
  110.           } else { 
  111.  
  112.               $_regular_price = $_POST[$key_regular_price][$i]; 
  113.               $_sale_price = $_POST[$key_sale_price][$i]; 
  114.               $_price_method = $_POST[$key_price_method][$i];         
  115.           } 
  116.  
  117.           $_price_method = ( $_price_method == 'exchange_rate' || empty( $_regular_price ) ) ? 'exchange_rate' : 'manual'; 
  118.  
  119.           if ( $_price_method == 'exchange_rate') { 
  120.  
  121.               $_regular_price = ''; 
  122.               $_sale_price = ''; 
  123.               $_price = ''; 
  124.  
  125.           } else { 
  126.  
  127.               $_regular_price = wc_format_decimal( $_regular_price ); 
  128.               $_sale_price = wc_format_decimal( $_sale_price ); 
  129.               // Update price if on sale 
  130.               $_price = ( '' !== $_sale_price ) ? $_sale_price : $_regular_price;                                 
  131.           }                 
  132.  
  133.           update_post_meta( $post_id, $key_regular_price, $_regular_price );                             
  134.           update_post_meta( $post_id, $key_sale_price, $_sale_price );                 
  135.           update_post_meta( $post_id, $key_price, $_price );                             
  136.           update_post_meta( $post_id, $key_price_method, $_price_method ); 
  137.  
  138.           do_action('wc_price_based_country_process_product_pricing', $key_prefix, $post_id, $_price_method, $i );                             
  139.            
  140.       }

  216.   public static function variable_product_sync( $product_id, $children ) {         
  217.        
  218.       foreach ( WCPBC()->get_regions() as $region_key => $region ) { 
  219.            
  220.           // Main active prices 
  221.           $min_price = null; 
  222.           $max_price = null; 
  223.           $min_price_id = null; 
  224.           $max_price_id = null; 
  225.  
  226.           // Regular prices 
  227.           $min_regular_price = null; 
  228.           $max_regular_price = null; 
  229.           $min_regular_price_id = null; 
  230.           $max_regular_price_id = null; 
  231.  
  232.           // Sale prices 
  233.           $min_sale_price = null; 
  234.           $max_sale_price = null; 
  235.           $min_sale_price_id = null; 
  236.           $max_sale_price_id = null; 
  237.  
  238.           foreach ( array( 'price', 'regular_price', 'sale_price' ) as $price_type ) {                                 
  239.  
  240.               foreach ( $children as $child_id ) {                 
  241.  
  242.                   $child_price_method = get_post_meta( $child_id, '_' . $region_key . '_variable_price_method', true ); 
  243.  
  244.                   if ( $child_price_method !== 'manual' ) { 
  245.  
  246.                       $child_price = get_post_meta( $child_id, '_' . $price_type, true );                     
  247.                       $child_price = ( !empty( $child_price ) && $region['exchange_rate'] ) ? ( $region['exchange_rate'] * $child_price ) : $child_price;                             
  248.  
  249.                   } else{ 
  250.  
  251.                       $child_price = get_post_meta( $child_id, '_' . $region_key . '_variable_' . $price_type, true );                     
  252.                   }                     
  253.  
  254.                   // Skip non-priced variations 
  255.                   if ( $child_price === '' ) { 
  256.                       continue; 
  257.                   } 
  258.  
  259.                   // Skip hidden variations 
  260.                   if ( 'yes' === get_option( 'woocommerce_hide_out_of_stock_items' ) ) { 
  261.                       $stock = get_post_meta( $child_id, '_stock', true ); 
  262.                       if ( $stock !== "" && $stock <= get_option( 'woocommerce_notify_no_stock_amount' ) ) { 
  263.                           continue; 
  264.                       } 
  265.                   } 
  266.  
  267.                   // Find min price 
  268.                   if ( is_null( ${"min_{$price_type}"} ) || $child_price < ${"min_{$price_type}"} ) { 
  269.                       ${"min_{$price_type}"} = $child_price; 
  270.                       ${"min_{$price_type}_id"} = $child_id; 
  271.                   } 
  272.  
  273.                   // Find max price 
  274.                   if ( is_null( ${"max_{$price_type}"} ) || $child_price > ${"max_{$price_type}"} ) { 
  275.                       ${"max_{$price_type}"} = $child_price; 
  276.                       ${"max_{$price_type}_id"} = $child_id; 
  277.                   } 
  278.               } 
  279.            
  280.               // Store ids 
  281.               update_post_meta( $product_id, '_' . $region_key . '_min_' . $price_type . '_variation_id', ${"min_{$price_type}_id"} ); 
  282.               update_post_meta( $product_id, '_' . $region_key . '_max_' . $price_type . '_variation_id', ${"max_{$price_type}_id"} ); 
  283.           }             
  284.       }         
  285.   }
