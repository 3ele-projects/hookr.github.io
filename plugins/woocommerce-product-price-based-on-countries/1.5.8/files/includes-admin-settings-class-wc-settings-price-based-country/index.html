  271.   /** 
  272.    * Get a unique slug that indentify a region 
  273.    * 
  274.    * @param  string $new_slug 
  275.    * @param  array $slugs 
  276.    * @return array 
  277.    */ 
  278.   private static function get_unique_slug( $new_slug, $slugs ) {                 
  279.        
  280.       $seqs = array(); 
  281.  
  282.       foreach ( $slugs as $slug ) { 
  283.           $slug_parts = explode( '-', $slug, 2 ); 
  284.           if ( $slug_parts[0] == $new_slug && ( count( $slug_parts ) == 1 || is_numeric( $slug_parts[1] ) ) ) { 
  285.               $seqs[] = isset( $slug_parts[1] ) ? $slug_parts[1] : 0; 
  286.           }             
  287.       }         
  288.        
  289.       if ($seqs ) { 
  290.           rsort($seqs);                     
  291.           $new_slug = $new_slug .'-' . ( $seqs[0]+1 );         
  292.       } 
  293.  
  294.       return $new_slug; 
  295.   }
