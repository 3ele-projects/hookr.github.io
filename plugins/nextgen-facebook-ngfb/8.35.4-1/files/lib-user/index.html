  357.       public function sanitize_submit_cm( $user_id ) { 
  358.  
  359.           if ( ! current_user_can( 'edit_user', $user_id ) ) 
  360.               return; 
  361.  
  362.           foreach ( $this->p->cf['opt']['pre'] as $cm_id => $cm_pre ) { 
  363.               $cm_opt = 'plugin_cm_'.$cm_pre.'_'; 
  364.               // not all social websites have contact fields, so check 
  365.               if ( array_key_exists( $cm_opt.'name', $this->p->options ) ) { 
  366.  
  367.                   $enabled = $this->p->options[$cm_opt.'enabled']; 
  368.                   $name = $this->p->options[$cm_opt.'name']; 
  369.                   $label = $this->p->options[$cm_opt.'label']; 
  370.  
  371.                   if ( isset( $_POST[$name] ) &&  
  372.                       ! empty( $enabled ) &&  
  373.                       ! empty( $name ) &&  
  374.                       ! empty( $label ) ) { 
  375.  
  376.                       // sanitize values only for those enabled contact methods 
  377.                       $val = wp_filter_nohtml_kses( $_POST[$name] ); 
  378.                       if ( ! empty( $val ) ) { 
  379.                           switch ( $name ) { 
  380.                               case $this->p->options['plugin_cm_skype_name']: 
  381.                                   // no change 
  382.                                   break; 
  383.                               case $this->p->options['plugin_cm_twitter_name']: 
  384.                                   $val = substr( preg_replace( '/[^a-zA-Z0-9_]/', '', $val ), 0, 15 ); 
  385.                                   if ( ! empty( $val ) )  
  386.                                       $val = '@'.$val; 
  387.                                   break; 
  388.                               default: 
  389.                                   // all other contact methods are assumed to be URLs 
  390.                                   if ( strpos( $val, '://' ) === false ) 
  391.                                       $val = ''; 
  392.                                   break; 
  393.                           } 
  394.                       } 
  395.                       $_POST[$name] = $val; 
  396.                   } 
  397.               } 
  398.           } 
  399.           return $user_id; 
  400.       }

  512.       public function reset_metabox_prefs( $pagehook, $box_ids = array(), $meta_name = '', $section = '', $force = false ) { 
  513.           $user_id = get_current_user_id();    // since wp 3.0 
  514.           // define a new state to set for the box_ids given 
  515.           switch ( $meta_name ) { 
  516.               case 'order':    $meta_states = array( 'meta-box-order' ); break ; 
  517.               case 'hidden':    $meta_states = array( 'metaboxhidden' ); break ; 
  518.               case 'closed':    $meta_states = array( 'closedpostboxes' ); break ; 
  519.               default: $meta_states = array( 'meta-box-order', 'metaboxhidden', 'closedpostboxes' ); break; 
  520.           } 
  521.           foreach ( $meta_states as $state ) { 
  522.               // define the meta_key for that option 
  523.               $meta_key = $state.'_'.$pagehook;  
  524.               // an empty box_ids array means reset the whole page 
  525.               if ( $force && empty( $box_ids ) ) 
  526.                   delete_user_option( $user_id, $meta_key, true ); 
  527.               $is_changed = false; 
  528.               $is_default = false; 
  529.               $opts = get_user_option( $meta_key, $user_id ); 
  530.               if ( ! is_array( $opts ) ) { 
  531.                   $is_changed = true; 
  532.                   $is_default = true; 
  533.                   $opts = array(); 
  534.               } 
  535.               if ( $is_default || $force ) { 
  536.                   foreach ( $box_ids as $box_id ) { 
  537.                       // change the order only if forced (default is controlled by add_meta_box() order) 
  538.                       if ( $force && $state == 'meta-box-order' && ! empty( $opts[$section] ) ) { 
  539.                           // don't proceed if the metabox is already first 
  540.                           if ( strpos( $opts[$section], $pagehook.'_'.$box_id ) !== 0 ) { 
  541.                               $boxes = explode( ', ', $opts[$section] ); 
  542.                               // remove the box, no matter its position in the array 
  543.                               if ( $key = array_search( $pagehook.'_'.$box_id, $boxes ) !== false ) 
  544.                                   unset( $boxes[$key] ); 
  545.                               // assume we want to be top-most 
  546.                               array_unshift( $boxes, $pagehook.'_'.$box_id ); 
  547.                               $opts[$section] = implode( ', ', $boxes ); 
  548.                               $is_changed = true; 
  549.                           } 
  550.                       } else { 
  551.                           // check to see if the metabox is present for that state 
  552.                           $key = array_search( $pagehook.'_'.$box_id, $opts ); 
  553.  
  554.                           // if we're not targetting , then clear it 
  555.                           if ( empty( $meta_name ) && $key !== false ) { 
  556.                               unset( $opts[$key] ); 
  557.                               $is_changed = true; 
  558.                           // otherwise if we want a state, add if it's missing 
  559.                           } elseif ( ! empty( $meta_name ) && $key === false ) { 
  560.                               $opts[] = $pagehook.'_'.$box_id; 
  561.                               $is_changed = true; 
  562.                           } 
  563.                       } 
  564.                   } 
  565.               } 
  566.               if ( $is_default || $is_changed ) 
  567.                   update_user_option( $user_id, $meta_key, array_unique( $opts ), true ); 
  568.           } 
  569.       }

  600.       public static function save_pref( $user_prefs, $user_id = false ) { 
  601.           $user_id = $user_id === false ?  
  602.               get_current_user_id() : $user_id; 
  603.  
  604.           if ( ! current_user_can( 'edit_user', $user_id ) ) 
  605.               return false; 
  606.  
  607.           if ( ! is_array( $user_prefs ) ||  
  608.               empty( $user_prefs ) ) 
  609.                   return false; 
  610.  
  611.           $old_prefs = self::get_pref( false, $user_id );    // get all prefs for user 
  612.           $new_prefs = array_merge( $old_prefs, $user_prefs ); 
  613.  
  614.           // don't bother saving unless we have to 
  615.           if ( $old_prefs !== $new_prefs ) { 
  616.               self::$pref[$user_id] = $new_prefs;    // update the pref cache 
  617.               unset( $new_prefs['options_filtered'] ); 
  618.               update_user_meta( $user_id, NGFB_PREF_NAME, $new_prefs ); 
  619.               return true; 
  620.           } else return false; 
  621.       }

  623.       public static function get_pref( $idx = false, $user_id = false ) { 
  624.           $user_id = $user_id === false ?  
  625.               get_current_user_id() : $user_id; 
  626.  
  627.           if ( ! isset( self::$pref[$user_id]['options_filtered'] ) ||  
  628.               self::$pref[$user_id]['options_filtered'] !== true ) { 
  629.  
  630.               $ngfb = Ngfb::get_instance(); 
  631.

  633.                   get_user_meta( $user_id, NGFB_PREF_NAME, true ), $user_id ); 
  634.  
  635.               if ( ! is_array( self::$pref[$user_id] ) ) 
  636.                   self::$pref[$user_id] = array(); 
  637.  
  638.               if ( ! isset( self::$pref[$user_id]['show_opts'] ) ) 
  639.                   self::$pref[$user_id]['show_opts'] = $ngfb->options['plugin_show_opts']; 
  640.  
  641.               self::$pref[$user_id]['options_filtered'] = true; 
  642.           } 
  643.  
  644.           if ( $idx !== false ) { 
  645.               if ( isset( self::$pref[$user_id][$idx] ) )  
  646.                   return self::$pref[$user_id][$idx]; 
  647.               else return false; 
  648.           } else return self::$pref[$user_id]; 
  649.       }
