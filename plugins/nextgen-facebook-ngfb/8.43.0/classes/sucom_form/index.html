  591.     public function get_textarea( $name, $class = '', $id = '', $len = 0, $placeholder = '', $disabled = false ) { 
  592.         if ( empty( $name ) ) {

  686.     private function get_sanitized_placeholder( $name, $placeholder ) { 
  687.         if ( empty( $name ) ) { 
  688.             return $placeholder;    // just in case 
  689.         } 
  690.         if ( $placeholder === true ) { 
  691.             if ( isset( $this->defaults[$name] ) ) { 
  692.                 $placeholder = $this->defaults[$name]; 
  693.             } 
  694.         } 
  695.         if ( $placeholder === true || $placeholder === '' ) { 
  696.             if ( ( $pos = strpos( $name, '#' ) ) > 0 ) { 
  697.                 $key_default = SucomUtil::get_key_locale( substr( $name, 0, $pos ), $this->options, 'default' ); 
  698.                 if ( $name !== $key_default ) { 
  699.                     if ( isset( $this->options[$key_default] ) ) { 
  700.                         $placeholder = $this->options[$key_default]; 
  701.                     } elseif ( $placeholder === true && isset( $this->defaults[$key_default] ) ) { 
  702.                         $placeholder = $this->defaults[$key_default]; 
  703.                     } 
  704.                 } 
  705.             } 
  706.         } 
  707.         if ( $placeholder === true ) { 
  708.             $placeholder = '';    // must be a string 
  709.         } 
  710.         return $placeholder; 
  711.     }

  713.     private function get_placeholder_events( $type = 'input', $placeholder ) { 
  714.         if ( $placeholder === '' ) 
  715.             return ''; 
  716.  
  717.         $js_if_empty = 'if ( this.value == \'\' ) this.value = \''.esc_js( $placeholder ).'\';'; 
  718.         $js_if_same = 'if ( this.value == \''.esc_js( $placeholder ).'\' ) this.value = \'\';'; 
  719.  
  720.         $html = ' placeholder="'.esc_attr( $placeholder ).'"'. 
  721.             ' onFocus="'.$js_if_empty.'"'. 
  722.             ' onBlur="'.$js_if_same.'"'; 
  723.  
  724.         if ( $type === 'input' ) { 
  725.             $html .= ' onKeyPress="if ( event.keyCode === 13 ) { '.$js_if_same.' }"'; 
  726.         } elseif ( $type === 'textarea' ) { 
  727.             $html .= ' onMouseOut="'.$js_if_same.'"'; 
  728.         } 
  729.  
  730.         return $html; 
  731.     }
