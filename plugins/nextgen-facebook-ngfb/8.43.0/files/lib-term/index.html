  393.                           delete_option( $key_name.'_term_'.$term_id ); 
  394.                           $term_meta = get_term_meta( $term_id, $key_name, $single ); 
  395.                       } else {

  397.                               array( $opt_term_meta ) : $opt_term_meta;

  399.                   }

  402.               $term_meta = $single === false ?
