  376.                           if ( ! empty( $this->p->options[$cm_label] ) ) 
  377.                               $fields[$id] = $this->p->options[$cm_label]; 
  378.                       } else unset( $fields[$id] ); 
  379.                   } 
  380.               }

  388.                   $cm_name = 'plugin_cm_'.$opt_pre.'_name'; 
  389.                   $cm_label = 'plugin_cm_'.$opt_pre.'_label'; 
  390.  
  391.                   // not all social websites have a contact fields, so check 
  392.                   if ( isset( $this->p->options[$cm_name] ) ) { 
  393.                       if ( ! empty( $this->p->options[$cm_enabled] ) &&  
  394.                           ! empty( $this->p->options[$cm_name] ) &&  
  395.                           ! empty( $this->p->options[$cm_label] ) ) { 
  396.  
  397.                           $fields[$this->p->options[$cm_name]] = $this->p->options[$cm_label]; 
  398.                       } 
  399.                   }

  408.       public function sanitize_submit_cm( $user_id ) { 
  409.  
  410.           if ( ! current_user_can( 'edit_user', $user_id ) ) 
  411.               return; 
  412.  
  413.           foreach ( $this->p->cf['opt']['cm_prefix'] as $id => $opt_pre ) { 
  414.               // not all social websites have contact fields, so check 
  415.               if ( isset( $this->p->options['plugin_cm_'.$opt_pre.'_name'] ) ) { 
  416.  
  417.                   $cm_enabled = $this->p->options['plugin_cm_'.$opt_pre.'_enabled']; 
  418.                   $cm_name = $this->p->options['plugin_cm_'.$opt_pre.'_name']; 
  419.                   $cm_label = $this->p->options['plugin_cm_'.$opt_pre.'_label']; 
  420.  
  421.                   // sanitize values only for those enabled contact methods 
  422.                   if ( isset( $_POST[$cm_name] ) && ! empty( $cm_enabled ) && ! empty( $cm_name ) && ! empty( $cm_label ) ) { 
  423.                       $val = wp_filter_nohtml_kses( $_POST[$cm_name] ); 
  424.                       if ( ! empty( $val ) ) { 
  425.                           switch ( $cm_name ) { 
  426.                               case $this->p->options['plugin_cm_skype_name']: 
  427.                                   // no change 
  428.                                   break; 
  429.                               case $this->p->options['plugin_cm_twitter_name']: 
  430.                                   $val = SucomUtil::get_at_name( $val ); 
  431.                                   break; 
  432.                               default: 
  433.                                   // all other contact methods are assumed to be URLs 
  434.                                   if ( strpos( $val, '://' ) === false ) 
  435.                                       $val = ''; 
  436.                                   break; 
  437.                           } 
  438.                       } 
  439.                       $_POST[$cm_name] = $val; 
  440.                   } 
  441.               } 
  442.           } 
  443.           return $user_id; 
  444.       }

  583.       public static function reset_metabox_prefs( $pagehook, $box_ids = array(), $meta_name = '', $context = '', $force = false ) { 
  584.           $user_id = get_current_user_id();    // since wp 3.0 
  585.           switch ( $meta_name ) { 
  586.               case 'order': 
  587.               case 'meta-box-order': 
  588.                   $meta_states = array( 'meta-box-order' ); 
  589.                   break; 
  590.               case 'hidden': 
  591.               case 'metaboxhidden': 
  592.                   $meta_states = array( 'metaboxhidden' ); 
  593.                   break; 
  594.               case 'closed': 
  595.               case 'closedpostboxes': 
  596.                   $meta_states = array( 'closedpostboxes' ); 
  597.                   break; 
  598.               default: 
  599.                   $meta_states = array( 'meta-box-order', 'metaboxhidden', 'closedpostboxes' ); 
  600.                   break; 
  601.           } 
  602.           foreach ( $meta_states as $state ) { 
  603.               $meta_key = $state.'_'.$pagehook;  
  604.               if ( $force && empty( $box_ids ) ) { 
  605.                   delete_user_option( $user_id, $meta_key, true ); 
  606.               } 
  607.               $is_changed = false; 
  608.               $is_default = false; 
  609.               $user_opts = get_user_option( $meta_key, $user_id ); 
  610.               if ( empty( $user_opts ) ) { 
  611.                   $is_changed = true; 
  612.                   $is_default = true; 
  613.                   $user_opts = array(); 
  614.               } 
  615.               if ( $is_default || $force ) { 
  616.                   foreach ( $box_ids as $box_id ) { 
  617.                       // change the order only if forced (default is controlled by add_meta_box() order) 
  618.                       if ( $force && $state == 'meta-box-order' && ! empty( $user_opts[$context] ) ) { 
  619.                           // don't proceed if the metabox is already first 
  620.                           if ( strpos( $user_opts[$context], $pagehook.'_'.$box_id ) !== 0 ) { 
  621.                               $boxes = explode( ', ', $user_opts[$context] ); 
  622.                               // remove the box, no matter its position in the array 
  623.                               if ( $key = array_search( $pagehook.'_'.$box_id, $boxes ) !== false ) { 
  624.                                   unset( $boxes[$key] ); 
  625.                               } 
  626.                               // assume we want to be top-most 
  627.                               array_unshift( $boxes, $pagehook.'_'.$box_id ); 
  628.                               $user_opts[$context] = implode( ', ', $boxes ); 
  629.                               $is_changed = true; 
  630.                           } 
  631.                       } else { 
  632.                           // check to see if the metabox is present for that state 
  633.                           $key = array_search( $pagehook.'_'.$box_id, $user_opts ); 
  634.  
  635.                           // if we're not targetting, then clear it 
  636.                           if ( empty( $meta_name ) && $key !== false ) { 
  637.                               unset( $user_opts[$key] ); 
  638.                               $is_changed = true; 
  639.                           // otherwise if we want a state, add if it's missing 
  640.                           } elseif ( ! empty( $meta_name ) && $key === false ) { 
  641.                               $user_opts[] = $pagehook.'_'.$box_id; 
  642.                               $is_changed = true; 
  643.                           } 
  644.                       } 
  645.                   } 
  646.               } 
  647.               if ( $is_default || $is_changed ) { 
  648.                   update_user_option( $user_id, $meta_key, array_unique( $user_opts ), true ); 
  649.               } 
  650.           } 
  651.       }

  689.           $user_id = $user_id === false ?  
  690.               get_current_user_id() : $user_id; 
  691.  
  692.           if ( ! current_user_can( 'edit_user', $user_id ) )

  696.               empty( $user_prefs ) )

  698.

  700.           $new_prefs = array_merge( $old_prefs, $user_prefs ); 
  701.  
  702.           // don't bother saving unless we have to

  707.               return true; 
  708.           } else return false;
