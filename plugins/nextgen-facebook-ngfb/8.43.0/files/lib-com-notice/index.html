  176.                       // clear a specific message string 
  177.                       } elseif ( isset( $user_notices[$msg_type][$msg_txt] ) ) { 
  178.                           unset( $user_notices[$msg_type][$msg_txt] );

  211.       public function show_admin_notices() { 
  212.  
  213.           $hidden = array(); 
  214.           $msg_html = ''; 
  215.           $nag_msgs = ''; 
  216.           $seen_msgs = array();    // duplicate check 
  217.           $dismissed_updated = false; 
  218.           $user_id = (int) get_current_user_id(); 
  219.           $user_notices =& $this->get_user_notices( $user_id );    // returns reference 
  220.           $user_dismissed = empty( $user_id ) ? false :         // just in case 
  221.               get_user_option( $this->dis_name, $user_id );    // get dismissed message ids 
  222.           $this->has_shown = true; 
  223.  
  224.           if ( isset( $this->p->cf['plugin'] ) && class_exists( 'SucomUpdate' ) ) { 
  225.               foreach ( array_keys( $this->p->cf['plugin'] ) as $lca ) { 
  226.                   if ( ! empty( $this->p->options['plugin_'.$lca.'_tid'] ) ) { 
  227.                       $uerr = SucomUpdate::get_umsg( $lca ); 
  228.                       if ( ! empty( $uerr ) ) { 
  229.                           $user_notices['err'][$uerr] = array(); 
  230.                       } 
  231.                   } 
  232.               } 
  233.           } 
  234.  
  235.           // loop through all the msg types and show them all 
  236.           foreach ( $this->all_types as $msg_type ) { 
  237.  
  238.               if ( ! isset( $user_notices[$msg_type] ) ) {    // just in case 
  239.                   continue; 
  240.               } 
  241.  
  242.               foreach ( $user_notices[$msg_type] as $msg_txt => $payload ) { 
  243.  
  244.                   if ( empty( $msg_txt ) || isset( $seen_msgs[$msg_txt] ) ) {    // skip duplicates 
  245.                       continue; 
  246.                   } 
  247.  
  248.                   $seen_msgs[$msg_txt] = true;    // avoid duplicates 
  249.  
  250.                   switch ( $msg_type ) { 
  251.                       case 'nag': 
  252.                           $nag_msgs .= $msg_txt;    // append to echo a single msg block 
  253.                           continue; 
  254.                       default: 
  255.                           // dis_time will be false if there's no dis_key 
  256.                           if ( ! empty( $payload['dis_time'] ) ) { 
  257.  
  258.                               // initialize the count 
  259.                               if ( ! isset( $hidden[$msg_type] ) ) { 
  260.                                   $hidden[$msg_type] = 0; 
  261.                               } 
  262.  
  263.                               // check for automatically hidden errors and/or warnings 
  264.                               if ( ( $msg_type === 'err' && $this->hide_err ) || 
  265.                                   ( $msg_type === 'warn' && $this->hide_warn ) ) { 
  266.  
  267.                                   $payload['hidden'] = true; 
  268.                                   if ( empty( $payload['silent'] ) ) { 
  269.                                       $hidden[$msg_type]++; 
  270.                                   } 
  271.  
  272.                               // dis_key has been dismissed 
  273.                               } elseif ( ! empty( $payload['dis_key'] ) && 
  274.                                   isset( $user_dismissed[$payload['dis_key']] ) ) { 
  275.  
  276.                                   $now_time = time(); 
  277.                                   $dis_time = $user_dismissed[$payload['dis_key']]; 
  278.  
  279.                                   if ( empty( $dis_time ) || $dis_time > $now_time ) { 
  280.                                       $payload['hidden'] = true; 
  281.                                       if ( empty( $payload['silent'] ) ) { 
  282.                                           $hidden[$msg_type]++; 
  283.                                       } 
  284.                                   } else {    // dismiss has expired 
  285.                                       $dismissed_updated = true;    // update the array when done 
  286.                                       unset( $user_dismissed[$payload['dis_key']] ); 
  287.                                   } 
  288.                               } 
  289.                           } 
  290.                           $msg_html .= $this->get_notice_html( $msg_type, $msg_txt, $payload ); 
  291.                           break; 
  292.                   } 
  293.               } 
  294.           } 
  295.  
  296.           // delete all notices for the current user id 
  297.           $this->trunc(); 
  298.  
  299.           // don't save unless we've changed something 
  300.           if ( $dismissed_updated === true && ! empty( $user_id ) ) { 
  301.               if ( empty( $user_dismissed ) ) { 
  302.                   delete_user_option( $user_id, $this->dis_name ); 
  303.               } else { 
  304.                   update_user_option( $user_id, $this->dis_name, $user_dismissed ); 
  305.               } 
  306.           } 
  307.  
  308.           echo "\n"; 
  309.           echo '<!-- '.$this->lca.' admin notices begin -->'."\n"; 
  310.  
  311.           if ( ! empty( $nag_msgs ) ) { 
  312.               echo $this->get_nag_style(); 
  313.               echo $this->get_notice_html( 'nag', $nag_msgs ); 
  314.           } 
  315.  
  316.           // remind the user that there are hidden error messages 
  317.           foreach ( array( 
  318.               'err' => _x( 'error', 'notification type', $this->text_dom ),  
  319.               'warn' => _x( 'warning', 'notification type', $this->text_dom ),  
  320.  ) as $msg_type => $log_name ) { 
  321.               if ( empty( $hidden[$msg_type] ) ) { 
  322.                   continue; 
  323.               } elseif ( $hidden[$msg_type] > 1 ) { 
  324.                   $msg_text = __( '%1$d important %2$s notices have been hidden and/or dismissed — <a id="%3$s">unhide and view the %2$s messages</a>.', $this->text_dom ); 
  325.               } else { 
  326.                   $msg_text = __( '%1$d important %2$s notice has been hidden and/or dismissed — <a id="%3$s">unhide and view the %2$s message</a>.', $this->text_dom ); 
  327.               } 
  328.               echo $this->get_notice_html( $msg_type, sprintf( $msg_text, $hidden[$msg_type], $log_name, $this->lca.'-unhide-notices' ) ); 
  329.           } 
  330.  
  331.           echo $msg_html; 
  332.           echo '<!-- '.$this->lca.' admin notices end -->'."\n"; 
  333.       }

  338.  
  339.           if ( empty( $user_id ) || ! current_user_can( 'edit_user', $user_id ) ) {

  342.

  348.           } 
  349.  
  350.           if ( empty( $dis_info['key'] ) ) { 
  351.               die( '-1' ); 
  352.           } 
  353.  
  354.           // site specific user options

  357.           if ( ! is_array( $user_dismissed ) ) { 
  358.               $user_dismissed = array(); 
  359.           } 
  360.  
  361.           // save the message id and expiration time (0 = never) 
  362.           $user_dismissed[$dis_info['key']] = empty( $dis_info['time'] ) || 
  363.               ! is_numeric( $dis_info['time'] ) ? 0 : time() + $dis_info['time']; 
  364.

  379.       public function is_dismissed( $dis_key = false, $user_id = true ) { 
  380.  
  381.           if ( empty( $dis_key ) || ! $this->can_dismiss() ) {    // just in case

  384.  
  385.           if ( $user_id === true ) { 
  386.               $user_id = (int) get_current_user_id(); 
  387.           } else { 
  388.               $user_id = (int) $user_id;    // false = 0 
  389.           } 
  390.  
  391.           if ( empty( $user_id ) ) { 
  392.               return false; 
  393.           } 
  394.  
  395.           $user_dismissed = get_user_option( $this->dis_name, $user_id ); 
  396.  
  397.           if ( ! is_array( $user_dismissed ) ) { 
  398.               return false; 
  399.           } 
  400.  
  401.           // notice has been dismissed 
  402.           if ( isset( $user_dismissed[$dis_key] ) ) { 
  403.  
  404.               $now_time = time(); 
  405.               $dis_time = $user_dismissed[$dis_key]; 
  406.  
  407.               if ( empty( $dis_time ) || $dis_time > $now_time ) { 
  408.                   return true; 
  409.  
  410.               // dismiss time has expired 
  411.               } else { 
  412.                   unset( $user_dismissed[$dis_key] ); 
  413.                   if ( empty( $user_dismissed ) ) { 
  414.                       delete_user_option( $user_id, $this->dis_name ); 
  415.                   } else { 
  416.                       update_user_option( $user_id, $this->dis_name, $user_dismissed ); 
  417.                   } 
  418.               } 
  419.           } 
  420.  
  421.           return false;

  453.       private function get_notice_html( $msg_type, $msg_txt, $payload = array() ) { 
  454.  
  455.           $charset = get_bloginfo( 'charset' ); 
  456.  
  457.           if ( ! isset( $payload['label'] ) ) { 
  458.               $payload['label'] = sprintf( __( '%s Note', $this->text_dom ), strtoupper( $this->lca ) ); 
  459.           } 
  460.  
  461.           switch ( $msg_type ) { 
  462.               case 'nag': 
  463.                   $payload['label'] = ''; 
  464.                   $msg_class = 'update-nag'; 
  465.                   break; 
  466.               case 'warn': 
  467.                   $msg_class = 'notice notice-warning'; 
  468.                   break; 
  469.               case 'err': 
  470.                   $msg_class = 'notice notice-error error'; 
  471.                   break; 
  472.               case 'upd': 
  473.                   $msg_class = 'notice notice-success updated'; 
  474.                   break; 
  475.               case 'inf': 
  476.               default: 
  477.                   $msg_type = 'inf';    // just in case 
  478.                   $msg_class = 'notice notice-info'; 
  479.                   break; 
  480.           } 
  481.  
  482.           // dis_key and dis_time must have values to create a dismissible notice 
  483.           $is_dismissible = empty( $payload['dis_key'] ) || empty( $payload['dis_time'] ) ? false : true; 
  484.  
  485.           $css_id_attr = empty( $payload['dis_key'] ) ? '' : ' id="'.$msg_type.'_'.$payload['dis_key'].'"'; 
  486.  
  487.           $data_attr = ! $is_dismissible ? 
  488.               '' : ' data-dismiss-nonce="'.wp_create_nonce( __FILE__ ).'"'. 
  489.                   ' data-dismiss-key="'.$payload['dis_key'].'"'. 
  490.                   ' data-dismiss-time="'.( is_numeric( $payload['dis_time'] ) ? 
  491.                       $payload['dis_time'] : 0 ).'"'; 
  492.  
  493.           // optionally hide notices if required 
  494.           $style_attr = ' style="'. 
  495.               ( empty( $payload['style'] ) ? '' : $payload['style'] ). 
  496.               ( empty( $payload['hidden'] ) ? 'display:block !important; visibility:visible !important;' : 'display:none;' ).'"'; 
  497.  
  498.           $msg_html = '<div class="'.$this->lca.'-notice '. 
  499.               ( ! $is_dismissible ? '' : $this->lca.'-dismissible ' ). 
  500.                   $msg_class.'"'.$css_id_attr.$style_attr.$data_attr.'>';    // display block or none 
  501.  
  502.           if ( ! empty( $payload['dis_time'] ) ) { 
  503.               $msg_html .= '<div class="notice-dismiss"><div class="notice-dismiss-text">'. 
  504.                   __( 'Dismiss', $this->text_dom ).'</div></div>'; 
  505.           } 
  506.  
  507.           if ( ! empty( $payload['label'] ) ) { 
  508.               $msg_html .= '<div class="notice-label">'. 
  509.                   $payload['label'].'</div>'; 
  510.           } 
  511.  
  512.           $msg_html .= '<div class="notice-message">'.$msg_txt.'</div>';
