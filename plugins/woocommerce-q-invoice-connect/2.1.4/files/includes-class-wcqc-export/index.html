  641.       /** 
  642.        * Get the tax rates/percentages for a given tax class 
  643.        * @param  string $tax_class tax class slug 
  644.        * @return string $tax_rates imploded list of tax rates 
  645.        */ 
  646.       public function get_tax_rate( $tax_class, $line_total, $line_tax ) { 
  647.           if ( version_compare( WOOCOMMERCE_VERSION, '2.1' ) >= 0 ) { 
  648.               // WC 2.1 or newer is used 
  649.               if ( $line_tax == 0 ) { 
  650.                   return '-'; // no need to determine tax rate... 
  651.               } 
  652.  
  653.               // if (empty($tax_class)) 
  654.               // $tax_class = 'standard';// does not appear to work anymore - get_rates does accept an empty tax_class though! 
  655.                
  656.               $tax = new WC_Tax(); 
  657.               $taxes = $tax->get_rates( $tax_class ); 
  658.  
  659.               $tax_rates = array(); 
  660.  
  661.               foreach ($taxes as $tax) { 
  662.                   $tax_rates[$tax['label']] = round( $tax['rate'], 2 ); 
  663.               } 
  664.  
  665.               if (empty($tax_rates)) { 
  666.                   // one last try: manually calculate 
  667.                   if ( $line_total != 0) { 
  668.                       $tax_rates[] = round( ($line_tax / $line_total)*100, 1 ); 
  669.                   } else { 
  670.                       $tax_rates[] = '-'; 
  671.                   } 
  672.               } 
  673.  
  674.               $tax_rates = implode(' , ', $tax_rates ); 
  675.           } else { 
  676.               // Backwards compatibility: calculate tax from line items 
  677.               if ( $line_total != 0) { 
  678.                   $tax_rates = round( ($line_tax / $line_total)*100, 1 ); 
  679.               } else { 
  680.                   $tax_rates = '-'; 
  681.               } 
  682.           } 
  683.            
  684.           return $tax_rates; 
  685.       }
