  510.       public function text_element_callback( $args ) { 
  511.           $menu = $args['menu']; 
  512.           $id = $args['id']; 
  513.           $size = isset( $args['size'] ) ? $args['size'] : '25'; 
  514.           $disabled = isset( $args['disabled'] ) ? "disabled='". $args['disabled'] ."'" : ""; 
  515.        
  516.           $options = get_option( $menu ); 
  517.        
  518.           if ( isset( $options[$id] ) ) { 
  519.               $current = $options[$id]; 
  520.           } else { 
  521.               $current = isset( $args['default'] ) ? $args['default'] : ''; 
  522.           } 
  523.        
  524.           $html = sprintf( '<input type="text" id="%1$s" name="%2$s[%1$s]" value="%3$s" size="%4$s" %5$s/>', $id, $menu, $current, $size, $disabled ); 
  525.        
  526.           // Displays option description. 
  527.           if ( isset( $args['description'] ) ) { 
  528.               $html .= sprintf( '<span class="description">%s</span>', $args['description'] ); 
  529.           } 
  530.        
  531.           echo $html;
