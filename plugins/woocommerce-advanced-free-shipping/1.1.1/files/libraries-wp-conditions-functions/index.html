  153. if ( ! function_exists( 'wpc_match_conditions' ) ) { 
  154.  
  155.   /** 
  156.    * Match conditions. 
  157.    * 
  158.    * Check if conditions match, if all conditions in one condition group 
  159.    * matches it will return TRUE and the fee will be applied. 
  160.    * 
  161.    * @since 1.0.0 
  162.    * 
  163.    * @param  array $condition_groups List of condition groups containing their conditions. 
  164.    * @param array $args Arguments to pass to the matching method. 
  165.    * @return BOOL TRUE if all the conditions in one of the condition groups matches true. 
  166.    */ 
  167.   function wpc_match_conditions( $condition_groups = array(), $args = array() ) { 
  168.  
  169.       if ( empty( $condition_groups ) || ! is_array( $condition_groups ) ) : 
  170.           return false; 
  171.       endif; 
  172.  
  173.       foreach ( $condition_groups as $condition_group => $conditions ) : 
  174.  
  175.           $match_condition_group = true; 
  176.  
  177.           foreach ( $conditions as $condition ) : 
  178.  
  179.               $condition = apply_filters( 'wp-conditions\condition', $condition ); // BC helper 
  180.               $wpc_condition = wpc_get_condition( $condition['condition'] ); 
  181.  
  182.               // Match the condition - pass any custom ($)args as parameters. 
  183.               $match = call_user_func_array( array( $wpc_condition, 'match' ), array( false, $condition['operator'], $condition['value'], $args ) ); 
  184.  
  185.               // Filter the matched result - BC helper 
  186.               $parameters = array( 'wp-conditions\condition\match', $match, $condition['condition'], $condition['operator'], $condition['value'], $args ); 
  187.               $match = call_user_func_array( 'apply_filters', $parameters ); 
  188.  
  189.               // Original - simple - way 
  190. //                $match = $wpc_condition->match( false, $condition['operator'], $condition['value'] ); 
  191. //                $match = apply_filters( 'wp-conditions\condition\match', $match, $condition['condition'], $condition['operator'], $condition['value'] ); 
  192.  
  193.               if ( false == $match ) : 
  194.                   $match_condition_group = false; 
  195.               endif; 
  196.  
  197.           endforeach; 
  198.  
  199.           // return true if one condition group matches 
  200.           if ( true == $match_condition_group ) : 
  201.               return true; 
  202.           endif; 
  203.  
  204.       endforeach; 
  205.  
  206.       return false; 
  207.  
  208.   }

  213. if ( ! function_exists( 'wpc_sanitize_conditions' ) ) { 
  214.  
  215.   /** 
  216.    * Sanitize conditions. 
  217.    * 
  218.    * Go over all the conditions and sanitize them. Used when the conditions are being saved. 
  219.    * 
  220.    * @since 1.0.0 
  221.    * 
  222.    * @param  array $conditions The list of conditions. 
  223.    * @return array 
  224.    */ 
  225.   function wpc_sanitize_conditions( $conditions ) { 
  226.  
  227.       $sanitized_conditions = array(); 
  228.       foreach ( $conditions as $group_key => $condition_group ) : 
  229.           if ( $group_key == '9999' ) continue; // Template group 
  230.  
  231.           foreach ( $condition_group as $condition_id => $condition_values ) : 
  232.               if ( $condition_id == '9999' ) continue; // Template condition 
  233.               if ( ! isset( $condition_values['value'] ) ) $condition_values['value'] = ''; 
  234.  
  235.               foreach ( $condition_values as $condition_key => $condition_value ) : 
  236.  
  237.                   switch ( $condition_key ) : 
  238.  
  239.                       default : 
  240.                           $condition_value = sanitize_text_field( $condition_value ); 
  241.                           break; 
  242.  
  243.                       case 'condition' : 
  244.                           $condition_value = sanitize_key( $condition_value ); 
  245.                           break; 
  246.  
  247.                       case 'operator' : 
  248.                           $condition_value = in_array( $condition_value, array( '==', '!=', '>=', '<=' ) ) ? $condition_value : '=='; 
  249.                           break; 
  250.  
  251.                       case 'value' : 
  252.                           if ( is_array( $condition_value ) ) : 
  253.                               $condition_value = array_map( 'sanitize_text_field', $condition_value ); 
  254.                           elseif ( is_string( $condition_value ) ) : 
  255.                               $condition_value = sanitize_text_field( $condition_value ); 
  256.                           endif; 
  257.                           break; 
  258.  
  259.                   endswitch; 
  260.  
  261.                   $sanitized_conditions[ $group_key ][ $condition_id ][ $condition_key ] = $condition_value; 
  262.  
  263.               endforeach; 
  264.  
  265.           endforeach; 
  266.  
  267.       endforeach; 
  268.  
  269.       return $sanitized_conditions; 
  270.  
  271.   }
