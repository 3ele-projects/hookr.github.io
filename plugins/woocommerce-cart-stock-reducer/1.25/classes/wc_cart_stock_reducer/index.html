  458.                     $info[ 'class' ]        = 'out-of-stock'; 
  459.                 } 
  460.             } 
  461.         } 
  462.  
  463.         return $info; 
  464.     }

  466.     public function replace_stock_pending_text( $pending_text, $info = null, $product = null ) { 
  467.  
  468.         if ( null != $product && $item = $this->item_managing_stock( $product->id, $product->variation_id ) ) { 
  469.             if ( !empty( $this->stock_pending_include_cart_items ) && $this->items_in_cart( $item ) ) { 
  470.                 // Only append text if enabled and there are items actually in this users cart 
  471.                 $pending_include_cart_items = str_ireplace( '%CSR_NUM_ITEMS%', $this->items_in_cart( $item ), $this->stock_pending_include_cart_items ); 
  472.                 $pending_text .= ' ' . $pending_include_cart_items; 
  473.             } 
  474.  
  475.             if ( !empty( $this->stock_pending_expire_time ) && $this->expiration_time_cache( $item ) ) { 
  476.                 // Only append text if enabled and there are items that will expire 
  477.                 $pending_expire_text = str_ireplace( '%CSR_EXPIRE_TIME%', human_time_diff( time(), $this->expiration_time_cache( $item ) ), $this->stock_pending_expire_time ); 
  478.                 // Was really hoping to use the jquery countdown here but the default WooCommerce templates 
  479.                 // call esc_html so I can't easily include a class here :( 
  480.                 $pending_text .= ' ' . $pending_expire_text; 
  481.             } 
  482.         } 
  483.  
  484.         return $pending_text; 
  485.     }

  488.         if ( isset( $this->expiration_time_cache[ $item_id ] ) ) { 
  489.             return $this->expiration_time_cache[ $item_id ]; 
  490.         } 
  491.         return false;

  494.     private function items_in_cart( $item_id ) { 
  495.         $count = 0; 
  496.         if ( $cart = WC()->cart ) { 
  497.             foreach ($cart->cart_contents as $cart_id => $item) { 
  498.                 if ( $item_id === $item[ 'product_id' ] || $item_id === $item[ 'variation_id' ] ) { 
  499.                     $count += $item[ 'quantity' ]; 
  500.                 } 
  501.             } 
  502.         } 
  503.         return $count; 
  504.     }

  506.     /** 
  507.      * Get the quantity available of a specific item 
  508.      * 
  509.      * @param int $item The item ID 
  510.      * @param object $product WooCommerce WC_Product based class, if not passed the item ID will be used to query 
  511.      * @param string $ignore Cart Item Key to ignore in the count 
  512.      * 
  513.      * @return int Quantity of items in stock 
  514.      */ 
  515.     public function get_stock_available( $product_id, $variation_id = null, $product = null, $ignore = false ) { 
  516.         $stock = 0; 
  517.  
  518.         $id = $this->item_managing_stock( $product_id, $variation_id ); 
  519.  
  520.         if ( null === $product ) { 
  521.             $product = wc_get_product( $id ); 
  522.         } 
  523.         if ( isset( $this->pseudo_stock[ $id ] ) ) { 
  524.             $stock = $this->pseudo_stock[ $id ]; 
  525.         } else { 
  526.             $stock = $product->get_total_stock(); 
  527.             // Cache our retrieved stock value or else we will decrement the "fake" stock count every time this is called 
  528.             $this->pseudo_stock[ $id ] = $stock; 
  529.         } 
  530.  
  531.         if ( false !== $id ) { 
  532.             if ( $id === $variation_id ) { 
  533.                 $product_field = 'variation_id'; 
  534.             } else { 
  535.                 $product_field = 'product_id'; 
  536.             } 
  537.  
  538.             // The minimum quantity of stock to have in order to skip checking carts.  This should be higher than the amount you expect could sell before the carts expire. 
  539.             // Originally was a configuration variable, but this is such an advanced option I thought it would be better as a filter. 
  540.             // Plus you can use some math to make this decision 
  541.             $min_no_check = apply_filters( 'wc_csr_min_no_check', false, $id, $stock ); 
  542.             if ( false != $min_no_check && $min_no_check < (int) $stock ) { 
  543.                 // Don't bother searching through all the carts if there is more than 'min_no_check' quantity 
  544.                 return $stock; 
  545.             } 
  546.  
  547.             $in_carts = $this->quantity_in_carts( $id, $product_field, $ignore ); 
  548.             if ( 0 < $in_carts ) { 
  549.                 $stock = ( $stock - $in_carts ); 
  550.                 // Trick WooCommerce into thinking there is no stock available, this does NOT get updated in the DB 
  551.                 $product->stock = $stock; 
  552.             } 
  553.         } 
  554.         return $stock; 
  555.     }

  557.     /** 
  558.      * Search through all sessions and count quantity of $item in all carts 
  559.      * 
  560.      * @param int $item WooCommerce item ID 
  561.      * @param string $field Which field to use to match.  'variation_id' or 'product_id' 
  562.      * @param bool $ignore true if active users count should be ignored 
  563.      * 
  564.      * @return int Total number of items 
  565.      */ 
  566.     public function quantity_in_carts( $item, $field = 'product_id', $ignore = false ) { 
  567.         global $wpdb; 
  568.         $quantity = 0; 
  569.         $item = (int) $item; 
  570.  
  571.         /** This should be the most efficient way to do this, though the proper way 
  572.          * would be to retrieve a list of all the session and use get_option or 
  573.          * the WooCommerce API to iterate through the sessions.  Though it is possible 
  574.          * that using get_option and an external cache will be faster on a heavy site. 
  575.          * Need to benchmark... In my free time. 
  576.         */ 
  577.         if ( version_compare( WOOCOMMERCE_VERSION, '2.5' ) >= 0 ) { 
  578.             // WooCommerce >= 2.5 stores session data in a separate table 
  579.             $results = $wpdb->get_results( "SELECT session_key, session_value FROM {$wpdb->prefix}woocommerce_sessions WHERE session_value LIKE '%\"{$field}\";i:{$item};%'", OBJECT ); 
  580.         } else { 
  581.             $results = $wpdb->get_results( "SELECT option_name, option_value as session_value FROM {$wpdb->options} WHERE option_name LIKE '_wc_session_%' AND option_value LIKE '%\"{$field}\";i:{$item};%'", OBJECT ); 
  582.         } 
  583.  
  584.         if ( $results ) { 
  585.             $WC = WC(); 
  586.             if ( isset( $WC, $WC->session ) ) { 
  587.                 // A user report a fatal error when trying to call get_customer_id. 
  588.                 // Even though it was likely some other plugin/themes fault, lets play safely. 
  589.                 $customer_id = $WC->session->get_customer_id(); 
  590.             } else { 
  591.                 $customer_id = null; 
  592.             } 
  593.             foreach ( $results as $result ) { 
  594.                 if ( null !== $customer_id && 
  595.                     ( isset( $result->session_key ) && $result->session_key === $customer_id ) || 
  596.                     ( isset( $result->option_name ) && $result->option_name === '_wc_session_' . $customer_id ) ) { 
  597.                     $row_in_own_cart = true; 
  598.                 } else { 
  599.                     $row_in_own_cart = false; 
  600.                 } 
  601.                 if ( true === $ignore && true === $row_in_own_cart ) { 
  602.                     continue; 
  603.                 } 
  604.                 $session = unserialize( $result->session_value ); 
  605.                 if ( isset( $session[ 'cart' ] ) ) { 
  606.                     $cart = unserialize( $session[ 'cart' ] ); 
  607.                     foreach ( $cart as $key => $row ) { 
  608.                         if ( isset( $row[ 'csr_expire_time' ] ) && $this->is_expired( $row[ 'csr_expire_time' ] ) ) { 
  609.                             // Skip items that are expired in carts 
  610.                             continue; 
  611.                         } 
  612.                         if ( $item === $row[ $field ] ) { 
  613.                             $quantity += $row[ 'quantity' ]; 
  614.                         } 
  615.                         if ( isset( $row[ 'csr_expire_time' ] ) && false === $row_in_own_cart ) { 
  616.                             // Don't track expiration time of items in the users own cart 
  617.                             if ( ! isset( $this->expiration_time_cache[ $item ] ) || $row[ 'csr_expire_time' ] < $this->expiration_time_cache[ $item ] ) { 
  618.                                 // Cache the earliest time an item is expiring 
  619.                                 $this->expiration_time_cache[ $item ] = $row[ 'csr_expire_time' ]; 
  620.                             } 
  621.                         } 
  622.                     } 
  623.                 } 
  624.  
  625.             } 
  626.         } 
  627.  
  628.         return $quantity; 
  629.     }

  631.     /** 
  632.      * Check if $expire_time has passed 
  633.      * 
  634.      * @param int|string $expire_time UNIX timestamp for expiration or 'never' if item never expires 
  635.      * 
  636.      * @return bool true if expired 
  637.      */ 
  638.     protected function is_expired( $expire_time = 'never' ) { 
  639.         $expired = false; 
  640.         if ( 'never' === $expire_time ) { 
  641.             // This should never happen, but better to be safe 
  642.             $expired = false; 
  643.         } elseif ( $expire_time < time() ) { 
  644.             $expired = true; 
  645.         } 
  646.         return $expired; 
  647.     }

  650.     /** 
  651.      * Validate and sanitize the 'expire_time_field' 
  652.      * 
  653.      * @param string $key Field name to validate 
  654.      * 
  655.      * @return string 
  656.      */ 
  657.     public function validate_expire_time_field( $key ) { 
  658.         $expire_time = sanitize_text_field( $_POST[ $this->plugin_id . $this->id . '_' . $key ] ); 
  659.         $expire_enabled_key = $this->plugin_id . $this->id . '_expire_items'; 
  660.         $expire_enabled = isset( $_POST[ $expire_enabled_key ] ) ? absint( $_POST[ $expire_enabled_key ] ) : 0; 
  661.  
  662.         if ( !empty( $expire_time ) ) { 
  663.             $time = strtotime( $expire_time ); 
  664.             if ( ! $time ) { 
  665.                 $this->errors[] = sprintf( __( 'Invalid Expire Time: %s', 'woocommerce-cart-stock-reducer' ), $expire_time ); 
  666.             } elseif ( false !== $time && $time < time() ) { 
  667.                 $this->errors[] = sprintf( __( 'Cannot set Expire Time that would be in the past: %s', 'woocommerce-cart-stock-reducer' ), $expire_time ); 
  668.             } 
  669.         } elseif ( 1 === $expire_enabled ) { 
  670.             $this->errors[] = sprintf( __( 'Expire time must be set if expiration is enabled', 'woocommerce-cart-stock-reducer' ) ); 
  671.         } 
  672.  
  673.         return $expire_time; 
  674.     }
