  620.             }

  626.             } elseif ( $minimum_amount < 2000 || $minimum_amount > 500000 ) {

  629.             }

  631.             // Validate GUARANTEE maximum amount. 
  632.             if ( ! ctype_digit( $maximum_amount ) ) {

  634.                 wc_novalnet_safe_redirect( admin_url( 'admin.php?page=wc-settings&tab=checkout§ion=' . $request ['section'] ) );
