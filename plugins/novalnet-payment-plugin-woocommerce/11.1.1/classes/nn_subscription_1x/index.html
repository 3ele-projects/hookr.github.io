  499.     /** 
  500.      * Reactivate the subscription process. 
  501.      * 
  502.      * @since 11.0.0 
  503.      * @param boolean         $can_update   For process reactivate action. 
  504.      * @param WC_Subscription $subscription The subscription object. 
  505.      * @param WC_Order        $wc_order     The order object. 
  506.      * 
  507.      * @return boolean 
  508.      */ 
  509.     public function reactivate_subscription_process( $can_update, $subscription, $wc_order ) { 
  510.  
  511.         $request = $_REQUEST; // input var okay. 
  512.  
  513.         // Checks Novalnet payment. 
  514.         if ( $can_update && wc_novalnet_check_string( $wc_order->payment_method ) && ! empty( $request ['new_status'] ) && 'active' === $request ['new_status'] ) { // input var okay. 
  515.  
  516.             // Return the status if subscription already updated. 
  517.             if ( get_post_meta( $wc_order->id, '_nn_subscription_updated', true ) ) { 
  518.                 return $can_update; 
  519.             } 
  520.  
  521.             // Get transaction details. 
  522.             $transaction_details = wc_novalnet_get_transaction_details( $wc_order->id, 'subscription' ); 
  523.             $period = $subscription ['period']; 
  524.             $interval = $subscription ['interval']; 
  525.  
  526.             // Get subscription details. 
  527.             $subscription_details = wc_novalnet_get_subs_details( '', $wc_order->id ); 
  528.             $formatted_previous_cycle = date( 'Y-m-d', strtotime( $subscription_details ['next_payment_date'] ) ); 
  529.             $previous_next_subs_cycle = strtotime( $formatted_previous_cycle ); 
  530.             $next_subs_cycle = $formatted_previous_cycle; 
  531.             $current_date = strtotime( date( 'Y-m-d' ) ); 
  532.  
  533.             // Form request parameters. 
  534.             $request_parameters = array( 
  535.                 'vendor_id' => $transaction_details ['vendor_id'],  
  536.                 'vendor_authcode' => $transaction_details ['auth_code'],  
  537.                 'product_id' => $transaction_details ['product_id'],  
  538.                 'request_type' => 'SUBSCRIPTION_PAUSE',  
  539.                 'tid' => $transaction_details ['tid'],  
  540.                 'subs_id' => $transaction_details ['subs_id'],  
  541.                 'suspend' => 0,  
  542.  ); 
  543.  
  544.             if ( $previous_next_subs_cycle <= $current_date ) { 
  545.  
  546.                 while ( strtotime( $next_subs_cycle ) <= $current_date ) { 
  547.                     $next_subs_cycle = date( 'Y-m-d', strtotime( $next_subs_cycle . '+' . $interval . ' ' . $period ) ); 
  548.                 } 
  549.  
  550.                 if ( strtotime( $next_subs_cycle ) > $current_date ) { 
  551.  
  552.                     // Calculate date difference. 
  553.                     $difference = date_diff( date_create( $formatted_previous_cycle ), date_create( $next_subs_cycle ) ); 
  554.  
  555.                     if ( $difference->days > 0 ) { 
  556.                         $request_parameters ['pause_period'] = $difference->days; 
  557.                         $request_parameters ['pause_time_unit'] = 'd'; 
  558.                     } 
  559.                 } 
  560.             } 
  561.  
  562.             // Submit XML call. 
  563.             $response = novalnet_instance()->novalnet_functions()->perform_xmlrequest( $request_parameters ); 
  564.  
  565.             // Log for subscription reactive process. 
  566.             $this->maintain_debug_log( 'Subscription reactive call initiated for the order id ' . $wc_order->id . ' and the status was ' . $response ['status'] ); 
  567.             if ( wc_novalnet_status_check( $response ) ) { 
  568.                 $next_payment_date = wc_novalnet_next_subscription_date( $response ); 
  569.  
  570.                 $this->update_next_payment_date( 
  571.                     $next_payment_date, $wc_order, array( 
  572.                     'subscription_key' => $subscription ['order_id'] . '_' . $subscription ['product_id'],  
  573.  ) 
  574.  ); 
  575.  
  576.                 // Update Next payment date in Novalnet subscription table. 
  577.                 wc_novalnet_db_update_query( 
  578.                     array( 
  579.                     'next_payment_date' => $next_payment_date,  
  580.  ), array( 
  581.                     'order_no' => $wc_order->id,  
  582.  ), 'novalnet_subscription_details' 
  583.  ); 
  584.  
  585.                 novalnet_instance()->novalnet_functions()->update_comments( $wc_order, wc_novalnet_format_text( sprintf( __( 'Subscription has been successfully activated on %s', 'wc-novalnet' ), wc_novalnet_formatted_date() ) . wc_novalnet_format_text( __( ' Next charging date: ', 'wc-novalnet' ) . wc_novalnet_formatted_date( $next_payment_date ) ) ) ); 
  586.             } else { 
  587.  
  588.                 // Process subscription error. 
  589.                 $this->subscription_error_process( wc_novalnet_format_text( sprintf( __( 'Subscription activation has been failed due to: %s', 'wc-novalnet' ), wc_novalnet_response_text( $response ) ) ), 'error' ); 
  590.                 return false; 
  591.             } 
  592.  
  593.             // Set value to notify subscription updated. 
  594.             update_post_meta( $wc_order->id, '_nn_subscription_updated', true ); 
  595.         } 
  596.         return $can_update; 
  597.     }

  697.     public function form_subscription_params( $payment_parameters, $wc_order, $subs_tariff, $subscription_enabled, $is_change_payment = true ) { 
  698.  
  699.         $novalnet_payment = $this->is_novalnet_recurring_payment_method( $wc_order ); 
  700.  
  701.         if ( $is_change_payment && $novalnet_payment ) { 
  702.             $payment_parameters ['amount'] = '0';

  711.             $trial_period = wc_novalnet_calculate_subscription_period( WC_Subscriptions_Order::get_subscription_trial_length( $wc_order ), WC_Subscriptions_Order::get_subscription_trial_period( $wc_order ) ); 
  712.  
  713.             $is_trial_period = empty( $trial_period ); 
  714.  
  715.             if ( $this->get_order_subscription_length( $wc_order ) !== '1' || ! $is_trial_period ) { 
  716.  
  717.                 // Calculate recurring period. 
  718.                 $recurring_period = wc_novalnet_calculate_subscription_period( WC_Subscriptions_Order::get_subscription_interval( $wc_order ), WC_Subscriptions_Order::get_subscription_period( $wc_order ) ); 
  719.

  721.                 $payment_parameters ['tariff_period'] = $recurring_period; 
  722.                 if ( ! $is_trial_period ) { 
  723.                     $payment_parameters ['tariff_period'] = $trial_period; 
  724.                 } 
  725.                 $payment_parameters ['tariff_period2'] = $recurring_period; 
  726.                 $payment_parameters ['tariff'] = $subs_tariff; 
  727.                 $payment_parameters ['tariff_period2_amount'] = novalnet_instance()->novalnet_functions()->get_recurring_amount_cart( $wc_order ); 
  728.  
  729.                 if ( $is_change_payment ) { 
  730.                     $payment_parameters ['amount'] = '0';

  732.                     $next_payment_date = date( 'Y-m-d', WC_Subscriptions_Manager::get_next_payment_date( $subscription_key, 'timestamp', $wc_order->order_date ) ); 
  733.  
  734.                     // Assign tariff period as days.

  739.                         } 
  740.                     } 
  741.                 } 
  742.             } 
  743.         } 
  744.         return $payment_parameters;
