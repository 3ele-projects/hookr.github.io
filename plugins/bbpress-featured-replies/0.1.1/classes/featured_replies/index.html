  208.         $action = $_POST['do'];

  210.         $actions = array_keys( self::$actions ); 
  211.  
  212.         if ( in_array( $action, $actions ) ) { 
  213.  
  214.             $reply_id = absint( $_POST['reply_id'] ); 
  215.  
  216.             // Verify reply exists 
  217.             if ( ! $reply = bbp_get_reply( $reply_id ) ) { 
  218.                 die(); 
  219.             } 
  220.  
  221.             // Verify whether user has the capability to moderate it 
  222.             if ( ! current_user_can( 'feature_replies', $reply_id ) ) { 
  223.                 die(); 
  224.             } 
  225.  
  226.             // TAKE ACTION! 
  227.             switch ( $action ) { 
  228.  
  229.                 case 'feature': 
  230.                     add_post_meta( $reply_id, 'featured', '1' );

  252.     public function add_reply_actions( $content, $reply_id ) { 
  253.         if ( is_admin() || ! current_user_can( 'feature_replies', $reply_id ) ) {
