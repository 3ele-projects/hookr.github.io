  166.         // $post_id and $post are required 
  167.         if ( empty( $post_id ) || empty( $post ) || self::$saved_meta_boxes ) { 
  168.             return; 
  169.         } 
  170.  
  171.         // Dont' save meta boxes for revisions or autosaves 
  172.         if ( defined( 'DOING_AUTOSAVE' ) || is_int( wp_is_post_revision( $post ) ) || is_int( wp_is_post_autosave( $post ) ) ) { 
  173.             return; 
  174.         } 
  175.  
  176.         // Check the nonce 
  177.         if ( empty( $_POST['woocommerce_meta_nonce'] ) || ! wp_verify_nonce( $_POST['woocommerce_meta_nonce'], 'woocommerce_save_data' ) ) { 
  178.             return; 
  179.         } 
  180.  
  181.         // Check the post being saved == the $post_id to prevent triggering this call for other save_post events 
  182.         if ( empty( $_POST['post_ID'] ) || $_POST['post_ID'] != $post_id ) { 
  183.             return; 
  184.         } 
  185.  
  186.         // Check user has permission to edit 
  187.         if ( ! current_user_can( 'edit_post', $post_id ) ) { 
  188.             return; 
  189.         } 
  190.  
  191.         // We need this save event to run once to avoid potential endless loops. This would have been perfect: 
  192.         // remove_action( current_filter(), __METHOD__ ); 
  193.         // But cannot be used due to https://github.com/woocommerce/woocommerce/issues/6485 
  194.         // When that is patched in core we can use the above. For now: 
  195.         self::$saved_meta_boxes = true; 
  196.  
  197.         // Check the post type 
  198.         if ( in_array( $post->post_type, wc_get_order_types( 'order-meta-boxes' ) ) ) { 
  199.             do_action( 'woocommerce_process_shop_order_meta', $post_id, $post ); 
  200.         } elseif ( in_array( $post->post_type, array( 'product', 'shop_coupon' ) ) ) { 
  201.             do_action( 'woocommerce_process_' . $post->post_type . '_meta', $post_id, $post );
