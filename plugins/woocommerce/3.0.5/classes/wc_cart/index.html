  855.      * @param array $cart_item_data extra cart item data we want to pass into the item 
  856.      * @return string|bool $cart_item_key

  858.     public function add_to_cart( $product_id = 0, $quantity = 1, $variation_id = 0, $variation = array(), $cart_item_data = array() ) {

  860.         try { 
  861.             $product_id = absint( $product_id );

  867.                 $product_id = wp_get_post_parent_id( $variation_id );

  869.  
  870.             // Get the product 
  871.             $product_data = wc_get_product( $variation_id ? $variation_id : $product_id ); 
  872.

  874.             if ( $quantity <= 0 || ! $product_data || 'trash' === $product_data->get_status() ) { 
  875.                 return false; 
  876.             } 
  877.  
  878.             // Load cart item data - may be added by other plugins 
  879.             $cart_item_data = (array) apply_filters( 'woocommerce_add_cart_item_data', $cart_item_data, $product_id, $variation_id ); 
  880.  
  881.             // Generate a ID based on product ID, variation ID, variation data, and other cart item data 
  882.             $cart_id = $this->generate_cart_id( $product_id, $variation_id, $variation, $cart_item_data ); 
  883.  
  884.             // Find the cart item key in the existing cart 
  885.             $cart_item_key = $this->find_product_in_cart( $cart_id ); 
  886.  
  887.             // Force quantity to 1 if sold individually and check for existing item in cart 
  888.             if ( $product_data->is_sold_individually() ) { 
  889.                 $quantity = apply_filters( 'woocommerce_add_to_cart_sold_individually_quantity', 1, $quantity, $product_id, $variation_id, $cart_item_data );

  891.  
  892.                 if ( $in_cart_quantity > 0 ) { 
  893.                     /** translators: %s: product name */ 
  894.                     throw new Exception( sprintf( '<a href="%s" class="button wc-forward">%s</a> %s', wc_get_cart_url(), __( 'View cart', 'woocommerce' ), sprintf( __( 'You cannot add another "%s" to your cart.', 'woocommerce' ), $product_data->get_name() ) ) ); 
  895.                 } 
  896.             } 
  897.  
  898.             // Check product is_purchasable

  904.             if ( ! $product_data->is_in_stock() ) { 
  905.                 throw new Exception( sprintf( __( 'You cannot add "%s" to the cart because the product is out of stock.', 'woocommerce' ), $product_data->get_name() ) ); 
  906.             }

  916.

  919.                         '<a href="%s" class="button wc-forward">%s</a> %s',  
  920.                         wc_get_cart_url(),  
  921.                         __( 'View Cart', 'woocommerce' ),  
  922.                         sprintf( __( 'You cannot add that amount to the cart — we have %1$s in stock and you already have %2$s in your cart.', 'woocommerce' ), wc_format_stock_quantity_for_display( $product_data->get_stock_quantity(), $product_data ), wc_format_stock_quantity_for_display( $products_qty_in_cart[ $product_data->get_stock_managed_by_id() ], $product_data ) ) 
  923.  ) ); 
  924.                 } 
  925.             } 
  926.  
  927.             // If cart_item_key is set, the item is already in the cart 
  928.             if ( $cart_item_key ) { 
  929.                 $new_quantity = $quantity + $this->cart_contents[ $cart_item_key ]['quantity'];

  936.                     'product_id' => $product_id,  
  937.                     'variation_id' => $variation_id,  
  938.                     'variation' => $variation,  
  939.                     'quantity' => $quantity,  
  940.                     'data' => $product_data,  
  941.  ) ), $cart_item_key ); 
  942.             } 
  943.  
  944.             if ( did_action( 'wp' ) ) { 
  945.                 $this->set_cart_cookies( ! $this->is_empty() ); 
  946.             } 
  947.  
  948.             do_action( 'woocommerce_add_to_cart', $cart_item_key, $product_id, $quantity, $variation_id, $variation, $cart_item_data );
