  510.      * @param mixed $ssl (default: null) 
  511.      * @return string 
  512.      */ 
  513.     public function api_request_url( $request, $ssl = null ) { 
  514.         if ( is_null( $ssl ) ) { 
  515.             $scheme = parse_url( home_url(), PHP_URL_SCHEME ); 
  516.         } elseif ( $ssl ) { 
  517.             $scheme = 'https'; 
  518.         } else { 
  519.             $scheme = 'http'; 
  520.         } 
  521.  
  522.         if ( strstr( get_option( 'permalink_structure' ), '/index.php/' ) ) { 
  523.             $api_request_url = trailingslashit( home_url( '/index.php/wc-api/' . $request, $scheme ) ); 
  524.         } elseif ( get_option( 'permalink_structure' ) ) { 
  525.             $api_request_url = trailingslashit( home_url( '/wc-api/' . $request, $scheme ) ); 
  526.         } else { 
  527.             $api_request_url = add_query_arg( 'wc-api', $request, trailingslashit( home_url( '', $scheme ) ) ); 
  528.         } 
  529.  
  530.         return esc_url_raw( apply_filters( 'woocommerce_api_request_url', $api_request_url, $request, $ssl ) ); 
  531.     }
