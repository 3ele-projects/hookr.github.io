  39.  * @param  string $replacement 
  40.  */ 
  41. function wc_deprecated_function( $function, $version, $replacement = null ) { 
  42.   if ( is_ajax() ) { 
  43.       do_action( 'deprecated_function_run', $function, $replacement, $version ); 
  44.       $log_string = "The {$function} function is deprecated since version {$version}."; 
  45.       $log_string .= $replacement ? " Replace with {$replacement}." : ''; 
  46.       error_log( $log_string ); 
  47.   } else { 
  48.       _deprecated_function( $function, $version, $replacement ); 
  49.   } 
  50. }

  889. /** 
  890.  * Save product price. 
  891.  * 
  892.  * This is a private function (internal use ONLY) used until a data manipulation api is built. 
  893.  * 
  894.  * @deprecated 3.0.0 
  895.  * @param int $product_id 
  896.  * @param float $regular_price 
  897.  * @param float $sale_price 
  898.  * @param string $date_from 
  899.  * @param string $date_to 
  900.  */ 
  901. function _wc_save_product_price( $product_id, $regular_price, $sale_price = '', $date_from = '', $date_to = '' ) { 
  902.   wc_doing_it_wrong( '_wc_save_product_price()', 'This function is not for developer use and is deprecated.', '3.0' ); 
  903.  
  904.   $product_id = absint( $product_id ); 
  905.   $regular_price = wc_format_decimal( $regular_price ); 
  906.   $sale_price = '' === $sale_price ? '' : wc_format_decimal( $sale_price ); 
  907.   $date_from = wc_clean( $date_from ); 
  908.   $date_to = wc_clean( $date_to ); 
  909.  
  910.   update_post_meta( $product_id, '_regular_price', $regular_price ); 
  911.   update_post_meta( $product_id, '_sale_price', $sale_price ); 
  912.  
  913.   // Save Dates 
  914.   update_post_meta( $product_id, '_sale_price_dates_from', $date_from ? strtotime( $date_from ) : '' ); 
  915.   update_post_meta( $product_id, '_sale_price_dates_to', $date_to ? strtotime( $date_to ) : '' ); 
  916.  
  917.   if ( $date_to && ! $date_from ) { 
  918.       $date_from = strtotime( 'NOW', current_time( 'timestamp' ) ); 
  919.       update_post_meta( $product_id, '_sale_price_dates_from', $date_from ); 
  920.   } 
  921.  
  922.   // Update price if on sale 
  923.   if ( '' !== $sale_price && '' === $date_to && '' === $date_from ) { 
  924.       update_post_meta( $product_id, '_price', $sale_price ); 
  925.   } else { 
  926.       update_post_meta( $product_id, '_price', $regular_price ); 
  927.   } 
  928.  
  929.   if ( '' !== $sale_price && $date_from && strtotime( $date_from ) < strtotime( 'NOW', current_time( 'timestamp' ) ) ) { 
  930.       update_post_meta( $product_id, '_price', $sale_price ); 
  931.   } 
  932.  
  933.   if ( $date_to && strtotime( $date_to ) < strtotime( 'NOW', current_time( 'timestamp' ) ) ) { 
  934.       update_post_meta( $product_id, '_price', $regular_price ); 
  935.       update_post_meta( $product_id, '_sale_price_dates_from', '' ); 
  936.       update_post_meta( $product_id, '_sale_price_dates_to', '' ); 
  937.   } 
  938. }
