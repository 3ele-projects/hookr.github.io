  516.     /** 
  517.      * Helper method to get customer user objects 
  518.      * 
  519.      * Note that WP_User_Query does not have built-in pagination so limit & offset are used to provide limited 
  520.      * pagination support 
  521.      * 
  522.      * The filter for role can only be a single role in a string. 
  523.      * 
  524.      * @since 2.3 
  525.      * @param array $args request arguments for filtering query 
  526.      * @return WP_User_Query 
  527.      */ 
  528.     private function query_customers( $args = array() ) { 
  529.  
  530.         // default users per page 
  531.         $users_per_page = get_option( 'posts_per_page' ); 
  532.  
  533.         // Set base query arguments 
  534.         $query_args = array( 
  535.             'fields' => 'ID',  
  536.             'role' => 'customer',  
  537.             'orderby' => 'registered',  
  538.             'number' => $users_per_page,  
  539.  ); 
  540.  
  541.         // Custom Role 
  542.         if ( ! empty( $args['role'] ) ) { 
  543.             $query_args['role'] = $args['role']; 
  544.         } 
  545.  
  546.         // Search 
  547.         if ( ! empty( $args['q'] ) ) { 
  548.             $query_args['search'] = $args['q']; 
  549.         } 
  550.  
  551.         // Limit number of users returned 
  552.         if ( ! empty( $args['limit'] ) ) { 
  553.             if ( -1 == $args['limit'] ) { 
  554.                 unset( $query_args['number'] ); 
  555.             } else { 
  556.                 $query_args['number'] = absint( $args['limit'] ); 
  557.                 $users_per_page = absint( $args['limit'] ); 
  558.             } 
  559.         } else { 
  560.             $args['limit'] = $query_args['number']; 
  561.         } 
  562.  
  563.         // Page 
  564.         $page = ( isset( $args['page'] ) ) ? absint( $args['page'] ) : 1; 
  565.  
  566.         // Offset 
  567.         if ( ! empty( $args['offset'] ) ) { 
  568.             $query_args['offset'] = absint( $args['offset'] ); 
  569.         } else { 
  570.             $query_args['offset'] = $users_per_page * ( $page - 1 ); 
  571.         } 
  572.  
  573.         // Created date 
  574.         if ( ! empty( $args['created_at_min'] ) ) { 
  575.             $this->created_at_min = $this->server->parse_datetime( $args['created_at_min'] ); 
  576.         } 
  577.  
  578.         if ( ! empty( $args['created_at_max'] ) ) { 
  579.             $this->created_at_max = $this->server->parse_datetime( $args['created_at_max'] ); 
  580.         } 
  581.  
  582.         // Order (ASC or DESC, ASC by default) 
  583.         if ( ! empty( $args['order'] ) ) { 
  584.             $query_args['order'] = $args['order']; 
  585.         } 
  586.  
  587.         // Orderby 
  588.         if ( ! empty( $args['orderby'] ) ) { 
  589.             $query_args['orderby'] = $args['orderby']; 
  590.  
  591.             // Allow sorting by meta value 
  592.             if ( ! empty( $args['orderby_meta_key'] ) ) { 
  593.                 $query_args['meta_key'] = $args['orderby_meta_key']; 
  594.             } 
  595.         } 
  596.  
  597.         $query = new WP_User_Query( $query_args ); 
  598.  
  599.         // Helper members for pagination headers 
  600.         $query->total_pages = ( -1 == $args['limit'] ) ? 1 : ceil( $query->get_total() / $users_per_page ); 
  601.         $query->page = $page; 
  602.  
  603.         return $query; 
  604.     }

  501.     /** 
  502.      * Helper method to get customer user objects 
  503.      * 
  504.      * Note that WP_User_Query does not have built-in pagination so limit & offset are used to provide limited 
  505.      * pagination support 
  506.      * 
  507.      * The filter for role can only be a single role in a string. 
  508.      * 
  509.      * @since 2.3 
  510.      * @param array $args request arguments for filtering query 
  511.      * @return WP_User_Query 
  512.      */ 
  513.     private function query_customers( $args = array() ) { 
  514.  
  515.         // default users per page 
  516.         $users_per_page = get_option( 'posts_per_page' ); 
  517.  
  518.         // Set base query arguments 
  519.         $query_args = array( 
  520.             'fields' => 'ID',  
  521.             'role' => 'customer',  
  522.             'orderby' => 'registered',  
  523.             'number' => $users_per_page,  
  524.  ); 
  525.  
  526.         // Custom Role 
  527.         if ( ! empty( $args['role'] ) ) { 
  528.             $query_args['role'] = $args['role']; 
  529.  
  530.             // Show users on all roles 
  531.             if ( 'all' === $query_args['role'] ) { 
  532.                 unset( $query_args['role'] ); 
  533.             } 
  534.         } 
  535.  
  536.         // Search 
  537.         if ( ! empty( $args['q'] ) ) { 
  538.             $query_args['search'] = $args['q']; 
  539.         } 
  540.  
  541.         // Limit number of users returned 
  542.         if ( ! empty( $args['limit'] ) ) { 
  543.             if ( -1 == $args['limit'] ) { 
  544.                 unset( $query_args['number'] ); 
  545.             } else { 
  546.                 $query_args['number'] = absint( $args['limit'] ); 
  547.                 $users_per_page = absint( $args['limit'] ); 
  548.             } 
  549.         } else { 
  550.             $args['limit'] = $query_args['number']; 
  551.         } 
  552.  
  553.         // Page 
  554.         $page = ( isset( $args['page'] ) ) ? absint( $args['page'] ) : 1; 
  555.  
  556.         // Offset 
  557.         if ( ! empty( $args['offset'] ) ) { 
  558.             $query_args['offset'] = absint( $args['offset'] ); 
  559.         } else { 
  560.             $query_args['offset'] = $users_per_page * ( $page - 1 ); 
  561.         } 
  562.  
  563.         // Created date 
  564.         if ( ! empty( $args['created_at_min'] ) ) { 
  565.             $this->created_at_min = $this->server->parse_datetime( $args['created_at_min'] ); 
  566.         } 
  567.  
  568.         if ( ! empty( $args['created_at_max'] ) ) { 
  569.             $this->created_at_max = $this->server->parse_datetime( $args['created_at_max'] ); 
  570.         } 
  571.  
  572.         // Order (ASC or DESC, ASC by default) 
  573.         if ( ! empty( $args['order'] ) ) { 
  574.             $query_args['order'] = $args['order']; 
  575.         } 
  576.  
  577.         // Orderby 
  578.         if ( ! empty( $args['orderby'] ) ) { 
  579.             $query_args['orderby'] = $args['orderby']; 
  580.  
  581.             // Allow sorting by meta value 
  582.             if ( ! empty( $args['orderby_meta_key'] ) ) { 
  583.                 $query_args['meta_key'] = $args['orderby_meta_key']; 
  584.             } 
  585.         } 
  586.  
  587.         $query = new WP_User_Query( $query_args ); 
  588.  
  589.         // Helper members for pagination headers 
  590.         $query->total_pages = ( -1 == $args['limit'] ) ? 1 : ceil( $query->get_total() / $users_per_page ); 
  591.         $query->page = $page; 
  592.  
  593.         return $query; 
  594.     }
