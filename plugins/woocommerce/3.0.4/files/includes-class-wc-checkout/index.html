  254.   /** 
  255.    * Create an order. Error codes: 
  256.    *         520 - Cannot insert order into the database. 
  257.    *         521 - Cannot get order after creation. 
  258.    *         522 - Cannot update order. 
  259.    *         525 - Cannot create line item. 
  260.    *         526 - Cannot create fee item. 
  261.    *         527 - Cannot create shipping item. 
  262.    *         528 - Cannot create tax item. 
  263.    *         529 - Cannot create coupon item. 
  264.    * 
  265.    * @throws Exception 
  266.    * @param  $data Posted data. 
  267.    * @return int|WP_ERROR 
  268.    */ 
  269.   public function create_order( $data ) { 
  270.       // Give plugins the opportunity to create an order themselves. 
  271.       if ( $order_id = apply_filters( 'woocommerce_create_order', null, $this ) ) { 
  272.           return $order_id; 
  273.       } 
  274.  
  275.       try { 
  276.           $order_id = absint( WC()->session->get( 'order_awaiting_payment' ) ); 
  277.           $cart_hash = md5( json_encode( wc_clean( WC()->cart->get_cart_for_session() ) ) . WC()->cart->total ); 
  278.           $available_gateways = WC()->payment_gateways->get_available_payment_gateways(); 
  279.  
  280.           /** 
  281.            * If there is an order pending payment, we can resume it here so 
  282.            * long as it has not changed. If the order has changed, i.e. 
  283.            * different items or cost, create a new order. We use a hash to 
  284.            * detect changes which is based on cart items + order total. 
  285.            */ 
  286.           if ( $order_id && ( $order = wc_get_order( $order_id ) ) && $order->has_cart_hash( $cart_hash ) && $order->has_status( array( 'pending', 'failed' ) ) ) { 
  287.               // Action for 3rd parties. 
  288.               do_action( 'woocommerce_resume_order', $order_id ); 
  289.  
  290.               // Remove all items - we will re-add them later. 
  291.               $order->remove_order_items(); 
  292.           } else { 
  293.               $order = new WC_Order(); 
  294.           } 
  295.  
  296.           foreach ( $data as $key => $value ) { 
  297.               if ( is_callable( array( $order, "set_{$key}" ) ) ) { 
  298.                   $order->{"set_{$key}"}( $value ); 
  299.  
  300.               // Store custom fields prefixed with wither shipping_ or billing_. This is for backwards compatibility with 2.6.x. 
  301.               } elseif ( 0 === stripos( $key, 'billing_' ) || 0 === stripos( $key, 'shipping_' ) ) { 
  302.                   $order->update_meta_data( '_' . $key, $value ); 
  303.               } 
  304.           } 
  305.  
  306.           $order->set_created_via( 'checkout' ); 
  307.           $order->set_cart_hash( $cart_hash ); 
  308.           $order->set_customer_id( apply_filters( 'woocommerce_checkout_customer_id', get_current_user_id() ) ); 
  309.           $order->set_currency( get_woocommerce_currency() ); 
  310.           $order->set_prices_include_tax( 'yes' === get_option( 'woocommerce_prices_include_tax' ) ); 
  311.           $order->set_customer_ip_address( WC_Geolocation::get_ip_address() ); 
  312.           $order->set_customer_user_agent( wc_get_user_agent() ); 
  313.           $order->set_customer_note( isset( $data['order_comments'] ) ? $data['order_comments'] : '' ); 
  314.           $order->set_payment_method( isset( $available_gateways[ $data['payment_method'] ] ) ? $available_gateways[ $data['payment_method'] ]  : $data['payment_method'] ); 
  315.           $order->set_shipping_total( WC()->cart->shipping_total ); 
  316.           $order->set_discount_total( WC()->cart->get_cart_discount_total() ); 
  317.           $order->set_discount_tax( WC()->cart->get_cart_discount_tax_total() ); 
  318.           $order->set_cart_tax( WC()->cart->tax_total ); 
  319.           $order->set_shipping_tax( WC()->cart->shipping_tax_total ); 
  320.           $order->set_total( WC()->cart->total ); 
  321.           $this->create_order_line_items( $order, WC()->cart ); 
  322.           $this->create_order_fee_lines( $order, WC()->cart ); 
  323.           $this->create_order_shipping_lines( $order, WC()->session->get( 'chosen_shipping_methods' ), WC()->shipping->get_packages() ); 
  324.           $this->create_order_tax_lines( $order, WC()->cart ); 
  325.           $this->create_order_coupon_lines( $order, WC()->cart ); 
  326.  
  327.           /** 
  328.            * Action hook to adjust order before save. 
  329.            * @since 3.0.0 
  330.            */ 
  331.           do_action( 'woocommerce_checkout_create_order', $order, $data ); 
  332.  
  333.           // Save the order. 
  334.           $order_id = $order->save(); 
  335.  
  336.           do_action( 'woocommerce_checkout_update_order_meta', $order_id, $data ); 
  337.  
  338.           return $order_id; 
  339.       } catch ( Exception $e ) { 
  340.           return new WP_Error( 'checkout-error', $e->getMessage() ); 
  341.       } 
  342.   }

  516.       if ( 'shipping' === $fieldset_key && ( ! $data['ship_to_different_address'] || ! WC()->cart->needs_shipping_address() ) ) { 
  517.           return true; 
  518.       } 
  519.       if ( 'account' === $fieldset_key && ( is_user_logged_in() || ( ! $this->is_registration_required() && empty( $data['createaccount'] ) ) ) ) { 
  520.           return true; 
  521.       } 
  522.       return false; 
  523.   }

  525.   /** 
  526.    * Get posted data from the checkout form. 
  527.    * 
  528.    * @since  3.0.0 
  529.    * @return array of data. 
  530.    */ 
  531.   protected function get_posted_data() { 
  532.       $skipped = array(); 
  533.       $data = array( 
  534.           'terms' => (int) isset( $_POST['terms'] ),  
  535.           'createaccount' => (int) ! empty( $_POST['createaccount'] ),  
  536.           'payment_method' => isset( $_POST['payment_method'] ) ? wc_clean( $_POST['payment_method'] ) : '',  
  537.           'shipping_method' => isset( $_POST['shipping_method'] ) ? wc_clean( $_POST['shipping_method'] ) : '',  
  538.           'ship_to_different_address' => ! empty( $_POST['ship_to_different_address'] ) && ! wc_ship_to_billing_address_only(),  
  539.           'woocommerce_checkout_update_totals' => isset( $_POST['woocommerce_checkout_update_totals'] ),  
  540.  ); 
  541.       foreach ( $this->get_checkout_fields() as $fieldset_key => $fieldset ) { 
  542.           if ( $this->maybe_skip_fieldset( $fieldset_key, $data ) ) { 
  543.               $skipped[] = $fieldset_key; 
  544.               continue; 
  545.           } 
  546.           foreach ( $fieldset as $key => $field ) { 
  547.               $type = sanitize_title( isset( $field['type'] ) ? $field['type'] : 'text' ); 
  548.  
  549.               switch ( $type ) { 
  550.                   case 'checkbox' : 
  551.                       $value = (int) isset( $_POST[ $key ] ); 
  552.                       break; 
  553.                   case 'multiselect' : 
  554.                       $value = isset( $_POST[ $key ] ) ? implode( ', ', wc_clean( $_POST[ $key ] ) ) : ''; 
  555.                       break; 
  556.                   case 'textarea' : 
  557.                       $value = isset( $_POST[ $key ] ) ? wc_sanitize_textarea( $_POST[ $key ] ) : ''; 
  558.                       break; 
  559.                   default : 
  560.                       $value = isset( $_POST[ $key ] ) ? wc_clean( $_POST[ $key ] ) : ''; 
  561.                       break; 
  562.               } 
  563.  
  564.               $data[ $key ] = apply_filters( 'woocommerce_process_checkout_' . $type . '_field', apply_filters( 'woocommerce_process_checkout_field_' . $key, $value ) ); 
  565.  
  566.               // BW compatibility. 
  567.               $this->legacy_posted_data[ $key ] = $data[ $key ]; 
  568.           } 
  569.       } 
  570.  
  571.       if ( in_array( 'shipping', $skipped ) && ( WC()->cart->needs_shipping_address() || wc_ship_to_billing_address_only() ) ) { 
  572.           foreach ( $this->get_checkout_fields( 'shipping' ) as $key => $field ) { 
  573.               $data[ $key ] = isset( $data[ 'billing_' . substr( $key, 9 ) ] ) ? $data[ 'billing_' . substr( $key, 9 ) ] : ''; 
  574.           } 
  575.       } 
  576.  
  577.       return $data; 
  578.   }

  580.   /** 
  581.    * Validates the posted checkout data based on field properties. 
  582.    * 
  583.    * @since  3.0.0 
  584.    * @param  array $data An array of posted data. 
  585.    * @param  WP_Error $errors 
  586.    */ 
  587.   protected function validate_posted_data( &$data, &$errors ) { 
  588.       foreach ( $this->get_checkout_fields() as $fieldset_key => $fieldset ) { 
  589.           if ( $this->maybe_skip_fieldset( $fieldset_key, $data ) ) { 
  590.               continue; 
  591.           } 
  592.           foreach ( $fieldset as $key => $field ) { 
  593.               if ( ! isset( $data[ $key ] ) ) { 
  594.                   continue; 
  595.               } 
  596.               $required = ! empty( $field['required'] ); 
  597.               $format = array_filter( isset( $field['validate'] ) ? (array) $field['validate'] : array() ); 
  598.               $field_label = isset( $field['label'] ) ? $field['label'] : ''; 
  599.  
  600.               switch ( $fieldset_key ) { 
  601.                   case 'shipping' : 
  602.                       /** translators: %s: field name */ 
  603.                       $field_label = sprintf( __( 'Shipping %s', 'woocommerce' ), strtolower( $field_label ) ); 
  604.                   break; 
  605.                   case 'billing' : 
  606.                       /** translators: %s: field name */ 
  607.                       $field_label = sprintf( __( 'Billing %s', 'woocommerce' ), strtolower( $field_label ) ); 
  608.                   break; 
  609.               } 
  610.  
  611.               if ( in_array( 'postcode', $format ) ) { 
  612.                   $country = isset( $data[ $fieldset_key . '_country' ] ) ? $data[ $fieldset_key . '_country' ] : WC()->customer->{"get_{$fieldset_key}_country"}(); 
  613.                   $data[ $key ] = wc_format_postcode( $data[ $key ], $country ); 
  614.  
  615.                   if ( '' !== $data[ $key ] && ! WC_Validation::is_postcode( $data[ $key ], $country ) ) { 
  616.                       $errors->add( 'validation', __( 'Please enter a valid postcode / ZIP.', 'woocommerce' ) ); 
  617.                   } 
  618.               } 
  619.  
  620.               if ( in_array( 'phone', $format ) ) { 
  621.                   $data[ $key ] = wc_format_phone_number( $data[ $key ] ); 
  622.  
  623.                   if ( '' !== $data[ $key ] && ! WC_Validation::is_phone( $data[ $key ] ) ) { 
  624.                       /** translators: %s: phone number */ 
  625.                       $errors->add( 'validation', sprintf( __( '%s is not a valid phone number.', 'woocommerce' ), '<strong>' . $field_label . '</strong>' ) ); 
  626.                   } 
  627.               } 
  628.  
  629.               if ( in_array( 'email', $format ) && '' !== $data[ $key ] ) { 
  630.                   $data[ $key ] = sanitize_email( $data[ $key ] ); 
  631.  
  632.                   if ( ! is_email( $data[ $key ] ) ) { 
  633.                       /** translators: %s: email address */ 
  634.                       $errors->add( 'validation', sprintf( __( '%s is not a valid email address.', 'woocommerce' ), '<strong>' . $field_label . '</strong>' ) ); 
  635.                       continue; 
  636.                   } 
  637.               } 
  638.  
  639.               if ( '' !== $data[ $key ] && in_array( 'state', $format ) ) { 
  640.                   $country = isset( $data[ $fieldset_key . '_country' ] ) ? $data[ $fieldset_key . '_country' ] : WC()->customer->{"get_{$fieldset_key}_country"}(); 
  641.                   $valid_states = WC()->countries->get_states( $country ); 
  642.  
  643.                   if ( ! empty( $valid_states ) && is_array( $valid_states ) && sizeof( $valid_states ) > 0 ) { 
  644.                       $valid_state_values = array_flip( array_map( 'strtolower', $valid_states ) ); 
  645.  
  646.                       // Convert value to key if set 
  647.                       if ( isset( $valid_state_values[ strtolower( $data[ $key ] ) ] ) ) { 
  648.                           $data[ $key ] = $valid_state_values[ strtolower( $data[ $key ] ) ]; 
  649.                       } 
  650.  
  651.                       if ( ! in_array( $data[ $key ], array_keys( $valid_states ) ) ) { 
  652.                           /** translators: 1: state field 2: valid states */ 
  653.                           $errors->add( 'validation', sprintf( __( '%1$s is not valid. Please enter one of the following: %2$s', 'woocommerce' ), '<strong>' . $field_label . '</strong>', implode( ', ', $valid_states ) ) ); 
  654.                       } 
  655.                   } 
  656.               } 
  657.  
  658.               if ( $required && '' === $data[ $key ] ) { 
  659.                   /** translators: %s: field name */ 
  660.                   $errors->add( 'required-field', apply_filters( 'woocommerce_checkout_required_field_notice', sprintf( __( '%s is a required field.', 'woocommerce' ), '<strong>' . $field_label . '</strong>' ), $field_label ) ); 
  661.               } 
  662.           } 
  663.       } 
  664.   }

  666.   /** 
  667.    * Validates that the checkout has enough info to proceed. 
  668.    * 
  669.    * @since  3.0.0 
  670.    * @param  array $data An array of posted data. 
  671.    * @param  WP_Error $errors 
  672.    */ 
  673.   protected function validate_checkout( &$data, &$errors ) { 
  674.       $this->validate_posted_data( $data, $errors ); 
  675.       $this->check_cart_items(); 
  676.  
  677.       if ( empty( $data['woocommerce_checkout_update_totals'] ) && empty( $data['terms'] ) && apply_filters( 'woocommerce_checkout_show_terms', wc_get_page_id( 'terms' ) > 0 ) ) { 
  678.           $errors->add( 'terms', __( 'You must accept our Terms & Conditions.', 'woocommerce' ) ); 
  679.       } 
  680.  
  681.       if ( WC()->cart->needs_shipping() ) { 
  682.           $shipping_country = WC()->customer->get_shipping_country(); 
  683.  
  684.           if ( empty( $shipping_country ) ) { 
  685.               $errors->add( 'shipping', __( 'Please enter an address to continue.', 'woocommerce' ) ); 
  686.           } elseif ( ! in_array( WC()->customer->get_shipping_country(), array_keys( WC()->countries->get_shipping_countries() ) ) ) { 
  687.               $errors->add( 'shipping', sprintf( __( 'Unfortunately <strong>we do not ship %s</strong>. Please enter an alternative shipping address.', 'woocommerce' ), WC()->countries->shipping_to_prefix() . ' ' . WC()->customer->get_shipping_country() ) ); 
  688.           } else { 
  689.               $chosen_shipping_methods = WC()->session->get( 'chosen_shipping_methods' ); 
  690.  
  691.               foreach ( WC()->shipping->get_packages() as $i => $package ) { 
  692.                   if ( ! isset( $chosen_shipping_methods[ $i ], $package['rates'][ $chosen_shipping_methods[ $i ] ] ) ) { 
  693.                       $errors->add( 'shipping', __( 'No shipping method has been selected. Please double check your address, or contact us if you need any help.', 'woocommerce' ) ); 
  694.                   } 
  695.               } 
  696.           } 
  697.       } 
  698.  
  699.       if ( WC()->cart->needs_payment() ) { 
  700.           $available_gateways = WC()->payment_gateways->get_available_payment_gateways(); 
  701.  
  702.           if ( ! isset( $available_gateways[ $data['payment_method'] ] ) ) { 
  703.               $errors->add( 'payment', __( 'Invalid payment method.', 'woocommerce' ) ); 
  704.           } else { 
  705.               $available_gateways[ $data['payment_method'] ]->validate_fields(); 
  706.           } 
  707.       } 
  708.  
  709.       do_action( 'woocommerce_after_checkout_validation', $data, $errors ); 
  710.   }

  816.   /** 
  817.    * Create a new customer account if needed. 
  818.    * @param  array $data 
  819.    * @throws Exception 
  820.    */ 
  821.   protected function process_customer( $data ) { 
  822.       $customer_id = get_current_user_id(); 
  823.  
  824.       if ( ! is_user_logged_in() && ( $this->is_registration_required() || ! empty( $data['createaccount'] ) ) ) { 
  825.           $username = ! empty( $data['account_username'] ) ? $data['account_username'] : ''; 
  826.           $password = ! empty( $data['account_password'] ) ? $data['account_password'] : ''; 
  827.           $customer_id = wc_create_new_customer( $data['billing_email'], $username, $password ); 
  828.  
  829.           if ( is_wp_error( $customer_id ) ) { 
  830.               throw new Exception( $customer_id->get_error_message() ); 
  831.           } 
  832.  
  833.           wp_set_current_user( $customer_id ); 
  834.           wc_set_customer_auth_cookie( $customer_id ); 
  835.  
  836.           // As we are now logged in, checkout will need to refresh to show logged in data 
  837.           WC()->session->set( 'reload_checkout', true ); 
  838.  
  839.           // Also, recalculate cart totals to reveal any role-based discounts that were unavailable before registering 
  840.           WC()->cart->calculate_totals(); 
  841.       } 
  842.  
  843.       // Add customer info from other fields. 
  844.       if ( $customer_id && apply_filters( 'woocommerce_checkout_update_customer_data', true, $this ) ) { 
  845.           $customer = new WC_Customer( $customer_id ); 
  846.  
  847.           if ( ! empty( $data['billing_first_name'] ) ) { 
  848.               $customer->set_first_name( $data['billing_first_name'] ); 
  849.           } 
  850.  
  851.           if ( ! empty( $data['billing_last_name'] ) ) { 
  852.               $customer->set_last_name( $data['billing_last_name'] ); 
  853.           } 
  854.  
  855.           foreach ( $data as $key => $value ) { 
  856.               // Use setters where available. 
  857.               if ( is_callable( array( $customer, "set_{$key}" ) ) ) { 
  858.                   $customer->{"set_{$key}"}( $value ); 
  859.  
  860.               // Store custom fields prefixed with wither shipping_ or billing_. 
  861.               } elseif ( 0 === stripos( $key, 'billing_' ) || 0 === stripos( $key, 'shipping_' ) ) { 
  862.                   $customer->update_meta_data( $key, $value ); 
  863.               } 
  864.           } 
  865.  
  866.           /** 
  867.            * Action hook to adjust customer before save. 
  868.            * @since 3.0.0 
  869.            */ 
  870.           do_action( 'woocommerce_checkout_update_customer', $customer, $data ); 
  871.  
  872.           $customer->save(); 
  873.       } 
  874.  
  875.       do_action( 'woocommerce_checkout_update_user_meta', $customer_id, $data ); 
  876.   }

  903.   /** 
  904.    * Process the checkout after the confirm order button is pressed. 
  905.    */ 
  906.   public function process_checkout() { 
  907.       try { 
  908.           if ( empty( $_POST['_wpnonce'] ) || ! wp_verify_nonce( $_POST['_wpnonce'], 'woocommerce-process_checkout' ) ) { 
  909.               WC()->session->set( 'refresh_totals', true ); 
  910.               throw new Exception( __( 'We were unable to process your order, please try again.', 'woocommerce' ) ); 
  911.           } 
  912.  
  913.           wc_maybe_define_constant( 'WOOCOMMERCE_CHECKOUT', true ); 
  914.           wc_set_time_limit( 0 ); 
  915.  
  916.           do_action( 'woocommerce_before_checkout_process' ); 
  917.  
  918.           if ( WC()->cart->is_empty() ) { 
  919.               throw new Exception( sprintf( __( 'Sorry, your session has expired. <a href="%s" class="wc-backward">Return to shop</a>', 'woocommerce' ), esc_url( wc_get_page_permalink( 'shop' ) ) ) ); 
  920.           } 
  921.  
  922.           do_action( 'woocommerce_checkout_process' ); 
  923.  
  924.           $errors = new WP_Error(); 
  925.           $posted_data = $this->get_posted_data(); 
  926.  
  927.           // Update session for customer and totals. 
  928.           $this->update_session( $posted_data ); 
  929.  
  930.           // Validate posted data and cart items before proceeding. 
  931.           $this->validate_checkout( $posted_data, $errors ); 
  932.  
  933.           foreach ( $errors->get_error_messages() as $message ) { 
  934.               wc_add_notice( $message, 'error' ); 
  935.           } 
  936.  
  937.           if ( empty( $posted_data['woocommerce_checkout_update_totals'] ) && 0 === wc_notice_count( 'error' ) ) { 
  938.               $this->process_customer( $posted_data ); 
  939.               $order_id = $this->create_order( $posted_data ); 
  940.               $order = wc_get_order( $order_id ); 
  941.  
  942.               if ( is_wp_error( $order_id ) ) { 
  943.                   throw new Exception( $order_id->get_error_message() ); 
  944.               } 
  945.  
  946.               do_action( 'woocommerce_checkout_order_processed', $order_id, $posted_data, $order ); 
  947.  
  948.               if ( WC()->cart->needs_payment() ) { 
  949.                   $this->process_order_payment( $order_id, $posted_data['payment_method'] ); 
  950.               } else { 
  951.                   $this->process_order_without_payment( $order_id ); 
  952.               } 
  953.           } 
  954.       } catch ( Exception $e ) { 
  955.           wc_add_notice( $e->getMessage(), 'error' ); 
  956.       } 
  957.       $this->send_ajax_failure_response(); 
  958.   }

  976.   /** 
  977.    * Gets the value either from the posted data, or from the users meta data. 
  978.    * 
  979.    * @param string $input 
  980.    * @return string 
  981.    */ 
  982.   public function get_value( $input ) {

  984.           return wc_clean( $_POST[ $input ] ); 
  985.  
  986.       } else { 
  987.  
  988.           $value = apply_filters( 'woocommerce_checkout_get_value', null, $input );
