  17. /** 
  18.  * Parses and formats a date for ISO8601/RFC3339. 
  19.  * 
  20.  * Required WP 4.4 or later. 
  21.  * See https://developer.wordpress.org/reference/functions/mysql_to_rfc3339/ 
  22.  * 
  23.  * @since  2.6.0 
  24.  * @param  string|null|WC_DateTime $date 
  25.  * @param  bool Send false to get local/offset time. 
  26.  * @return string|null ISO8601/RFC3339 formatted datetime. 
  27.  */ 
  28. function wc_rest_prepare_date_response( $date, $utc = true ) { 
  29.   if ( is_numeric( $date ) ) { 
  30.       $date = new WC_DateTime( "@$date", new DateTimeZone( 'UTC' ) ); 
  31.       $date->setTimezone( new DateTimeZone( wc_timezone_string() ) ); 
  32.   } elseif ( is_string( $date ) ) { 
  33.       $date = new WC_DateTime( $date, new DateTimeZone( 'UTC' ) ); 
  34.       $date->setTimezone( new DateTimeZone( wc_timezone_string() ) ); 
  35.   } 
  36.  
  37.   if ( ! is_a( $date, 'WC_DateTime' ) ) { 
  38.       return null; 
  39.   } 
  40.  
  41.   // Get timestamp before changing timezone to UTC. 
  42.   return gmdate( 'Y-m-d\TH:i:s', $utc ? $date->getTimestamp() : $date->getOffsetTimestamp() ); 
  43. }

  61. /** 
  62.  * Upload image from URL. 
  63.  * 
  64.  * @since 2.6.0 
  65.  * @param string $image_url 
  66.  * @return array|WP_Error Attachment data or error message. 
  67.  */ 
  68. function wc_rest_upload_image_from_url( $image_url ) { 
  69.   $file_name = basename( current( explode( '?', $image_url ) ) ); 
  70.   $parsed_url = @parse_url( $image_url ); 
  71.  
  72.   // Check parsed URL. 
  73.   if ( ! $parsed_url || ! is_array( $parsed_url ) ) { 
  74.       return new WP_Error( 'woocommerce_rest_invalid_image_url', sprintf( __( 'Invalid URL %s.', 'woocommerce' ), $image_url ), array( 'status' => 400 ) ); 
  75.   } 
  76.  
  77.   // Ensure url is valid. 
  78.   $image_url = esc_url_raw( $image_url ); 
  79.  
  80.   // Get the file. 
  81.   $response = wp_safe_remote_get( $image_url, array( 
  82.       'timeout' => 10,  
  83.  ) ); 
  84.  
  85.   if ( is_wp_error( $response ) ) { 
  86.       return new WP_Error( 'woocommerce_rest_invalid_remote_image_url', sprintf( __( 'Error getting remote image %s.', 'woocommerce' ), $image_url ) . ' ' . sprintf( __( 'Error: %s.', 'woocommerce' ), $response->get_error_message() ), array( 'status' => 400 ) ); 
  87.   } elseif ( 200 !== wp_remote_retrieve_response_code( $response ) ) { 
  88.       return new WP_Error( 'woocommerce_rest_invalid_remote_image_url', sprintf( __( 'Error getting remote image %s.', 'woocommerce' ), $image_url ), array( 'status' => 400 ) ); 
  89.   } 
  90.  
  91.   // Ensure we have a file name and type. 
  92.   $wp_filetype = wp_check_filetype( $file_name, wc_rest_allowed_image_mime_types() ); 
  93.  
  94.   if ( ! $wp_filetype['type'] ) { 
  95.       $headers = wp_remote_retrieve_headers( $response ); 
  96.       if ( isset( $headers['content-disposition'] ) && strstr( $headers['content-disposition'], 'filename=' ) ) { 
  97.           $disposition = end( explode( 'filename=', $headers['content-disposition'] ) ); 
  98.           $disposition = sanitize_file_name( $disposition ); 
  99.           $file_name = $disposition; 
  100.       } elseif ( isset( $headers['content-type'] ) && strstr( $headers['content-type'], 'image/' ) ) { 
  101.           $file_name = 'image.' . str_replace( 'image/', '', $headers['content-type'] ); 
  102.       } 
  103.       unset( $headers ); 
  104.  
  105.       // Recheck filetype 
  106.       $wp_filetype = wp_check_filetype( $file_name, wc_rest_allowed_image_mime_types() ); 
  107.  
  108.       if ( ! $wp_filetype['type'] ) { 
  109.           return new WP_Error( 'woocommerce_rest_invalid_image_type', __( 'Invalid image type.', 'woocommerce' ), array( 'status' => 400 ) ); 
  110.       } 
  111.   } 
  112.  
  113.   // Upload the file. 
  114.   $upload = wp_upload_bits( $file_name, '', wp_remote_retrieve_body( $response ) ); 
  115.  
  116.   if ( $upload['error'] ) { 
  117.       return new WP_Error( 'woocommerce_rest_image_upload_error', $upload['error'], array( 'status' => 400 ) ); 
  118.   } 
  119.  
  120.   // Get filesize. 
  121.   $filesize = filesize( $upload['file'] ); 
  122.  
  123.   if ( 0 == $filesize ) { 
  124.       @unlink( $upload['file'] ); 
  125.       unset( $upload ); 
  126.  
  127.       return new WP_Error( 'woocommerce_rest_image_upload_file_error', __( 'Zero size file downloaded.', 'woocommerce' ), array( 'status' => 400 ) ); 
  128.   } 
  129.  
  130.   do_action( 'woocommerce_rest_api_uploaded_image_from_url', $upload, $image_url ); 
  131.  
  132.   return $upload; 
  133. }

  135. /** 
  136.  * Set uploaded image as attachment. 
  137.  * 
  138.  * @since 2.6.0 
  139.  * @param array $upload Upload information from wp_upload_bits. 
  140.  * @param int $id Post ID. Default to 0. 
  141.  * @return int Attachment ID 
  142.  */ 
  143. function wc_rest_set_uploaded_image_as_attachment( $upload, $id = 0 ) { 
  144.   $info = wp_check_filetype( $upload['file'] ); 
  145.   $title = ''; 
  146.   $content = ''; 
  147.  
  148.   if ( ! function_exists( 'wp_generate_attachment_metadata' ) ) { 
  149.       include_once( ABSPATH . 'wp-admin/includes/image.php' ); 
  150.   } 
  151.  
  152.   if ( $image_meta = wp_read_image_metadata( $upload['file'] ) ) { 
  153.       if ( trim( $image_meta['title'] ) && ! is_numeric( sanitize_title( $image_meta['title'] ) ) ) { 
  154.           $title = wc_clean( $image_meta['title'] ); 
  155.       } 
  156.       if ( trim( $image_meta['caption'] ) ) { 
  157.           $content = wc_clean( $image_meta['caption'] ); 
  158.       } 
  159.   } 
  160.  
  161.   $attachment = array( 
  162.       'post_mime_type' => $info['type'],  
  163.       'guid' => $upload['url'],  
  164.       'post_parent' => $id,  
  165.       'post_title' => $title,  
  166.       'post_content' => $content,  
  167.  ); 
  168.  
  169.   $attachment_id = wp_insert_attachment( $attachment, $upload['file'], $id ); 
  170.   if ( ! is_wp_error( $attachment_id ) ) { 
  171.       wp_update_attachment_metadata( $attachment_id, wp_generate_attachment_metadata( $attachment_id, $upload['file'] ) ); 
  172.   }

  177. /** 
  178.  * Validate reports request arguments. 
  179.  * 
  180.  * @since 2.6.0 
  181.  * @param mixed $value 
  182.  * @param WP_REST_Request $request 
  183.  * @param string $param 
  184.  * @return WP_Error|boolean 
  185.  */ 
  186. function wc_rest_validate_reports_request_arg( $value, $request, $param ) { 
  187.  
  188.   $attributes = $request->get_attributes(); 
  189.   if ( ! isset( $attributes['args'][ $param ] ) || ! is_array( $attributes['args'][ $param ] ) ) { 
  190.       return true; 
  191.   } 
  192.   $args = $attributes['args'][ $param ]; 
  193.  
  194.   if ( 'string' === $args['type'] && ! is_string( $value ) ) { 
  195.       return new WP_Error( 'woocommerce_rest_invalid_param', sprintf( __( '%1$s is not of type %2$s', 'woocommerce' ), $param, 'string' ) ); 
  196.   } 
  197.  
  198.   if ( 'date' === $args['format'] ) { 
  199.       $regex = '#^\d{4}-\d{2}-\d{2}$#'; 
  200.  
  201.       if ( ! preg_match( $regex, $value, $matches ) ) { 
  202.           return new WP_Error( 'woocommerce_rest_invalid_date', __( 'The date you provided is invalid.', 'woocommerce' ) ); 
  203.       } 
  204.   } 
  205.  
  206.   return true; 
  207. }

  209. /** 
  210.  * Encodes a value according to RFC 3986. 
  211.  * Supports multidimensional arrays. 
  212.  * 
  213.  * @since 2.6.0 
  214.  * @param string|array $value The value to encode. 
  215.  * @return string|array       Encoded values. 
  216.  */ 
  217. function wc_rest_urlencode_rfc3986( $value ) { 
  218.   if ( is_array( $value ) ) { 
  219.       return array_map( 'wc_rest_urlencode_rfc3986', $value ); 
  220.   } else { 
  221.       // Percent symbols (%) must be double-encoded. 
  222.       return str_replace( '%', '%25', rawurlencode( rawurldecode( $value ) ) ); 
  223.   } 
  224. }

  226. /**

  235. function wc_rest_check_post_permissions( $post_type, $context = 'read', $object_id = 0 ) { 
  236.   $contexts = array( 
  237.       'read' => 'read_private_posts',  
  238.       'create' => 'publish_posts',  
  239.       'edit' => 'edit_post',  
  240.       'delete' => 'delete_post',  
  241.       'batch' => 'edit_others_posts',  
  242.  ); 
  243.  
  244.   if ( 'revision' === $post_type ) { 
  245.       $permission = false; 
  246.   } else { 
  247.       $cap = $contexts[ $context ]; 
  248.       $post_type_object = get_post_type_object( $post_type ); 
  249.       $permission = current_user_can( $post_type_object->cap->$cap, $object_id ); 
  250.   } 
  251.  
  252.   return apply_filters( 'woocommerce_rest_check_permissions', $permission, $context, $object_id, $post_type ); 
  253. }
