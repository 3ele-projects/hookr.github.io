  120.   /** 
  121.    * Convert array into XML by recursively generating child elements 
  122.    * 
  123.    * @since 2.1 
  124.    * @param string|array $element_key - name for element, e.g. <OrderID> 
  125.    * @param string|array $element_value - value for element, e.g. 1234 
  126.    * @return string - generated XML 
  127.    */ 
  128.   private function array_to_xml( $element_key, $element_value = array() ) { 
  129.  
  130.       if ( is_array( $element_value ) ) { 
  131.  
  132.           // handle attributes 
  133.           if ( '@attributes' === $element_key ) { 
  134.               foreach ( $element_value as $attribute_key => $attribute_value ) { 
  135.  
  136.                   $this->xml->startAttribute( $attribute_key ); 
  137.                   $this->xml->text( $attribute_value ); 
  138.                   $this->xml->endAttribute(); 
  139.               } 
  140.               return; 
  141.           } 
  142.  
  143.           // handle multi-elements (e.g. multiple <Order> elements) 
  144.           if ( is_numeric( key( $element_value ) ) ) { 
  145.  
  146.               // recursively generate child elements 
  147.               foreach ( $element_value as $child_element_key => $child_element_value ) { 
  148.  
  149.                   $this->xml->startElement( $element_key ); 
  150.  
  151.                   foreach ( $child_element_value as $sibling_element_key => $sibling_element_value ) { 
  152.                       $this->array_to_xml( $sibling_element_key, $sibling_element_value ); 
  153.                   } 
  154.  
  155.                   $this->xml->endElement(); 
  156.               } 
  157.           } else { 
  158.  
  159.               // start root element 
  160.               $this->xml->startElement( $element_key ); 
  161.  
  162.               // recursively generate child elements 
  163.               foreach ( $element_value as $child_element_key => $child_element_value ) { 
  164.                   $this->array_to_xml( $child_element_key, $child_element_value ); 
  165.               } 
  166.  
  167.               // end root element 
  168.               $this->xml->endElement(); 
  169.           } 
  170.       } else { 
  171.  
  172.           // handle single elements 
  173.           if ( '@value' == $element_key ) { 
  174.  
  175.               $this->xml->text( $element_value ); 
  176.  
  177.           } else { 
  178.  
  179.               // wrap element in CDATA tags if it contains illegal characters 
  180.               if ( false !== strpos( $element_value, '<' ) || false !== strpos( $element_value, '>' ) ) { 
  181.  
  182.                   $this->xml->startElement( $element_key ); 
  183.                   $this->xml->writeCdata( $element_value ); 
  184.                   $this->xml->endElement(); 
  185.  
  186.               } else { 
  187.  
  188.                   $this->xml->writeElement( $element_key, $element_value ); 
  189.               } 
  190.           } 
  191.  
  192.           return; 
  193.       } 
  194.   }

  257.

  279.                   if ( ! empty( $attribute['options'] ) && is_array( $attribute['options'] ) ) { 
  280.  
  281.                       foreach ( $attribute['options'] as $option_key => $option ) {

  283.                           unset( $data['parent']['attributes'][ $attribute_key ]['options'][ $option_key ] );
