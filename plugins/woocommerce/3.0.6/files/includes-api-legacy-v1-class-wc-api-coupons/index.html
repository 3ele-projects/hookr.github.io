  18. class WC_API_Coupons extends WC_API_Resource {

  23.   /**

  59.   /** 
  60.    * Get all coupons 
  61.    * 
  62.    * @since 2.1 
  63.    * @param string $fields 
  64.    * @param array $filter 
  65.    * @param int $page 
  66.    * @return array

  68.   public function get_coupons( $fields = null, $filter = array(), $page = 1 ) { 
  69.  
  70.       $filter['page'] = $page; 
  71.  
  72.       $query = $this->query_coupons( $filter ); 
  73.  
  74.       $coupons = array(); 
  75.  
  76.       foreach ( $query->posts as $coupon_id ) { 
  77.  
  78.           if ( ! $this->is_readable( $coupon_id ) ) 
  79.               continue; 
  80.  
  81.           $coupons[] = current( $this->get_coupon( $coupon_id, $fields ) ); 
  82.       } 
  83.  
  84.       $this->server->add_pagination_headers( $query ); 
  85.  
  86.       return array( 'coupons' => $coupons ); 
  87.   }

  89.   /** 
  90.    * Get the coupon for the given ID 
  91.    * 
  92.    * @since 2.1

  94.    * @param string $fields fields to include in response

  96.    */ 
  97.   public function get_coupon( $id, $fields = null ) { 
  98.       $id = $this->validate_request( $id, 'shop_coupon', 'read' ); 
  99.  
  100.       if ( is_wp_error( $id ) ) { 
  101.           return $id; 
  102.       } 
  103.  
  104.       $coupon = new WC_Coupon( $id ); 
  105.  
  106.       if ( 0 === $coupon->get_id() ) { 
  107.           throw new WC_API_Exception( 'woocommerce_api_invalid_coupon_id', __( 'Invalid coupon ID', 'woocommerce' ), 404 ); 
  108.       } 
  109.  
  110.       $coupon_data = array( 
  111.           'id' => $coupon->get_id(),  
  112.           'code' => $coupon->get_code(),  
  113.           'type' => $coupon->get_discount_type(),  
  114.           'created_at' => $this->server->format_datetime( $coupon->get_date_created() ? $coupon->get_date_created()->getTimestamp() : 0 ), // API gives UTC times. 
  115.           'updated_at' => $this->server->format_datetime( $coupon->get_date_modified() ? $coupon->get_date_modified()->getTimestamp() : 0 ), // API gives UTC times. 
  116.           'amount' => wc_format_decimal( $coupon->get_amount(), 2 ),  
  117.           'individual_use' => $coupon->get_individual_use(),  
  118.           'product_ids' => array_map( 'absint', (array) $coupon->get_product_ids() ),  
  119.           'exclude_product_ids' => array_map( 'absint', (array) $coupon->get_excluded_product_ids() ),  
  120.           'usage_limit' => $coupon->get_usage_limit() ? $coupon->get_usage_limit() : null,  
  121.           'usage_limit_per_user' => $coupon->get_usage_limit_per_user() ? $coupon->get_usage_limit_per_user() : null,  
  122.           'limit_usage_to_x_items' => (int) $coupon->get_limit_usage_to_x_items(),  
  123.           'usage_count' => (int) $coupon->get_usage_count(),  
  124.           'expiry_date' => $this->server->format_datetime( $coupon->get_date_expires() ? $coupon->get_date_expires()->getTimestamp() : 0 ), // API gives UTC times. 
  125.           'enable_free_shipping' => $coupon->get_free_shipping(),  
  126.           'product_category_ids' => array_map( 'absint', (array) $coupon->get_product_categories() ),  
  127.           'exclude_product_category_ids' => array_map( 'absint', (array) $coupon->get_excluded_product_categories() ),  
  128.           'exclude_sale_items' => $coupon->get_exclude_sale_items(),  
  129.           'minimum_amount' => wc_format_decimal( $coupon->get_minimum_amount(), 2 ),  
  130.           'customer_emails' => $coupon->get_email_restrictions(),  
  131.  ); 
  132.  
  133.       return array( 'coupon' => apply_filters( 'woocommerce_api_coupon_response', $coupon_data, $coupon, $fields, $this->server ) ); 
  134.   }

  136.   /**

  138.    *

  143.   public function get_coupons_count( $filter = array() ) {

  145.       $query = $this->query_coupons( $filter ); 
  146.  
  147.       if ( ! current_user_can( 'read_private_shop_coupons' ) )

  150.       return array( 'count' => (int) $query->found_posts );

  166.       if ( is_null( $id ) )
