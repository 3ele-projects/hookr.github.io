/includes/cli/class-wc-cli-runner.php

  1. <?php 
  2. /** 
  3.  * WC API to WC CLI Bridge. 
  4.  * 
  5.  * Hooks into the REST API, figures out which endpoints come from WC,  
  6.  * and registers them as CLI commands. 
  7.  * 
  8.  * Forked from wp-cli/restful (by Daniel Bachhuber, released under the MIT license https://opensource.org/licenses/MIT). 
  9.  * https://github.com/wp-cli/restful 
  10.  * 
  11.  * @version 3.0.0 
  12.  * @package WooCommerce 
  13.  */ 
  14. class WC_CLI_Runner { 
  15.   /** 
  16.    * Endpoints to disable (meaning they will not be avaiable as CLI commands). 
  17.    * Some of these can either be done via WP already, or are offered with 
  18.    * some other changes (like tools). 
  19.    */ 
  20.   private static $disabled_endpoints = array( 
  21.       'settings',  
  22.       'settings/(?P<group_id>[\w-]+)',  
  23.       'settings/(?P<group_id>[\w-]+)/batch',  
  24.       'settings/(?P<group_id>[\w-]+)/(?P<id>[\w-]+)',  
  25.       'system_status',  
  26.       'system_status/tools',  
  27.       'system_status/tools/(?P<id>[\w-]+)',  
  28.       'reports',  
  29.       'reports/sales',  
  30.       'reports/top_sellers',  
  31.  ); 
  32.  
  33.   /** 
  34.    * The version of the REST API we should target to 
  35.    * generate commands. 
  36.    */ 
  37.   private static $target_rest_version = 'v2'; 
  38.  
  39.   /** 
  40.    * Register's all endpoints as commands once WP and WC have all loaded. 
  41.    */ 
  42.   public static function after_wp_load() { 
  43.       global $wp_rest_server; 
  44.       $wp_rest_server = new WP_REST_Server; 
  45.       do_action( 'rest_api_init', $wp_rest_server ); 
  46.  
  47.       $request = new WP_REST_Request( 'GET', '/' ); 
  48.       $request->set_param( 'context', 'help' ); 
  49.       $response = $wp_rest_server->dispatch( $request ); 
  50.       $response_data = $response->get_data(); 
  51.       if ( empty( $response_data ) ) { 
  52.           return; 
  53.       } 
  54.  
  55.       // Loop through all of our endpoints and register any valid WC endpoints. 
  56.       foreach ( $response_data['routes'] as $route => $route_data ) { 
  57.           // Only register endpoints for WC and our target version. 
  58.           if ( substr( $route, 0, 4 + strlen( self::$target_rest_version ) ) !== '/wc/' . self::$target_rest_version ) { 
  59.               continue; 
  60.           } 
  61.  
  62.           // Only register endpoints with schemas 
  63.           if ( empty( $route_data['schema']['title'] ) ) { 
  64.               WP_CLI::debug( sprintf( __( 'No schema title found for %s, skipping REST command registration.', 'woocommerce' ), $route ), 'wc' ); 
  65.               continue; 
  66.           } 
  67.           // Ignore batch endpoints 
  68.           if ( 'batch' === $route_data['schema']['title'] ) { 
  69.               continue; 
  70.           } 
  71.           // Disable specific endpoints 
  72.           $route_pieces = explode( '/', $route ); 
  73.           $endpoint_piece = str_replace( '/wc/' . $route_pieces[2] . '/', '', $route ); 
  74.           if ( in_array( $endpoint_piece, self::$disabled_endpoints ) ) { 
  75.               continue; 
  76.           } 
  77.  
  78.           self::register_route_commands( new WC_CLI_REST_Command( $route_data['schema']['title'], $route, $route_data['schema'] ), $route, $route_data ); 
  79.       } 
  80.   } 
  81.  
  82.   /** 
  83.    * Generates command information and tells WP CLI about all 
  84.    * commands avaiable from a route. 
  85.    * 
  86.    * @param string $rest_command 
  87.    * @param string $route 
  88.    * @param array  $route_data 
  89.    * @param array  $command_args 
  90.    */ 
  91.   private static function register_route_commands( $rest_command, $route, $route_data, $command_args = array() ) { 
  92.       // Define IDs that we are looking for in the routes (in addition to id) 
  93.       // so that we can pass it to the rest command, and use it here to generate documentation. 
  94.       $supported_ids = array( 
  95.               'id' => __( 'ID.', 'woocommerce' ),  
  96.               'product_id' => __( 'Product ID.', 'woocommerce' ),  
  97.               'customer_id' => __( 'Customer ID.', 'woocommerce' ),  
  98.               'order_id' => __( 'Order ID.', 'woocommerce' ),  
  99.               'refund_id' => __( 'Refund ID.', 'woocommerce' ),  
  100.               'attribute_id' => __( 'Attribute ID.', 'woocommerce' ),  
  101.               'zone_id' => __( 'Zone ID.', 'woocommerce' ),  
  102.  ); 
  103.       $rest_command->set_supported_ids( $supported_ids ); 
  104.       $positional_args = array_keys( $supported_ids ); 
  105.  
  106.       $parent = "wc {$route_data['schema']['title']}"; 
  107.       $supported_commands = array(); 
  108.  
  109.       // Get a list of supported commands for each route. 
  110.       foreach ( $route_data['endpoints'] as $endpoint ) { 
  111.           $parsed_args = preg_match_all( '#\([^\)]+\)#', $route, $matches ); 
  112.           $first_match = $matches[0]; 
  113.           $resource_id = ! empty( $matches[0] ) ? array_pop( $matches[0] ) : null; 
  114.           $trimmed_route = rtrim( $route ); 
  115.           $is_singular = substr( $trimmed_route, - strlen( $resource_id ) ) === $resource_id; 
  116.           if ( ! $is_singular ) { 
  117.               $resource_id = $first_match; 
  118.           } 
  119.           $command = ''; 
  120.  
  121.           // List a collection 
  122.           if ( array( 'GET' ) == $endpoint['methods'] && ! $is_singular ) { 
  123.               $supported_commands['list'] = ! empty( $endpoint['args'] ) ? $endpoint['args'] : array(); 
  124.           } 
  125.           // Create a specific resource 
  126.           if ( array( 'POST' ) == $endpoint['methods'] && ! $is_singular ) { 
  127.               $supported_commands['create'] = ! empty( $endpoint['args'] ) ? $endpoint['args'] : array(); 
  128.           } 
  129.           // Get a specific resource 
  130.           if ( array( 'GET' ) == $endpoint['methods'] && $is_singular ) { 
  131.               $supported_commands['get'] = ! empty( $endpoint['args'] ) ? $endpoint['args'] : array(); 
  132.           } 
  133.           // Update a specific resource 
  134.           if ( in_array( 'POST', $endpoint['methods'] ) && $is_singular ) { 
  135.               $supported_commands['update'] = ! empty( $endpoint['args'] ) ? $endpoint['args'] : array(); 
  136.           } 
  137.           // Delete a specific resource 
  138.           if ( array( 'DELETE' ) == $endpoint['methods'] && $is_singular ) { 
  139.               $supported_commands['delete'] = ! empty( $endpoint['args'] ) ? $endpoint['args'] : array(); 
  140.           } 
  141.       } 
  142.  
  143.       foreach ( $supported_commands as $command => $endpoint_args ) { 
  144.           $synopsis = array(); 
  145.           $arg_regs = array(); 
  146.  
  147.           foreach ( $supported_ids as $id_name => $id_desc ) { 
  148.               if ( strpos( $route, '<' . $id_name . '>' ) !== false ) { 
  149.                   $synopsis[] = array( 
  150.                       'name' => $id_name,  
  151.                       'type' => 'positional',  
  152.                       'description' => $id_desc,  
  153.                       'optional' => false,  
  154.  ); 
  155.               } 
  156.           } 
  157.  
  158.           if ( in_array( $command, array( 'delete', 'get', 'update' ) ) ) { 
  159.               $synopsis[] = array( 
  160.                   'name' => 'id',  
  161.                   'type' => 'positional',  
  162.                   'description' => __( 'The id for the resource.', 'woocommerce' ),  
  163.                   'optional' => false,  
  164.  ); 
  165.           } 
  166.  
  167.           foreach ( $endpoint_args as $name => $args ) { 
  168.               if ( ! in_array( $name, $positional_args ) ) { 
  169.                   $arg_regs[] = array( 
  170.                       'name' => $name,  
  171.                       'type' => 'assoc',  
  172.                       'description' => ! empty( $args['description'] ) ? $args['description'] : '',  
  173.                       'optional' => empty( $args['required'] ) ? true : false,  
  174.  ); 
  175.               } 
  176.           } 
  177.  
  178.           foreach ( $arg_regs as $arg_reg ) { 
  179.               $synopsis[] = $arg_reg; 
  180.           } 
  181.  
  182.           if ( in_array( $command, array( 'list', 'get' ) ) ) { 
  183.               $synopsis[] = array( 
  184.                   'name' => 'fields',  
  185.                   'type' => 'assoc',  
  186.                   'description' => __( 'Limit response to specific fields. Defaults to all fields.', 'woocommerce' ),  
  187.                   'optional' => true,  
  188.  ); 
  189.               $synopsis[] = array( 
  190.                   'name' => 'field',  
  191.                   'type' => 'assoc',  
  192.                   'description' => __( 'Get the value of an individual field.', 'woocommerce' ),  
  193.                   'optional' => true,  
  194.  ); 
  195.               $synopsis[] = array( 
  196.                   'name' => 'format',  
  197.                   'type' => 'assoc',  
  198.                   'description' => __( 'Render response in a particular format.', 'woocommerce' ),  
  199.                   'optional' => true,  
  200.                   'default' => 'table',  
  201.                   'options' => array( 
  202.                       'table',  
  203.                       'json',  
  204.                       'csv',  
  205.                       'ids',  
  206.                       'yaml',  
  207.                       'count',  
  208.                       'headers',  
  209.                       'body',  
  210.                       'envelope',  
  211.  ),  
  212.  ); 
  213.           } 
  214.  
  215.           if ( in_array( $command, array( 'create', 'update', 'delete' ) ) ) { 
  216.               $synopsis[] = array( 
  217.                   'name' => 'porcelain',  
  218.                   'type' => 'flag',  
  219.                   'description' => __( 'Output just the id when the operation is successful.', 'woocommerce' ),  
  220.                   'optional' => true,  
  221.  ); 
  222.           } 
  223.  
  224.           $methods = array( 
  225.               'list' => 'list_items',  
  226.               'create' => 'create_item',  
  227.               'delete' => 'delete_item',  
  228.               'get' => 'get_item',  
  229.               'update' => 'update_item',  
  230.  ); 
  231.  
  232.           $before_invoke = null; 
  233.           if ( empty( $command_args['when'] ) && \WP_CLI::get_config( 'debug' ) ) { 
  234.               $before_invoke = function() { 
  235.                   if ( ! defined( 'SAVEQUERIES' ) ) { 
  236.                       define( 'SAVEQUERIES', true ); 
  237.                   } 
  238.               }; 
  239.           } 
  240.  
  241.           WP_CLI::add_command( "{$parent} {$command}", array( $rest_command, $methods[ $command ] ), array( 
  242.               'synopsis' => $synopsis,  
  243.               'when' => ! empty( $command_args['when'] ) ? $command_args['when'] : '',  
  244.               'before_invoke' => $before_invoke,  
  245.  ) ); 
  246.       } 
  247.   } 
  248. }
