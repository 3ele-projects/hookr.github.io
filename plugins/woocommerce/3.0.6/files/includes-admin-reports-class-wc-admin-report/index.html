  343.   /** 
  344.    * Put data with post_date's into an array of times. 
  345.    * 
  346.    * @param  array $data array of your data 
  347.    * @param  string $date_key key for the 'date' field. e.g. 'post_date' 
  348.    * @param  string $data_key key for the data you are charting 
  349.    * @param  int $interval 
  350.    * @param  string $start_date 
  351.    * @param  string $group_by 
  352.    * @return array 
  353.    */ 
  354.   public function prepare_chart_data( $data, $date_key, $data_key, $interval, $start_date, $group_by ) { 
  355.       $prepared_data = array(); 
  356.  
  357.       // Ensure all days (or months) have values in this range. 
  358.       if ( 'day' === $group_by ) { 
  359.           for ( $i = 0; $i <= $interval; $i ++ ) { 
  360.               $time = strtotime( date( 'Ymd', strtotime( "+{$i} DAY", $start_date ) ) ) . '000'; 
  361.  
  362.               if ( ! isset( $prepared_data[ $time ] ) ) { 
  363.                   $prepared_data[ $time ] = array( esc_js( $time ), 0 ); 
  364.               } 
  365.           } 
  366.       } else { 
  367.           $current_yearnum = date( 'Y', $start_date ); 
  368.           $current_monthnum = date( 'm', $start_date ); 
  369.  
  370.           for ( $i = 0; $i <= $interval; $i ++ ) { 
  371.               $time = strtotime( $current_yearnum . str_pad( $current_monthnum, 2, '0', STR_PAD_LEFT ) . '01' ) . '000'; 
  372.  
  373.               if ( ! isset( $prepared_data[ $time ] ) ) { 
  374.                   $prepared_data[ $time ] = array( esc_js( $time ), 0 ); 
  375.               } 
  376.  
  377.               $current_monthnum ++; 
  378.  
  379.               if ( $current_monthnum > 12 ) { 
  380.                   $current_monthnum = 1; 
  381.                   $current_yearnum  ++; 
  382.               } 
  383.           } 
  384.       } 
  385.  
  386.       foreach ( $data as $d ) { 
  387.           switch ( $group_by ) { 
  388.               case 'day' : 
  389.                   $time = strtotime( date( 'Ymd', strtotime( $d->$date_key ) ) ) . '000'; 
  390.               break; 
  391.               case 'month' : 
  392.               default : 
  393.                   $time = strtotime( date( 'Ym', strtotime( $d->$date_key ) ) . '01' ) . '000'; 
  394.               break; 
  395.           } 
  396.  
  397.           if ( ! isset( $prepared_data[ $time ] ) ) { 
  398.               continue; 
  399.           } 
  400.  
  401.           if ( $data_key ) { 
  402.               $prepared_data[ $time ][1] += $d->$data_key; 
  403.           } else { 
  404.               $prepared_data[ $time ][1] ++; 
  405.           } 
  406.       } 
  407.  
  408.       return $prepared_data; 
  409.   }

  520.                   $this->end_date = strtotime( 'midnight', current_time( 'timestamp' ) );

  522.                   $this->end_date = strtotime( 'midnight', strtotime( sanitize_text_field( $_GET['end_date'] ) ) ); 
  523.               } 
  524.  
  525.               $interval = 0; 
  526.               $min_date = $this->start_date; 
  527.  
  528.               while ( ( $min_date = strtotime( "+1 MONTH", $min_date ) ) <= $this->end_date ) { 
  529.                   $interval ++; 
  530.               } 
  531.  
  532.               // 3 months max for day view 
  533.               if ( $interval > 3 ) { 
  534.                   $this->chart_groupby = 'month'; 
  535.               } else { 
  536.                   $this->chart_groupby = 'day'; 
  537.               } 
  538.           break; 
  539.  
  540.           case 'year' : 
  541.               $this->start_date = strtotime( date( 'Y-01-01', current_time( 'timestamp' ) ) ); 
  542.               $this->end_date = strtotime( 'midnight', current_time( 'timestamp' ) ); 
  543.               $this->chart_groupby = 'month'; 
  544.           break; 
  545.  
  546.           case 'last_month' : 
  547.               $first_day_current_month = strtotime( date( 'Y-m-01', current_time( 'timestamp' ) ) ); 
  548.               $this->start_date = strtotime( date( 'Y-m-01', strtotime( '-1 DAY', $first_day_current_month ) ) ); 
  549.               $this->end_date = strtotime( date( 'Y-m-t', strtotime( '-1 DAY', $first_day_current_month ) ) ); 
  550.               $this->chart_groupby = 'day'; 
  551.           break; 
  552.  
  553.           case 'month' : 
  554.               $this->start_date = strtotime( date( 'Y-m-01', current_time( 'timestamp' ) ) ); 
  555.               $this->end_date = strtotime( 'midnight', current_time( 'timestamp' ) ); 
  556.               $this->chart_groupby = 'day'; 
  557.           break; 
  558.  
  559.           case '7day' : 
  560.               $this->start_date = strtotime( '-6 days', current_time( 'timestamp' ) ); 
  561.               $this->end_date = strtotime( 'midnight', current_time( 'timestamp' ) ); 
  562.               $this->chart_groupby = 'day'; 
  563.           break; 
  564.       } 
  565.  
  566.       // Group by 
  567.       switch ( $this->chart_groupby ) { 
  568.  
  569.           case 'day' : 
  570.               $this->group_by_query = 'YEAR(posts.post_date), MONTH(posts.post_date), DAY(posts.post_date)'; 
  571.               $this->chart_interval = absint( ceil( max( 0, ( $this->end_date - $this->start_date ) / ( 60 * 60 * 24 ) ) ) ); 
  572.               $this->barwidth = 60 * 60 * 24 * 1000; 
  573.           break; 
  574.  
  575.           case 'month' : 
  576.               $this->group_by_query = 'YEAR(posts.post_date), MONTH(posts.post_date)'; 
  577.               $this->chart_interval = 0; 
  578.               $min_date = strtotime( date( 'Y-m-01', $this->start_date ) ); 
  579.  
  580.               while ( ( $min_date = strtotime( "+1 MONTH", $min_date ) ) <= $this->end_date ) { 
  581.                   $this->chart_interval ++; 
  582.               } 
  583.  
  584.               $this->barwidth = 60 * 60 * 24 * 7 * 4 * 1000; 
  585.           break; 
  586.       } 
  587.   }
