  74.       $response = array(); 
  75.  
  76.       foreach ( $mappings as $section => $values ) { 
  77.           foreach ( $values as $key => $value ) { 
  78.               if ( isset( $schema['properties'][ $section ]['properties'][ $key ]['type'] ) ) { 
  79.                   settype( $values[ $key ], $schema['properties'][ $section ]['properties'][ $key ]['type'] ); 
  80.               } 
  81.           } 
  82.           settype( $values, $schema['properties'][ $section ]['type'] ); 
  83.           $response[ $section ] = $values; 
  84.       } 
  85.  
  86.       $response = $this->prepare_item_for_response( $response, $request ); 
  87.  
  88.       return rest_ensure_response( $response );
