  173.   /** 
  174.    * Prepare a single order output for response. 
  175.    * 
  176.    * @since  3.0.0

  178.    * @param  WP_REST_Request $request Request object.

  180.    */ 
  181.   public function prepare_object_for_response( $object, $request ) { 
  182.       $this->request = $request; 
  183.       $order = wc_get_order( (int) $request['order_id'] ); 
  184.  
  185.       if ( ! $order ) { 
  186.           return new WP_Error( 'woocommerce_rest_invalid_order_id', __( 'Invalid order ID.', 'woocommerce' ), 404 ); 
  187.       } 
  188.  
  189.       if ( ! $object || $object->get_parent_id() !== $order->get_id() ) { 
  190.           return new WP_Error( 'woocommerce_rest_invalid_order_refund_id', __( 'Invalid order refund ID.', 'woocommerce' ), 404 ); 
  191.       } 
  192.  
  193.       $data = $this->get_formatted_item_data( $object ); 
  194.       $context = ! empty( $request['context'] ) ? $request['context'] : 'view'; 
  195.       $data = $this->add_additional_fields_to_object( $data, $request ); 
  196.       $data = $this->filter_response_by_context( $data, $context ); 
  197.  
  198.       // Wrap the data in a response object. 
  199.       $response = rest_ensure_response( $data ); 
  200.  
  201.       $response->add_links( $this->prepare_links( $object, $request ) ); 
  202.  
  203.       /** 
  204.        * Filter the data for a response. 
  205.        * 
  206.        * The dynamic portion of the hook name, $this->post_type,  
  207.        * refers to object type being prepared for the response. 
  208.        * 
  209.        * @param WP_REST_Response $response The response object. 
  210.        * @param WC_Data          $object   Object data. 
  211.        * @param WP_REST_Request  $request  Request object. 
  212.        */ 
  213.       return apply_filters( "woocommerce_rest_prepare_{$this->post_type}_object", $response, $object, $request ); 
  214.   }

  256.   /** 
  257.    * Prepares one object for create or update operation. 
  258.    * 
  259.    * @since  3.0.0 
  260.    * @param  WP_REST_Request $request Request object. 
  261.    * @param  bool            $creating If is creating a new object. 
  262.    * @return WP_Error|WC_Data The prepared item, or WP_Error object on failure. 
  263.    */ 
  264.   protected function prepare_object_for_database( $request, $creating = false ) { 
  265.       $order = wc_get_order( (int) $request['order_id'] ); 
  266.  
  267.       if ( ! $order ) { 
  268.           return new WP_Error( 'woocommerce_rest_invalid_order_id', __( 'Invalid order ID.', 'woocommerce' ), 404 ); 
  269.       } 
  270.  
  271.       if ( 0 > $request['amount'] ) { 
  272.           return new WP_Error( 'woocommerce_rest_invalid_order_refund', __( 'Refund amount must be greater than zero.', 'woocommerce' ), 400 ); 
  273.       } 
  274.  
  275.       // Create the refund. 
  276.       $refund = wc_create_refund( array( 
  277.           'order_id' => $order->get_id(),  
  278.           'amount' => $request['amount'],  
  279.           'reason' => empty( $request['reason'] ) ? null : $request['reason'],  
  280.           'line_items' => $request['line_items'],  
  281.           'refund_payment' => is_bool( $request['api_refund'] ) ? $request['api_refund'] : true,  
  282.           'restock_items' => true,  
  283.  ) ); 
  284.  
  285.       if ( is_wp_error( $refund ) ) { 
  286.           return new WP_Error( 'woocommerce_rest_cannot_create_order_refund', $refund->get_error_message(), 500 ); 
  287.       } 
  288.  
  289.       if ( ! $refund ) { 
  290.           return new WP_Error( 'woocommerce_rest_cannot_create_order_refund', __( 'Cannot create order refund, please try again.', 'woocommerce' ), 500 ); 
  291.       } 
  292.  
  293.       /** 
  294.        * Filters an object before it is inserted via the REST API. 
  295.        * 
  296.        * The dynamic portion of the hook name, `$this->post_type`,

  298.        * 
  299.        * @param WC_Data         $coupon   Object object.

  301.        * @param bool            $creating If is creating a new object. 
  302.        */ 
  303.       return apply_filters( "woocommerce_rest_pre_insert_{$this->post_type}_object", $refund, $request, $creating ); 
  304.   }

  308.    *

  310.    * @param  WP_REST_Request $request  Full details about the request.

  313.    */ 
  314.   protected function save_object( $request, $creating = false ) { 
  315.       try { 
  316.           $object = $this->prepare_object_for_database( $request, $creating ); 
  317.  
  318.           if ( is_wp_error( $object ) ) { 
  319.               return $object; 
  320.           } 
  321.  
  322.           return $this->get_object( $object->get_id() ); 
  323.       } catch ( WC_Data_Exception $e ) { 
  324.           return new WP_Error( $e->getErrorCode(), $e->getMessage(), $e->getErrorData() ); 
  325.       } catch ( WC_REST_Exception $e ) { 
  326.           return new WP_Error( $e->getErrorCode(), $e->getMessage(), array( 'status' => $e->getCode() ) ); 
  327.       } 
  328.   }
