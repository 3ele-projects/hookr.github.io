  252.   /** 
  253.    * Prepare a single coupon for create or update. 
  254.    * 
  255.    * @param  WP_REST_Request $request Request object. 
  256.    * @param  bool            $creating If is creating a new object. 
  257.    * @return WP_Error|WC_Data 
  258.    */ 
  259.   protected function prepare_object_for_database( $request, $creating = false ) { 
  260.       $id = isset( $request['id'] ) ? absint( $request['id'] ) : 0; 
  261.       $coupon = new WC_Coupon( $id ); 
  262.       $schema = $this->get_item_schema(); 
  263.       $data_keys = array_keys( array_filter( $schema['properties'], array( $this, 'filter_writable_props' ) ) ); 
  264.  
  265.       // Validate required POST fields. 
  266.       if ( $creating && empty( $request['code'] ) ) { 
  267.           return new WP_Error( 'woocommerce_rest_empty_coupon_code', sprintf( __( 'The coupon code cannot be empty.', 'woocommerce' ), 'code' ), array( 'status' => 400 ) ); 
  268.       } 
  269.  
  270.       // Handle all writable props. 
  271.       foreach ( $data_keys as $key ) { 
  272.           $value = $request[ $key ]; 
  273.  
  274.           if ( ! is_null( $value ) ) { 
  275.               switch ( $key ) { 
  276.                   case 'code' : 
  277.                       $coupon_code = wc_format_coupon_code( $value ); 
  278.                       $id = $coupon->get_id() ? $coupon->get_id() : 0; 
  279.                       $id_from_code = wc_get_coupon_id_by_code( $coupon_code, $id ); 
  280.  
  281.                       if ( $id_from_code ) { 
  282.                           return new WP_Error( 'woocommerce_rest_coupon_code_already_exists', __( 'The coupon code already exists', 'woocommerce' ), array( 'status' => 400 ) ); 
  283.                       } 
  284.  
  285.                       $coupon->set_code( $coupon_code ); 
  286.                       break; 
  287.                   case 'meta_data' : 
  288.                       if ( is_array( $value ) ) { 
  289.                           foreach ( $value as $meta ) { 
  290.                               $coupon->update_meta_data( $meta['key'], $meta['value'], isset( $meta['id'] ) ? $meta['id'] : '' ); 
  291.                           } 
  292.                       } 
  293.                       break; 
  294.                   case 'description' : 
  295.                       $coupon->set_description( wp_filter_post_kses( $value ) ); 
  296.                       break; 
  297.                   default : 
  298.                       if ( is_callable( array( $coupon, "set_{$key}" ) ) ) { 
  299.                           $coupon->{"set_{$key}"}( $value ); 
  300.                       } 
  301.                       break; 
  302.               } 
  303.           } 
  304.       }

  317.   }
