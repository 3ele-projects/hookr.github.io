  46.   private function get_top_seller() { 
  47.       global $wpdb; 
  48.  
  49.       $query = array(); 
  50.       $query['fields'] = "SELECT SUM( order_item_meta.meta_value ) as qty, order_item_meta_2.meta_value as product_id 
  51.           FROM {$wpdb->posts} as posts"; 
  52.       $query['join'] = "INNER JOIN {$wpdb->prefix}woocommerce_order_items AS order_items ON posts.ID = order_id "; 
  53.       $query['join']   .= "INNER JOIN {$wpdb->prefix}woocommerce_order_itemmeta AS order_item_meta ON order_items.order_item_id = order_item_meta.order_item_id "; 
  54.       $query['join']   .= "INNER JOIN {$wpdb->prefix}woocommerce_order_itemmeta AS order_item_meta_2 ON order_items.order_item_id = order_item_meta_2.order_item_id "; 
  55.       $query['where'] = "WHERE posts.post_type IN ( '" . implode( "', '", wc_get_order_types( 'order-count' ) ) . "' ) "; 
  56.       $query['where']  .= "AND posts.post_status IN ( 'wc-" . implode( "', 'wc-", apply_filters( 'woocommerce_reports_order_statuses', array( 'completed', 'processing', 'on-hold' ) ) ) . "' ) "; 
  57.       $query['where']  .= "AND order_item_meta.meta_key = '_qty' "; 
  58.       $query['where']  .= "AND order_item_meta_2.meta_key = '_product_id' "; 
  59.       $query['where']  .= "AND posts.post_date >= '" . date( 'Y-m-01', current_time( 'timestamp' ) ) . "' "; 
  60.       $query['where']  .= "AND posts.post_date <= '" . date( 'Y-m-d H:i:s', current_time( 'timestamp' ) ) . "' "; 
  61.       $query['groupby'] = "GROUP BY product_id"; 
  62.       $query['orderby'] = "ORDER BY qty DESC"; 
  63.       $query['limits'] = "LIMIT 1"; 
  64.  
  65.       return $wpdb->get_row( implode( ' ', apply_filters( 'woocommerce_dashboard_status_widget_top_seller_query', $query ) ) ); 
  66.   }

  181.       global $wpdb;

  183.       // Get products using a query - this is too advanced for get_posts :( 
  184.       $stock = absint( max( get_option( 'woocommerce_notify_low_stock_amount' ), 1 ) );

  186.       $transient_name = 'wc_low_stock_count'; 
  187.  
  188.       if ( false === ( $lowinstock_count = get_transient( $transient_name ) ) ) { 
  189.           $query_from = apply_filters( 'woocommerce_report_low_in_stock_query_from', "FROM {$wpdb->posts} as posts 
  190.               INNER JOIN {$wpdb->postmeta} AS postmeta ON posts.ID = postmeta.post_id 
  191.               INNER JOIN {$wpdb->postmeta} AS postmeta2 ON posts.ID = postmeta2.post_id 
  192.               WHERE 1=1 
  193.               AND posts.post_type IN ( 'product', 'product_variation' ) 
  194.               AND posts.post_status = 'publish' 
  195.               AND postmeta2.meta_key = '_manage_stock' AND postmeta2.meta_value = 'yes' 
  196.               AND postmeta.meta_key = '_stock' AND CAST(postmeta.meta_value AS SIGNED) <= '{$stock}'

  200.           set_transient( $transient_name, $lowinstock_count, DAY_IN_SECONDS * 30 );

  247.   public function recent_reviews() { 
  248.       global $wpdb; 
  249.       $comments = $wpdb->get_results( " 
  250.           SELECT posts.ID, posts.post_title, comments.comment_author, comments.comment_ID, SUBSTRING(comments.comment_content, 1, 100) AS comment_excerpt 
  251.           FROM $wpdb->comments comments 
  252.           LEFT JOIN $wpdb->posts posts ON (comments.comment_post_ID = posts.ID) 
  253.           WHERE comments.comment_approved = '1' 
  254.           AND comments.comment_type = '' 
  255.           AND posts.post_password = '' 
  256.           AND posts.post_type = 'product' 
  257.           ORDER BY comments.comment_date_gmt DESC 
  258.           LIMIT 5 
  259.       " ); 
  260.  
  261.       if ( $comments ) { 
  262.           echo '<ul>'; 
  263.           foreach ( $comments as $comment ) { 
  264.  
  265.               echo '<li>'; 
  266.  
  267.               echo get_avatar( $comment->comment_author, '32' ); 
  268.  
  269.               $rating = intval( get_comment_meta( $comment->comment_ID, 'rating', true ) ); 
  270.  
  271.               /** translators: %s: rating */ 
  272.               echo '<div class="star-rating"><span style="width:' . ( $rating * 20 ) . '%">' . sprintf( __( '%s out of 5', 'woocommerce' ), $rating ) . '</span></div>'; 
  273.  
  274.               /** translators: %s: review author */ 
  275.               echo '<h4 class="meta"><a href="' . get_permalink( $comment->ID ) . '#comment-' . absint( $comment->comment_ID ) . '">' . esc_html( apply_filters( 'woocommerce_admin_dashboard_recent_reviews', $comment->post_title, $comment ) ) . '</a> ' . sprintf( __( 'reviewed by %s', 'woocommerce' ), esc_html( $comment->comment_author ) ) . '</h4>'; 
  276.               echo '<blockquote>' . wp_kses_data( $comment->comment_excerpt ) . ' [...]</blockquote></li>'; 
  277.  
  278.           } 
  279.           echo '</ul>'; 
  280.       } else { 
  281.           echo '<p>' . __( 'There are no product reviews yet.', 'woocommerce' ) . '</p>'; 
  282.       } 
  283.   }
