  190. /** 
  191.  * Checks if a user (by email or ID or both) has bought an item.

  197. function wc_customer_bought_product( $customer_email, $user_id, $product_id ) { 
  198.   global $wpdb; 
  199.  
  200.   $transient_name = 'wc_cbp_' . md5( $customer_email . $user_id . WC_Cache_Helper::get_transient_version( 'orders' ) ); 
  201.  
  202.   if ( false === ( $result = get_transient( $transient_name ) ) ) { 
  203.       $customer_data = array( $user_id ); 
  204.  
  205.       if ( $user_id ) { 
  206.           $user = get_user_by( 'id', $user_id ); 
  207.  
  208.           if ( isset( $user->user_email ) ) { 
  209.               $customer_data[] = $user->user_email; 
  210.           } 
  211.       } 
  212.  
  213.       if ( is_email( $customer_email ) ) { 
  214.           $customer_data[] = $customer_email; 
  215.       } 
  216.  
  217.       $customer_data = array_map( 'esc_sql', array_filter( array_unique( $customer_data ) ) ); 
  218.       $statuses = array_map( 'esc_sql', wc_get_is_paid_statuses() ); 
  219.  
  220.       if ( sizeof( $customer_data ) == 0 ) { 
  221.           return false; 
  222.       } 
  223.  
  224.       $result = $wpdb->get_col( " 
  225.           SELECT im.meta_value FROM {$wpdb->posts} AS p 
  226.           INNER JOIN {$wpdb->postmeta} AS pm ON p.ID = pm.post_id 
  227.           INNER JOIN {$wpdb->prefix}woocommerce_order_items AS i ON p.ID = i.order_id 
  228.           INNER JOIN {$wpdb->prefix}woocommerce_order_itemmeta AS im ON i.order_item_id = im.order_item_id 
  229.           WHERE p.post_status IN ( 'wc-" . implode( "', 'wc-", $statuses ) . "' ) 
  230.           AND pm.meta_key IN ( '_billing_email', '_customer_user' ) 
  231.           AND im.meta_key IN ( '_product_id', '_variation_id' ) 
  232.           AND im.meta_value != 0 
  233.           AND pm.meta_value IN ( '" . implode( "', '", $customer_data ) . "' ) 
  234.       " ); 
  235.       $result = array_map( 'absint', $result ); 
  236.  
  237.       set_transient( $transient_name, $result, DAY_IN_SECONDS * 30 ); 
  238.   } 
  239.   return in_array( absint( $product_id ), $result ); 
  240. }

  242. /**

  251. function wc_customer_has_capability( $allcaps, $caps, $args ) { 
  252.   if ( isset( $caps[0] ) ) { 
  253.       switch ( $caps[0] ) { 
  254.           case 'view_order' : 
  255.               $user_id = $args[1]; 
  256.               $order = wc_get_order( $args[2] ); 
  257.  
  258.               if ( $order && $user_id == $order->get_user_id() ) { 
  259.                   $allcaps['view_order'] = true; 
  260.               } 
  261.           break; 
  262.           case 'pay_for_order' : 
  263.               $user_id = $args[1]; 
  264.               $order_id = isset( $args[2] ) ? $args[2] : null; 
  265.  
  266.               // When no order ID, we assume it's a new order 
  267.               // and thus, customer can pay for it 
  268.               if ( ! $order_id ) { 
  269.                   $allcaps['pay_for_order'] = true; 
  270.                   break; 
  271.               } 
  272.  
  273.               $order = wc_get_order( $order_id ); 
  274.  
  275.               if ( $order && ( $user_id == $order->get_user_id() || ! $order->get_user_id() ) ) { 
  276.                   $allcaps['pay_for_order'] = true; 
  277.               } 
  278.           break; 
  279.           case 'order_again' : 
  280.               $user_id = $args[1]; 
  281.               $order = wc_get_order( $args[2] ); 
  282.  
  283.               if ( $order && $user_id == $order->get_user_id() ) { 
  284.                   $allcaps['order_again'] = true; 
  285.               } 
  286.           break;

  288.               $user_id = $args[1]; 
  289.               $order = wc_get_order( $args[2] ); 
  290.  
  291.               if ( $order && $user_id == $order->get_user_id() ) { 
  292.                   $allcaps['cancel_order'] = true;

  554.       wc_set_user_last_update_time( $user_id );

  606. /** 
  607.  * Wrapper for @see get_avatar() which doesn't simply return 
  608.  * the URL so we need to pluck it from the HTML img tag. 
  609.  * 
  610.  * Kudos to https://github.com/WP-API/WP-API for offering a better solution. 
  611.  * 
  612.  * @since 2.6.0 
  613.  * @param string $email the customer's email. 
  614.  * @return string the URL to the customer's avatar. 
  615.  */ 
  616. function wc_get_customer_avatar_url( $email ) { 
  617.   $avatar_html = get_avatar( $email ); 
  618.  
  619.   // Get the URL of the avatar from the provided HTML. 
  620.   preg_match( '/src=["|\'](.+)[\&|"|\']/U', $avatar_html, $matches ); 
  621.  
  622.   if ( isset( $matches[1] ) && ! empty( $matches[1] ) ) { 
  623.       return esc_url_raw( $matches[1] ); 
  624.   }
