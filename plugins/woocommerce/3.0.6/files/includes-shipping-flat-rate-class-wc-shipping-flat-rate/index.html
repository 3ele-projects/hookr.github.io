  48.   /** 
  49.    * Evaluate a cost from a sum/string. 
  50.    * @param  string $sum 
  51.    * @param  array  $args 
  52.    * @return string 
  53.    */ 
  54.   protected function evaluate_cost( $sum, $args = array() ) { 
  55.       include_once( WC()->plugin_path() . '/includes/libraries/class-wc-eval-math.php' ); 
  56.  
  57.       // Allow 3rd parties to process shipping cost arguments 
  58.       $args = apply_filters( 'woocommerce_evaluate_shipping_cost_args', $args, $sum, $this ); 
  59.       $locale = localeconv(); 
  60.       $decimals = array( wc_get_price_decimal_separator(), $locale['decimal_point'], $locale['mon_decimal_point'], ', ' ); 
  61.       $this->fee_cost = $args['cost']; 
  62.  
  63.       // Expand shortcodes 
  64.       add_shortcode( 'fee', array( $this, 'fee' ) ); 
  65.  
  66.       $sum = do_shortcode( str_replace( 
  67.           array( 
  68.               '[qty]',  
  69.               '[cost]',  
  70.  ),  
  71.           array( 
  72.               $args['qty'],  
  73.               $args['cost'],  
  74.  ),  
  75.           $sum 
  76.  ) ); 
  77.  
  78.       remove_shortcode( 'fee', array( $this, 'fee' ) ); 
  79.  
  80.       // Remove whitespace from string 
  81.       $sum = preg_replace( '/\s+/', '', $sum ); 
  82.  
  83.       // Remove locale from string 
  84.       $sum = str_replace( $decimals, '.', $sum ); 
  85.  
  86.       // Trim invalid start/end characters 
  87.       $sum = rtrim( ltrim( $sum, "\t\n\r\0\x0B+*/" ), "\t\n\r\0\x0B+-*/" ); 
  88.  
  89.       // Do the math 
  90.       return $sum ? WC_Eval_Math::evaluate( $sum ) : 0; 
  91.   }

  93.   /** 
  94.    * Work out fee (shortcode). 
  95.    * @param  array $atts 
  96.    * @return string 
  97.    */ 
  98.   public function fee( $atts ) { 
  99.       $atts = shortcode_atts( array( 
  100.           'percent' => '',  
  101.           'min_fee' => '',  
  102.           'max_fee' => '',  
  103.  ), $atts, 'fee' ); 
  104.  
  105.       $calculated_fee = 0; 
  106.  
  107.       if ( $atts['percent'] ) { 
  108.           $calculated_fee = $this->fee_cost * ( floatval( $atts['percent'] ) / 100 ); 
  109.       } 
  110.  
  111.       if ( $atts['min_fee'] && $calculated_fee < $atts['min_fee'] ) { 
  112.           $calculated_fee = $atts['min_fee']; 
  113.       } 
  114.  
  115.       if ( $atts['max_fee'] && $calculated_fee > $atts['max_fee'] ) { 
  116.           $calculated_fee = $atts['max_fee']; 
  117.       } 
  118.  
  119.       return $calculated_fee; 
  120.   }

  122.   /** 
  123.    * calculate_shipping function. 
  124.    * 
  125.    * @param array $package (default: array()) 
  126.    */ 
  127.   public function calculate_shipping( $package = array() ) { 
  128.       $rate = array( 
  129.           'id' => $this->get_rate_id(),  
  130.           'label' => $this->title,  
  131.           'cost' => 0,  
  132.           'package' => $package,  
  133.  ); 
  134.  
  135.       // Calculate the costs 
  136.       $has_costs = false; // True when a cost is set. False if all costs are blank strings. 
  137.       $cost = $this->get_option( 'cost' ); 
  138.  
  139.       if ( '' !== $cost ) { 
  140.           $has_costs = true; 
  141.           $rate['cost'] = $this->evaluate_cost( $cost, array( 
  142.               'qty' => $this->get_package_item_qty( $package ),  
  143.               'cost' => $package['contents_cost'],  
  144.  ) ); 
  145.       } 
  146.  
  147.       // Add shipping class costs. 
  148.       $shipping_classes = WC()->shipping->get_shipping_classes(); 
  149.  
  150.       if ( ! empty( $shipping_classes ) ) { 
  151.           $found_shipping_classes = $this->find_shipping_classes( $package ); 
  152.           $highest_class_cost = 0; 
  153.  
  154.           foreach ( $found_shipping_classes as $shipping_class => $products ) { 
  155.               // Also handles BW compatibility when slugs were used instead of ids 
  156.               $shipping_class_term = get_term_by( 'slug', $shipping_class, 'product_shipping_class' ); 
  157.               $class_cost_string = $shipping_class_term && $shipping_class_term->term_id ? $this->get_option( 'class_cost_' . $shipping_class_term->term_id, $this->get_option( 'class_cost_' . $shipping_class, '' ) ) : $this->get_option( 'no_class_cost', '' ); 
  158.  
  159.               if ( '' === $class_cost_string ) { 
  160.                   continue; 
  161.               } 
  162.  
  163.               $has_costs = true; 
  164.               $class_cost = $this->evaluate_cost( $class_cost_string, array( 
  165.                   'qty' => array_sum( wp_list_pluck( $products, 'quantity' ) ),  
  166.                   'cost' => array_sum( wp_list_pluck( $products, 'line_total' ) ),  
  167.  ) ); 
  168.  
  169.               if ( 'class' === $this->type ) { 
  170.                   $rate['cost'] += $class_cost; 
  171.               } else { 
  172.                   $highest_class_cost = $class_cost > $highest_class_cost ? $class_cost : $highest_class_cost; 
  173.               } 
  174.           } 
  175.  
  176.           if ( 'order' === $this->type && $highest_class_cost ) { 
  177.               $rate['cost'] += $highest_class_cost; 
  178.           } 
  179.       } 
  180.  
  181.       // Add the rate 
  182.       if ( $has_costs ) { 
  183.           $this->add_rate( $rate ); 
  184.       } 
  185.  
  186.       /** 
  187.        * Developers can add additional flat rates based on this one via this action since @version 2.4. 
  188.        * 
  189.        * Previously there were (overly complex) options to add additional rates however this was not user. 
  190.        * friendly and goes against what Flat Rate Shipping was originally intended for. 
  191.        * 
  192.        * This example shows how you can add an extra rate based on this flat rate via custom function: 
  193.        * 
  194.        *         add_action( 'woocommerce_flat_rate_shipping_add_rate', 'add_another_custom_flat_rate', 10, 2 ); 
  195.        * 
  196.        *         function add_another_custom_flat_rate( $method, $rate ) { 
  197.        *             $new_rate = $rate; 
  198.        *             $new_rate['id']    .= ':' . 'custom_rate_name'; // Append a custom ID. 
  199.        *             $new_rate['label'] = 'Rushed Shipping'; // Rename to 'Rushed Shipping'. 
  200.        *             $new_rate['cost']  += 2; // Add $2 to the cost. 
  201.        * 
  202.        *             // Add it to WC. 
  203.        *             $method->add_rate( $new_rate ); 
  204.        *         }. 
  205.        */ 
  206.       do_action( 'woocommerce_' . $this->id . '_shipping_add_rate', $this, $rate ); 
  207.   }

  209.   /** 
  210.    * Get items in package. 
  211.    * @param  array $package 
  212.    * @return int 
  213.    */ 
  214.   public function get_package_item_qty( $package ) { 
  215.       $total_quantity = 0; 
  216.       foreach ( $package['contents'] as $item_id => $values ) { 
  217.           if ( $values['quantity'] > 0 && $values['data']->needs_shipping() ) { 
  218.               $total_quantity += $values['quantity']; 
  219.           } 
  220.       } 
  221.       return $total_quantity; 
  222.   }

  224.   /** 
  225.    * Finds and returns shipping classes and the products with said class. 
  226.    * @param mixed $package 
  227.    * @return array 
  228.    */ 
  229.   public function find_shipping_classes( $package ) { 
  230.       $found_shipping_classes = array(); 
  231.  
  232.       foreach ( $package['contents'] as $item_id => $values ) { 
  233.           if ( $values['data']->needs_shipping() ) { 
  234.               $found_class = $values['data']->get_shipping_class(); 
  235.  
  236.               if ( ! isset( $found_shipping_classes[ $found_class ] ) ) { 
  237.                   $found_shipping_classes[ $found_class ] = array(); 
  238.               } 
  239.  
  240.               $found_shipping_classes[ $found_class ][ $item_id ] = $values; 
  241.           } 
  242.       } 
  243.  
  244.       return $found_shipping_classes; 
  245.   }
