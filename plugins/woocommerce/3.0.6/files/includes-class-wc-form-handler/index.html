  56.    * Save and and update a billing or shipping address if the

  59.   public static function save_address() { 
  60.       global $wp; 
  61.  
  62.       if ( 'POST' !== strtoupper( $_SERVER['REQUEST_METHOD'] ) ) { 
  63.           return; 
  64.       } 
  65.  
  66.       if ( empty( $_POST['action'] ) || 'edit_address' !== $_POST['action'] || empty( $_POST['_wpnonce'] ) || ! wp_verify_nonce( $_POST['_wpnonce'], 'woocommerce-edit_address' ) ) { 
  67.           return; 
  68.       } 
  69.  
  70.       $user_id = get_current_user_id(); 
  71.  
  72.       if ( $user_id <= 0 ) { 
  73.           return; 
  74.       } 
  75.  
  76.       $load_address = isset( $wp->query_vars['edit-address'] ) ? wc_edit_address_i18n( sanitize_title( $wp->query_vars['edit-address'] ), true ) : 'billing'; 
  77.  
  78.       $address = WC()->countries->get_address_fields( esc_attr( $_POST[ $load_address . '_country' ] ), $load_address . '_' ); 
  79.  
  80.       foreach ( $address as $key => $field ) { 
  81.  
  82.           if ( ! isset( $field['type'] ) ) { 
  83.               $field['type'] = 'text'; 
  84.           } 
  85.  
  86.           // Get Value. 
  87.           switch ( $field['type'] ) { 
  88.               case 'checkbox' : 
  89.                   $_POST[ $key ] = (int) isset( $_POST[ $key ] ); 
  90.                   break; 
  91.               default : 
  92.                   $_POST[ $key ] = isset( $_POST[ $key ] ) ? wc_clean( $_POST[ $key ] ) : ''; 
  93.                   break; 
  94.           } 
  95.  
  96.           // Hook to allow modification of value. 
  97.           $_POST[ $key ] = apply_filters( 'woocommerce_process_myaccount_field_' . $key, $_POST[ $key ] ); 
  98.  
  99.           // Validation: Required fields. 
  100.           if ( ! empty( $field['required'] ) && empty( $_POST[ $key ] ) ) { 
  101.               wc_add_notice( sprintf( __( '%s is a required field.', 'woocommerce' ), $field['label'] ), 'error' ); 
  102.           } 
  103.  
  104.           if ( ! empty( $_POST[ $key ] ) ) { 
  105.  
  106.               // Validation rules. 
  107.               if ( ! empty( $field['validate'] ) && is_array( $field['validate'] ) ) { 
  108.                   foreach ( $field['validate'] as $rule ) { 
  109.                       switch ( $rule ) { 
  110.                           case 'postcode' : 
  111.                               $_POST[ $key ] = strtoupper( str_replace( ' ', '', $_POST[ $key ] ) ); 
  112.  
  113.                               if ( ! WC_Validation::is_postcode( $_POST[ $key ], $_POST[ $load_address . '_country' ] ) ) { 
  114.                                   wc_add_notice( __( 'Please enter a valid postcode / ZIP.', 'woocommerce' ), 'error' ); 
  115.                               } else { 
  116.                                   $_POST[ $key ] = wc_format_postcode( $_POST[ $key ], $_POST[ $load_address . '_country' ] ); 
  117.                               } 
  118.                               break; 
  119.                           case 'phone' : 
  120.                               $_POST[ $key ] = wc_format_phone_number( $_POST[ $key ] ); 
  121.  
  122.                               if ( ! WC_Validation::is_phone( $_POST[ $key ] ) ) { 
  123.                                   wc_add_notice( sprintf( __( '%s is not a valid phone number.', 'woocommerce' ), '<strong>' . $field['label'] . '</strong>' ), 'error' ); 
  124.                               } 
  125.                               break; 
  126.                           case 'email' : 
  127.                               $_POST[ $key ] = strtolower( $_POST[ $key ] ); 
  128.  
  129.                               if ( ! is_email( $_POST[ $key ] ) ) { 
  130.                                   wc_add_notice( sprintf( __( '%s is not a valid email address.', 'woocommerce' ), '<strong>' . $field['label'] . '</strong>' ), 'error' ); 
  131.                               } 
  132.                               break; 
  133.                       } 
  134.                   } 
  135.               } 
  136.           } 
  137.       } 
  138.  
  139.       do_action( 'woocommerce_after_save_address_validation', $user_id, $load_address, $address ); 
  140.  
  141.       if ( 0 === wc_notice_count( 'error' ) ) { 
  142.  
  143.           foreach ( $address as $key => $field ) { 
  144.               update_user_meta( $user_id, $key, $_POST[ $key ] ); 
  145.           } 
  146.  
  147.           wc_add_notice( __( 'Address changed successfully.', 'woocommerce' ) ); 
  148.  
  149.           do_action( 'woocommerce_customer_save_address', $user_id, $load_address ); 
  150.  
  151.           wp_safe_redirect( wc_get_endpoint_url( 'edit-address', '', wc_get_page_permalink( 'myaccount' ) ) ); 
  152.           exit; 
  153.       } 
  154.   }

  159.   public static function save_account_details() { 
  160.  
  161.       if ( 'POST' !== strtoupper( $_SERVER['REQUEST_METHOD'] ) ) { 
  162.           return; 
  163.       } 
  164.  
  165.       if ( empty( $_POST['action'] ) || 'save_account_details' !== $_POST['action'] || empty( $_POST['_wpnonce'] ) || ! wp_verify_nonce( $_POST['_wpnonce'], 'save_account_details' ) ) { 
  166.           return; 
  167.       } 
  168.  
  169.       $errors = new WP_Error(); 
  170.       $user = new stdClass(); 
  171.  
  172.       $user->ID = (int) get_current_user_id(); 
  173.       $current_user = get_user_by( 'id', $user->ID ); 
  174.  
  175.       if ( $user->ID <= 0 ) { 
  176.           return; 
  177.       } 
  178.  
  179.       $account_first_name = ! empty( $_POST['account_first_name'] ) ? wc_clean( $_POST['account_first_name'] ) : ''; 
  180.       $account_last_name = ! empty( $_POST['account_last_name'] ) ? wc_clean( $_POST['account_last_name'] ) : ''; 
  181.       $account_email = ! empty( $_POST['account_email'] ) ? wc_clean( $_POST['account_email'] ) : ''; 
  182.       $pass_cur = ! empty( $_POST['password_current'] ) ? $_POST['password_current'] : ''; 
  183.       $pass1 = ! empty( $_POST['password_1'] ) ? $_POST['password_1'] : ''; 
  184.       $pass2 = ! empty( $_POST['password_2'] ) ? $_POST['password_2'] : ''; 
  185.       $save_pass = true; 
  186.  
  187.       $user->first_name = $account_first_name; 
  188.       $user->last_name = $account_last_name; 
  189.  
  190.       // Prevent emails being displayed, or leave alone. 
  191.       $user->display_name = is_email( $current_user->display_name ) ? $user->first_name : $current_user->display_name; 
  192.  
  193.       // Handle required fields 
  194.       $required_fields = apply_filters( 'woocommerce_save_account_details_required_fields', array( 
  195.           'account_first_name' => __( 'First name', 'woocommerce' ),  
  196.           'account_last_name' => __( 'Last name', 'woocommerce' ),  
  197.           'account_email' => __( 'Email address', 'woocommerce' ),  
  198.  ) ); 
  199.  
  200.       foreach ( $required_fields as $field_key => $field_name ) { 
  201.           if ( empty( $_POST[ $field_key ] ) ) { 
  202.               wc_add_notice( sprintf( __( '%s is a required field.', 'woocommerce' ), '<strong>' . esc_html( $field_name ) . '</strong>' ), 'error' ); 
  203.           } 
  204.       } 
  205.  
  206.       if ( $account_email ) { 
  207.           $account_email = sanitize_email( $account_email ); 
  208.           if ( ! is_email( $account_email ) ) { 
  209.               wc_add_notice( __( 'Please provide a valid email address.', 'woocommerce' ), 'error' ); 
  210.           } elseif ( email_exists( $account_email ) && $account_email !== $current_user->user_email ) { 
  211.               wc_add_notice( __( 'This email address is already registered.', 'woocommerce' ), 'error' ); 
  212.           } 
  213.           $user->user_email = $account_email; 
  214.       } 
  215.  
  216.       if ( ! empty( $pass_cur ) && empty( $pass1 ) && empty( $pass2 ) ) { 
  217.           wc_add_notice( __( 'Please fill out all password fields.', 'woocommerce' ), 'error' ); 
  218.           $save_pass = false; 
  219.       } elseif ( ! empty( $pass1 ) && empty( $pass_cur ) ) { 
  220.           wc_add_notice( __( 'Please enter your current password.', 'woocommerce' ), 'error' ); 
  221.           $save_pass = false; 
  222.       } elseif ( ! empty( $pass1 ) && empty( $pass2 ) ) { 
  223.           wc_add_notice( __( 'Please re-enter your password.', 'woocommerce' ), 'error' ); 
  224.           $save_pass = false; 
  225.       } elseif ( ( ! empty( $pass1 ) || ! empty( $pass2 ) ) && $pass1 !== $pass2 ) { 
  226.           wc_add_notice( __( 'New passwords do not match.', 'woocommerce' ), 'error' ); 
  227.           $save_pass = false; 
  228.       } elseif ( ! empty( $pass1 ) && ! wp_check_password( $pass_cur, $current_user->user_pass, $current_user->ID ) ) { 
  229.           wc_add_notice( __( 'Your current password is incorrect.', 'woocommerce' ), 'error' ); 
  230.           $save_pass = false; 
  231.       } 
  232.  
  233.       if ( $pass1 && $save_pass ) { 
  234.           $user->user_pass = $pass1; 
  235.       } 
  236.  
  237.       // Allow plugins to return their own errors. 
  238.       do_action_ref_array( 'woocommerce_save_account_details_errors', array( &$errors, &$user ) ); 
  239.  
  240.       if ( $errors->get_error_messages() ) { 
  241.           foreach ( $errors->get_error_messages() as $error ) { 
  242.               wc_add_notice( $error, 'error' ); 
  243.           } 
  244.       } 
  245.  
  246.       if ( wc_notice_count( 'error' ) === 0 ) { 
  247.  
  248.           wp_update_user( $user ); 
  249.  
  250.           wc_add_notice( __( 'Account details changed successfully.', 'woocommerce' ) ); 
  251.  
  252.           do_action( 'woocommerce_save_account_details', $user->ID ); 
  253.  
  254.           wp_safe_redirect( wc_get_page_permalink( 'myaccount' ) ); 
  255.           exit; 
  256.       } 
  257.   }

  281.   public static function pay_action() { 
  282.       global $wp; 
  283.  
  284.       if ( isset( $_POST['woocommerce_pay'] ) && isset( $_POST['_wpnonce'] ) && wp_verify_nonce( $_POST['_wpnonce'], 'woocommerce-pay' ) ) { 
  285.  
  286.           ob_start(); 
  287.  
  288.           // Pay for existing order 
  289.           $order_key = $_GET['key']; 
  290.           $order_id = absint( $wp->query_vars['order-pay'] ); 
  291.           $order = wc_get_order( $order_id ); 
  292.  
  293.           if ( $order->get_id() == $order_id && $order->get_order_key() == $order_key && $order->needs_payment() ) { 
  294.  
  295.               do_action( 'woocommerce_before_pay_action', $order ); 
  296.  
  297.               WC()->customer->set_props( array( 
  298.                   'billing_country' => $order->get_billing_country() ? $order->get_billing_country()   : null,  
  299.                   'billing_state' => $order->get_billing_state() ? $order->get_billing_state()       : null,  
  300.                   'billing_postcode' => $order->get_billing_postcode() ? $order->get_billing_postcode() : null,  
  301.                   'billing_city' => $order->get_billing_city() ? $order->get_billing_city()         : null,  
  302.  ) ); 
  303.               WC()->customer->save(); 
  304.  
  305.               // Terms 
  306.               if ( ! empty( $_POST['terms-field'] ) && empty( $_POST['terms'] ) ) { 
  307.                   wc_add_notice( __( 'You must accept our Terms & Conditions.', 'woocommerce' ), 'error' ); 
  308.                   return; 
  309.               } 
  310.  
  311.               // Update payment method 
  312.               if ( $order->needs_payment() ) { 
  313.                   $payment_method = isset( $_POST['payment_method'] ) ? wc_clean( $_POST['payment_method'] ) : false; 
  314.                   $available_gateways = WC()->payment_gateways->get_available_payment_gateways(); 
  315.  
  316.                   if ( ! $payment_method ) { 
  317.                       wc_add_notice( __( 'Invalid payment method.', 'woocommerce' ), 'error' ); 
  318.                       return; 
  319.                   } 
  320.  
  321.                   // Update meta 
  322.                   update_post_meta( $order_id, '_payment_method', $payment_method ); 
  323.  
  324.                   if ( isset( $available_gateways[ $payment_method ] ) ) { 
  325.                       $payment_method_title = $available_gateways[ $payment_method ]->get_title(); 
  326.                   } else { 
  327.                       $payment_method_title = ''; 
  328.                   } 
  329.  
  330.                   update_post_meta( $order_id, '_payment_method_title', $payment_method_title ); 
  331.  
  332.                   // Validate 
  333.                   $available_gateways[ $payment_method ]->validate_fields(); 
  334.  
  335.                   // Process 
  336.                   if ( wc_notice_count( 'error' ) == 0 ) { 
  337.  
  338.                       $result = $available_gateways[ $payment_method ]->process_payment( $order_id ); 
  339.  
  340.                       // Redirect to success/confirmation/payment page 
  341.                       if ( 'success' === $result['result'] ) { 
  342.                           wp_redirect( $result['redirect'] ); 
  343.                           exit; 
  344.                       } 
  345.                   } 
  346.               } else { 
  347.                   // No payment was required for order 
  348.                   $order->payment_complete(); 
  349.                   wp_safe_redirect( $order->get_checkout_order_received_url() ); 
  350.                   exit; 
  351.               } 
  352.  
  353.               do_action( 'woocommerce_after_pay_action', $order ); 
  354.  
  355.           } 
  356.       } 
  357.   }

  442.           // Add Discount 
  443.           WC()->cart->add_discount( sanitize_text_field( $_POST['coupon_code'] ) ); 
  444.  
  445.       } elseif ( isset( $_GET['remove_coupon'] ) ) { 
  446.  
  447.           // Remove Coupon Codes 
  448.           WC()->cart->remove_coupon( wc_clean( $_GET['remove_coupon'] ) ); 
  449.  
  450.       } elseif ( ! empty( $_GET['remove_item'] ) && isset( $_GET['_wpnonce'] ) && wp_verify_nonce( $_GET['_wpnonce'], 'woocommerce-cart' ) ) { 
  451.  
  452.           // Remove from cart 
  453.           $cart_item_key = sanitize_text_field( $_GET['remove_item'] ); 
  454.  
  455.           if ( $cart_item = WC()->cart->get_cart_item( $cart_item_key ) ) { 
  456.               WC()->cart->remove_cart_item( $cart_item_key ); 
  457.  
  458.               $product = wc_get_product( $cart_item['product_id'] ); 
  459.  
  460.               $item_removed_title = apply_filters( 'woocommerce_cart_item_removed_title', $product ? sprintf( _x( '“%s”', 'Item name in quotes', 'woocommerce' ), $product->get_name() ) : __( 'Item', 'woocommerce' ), $cart_item ); 
  461.  
  462.               // Don't show undo link if removed item is out of stock. 
  463.               if ( $product->is_in_stock() && $product->has_enough_stock( $cart_item['quantity'] ) ) { 
  464.                   $removed_notice = sprintf( __( '%s removed.', 'woocommerce' ), $item_removed_title ); 
  465.                   $removed_notice .= ' <a href="' . esc_url( WC()->cart->get_undo_url( $cart_item_key ) ) . '">' . __( 'Undo?', 'woocommerce' ) . '</a>'; 
  466.               } else { 
  467.                   $removed_notice = sprintf( __( '%s removed.', 'woocommerce' ), $item_removed_title ); 
  468.               } 
  469.  
  470.               wc_add_notice( $removed_notice ); 
  471.           } 
  472.  
  473.           $referer = wp_get_referer() ? remove_query_arg( array( 'remove_item', 'add-to-cart', 'added-to-cart' ), add_query_arg( 'removed_item', '1', wp_get_referer() ) ) : wc_get_cart_url(); 
  474.           wp_safe_redirect( $referer ); 
  475.           exit; 
  476.  
  477.       } elseif ( ! empty( $_GET['undo_item'] ) && isset( $_GET['_wpnonce'] ) && wp_verify_nonce( $_GET['_wpnonce'], 'woocommerce-cart' ) ) { 
  478.  
  479.           // Undo Cart Item 
  480.           $cart_item_key = sanitize_text_field( $_GET['undo_item'] ); 
  481.  
  482.           WC()->cart->restore_cart_item( $cart_item_key ); 
  483.  
  484.           $referer = wp_get_referer() ? remove_query_arg( array( 'undo_item', '_wpnonce' ), wp_get_referer() ) : wc_get_cart_url(); 
  485.           wp_safe_redirect( $referer ); 
  486.           exit; 
  487.  
  488.       } 
  489.  
  490.       // Update Cart - checks apply_coupon too because they are in the same form 
  491.       if ( ( ! empty( $_POST['apply_coupon'] ) || ! empty( $_POST['update_cart'] ) || ! empty( $_POST['proceed'] ) ) && isset( $_POST['_wpnonce'] ) && wp_verify_nonce( $_POST['_wpnonce'], 'woocommerce-cart' ) ) { 
  492.  
  493.           $cart_updated = false; 
  494.           $cart_totals = isset( $_POST['cart'] ) ? $_POST['cart'] : ''; 
  495.  
  496.           if ( ! WC()->cart->is_empty() && is_array( $cart_totals ) ) { 
  497.               foreach ( WC()->cart->get_cart() as $cart_item_key => $values ) { 
  498.  
  499.                   $_product = $values['data']; 
  500.  
  501.                   // Skip product if no updated quantity was posted 
  502.                   if ( ! isset( $cart_totals[ $cart_item_key ] ) || ! isset( $cart_totals[ $cart_item_key ]['qty'] ) ) { 
  503.                       continue; 
  504.                   } 
  505.  
  506.                   // Sanitize 
  507.                   $quantity = apply_filters( 'woocommerce_stock_amount_cart_item', wc_stock_amount( preg_replace( "/[^0-9\.]/", '', $cart_totals[ $cart_item_key ]['qty'] ) ), $cart_item_key ); 
  508.  
  509.                   if ( '' === $quantity || $quantity == $values['quantity'] ) 
  510.                       continue; 
  511.  
  512.                   // Update cart validation 
  513.                   $passed_validation = apply_filters( 'woocommerce_update_cart_validation', true, $cart_item_key, $values, $quantity ); 
  514.  
  515.                   // is_sold_individually 
  516.                   if ( $_product->is_sold_individually() && $quantity > 1 ) { 
  517.                       wc_add_notice( sprintf( __( 'You can only have 1 %s in your cart.', 'woocommerce' ), $_product->get_name() ), 'error' ); 
  518.                       $passed_validation = false; 
  519.                   } 
  520.  
  521.                   if ( $passed_validation ) { 
  522.                       WC()->cart->set_quantity( $cart_item_key, $quantity, false ); 
  523.                       $cart_updated = true; 
  524.                   } 
  525.               } 
  526.           } 
  527.  
  528.           // Trigger action - let 3rd parties update the cart if they need to and update the $cart_updated variable 
  529.           $cart_updated = apply_filters( 'woocommerce_update_cart_action_cart_updated', $cart_updated ); 
  530.  
  531.           if ( $cart_updated ) { 
  532.               // Recalc our totals 
  533.               WC()->cart->calculate_totals(); 
  534.           } 
  535.  
  536.           if ( ! empty( $_POST['proceed'] ) ) { 
  537.               wp_safe_redirect( wc_get_checkout_url() ); 
  538.               exit; 
  539.           } elseif ( $cart_updated ) { 
  540.               wc_add_notice( __( 'Cart updated.', 'woocommerce' ) ); 
  541.               $referer = remove_query_arg( array( 'remove_coupon', 'add-to-cart' ), ( wp_get_referer() ? wp_get_referer() : wc_get_cart_url() ) ); 
  542.               wp_safe_redirect( $referer ); 
  543.               exit; 
  544.           } 
  545.       } 
  546.   }

  548.   /** 
  549.    * Place a previous order again. 
  550.    */ 
  551.   public static function order_again() { 
  552.  
  553.       // Nothing to do 
  554.       if ( ! isset( $_GET['order_again'] ) || ! is_user_logged_in() || ! isset( $_GET['_wpnonce'] ) || ! wp_verify_nonce( $_GET['_wpnonce'], 'woocommerce-order_again' ) ) { 
  555.           return; 
  556.       } 
  557.  
  558.       // Clear current cart 
  559.       WC()->cart->empty_cart(); 
  560.  
  561.       // Load the previous order - Stop if the order does not exist 
  562.       $order = wc_get_order( absint( $_GET['order_again'] ) ); 
  563.  
  564.       if ( ! $order->get_id() ) { 
  565.           return; 
  566.       } 
  567.  
  568.       if ( ! $order->has_status( apply_filters( 'woocommerce_valid_order_statuses_for_order_again', array( 'completed' ) ) ) ) { 
  569.           return; 
  570.       } 
  571.  
  572.       // Make sure the user is allowed to order again. By default it check if the 
  573.       // previous order belonged to the current user. 
  574.       if ( ! current_user_can( 'order_again', $order->get_id() ) ) { 
  575.           return; 
  576.       } 
  577.  
  578.       // Copy products from the order to the cart 
  579.       $order_items = $order->get_items(); 
  580.       foreach ( $order_items as $item ) { 
  581.           // Load all product info including variation data 
  582.           $product_id = (int) apply_filters( 'woocommerce_add_to_cart_product_id', $item->get_product_id() ); 
  583.           $quantity = $item->get_quantity(); 
  584.           $variation_id = $item->get_variation_id(); 
  585.           $variations = array(); 
  586.           $cart_item_data = apply_filters( 'woocommerce_order_again_cart_item_data', array(), $item, $order ); 
  587.  
  588.           foreach ( $item->get_meta_data() as $meta ) { 
  589.               if ( taxonomy_is_product_attribute( $meta->key ) ) { 
  590.                   $term = get_term_by( 'slug', $meta->value, $meta->key ); 
  591.                   $variations[ $meta->key ] = $term ? $term->name : $meta->value; 
  592.               } elseif ( meta_is_product_attribute( $meta->key, $meta->value, $product_id ) ) { 
  593.                   $variations[ $meta->key ] = $meta->value; 
  594.               } 
  595.           } 
  596.  
  597.           // Add to cart validation 
  598.           if ( ! apply_filters( 'woocommerce_add_to_cart_validation', true, $product_id, $quantity, $variation_id, $variations, $cart_item_data ) ) { 
  599.               continue; 
  600.           } 
  601.  
  602.           WC()->cart->add_to_cart( $product_id, $quantity, $variation_id, $variations, $cart_item_data ); 
  603.       } 
  604.  
  605.       do_action( 'woocommerce_ordered_again', $order->get_id() ); 
  606.  
  607.       $num_items_in_cart = count( WC()->cart->get_cart() ); 
  608.       $num_items_in_original_order = count( $order_items ); 
  609.  
  610.       if ( $num_items_in_original_order > $num_items_in_cart ) { 
  611.           wc_add_notice( 
  612.               sprintf( _n( 
  613.                   '%d item from your previous order is currently unavailable and could not be added to your cart.',  
  614.                   '%d items from your previous order are currently unavailable and could not be added to your cart.',  
  615.                   $num_items_in_original_order - $num_items_in_cart,  
  616.                   'woocommerce' 
  617.  ), $num_items_in_original_order - $num_items_in_cart ),  
  618.               'error' 
  619.  ); 
  620.       } 
  621.  
  622.       if ( $num_items_in_cart > 0 ) { 
  623.           wc_add_notice( __( 'The cart has been filled with the items from your previous order.', 'woocommerce' ) ); 
  624.       } 
  625.  
  626.       // Redirect to cart 
  627.       wp_safe_redirect( wc_get_cart_url() ); 
  628.       exit; 
  629.   }

  670.   /** 
  671.    * Add to cart action. 
  672.    * 
  673.    * Checks for a valid request, does validation (via hooks) and then redirects if valid. 
  674.    * 
  675.    * @param bool $url (default: false) 
  676.    */ 
  677.   public static function add_to_cart_action( $url = false ) { 
  678.       if ( empty( $_REQUEST['add-to-cart'] ) || ! is_numeric( $_REQUEST['add-to-cart'] ) ) { 
  679.           return; 
  680.       } 
  681.  
  682.       $product_id = apply_filters( 'woocommerce_add_to_cart_product_id', absint( $_REQUEST['add-to-cart'] ) ); 
  683.       $was_added_to_cart = false; 
  684.       $adding_to_cart = wc_get_product( $product_id ); 
  685.  
  686.       if ( ! $adding_to_cart ) { 
  687.           return; 
  688.       } 
  689.  
  690.       $add_to_cart_handler = apply_filters( 'woocommerce_add_to_cart_handler', $adding_to_cart->get_type(), $adding_to_cart ); 
  691.  
  692.       // Variable product handling 
  693.       if ( 'variable' === $add_to_cart_handler ) { 
  694.           $was_added_to_cart = self::add_to_cart_handler_variable( $product_id ); 
  695.  
  696.       // Grouped Products 
  697.       } elseif ( 'grouped' === $add_to_cart_handler ) { 
  698.           $was_added_to_cart = self::add_to_cart_handler_grouped( $product_id ); 
  699.  
  700.       // Custom Handler 
  701.       } elseif ( has_action( 'woocommerce_add_to_cart_handler_' . $add_to_cart_handler ) ) { 
  702.           do_action( 'woocommerce_add_to_cart_handler_' . $add_to_cart_handler, $url ); 
  703.  
  704.       // Simple Products 
  705.       } else { 
  706.           $was_added_to_cart = self::add_to_cart_handler_simple( $product_id ); 
  707.       } 
  708.  
  709.       // If we added the product to the cart we can now optionally do a redirect. 
  710.       if ( $was_added_to_cart && wc_notice_count( 'error' ) === 0 ) { 
  711.           // If has custom URL redirect there 
  712.           if ( $url = apply_filters( 'woocommerce_add_to_cart_redirect', $url ) ) { 
  713.               wp_safe_redirect( $url ); 
  714.               exit; 
  715.           } elseif ( get_option( 'woocommerce_cart_redirect_after_add' ) === 'yes' ) { 
  716.               wp_safe_redirect( wc_get_cart_url() ); 
  717.               exit; 
  718.           } 
  719.       } 
  720.   }

  780.   /** 
  781.    * Handle adding variable products to the cart. 
  782.    * @since 2.4.6 Split from add_to_cart_action 
  783.    * @param int $product_id 
  784.    * @return bool success or not 
  785.    */ 
  786.   private static function add_to_cart_handler_variable( $product_id ) { 
  787.       $adding_to_cart = wc_get_product( $product_id ); 
  788.       $variation_id = empty( $_REQUEST['variation_id'] ) ? '' : absint( $_REQUEST['variation_id'] ); 
  789.       $quantity = empty( $_REQUEST['quantity'] ) ? 1 : wc_stock_amount( $_REQUEST['quantity'] ); 
  790.       $missing_attributes = array(); 
  791.       $variations = array(); 
  792.       $attributes = $adding_to_cart->get_attributes(); 
  793.  
  794.       // If no variation ID is set, attempt to get a variation ID from posted attributes. 
  795.       if ( empty( $variation_id ) ) { 
  796.           $data_store = WC_Data_Store::load( 'product' ); 
  797.           $variation_id = $data_store->find_matching_product_variation( $adding_to_cart, wp_unslash( $_POST ) ); 
  798.       } 
  799.  
  800.       // Validate the attributes. 
  801.       try { 
  802.           if ( empty( $variation_id ) ) { 
  803.               throw new Exception( __( 'Please choose product options…', 'woocommerce' ) ); 
  804.           } 
  805.  
  806.           $variation_data = wc_get_product_variation_attributes( $variation_id ); 
  807.  
  808.           foreach ( $attributes as $attribute ) { 
  809.               if ( ! $attribute['is_variation'] ) { 
  810.                   continue; 
  811.               } 
  812.  
  813.               $taxonomy = 'attribute_' . sanitize_title( $attribute['name'] ); 
  814.  
  815.               if ( isset( $_REQUEST[ $taxonomy ] ) ) { 
  816.                   // Get value from post data 
  817.                   if ( $attribute['is_taxonomy'] ) { 
  818.                       // Don't use wc_clean as it destroys sanitized characters 
  819.                       $value = sanitize_title( stripslashes( $_REQUEST[ $taxonomy ] ) ); 
  820.                   } else { 
  821.                       $value = wc_clean( stripslashes( $_REQUEST[ $taxonomy ] ) ); 
  822.                   } 
  823.  
  824.                   // Get valid value from variation 
  825.                   $valid_value = isset( $variation_data[ $taxonomy ] ) ? $variation_data[ $taxonomy ] : ''; 
  826.  
  827.                   // Allow if valid or show error. 
  828.                   if ( '' === $valid_value || $valid_value === $value ) { 
  829.                       $variations[ $taxonomy ] = $value; 
  830.                   } else { 
  831.                       throw new Exception( sprintf( __( 'Invalid value posted for %s', 'woocommerce' ), wc_attribute_label( $attribute['name'] ) ) ); 
  832.                   } 
  833.               } else { 
  834.                   $missing_attributes[] = wc_attribute_label( $attribute['name'] ); 
  835.               } 
  836.           } 
  837.           if ( ! empty( $missing_attributes ) ) { 
  838.               throw new Exception( sprintf( _n( '%s is a required field', '%s are required fields', sizeof( $missing_attributes ), 'woocommerce' ), wc_format_list_of_items( $missing_attributes ) ) ); 
  839.           } 
  840.       } catch ( Exception $e ) { 
  841.           wc_add_notice( $e->getMessage(), 'error' ); 
  842.           return false; 
  843.       } 
  844.  
  845.       // Add to cart validation 
  846.       $passed_validation = apply_filters( 'woocommerce_add_to_cart_validation', true, $product_id, $quantity, $variation_id, $variations ); 
  847.  
  848.       if ( $passed_validation && WC()->cart->add_to_cart( $product_id, $quantity, $variation_id, $variations ) !== false ) { 
  849.           wc_add_to_cart_message( array( $product_id => $quantity ), true ); 
  850.           return true; 
  851.       } 
  852.  
  853.       return false; 
  854.   }

  856.   /** 
  857.    * Process the login form. 
  858.    */ 
  859.   public static function process_login() { 
  860.       $nonce_value = isset( $_POST['_wpnonce'] ) ? $_POST['_wpnonce'] : ''; 
  861.       $nonce_value = isset( $_POST['woocommerce-login-nonce'] ) ? $_POST['woocommerce-login-nonce'] : $nonce_value; 
  862.  
  863.       if ( ! empty( $_POST['login'] ) && wp_verify_nonce( $nonce_value, 'woocommerce-login' ) ) { 
  864.  
  865.           try { 
  866.               $creds = array( 
  867.                   'user_password' => $_POST['password'],  
  868.                   'remember' => isset( $_POST['rememberme'] ),  
  869.  ); 
  870.  
  871.               $username = trim( $_POST['username'] ); 
  872.               $validation_error = new WP_Error(); 
  873.               $validation_error = apply_filters( 'woocommerce_process_login_errors', $validation_error, $_POST['username'], $_POST['password'] ); 
  874.  
  875.               if ( $validation_error->get_error_code() ) { 
  876.                   throw new Exception( '<strong>' . __( 'Error:', 'woocommerce' ) . '</strong> ' . $validation_error->get_error_message() ); 
  877.               } 
  878.  
  879.               if ( empty( $username ) ) { 
  880.                   throw new Exception( '<strong>' . __( 'Error:', 'woocommerce' ) . '</strong> ' . __( 'Username is required.', 'woocommerce' ) ); 
  881.               } 
  882.  
  883.               if ( is_email( $username ) && apply_filters( 'woocommerce_get_username_from_email', true ) ) { 
  884.                   $user = get_user_by( 'email', $username ); 
  885.  
  886.                   if ( isset( $user->user_login ) ) { 
  887.                       $creds['user_login'] = $user->user_login; 
  888.                   } else { 
  889.                       throw new Exception( '<strong>' . __( 'Error:', 'woocommerce' ) . '</strong> ' . __( 'A user could not be found with this email address.', 'woocommerce' ) ); 
  890.                   } 
  891.               } else { 
  892.                   $creds['user_login'] = $username; 
  893.               } 
  894.  
  895.               // On multisite, ensure user exists on current site, if not add them before allowing login. 
  896.               if ( is_multisite() ) { 
  897.                   $user_data = get_user_by( 'login', $username ); 
  898.  
  899.                   if ( $user_data && ! is_user_member_of_blog( $user_data->ID, get_current_blog_id() ) ) { 
  900.                       add_user_to_blog( get_current_blog_id(), $user_data->ID, 'customer' ); 
  901.                   } 
  902.               } 
  903.  
  904.               // Perform the login 
  905.               $user = wp_signon( apply_filters( 'woocommerce_login_credentials', $creds ), is_ssl() ); 
  906.  
  907.               if ( is_wp_error( $user ) ) { 
  908.                   $message = $user->get_error_message(); 
  909.                   $message = str_replace( '<strong>' . esc_html( $creds['user_login'] ) . '</strong>', '<strong>' . esc_html( $username ) . '</strong>', $message ); 
  910.                   throw new Exception( $message ); 
  911.               } else { 
  912.  
  913.                   if ( ! empty( $_POST['redirect'] ) ) { 
  914.                       $redirect = $_POST['redirect']; 
  915.                   } elseif ( wp_get_referer() ) { 
  916.                       $redirect = wp_get_referer(); 
  917.                   } else { 
  918.                       $redirect = wc_get_page_permalink( 'myaccount' ); 
  919.                   } 
  920.  
  921.                   wp_redirect( apply_filters( 'woocommerce_login_redirect', $redirect, $user ) ); 
  922.                   exit; 
  923.               } 
  924.           } catch ( Exception $e ) { 
  925.               wc_add_notice( apply_filters( 'login_errors', $e->getMessage() ), 'error' ); 
  926.               do_action( 'woocommerce_login_failed' ); 
  927.           } 
  928.       } 
  929.   }
