  543.   /** 
  544.    * Sync the variable product's attributes with the variations. 
  545.    */ 
  546.   public static function sync_attributes( $product, $children = false ) { 
  547.       if ( ! is_a( $product, 'WC_Product' ) ) { 
  548.           $product = wc_get_product( $product ); 
  549.       } 
  550.  
  551.       /** 
  552.        * Pre 2.4 handling where 'slugs' were saved instead of the full text attribute. 
  553.        * Attempt to get full version of the text attribute from the parent and UPDATE meta. 
  554.        */ 
  555.       if ( version_compare( get_post_meta( $product->get_id(), '_product_version', true ), '2.4.0', '<' ) ) { 
  556.           $parent_attributes = array_filter( (array) get_post_meta( $product->get_id(), '_product_attributes', true ) ); 
  557.  
  558.           if ( ! $children ) { 
  559.               $children = $product->get_children( 'edit' ); 
  560.           } 
  561.  
  562.           foreach ( $children as $child_id ) { 
  563.               $all_meta = get_post_meta( $child_id ); 
  564.  
  565.               foreach ( $all_meta as $name => $value ) { 
  566.                   if ( 0 !== strpos( $name, 'attribute_' ) ) { 
  567.                       continue; 
  568.                   } 
  569.                   if ( sanitize_title( $value[0] ) === $value[0] ) { 
  570.                       foreach ( $parent_attributes as $attribute ) { 
  571.                           if ( 'attribute_' . sanitize_title( $attribute['name'] ) !== $name ) { 
  572.                               continue; 
  573.                           } 
  574.                           $text_attributes = wc_get_text_attributes( $attribute['value'] ); 
  575.                           foreach ( $text_attributes as $text_attribute ) { 
  576.                               if ( sanitize_title( $text_attribute ) === $value[0] ) { 
  577.                                   update_post_meta( $child_id, $name, $text_attribute ); 
  578.                                   break; 
  579.                               } 
  580.                           } 
  581.                       } 
  582.                   } 
  583.               } 
  584.           } 
  585.       } 
  586.   }
