  32.    * Reads a product from the database and sets its data to the class. 
  33.    * 
  34.    * @since 3.0.0 
  35.    * @param WC_Product 
  36.    */ 
  37.   public function read( &$product ) { 
  38.       $product->set_defaults(); 
  39.  
  40.       if ( ! $product->get_id() || ! ( $post_object = get_post( $product->get_id() ) ) || 'product_variation' !== $post_object->post_type ) { 
  41.           return; 
  42.       } 
  43.  
  44.       $id = $product->get_id(); 
  45.       $product->set_parent_id( $post_object->post_parent ); 
  46.       $parent_id = $product->get_parent_id(); 
  47.  
  48.       // The post doesn't have a parent id, therefore its invalid and we should prevent this being created. 
  49.       if ( empty( $parent_id ) ) { 
  50.           throw new Exception( sprintf( 'No parent product set for variation #%d', $product->get_id() ), 422 ); 
  51.       } 
  52.  
  53.       // The post parent is not a valid variable product so we should prevent this being created. 
  54.       if ( 'product' !== get_post_type( $product->get_parent_id() ) ) { 
  55.           throw new Exception( sprintf( 'Invalid parent for variation #%d', $product->get_id() ), 422 ); 
  56.       } 
  57.  
  58.       $product->set_props( array( 
  59.           'name' => $post_object->post_title,  
  60.           'slug' => $post_object->post_name,  
  61.           'date_created' => 0 < $post_object->post_date_gmt ? wc_string_to_timestamp( $post_object->post_date_gmt ) : null,  
  62.           'date_modified' => 0 < $post_object->post_modified_gmt ? wc_string_to_timestamp( $post_object->post_modified_gmt ) : null,  
  63.           'status' => $post_object->post_status,  
  64.           'menu_order' => $post_object->menu_order,  
  65.           'reviews_allowed' => 'open' === $post_object->comment_status,  
  66.  ) ); 
  67.  
  68.       $this->read_downloads( $product ); 
  69.       $this->read_product_data( $product ); 
  70.       $this->read_extra_data( $product ); 
  71.       $product->set_attributes( wc_get_product_variation_attributes( $product->get_id() ) ); 
  72.  
  73.       /** 
  74.        * If a variation title is not in sync with the parent e.g. saved prior to 3.0, or if the parent title has changed, detect here and update. 
  75.        */ 
  76.       if ( version_compare( get_post_meta( $product->get_id(), '_product_version', true ), '3.0', '<' ) && 0 !== strpos( $post_object->post_title, get_post_field( 'post_title', $product->get_parent_id() ) ) ) { 
  77.           global $wpdb; 
  78.  
  79.           $new_title = $this->generate_product_title( $product ); 
  80.           $product->set_name( $new_title ); 
  81.           $wpdb->update( $wpdb->posts, array( 'post_title' => $new_title ), array( 'ID' => $product->get_id() ) ); 
  82.           clean_post_cache( $product->get_id() ); 
  83.       } 
  84.  
  85.       // Set object_read true once all data is read. 
  86.       $product->set_object_read( true ); 
  87.   }

  98.       }

  100.       $id = wp_insert_post( apply_filters( 'woocommerce_new_product_variation_data', array( 
  101.           'post_type' => 'product_variation',  
  102.           'post_status' => $product->get_status() ? $product->get_status() : 'publish',  
  103.           'post_author' => get_current_user_id(),  
  104.           'post_title' => $this->generate_product_title( $product ),  
  105.           'post_content' => '',  
  106.           'post_parent' => $product->get_parent_id(),

  109.           'menu_order' => $product->get_menu_order(),

  111.           'post_date_gmt' => gmdate( 'Y-m-d H:i:s', $product->get_date_created( 'edit' )->getTimestamp() ),

  122.           $product->save_meta_data(); 
  123.           $product->apply_changes(); 
  124.  
  125.           $this->update_version_and_type( $product );

  133.   /** 
  134.    * Updates an existing product. 
  135.    * 
  136.    * @since 3.0.0 
  137.    * @param WC_Product 
  138.    */ 
  139.   public function update( &$product ) { 
  140.       $product->save_meta_data(); 
  141.       $changes = $product->get_changes(); 
  142.       $title = $this->generate_product_title( $product ); 
  143.  
  144.       // Only update the post when the post data changes. 
  145.       if ( $title !== $product->get_name( 'edit' ) || array_intersect( array( 'parent_id', 'status', 'menu_order', 'date_created', 'date_modified' ), array_keys( $changes ) ) ) { 
  146.           $post_data = array( 
  147.               'post_title' => $title,  
  148.               'post_parent' => $product->get_parent_id( 'edit' ),  
  149.               'comment_status' => 'closed',  
  150.               'post_status' => $product->get_status( 'edit' ) ? $product->get_status( 'edit' ) : 'publish',  
  151.               'menu_order' => $product->get_menu_order( 'edit' ),  
  152.               'post_date' => gmdate( 'Y-m-d H:i:s', $product->get_date_created( 'edit' )->getOffsetTimestamp() ),  
  153.               'post_date_gmt' => gmdate( 'Y-m-d H:i:s', $product->get_date_created( 'edit' )->getTimestamp() ),  
  154.               'post_modified' => isset( $changes['date_modified'] ) ? gmdate( 'Y-m-d H:i:s', $product->get_date_modified( 'edit' )->getOffsetTimestamp() ) : current_time( 'mysql' ),  
  155.               'post_modified_gmt' => isset( $changes['date_modified'] ) ? gmdate( 'Y-m-d H:i:s', $product->get_date_modified( 'edit' )->getTimestamp() ) : current_time( 'mysql', 1 ),  
  156.  ); 
  157.  
  158.           /** 
  159.            * When updating this object, to prevent infinite loops, use $wpdb 
  160.            * to update data, since wp_update_post spawns more calls to the 
  161.            * save_post action. 
  162.            * 
  163.            * This ensures hooks are fired by either WP itself (admin screen save),  
  164.            * or an update purely from CRUD. 
  165.            */ 
  166.           if ( doing_action( 'save_post' ) ) { 
  167.               $GLOBALS['wpdb']->update( $GLOBALS['wpdb']->posts, $post_data, array( 'ID' => $product->get_id() ) ); 
  168.               clean_post_cache( $product->get_id() ); 
  169.           } else { 
  170.               wp_update_post( array_merge( array( 'ID' => $product->get_id() ), $post_data ) ); 
  171.           } 
  172.           $product->read_meta_data( true ); // Refresh internal meta data, in case things were hooked into `save_post` or another WP hook. 
  173.       } 
  174.  
  175.       $this->update_post_meta( $product ); 
  176.       $this->update_terms( $product ); 
  177.       $this->update_attributes( $product ); 
  178.       $this->handle_updated_props( $product ); 
  179.  
  180.       $product->apply_changes(); 
  181.  
  182.       $this->update_version_and_type( $product ); 
  183.  
  184.       $this->clear_caches( $product ); 
  185.  
  186.       do_action( 'woocommerce_update_product_variation', $product->get_id() ); 
  187.   }

  195.   /** 
  196.    * Generates a title with attribute information for a variation. 
  197.    * Products will get a title of the form "Name - Value, Value" or just "Name". 
  198.    * 
  199.    * @since 3.0.0 
  200.    * @param WC_Product 
  201.    * @return string 
  202.    */ 
  203.   protected function generate_product_title( $product ) { 
  204.       $attributes = (array) $product->get_attributes(); 
  205.  
  206.       // Don't include attributes if the product has 3+ attributes. 
  207.       $should_include_attributes = count( $attributes ) < 3; 
  208.  
  209.       // Don't include attributes if an attribute name has 2+ words. 
  210.       if ( $should_include_attributes ) { 
  211.           foreach ( $attributes as $name => $value ) { 
  212.               if ( false !== strpos( $name, '-' ) ) { 
  213.                   $should_include_attributes = false; 
  214.                   break; 
  215.               } 
  216.           } 
  217.       } 
  218.  
  219.       $should_include_attributes = apply_filters( 'woocommerce_product_variation_title_include_attributes', $should_include_attributes, $product ); 
  220.       $separator = apply_filters( 'woocommerce_product_variation_title_attributes_separator', ' - ', $product ); 
  221.       $title_base = get_post_field( 'post_title', $product->get_parent_id() ); 
  222.       $title_suffix = $should_include_attributes ? wc_get_formatted_variation( $product, true, false ) : ''; 
  223.  
  224.       return apply_filters( 'woocommerce_product_variation_title', rtrim( $title_base . $separator . $title_suffix, $separator ), $product, $title_base, $title_suffix ); 
  225.   }

  279.           'sku' => get_post_meta( $product->get_parent_id(), '_sku', true ),

  313.   /** 
  314.    * Update attribute meta values. 
  315.    * 
  316.    * @since 3.0.0 
  317.    * @param WC_Product 
  318.    * @param bool Force update. Used during create. 
  319.    */ 
  320.   protected function update_attributes( &$product, $force = false ) { 
  321.       $changes = $product->get_changes(); 
  322.  
  323.       if ( $force || array_key_exists( 'attributes', $changes ) ) { 
  324.           global $wpdb; 
  325.           $attributes = $product->get_attributes(); 
  326.           $updated_attribute_keys = array(); 
  327.           foreach ( $attributes as $key => $value ) { 
  328.               update_post_meta( $product->get_id(), 'attribute_' . $key, $value ); 
  329.               $updated_attribute_keys[] = 'attribute_' . $key; 
  330.           } 
  331.  
  332.           // Remove old taxonomies attributes so data is kept up to date - first get attribute key names. 
  333.           $delete_attribute_keys = $wpdb->get_col( $wpdb->prepare( "SELECT meta_key FROM {$wpdb->postmeta} WHERE meta_key LIKE 'attribute_%%' AND meta_key NOT IN ( '" . implode( "', '", array_map( 'esc_sql', $updated_attribute_keys ) ) . "' ) AND post_id = %d;", $product->get_id() ) ); 
  334.  
  335.           foreach ( $delete_attribute_keys as $key ) { 
  336.               delete_post_meta( $product->get_id(), $key ); 
  337.           } 
  338.       } 
  339.   }

  341.   /** 
  342.    * Helper method that updates all the post meta for a product based on it's settings in the WC_Product class. 
  343.    * 
  344.    * @since 3.0.0 
  345.    * @param WC_Product 
  346.    * @param bool Force update. Used during create. 
  347.    */ 
  348.   public function update_post_meta( &$product, $force = false ) { 
  349.       $meta_key_to_props = array( 
  350.           '_variation_description' => 'description',  
  351.  ); 
  352.  
  353.       $props_to_update = $force ? $meta_key_to_props : $this->get_props_to_update( $product, $meta_key_to_props ); 
  354.  
  355.       foreach ( $props_to_update as $meta_key => $prop ) { 
  356.           $value = $product->{"get_$prop"}( 'edit' ); 
  357.           $updated = update_post_meta( $product->get_id(), $meta_key, $value ); 
  358.           if ( $updated ) { 
  359.               $this->updated_props[] = $prop; 
  360.           } 
  361.       } 
  362.  
  363.       parent::update_post_meta( $product, $force ); 
  364.   }
