  7.  * Order Item Meta

  11.  * @deprecated  3.0.0 wc_display_item_meta function is used instead. 
  12.  * @class         order_item_meta 
  13.  * @version        2.4 
  14.  * @package        WooCommerce/Classes

  37.   public function __construct( $item = array(), $product = null ) {

  41.           $this->meta = array_filter( (array) $item );

  49.   /** 
  50.    * Display meta in a formatted list. 
  51.    * 
  52.    * @param bool $flat (default: false) 
  53.    * @param bool $return (default: false) 
  54.    * @param string $hideprefix (default: _) 
  55.    * @param  string $delimiter Delimiter used to separate items when $flat is true 
  56.    * @return string|void 
  57.    */ 
  58.   public function display( $flat = false, $return = false, $hideprefix = '_', $delimiter = ", \n" ) { 
  59.       $output = ''; 
  60.       $formatted_meta = $this->get_formatted( $hideprefix ); 
  61.  
  62.       if ( ! empty( $formatted_meta ) ) { 
  63.           $meta_list = array(); 
  64.  
  65.           foreach ( $formatted_meta as $meta ) { 
  66.               if ( $flat ) { 
  67.                   $meta_list[] = wp_kses_post( $meta['label'] . ': ' . $meta['value'] ); 
  68.               } else { 
  69.                   $meta_list[] = ' 
  70.                       <dt class="variation-' . sanitize_html_class( sanitize_text_field( $meta['key'] ) ) . '">' . wp_kses_post( $meta['label'] ) . ':</dt> 
  71.                       <dd class="variation-' . sanitize_html_class( sanitize_text_field( $meta['key'] ) ) . '">' . wp_kses_post( wpautop( make_clickable( $meta['value'] ) ) ) . '</dd> 
  72.                   '; 
  73.               } 
  74.           } 
  75.  
  76.           if ( ! empty( $meta_list ) ) { 
  77.               if ( $flat ) { 
  78.                   $output .= implode( $delimiter, $meta_list ); 
  79.               } else { 
  80.                   $output .= '<dl class="variation">' . implode( '', $meta_list ) . '</dl>'; 
  81.               } 
  82.           } 
  83.       } 
  84.  
  85.       $output = apply_filters( 'woocommerce_order_items_meta_display', $output, $this ); 
  86.  
  87.       if ( $return ) { 
  88.           return $output; 
  89.       } else { 
  90.           echo $output; 
  91.       } 
  92.   }

  94.   /** 
  95.    * Return an array of formatted item meta in format e.g. 
  96.    * 
  97.    * array( 
  98.    *   'pa_size' => array( 
  99.    *     'label' => 'Size',  
  100.    *     'value' => 'Medium',  
  101.    * ) 
  102.    * ) 
  103.    * 
  104.    * @since 2.4 
  105.    * @param string $hideprefix exclude meta when key is prefixed with this, defaults to `_` 
  106.    * @return array 
  107.    */ 
  108.   public function get_formatted( $hideprefix = '_' ) { 
  109.       if ( $this->legacy ) { 
  110.           return $this->get_formatted_legacy( $hideprefix ); 
  111.       } 
  112.  
  113.       $formatted_meta = array(); 
  114.  
  115.       if ( ! empty( $this->item['item_meta_array'] ) ) { 
  116.           foreach ( $this->item['item_meta_array'] as $meta_id => $meta ) { 
  117.               if ( "" === $meta->value || is_serialized( $meta->value ) || ( ! empty( $hideprefix ) && substr( $meta->key, 0, 1 ) === $hideprefix ) ) { 
  118.                   continue; 
  119.               } 
  120.  
  121.               $attribute_key = urldecode( str_replace( 'attribute_', '', $meta->key ) ); 
  122.               $meta_value = $meta->value; 
  123.  
  124.               // If this is a term slug, get the term's nice name 
  125.               if ( taxonomy_exists( $attribute_key ) ) { 
  126.                   $term = get_term_by( 'slug', $meta_value, $attribute_key ); 
  127.  
  128.                   if ( ! is_wp_error( $term ) && is_object( $term ) && $term->name ) { 
  129.                       $meta_value = $term->name; 
  130.                   } 
  131.               } 
  132.  
  133.               $formatted_meta[ $meta_id ] = array( 
  134.                   'key' => $meta->key,  
  135.                   'label' => wc_attribute_label( $attribute_key, $this->product ),  
  136.                   'value' => apply_filters( 'woocommerce_order_item_display_meta_value', $meta_value ),  
  137.  ); 
  138.           } 
  139.       } 
  140.  
  141.       return apply_filters( 'woocommerce_order_items_meta_get_formatted', $formatted_meta, $this ); 
  142.   }

  144.   /** 
  145.    * Return an array of formatted item meta in format e.g. 
  146.    * Handles @deprecated args. 
  147.    * @return array 
  148.    */ 
  149.   public function get_formatted_legacy( $hideprefix = '_' ) { 
  150.       if ( ! is_ajax() ) { 
  151.           wc_deprecated_function( 'get_formatted_legacy', '2.4', 'Item Meta Data is being called with legacy arguments' ); 
  152.       } 
  153.  
  154.       $formatted_meta = array(); 
  155.  
  156.       foreach ( $this->meta as $meta_key => $meta_values ) { 
  157.           if ( empty( $meta_values ) || ( ! empty( $hideprefix ) && substr( $meta_key, 0, 1 ) == $hideprefix ) ) { 
  158.               continue; 
  159.           } 
  160.           foreach ( (array) $meta_values as $meta_value ) { 
  161.               // Skip serialised meta 
  162.               if ( is_serialized( $meta_value ) ) { 
  163.                   continue; 
  164.               } 
  165.  
  166.               $attribute_key = urldecode( str_replace( 'attribute_', '', $meta_key ) ); 
  167.  
  168.               // If this is a term slug, get the term's nice name 
  169.               if ( taxonomy_exists( $attribute_key ) ) { 
  170.                   $term = get_term_by( 'slug', $meta_value, $attribute_key ); 
  171.                   if ( ! is_wp_error( $term ) && is_object( $term ) && $term->name ) { 
  172.                       $meta_value = $term->name; 
  173.                   } 
  174.               } 
  175.  
  176.               // Unique key required 
  177.               $formatted_meta_key = $meta_key; 
  178.               $loop = 0; 
  179.               while ( isset( $formatted_meta[ $formatted_meta_key ] ) ) { 
  180.                   $loop ++; 
  181.                   $formatted_meta_key = $meta_key . '-' . $loop; 
  182.               } 
  183.  
  184.               $formatted_meta[ $formatted_meta_key ] = array( 
  185.                   'key' => $meta_key,  
  186.                   'label' => wc_attribute_label( $attribute_key, $this->product ),  
  187.                   'value' => apply_filters( 'woocommerce_order_item_display_meta_value', $meta_value ),  
  188.  ); 
  189.           } 
  190.       } 
  191.  
  192.       return $formatted_meta; 
  193.   }
