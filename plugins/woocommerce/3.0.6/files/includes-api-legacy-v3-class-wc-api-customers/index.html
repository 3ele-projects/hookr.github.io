  539.           'number' => $users_per_page,  
  540.  ); 
  541.  
  542.       // Custom Role 
  543.       if ( ! empty( $args['role'] ) ) { 
  544.           $query_args['role'] = $args['role']; 
  545.  
  546.           // Show users on all roles 
  547.           if ( 'all' === $query_args['role'] ) { 
  548.               unset( $query_args['role'] ); 
  549.           } 
  550.       } 
  551.  
  552.       // Search 
  553.       if ( ! empty( $args['q'] ) ) { 
  554.           $query_args['search'] = $args['q']; 
  555.       } 
  556.  
  557.       // Limit number of users returned 
  558.       if ( ! empty( $args['limit'] ) ) { 
  559.           if ( -1 == $args['limit'] ) { 
  560.               unset( $query_args['number'] ); 
  561.           } else { 
  562.               $query_args['number'] = absint( $args['limit'] ); 
  563.               $users_per_page = absint( $args['limit'] ); 
  564.           } 
  565.       } else { 
  566.           $args['limit'] = $query_args['number']; 
  567.       } 
  568.  
  569.       // Page 
  570.       $page = ( isset( $args['page'] ) ) ? absint( $args['page'] ) : 1; 
  571.  
  572.       // Offset 
  573.       if ( ! empty( $args['offset'] ) ) { 
  574.           $query_args['offset'] = absint( $args['offset'] ); 
  575.       } else { 
  576.           $query_args['offset'] = $users_per_page * ( $page - 1 ); 
  577.       } 
  578.  
  579.       // Created date 
  580.       if ( ! empty( $args['created_at_min'] ) ) { 
  581.           $this->created_at_min = $this->server->parse_datetime( $args['created_at_min'] ); 
  582.       } 
  583.  
  584.       if ( ! empty( $args['created_at_max'] ) ) { 
  585.           $this->created_at_max = $this->server->parse_datetime( $args['created_at_max'] ); 
  586.       } 
  587.  
  588.       // Order (ASC or DESC, ASC by default) 
  589.       if ( ! empty( $args['order'] ) ) { 
  590.           $query_args['order'] = $args['order']; 
  591.       } 
  592.  
  593.       // Orderby 
  594.       if ( ! empty( $args['orderby'] ) ) { 
  595.           $query_args['orderby'] = $args['orderby']; 
  596.  
  597.           // Allow sorting by meta value 
  598.           if ( ! empty( $args['orderby_meta_key'] ) ) { 
  599.               $query_args['meta_key'] = $args['orderby_meta_key']; 
  600.           } 
  601.       } 
  602.  
  603.       $query = new WP_User_Query( $query_args ); 
  604.  
  605.       // Helper members for pagination headers 
  606.       $query->total_pages = ( -1 == $args['limit'] ) ? 1 : ceil( $query->get_total() / $users_per_page ); 
  607.       $query->page = $page; 
  608.  
  609.       return $query; 
  610.   }

  681.   /** 
  682.    * Validate the request by checking: 
  683.    * 
  684.    * 1) the ID is a valid integer 
  685.    * 2) the ID returns a valid WP_User 
  686.    * 3) the current user has the proper permissions 
  687.    * 
  688.    * @since 2.1 
  689.    * @see WC_API_Resource::validate_request() 
  690.    * @param integer $id the customer ID 
  691.    * @param string $type the request type, unused because this method overrides the parent class 
  692.    * @param string $context the context of the request, either `read`, `edit` or `delete` 
  693.    * @return int|WP_Error valid user ID or WP_Error if any of the checks fails 
  694.    */ 
  695.   protected function validate_request( $id, $type, $context ) { 
  696.  
  697.       try { 
  698.           $id = absint( $id ); 
  699.  
  700.           // validate ID 
  701.           if ( empty( $id ) ) { 
  702.               throw new WC_API_Exception( 'woocommerce_api_invalid_customer_id', __( 'Invalid customer ID', 'woocommerce' ), 404 ); 
  703.           } 
  704.  
  705.           // non-existent IDs return a valid WP_User object with the user ID = 0 
  706.           $customer = new WP_User( $id ); 
  707.  
  708.           if ( 0 === $customer->ID ) { 
  709.               throw new WC_API_Exception( 'woocommerce_api_invalid_customer', __( 'Invalid customer', 'woocommerce' ), 404 ); 
  710.           } 
  711.  
  712.           // validate permissions 
  713.           switch ( $context ) { 
  714.  
  715.               case 'read': 
  716.                   if ( ! current_user_can( 'list_users' ) ) { 
  717.                       throw new WC_API_Exception( 'woocommerce_api_user_cannot_read_customer', __( 'You do not have permission to read this customer', 'woocommerce' ), 401 ); 
  718.                   } 
  719.                   break; 
  720.  
  721.               case 'edit': 
  722.                   if ( ! current_user_can( 'edit_users' ) ) { 
  723.                       throw new WC_API_Exception( 'woocommerce_api_user_cannot_edit_customer', __( 'You do not have permission to edit this customer', 'woocommerce' ), 401 ); 
  724.                   } 
  725.                   break; 
  726.  
  727.               case 'delete': 
  728.                   if ( ! current_user_can( 'delete_users' ) ) { 
  729.                       throw new WC_API_Exception( 'woocommerce_api_user_cannot_delete_customer', __( 'You do not have permission to delete this customer', 'woocommerce' ), 401 ); 
  730.                   } 
  731.                   break; 
  732.           } 
  733.  
  734.           return $id; 
  735.       } catch ( WC_API_Exception $e ) {

  752.   /** 
  753.    * Bulk update or insert customers 
  754.    * Accepts an array with customers in the formats supported by 
  755.    * WC_API_Customers->create_customer() and WC_API_Customers->edit_customer() 
  756.    * 
  757.    * @since 2.4.0 
  758.    * @param array $data 
  759.    * @return array 
  760.    */ 
  761.   public function bulk( $data ) { 
  762.  
  763.       try { 
  764.           if ( ! isset( $data['customers'] ) ) { 
  765.               throw new WC_API_Exception( 'woocommerce_api_missing_customers_data', sprintf( __( 'No %1$s data specified to create/edit %1$s', 'woocommerce' ), 'customers' ), 400 ); 
  766.           } 
  767.  
  768.           $data = $data['customers']; 
  769.           $limit = apply_filters( 'woocommerce_api_bulk_limit', 100, 'customers' ); 
  770.  
  771.           // Limit bulk operation 
  772.           if ( count( $data ) > $limit ) { 
  773.               throw new WC_API_Exception( 'woocommerce_api_customers_request_entity_too_large', sprintf( __( 'Unable to accept more than %s items for this request.', 'woocommerce' ), $limit ), 413 ); 
  774.           } 
  775.  
  776.           $customers = array(); 
  777.  
  778.           foreach ( $data as $_customer ) { 
  779.               $customer_id = 0; 
  780.  
  781.               // Try to get the customer ID 
  782.               if ( isset( $_customer['id'] ) ) { 
  783.                   $customer_id = intval( $_customer['id'] ); 
  784.               } 
  785.  
  786.               if ( $customer_id ) { 
  787.  
  788.                   // Customer exists / edit customer 
  789.                   $edit = $this->edit_customer( $customer_id, array( 'customer' => $_customer ) ); 
  790.  
  791.                   if ( is_wp_error( $edit ) ) { 
  792.                       $customers[] = array( 
  793.                           'id' => $customer_id,  
  794.                           'error' => array( 'code' => $edit->get_error_code(), 'message' => $edit->get_error_message() ),  
  795.  ); 
  796.                   } else { 
  797.                       $customers[] = $edit['customer']; 
  798.                   } 
  799.               } else { 
  800.  
  801.                   // Customer don't exists / create customer 
  802.                   $new = $this->create_customer( array( 'customer' => $_customer ) ); 
  803.  
  804.                   if ( is_wp_error( $new ) ) { 
  805.                       $customers[] = array( 
  806.                           'id' => $customer_id,  
  807.                           'error' => array( 'code' => $new->get_error_code(), 'message' => $new->get_error_message() ),  
  808.  ); 
  809.                   } else { 
  810.                       $customers[] = $new['customer']; 
  811.                   } 
  812.               } 
  813.           } 
  814.  
  815.           return array( 'customers' => apply_filters( 'woocommerce_api_customers_bulk_response', $customers, $this ) ); 
  816.       } catch ( WC_API_Exception $e ) { 
  817.           return new WP_Error( $e->getErrorCode(), $e->getMessage(), array( 'status' => $e->getCode() ) ); 
  818.       } 
  819.   }
