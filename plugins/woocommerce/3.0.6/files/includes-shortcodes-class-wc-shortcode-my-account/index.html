  82.                   }

  189.   public static function lost_password() { 
  190.       /** 
  191.        * After sending the reset link, don't show the form again. 
  192.        */ 
  193.       if ( ! empty( $_GET['reset-link-sent'] ) ) { 
  194.           return wc_get_template( 'myaccount/lost-password-confirmation.php' ); 
  195.  
  196.       /** 
  197.        * Process reset key / login from email confirmation link 
  198.        */ 
  199.       } elseif ( ! empty( $_GET['show-reset-form'] ) ) { 
  200.           if ( isset( $_COOKIE[ 'wp-resetpass-' . COOKIEHASH ] ) && 0 < strpos( $_COOKIE[ 'wp-resetpass-' . COOKIEHASH ], ':' ) ) { 
  201.               list( $rp_login, $rp_key ) = array_map( 'wc_clean', explode( ':', wp_unslash( $_COOKIE[ 'wp-resetpass-' . COOKIEHASH ] ), 2 ) ); 
  202.               $user = self::check_password_reset_key( $rp_key, $rp_login ); 
  203.  
  204.               // reset key / login is correct, display reset password form with hidden key / login values 
  205.               if ( is_object( $user ) ) { 
  206.                   return wc_get_template( 'myaccount/form-reset-password.php', array( 
  207.                       'key' => $rp_key,  
  208.                       'login' => $rp_login,  
  209.  ) ); 
  210.               } else { 
  211.                   self::set_reset_password_cookie(); 
  212.               } 
  213.           } 
  214.       } 
  215.  
  216.       // Show lost password form by default 
  217.       wc_get_template( 'myaccount/form-lost-password.php', array( 
  218.           'form' => 'lost_password',  
  219.  ) ); 
  220.   }

  222.   /** 
  223.    * Handles sending password retrieval email to customer. 
  224.    * 
  225.    * Based on retrieve_password() in core wp-login.php. 
  226.    * 
  227.    * @uses $wpdb WordPress Database object 
  228.    * @return bool True: when finish. False: on error 
  229.    */ 
  230.   public static function retrieve_password() { 
  231.       global $wpdb, $wp_hasher; 
  232.  
  233.       $login = trim( $_POST['user_login'] ); 
  234.  
  235.       if ( empty( $login ) ) { 
  236.  
  237.           wc_add_notice( __( 'Enter a username or email address.', 'woocommerce' ), 'error' ); 
  238.           return false; 
  239.  
  240.       } else { 
  241.           // Check on username first, as customers can use emails as usernames. 
  242.           $user_data = get_user_by( 'login', $login ); 
  243.       } 
  244.  
  245.       // If no user found, check if it login is email and lookup user based on email. 
  246.       if ( ! $user_data && is_email( $login ) && apply_filters( 'woocommerce_get_username_from_email', true ) ) { 
  247.           $user_data = get_user_by( 'email', $login ); 
  248.       } 
  249.  
  250.       $errors = new WP_Error(); 
  251.  
  252.       do_action( 'lostpassword_post', $errors ); 
  253.  
  254.       if ( $errors->get_error_code() ) { 
  255.           wc_add_notice( $errors->get_error_message(), 'error' ); 
  256.           return false; 
  257.       } 
  258.  
  259.       if ( ! $user_data ) { 
  260.           wc_add_notice( __( 'Invalid username or email.', 'woocommerce' ), 'error' ); 
  261.           return false; 
  262.       } 
  263.  
  264.       if ( is_multisite() && ! is_user_member_of_blog( $user_data->ID, get_current_blog_id() ) ) { 
  265.           wc_add_notice( __( 'Invalid username or email.', 'woocommerce' ), 'error' ); 
  266.           return false; 
  267.       } 
  268.  
  269.       // redefining user_login ensures we return the right case in the email 
  270.       $user_login = $user_data->user_login; 
  271.  
  272.       do_action( 'retrieve_password', $user_login ); 
  273.  
  274.       $allow = apply_filters( 'allow_password_reset', true, $user_data->ID ); 
  275.  
  276.       if ( ! $allow ) { 
  277.  
  278.           wc_add_notice( __( 'Password reset is not allowed for this user', 'woocommerce' ), 'error' ); 
  279.           return false; 
  280.  
  281.       } elseif ( is_wp_error( $allow ) ) { 
  282.  
  283.           wc_add_notice( $allow->get_error_message(), 'error' ); 
  284.           return false; 
  285.       } 
  286.  
  287.       // Get password reset key (function introduced in WordPress 4.4). 
  288.       $key = get_password_reset_key( $user_data ); 
  289.  
  290.       // Send email notification 
  291.       WC()->mailer(); // load email classes 
  292.       do_action( 'woocommerce_reset_password_notification', $user_login, $key ); 
  293.  
  294.       return true; 
  295.   }
