  105.        */

  111.   /** 
  112.    * Method to update an order in the database. 
  113.    * @param WC_Order $order 
  114.    */ 
  115.   public function update( &$order ) { 
  116.       $order->save_meta_data(); 
  117.       $order->set_version( WC_VERSION ); 
  118.  
  119.       $changes = $order->get_changes(); 
  120.  
  121.       // Only update the post when the post data changes. 
  122.       if ( array_intersect( array( 'date_created', 'date_modified', 'status', 'parent_id', 'post_excerpt' ), array_keys( $changes ) ) ) { 
  123.           $post_data = array( 
  124.               'post_date' => gmdate( 'Y-m-d H:i:s', $order->get_date_created( 'edit' )->getOffsetTimestamp() ),  
  125.               'post_date_gmt' => gmdate( 'Y-m-d H:i:s', $order->get_date_created( 'edit' )->getTimestamp() ),  
  126.               'post_status' => 'wc-' . ( $order->get_status( 'edit' ) ? $order->get_status( 'edit' ) : apply_filters( 'woocommerce_default_order_status', 'pending' ) ),  
  127.               'post_parent' => $order->get_parent_id(),  
  128.               'post_excerpt' => $this->get_post_excerpt( $order ),  
  129.               'post_modified' => isset( $changes['date_modified'] ) ? gmdate( 'Y-m-d H:i:s', $order->get_date_modified( 'edit' )->getOffsetTimestamp() ) : current_time( 'mysql' ),  
  130.               'post_modified_gmt' => isset( $changes['date_modified'] ) ? gmdate( 'Y-m-d H:i:s', $order->get_date_modified( 'edit' )->getTimestamp() ) : current_time( 'mysql', 1 ),  
  131.  ); 
  132.  
  133.           /** 
  134.            * When updating this object, to prevent infinite loops, use $wpdb 
  135.            * to update data, since wp_update_post spawns more calls to the 
  136.            * save_post action. 
  137.            * 
  138.            * This ensures hooks are fired by either WP itself (admin screen save),  
  139.            * or an update purely from CRUD. 
  140.            */ 
  141.           if ( doing_action( 'save_post' ) ) { 
  142.               $GLOBALS['wpdb']->update( $GLOBALS['wpdb']->posts, $post_data, array( 'ID' => $order->get_id() ) ); 
  143.               clean_post_cache( $order->get_id() ); 
  144.           } else { 
  145.               wp_update_post( array_merge( array( 'ID' => $order->get_id() ), $post_data ) ); 
  146.           } 
  147.           $order->read_meta_data( true ); // Refresh internal meta data, in case things were hooked into `save_post` or another WP hook. 
  148.       } 
  149.       $this->update_post_meta( $order ); 
  150.       $order->apply_changes(); 
  151.       $this->clear_caches( $order ); 
  152.   }

  284.   /** 
  285.    * Read order items of a specific type from the database for this order. 
  286.    * 
  287.    * @param  WC_Order $order 
  288.    * @param  string $type 
  289.    * @return array 
  290.    */ 
  291.   public function read_items( $order, $type ) { 
  292.       global $wpdb; 
  293.  
  294.       // Get from cache if available. 
  295.       $items = wp_cache_get( 'order-items-' . $order->get_id(), 'orders' ); 
  296.  
  297.       if ( false === $items ) { 
  298.           $get_items_sql = $wpdb->prepare( "SELECT order_item_type, order_item_id, order_id, order_item_name FROM {$wpdb->prefix}woocommerce_order_items WHERE order_id = %d ORDER BY order_item_id;", $order->get_id() ); 
  299.           $items = $wpdb->get_results( $get_items_sql ); 
  300.           foreach ( $items as $item ) { 
  301.               wp_cache_set( 'item-' . $item->order_item_id, $item, 'order-items' ); 
  302.           } 
  303.           wp_cache_set( 'order-items-' . $order->get_id(), $items, 'orders' ); 
  304.       } 
  305.  
  306.       $items = wp_list_filter( $items, array( 'order_item_type' => $type ) ); 
  307.  
  308.       if ( ! empty( $items ) ) { 
  309.           $items = array_map( array( 'WC_Order_Factory', 'get_order_item' ), array_combine( wp_list_pluck( $items, 'order_item_id' ), $items ) ); 
  310.       } else { 
  311.           $items = array(); 
  312.       } 
  313.  
  314.       return $items; 
  315.   }
