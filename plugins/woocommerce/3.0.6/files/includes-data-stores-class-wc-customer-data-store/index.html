  123.    * Method to read a customer object. 
  124.    * 
  125.    * @since 3.0.0

  129.   public function read( &$customer ) { 
  130.       global $wpdb; 
  131.  
  132.       // User object is required. 
  133.       if ( ! $customer->get_id() || ! ( $user_object = get_user_by( 'id', $customer->get_id() ) ) || empty( $user_object->ID ) ) { 
  134.           throw new Exception( __( 'Invalid customer.', 'woocommerce' ) ); 
  135.       } 
  136.  
  137.       // Only users on this site should be read. 
  138.       if ( is_multisite() && ! is_user_member_of_blog( $customer->get_id() ) ) { 
  139.           throw new Exception( __( 'Invalid customer.', 'woocommerce' ) ); 
  140.       } 
  141.  
  142.       $customer_id = $customer->get_id(); 
  143.       // Load meta but exclude deprecated props. 
  144.       $user_meta = array_diff_key( array_map( 'wc_flatten_meta_callback', get_user_meta( $customer_id ) ), array_flip( array( 'country', 'state', 'postcode', 'city', 'address', 'address_2', 'default', 'location' ) ) ); 
  145.       $customer->set_props( $user_meta ); 
  146.       $customer->set_props( array( 
  147.           'is_paying_customer' => get_user_meta( $customer_id, 'paying_customer', true ),  
  148.           'email' => $user_object->user_email,  
  149.           'username' => $user_object->user_login,  
  150.           'date_created' => $user_object->user_registered, // Mysql string in local format. 
  151.           'date_modified' => get_user_meta( $customer_id, 'last_update', true ),  
  152.           'role' => ! empty( $user_object->roles[0] ) ? $user_object->roles[0] : 'customer',  
  153.  ) ); 
  154.       $customer->read_meta_data(); 
  155.       $customer->set_object_read( true ); 
  156.       do_action( 'woocommerce_customer_loaded', $customer ); 
  157.   }

  209.   private function update_user_meta( $customer ) { 
  210.       $updated_props = array(); 
  211.       $changed_props = $customer->get_changes(); 
  212.  
  213.       $meta_key_to_props = array( 
  214.           'paying_customer' => 'is_paying_customer',  
  215.           'first_name' => 'first_name',  
  216.           'last_name' => 'last_name',  
  217.  ); 
  218.  
  219.       foreach ( $meta_key_to_props as $meta_key => $prop ) { 
  220.           if ( ! array_key_exists( $prop, $changed_props ) ) { 
  221.               continue; 
  222.           } 
  223.  
  224.           if ( update_user_meta( $customer->get_id(), $meta_key, $customer->{"get_$prop"}( 'edit' ) ) ) { 
  225.               $updated_props[] = $prop; 
  226.           } 
  227.       } 
  228.  
  229.       $billing_address_props = array( 
  230.           'billing_first_name' => 'billing_first_name',  
  231.           'billing_last_name' => 'billing_last_name',  
  232.           'billing_company' => 'billing_company',  
  233.           'billing_address_1' => 'billing_address_1',  
  234.           'billing_address_2' => 'billing_address_2',  
  235.           'billing_city' => 'billing_city',  
  236.           'billing_state' => 'billing_state',  
  237.           'billing_postcode' => 'billing_postcode',  
  238.           'billing_country' => 'billing_country',  
  239.           'billing_email' => 'billing_email',  
  240.           'billing_phone' => 'billing_phone',  
  241.  ); 
  242.  
  243.       foreach ( $billing_address_props as $meta_key => $prop ) { 
  244.           $prop_key = substr( $prop, 8 ); 
  245.           if ( ! isset( $changed_props['billing'] ) || ! array_key_exists( $prop_key, $changed_props['billing'] ) ) { 
  246.               continue; 
  247.           } 
  248.           if ( update_user_meta( $customer->get_id(), $meta_key, $customer->{"get_$prop"}( 'edit' ) ) ) { 
  249.               $updated_props[] = $prop; 
  250.           } 
  251.       } 
  252.  
  253.       $shipping_address_props = array( 
  254.           'shipping_first_name' => 'shipping_first_name',  
  255.           'shipping_last_name' => 'shipping_last_name',  
  256.           'shipping_company' => 'shipping_company',  
  257.           'shipping_address_1' => 'shipping_address_1',

  332.   /**

  341.  
  342.       if ( '' === $spent ) {

  344.  
  345.           $statuses = array_map( 'esc_sql', wc_get_is_paid_statuses() ); 
  346.           $spent = $wpdb->get_var( "SELECT SUM(meta2.meta_value) 
  347.               FROM $wpdb->posts as posts

  352.               AND     posts.post_type = 'shop_order' 
  353.               AND     posts.post_status   IN ( 'wc-" . implode( "', 'wc-", $statuses ) . "' ) 
  354.               AND     meta2.meta_key = '_order_total' 
  355.           " ); 
  356.  
  357.           if ( ! $spent ) { 
  358.               $spent = 0; 
  359.           } 
  360.           update_user_meta( $customer->get_id(), '_money_spent', $spent );
