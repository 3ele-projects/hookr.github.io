  208.                   'terms' => $request[ $key ],  
  209.  ); 
  210.           }

  212.  
  213.       // Filter product type by slug. 
  214.       if ( ! empty( $request['type'] ) ) {

  216.               'taxonomy' => 'product_type',  
  217.               'field' => 'slug',

  222.       // Filter by attribute and term. 
  223.       if ( ! empty( $request['attribute'] ) && ! empty( $request['attribute_term'] ) ) { 
  224.           if ( in_array( $request['attribute'], wc_get_attribute_taxonomy_names(), true ) ) { 
  225.               $tax_query[] = array( 
  226.                   'taxonomy' => $request['attribute'],

  230.           }

  242.               'terms' => 'featured',

  249.           // Include the current string as a SKU too. 
  250.           if ( 1 < count( $skus ) ) { 
  251.               $skus[] = $request['sku']; 
  252.           }

  255.               'key' => '_sku',

  266.  ) );

  268.  
  269.       // Price filter.

  273.

  277.               'key' => '_stock_status',

  279.  ) ); 
  280.       } 
  281.  
  282.       // Filter by on sale products.

  288.       // Force the post_type argument, since it's not a user input variable. 
  289.       if ( ! empty( $request['sku'] ) ) {

  291.       } else { 
  292.           $args['post_type'] = $this->post_type; 
  293.       } 
  294.  
  295.       return $args;
