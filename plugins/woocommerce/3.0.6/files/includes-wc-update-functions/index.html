  41. function wc_update_200_permalinks() {

  53.  
  54.       if ( 'yes' == get_option( 'woocommerce_prepend_shop_page_to_products' ) ) {

  56.       } else { 
  57.           if ( ( $product_slug = get_option( 'woocommerce_product_slug' ) ) !== false && ! empty( $product_slug ) ) { 
  58.               $product_base = trailingslashit( $product_slug ); 
  59.           } else { 
  60.               $product_base = trailingslashit( _x( 'product', 'slug', 'woocommerce' ) ); 
  61.           } 
  62.       } 
  63.  
  64.       if ( get_option( 'woocommerce_prepend_category_to_products' ) == 'yes' ) 
  65.           $product_base .= trailingslashit( '%product_cat%' ); 
  66.  
  67.       $permalinks = array( 
  68.           'product_base' => untrailingslashit( $product_base ),  
  69.           'category_base' => untrailingslashit( $category_base . $category_slug ),  
  70.           'attribute_base' => untrailingslashit( $category_base ),

  192.   // Now its time for the massive update to line items - move them to the new DB tables

  207.               $order_item['line_total'] = $order_item['cost'] * $order_item['qty']; 
  208.               $order_item['line_subtotal_tax'] = $order_item['line_tax']; 
  209.               $order_item['line_subtotal'] = $order_item['line_total']; 
  210.           } 
  211.  
  212.           $order_item['line_tax'] = isset( $order_item['line_tax'] ) ? $order_item['line_tax'] : 0; 
  213.           $order_item['line_total'] = isset( $order_item['line_total'] ) ? $order_item['line_total'] : 0; 
  214.           $order_item['line_subtotal_tax'] = isset( $order_item['line_subtotal_tax'] ) ? $order_item['line_subtotal_tax'] : 0; 
  215.           $order_item['line_subtotal'] = isset( $order_item['line_subtotal'] ) ? $order_item['line_subtotal'] : 0; 
  216.  
  217.           $item_id = wc_add_order_item( $order_item_row->post_id, array( 
  218.                'order_item_name' => $order_item['name'],

  221.  
  222.            // Add line item meta

  231.                wc_add_order_item_meta( $item_id, '_line_tax', wc_format_decimal( $order_item['line_tax'] ) ); 
  232.  
  233.                $meta_rows = array();

  235.               // Insert meta 
  236.               if ( ! empty( $order_item['item_meta'] ) ) { 
  237.                   foreach ( $order_item['item_meta'] as $key => $meta ) { 
  238.                       // Backwards compatibility 
  239.                       if ( is_array( $meta ) && isset( $meta['meta_name'] ) ) {

  614. function wc_update_240_shipping_methods() { 
  615.   global $wpdb; 
  616.   /** 
  617.    * Flat Rate Shipping. 
  618.    * Update legacy options to new math based options. 
  619.    */ 
  620.   $shipping_methods = array( 
  621.       'woocommerce_flat_rates' => new WC_Shipping_Legacy_Flat_Rate(),  
  622.       'woocommerce_international_delivery_flat_rates' => new WC_Shipping_Legacy_International_Delivery(),  
  623.  ); 
  624.   foreach ( $shipping_methods as $flat_rate_option_key => $shipping_method ) { 
  625.       // Stop this running more than once if routine is repeated 
  626.       if ( version_compare( $shipping_method->get_option( 'version', 0 ), '2.4.0', '<' ) ) { 
  627.           $has_classes = sizeof( WC()->shipping->get_shipping_classes() ) > 0; 
  628.           $cost_key = $has_classes ? 'no_class_cost' : 'cost'; 
  629.           $min_fee = $shipping_method->get_option( 'minimum_fee' ); 
  630.           $math_cost_strings = array( 'cost' => array(), 'no_class_cost' => array() ); 
  631.           $math_cost_strings[ $cost_key ][] = $shipping_method->get_option( 'cost' ); 
  632.  
  633.           if ( $fee = $shipping_method->get_option( 'fee' ) ) { 
  634.               $math_cost_strings[ $cost_key ][] = strstr( $fee, '%' ) ? '[fee percent="' . str_replace( '%', '', $fee ) . '" min="' . esc_attr( $min_fee ) . '"]' : $fee; 
  635.           } 
  636.  
  637.           foreach ( WC()->shipping->get_shipping_classes() as $shipping_class ) { 
  638.               $rate_key = 'class_cost_' . $shipping_class->slug; 
  639.               $math_cost_strings[ $rate_key ] = $math_cost_strings['no_class_cost']; 
  640.           } 
  641.  
  642.           if ( $flat_rates = array_filter( (array) get_option( $flat_rate_option_key, array() ) ) ) { 
  643.               foreach ( $flat_rates as $shipping_class => $rate ) { 
  644.                   $rate_key = 'class_cost_' . $shipping_class; 
  645.                   if ( $rate['cost'] || $rate['fee'] ) { 
  646.                       $math_cost_strings[ $rate_key ][] = $rate['cost']; 
  647.                       $math_cost_strings[ $rate_key ][] = strstr( $rate['fee'], '%' ) ? '[fee percent="' . str_replace( '%', '', $rate['fee'] ) . '" min="' . esc_attr( $min_fee ) . '"]' : $rate['fee']; 
  648.                   } 
  649.               } 
  650.           } 
  651.  
  652.           if ( 'item' === $shipping_method->type ) { 
  653.               foreach ( $math_cost_strings as $key => $math_cost_string ) { 
  654.                   $math_cost_strings[ $key ] = array_filter( array_map( 'trim', $math_cost_strings[ $key ] ) ); 
  655.                   if ( ! empty( $math_cost_strings[ $key ] ) ) { 
  656.                       $last_key = max( 0, sizeof( $math_cost_strings[ $key ] ) - 1 ); 
  657.                       $math_cost_strings[ $key ][0] = '( ' . $math_cost_strings[ $key ][0]; 
  658.                       $math_cost_strings[ $key ][ $last_key ] .= ' ) * [qty]'; 
  659.                   } 
  660.               } 
  661.           } 
  662.  
  663.           $math_cost_strings['cost'][] = $shipping_method->get_option( 'cost_per_order' ); 
  664.  
  665.           // Save settings 
  666.           foreach ( $math_cost_strings as $option_id => $math_cost_string ) { 
  667.               $shipping_method->settings[ $option_id ] = implode( ' + ', array_filter( $math_cost_string ) ); 
  668.           } 
  669.  
  670.           $shipping_method->settings['version'] = '2.4.0'; 
  671.           $shipping_method->settings['type'] = 'item' === $shipping_method->settings['type'] ? 'class' : $shipping_method->settings['type']; 
  672.  
  673.           update_option( $shipping_method->plugin_id . $shipping_method->id . '_settings', $shipping_method->settings ); 
  674.       } 
  675.   } 
  676. }
