  152.   public function create_item( $request ) { 
  153.       if ( ! empty( $request['id'] ) ) { 
  154.           /** translators: %s: post type */ 
  155.           return new WP_Error( "woocommerce_rest_{$this->post_type}_exists", sprintf( __( 'Cannot create existing %s.', 'woocommerce' ), $this->post_type ), array( 'status' => 400 ) ); 
  156.       } 
  157.  
  158.       // Validate topic. 
  159.       if ( empty( $request['topic'] ) || ! wc_is_webhook_valid_topic( strtolower( $request['topic'] ) ) ) { 
  160.           return new WP_Error( "woocommerce_rest_{$this->post_type}_invalid_topic", __( 'Webhook topic is required and must be valid.', 'woocommerce' ), array( 'status' => 400 ) ); 
  161.       } 
  162.  
  163.       // Validate delivery URL. 
  164.       if ( empty( $request['delivery_url'] ) || ! wc_is_valid_url( $request['delivery_url'] ) ) { 
  165.           return new WP_Error( "woocommerce_rest_{$this->post_type}_invalid_delivery_url", __( 'Webhook delivery URL must be a valid URL starting with http:// or https://.', 'woocommerce' ), array( 'status' => 400 ) ); 
  166.       } 
  167.  
  168.       $post = $this->prepare_item_for_database( $request ); 
  169.       if ( is_wp_error( $post ) ) { 
  170.           return $post; 
  171.       } 
  172.  
  173.       $post->post_type = $this->post_type; 
  174.       $post_id = wp_insert_post( $post, true ); 
  175.  
  176.       if ( is_wp_error( $post_id ) ) { 
  177.  
  178.           if ( in_array( $post_id->get_error_code(), array( 'db_insert_error' ) ) ) { 
  179.               $post_id->add_data( array( 'status' => 500 ) ); 
  180.           } else { 
  181.               $post_id->add_data( array( 'status' => 400 ) ); 
  182.           } 
  183.           return $post_id; 
  184.       } 
  185.       $post->ID = $post_id; 
  186.  
  187.       $webhook = new WC_Webhook( $post_id ); 
  188.  
  189.       // Set topic. 
  190.       $webhook->set_topic( $request['topic'] ); 
  191.  
  192.       // Set delivery URL. 
  193.       $webhook->set_delivery_url( $request['delivery_url'] ); 
  194.  
  195.       // Set secret. 
  196.       $webhook->set_secret( ! empty( $request['secret'] ) ? $request['secret'] : '' ); 
  197.  
  198.       // Set API version to WP API integration. 
  199.       $webhook->set_api_version( $this->get_default_api_version() ); 
  200.  
  201.       // Set status. 
  202.       if ( ! empty( $request['status'] ) ) { 
  203.           $webhook->update_status( $request['status'] ); 
  204.       } 
  205.  
  206.       $post = get_post( $post_id ); 
  207.       $this->update_additional_fields_for_object( $post, $request ); 
  208.  
  209.       /** 
  210.        * Fires after a single item is created or updated via the REST API. 
  211.        * 
  212.        * @param WP_Post         $post      Inserted object. 
  213.        * @param WP_REST_Request $request   Request object. 
  214.        * @param boolean         $creating  True when creating item, false when updating. 
  215.        */ 
  216.       do_action( "woocommerce_rest_insert_{$this->post_type}", $post, $request, true ); 
  217.  
  218.       $request->set_param( 'context', 'edit' ); 
  219.       $response = $this->prepare_item_for_response( $post, $request ); 
  220.       $response = rest_ensure_response( $response ); 
  221.       $response->set_status( 201 ); 
  222.       $response->header( 'Location', rest_url( sprintf( '/%s/%s/%d', $this->namespace, $this->rest_base, $post_id ) ) );

  226.  
  227.       // Clear cache. 
  228.       delete_transient( 'woocommerce_webhook_ids' ); 
  229.  
  230.       return $response;

  240.       $id = (int) $request['id'];

  245.       } 
  246.  
  247.       $webhook = new WC_Webhook( $id ); 
  248.  
  249.       // Update topic. 
  250.       if ( ! empty( $request['topic'] ) ) { 
  251.           if ( wc_is_webhook_valid_topic( strtolower( $request['topic'] ) ) ) { 
  252.               $webhook->set_topic( $request['topic'] ); 
  253.           } else { 
  254.               return new WP_Error( "woocommerce_rest_{$this->post_type}_invalid_topic", __( 'Webhook topic must be valid.', 'woocommerce' ), array( 'status' => 400 ) ); 
  255.           } 
  256.       } 
  257.  
  258.       // Update delivery URL. 
  259.       if ( ! empty( $request['delivery_url'] ) ) { 
  260.           if ( wc_is_valid_url( $request['delivery_url'] ) ) { 
  261.               $webhook->set_delivery_url( $request['delivery_url'] ); 
  262.           } else { 
  263.               return new WP_Error( "woocommerce_rest_{$this->post_type}_invalid_delivery_url", __( 'Webhook delivery URL must be a valid URL starting with http:// or https://.', 'woocommerce' ), array( 'status' => 400 ) ); 
  264.           } 
  265.       } 
  266.  
  267.       // Update secret. 
  268.       if ( ! empty( $request['secret'] ) ) { 
  269.           $webhook->set_secret( $request['secret'] ); 
  270.       } 
  271.  
  272.       // Update status. 
  273.       if ( ! empty( $request['status'] ) ) { 
  274.           $webhook->update_status( $request['status'] ); 
  275.       } 
  276.  
  277.       $post = $this->prepare_item_for_database( $request ); 
  278.       if ( is_wp_error( $post ) ) { 
  279.           return $post;

  283.       $post_id = wp_update_post( (array) $post, true );

  285.           if ( in_array( $post_id->get_error_code(), array( 'db_update_error' ) ) ) { 
  286.               $post_id->add_data( array( 'status' => 500 ) ); 
  287.           } else { 
  288.               $post_id->add_data( array( 'status' => 400 ) );

  300.        * @param WP_REST_Request $request   Request object. 
  301.        * @param boolean         $creating  True when creating item, false when updating. 
  302.        */ 
  303.       do_action( "woocommerce_rest_insert_{$this->post_type}", $post, $request, false ); 
  304.  
  305.       $request->set_param( 'context', 'edit' ); 
  306.       $response = $this->prepare_item_for_response( $post, $request ); 
  307.  
  308.       // Clear cache. 
  309.       delete_transient( 'woocommerce_webhook_ids' ); 
  310.  
  311.       return rest_ensure_response( $response ); 
  312.   }

  314.   /** 
  315.    * Delete a single webhook. 
  316.    * 
  317.    * @param WP_REST_Request $request Full details about the request. 
  318.    * @return WP_REST_Response|WP_Error 
  319.    */ 
  320.   public function delete_item( $request ) { 
  321.       $id = (int) $request['id']; 
  322.       $force = isset( $request['force'] ) ? (bool) $request['force'] : false; 
  323.  
  324.       // We don't support trashing for this type, error out. 
  325.       if ( ! $force ) { 
  326.           return new WP_Error( 'woocommerce_rest_trash_not_supported', __( 'Webhooks do not support trashing.', 'woocommerce' ), array( 'status' => 501 ) ); 
  327.       } 
  328.  
  329.       $post = get_post( $id ); 
  330.  
  331.       if ( empty( $id ) || empty( $post->ID ) || $this->post_type !== $post->post_type ) { 
  332.           return new WP_Error( "woocommerce_rest_{$this->post_type}_invalid_id", __( 'Invalid post ID.', 'woocommerce' ), array( 'status' => 404 ) ); 
  333.       } 
  334.  
  335.       $request->set_param( 'context', 'edit' ); 
  336.       $response = $this->prepare_item_for_response( $post, $request ); 
  337.  
  338.       $result = wp_delete_post( $id, true ); 
  339.  
  340.       if ( ! $result ) { 
  341.           /** translators: %s: post type */ 
  342.           return new WP_Error( 'woocommerce_rest_cannot_delete', sprintf( __( 'The %s cannot be deleted.', 'woocommerce' ), $this->post_type ), array( 'status' => 500 ) ); 
  343.       } 
  344.  
  345.       /** 
  346.        * Fires after a single item is deleted or trashed via the REST API. 
  347.        * 
  348.        * @param object           $post     The deleted or trashed item. 
  349.        * @param WP_REST_Response $response The response data. 
  350.        * @param WP_REST_Request  $request  The request sent to the API. 
  351.        */ 
  352.       do_action( "woocommerce_rest_delete_{$this->post_type}", $post, $response, $request ); 
  353.  
  354.       // Clear cache. 
  355.       delete_transient( 'woocommerce_webhook_ids' ); 
  356.  
  357.       return $response; 
  358.   }

  366.   protected function prepare_item_for_database( $request ) { 
  367.       global $wpdb; 
  368.  
  369.       $data = new stdClass; 
  370.  
  371.       // Post ID. 
  372.       if ( isset( $request['id'] ) ) { 
  373.           $data->ID = absint( $request['id'] ); 
  374.       } 
  375.  
  376.       // Validate required POST fields. 
  377.       if ( 'POST' === $request->get_method() && empty( $data->ID ) ) { 
  378.           // @codingStandardsIgnoreStart 
  379.           $data->post_title = ! empty( $request['name'] ) ? $request['name'] : sprintf( __( 'Webhook created on %s', 'woocommerce' ), strftime( _x( '%b %d, %Y @ %I:%M %p', 'Webhook created on date parsed by strftime', 'woocommerce' ) ) ); 
  380.           // @codingStandardsIgnoreEnd 
  381.  
  382.           // Post author. 
  383.           $data->post_author = get_current_user_id(); 
  384.  
  385.           // Post password. 
  386.           $password = strlen( uniqid( 'webhook_' ) ); 
  387.           $data->post_password = $password > 20 ? substr( $password, 0, 20 ) : $password; 
  388.  
  389.           // Post status. 
  390.           $data->post_status = 'publish'; 
  391.       } else { 
  392.  
  393.           // Allow edit post title. 
  394.           if ( ! empty( $request['name'] ) ) { 
  395.               $data->post_title = $request['name']; 
  396.           } 
  397.       } 
  398.  
  399.       // Comment status. 
  400.       $data->comment_status = 'closed'; 
  401.  
  402.       // Ping status. 
  403.       $data->ping_status = 'closed'; 
  404.  
  405.       /** 
  406.        * Filter the query_vars used in `get_items` for the constructed query. 
  407.        * 
  408.        * The dynamic portion of the hook name, $this->post_type, refers to post_type of the post being 
  409.        * prepared for insertion. 
  410.        * 
  411.        * @param stdClass        $data An object representing a single item prepared 
  412.        *                                       for inserting or updating the database. 
  413.        * @param WP_REST_Request $request       Request object. 
  414.        */ 
  415.       return apply_filters( "woocommerce_rest_pre_insert_{$this->post_type}", $data, $request ); 
  416.   }

  426.       $webhook = new WC_Webhook( $id ); 
  427.       $data = array( 
  428.           'id' => $webhook->id,  
  429.           'name' => $webhook->get_name(),  
  430.           'status' => $webhook->get_status(),  
  431.           'topic' => $webhook->get_topic(),  
  432.           'resource' => $webhook->get_resource(),  
  433.           'event' => $webhook->get_event(),  
  434.           'hooks' => $webhook->get_hooks(),  
  435.           'delivery_url' => $webhook->get_delivery_url(),  
  436.           'date_created' => wc_rest_prepare_date_response( $webhook->get_post_data()->post_date_gmt ),  
  437.           'date_modified' => wc_rest_prepare_date_response( $webhook->get_post_data()->post_modified_gmt ),  
  438.  ); 
  439.  
  440.       $context = ! empty( $request['context'] ) ? $request['context'] : 'view'; 
  441.       $data = $this->add_additional_fields_to_object( $data, $request ); 
  442.       $data = $this->filter_response_by_context( $data, $context ); 
  443.  
  444.       // Wrap the data in a response object. 
  445.       $response = rest_ensure_response( $data ); 
  446.  
  447.       $response->add_links( $this->prepare_links( $post, $request ) ); 
  448.  
  449.       /** 
  450.        * Filter webhook object returned from the REST API. 
  451.        * 
  452.        * @param WP_REST_Response $response The response object. 
  453.        * @param WC_Webhook       $webhook  Webhook object used to create response. 
  454.        * @param WP_REST_Request  $request  Request object. 
  455.        */ 
  456.       return apply_filters( "woocommerce_rest_prepare_{$this->post_type}", $response, $webhook, $request ); 
  457.   }
