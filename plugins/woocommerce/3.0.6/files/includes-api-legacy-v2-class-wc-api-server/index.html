  263.    * used as the delimiter with preg_match()

  283.       }

  288.   /** 
  289.    * Match the request to a callback and call it 
  290.    * 
  291.    * @since 2.1 
  292.    * @return mixed The value returned by the callback, or a WP_Error instance 
  293.    */ 
  294.   public function dispatch() { 
  295.  
  296.       switch ( $this->method ) { 
  297.  
  298.           case 'HEAD' : 
  299.           case 'GET' : 
  300.               $method = self::METHOD_GET; 
  301.               break; 
  302.  
  303.           case 'POST' : 
  304.               $method = self::METHOD_POST; 
  305.               break; 
  306.  
  307.           case 'PUT' : 
  308.               $method = self::METHOD_PUT; 
  309.               break; 
  310.  
  311.           case 'PATCH' : 
  312.               $method = self::METHOD_PATCH; 
  313.               break; 
  314.  
  315.           case 'DELETE' : 
  316.               $method = self::METHOD_DELETE; 
  317.               break; 
  318.  
  319.           default : 
  320.               return new WP_Error( 'woocommerce_api_unsupported_method', __( 'Unsupported request method', 'woocommerce' ), array( 'status' => 400 ) ); 
  321.       } 
  322.  
  323.       foreach ( $this->get_routes() as $route => $handlers ) { 
  324.           foreach ( $handlers as $handler ) { 
  325.               $callback = $handler[0]; 
  326.               $supported = isset( $handler[1] ) ? $handler[1] : self::METHOD_GET; 
  327.  
  328.               if ( ! ( $supported & $method ) ) { 
  329.                   continue; 
  330.               } 
  331.  
  332.               $match = preg_match( '@^' . $route . '$@i', urldecode( $this->path ), $args ); 
  333.  
  334.               if ( ! $match ) { 
  335.                   continue; 
  336.               } 
  337.  
  338.               if ( ! is_callable( $callback ) ) { 
  339.                   return new WP_Error( 'woocommerce_api_invalid_handler', __( 'The handler for the route is invalid', 'woocommerce' ), array( 'status' => 500 ) ); 
  340.               } 
  341.  
  342.               $args = array_merge( $args, $this->params['GET'] ); 
  343.               if ( $method & self::METHOD_POST ) { 
  344.                   $args = array_merge( $args, $this->params['POST'] ); 
  345.               } 
  346.               if ( $supported & self::ACCEPT_DATA ) { 
  347.                   $data = $this->handler->parse_body( $this->get_raw_data() ); 
  348.                   $args = array_merge( $args, array( 'data' => $data ) ); 
  349.               } elseif ( $supported & self::ACCEPT_RAW_DATA ) { 
  350.                   $data = $this->get_raw_data(); 
  351.                   $args = array_merge( $args, array( 'data' => $data ) ); 
  352.               } 
  353.  
  354.               $args['_method'] = $method; 
  355.               $args['_route'] = $route; 
  356.               $args['_path'] = $this->path; 
  357.               $args['_headers'] = $this->headers; 
  358.               $args['_files'] = $this->files; 
  359.  
  360.               $args = apply_filters( 'woocommerce_api_dispatch_args', $args, $callback ); 
  361.  
  362.               // Allow plugins to halt the request via this filter 
  363.               if ( is_wp_error( $args ) ) { 
  364.                   return $args; 
  365.               } 
  366.  
  367.               $params = $this->sort_callback_params( $callback, $args ); 
  368.               if ( is_wp_error( $params ) ) { 
  369.                   return $params; 
  370.               } 
  371.  
  372.               return call_user_func_array( $callback, $params ); 
  373.           } 
  374.       } 
  375.  
  376.       return new WP_Error( 'woocommerce_api_no_route', __( 'No route was found matching the URL and request method', 'woocommerce' ), array( 'status' => 404 ) ); 
  377.   }

  394.   /** 
  395.    * Sort parameters by order specified in method declaration 
  396.    * 
  397.    * Takes a callback and a list of available params, then filters and sorts 
  398.    * by the parameters the method actually needs, using the Reflection API 
  399.    * 
  400.    * @since 2.2 
  401.    * @param callable|array $callback the endpoint callback 
  402.    * @param array $provided the provided request parameters 
  403.    * @return array 
  404.    */ 
  405.   protected function sort_callback_params( $callback, $provided ) { 
  406.       if ( is_array( $callback ) ) { 
  407.           $ref_func = new ReflectionMethod( $callback[0], $callback[1] ); 
  408.       } else { 
  409.           $ref_func = new ReflectionFunction( $callback ); 
  410.       } 
  411.  
  412.       $wanted = $ref_func->getParameters(); 
  413.       $ordered_parameters = array(); 
  414.  
  415.       foreach ( $wanted as $param ) { 
  416.           if ( isset( $provided[ $param->getName() ] ) ) { 
  417.               // We have this parameters in the list to choose from 
  418.               if ( 'data' == $param->getName() ) { 
  419.                   $ordered_parameters[] = $provided[ $param->getName() ]; 
  420.                   continue; 
  421.               } 
  422.  
  423.               $ordered_parameters[] = $this->urldecode_deep( $provided[ $param->getName() ] ); 
  424.           } elseif ( $param->isDefaultValueAvailable() ) { 
  425.               // We don't have this parameter, but it's optional 
  426.               $ordered_parameters[] = $param->getDefaultValue(); 
  427.           } else { 
  428.               // We don't have this parameter and it wasn't optional, abort! 
  429.               return new WP_Error( 'woocommerce_api_missing_callback_param', sprintf( __( 'Missing parameter %s', 'woocommerce' ), $param->getName() ), array( 'status' => 400 ) ); 
  430.           } 
  431.       } 
  432.  
  433.       return $ordered_parameters; 
  434.   }

  563.   /**

  568.    */ 
  569.   public function add_pagination_headers( $query ) { 
  570.  
  571.       // WP_User_Query 
  572.       if ( is_a( $query, 'WP_User_Query' ) ) { 
  573.  
  574.           $single = count( $query->get_results() ) == 1; 
  575.           $total = $query->get_total(); 
  576.  
  577.           if ( $query->get( 'number' ) > 0 ) { 
  578.               $page = ( $query->get( 'offset' ) / $query->get( 'number' ) ) + 1; 
  579.               $total_pages = ceil( $total / $query->get( 'number' ) ); 
  580.           } else { 
  581.               $page = 1; 
  582.               $total_pages = 1; 
  583.           } 
  584.  
  585.       // WP_Query 
  586.       } else { 
  587.  
  588.           $page = $query->get( 'paged' ); 
  589.           $single = $query->is_single(); 
  590.           $total = $query->found_posts; 
  591.           $total_pages = $query->max_num_pages; 
  592.       } 
  593.  
  594.       if ( ! $page ) { 
  595.           $page = 1; 
  596.       } 
  597.  
  598.       $next_page = absint( $page ) + 1; 
  599.  
  600.       if ( ! $single ) { 
  601.  
  602.           // first/prev 
  603.           if ( $page > 1 ) { 
  604.               $this->link_header( 'first', $this->get_paginated_url( 1 ) ); 
  605.               $this->link_header( 'prev', $this->get_paginated_url( $page -1 ) ); 
  606.           } 
  607.  
  608.           // next 
  609.           if ( $next_page <= $total_pages ) { 
  610.               $this->link_header( 'next', $this->get_paginated_url( $next_page ) ); 
  611.           } 
  612.  
  613.           // last 
  614.           if ( $page != $total_pages ) { 
  615.               $this->link_header( 'last', $this->get_paginated_url( $total_pages ) ); 
  616.           } 
  617.       } 
  618.  
  619.       $this->header( 'X-WC-Total', $total ); 
  620.       $this->header( 'X-WC-TotalPages', $total_pages ); 
  621.  
  622.       do_action( 'woocommerce_api_pagination_headers', $this, $query ); 
  623.   }

  625.   /** 
  626.    * Returns the request URL with the page query parameter set to the specified page 
  627.    * 
  628.    * @since 2.1 
  629.    * @param int $page 
  630.    * @return string 
  631.    */ 
  632.   private function get_paginated_url( $page ) { 
  633.  
  634.       // remove existing page query param 
  635.       $request = remove_query_arg( 'page' ); 
  636.  
  637.       // add provided page query param 
  638.       $request = urldecode( add_query_arg( 'page', $page, $request ) ); 
  639.  
  640.       // get the home host 
  641.       $host = parse_url( get_home_url(), PHP_URL_HOST ); 
  642.  
  643.       return set_url_scheme( "http://{$host}{$request}" ); 
  644.   }

  650.    * @return string 
  651.    */ 
  652.   public function get_raw_data() { 
  653.       // $HTTP_RAW_POST_DATA is deprecated on PHP 5.6 
  654.       if ( function_exists( 'phpversion' ) && version_compare( phpversion(), '5.6', '>=' ) ) { 
  655.           return file_get_contents( 'php://input' ); 
  656.       } 
  657.  
  658.       global $HTTP_RAW_POST_DATA; 
  659.  
  660.       // A bug in PHP < 5.2.2 makes $HTTP_RAW_POST_DATA not set by default,  
  661.       // but we can do it ourself. 
  662.       if ( ! isset( $HTTP_RAW_POST_DATA ) ) { 
  663.           $HTTP_RAW_POST_DATA = file_get_contents( 'php://input' ); 
  664.       }

  669.   /** 
  670.    * Parse an RFC3339 datetime into a MySQl datetime 
  671.    * 
  672.    * Invalid dates default to unix epoch 
  673.    * 
  674.    * @since 2.1 
  675.    * @param string $datetime RFC3339 datetime 
  676.    * @return string MySQl datetime (YYYY-MM-DD HH:MM:SS) 
  677.    */ 
  678.   public function parse_datetime( $datetime ) { 
  679.  
  680.       // Strip millisecond precision (a full stop followed by one or more digits) 
  681.       if ( strpos( $datetime, '.' ) !== false ) { 
  682.           $datetime = preg_replace( '/\.\d+/', '', $datetime ); 
  683.       } 
  684.  
  685.       // default timezone to UTC 
  686.       $datetime = preg_replace( '/[+-]\d+:+\d+$/', '+00:00', $datetime ); 
  687.  
  688.       try { 
  689.  
  690.           $datetime = new DateTime( $datetime, new DateTimeZone( 'UTC' ) ); 
  691.  
  692.       } catch ( Exception $e ) { 
  693.  
  694.           $datetime = new DateTime( '@0' ); 
  695.  
  696.       } 
  697.  
  698.       return $datetime->format( 'Y-m-d H:i:s' ); 
  699.   }

  701.   /** 
  702.    * Format a unix timestamp or MySQL datetime into an RFC3339 datetime 
  703.    * 
  704.    * @since 2.1 
  705.    * @param int|string $timestamp unix timestamp or MySQL datetime 
  706.    * @param bool $convert_to_utc 
  707.    * @param bool $convert_to_gmt Use GMT timezone. 
  708.    * @return string RFC3339 datetime 
  709.    */ 
  710.   public function format_datetime( $timestamp, $convert_to_utc = false, $convert_to_gmt = false ) { 
  711.       if ( $convert_to_gmt ) { 
  712.           if ( is_numeric( $timestamp ) ) { 
  713.               $timestamp = date( 'Y-m-d H:i:s', $timestamp ); 
  714.           } 
  715.  
  716.           $timestamp = get_gmt_from_date( $timestamp ); 
  717.       } 
  718.  
  719.       if ( $convert_to_utc ) { 
  720.           $timezone = new DateTimeZone( wc_timezone_string() ); 
  721.       } else { 
  722.           $timezone = new DateTimeZone( 'UTC' ); 
  723.       } 
  724.  
  725.       try { 
  726.  
  727.           if ( is_numeric( $timestamp ) ) { 
  728.               $date = new DateTime( "@{$timestamp}" ); 
  729.           } else { 
  730.               $date = new DateTime( $timestamp, $timezone ); 
  731.           } 
  732.  
  733.           // convert to UTC by adjusting the time based on the offset of the site's timezone 
  734.           if ( $convert_to_utc ) { 
  735.               $date->modify( -1 * $date->getOffset() . ' seconds' ); 
  736.           } 
  737.       } catch ( Exception $e ) { 
  738.  
  739.           $date = new DateTime( '@0' ); 
  740.       } 
  741.  
  742.       return $date->format( 'Y-m-d\TH:i:s\Z' ); 
  743.   }

  745.   /** 
  746.    * Extract headers from a PHP-style $_SERVER array 
  747.    * 
  748.    * @since 2.1 
  749.    * @param array $server Associative array similar to $_SERVER 
  750.    * @return array Headers extracted from the input 
  751.    */ 
  752.   public function get_headers( $server ) { 
  753.       $headers = array(); 
  754.       // CONTENT_* headers are not prefixed with HTTP_ 
  755.       $additional = array( 'CONTENT_LENGTH' => true, 'CONTENT_MD5' => true, 'CONTENT_TYPE' => true ); 
  756.  
  757.       foreach ( $server as $key => $value ) { 
  758.           if ( strpos( $key, 'HTTP_' ) === 0 ) { 
  759.               $headers[ substr( $key, 5 ) ] = $value; 
  760.           } elseif ( isset( $additional[ $key ] ) ) { 
  761.               $headers[ $key ] = $value; 
  762.           } 
  763.       } 
  764.  
  765.       return $headers; 
  766.   }
