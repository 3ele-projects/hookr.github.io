  83.   /** 
  84.    * Basic Authentication. 
  85.    * 
  86.    * SSL-encrypted requests are not subject to sniffing or man-in-the-middle 
  87.    * attacks, so the request can be authenticated by simply looking up the user 
  88.    * associated with the given consumer key and confirming the consumer secret 
  89.    * provided is valid. 
  90.    * 
  91.    * @return int|bool 
  92.    */ 
  93.   private function perform_basic_authentication() { 
  94.       global $wc_rest_authentication_error; 
  95.  
  96.       $consumer_key = ''; 
  97.       $consumer_secret = ''; 
  98.  
  99.       // If the $_GET parameters are present, use those first. 
  100.       if ( ! empty( $_GET['consumer_key'] ) && ! empty( $_GET['consumer_secret'] ) ) { 
  101.           $consumer_key = $_GET['consumer_key']; 
  102.           $consumer_secret = $_GET['consumer_secret']; 
  103.       } 
  104.  
  105.       // If the above is not present, we will do full basic auth. 
  106.       if ( ! $consumer_key && ! empty( $_SERVER['PHP_AUTH_USER'] ) && ! empty( $_SERVER['PHP_AUTH_PW'] ) ) { 
  107.           $consumer_key = $_SERVER['PHP_AUTH_USER']; 
  108.           $consumer_secret = $_SERVER['PHP_AUTH_PW']; 
  109.       } 
  110.  
  111.       // Stop if don't have any key. 
  112.       if ( ! $consumer_key || ! $consumer_secret ) { 
  113.           return false; 
  114.       } 
  115.  
  116.       // Get user data. 
  117.       $user = $this->get_user_data_by_consumer_key( $consumer_key ); 
  118.       if ( empty( $user ) ) { 
  119.           return false; 
  120.       } 
  121.  
  122.       // Validate user secret. 
  123.       if ( ! hash_equals( $user->consumer_secret, $consumer_secret ) ) { 
  124.           $wc_rest_authentication_error = new WP_Error( 'woocommerce_rest_authentication_error', __( 'Consumer secret is invalid.', 'woocommerce' ), array( 'status' => 401 ) ); 
  125.  
  126.           return false; 
  127.       } 
  128.  
  129.       // Check API Key permissions. 
  130.       if ( ! $this->check_permissions( $user->permissions ) ) { 
  131.           return false; 
  132.       } 
  133.  
  134.       // Update last access. 
  135.       $this->update_last_access( $user->key_id ); 
  136.  
  137.       return $user->user_id; 
  138.   }

  140.   /** 
  141.    * Parse the Authorization header into parameters. 
  142.    * 
  143.    * @since 3.0.0 
  144.    * 
  145.    * @param string $header Authorization header value (not including "Authorization: " prefix). 
  146.    * 
  147.    * @return array Map of parameter values. 
  148.    */ 
  149.   public function parse_header( $header ) { 
  150.       if ( 'OAuth ' !== substr( $header, 0, 6 ) ) { 
  151.           return array(); 
  152.       } 
  153.  
  154.       // From OAuth PHP library, used under MIT license. 
  155.       $params = array(); 
  156.       if ( preg_match_all( '/(oauth_[a-z_-]*)=(:?"([^"]*)"|([^, ]*))/', $header, $matches ) ) { 
  157.           foreach ( $matches[1] as $i => $h ) { 
  158.               $params[ $h ] = urldecode( empty( $matches[3][ $i ] ) ? $matches[4][ $i ] : $matches[3][ $i ] ); 
  159.           } 
  160.           if ( isset( $params['realm'] ) ) { 
  161.               unset( $params['realm'] ); 
  162.           } 
  163.       } 
  164.  
  165.       return $params; 
  166.   }

  168.   /** 
  169.    * Get the authorization header. 
  170.    * 
  171.    * On certain systems and configurations, the Authorization header will be 
  172.    * stripped out by the server or PHP. Typically this is then used to 
  173.    * generate `PHP_AUTH_USER`/`PHP_AUTH_PASS` but not passed on. We use 
  174.    * `getallheaders` here to try and grab it out instead. 
  175.    * 
  176.    * @since 3.0.0 
  177.    * 
  178.    * @return string Authorization header if set. 
  179.    */ 
  180.   public function get_authorization_header() { 
  181.       if ( ! empty( $_SERVER['HTTP_AUTHORIZATION'] ) ) { 
  182.           return wp_unslash( $_SERVER['HTTP_AUTHORIZATION'] ); 
  183.       } 
  184.  
  185.       if ( function_exists( 'getallheaders' ) ) { 
  186.           $headers = getallheaders(); 
  187.           // Check for the authoization header case-insensitively. 
  188.           foreach ( $headers as $key => $value ) { 
  189.               if ( 'authorization' === strtolower( $key ) ) { 
  190.                   return $value; 
  191.               } 
  192.           } 
  193.       } 
  194.  
  195.       return ''; 
  196.   }

  198.   /** 
  199.    * Get oAuth parameters from $_GET, $_POST or request header. 
  200.    * 
  201.    * @since 3.0.0 
  202.    * 
  203.    * @return array|WP_Error 
  204.    */ 
  205.   public function get_oauth_parameters() { 
  206.       global $wc_rest_authentication_error; 
  207.  
  208.       $params = array_merge( $_GET, $_POST ); 
  209.       $params = wp_unslash( $params ); 
  210.       $header = $this->get_authorization_header(); 
  211.  
  212.       if ( ! empty( $header ) ) { 
  213.           // Trim leading spaces. 
  214.           $header = trim( $header ); 
  215.           $header_params = $this->parse_header( $header ); 
  216.  
  217.           if ( ! empty( $header_params ) ) { 
  218.               $params = array_merge( $params, $header_params ); 
  219.           } 
  220.       } 
  221.  
  222.       $param_names = array( 
  223.           'oauth_consumer_key',  
  224.           'oauth_timestamp',  
  225.           'oauth_nonce',  
  226.           'oauth_signature',  
  227.           'oauth_signature_method',  
  228.  ); 
  229.  
  230.       $errors = array(); 
  231.       $have_one = false; 
  232.  
  233.       // Check for required OAuth parameters. 
  234.       foreach ( $param_names as $param_name ) { 
  235.           if ( empty( $params[ $param_name ] ) ) { 
  236.               $errors[] = $param_name; 
  237.           } else { 
  238.               $have_one = true; 
  239.           } 
  240.       } 
  241.  
  242.       // All keys are missing, so we're probably not even trying to use OAuth. 
  243.       if ( ! $have_one ) { 
  244.           return array(); 
  245.       } 
  246.  
  247.       // If we have at least one supplied piece of data, and we have an error,  
  248.       // then it's a failed authentication. 
  249.       if ( ! empty( $errors ) ) { 
  250.           $message = sprintf( 
  251.               _n( 'Missing OAuth parameter %s', 'Missing OAuth parameters %s', count( $errors ), 'woocommerce' ),  
  252.               implode( ', ', $errors ) 
  253.  ); 
  254.  
  255.           $wc_rest_authentication_error = new WP_Error( 'woocommerce_rest_authentication_missing_parameter', $message, array( 'status' => 401 ) ); 
  256.  
  257.           return array(); 
  258.       } 
  259.  
  260.       return $params; 
  261.   }

  263.   /** 
  264.    * Perform OAuth 1.0a "one-legged" (http://oauthbible.com/#oauth-10a-one-legged) authentication for non-SSL requests. 
  265.    * 
  266.    * This is required so API credentials cannot be sniffed or intercepted when making API requests over plain HTTP. 
  267.    * 
  268.    * This follows the spec for simple OAuth 1.0a authentication (RFC 5849) as closely as possible, with two exceptions: 
  269.    * 
  270.    * 1) There is no token associated with request/responses, only consumer keys/secrets are used. 
  271.    * 
  272.    * 2) The OAuth parameters are included as part of the request query string instead of part of the Authorization header,  
  273.    *    This is because there is no cross-OS function within PHP to get the raw Authorization header. 
  274.    * 
  275.    * @link http://tools.ietf.org/html/rfc5849 for the full spec. 
  276.    * 
  277.    * @return int|bool 
  278.    */ 
  279.   private function perform_oauth_authentication() { 
  280.       global $wc_rest_authentication_error; 
  281.  
  282.       $params = $this->get_oauth_parameters(); 
  283.       if ( empty( $params ) ) { 
  284.           return false; 
  285.       } 
  286.  
  287.       // Fetch WP user by consumer key. 
  288.       $user = $this->get_user_data_by_consumer_key( $params['oauth_consumer_key'] ); 
  289.  
  290.       if ( empty( $user ) ) { 
  291.           $wc_rest_authentication_error = new WP_Error( 'woocommerce_rest_authentication_error', __( 'Consumer key is invalid.', 'woocommerce' ), array( 'status' => 401 ) ); 
  292.  
  293.           return false; 
  294.       } 
  295.  
  296.       // Perform OAuth validation. 
  297.       $wc_rest_authentication_error = $this->check_oauth_signature( $user, $params ); 
  298.       if ( is_wp_error( $wc_rest_authentication_error ) ) { 
  299.           return false; 
  300.       } 
  301.  
  302.       $wc_rest_authentication_error = $this->check_oauth_timestamp_and_nonce( $user, $params['oauth_timestamp'], $params['oauth_nonce'] ); 
  303.       if ( is_wp_error( $wc_rest_authentication_error ) ) { 
  304.           return false; 
  305.       } 
  306.  
  307.       // Check API Key permissions. 
  308.       if ( ! $this->check_permissions( $user->permissions ) ) { 
  309.           return false; 
  310.       } 
  311.  
  312.       // Update last access. 
  313.       $this->update_last_access( $user->key_id ); 
  314.  
  315.       return $user->user_id; 
  316.   }

  318.   /** 
  319.    * Verify that the consumer-provided request signature matches our generated signature,  
  320.    * this ensures the consumer has a valid key/secret. 
  321.    * 
  322.    * @param stdClass $user 
  323.    * @param array $params The request parameters. 
  324.    * @return null|WP_Error 
  325.    */ 
  326.   private function check_oauth_signature( $user, $params ) { 
  327.       $http_method = strtoupper( $_SERVER['REQUEST_METHOD'] ); 
  328.       $request_path = parse_url( $_SERVER['REQUEST_URI'], PHP_URL_PATH ); 
  329.       $wp_base = get_home_url( null, '/', 'relative' ); 
  330.       if ( substr( $request_path, 0, strlen( $wp_base ) ) === $wp_base ) { 
  331.           $request_path = substr( $request_path, strlen( $wp_base ) ); 
  332.       } 
  333.       $base_request_uri = rawurlencode( get_home_url( null, $request_path ) ); 
  334.  
  335.       // Get the signature provided by the consumer and remove it from the parameters prior to checking the signature. 
  336.       $consumer_signature = rawurldecode( str_replace( ' ', '+', $params['oauth_signature'] ) ); 
  337.       unset( $params['oauth_signature'] ); 
  338.  
  339.       // Sort parameters. 
  340.       if ( ! uksort( $params, 'strcmp' ) ) { 
  341.           return new WP_Error( 'woocommerce_rest_authentication_error', __( 'Invalid signature - failed to sort parameters.', 'woocommerce' ), array( 'status' => 401 ) ); 
  342.       } 
  343.  
  344.       // Normalize parameter key/values. 
  345.       $params = $this->normalize_parameters( $params ); 
  346.       $query_parameters = array(); 
  347.       foreach ( $params as $param_key => $param_value ) { 
  348.           if ( is_array( $param_value ) ) { 
  349.               foreach ( $param_value as $param_key_inner => $param_value_inner ) { 
  350.                   $query_parameters[] = $param_key . '%255B' . $param_key_inner . '%255D%3D' . $param_value_inner; 
  351.               } 
  352.           } else { 
  353.               $query_parameters[] = $param_key . '%3D' . $param_value; // Join with equals sign. 
  354.           } 
  355.       } 
  356.       $query_string = implode( '%26', $query_parameters ); // Join with ampersand. 
  357.       $string_to_sign = $http_method . '&' . $base_request_uri . '&' . $query_string; 
  358.  
  359.       if ( 'HMAC-SHA1' !== $params['oauth_signature_method'] && 'HMAC-SHA256' !== $params['oauth_signature_method'] ) { 
  360.           return new WP_Error( 'woocommerce_rest_authentication_error', __( 'Invalid signature - signature method is invalid.', 'woocommerce' ), array( 'status' => 401 ) ); 
  361.       } 
  362.  
  363.       $hash_algorithm = strtolower( str_replace( 'HMAC-', '', $params['oauth_signature_method'] ) ); 
  364.       $secret = $user->consumer_secret . '&'; 
  365.       $signature = base64_encode( hash_hmac( $hash_algorithm, $string_to_sign, $secret, true ) ); 
  366.  
  367.       if ( ! hash_equals( $signature, $consumer_signature ) ) { 
  368.           return new WP_Error( 'woocommerce_rest_authentication_error', __( 'Invalid signature - provided signature does not match.', 'woocommerce' ), array( 'status' => 401 ) ); 
  369.       } 
  370.  
  371.       return true; 
  372.   }

  374.   /** 
  375.    * Normalize each parameter by assuming each parameter may have already been 
  376.    * encoded, so attempt to decode, and then re-encode according to RFC 3986. 
  377.    * 
  378.    * Note both the key and value is normalized so a filter param like: 
  379.    * 
  380.    * 'filter[period]' => 'week' 
  381.    * 
  382.    * is encoded to: 
  383.    * 
  384.    * 'filter%5Bperiod%5D' => 'week' 
  385.    * 
  386.    * This conforms to the OAuth 1.0a spec which indicates the entire query string 
  387.    * should be URL encoded. 
  388.    * 
  389.    * @see rawurlencode() 
  390.    * @param array $parameters Un-normalized pararmeters. 
  391.    * @return array Normalized parameters. 
  392.    */ 
  393.   private function normalize_parameters( $parameters ) { 
  394.       $keys = wc_rest_urlencode_rfc3986( array_keys( $parameters ) ); 
  395.       $values = wc_rest_urlencode_rfc3986( array_values( $parameters ) ); 
  396.       $parameters = array_combine( $keys, $values ); 
  397.  
  398.       return $parameters; 
  399.   }

  401.   /** 
  402.    * Verify that the timestamp and nonce provided with the request are valid. This prevents replay attacks where 
  403.    * an attacker could attempt to re-send an intercepted request at a later time. 
  404.    * 
  405.    * - A timestamp is valid if it is within 15 minutes of now. 
  406.    * - A nonce is valid if it has not been used within the last 15 minutes. 
  407.    * 
  408.    * @param stdClass $user 
  409.    * @param int $timestamp the unix timestamp for when the request was made 
  410.    * @param string $nonce a unique (for the given user) 32 alphanumeric string, consumer-generated 
  411.    * @return bool|WP_Error 
  412.    */ 
  413.   private function check_oauth_timestamp_and_nonce( $user, $timestamp, $nonce ) { 
  414.       global $wpdb; 
  415.  
  416.       $valid_window = 15 * 60; // 15 minute window. 
  417.  
  418.       if ( ( $timestamp < time() - $valid_window ) || ( $timestamp > time() + $valid_window ) ) { 
  419.           return new WP_Error( 'woocommerce_rest_authentication_error', __( 'Invalid timestamp.', 'woocommerce' ), array( 'status' => 401 ) ); 
  420.       } 
  421.  
  422.       $used_nonces = maybe_unserialize( $user->nonces ); 
  423.  
  424.       if ( empty( $used_nonces ) ) { 
  425.           $used_nonces = array(); 
  426.       } 
  427.  
  428.       if ( in_array( $nonce, $used_nonces ) ) { 
  429.           return new WP_Error( 'woocommerce_rest_authentication_error', __( 'Invalid nonce - nonce has already been used.', 'woocommerce' ), array( 'status' => 401 ) ); 
  430.       } 
  431.  
  432.       $used_nonces[ $timestamp ] = $nonce; 
  433.  
  434.       // Remove expired nonces. 
  435.       foreach ( $used_nonces as $nonce_timestamp => $nonce ) { 
  436.           if ( $nonce_timestamp < ( time() - $valid_window ) ) { 
  437.               unset( $used_nonces[ $nonce_timestamp ] ); 
  438.           } 
  439.       } 
  440.  
  441.       $used_nonces = maybe_serialize( $used_nonces ); 
  442.  
  443.       $wpdb->update( 
  444.           $wpdb->prefix . 'woocommerce_api_keys',  
  445.           array( 'nonces' => $used_nonces ),  
  446.           array( 'key_id' => $user->key_id ),  
  447.           array( '%s' ),  
  448.           array( '%d' ) 
  449.  ); 
  450.  
  451.       return true; 
  452.   }

  454.   /** 
  455.    * Return the user data for the given consumer_key. 
  456.    * 
  457.    * @param string $consumer_key 
  458.    * @return array

  460.   private function get_user_data_by_consumer_key( $consumer_key ) { 
  461.       global $wpdb; 
  462.  
  463.       $consumer_key = wc_api_hash( sanitize_text_field( $consumer_key ) ); 
  464.       $user = $wpdb->get_row( $wpdb->prepare( " 
  465.           SELECT key_id, user_id, permissions, consumer_key, consumer_secret, nonces 
  466.           FROM {$wpdb->prefix}woocommerce_api_keys 
  467.           WHERE consumer_key = %s 
  468.       ", $consumer_key ) ); 
  469.  
  470.       return $user; 
  471.   }

  473.   /** 
  474.    * Check that the API keys provided have the proper key-specific permissions to either read or write API resources. 
  475.    * 
  476.    * @param string $permissions 
  477.    * @return bool 
  478.    */ 
  479.   private function check_permissions( $permissions ) { 
  480.       global $wc_rest_authentication_error; 
  481.  
  482.       $valid = true; 
  483.  
  484.       if ( ! isset( $_SERVER['REQUEST_METHOD'] ) ) { 
  485.           return false; 
  486.       } 
  487.  
  488.       switch ( $_SERVER['REQUEST_METHOD'] ) { 
  489.  
  490.           case 'HEAD' : 
  491.           case 'GET' : 
  492.               if ( 'read' !== $permissions && 'read_write' !== $permissions ) { 
  493.                   $wc_rest_authentication_error = new WP_Error( 'woocommerce_rest_authentication_error', __( 'The API key provided does not have read permissions.', 'woocommerce' ), array( 'status' => 401 ) ); 
  494.                   $valid = false; 
  495.               } 
  496.               break; 
  497.  
  498.           case 'POST' : 
  499.           case 'PUT' : 
  500.           case 'PATCH' : 
  501.           case 'DELETE' : 
  502.               if ( 'write' !== $permissions && 'read_write' !== $permissions ) { 
  503.                   $wc_rest_authentication_error = new WP_Error( 'woocommerce_rest_authentication_error', __( 'The API key provided does not have write permissions.', 'woocommerce' ), array( 'status' => 401 ) ); 
  504.                   $valid = false; 
  505.               } 
  506.               break; 
  507.       } 
  508.  
  509.       return $valid; 
  510.   }
