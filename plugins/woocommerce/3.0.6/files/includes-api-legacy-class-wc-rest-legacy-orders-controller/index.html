  41.   public function query_args( $args, $request ) { 
  42.       global $wpdb; 
  43.  
  44.       // Set post_status. 
  45.       if ( 'any' !== $request['status'] ) { 
  46.           $args['post_status'] = 'wc-' . $request['status']; 
  47.       } else { 
  48.           $args['post_status'] = 'any'; 
  49.       } 
  50.  
  51.       if ( ! empty( $request['customer'] ) ) { 
  52.           if ( ! empty( $args['meta_query'] ) ) { 
  53.               $args['meta_query'] = array(); 
  54.           } 
  55.  
  56.           $args['meta_query'][] = array( 
  57.               'key' => '_customer_user',  
  58.               'value' => $request['customer'],  
  59.               'type' => 'NUMERIC',  
  60.  ); 
  61.       } 
  62.  
  63.       // Search by product. 
  64.       if ( ! empty( $request['product'] ) ) { 
  65.           $order_ids = $wpdb->get_col( $wpdb->prepare( " 
  66.               SELECT order_id 
  67.               FROM {$wpdb->prefix}woocommerce_order_items 
  68.               WHERE order_item_id IN ( SELECT order_item_id FROM {$wpdb->prefix}woocommerce_order_itemmeta WHERE meta_key = '_product_id' AND meta_value = %d ) 
  69.               AND order_item_type = 'line_item' 
  70.            ", $request['product'] ) ); 
  71.  
  72.           // Force WP_Query return empty if don't found any order. 
  73.           $order_ids = ! empty( $order_ids ) ? $order_ids : array( 0 ); 
  74.  
  75.           $args['post__in'] = $order_ids; 
  76.       } 
  77.  
  78.       // Search. 
  79.       if ( ! empty( $args['s'] ) ) { 
  80.           $order_ids = wc_order_search( $args['s'] ); 
  81.  
  82.           if ( ! empty( $order_ids ) ) { 
  83.               unset( $args['s'] ); 
  84.               $args['post__in'] = array_merge( $order_ids, array( 0 ) ); 
  85.           } 
  86.       } 
  87.  
  88.       return $args; 
  89.   }

  91.   /** 
  92.    * Prepare a single order output for response. 
  93.    * 
  94.    * @deprecated 3.0 
  95.    * 
  96.    * @param WP_Post $post Post object. 
  97.    * @param WP_REST_Request $request Request object. 
  98.    * @return WP_REST_Response $data 
  99.    */ 
  100.   public function prepare_item_for_response( $post, $request ) { 
  101.       $this->request = $request; 
  102.       $statuses = wc_get_order_statuses(); 
  103.       $order = wc_get_order( $post ); 
  104.       $data = array_merge( array( 'id' => $order->get_id() ), $order->get_data() ); 
  105.       $format_decimal = array( 'discount_total', 'discount_tax', 'shipping_total', 'shipping_tax', 'shipping_total', 'shipping_tax', 'cart_tax', 'total', 'total_tax' ); 
  106.       $format_date = array( 'date_created', 'date_modified', 'date_completed', 'date_paid' ); 
  107.       $format_line_items = array( 'line_items', 'tax_lines', 'shipping_lines', 'fee_lines', 'coupon_lines' ); 
  108.  
  109.       // Format decimal values. 
  110.       foreach ( $format_decimal as $key ) { 
  111.           $data[ $key ] = wc_format_decimal( $data[ $key ], $this->request['dp'] ); 
  112.       } 
  113.  
  114.       // Format date values. 
  115.       foreach ( $format_date as $key ) { 
  116.           $data[ $key ] = $data[ $key ] ? wc_rest_prepare_date_response( get_gmt_from_date( date( 'Y-m-d H:i:s', $data[ $key ] ) ) ) : false; 
  117.       } 
  118.  
  119.       // Format the order status. 
  120.       $data['status'] = 'wc-' === substr( $data['status'], 0, 3 ) ? substr( $data['status'], 3 ) : $data['status']; 
  121.  
  122.       // Format line items. 
  123.       foreach ( $format_line_items as $key ) { 
  124.           $data[ $key ] = array_values( array_map( array( $this, 'get_order_item_data' ), $data[ $key ] ) ); 
  125.       } 
  126.  
  127.       // Refunds. 
  128.       $data['refunds'] = array(); 
  129.       foreach ( $order->get_refunds() as $refund ) { 
  130.           $data['refunds'][] = array( 
  131.               'id' => $refund->get_id(),  
  132.               'refund' => $refund->get_reason() ? $refund->get_reason() : '',  
  133.               'total' => '-' . wc_format_decimal( $refund->get_amount(), $this->request['dp'] ),  
  134.  ); 
  135.       } 
  136.  
  137.       $context = ! empty( $request['context'] ) ? $request['context'] : 'view'; 
  138.       $data = $this->add_additional_fields_to_object( $data, $request ); 
  139.       $data = $this->filter_response_by_context( $data, $context ); 
  140.       $response = rest_ensure_response( $data ); 
  141.       $response->add_links( $this->prepare_links( $order, $request ) ); 
  142.  
  143.       /** 
  144.        * Filter the data for a response. 
  145.        * 
  146.        * The dynamic portion of the hook name, $this->post_type, refers to post_type of the post being 
  147.        * prepared for the response. 
  148.        * 
  149.        * @param WP_REST_Response   $response   The response object. 
  150.        * @param WP_Post            $post       Post object. 
  151.        * @param WP_REST_Request    $request    Request object. 
  152.        */ 
  153.       return apply_filters( "woocommerce_rest_prepare_{$this->post_type}", $response, $post, $request ); 
  154.   }

  156.   /** 
  157.    * Prepare a single order for create. 
  158.    * 
  159.    * @deprecated 3.0 
  160.    * 
  161.    * @param  WP_REST_Request $request Request object. 
  162.    * @return WP_Error|WC_Order $data Object. 
  163.    */ 
  164.   protected function prepare_item_for_database( $request ) { 
  165.       $id = isset( $request['id'] ) ? absint( $request['id'] ) : 0; 
  166.       $order = new WC_Order( $id ); 
  167.       $schema = $this->get_item_schema(); 
  168.       $data_keys = array_keys( array_filter( $schema['properties'], array( $this, 'filter_writable_props' ) ) ); 
  169.  
  170.       // Handle all writable props 
  171.       foreach ( $data_keys as $key ) { 
  172.           $value = $request[ $key ]; 
  173.  
  174.           if ( ! is_null( $value ) ) { 
  175.               switch ( $key ) { 
  176.                   case 'billing' : 
  177.                   case 'shipping' : 
  178.                       $this->update_address( $order, $value, $key ); 
  179.                       break; 
  180.                   case 'line_items' : 
  181.                   case 'shipping_lines' : 
  182.                   case 'fee_lines' : 
  183.                   case 'coupon_lines' : 
  184.                       if ( is_array( $value ) ) { 
  185.                           foreach ( $value as $item ) { 
  186.                               if ( is_array( $item ) ) { 
  187.                                   if ( $this->item_is_null( $item ) || ( isset( $item['quantity'] ) && 0 === $item['quantity'] ) ) { 
  188.                                       $order->remove_item( $item['id'] ); 
  189.                                   } else { 
  190.                                       $this->set_item( $order, $key, $item ); 
  191.                                   } 
  192.                               } 
  193.                           } 
  194.                       } 
  195.                       break; 
  196.                   case 'meta_data' : 
  197.                       if ( is_array( $value ) ) { 
  198.                           foreach ( $value as $meta ) { 
  199.                               $order->update_meta_data( $meta['key'], $meta['value'], isset( $meta['id'] ) ? $meta['id'] : '' ); 
  200.                           } 
  201.                       } 
  202.                       break; 
  203.                   default : 
  204.                       if ( is_callable( array( $order, "set_{$key}" ) ) ) { 
  205.                           $order->{"set_{$key}"}( $value ); 
  206.                       } 
  207.                       break; 
  208.               } 
  209.           } 
  210.       }

  236.   /** 
  237.    * Create order. 
  238.    * 
  239.    * @deprecated 3.0.0 
  240.    * 
  241.    * @param WP_REST_Request $request Full details about the request. 
  242.    * @return int|WP_Error 
  243.    */ 
  244.   protected function create_order( $request ) { 
  245.       try { 
  246.           // Make sure customer exists. 
  247.           if ( ! is_null( $request['customer_id'] ) && 0 !== $request['customer_id'] && false === get_user_by( 'id', $request['customer_id'] ) ) { 
  248.               throw new WC_REST_Exception( 'woocommerce_rest_invalid_customer_id', __( 'Customer ID is invalid.', 'woocommerce' ), 400 ); 
  249.           } 
  250.  
  251.           $order = $this->prepare_item_for_database( $request ); 
  252.           $order->set_created_via( 'rest-api' ); 
  253.           $order->set_prices_include_tax( 'yes' === get_option( 'woocommerce_prices_include_tax' ) ); 
  254.           $order->calculate_totals(); 
  255.           $order->save(); 
  256.  
  257.           // Handle set paid. 
  258.           if ( true === $request['set_paid'] ) { 
  259.               $order->payment_complete( $request['transaction_id'] ); 
  260.           } 
  261.  
  262.           return $order->get_id(); 
  263.       } catch ( WC_Data_Exception $e ) { 
  264.           return new WP_Error( $e->getErrorCode(), $e->getMessage(), $e->getErrorData() ); 
  265.       } catch ( WC_REST_Exception $e ) { 
  266.           return new WP_Error( $e->getErrorCode(), $e->getMessage(), array( 'status' => $e->getCode() ) ); 
  267.       } 
  268.   }

  278.   protected function update_order( $request ) { 
  279.       try { 
  280.           $order = $this->prepare_item_for_database( $request ); 
  281.           $order->save(); 
  282.  
  283.           // Handle set paid. 
  284.           if ( $order->needs_payment() && true === $request['set_paid'] ) { 
  285.               $order->payment_complete( $request['transaction_id'] ); 
  286.           } 
  287.  
  288.           // If items have changed, recalculate order totals. 
  289.           if ( isset( $request['billing'] ) || isset( $request['shipping'] ) || isset( $request['line_items'] ) || isset( $request['shipping_lines'] ) || isset( $request['fee_lines'] ) || isset( $request['coupon_lines'] ) ) { 
  290.               $order->calculate_totals(); 
  291.           } 
  292.  
  293.           return $order->get_id(); 
  294.       } catch ( WC_Data_Exception $e ) { 
  295.           return new WP_Error( $e->getErrorCode(), $e->getMessage(), $e->getErrorData() ); 
  296.       } catch ( WC_REST_Exception $e ) { 
  297.           return new WP_Error( $e->getErrorCode(), $e->getMessage(), array( 'status' => $e->getCode() ) );
