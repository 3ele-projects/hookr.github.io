  203.   protected function get_views() { 
  204.       $status_links = array();

  206.       $class = ''; 
  207.       $total_posts = array_sum( (array) $num_posts ); 
  208.  
  209.       // Subtract post types that are not included in the admin all list. 
  210.       foreach ( get_post_stati( array( 'show_in_admin_all_list' => false ) ) as $state ) {

  215.       /** translators: %s: count */ 
  216.       $status_links['all'] = "<a href='admin.php?page=wc-settings&tab=api&section=webhooks'$class>" . sprintf( _nx( 'All <span class="count">(%s)</span>', 'All <span class="count">(%s)</span>', $total_posts, 'posts', 'woocommerce' ), number_format_i18n( $total_posts ) ) . '</a>'; 
  217.  
  218.       foreach ( get_post_stati( array( 'show_in_admin_status_list' => true ), 'objects' ) as $status ) { 
  219.           $class = ''; 
  220.           $status_name = $status->name; 
  221.  
  222.           if ( ! in_array( $status_name, array( 'publish', 'draft', 'pending', 'trash', 'future', 'private', 'auto-draft' ) ) ) { 
  223.               continue; 
  224.           } 
  225.  
  226.           if ( empty( $num_posts->$status_name ) ) { 
  227.               continue; 
  228.           } 
  229.  
  230.           if ( isset( $_REQUEST['status'] ) && $status_name == $_REQUEST['status'] ) { 
  231.               $class = ' class="current"'; 
  232.           } 
  233.  
  234.           $label = $this->get_status_label( $status_name, $status ); 
  235.  
  236.           $status_links[ $status_name ] = "<a href='admin.php?page=wc-settings&tab=api&section=webhooks&status=$status_name'$class>" . sprintf( translate_nooped_plural( $label, $num_posts->$status_name ), number_format_i18n( $num_posts->$status_name ) ) . '</a>'; 
  237.       } 
  238.  
  239.       return $status_links; 
  240.   }
