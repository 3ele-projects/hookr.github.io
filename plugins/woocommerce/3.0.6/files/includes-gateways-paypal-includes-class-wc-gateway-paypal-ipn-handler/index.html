  34.   public function check_response() { 
  35.       if ( ! empty( $_POST ) && $this->validate_ipn() ) { 
  36.           $posted = wp_unslash( $_POST ); 
  37.  
  38.           // @codingStandardsIgnoreStart 
  39.           do_action( 'valid-paypal-standard-ipn-request', $posted ); 
  40.           // @codingStandardsIgnoreEnd 
  41.           exit; 
  42.       } 
  43.  
  44.       wp_die( 'PayPal IPN Request Failure', 'PayPal IPN', array( 'response' => 500 ) );

  47.   /** 
  48.    * There was a valid response. 
  49.    * @param  array $posted Post data after wp_unslash 
  50.    */ 
  51.   public function valid_response( $posted ) { 
  52.       if ( ! empty( $posted['custom'] ) && ( $order = $this->get_paypal_order( $posted['custom'] ) ) ) { 
  53.  
  54.           // Lowercase returned variables. 
  55.           $posted['payment_status'] = strtolower( $posted['payment_status'] ); 
  56.  
  57.           // Sandbox fix. 
  58.           if ( isset( $posted['test_ipn'] ) && 1 == $posted['test_ipn'] && 'pending' == $posted['payment_status'] ) { 
  59.               $posted['payment_status'] = 'completed'; 
  60.           } 
  61.  
  62.           WC_Gateway_Paypal::log( 'Found order #' . $order->get_id() ); 
  63.           WC_Gateway_Paypal::log( 'Payment status: ' . $posted['payment_status'] ); 
  64.  
  65.           if ( method_exists( $this, 'payment_status_' . $posted['payment_status'] ) ) { 
  66.               call_user_func( array( $this, 'payment_status_' . $posted['payment_status'] ), $order, $posted ); 
  67.           } 
  68.       } 
  69.   }

  71.   /** 
  72.    * Check PayPal IPN validity. 
  73.    */ 
  74.   public function validate_ipn() { 
  75.       WC_Gateway_Paypal::log( 'Checking IPN response is valid' ); 
  76.  
  77.       // Get received values from post data 
  78.       $validate_ipn = wp_unslash( $_POST ); 
  79.       $validate_ipn['cmd'] = '_notify-validate'; 
  80.  
  81.       // Send back post vars to paypal 
  82.       $params = array( 
  83.           'body' => $validate_ipn,  
  84.           'timeout' => 60,  
  85.           'httpversion' => '1.1',  
  86.           'compress' => false,  
  87.           'decompress' => false,  
  88.           'user-agent' => 'WooCommerce/' . WC()->version,  
  89.  ); 
  90.  
  91.       // Post back to get a response. 
  92.       $response = wp_safe_remote_post( $this->sandbox ? 'https://www.sandbox.paypal.com/cgi-bin/webscr' : 'https://www.paypal.com/cgi-bin/webscr', $params ); 
  93.  
  94.       WC_Gateway_Paypal::log( 'IPN Request: ' . wc_print_r( $params, true ) ); 
  95.       WC_Gateway_Paypal::log( 'IPN Response: ' . wc_print_r( $response, true ) ); 
  96.  
  97.       // Check to see if the request was valid. 
  98.       if ( ! is_wp_error( $response ) && $response['response']['code'] >= 200 && $response['response']['code'] < 300 && strstr( $response['body'], 'VERIFIED' ) ) { 
  99.           WC_Gateway_Paypal::log( 'Received valid response from PayPal' ); 
  100.           return true; 
  101.       } 
  102.  
  103.       WC_Gateway_Paypal::log( 'Received invalid response from PayPal' ); 
  104.  
  105.       if ( is_wp_error( $response ) ) { 
  106.           WC_Gateway_Paypal::log( 'Error response: ' . $response->get_error_message() ); 
  107.       } 
  108.  
  109.       return false; 
  110.   }

  112.   /** 
  113.    * Check for a valid transaction type. 
  114.    * @param string $txn_type 
  115.    */ 
  116.   protected function validate_transaction_type( $txn_type ) { 
  117.       $accepted_types = array( 'cart', 'instant', 'express_checkout', 'web_accept', 'masspay', 'send_money', 'paypal_here' ); 
  118.  
  119.       if ( ! in_array( strtolower( $txn_type ), $accepted_types ) ) { 
  120.           WC_Gateway_Paypal::log( 'Aborting, Invalid type:' . $txn_type ); 
  121.           exit; 
  122.       } 
  123.   }

  125.   /** 
  126.    * Check currency from IPN matches the order. 
  127.    * @param WC_Order $order 
  128.    * @param string $currency 
  129.    */ 
  130.   protected function validate_currency( $order, $currency ) { 
  131.       if ( $order->get_currency() != $currency ) { 
  132.           WC_Gateway_Paypal::log( 'Payment error: Currencies do not match (sent "' . $order->get_currency() . '" | returned "' . $currency . '")' ); 
  133.  
  134.           // Put this order on-hold for manual checking. 
  135.           $order->update_status( 'on-hold', sprintf( __( 'Validation error: PayPal currencies do not match (code %s).', 'woocommerce' ), $currency ) ); 
  136.           exit; 
  137.       } 
  138.   }

  140.   /** 
  141.    * Check payment amount from IPN matches the order. 
  142.    * @param WC_Order $order 
  143.    * @param int $amount 
  144.    */ 
  145.   protected function validate_amount( $order, $amount ) { 
  146.       if ( number_format( $order->get_total(), 2, '.', '' ) != number_format( $amount, 2, '.', '' ) ) { 
  147.           WC_Gateway_Paypal::log( 'Payment error: Amounts do not match (gross ' . $amount . ')' ); 
  148.  
  149.           // Put this order on-hold for manual checking. 
  150.           $order->update_status( 'on-hold', sprintf( __( 'Validation error: PayPal amounts do not match (gross %s).', 'woocommerce' ), $amount ) ); 
  151.           exit; 
  152.       } 
  153.   }

  155.   /** 
  156.    * Check receiver email from PayPal. If the receiver email in the IPN is different than what is stored in. 
  157.    * WooCommerce -> Settings -> Checkout -> PayPal, it will log an error about it. 
  158.    * @param WC_Order $order 
  159.    * @param string $receiver_email

  161.   protected function validate_receiver_email( $order, $receiver_email ) { 
  162.       if ( strcasecmp( trim( $receiver_email ), trim( $this->receiver_email ) ) != 0 ) { 
  163.           WC_Gateway_Paypal::log( "IPN Response is for another account: {$receiver_email}. Your email is {$this->receiver_email}" ); 
  164.  
  165.           // Put this order on-hold for manual checking. 
  166.           $order->update_status( 'on-hold', sprintf( __( 'Validation error: PayPal IPN response from a different email address (%s).', 'woocommerce' ), $receiver_email ) ); 
  167.           exit; 
  168.       } 
  169.   }

  171.   /** 
  172.    * Handle a completed payment. 
  173.    * @param WC_Order $order 
  174.    * @param array $posted 
  175.    */ 
  176.   protected function payment_status_completed( $order, $posted ) { 
  177.       if ( $order->has_status( wc_get_is_paid_statuses() ) ) { 
  178.           WC_Gateway_Paypal::log( 'Aborting, Order #' . $order->get_id() . ' is already complete.' ); 
  179.           exit; 
  180.       } 
  181.  
  182.       $this->validate_transaction_type( $posted['txn_type'] ); 
  183.       $this->validate_currency( $order, $posted['mc_currency'] ); 
  184.       $this->validate_amount( $order, $posted['mc_gross'] ); 
  185.       $this->validate_receiver_email( $order, $posted['receiver_email'] ); 
  186.       $this->save_paypal_meta_data( $order, $posted ); 
  187.  
  188.       if ( 'completed' === $posted['payment_status'] ) { 
  189.           if ( $order->has_status( 'cancelled' ) ) { 
  190.               $this->payment_status_paid_cancelled_order( $order, $posted ); 
  191.           } 
  192.  
  193.           $this->payment_complete( $order, ( ! empty( $posted['txn_id'] ) ? wc_clean( $posted['txn_id'] ) : '' ), __( 'IPN payment completed', 'woocommerce' ) ); 
  194.  
  195.           if ( ! empty( $posted['mc_fee'] ) ) { 
  196.               // Log paypal transaction fee. 
  197.               update_post_meta( $order->get_id(), 'PayPal Transaction Fee', wc_clean( $posted['mc_fee'] ) ); 
  198.           } 
  199.       } else { 
  200.           if ( 'authorization' === $posted['pending_reason'] ) { 
  201.               $this->payment_on_hold( $order, __( 'Payment authorized. Change payment status to processing or complete to capture funds.', 'woocommerce' ) ); 
  202.           } else { 
  203.               $this->payment_on_hold( $order, sprintf( __( 'Payment pending (%s).', 'woocommerce' ), $posted['pending_reason'] ) ); 
  204.           } 
  205.       } 
  206.   }
