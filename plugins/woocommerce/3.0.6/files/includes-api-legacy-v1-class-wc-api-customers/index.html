  277.   /** 
  278.    * Helper method to get customer user objects 
  279.    * 
  280.    * Note that WP_User_Query does not have built-in pagination so limit & offset are used to provide limited 
  281.    * pagination support 
  282.    * 
  283.    * @since 2.1 
  284.    * @param array $args request arguments for filtering query 
  285.    * @return WP_User_Query 
  286.    */ 
  287.   private function query_customers( $args = array() ) { 
  288.  
  289.       // default users per page 
  290.       $users_per_page = get_option( 'posts_per_page' ); 
  291.  
  292.       // set base query arguments 
  293.       $query_args = array( 
  294.           'fields' => 'ID',  
  295.           'role' => 'customer',  
  296.           'orderby' => 'registered',  
  297.           'number' => $users_per_page,  
  298.  ); 
  299.  
  300.       // search 
  301.       if ( ! empty( $args['q'] ) ) { 
  302.           $query_args['search'] = $args['q']; 
  303.       } 
  304.  
  305.       // limit number of users returned 
  306.       if ( ! empty( $args['limit'] ) ) { 
  307.  
  308.           $query_args['number'] = absint( $args['limit'] ); 
  309.  
  310.           $users_per_page = absint( $args['limit'] ); 
  311.       } 
  312.  
  313.       // page 
  314.       $page = ( isset( $args['page'] ) ) ? absint( $args['page'] ) : 1; 
  315.  
  316.       // offset 
  317.       if ( ! empty( $args['offset'] ) ) { 
  318.           $query_args['offset'] = absint( $args['offset'] ); 
  319.       } else { 
  320.           $query_args['offset'] = $users_per_page * ( $page - 1 ); 
  321.       } 
  322.  
  323.       // created date 
  324.       if ( ! empty( $args['created_at_min'] ) ) { 
  325.           $this->created_at_min = $this->server->parse_datetime( $args['created_at_min'] ); 
  326.       } 
  327.  
  328.       if ( ! empty( $args['created_at_max'] ) ) { 
  329.           $this->created_at_max = $this->server->parse_datetime( $args['created_at_max'] ); 
  330.       } 
  331.  
  332.       $query = new WP_User_Query( $query_args ); 
  333.  
  334.       // helper members for pagination headers 
  335.       $query->total_pages = ceil( $query->get_total() / $users_per_page ); 
  336.       $query->page = $page; 
  337.  
  338.       return $query; 
  339.   }

  408.   /** 
  409.    * Validate the request by checking: 
  410.    * 
  411.    * 1) the ID is a valid integer 
  412.    * 2) the ID returns a valid WP_User 
  413.    * 3) the current user has the proper permissions 
  414.    * 
  415.    * @since 2.1 
  416.    * @see WC_API_Resource::validate_request() 
  417.    * @param string|int $id the customer ID 
  418.    * @param string $type the request type, unused because this method overrides the parent class 
  419.    * @param string $context the context of the request, either `read`, `edit` or `delete` 
  420.    * @return int|WP_Error valid user ID or WP_Error if any of the checks fails 
  421.    */ 
  422.   protected function validate_request( $id, $type, $context ) { 
  423.  
  424.       $id = absint( $id ); 
  425.  
  426.       // validate ID 
  427.       if ( empty( $id ) ) 
  428.           return new WP_Error( 'woocommerce_api_invalid_customer_id', __( 'Invalid customer ID', 'woocommerce' ), array( 'status' => 404 ) ); 
  429.  
  430.       // non-existent IDs return a valid WP_User object with the user ID = 0 
  431.       $customer = new WP_User( $id ); 
  432.  
  433.       if ( 0 === $customer->ID ) 
  434.           return new WP_Error( 'woocommerce_api_invalid_customer', __( 'Invalid customer', 'woocommerce' ), array( 'status' => 404 ) ); 
  435.  
  436.       // validate permissions 
  437.       switch ( $context ) { 
  438.  
  439.           case 'read': 
  440.               if ( ! current_user_can( 'list_users' ) ) 
  441.                   return new WP_Error( 'woocommerce_api_user_cannot_read_customer', __( 'You do not have permission to read this customer', 'woocommerce' ), array( 'status' => 401 ) ); 
  442.               break; 
  443.  
  444.           case 'edit': 
  445.               if ( ! current_user_can( 'edit_users' ) ) 
  446.                   return new WP_Error( 'woocommerce_api_user_cannot_edit_customer', __( 'You do not have permission to edit this customer', 'woocommerce' ), array( 'status' => 401 ) ); 
  447.               break; 
  448.  
  449.           case 'delete': 
  450.               if ( ! current_user_can( 'delete_users' ) ) 
  451.                   return new WP_Error( 'woocommerce_api_user_cannot_delete_customer', __( 'You do not have permission to delete this customer', 'woocommerce' ), array( 'status' => 401 ) ); 
  452.               break; 
  453.       } 
  454.  
  455.       return $id; 
  456.   }
