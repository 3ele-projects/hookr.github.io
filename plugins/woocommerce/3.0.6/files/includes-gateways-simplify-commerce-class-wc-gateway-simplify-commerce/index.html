  341.   /** 
  342.    * Actually saves a customer token to the database. 
  343.    * 
  344.    * @param  WC_Payment_Token   $customer_token Payment Token 
  345.    * @param  string             $cart_token     CC Token 
  346.    * @param  array              $customer_info  'email', 'name' 
  347.    */ 
  348.   public function save_token( $customer_token, $cart_token, $customer_info ) { 
  349.       if ( ! is_null( $customer_token ) ) { 
  350.           $customer = Simplify_Customer::findCustomer( $customer_token->get_token() ); 
  351.           $updates = array( 'token' => $cart_token ); 
  352.           $customer->setAll( $updates ); 
  353.           $customer->updateCustomer(); 
  354.           $customer = Simplify_Customer::findCustomer( $customer_token->get_token() ); // get updated customer with new set card 
  355.           $token = $customer_token; 
  356.       } else { 
  357.           $customer = Simplify_Customer::createCustomer( array( 
  358.               'token' => $cart_token,  
  359.               'email' => $customer_info['email'],  
  360.               'name' => $customer_info['name'],  
  361.  ) ); 
  362.           $token = new WC_Payment_Token_CC(); 
  363.           $token->set_token( $customer->id ); 
  364.       } 
  365.  
  366.       // If we were able to create an save our card, save the data on our side too 
  367.       if ( is_object( $customer ) && '' != $customer->id ) { 
  368.           $customer_properties = $customer->getProperties(); 
  369.           $card = $customer_properties['card']; 
  370.           $token->set_gateway_id( $this->id ); 
  371.           $token->set_card_type( strtolower( $card->type ) ); 
  372.           $token->set_last4( $card->last4 ); 
  373.           $expiry_month = ( 1 === strlen( $card->expMonth ) ? '0' . $card->expMonth : $card->expMonth ); 
  374.           $token->set_expiry_month( $expiry_month ); 
  375.           $token->set_expiry_year( '20' . $card->expYear ); 
  376.           if ( is_user_logged_in() ) { 
  377.               $token->set_user_id( get_current_user_id() ); 
  378.           } 
  379.           $token->save(); 
  380.           return $token; 
  381.       } 
  382.  
  383.       return null; 
  384.   }

  407.   /** 
  408.    * Process standard payments. 
  409.    * 
  410.    * @param  WC_Order $order 
  411.    * @param  string   $cart_token 
  412.    * @uses   Simplify_ApiException 
  413.    * @uses   Simplify_BadRequestException 
  414.    * @return array 
  415.    */ 
  416.   protected function process_standard_payments( $order, $cart_token = '', $customer_token = '' ) { 
  417.       try { 
  418.  
  419.           if ( empty( $cart_token ) && empty( $customer_token ) ) { 
  420.               $error_msg = __( 'Please make sure your card details have been entered correctly and that your browser supports JavaScript.', 'woocommerce' ); 
  421.  
  422.               if ( 'yes' == $this->sandbox ) { 
  423.                   $error_msg .= ' ' . __( 'Developers: Please make sure that you\'re including jQuery and there are no JavaScript errors on the page.', 'woocommerce' ); 
  424.               } 
  425.  
  426.               throw new Simplify_ApiException( $error_msg ); 
  427.           } 
  428.  
  429.           // We need to figure out if we want to charge the card token (new unsaved token, no customer, etc) 
  430.           // or the customer token (just saved method, previously saved method) 
  431.           $pass_tokens = array(); 
  432.  
  433.           if ( ! empty( $cart_token ) ) { 
  434.               $pass_tokens['token'] = $cart_token; 
  435.           } 
  436.  
  437.           if ( ! empty( $customer_token ) ) { 
  438.               $pass_tokens['customer'] = $customer_token; 
  439.               // Use the customer token only, since we already saved the (one time use) card token to the customer 
  440.               if ( isset( $_POST['wc-simplify_commerce-new-payment-method'] ) && true === (bool) $_POST['wc-simplify_commerce-new-payment-method'] ) { 
  441.                   unset( $pass_tokens['token'] ); 
  442.               } 
  443.           } 
  444.  
  445.           // Did we create an account and save a payment method? We might need to use the customer token instead of the card token 
  446.           if ( isset( $_POST['createaccount'] ) && true === (bool) $_POST['createaccount'] && empty( $customer_token ) ) { 
  447.               $user_token = $this->get_users_token(); 
  448.               if ( ! is_null( $user_token ) ) { 
  449.                   $pass_tokens['customer'] = $user_token->get_token(); 
  450.                   unset( $pass_tokens['token'] ); 
  451.               } 
  452.           } 
  453.  
  454.           $payment_response = $this->do_payment( $order, $order->get_total(), $pass_tokens ); 
  455.  
  456.           if ( is_wp_error( $payment_response ) ) { 
  457.               throw new Simplify_ApiException( $payment_response->get_error_message() ); 
  458.           } else { 
  459.               // Remove cart 
  460.               WC()->cart->empty_cart(); 
  461.  
  462.               // Return thank you page redirect 
  463.               return array( 
  464.                   'result' => 'success',  
  465.                   'redirect' => $this->get_return_url( $order ),  
  466.  ); 
  467.           } 
  468.       } catch ( Simplify_ApiException $e ) { 
  469.           if ( $e instanceof Simplify_BadRequestException && $e->hasFieldErrors() && $e->getFieldErrors() ) { 
  470.               foreach ( $e->getFieldErrors() as $error ) {

  484.    /** 
  485.    * do payment function. 
  486.    * 
  487.    * @param WC_order $order 
  488.    * @param int $amount (default: 0) 
  489.    * @uses  Simplify_BadRequestException 
  490.    * @return bool|WP_Error 
  491.    */ 
  492.   public function do_payment( $order, $amount = 0, $token = array() ) { 
  493.       if ( $amount * 100 < 50 ) { 
  494.           return new WP_Error( 'simplify_error', __( 'Sorry, the minimum allowed order total is 0.50 to use this payment method.', 'woocommerce' ) ); 
  495.       } 
  496.  
  497.       try { 
  498.           // Charge the customer 
  499.           $data = array( 
  500.               'amount' => $amount * 100, // In cents. 
  501.               'description' => sprintf( __( '%1$s - Order #%2$s', 'woocommerce' ), esc_html( get_bloginfo( 'name', 'display' ) ), $order->get_order_number() ),  
  502.               'currency' => strtoupper( get_woocommerce_currency() ),  
  503.               'reference' => $order->get_id(),  
  504.  ); 
  505.  
  506.           $data = array_merge( $data, $token ); 
  507.           $payment = Simplify_Payment::createPayment( $data ); 
  508.  
  509.       } catch ( Exception $e ) { 
  510.  
  511.           $error_message = $e->getMessage(); 
  512.  
  513.           if ( $e instanceof Simplify_BadRequestException && $e->hasFieldErrors() && $e->getFieldErrors() ) { 
  514.               $error_message = ''; 
  515.               foreach ( $e->getFieldErrors() as $error ) { 
  516.                   $error_message .= ' ' . $error->getFieldName() . ': "' . $error->getMessage() . '" (' . $error->getErrorCode() . ')'; 
  517.               } 
  518.           } 
  519.  
  520.           $order->add_order_note( sprintf( __( 'Simplify payment error: %s.', 'woocommerce' ), $error_message ) ); 
  521.  
  522.           return new WP_Error( 'simplify_payment_declined', $e->getMessage(), array( 'status' => $e->getCode() ) ); 
  523.       } 
  524.  
  525.       if ( 'APPROVED' == $payment->paymentStatus ) { 
  526.           // Payment complete 
  527.           $order->payment_complete( $payment->id ); 
  528.  
  529.           // Add order note 
  530.           $order->add_order_note( sprintf( __( 'Simplify payment approved (ID: %1$s, Auth Code: %2$s)', 'woocommerce' ), $payment->id, $payment->authCode ) ); 
  531.  
  532.           return true; 
  533.       } else { 
  534.           $order->add_order_note( __( 'Simplify payment declined', 'woocommerce' ) ); 
  535.  
  536.           return new WP_Error( 'simplify_payment_declined', __( 'Payment was declined - please try another card.', 'woocommerce' ) ); 
  537.       } 
  538.   }

  567.   /** 
  568.    * Process the payment. 
  569.    * 
  570.    * @param int $order_id 
  571.    */ 
  572.   public function process_payment( $order_id ) { 
  573.       $order = wc_get_order( $order_id ); 
  574.  
  575.       // Payment/CC form is hosted on Simplify 
  576.       if ( 'hosted' === $this->mode ) { 
  577.           return $this->process_hosted_payments( $order ); 
  578.       } 
  579.  
  580.       // New CC info was entered 
  581.       if ( isset( $_POST['simplify_token'] ) ) { 
  582.           $cart_token = wc_clean( $_POST['simplify_token'] ); 
  583.           $customer_token = $this->get_users_token(); 
  584.           $customer_token_value = ( ! is_null( $customer_token ) ? $customer_token->get_token() : '' ); 
  585.           $this->process_customer( $order, $customer_token, $cart_token ); 
  586.           return $this->process_standard_payments( $order, $cart_token, $customer_token_value ); 
  587.       } 
  588.  
  589.       // Possibly Create (or update) customer/save payment token, use an existing token, and then process the payment 
  590.       if ( isset( $_POST['wc-simplify_commerce-payment-token'] ) && 'new' !== $_POST['wc-simplify_commerce-payment-token'] ) { 
  591.           $token_id = wc_clean( $_POST['wc-simplify_commerce-payment-token'] ); 
  592.           $token = WC_Payment_Tokens::get( $token_id ); 
  593.           if ( $token->get_user_id() !== get_current_user_id() ) { 
  594.               wc_add_notice( __( 'Please make sure your card details have been entered correctly and that your browser supports JavaScript.', 'woocommerce' ), 'error' ); 
  595.               return; 
  596.           } 
  597.           $this->process_customer( $order, $token ); 
  598.           return $this->process_standard_payments( $order, '', $token->get_token() ); 
  599.       } 
  600.   }

  670.               wp_redirect( $this->get_return_url( $order ) );
