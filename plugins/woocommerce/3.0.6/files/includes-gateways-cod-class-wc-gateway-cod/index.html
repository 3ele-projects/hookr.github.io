  117.   /** 
  118.    * Check If The Gateway Is Available For Use. 
  119.    * 
  120.    * @return bool 
  121.    */ 
  122.   public function is_available() { 
  123.       $order = null; 
  124.       $needs_shipping = false; 
  125.  
  126.       // Test if shipping is needed first 
  127.       if ( WC()->cart && WC()->cart->needs_shipping() ) { 
  128.           $needs_shipping = true; 
  129.       } elseif ( is_page( wc_get_page_id( 'checkout' ) ) && 0 < get_query_var( 'order-pay' ) ) { 
  130.           $order_id = absint( get_query_var( 'order-pay' ) ); 
  131.           $order = wc_get_order( $order_id ); 
  132.  
  133.           // Test if order needs shipping. 
  134.           if ( 0 < sizeof( $order->get_items() ) ) { 
  135.               foreach ( $order->get_items() as $item ) { 
  136.                   $_product = $item->get_product(); 
  137.                   if ( $_product && $_product->needs_shipping() ) { 
  138.                       $needs_shipping = true; 
  139.                       break; 
  140.                   } 
  141.               } 
  142.           } 
  143.       } 
  144.  
  145.       $needs_shipping = apply_filters( 'woocommerce_cart_needs_shipping', $needs_shipping ); 
  146.  
  147.       // Virtual order, with virtual disabled 
  148.       if ( ! $this->enable_for_virtual && ! $needs_shipping ) { 
  149.           return false; 
  150.       } 
  151.  
  152.       // Check methods 
  153.       if ( ! empty( $this->enable_for_methods ) && $needs_shipping ) { 
  154.  
  155.           // Only apply if all packages are being shipped via chosen methods, or order is virtual 
  156.           $chosen_shipping_methods_session = WC()->session->get( 'chosen_shipping_methods' ); 
  157.  
  158.           if ( isset( $chosen_shipping_methods_session ) ) { 
  159.               $chosen_shipping_methods = array_unique( $chosen_shipping_methods_session ); 
  160.           } else { 
  161.               $chosen_shipping_methods = array(); 
  162.           } 
  163.  
  164.           $check_method = false; 
  165.  
  166.           if ( is_object( $order ) ) { 
  167.               if ( $order->shipping_method ) { 
  168.                   $check_method = $order->shipping_method; 
  169.               } 
  170.           } elseif ( empty( $chosen_shipping_methods ) || sizeof( $chosen_shipping_methods ) > 1 ) { 
  171.               $check_method = false; 
  172.           } elseif ( sizeof( $chosen_shipping_methods ) == 1 ) { 
  173.               $check_method = $chosen_shipping_methods[0]; 
  174.           } 
  175.  
  176.           if ( ! $check_method ) { 
  177.               return false; 
  178.           } 
  179.  
  180.           $found = false; 
  181.  
  182.           foreach ( $this->enable_for_methods as $method_id ) { 
  183.               if ( strpos( $check_method, $method_id ) === 0 ) { 
  184.                   $found = true; 
  185.                   break; 
  186.               } 
  187.           } 
  188.  
  189.           if ( ! $found ) { 
  190.               return false; 
  191.           } 
  192.       } 
  193.  
  194.       return parent::is_available(); 
  195.   }

  204.   public function process_payment( $order_id ) { 
  205.       $order = wc_get_order( $order_id ); 
  206.  
  207.       // Mark as processing or on-hold (payment won't be taken until delivery) 
  208.       $order->update_status( apply_filters( 'woocommerce_cod_process_payment_order_status', $order->has_downloadable_item() ? 'on-hold' : 'processing', $order ), __( 'Payment to be made upon delivery.', 'woocommerce' ) ); 
  209.  
  210.       // Reduce stock levels 
  211.       wc_reduce_stock_levels( $order_id ); 
  212.  
  213.       // Remove cart 
  214.       WC()->cart->empty_cart(); 
  215.  
  216.       // Return thankyou redirect 
  217.       return array( 
  218.           'result' => 'success',  
  219.           'redirect' => $this->get_return_url( $order ),  
  220.  );
