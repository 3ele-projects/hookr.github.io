  243. /** 
  244.  * Format decimal numbers ready for DB storage. 
  245.  * 
  246.  * Sanitize, remove locale formatting, and optionally round + trim off zeros. 
  247.  * 
  248.  * @param  float|string $number Expects either a float or a string with a decimal separator only (no thousands) 
  249.  * @param  mixed $dp number of decimal points to use, blank to use woocommerce_price_num_decimals, or false to avoid all rounding. 
  250.  * @param  bool $trim_zeros from end of string 
  251.  * @return string 
  252.  */ 
  253. function wc_format_decimal( $number, $dp = false, $trim_zeros = false ) { 
  254.   $locale = localeconv(); 
  255.   $decimals = array( wc_get_price_decimal_separator(), $locale['decimal_point'], $locale['mon_decimal_point'] ); 
  256.  
  257.   // Remove locale from string. 
  258.   if ( ! is_float( $number ) ) { 
  259.       $number = wc_clean( str_replace( $decimals, '.', $number ) ); 
  260.       $number = preg_replace( '/[^0-9\., -]/', '', $number ); 
  261.   } 
  262.  
  263.   if ( false !== $dp ) { 
  264.       $dp = intval( '' == $dp ? wc_get_price_decimals() : $dp ); 
  265.       $number = number_format( floatval( $number ), $dp, '.', '' ); 
  266.  
  267.   // DP is false - don't use number format, just return a string in our format 
  268.   } elseif ( is_float( $number ) ) { 
  269.       $number = wc_clean( str_replace( $decimals, '.', strval( $number ) ) ); 
  270.   } 
  271.  
  272.   if ( $trim_zeros && strstr( $number, '.' ) ) { 
  273.       $number = rtrim( rtrim( $number, '0' ), '.' ); 
  274.   } 
  275.  
  276.   return $number; 
  277. }

  563. /** 
  564.  * WooCommerce Timezone - helper to retrieve the timezone string for a site until. 
  565.  * a WP core method exists (see https://core.trac.wordpress.org/ticket/24730). 
  566.  * 
  567.  * Adapted from https://secure.php.net/manual/en/function.timezone-name-from-abbr.php#89155. 
  568.  * 
  569.  * @since 2.1 
  570.  * @return string PHP timezone string for the site 
  571.  */ 
  572. function wc_timezone_string() { 
  573.  
  574.   // if site timezone string exists, return it 
  575.   if ( $timezone = get_option( 'timezone_string' ) ) { 
  576.       return $timezone; 
  577.   } 
  578.  
  579.   // get UTC offset, if it isn't set then return UTC 
  580.   if ( 0 === ( $utc_offset = intval( get_option( 'gmt_offset', 0 ) ) ) ) { 
  581.       return 'UTC'; 
  582.   } 
  583.  
  584.   // adjust UTC offset from hours to seconds 
  585.   $utc_offset *= 3600; 
  586.  
  587.   // attempt to guess the timezone string from the UTC offset 
  588.   if ( $timezone = timezone_name_from_abbr( '', $utc_offset ) ) { 
  589.       return $timezone; 
  590.   }
