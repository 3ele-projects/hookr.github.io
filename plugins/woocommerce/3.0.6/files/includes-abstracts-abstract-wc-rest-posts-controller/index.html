  512.                * Filter the query_vars used in `get_items` for the constructed query. 
  513.                * 
  514.                * The dynamic portion of the hook name, $var, refers to the query_var key. 
  515.                * 
  516.                * @param mixed $prepared_args[ $var ] The query_var value. 
  517.                * 
  518.                */ 
  519.               $query_args[ $var ] = apply_filters( "woocommerce_rest_query_var-{$var}", $prepared_args[ $var ] ); 
  520.           } 
  521.       } 
  522.  
  523.       $query_args['ignore_sticky_posts'] = true; 
  524.  
  525.       if ( 'include' === $query_args['orderby'] ) { 
  526.           $query_args['orderby'] = 'post__in'; 
  527.       } elseif ( 'id' === $query_args['orderby'] ) { 
  528.           $query_args['orderby'] = 'ID'; // ID must be capitalized 
  529.       } 
  530.  
  531.       return $query_args; 
  532.   }

  535.    * Get all the WP Query vars that are allowed for the API request. 
  536.    * 
  537.    * @return array

  539.   protected function get_allowed_query_vars() { 
  540.       global $wp; 
  541.  
  542.       /** 
  543.        * Filter the publicly allowed query vars. 
  544.        * 
  545.        * Allows adjusting of the default query vars that are made public. 
  546.        * 
  547.        * @param array  Array of allowed WP_Query query vars. 
  548.        */ 
  549.       $valid_vars = apply_filters( 'query_vars', $wp->public_query_vars ); 
  550.  
  551.       $post_type_obj = get_post_type_object( $this->post_type ); 
  552.       if ( current_user_can( $post_type_obj->cap->edit_posts ) ) { 
  553.           /** 
  554.            * Filter the allowed 'private' query vars for authorized users. 
  555.            * 
  556.            * If the user has the `edit_posts` capability, we also allow use of 
  557.            * private query parameters, which are only undesirable on the 
  558.            * frontend, but are safe for use in query strings. 
  559.            * 
  560.            * To disable anyway, use 
  561.            * `add_filter( 'woocommerce_rest_private_query_vars', '__return_empty_array' );` 
  562.            * 
  563.            * @param array $private_query_vars Array of allowed query vars for authorized users. 
  564.            * } 
  565.            */ 
  566.           $private = apply_filters( 'woocommerce_rest_private_query_vars', $wp->private_query_vars ); 
  567.           $valid_vars = array_merge( $valid_vars, $private ); 
  568.       } 
  569.       // Define our own in addition to WP's normal vars. 
  570.       $rest_valid = array( 
  571.           'date_query',  
  572.           'ignore_sticky_posts',  
  573.           'offset',  
  574.           'post__in',  
  575.           'post__not_in',  
  576.           'post_parent',  
  577.           'post_parent__in',  
  578.           'post_parent__not_in',  
  579.           'posts_per_page',  
  580.           'meta_query',  
  581.           'tax_query',  
  582.           'meta_key',  
  583.           'meta_value',  
  584.           'meta_compare',  
  585.           'meta_value_num',  
  586.  ); 
  587.       $valid_vars = array_merge( $valid_vars, $rest_valid ); 
  588.  
  589.       /** 
  590.        * Filter allowed query vars for the REST API. 
  591.        * 
  592.        * This filter allows you to add or remove query vars from the final allowed 
  593.        * list for all requests, including unauthenticated ones. To alter the 
  594.        * vars for editors only. 
  595.        * 
  596.        * @param array { 
  597.        *    Array of allowed WP_Query query vars. 
  598.        * 
  599.        *    @param string $allowed_query_var The query var to allow. 
  600.        * } 
  601.        */ 
  602.       $valid_vars = apply_filters( 'woocommerce_rest_query_vars', $valid_vars ); 
  603.  
  604.       return $valid_vars; 
  605.   }
