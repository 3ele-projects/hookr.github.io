  243.       if ( isset( $request['id'] ) ) {

  254.       } 
  255.  
  256.       // SKU. 
  257.       if ( isset( $request['sku'] ) ) { 
  258.           $variation->set_sku( wc_clean( $request['sku'] ) ); 
  259.       } 
  260.  
  261.       // Thumbnail. 
  262.       if ( isset( $request['image'] ) ) { 
  263.           if ( is_array( $request['image'] ) ) {

  265.               if ( is_array( $image ) ) { 
  266.                   $image['position'] = 0; 
  267.               }

  269.               $variation = $this->set_product_images( $variation, array( $image ) );

  272.           } 
  273.       } 
  274.  
  275.       // Virtual variation. 
  276.       if ( isset( $request['virtual'] ) ) { 
  277.           $variation->set_virtual( $request['virtual'] ); 
  278.       } 
  279.  
  280.       // Downloadable variation. 
  281.       if ( isset( $request['downloadable'] ) ) { 
  282.           $variation->set_downloadable( $request['downloadable'] ); 
  283.       } 
  284.  
  285.       // Downloads. 
  286.       if ( $variation->get_downloadable() ) { 
  287.           // Downloadable files. 
  288.           if ( isset( $request['downloads'] ) && is_array( $request['downloads'] ) ) { 
  289.               $variation = $this->save_downloadable_files( $variation, $request['downloads'] ); 
  290.           } 
  291.  
  292.           // Download limit. 
  293.           if ( isset( $request['download_limit'] ) ) { 
  294.               $variation->set_download_limit( $request['download_limit'] ); 
  295.           } 
  296.  
  297.           // Download expiry. 
  298.           if ( isset( $request['download_expiry'] ) ) { 
  299.               $variation->set_download_expiry( $request['download_expiry'] );

  303.       // Shipping data. 
  304.       $variation = $this->save_product_shipping_data( $variation, $request ); 
  305.  
  306.       // Stock handling. 
  307.       if ( isset( $request['manage_stock'] ) ) { 
  308.           $variation->set_manage_stock( $request['manage_stock'] ); 
  309.       } 
  310.  
  311.       if ( isset( $request['in_stock'] ) ) { 
  312.           $variation->set_stock_status( true === $request['in_stock'] ? 'instock' : 'outofstock' ); 
  313.       } 
  314.  
  315.       if ( isset( $request['backorders'] ) ) { 
  316.           $variation->set_backorders( $request['backorders'] );

  318.

  321.               $variation->set_stock_quantity( $request['stock_quantity'] );

  326.           } 
  327.       } else {

  330.       } 
  331.  
  332.       // Regular Price. 
  333.       if ( isset( $request['regular_price'] ) ) { 
  334.           $variation->set_regular_price( $request['regular_price'] ); 
  335.       } 
  336.  
  337.       // Sale Price. 
  338.       if ( isset( $request['sale_price'] ) ) { 
  339.           $variation->set_sale_price( $request['sale_price'] ); 
  340.       } 
  341.  
  342.       if ( isset( $request['date_on_sale_from'] ) ) { 
  343.           $variation->set_date_on_sale_from( $request['date_on_sale_from'] ); 
  344.       } 
  345.  
  346.       if ( isset( $request['date_on_sale_from_gmt'] ) ) { 
  347.           $variation->set_date_on_sale_from( $request['date_on_sale_from_gmt'] ? strtotime( $request['date_on_sale_from_gmt'] ) : null );

  349.  
  350.       if ( isset( $request['date_on_sale_to'] ) ) { 
  351.           $variation->set_date_on_sale_to( $request['date_on_sale_to'] );

  353.  
  354.       if ( isset( $request['date_on_sale_to_gmt'] ) ) { 
  355.           $variation->set_date_on_sale_to( $request['date_on_sale_to_gmt'] ? strtotime( $request['date_on_sale_to_gmt'] ) : null );

  449.   /** 
  450.    * Delete a variation. 
  451.    * 
  452.    * @param WP_REST_Request $request Full details about the request 
  453.    * @return WP_Error|boolean 
  454.    */ 
  455.   public function delete_item( $request ) { 
  456.       $id = absint( is_array( $request['id'] ) ? $request['id']['id'] : $request['id'] ); 
  457.       $force = (bool) $request['force']; 
  458.       $object = $this->get_object( (int) $request['id'] ); 
  459.       $result = false; 
  460.  
  461.       if ( ! $object || 0 === $object->get_id() ) { 
  462.           return new WP_Error( "woocommerce_rest_{$this->post_type}_invalid_id", __( 'Invalid ID.', 'woocommerce' ), array( 'status' => 404 ) ); 
  463.       } 
  464.  
  465.       $supports_trash = EMPTY_TRASH_DAYS > 0 && is_callable( array( $object, 'get_status' ) ); 
  466.  
  467.       /** 
  468.        * Filter whether an object is trashable. 
  469.        * 
  470.        * Return false to disable trash support for the object. 
  471.        * 
  472.        * @param boolean $supports_trash Whether the object type support trashing. 
  473.        * @param WC_Data $object         The object being considered for trashing support. 
  474.        */ 
  475.       $supports_trash = apply_filters( "woocommerce_rest_{$this->post_type}_object_trashable", $supports_trash, $object ); 
  476.  
  477.       if ( ! wc_rest_check_post_permissions( $this->post_type, 'delete', $object->get_id() ) ) { 
  478.           /** translators: %s: post type */ 
  479.           return new WP_Error( "woocommerce_rest_user_cannot_delete_{$this->post_type}", sprintf( __( 'Sorry, you are not allowed to delete %s.', 'woocommerce' ), $this->post_type ), array( 'status' => rest_authorization_required_code() ) ); 
  480.       } 
  481.  
  482.       $request->set_param( 'context', 'edit' ); 
  483.       $response = $this->prepare_object_for_response( $object, $request ); 
  484.  
  485.       // If we're forcing, then delete permanently. 
  486.       if ( $force ) { 
  487.           $object->delete( true ); 
  488.           $result = 0 === $object->get_id(); 
  489.       } else { 
  490.           // If we don't support trashing for this type, error out. 
  491.           if ( ! $supports_trash ) { 
  492.               /** translators: %s: post type */ 
  493.               return new WP_Error( 'woocommerce_rest_trash_not_supported', sprintf( __( 'The %s does not support trashing.', 'woocommerce' ), $this->post_type ), array( 'status' => 501 ) ); 
  494.           } 
  495.  
  496.           // Otherwise, only trash if we haven't already. 
  497.           if ( is_callable( array( $object, 'get_status' ) ) ) { 
  498.               if ( 'trash' === $object->get_status() ) { 
  499.                   /** translators: %s: post type */ 
  500.                   return new WP_Error( 'woocommerce_rest_already_trashed', sprintf( __( 'The %s has already been deleted.', 'woocommerce' ), $this->post_type ), array( 'status' => 410 ) ); 
  501.               } 
  502.  
  503.               $object->delete(); 
  504.               $result = 'trash' === $object->get_status(); 
  505.           } 
  506.       } 
  507.  
  508.       if ( ! $result ) { 
  509.           /** translators: %s: post type */ 
  510.           return new WP_Error( 'woocommerce_rest_cannot_delete', sprintf( __( 'The %s cannot be deleted.', 'woocommerce' ), $this->post_type ), array( 'status' => 500 ) ); 
  511.       } 
  512.  
  513.       // Delete parent product transients. 
  514.       if ( 0 !== $object->get_parent_id() ) { 
  515.           wc_delete_product_transients( $object->get_parent_id() ); 
  516.       }

  519.        * Fires after a single object is deleted or trashed via the REST API. 
  520.        * 
  521.        * @param WC_Data          $object   The deleted or trashed object.

  523.        * @param WP_REST_Request  $request  The request sent to the API. 
  524.        */ 
  525.       do_action( "woocommerce_rest_delete_{$this->post_type}_object", $object, $response, $request ); 
  526.  
  527.       return $response; 
  528.   }

  530.   /**

  532.    * 
  533.    * @since  3.0.0

  535.    * @return array Of WP_Error or WP_REST_Response. 
  536.    */ 
  537.   public function batch_items( $request ) { 
  538.       $items = array_filter( $request->get_params() ); 
  539.       $params = $request->get_url_params(); 
  540.       $product_id = $params['product_id']; 
  541.       $body_params = array(); 
  542.  
  543.       foreach ( array( 'update', 'create', 'delete' ) as $batch_type ) { 
  544.           if ( ! empty( $items[ $batch_type ] ) ) { 
  545.               $injected_items = array(); 
  546.               foreach ( $items[ $batch_type ] as $item ) { 
  547.                   $injected_items[] = is_array( $item ) ? array_merge( array( 'product_id' => $product_id ), $item ) : $item; 
  548.               } 
  549.               $body_params[ $batch_type ] = $injected_items; 
  550.           } 
  551.       } 
  552.  
  553.       $request = new WP_REST_Request( $request->get_method() ); 
  554.       $request->set_body_params( $body_params ); 
  555.  
  556.       return parent::batch_items( $request ); 
  557.   }
