  31.    */ 
  32.   public function __construct( WC_API_Server $server ) { 
  33.  
  34.       $this->server = $server; 
  35.  
  36.       // automatically register routes for sub-classes 
  37.       add_filter( 'woocommerce_api_endpoints', array( $this, 'register_routes' ) ); 
  38.  
  39.       // maybe add meta to top-level resource responses 
  40.       foreach ( array( 'order', 'coupon', 'customer', 'product', 'report' ) as $resource ) { 
  41.           add_filter( "woocommerce_api_{$resource}_response", array( $this, 'maybe_add_meta' ), 15, 2 );

  43.  
  44.       $response_names = array( 
  45.           'order',  
  46.           'coupon',  
  47.           'customer',  
  48.           'product',  
  49.           'report',

  51.           'customer_downloads',  
  52.           'order_note',  
  53.           'order_refund',  
  54.           'product_reviews',  
  55.           'product_category',  
  56.  ); 
  57.  
  58.       foreach ( $response_names as $name ) { 
  59.  
  60.           /** 
  61.            * Remove fields from responses when requests specify certain fields 
  62.            * note these are hooked at a later priority so data added via 
  63.            * filters (e.g. customer data to the order response) still has the 
  64.            * fields filtered properly

  70.   /** 
  71.    * Validate the request by checking: 
  72.    * 
  73.    * 1) the ID is a valid integer 
  74.    * 2) the ID returns a valid post object and matches the provided post type 
  75.    * 3) the current user has the proper permissions to read/edit/delete the post 
  76.    * 
  77.    * @since 2.1 
  78.    * @param string|int $id the post ID 
  79.    * @param string $type the post type, either `shop_order`, `shop_coupon`, or `product` 
  80.    * @param string $context the context of the request, either `read`, `edit` or `delete` 
  81.    * @return int|WP_Error valid post ID or WP_Error if any of the checks fails 
  82.    */ 
  83.   protected function validate_request( $id, $type, $context ) { 
  84.  
  85.       if ( 'shop_order' === $type || 'shop_coupon' === $type || 'shop_webhook' === $type ) { 
  86.           $resource_name = str_replace( 'shop_', '', $type ); 
  87.       } else { 
  88.           $resource_name = $type; 
  89.       } 
  90.  
  91.       $id = absint( $id ); 
  92.  
  93.       // Validate ID 
  94.       if ( empty( $id ) ) { 
  95.           return new WP_Error( "woocommerce_api_invalid_{$resource_name}_id", sprintf( __( 'Invalid %s ID', 'woocommerce' ), $type ), array( 'status' => 404 ) ); 
  96.       } 
  97.  
  98.       // Only custom post types have per-post type/permission checks 
  99.       if ( 'customer' !== $type ) { 
  100.  
  101.           $post = get_post( $id ); 
  102.  
  103.           if ( null === $post ) { 
  104.               return new WP_Error( "woocommerce_api_no_{$resource_name}_found", sprintf( __( 'No %1$s found with the ID equal to %2$s', 'woocommerce' ), $resource_name, $id ), array( 'status' => 404 ) ); 
  105.           } 
  106.  
  107.           // For checking permissions, product variations are the same as the product post type 
  108.           $post_type = ( 'product_variation' === $post->post_type ) ? 'product' : $post->post_type; 
  109.  
  110.           // Validate post type 
  111.           if ( $type !== $post_type ) { 
  112.               return new WP_Error( "woocommerce_api_invalid_{$resource_name}", sprintf( __( 'Invalid %s', 'woocommerce' ), $resource_name ), array( 'status' => 404 ) ); 
  113.           } 
  114.  
  115.           // Validate permissions 
  116.           switch ( $context ) { 
  117.  
  118.               case 'read': 
  119.                   if ( ! $this->is_readable( $post ) ) 
  120.                       return new WP_Error( "woocommerce_api_user_cannot_read_{$resource_name}", sprintf( __( 'You do not have permission to read this %s', 'woocommerce' ), $resource_name ), array( 'status' => 401 ) ); 
  121.                   break; 
  122.  
  123.               case 'edit': 
  124.                   if ( ! $this->is_editable( $post ) ) 
  125.                       return new WP_Error( "woocommerce_api_user_cannot_edit_{$resource_name}", sprintf( __( 'You do not have permission to edit this %s', 'woocommerce' ), $resource_name ), array( 'status' => 401 ) ); 
  126.                   break; 
  127.  
  128.               case 'delete': 
  129.                   if ( ! $this->is_deletable( $post ) ) 
  130.                       return new WP_Error( "woocommerce_api_user_cannot_delete_{$resource_name}", sprintf( __( 'You do not have permission to delete this %s', 'woocommerce' ), $resource_name ), array( 'status' => 401 ) ); 
  131.                   break; 
  132.           } 
  133.       } 
  134.  
  135.       return $id; 
  136.   }

  138.   /** 
  139.    * Add common request arguments to argument list before WP_Query is run 
  140.    * 
  141.    * @since 2.1 
  142.    * @param array $base_args required arguments for the query (e.g. `post_type`, etc) 
  143.    * @param array $request_args arguments provided in the request 
  144.    * @return array 
  145.    */ 
  146.   protected function merge_query_args( $base_args, $request_args ) { 
  147.  
  148.       $args = array(); 
  149.  
  150.       // date 
  151.       if ( ! empty( $request_args['created_at_min'] ) || ! empty( $request_args['created_at_max'] ) || ! empty( $request_args['updated_at_min'] ) || ! empty( $request_args['updated_at_max'] ) ) { 
  152.  
  153.           $args['date_query'] = array(); 
  154.  
  155.           // resources created after specified date 
  156.           if ( ! empty( $request_args['created_at_min'] ) ) { 
  157.               $args['date_query'][] = array( 'column' => 'post_date_gmt', 'after' => $this->server->parse_datetime( $request_args['created_at_min'] ), 'inclusive' => true ); 
  158.           } 
  159.  
  160.           // resources created before specified date 
  161.           if ( ! empty( $request_args['created_at_max'] ) ) { 
  162.               $args['date_query'][] = array( 'column' => 'post_date_gmt', 'before' => $this->server->parse_datetime( $request_args['created_at_max'] ), 'inclusive' => true ); 
  163.           } 
  164.  
  165.           // resources updated after specified date 
  166.           if ( ! empty( $request_args['updated_at_min'] ) ) { 
  167.               $args['date_query'][] = array( 'column' => 'post_modified_gmt', 'after' => $this->server->parse_datetime( $request_args['updated_at_min'] ), 'inclusive' => true ); 
  168.           } 
  169.  
  170.           // resources updated before specified date 
  171.           if ( ! empty( $request_args['updated_at_max'] ) ) { 
  172.               $args['date_query'][] = array( 'column' => 'post_modified_gmt', 'before' => $this->server->parse_datetime( $request_args['updated_at_max'] ), 'inclusive' => true ); 
  173.           } 
  174.       } 
  175.  
  176.       // search 
  177.       if ( ! empty( $request_args['q'] ) ) { 
  178.           $args['s'] = $request_args['q']; 
  179.       } 
  180.  
  181.       // resources per response 
  182.       if ( ! empty( $request_args['limit'] ) ) { 
  183.           $args['posts_per_page'] = $request_args['limit']; 
  184.       } 
  185.  
  186.       // resource offset 
  187.       if ( ! empty( $request_args['offset'] ) ) { 
  188.           $args['offset'] = $request_args['offset']; 
  189.       } 
  190.  
  191.       // order (ASC or DESC, ASC by default) 
  192.       if ( ! empty( $request_args['order'] ) ) { 
  193.           $args['order'] = $request_args['order']; 
  194.       } 
  195.  
  196.       // orderby 
  197.       if ( ! empty( $request_args['orderby'] ) ) { 
  198.           $args['orderby'] = $request_args['orderby']; 
  199.  
  200.           // allow sorting by meta value 
  201.           if ( ! empty( $request_args['orderby_meta_key'] ) ) { 
  202.               $args['meta_key'] = $request_args['orderby_meta_key']; 
  203.           } 
  204.       } 
  205.  
  206.       // allow post status change 
  207.       if ( ! empty( $request_args['post_status'] ) ) { 
  208.           $args['post_status'] = $request_args['post_status']; 
  209.           unset( $request_args['post_status'] ); 
  210.       } 
  211.  
  212.       // filter by a list of post id 
  213.       if ( ! empty( $request_args['in'] ) ) { 
  214.           $args['post__in'] = explode( ', ', $request_args['in'] ); 
  215.           unset( $request_args['in'] ); 
  216.       } 
  217.  
  218.       // filter by a list of post id 
  219.       if ( ! empty( $request_args['in'] ) ) { 
  220.           $args['post__in'] = explode( ', ', $request_args['in'] ); 
  221.           unset( $request_args['in'] ); 
  222.       } 
  223.  
  224.       // resource page 
  225.       $args['paged'] = ( isset( $request_args['page'] ) ) ? absint( $request_args['page'] ) : 1; 
  226.  
  227.       $args = apply_filters( 'woocommerce_api_query_args', $args, $request_args ); 
  228.  
  229.       return array_merge( $base_args, $args ); 
  230.   }

  232.   /** 
  233.    * Add meta to resources when requested by the client. Meta is added as a top-level 
  234.    * `<resource_name>_meta` attribute (e.g. `order_meta`) as a list of key/value pairs 
  235.    * 
  236.    * @since 2.1 
  237.    * @param array $data the resource data 
  238.    * @param object $resource the resource object (e.g WC_Order) 
  239.    * @return mixed 
  240.    */ 
  241.   public function maybe_add_meta( $data, $resource ) { 
  242.  
  243.       if ( isset( $this->server->params['GET']['filter']['meta'] ) && 'true' === $this->server->params['GET']['filter']['meta'] && is_object( $resource ) ) { 
  244.  
  245.           // don't attempt to add meta more than once 
  246.           if ( preg_grep( '/[a-z]+_meta/', array_keys( $data ) ) ) 
  247.               return $data; 
  248.  
  249.           // define the top-level property name for the meta 
  250.           switch ( get_class( $resource ) ) { 
  251.  
  252.               case 'WC_Order': 
  253.                   $meta_name = 'order_meta'; 
  254.                   break; 
  255.  
  256.               case 'WC_Coupon': 
  257.                   $meta_name = 'coupon_meta'; 
  258.                   break; 
  259.  
  260.               case 'WP_User': 
  261.                   $meta_name = 'customer_meta'; 
  262.                   break; 
  263.  
  264.               default: 
  265.                   $meta_name = 'product_meta'; 
  266.                   break; 
  267.           } 
  268.  
  269.           if ( is_a( $resource, 'WP_User' ) ) { 
  270.  
  271.               // customer meta 
  272.               $meta = (array) get_user_meta( $resource->ID ); 
  273.  
  274.           } elseif ( is_a( $resource, 'WC_Product_Variation' ) ) { 
  275.  
  276.               // product variation meta 
  277.               $meta = (array) get_post_meta( $resource->get_variation_id() ); 
  278.  
  279.           } else { 
  280.  
  281.               // coupon/order/product meta 
  282.               $meta = (array) get_post_meta( $resource->id ); 
  283.           } 
  284.  
  285.           foreach ( $meta as $meta_key => $meta_value ) { 
  286.  
  287.               // don't add hidden meta by default 
  288.               if ( ! is_protected_meta( $meta_key ) ) { 
  289.                   $data[ $meta_name ][ $meta_key ] = maybe_unserialize( $meta_value[0] ); 
  290.               } 
  291.           } 
  292.       } 
  293.  
  294.       return $data; 
  295.   }

  297.   /** 
  298.    * Restrict the fields included in the response if the request specified certain only certain fields should be returned 
  299.    * 
  300.    * @since 2.1 
  301.    * @param array $data the response data 
  302.    * @param object $resource the object that provided the response data, e.g. WC_Coupon or WC_Order 
  303.    * @param array|string the requested list of fields to include in the response 
  304.    * @return array response data 
  305.    */ 
  306.   public function filter_response_fields( $data, $resource, $fields ) { 
  307.  
  308.       if ( ! is_array( $data ) || empty( $fields ) ) { 
  309.           return $data; 
  310.       } 
  311.  
  312.       $fields = explode( ', ', $fields ); 
  313.       $sub_fields = array(); 
  314.  
  315.       // get sub fields 
  316.       foreach ( $fields as $field ) { 
  317.  
  318.           if ( false !== strpos( $field, '.' ) ) { 
  319.  
  320.               list( $name, $value ) = explode( '.', $field ); 
  321.  
  322.               $sub_fields[ $name ] = $value; 
  323.           } 
  324.       } 
  325.  
  326.       // iterate through top-level fields 
  327.       foreach ( $data as $data_field => $data_value ) { 
  328.  
  329.           // if a field has sub-fields and the top-level field has sub-fields to filter 
  330.           if ( is_array( $data_value ) && in_array( $data_field, array_keys( $sub_fields ) ) ) { 
  331.  
  332.               // iterate through each sub-field 
  333.               foreach ( $data_value as $sub_field => $sub_field_value ) { 
  334.  
  335.                   // remove non-matching sub-fields 
  336.                   if ( ! in_array( $sub_field, $sub_fields ) ) { 
  337.                       unset( $data[ $data_field ][ $sub_field ] ); 
  338.                   } 
  339.               } 
  340.           } else { 
  341.  
  342.               // remove non-matching top-level fields 
  343.               if ( ! in_array( $data_field, $fields ) ) { 
  344.                   unset( $data[ $data_field ] ); 
  345.               } 
  346.           } 
  347.       } 
  348.  
  349.       return $data; 
  350.   }

  352.   /** 
  353.    * Delete a given resource 
  354.    * 
  355.    * @since 2.1 
  356.    * @param int $id the resource ID 
  357.    * @param string $type the resource post type, or `customer` 
  358.    * @param bool $force true to permanently delete resource, false to move to trash (not supported for `customer`) 
  359.    * @return array|WP_Error 
  360.    */ 
  361.   protected function delete( $id, $type, $force = false ) { 
  362.  
  363.       if ( 'shop_order' === $type || 'shop_coupon' === $type ) { 
  364.           $resource_name = str_replace( 'shop_', '', $type ); 
  365.       } else { 
  366.           $resource_name = $type; 
  367.       } 
  368.  
  369.       if ( 'customer' === $type ) { 
  370.  
  371.           $result = wp_delete_user( $id ); 
  372.  
  373.           if ( $result ) { 
  374.               return array( 'message' => __( 'Permanently deleted customer', 'woocommerce' ) ); 
  375.           } else { 
  376.               return new WP_Error( 'woocommerce_api_cannot_delete_customer', __( 'The customer cannot be deleted', 'woocommerce' ), array( 'status' => 500 ) ); 
  377.           } 
  378.       } else { 
  379.  
  380.           // delete order/coupon/webhook 
  381.           $result = ( $force ) ? wp_delete_post( $id, true ) : wp_trash_post( $id ); 
  382.  
  383.           if ( ! $result ) { 
  384.               return new WP_Error( "woocommerce_api_cannot_delete_{$resource_name}", sprintf( __( 'This %s cannot be deleted', 'woocommerce' ), $resource_name ), array( 'status' => 500 ) ); 
  385.           } 
  386.  
  387.           if ( $force ) { 
  388.               return array( 'message' => sprintf( __( 'Permanently deleted %s', 'woocommerce' ), $resource_name ) ); 
  389.           } else { 
  390.               $this->server->send_status( '202' ); 
  391.  
  392.               return array( 'message' => sprintf( __( 'Deleted %s', 'woocommerce' ), $resource_name ) ); 
  393.           } 
  394.       } 
  395.   }

  438.   /** 
  439.    * Checks the permissions for the current user given a post and context 
  440.    * 
  441.    * @since 2.1 
  442.    * @param WP_Post|int $post 
  443.    * @param string $context the type of permission to check, either `read`, `write`, or `delete` 
  444.    * @return bool true if the current user has the permissions to perform the context on the post 
  445.    */ 
  446.   private function check_permission( $post, $context ) { 
  447.  
  448.       if ( ! is_a( $post, 'WP_Post' ) ) { 
  449.           $post = get_post( $post ); 
  450.       } 
  451.  
  452.       if ( is_null( $post ) ) { 
  453.           return false; 
  454.       } 
  455.  
  456.       $post_type = get_post_type_object( $post->post_type ); 
  457.  
  458.       if ( 'read' === $context ) { 
  459.           return ( 'revision' !== $post->post_type && current_user_can( $post_type->cap->read_private_posts, $post->ID ) ); 
  460.       } elseif ( 'edit' === $context ) { 
  461.           return current_user_can( $post_type->cap->edit_post, $post->ID ); 
  462.       } elseif ( 'delete' === $context ) { 
  463.           return current_user_can( $post_type->cap->delete_post, $post->ID ); 
  464.       } else { 
  465.           return false; 
  466.       } 
  467.   }
