  220.   /** 
  221.    * Do a REST Request 
  222.    * 
  223.    * @param string $method 
  224.    * 
  225.    */ 
  226.   private function do_request( $method, $route, $assoc_args ) { 
  227.       if ( ! defined( 'REST_REQUEST' ) ) { 
  228.           define( 'REST_REQUEST', true ); 
  229.       } 
  230.       $request = new WP_REST_Request( $method, $route ); 
  231.       if ( in_array( $method, array( 'POST', 'PUT' ) ) ) { 
  232.           $request->set_body_params( $assoc_args ); 
  233.       } else { 
  234.           foreach ( $assoc_args as $key => $value ) { 
  235.               $request->set_param( $key, $value ); 
  236.           } 
  237.       } 
  238.       if ( defined( 'SAVEQUERIES' ) && SAVEQUERIES ) { 
  239.           $original_queries = is_array( $GLOBALS['wpdb']->queries ) ? array_keys( $GLOBALS['wpdb']->queries ) : array(); 
  240.       } 
  241.       $response = rest_do_request( $request ); 
  242.       if ( defined( 'SAVEQUERIES' ) && SAVEQUERIES ) { 
  243.           $performed_queries = array(); 
  244.           foreach ( (array) $GLOBALS['wpdb']->queries as $key => $query ) { 
  245.               if ( in_array( $key, $original_queries ) ) { 
  246.                   continue; 
  247.               } 
  248.               $performed_queries[] = $query; 
  249.           } 
  250.           usort( $performed_queries, function( $a, $b ) { 
  251.               if ( $a[1] === $b[1] ) { 
  252.                   return 0; 
  253.               } 
  254.               return ( $a[1] > $b[1] ) ? -1 : 1; 
  255.           }); 
  256.  
  257.           $query_count = count( $performed_queries ); 
  258.           $query_total_time = 0; 
  259.           foreach ( $performed_queries as $query ) { 
  260.               $query_total_time += $query[1]; 
  261.           } 
  262.           $slow_query_message = ''; 
  263.           if ( $performed_queries && 'wc' === WP_CLI::get_config( 'debug' ) ) { 
  264.               $slow_query_message .= '. Ordered by slowness, the queries are:' . PHP_EOL; 
  265.               foreach ( $performed_queries as $i => $query ) { 
  266.                   $i++; 
  267.                   $bits = explode( ', ', $query[2] ); 
  268.                   $backtrace = implode( ', ', array_slice( $bits, 13 ) ); 
  269.                   $seconds = round( $query[1], 6 ); 
  270.                   $slow_query_message .= <<<EOT 
  271. {$i}: 
  272. - {$seconds} seconds 
  273. - {$backtrace} 
  274. - {$query[0]} 
  275. EOT; 
  276.                   $slow_query_message .= PHP_EOL; 
  277.               } 
  278.           } elseif ( 'wc' !== WP_CLI::get_config( 'debug' ) ) { 
  279.               $slow_query_message = '. Use --debug=wc to see all queries.'; 
  280.           } 
  281.           $query_total_time = round( $query_total_time, 6 ); 
  282.           WP_CLI::debug( "wc command executed {$query_count} queries in {$query_total_time} seconds{$slow_query_message}", 'wc' ); 
  283.       } 
  284.  
  285.       if ( $error = $response->as_error() ) { 
  286.           // For authentication errors (status 401), include a reminder to set the --user flag. 
  287.           // WP_CLI::error will only return the first message from WP_Error, so we will pass a string containing both instead. 
  288.           if ( 401 === $response->get_status() ) { 
  289.               $errors = $error->get_error_messages(); 
  290.               $errors[] = __( 'Make sure to include the --user flag with an account that has permissions for this action.', 'woocommerce' ) . ' {"status":401}'; 
  291.               $error = implode( "\n", $errors ); 
  292.           } 
  293.           WP_CLI::error( $error ); 
  294.       } 
  295.       return array( $response->get_status(), $response->get_data(), $response->get_headers() ); 
  296.   }

  306.           if ( is_string( $assoc_args['fields'] ) ) { 
  307.               $fields = explode( ', ', $assoc_args['fields'] ); 
  308.           } else { 
  309.               $fields = $assoc_args['fields']; 
  310.           }

  337.   /** 
  338.    * Get the route for this resource 
  339.    * 
  340.    * @param  array $args 
  341.    * @return string 
  342.    */ 
  343.   private function get_filled_route( $args = array() ) { 
  344.       $supported_id_matched = false; 
  345.       $route = $this->route; 
  346.  
  347.       foreach ( $this->get_supported_ids() as $id_name => $id_desc ) { 
  348.           if ( strpos( $route, '<' . $id_name . '>' ) !== false && ! empty( $args ) ) { 
  349.               $route = str_replace( '(?P<' . $id_name . '>[\d]+)', $args[0], $route ); 
  350.               $supported_id_matched = true; 
  351.           } 
  352.       } 
  353.  
  354.       if ( ! empty( $args ) ) { 
  355.           $id_replacement = $supported_id_matched && ! empty( $args[1] ) ? $args[1] : $args[0]; 
  356.           $route = str_replace( array( '(?P<id>[\d]+)', '(?P<id>[\w-]+)' ), $id_replacement, $route ); 
  357.       } 
  358.  
  359.       return rtrim( $route ); 
  360.   }

  384.       if ( 'add' == $change ) {

  386.       } elseif ( 'remove' == $change ) { 
  387.           $label = '- ';

  435.   /**

  441.   protected function decode_json( $arr ) { 
  442.       foreach ( $arr as $key => $value ) { 
  443.           if ( '[' === substr( $value, 0, 1 ) || '{' === substr( $value, 0, 1 ) ) { 
  444.               $arr[ $key ] = json_decode( $value, true );
