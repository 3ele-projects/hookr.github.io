  152.    * Validate a request argument based on details registered to the route. 
  153.    * 
  154.    * @param  mixed            $value

  156.    * @param  string           $param 
  157.    * @return WP_Error|boolean 
  158.    */ 
  159.   function rest_validate_request_arg( $value, $request, $param ) { 
  160.  
  161.       $attributes = $request->get_attributes(); 
  162.       if ( ! isset( $attributes['args'][ $param ] ) || ! is_array( $attributes['args'][ $param ] ) ) { 
  163.           return true; 
  164.       } 
  165.       $args = $attributes['args'][ $param ]; 
  166.  
  167.       if ( ! empty( $args['enum'] ) ) { 
  168.           if ( ! in_array( $value, $args['enum'] ) ) { 
  169.               /** translators: 1: parameter 2: arguments */ 
  170.               return new WP_Error( 'rest_invalid_param', sprintf( __( '%1$s is not one of %2$s', 'woocommerce' ), $param, implode( ', ', $args['enum'] ) ) ); 
  171.           } 
  172.       } 
  173.  
  174.       if ( 'integer' === $args['type'] && ! is_numeric( $value ) ) { 
  175.           /** translators: 1: parameter 2: integer type */ 
  176.           return new WP_Error( 'rest_invalid_param', sprintf( __( '%1$s is not of type %2$s', 'woocommerce' ), $param, 'integer' ) ); 
  177.       } 
  178.  
  179.       if ( 'boolean' === $args['type'] && ! rest_is_boolean( $value ) ) { 
  180.           /** translators: 1: parameter 2: boolean type */ 
  181.           return new WP_Error( 'rest_invalid_param', sprintf( __( '%1$s is not of type %2$s', 'woocommerce' ), $value, 'boolean' ) ); 
  182.       } 
  183.  
  184.       if ( 'string' === $args['type'] && ! is_string( $value ) ) { 
  185.           /** translators: 1: parameter 2: string type */ 
  186.           return new WP_Error( 'rest_invalid_param', sprintf( __( '%1$s is not of type %2$s', 'woocommerce' ), $param, 'string' ) ); 
  187.       } 
  188.  
  189.       if ( isset( $args['format'] ) ) { 
  190.           switch ( $args['format'] ) { 
  191.               case 'date-time' : 
  192.                   if ( ! rest_parse_date( $value ) ) { 
  193.                       return new WP_Error( 'rest_invalid_date', __( 'The date you provided is invalid.', 'woocommerce' ) ); 
  194.                   } 
  195.                   break; 
  196.  
  197.               case 'email' : 
  198.                   if ( ! is_email( $value ) ) { 
  199.                       return new WP_Error( 'rest_invalid_email', __( 'The email address you provided is invalid.', 'woocommerce' ) ); 
  200.                   } 
  201.                   break; 
  202.               case 'ipv4' : 
  203.                   if ( ! rest_is_ip_address( $value ) ) { 
  204.                       /** translators: %s: IP address */ 
  205.                       return new WP_Error( 'rest_invalid_param', sprintf( __( '%s is not a valid IP address.', 'woocommerce' ), $value ) ); 
  206.                   } 
  207.                   break; 
  208.           } 
  209.       } 
  210.  
  211.       if ( in_array( $args['type'], array( 'numeric', 'integer' ) ) && ( isset( $args['minimum'] ) || isset( $args['maximum'] ) ) ) { 
  212.           if ( isset( $args['minimum'] ) && ! isset( $args['maximum'] ) ) { 
  213.               if ( ! empty( $args['exclusiveMinimum'] ) && $value <= $args['minimum'] ) { 
  214.                   return new WP_Error( 'rest_invalid_param', sprintf( __( '%1$s must be greater than %2$d (exclusive)', 'woocommerce' ), $param, $args['minimum'] ) ); 
  215.               } elseif ( empty( $args['exclusiveMinimum'] ) && $value < $args['minimum'] ) { 
  216.                   return new WP_Error( 'rest_invalid_param', sprintf( __( '%1$s must be greater than %2$d (inclusive)', 'woocommerce' ), $param, $args['minimum'] ) ); 
  217.               } 
  218.           } elseif ( isset( $args['maximum'] ) && ! isset( $args['minimum'] ) ) { 
  219.               if ( ! empty( $args['exclusiveMaximum'] ) && $value >= $args['maximum'] ) { 
  220.                   return new WP_Error( 'rest_invalid_param', sprintf( __( '%1$s must be less than %2$d (exclusive)', 'woocommerce' ), $param, $args['maximum'] ) ); 
  221.               } elseif ( empty( $args['exclusiveMaximum'] ) && $value > $args['maximum'] ) { 
  222.                   return new WP_Error( 'rest_invalid_param', sprintf( __( '%1$s must be less than %2$d (inclusive)', 'woocommerce' ), $param, $args['maximum'] ) ); 
  223.               } 
  224.           } elseif ( isset( $args['maximum'] ) && isset( $args['minimum'] ) ) { 
  225.               if ( ! empty( $args['exclusiveMinimum'] ) && ! empty( $args['exclusiveMaximum'] ) ) { 
  226.                   if ( $value >= $args['maximum'] || $value <= $args['minimum'] ) { 
  227.                       return new WP_Error( 'rest_invalid_param', sprintf( __( '%1$s must be between %2$d (exclusive) and %3$d (exclusive)', 'woocommerce' ), $param, $args['minimum'], $args['maximum'] ) );
