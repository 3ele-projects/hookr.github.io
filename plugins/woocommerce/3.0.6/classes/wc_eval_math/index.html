  235.

  237.             // if the token is a binary operator, pop two values off the stack, do the operation, and push the result back on 
  238.             if ( in_array( $token, array( '+', '-', '*', '/', '^' ) ) ) { 
  239.                 if ( is_null( $op2 = $stack->pop() ) ) return self::trigger( "internal error" );

  242.                     case '+': 
  243.                         $stack->push( $op1 + $op2 ); 
  244.                         break; 
  245.                     case '-': 
  246.                         $stack->push( $op1 - $op2 ); 
  247.                         break; 
  248.                     case '*': 
  249.                         $stack->push( $op1 * $op2 ); 
  250.                         break; 
  251.                     case '/': 
  252.                         if ( 0 == $op2 ) { 
  253.                             return self::trigger( 'division by zero' ); 
  254.                         } 
  255.                         $stack->push( $op1 / $op2 ); 
  256.                         break; 
  257.                     case '^': 
  258.                         $stack->push( pow( $op1, $op2 ) ); 
  259.                         break; 
  260.                 } 
  261.                 // if the token is a unary operator, pop one value off the stack, do the operation, and push it back on 
  262.             } elseif ( '_' === $token ) { 
  263.                 $stack->push( -1 * $stack->pop() ); 
  264.                 // if the token is a function, pop arguments off the stack, hand them to the function, and push the result back on 
  265.             } elseif ( ! preg_match( "/^([a-z]\w*)\($/", $token, $matches ) ) { 
  266.                 if ( is_numeric( $token ) ) { 
  267.                     $stack->push( $token ); 
  268.                 } elseif ( array_key_exists( $token, self::$v ) ) { 
  269.                     $stack->push( self::$v[ $token ] ); 
  270.                 } elseif ( array_key_exists( $token, $vars ) ) { 
  271.                     $stack->push( $vars[ $token ] ); 
  272.                 } else { 
  273.                     return self::trigger( "undefined variable '$token'" ); 
  274.                 } 
  275.             } 
  276.         } 
  277.         // when we're out of tokens, the stack should have a single element, the final result 
  278.         if ( 1 != $stack->count ) { 
  279.             return self::trigger( "internal error" ); 
  280.         } 
  281.         return $stack->pop(); 
  282.     }
