  676.     /** 
  677.      * Wrapper method to create/update order items.

  681.      * @param WC_Order $order order 
  682.      * @param string $item_type 
  683.      * @param array $posted item provided in the request body 
  684.      * @throws WC_REST_Exception If item ID is not associated with order 
  685.      */ 
  686.     protected function set_item( $order, $item_type, $posted ) { 
  687.         global $wpdb; 
  688.  
  689.         if ( ! empty( $posted['id'] ) ) { 
  690.             $action = 'update';

  692.             $action = 'create';

  694.  
  695.         $method = 'prepare_' . $item_type; 
  696.  
  697.         // Verify provided line item ID is associated with order.

  700.                 $wpdb->prepare( "SELECT * FROM {$wpdb->prefix}woocommerce_order_items WHERE order_item_id = %d AND order_id = %d",  
  701.                 absint( $posted['id'] ),
