  468.      /** 
  469.      * do payment function. 
  470.      * 
  471.      * @param WC_order $order 
  472.      * @param int $amount (default: 0) 
  473.      * @uses  Simplify_BadRequestException 
  474.      * @return bool|WP_Error 
  475.      */ 
  476.     public function do_payment( $order, $amount = 0, $token = array() ) { 
  477.         if ( $amount * 100 < 50 ) { 
  478.             return new WP_Error( 'simplify_error', __( 'Sorry, the minimum allowed order total is 0.50 to use this payment method.', 'woocommerce' ) ); 
  479.         } 
  480.  
  481.         try { 
  482.             // Charge the customer 
  483.             $data = array( 
  484.                 'amount' => $amount * 100, // In cents. 
  485.                 'description' => sprintf( __( '%1$s - Order #%2$s', 'woocommerce' ), esc_html( get_bloginfo( 'name', 'display' ) ), $order->get_order_number() ),  
  486.                 'currency' => strtoupper( get_woocommerce_currency() ),  
  487.                 'reference' => $order->get_id(),  
  488.  ); 
  489.  
  490.             $data = array_merge( $data, $token ); 
  491.             $payment = Simplify_Payment::createPayment( $data ); 
  492.  
  493.         } catch ( Exception $e ) { 
  494.  
  495.             $error_message = $e->getMessage(); 
  496.  
  497.             if ( $e instanceof Simplify_BadRequestException && $e->hasFieldErrors() && $e->getFieldErrors() ) { 
  498.                 $error_message = ''; 
  499.                 foreach ( $e->getFieldErrors() as $error ) { 
  500.                     $error_message .= ' ' . $error->getFieldName() . ': "' . $error->getMessage() . '" (' . $error->getErrorCode() . ')'; 
  501.                 } 
  502.             } 
  503.  
  504.             $order->add_order_note( sprintf( __( 'Simplify payment error: %s.', 'woocommerce' ), $error_message ) ); 
  505.  
  506.             return new WP_Error( 'simplify_payment_declined', $e->getMessage(), array( 'status' => $e->getCode() ) ); 
  507.         } 
  508.  
  509.         if ( 'APPROVED' == $payment->paymentStatus ) { 
  510.             // Payment complete 
  511.             $order->payment_complete( $payment->id ); 
  512.  
  513.             // Add order note 
  514.             $order->add_order_note( sprintf( __( 'Simplify payment approved (ID: %1$s, Auth Code: %2$s)', 'woocommerce' ), $payment->id, $payment->authCode ) ); 
  515.  
  516.             return true; 
  517.         } else { 
  518.             $order->add_order_note( __( 'Simplify payment declined', 'woocommerce' ) ); 
  519.  
  520.             return new WP_Error( 'simplify_payment_declined', __( 'Payment was declined - please try another card.', 'woocommerce' ) ); 
  521.         } 
  522.     }

  551.     /** 
  552.      * Process the payment. 
  553.      * 
  554.      * @param int $order_id 
  555.      */ 
  556.     public function process_payment( $order_id ) { 
  557.         $order = wc_get_order( $order_id ); 
  558.  
  559.         // Payment/CC form is hosted on Simplify 
  560.         if ( 'hosted' === $this->mode ) { 
  561.             return $this->process_hosted_payments( $order ); 
  562.         } 
  563.  
  564.         // New CC info was entered 
  565.         if ( isset( $_POST['simplify_token'] ) ) { 
  566.             $cart_token = wc_clean( $_POST['simplify_token'] ); 
  567.             $customer_token = $this->get_users_token(); 
  568.             $customer_token_value = ( ! is_null( $customer_token ) ? $customer_token->get_token() : '' ); 
  569.             $this->process_customer( $order, $customer_token, $cart_token ); 
  570.             return $this->process_standard_payments( $order, $cart_token, $customer_token_value ); 
  571.         } 
  572.  
  573.         // Possibly Create (or update) customer/save payment token, use an existing token, and then process the payment 
  574.         if ( isset( $_POST['wc-simplify_commerce-payment-token'] ) && 'new' !== $_POST['wc-simplify_commerce-payment-token'] ) { 
  575.             $token_id = wc_clean( $_POST['wc-simplify_commerce-payment-token'] ); 
  576.             $token = WC_Payment_Tokens::get( $token_id ); 
  577.             if ( $token->get_user_id() !== get_current_user_id() ) { 
  578.                 wc_add_notice( __( 'Please make sure your card details have been entered correctly and that your browser supports JavaScript.', 'woocommerce' ), 'error' ); 
  579.                 return; 
  580.             } 
  581.             $this->process_customer( $order, $token ); 
  582.             return $this->process_standard_payments( $order, '', $token->get_token() ); 
  583.         } 
  584.     }
