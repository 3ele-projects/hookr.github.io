  649.     /** 
  650.      * Send a test ping to the delivery URL, sent when the webhook is first created. 
  651.      * 
  652.      * @since 2.2 
  653.      * @return bool|WP_Error 
  654.      */ 
  655.     public function deliver_ping() { 
  656.         $args = array( 
  657.             'user-agent' => sprintf( 'WooCommerce/%s Hookshot (WordPress/%s)', WC_VERSION, $GLOBALS['wp_version'] ),  
  658.             'body' => "webhook_id={$this->id}",  
  659.  ); 
  660.  
  661.         $test = wp_safe_remote_post( $this->get_delivery_url(), $args ); 
  662.         $response_code = wp_remote_retrieve_response_code( $test ); 
  663.  
  664.         if ( is_wp_error( $test ) ) { 
  665.             return new WP_Error( 'error', sprintf( __( 'Error: Delivery URL cannot be reached: %s', 'woocommerce' ), $test->get_error_message() ) ); 
  666.         } 
  667.  
  668.         if ( 200 !== $response_code ) { 
  669.             return new WP_Error( 'error', sprintf( __( 'Error: Delivery URL returned response code: %s', 'woocommerce' ), absint( $response_code ) ) ); 
  670.         } 
  671.  
  672.         return true; 
  673.     }

  675.     /** 
  676.      * Get the webhook status: 
  677.      * 
  678.      * + `active` - delivers payload. 
  679.      * + `paused` - does not deliver payload, paused by admin. 
  680.      * + `disabled` - does not delivery payload, paused automatically due to. 
  681.      * consecutive failures. 
  682.      * 
  683.      * @since 2.2 
  684.      * @return string status 
  685.      */ 
  686.     public function get_status() { 
  687.  
  688.         switch ( $this->get_post_data()->post_status ) { 
  689.  
  690.             case 'publish': 
  691.                 $status = 'active'; 
  692.                 break; 
  693.  
  694.             case 'draft': 
  695.                 $status = 'paused'; 
  696.                 break; 
  697.  
  698.             case 'pending': 
  699.                 $status = 'disabled'; 
  700.                 break; 
  701.  
  702.             default: 
  703.                 $status = 'paused'; 
  704.         } 
  705.  
  706.         return apply_filters( 'woocommerce_webhook_status', $status, $this->id ); 
  707.     }
