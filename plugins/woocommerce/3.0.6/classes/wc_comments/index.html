  259.                     SELECT comment_approved, COUNT(*) AS num_comments

  261.                     WHERE comment_type NOT IN ('order_note', 'webhook_delivery') 
  262.                     GROUP BY comment_approved 
  263.                 ", ARRAY_A ); 
  264.  
  265.                 $total = 0; 
  266.                 $approved = array( 
  267.                     '0' => 'moderated',  
  268.                     '1' => 'approved',  
  269.                     'spam' => 'spam',

  271.                     'post-trashed' => 'post-trashed',  
  272.  ); 
  273.

  278.                     } 
  279.                     if ( isset( $approved[ $row['comment_approved'] ] ) ) {

  281.                     }

  285.                 $stats['all'] = $total;
