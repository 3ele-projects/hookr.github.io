  322.  
  323.         // Check if we need to recalculate shipping for this package 
  324.         $package_to_hash = $package; 
  325.  
  326.         // Remove data objects so hashes are consistent 
  327.         foreach ( $package_to_hash['contents'] as $item_id => $item ) { 
  328.             unset( $package_to_hash['contents'][ $item_id ]['data'] ); 
  329.         } 
  330.

  333.         $stored_rates = WC()->session->get( $session_key ); 
  334.

  353.  ) );
