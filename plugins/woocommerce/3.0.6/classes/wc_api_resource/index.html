  241.  
  242.                 case 'WC_Coupon': 
  243.                     $meta_name = 'coupon_meta'; 
  244.                     break; 
  245.  
  246.                 case 'WP_User': 
  247.                     $meta_name = 'customer_meta'; 
  248.                     break; 
  249.  
  250.                 default: 
  251.                     $meta_name = 'product_meta'; 
  252.                     break; 
  253.             } 
  254.  
  255.             if ( is_a( $resource, 'WP_User' ) ) { 
  256.  
  257.                 // customer meta 
  258.                 $meta = (array) get_user_meta( $resource->ID ); 
  259.  
  260.             } elseif ( is_a( $resource, 'WC_Product_Variation' ) ) { 
  261.  
  262.                 // product variation meta 
  263.                 $meta = (array) get_post_meta( $resource->get_variation_id() ); 
  264.  
  265.             } else { 
  266.  
  267.                 // coupon/order/product meta 
  268.                 $meta = (array) get_post_meta( $resource->id ); 
  269.             } 
  270.  
  271.             foreach ( $meta as $meta_key => $meta_value ) { 
  272.  
  273.                 // don't add hidden meta by default 
  274.                 if ( ! is_protected_meta( $meta_key ) ) { 
  275.                     $data[ $meta_name ][ $meta_key ] = maybe_unserialize( $meta_value[0] ); 
  276.                 } 
  277.             } 
  278.         } 
  279.  
  280.         return $data; 
  281.     }

  283.     /** 
  284.      * Restrict the fields included in the response if the request specified certain only certain fields should be returned 
  285.      * 
  286.      * @since 2.1 
  287.      * @param array $data the response data 
  288.      * @param object $resource the object that provided the response data, e.g. WC_Coupon or WC_Order 
  289.      * @param array|string the requested list of fields to include in the response 
  290.      * @return array response data 
  291.      */ 
  292.     public function filter_response_fields( $data, $resource, $fields ) { 
  293.  
  294.         if ( ! is_array( $data ) || empty( $fields ) ) { 
  295.             return $data; 
  296.         } 
  297.  
  298.         $fields = explode( ', ', $fields ); 
  299.         $sub_fields = array(); 
  300.  
  301.         // get sub fields 
  302.         foreach ( $fields as $field ) { 
  303.  
  304.             if ( false !== strpos( $field, '.' ) ) { 
  305.  
  306.                 list( $name, $value ) = explode( '.', $field ); 
  307.  
  308.                 $sub_fields[ $name ] = $value; 
  309.             } 
  310.         } 
  311.  
  312.         // iterate through top-level fields 
  313.         foreach ( $data as $data_field => $data_value ) { 
  314.  
  315.             // if a field has sub-fields and the top-level field has sub-fields to filter 
  316.             if ( is_array( $data_value ) && in_array( $data_field, array_keys( $sub_fields ) ) ) { 
  317.  
  318.                 // iterate through each sub-field 
  319.                 foreach ( $data_value as $sub_field => $sub_field_value ) { 
  320.  
  321.                     // remove non-matching sub-fields 
  322.                     if ( ! in_array( $sub_field, $sub_fields ) ) { 
  323.                         unset( $data[ $data_field ][ $sub_field ] ); 
  324.                     } 
  325.                 } 
  326.             } else { 
  327.  
  328.                 // remove non-matching top-level fields 
  329.                 if ( ! in_array( $data_field, $fields ) ) { 
  330.                     unset( $data[ $data_field ] ); 
  331.                 } 
  332.             } 
  333.         } 
  334.  
  335.         return $data; 
  336.     }

  338.     /** 
  339.      * Delete a given resource 
  340.      * 
  341.      * @since 2.1 
  342.      * @param int $id the resource ID 
  343.      * @param string $type the resource post type, or `customer` 
  344.      * @param bool $force true to permanently delete resource, false to move to trash (not supported for `customer`) 
  345.      * @return array|WP_Error 
  346.      */ 
  347.     protected function delete( $id, $type, $force = false ) { 
  348.  
  349.         if ( 'shop_order' === $type || 'shop_coupon' === $type ) { 
  350.             $resource_name = str_replace( 'shop_', '', $type ); 
  351.         } else { 
  352.             $resource_name = $type; 
  353.         } 
  354.  
  355.         if ( 'customer' === $type ) { 
  356.  
  357.             $result = wp_delete_user( $id ); 
  358.  
  359.             if ( $result ) 
  360.                 return array( 'message' => __( 'Permanently deleted customer', 'woocommerce' ) ); 
  361.             else 
  362.                 return new WP_Error( 'woocommerce_api_cannot_delete_customer', __( 'The customer cannot be deleted', 'woocommerce' ), array( 'status' => 500 ) ); 
  363.  
  364.         } else { 
  365.  
  366.             // delete order/coupon/product/webhook 
  367.             $result = ( $force ) ? wp_delete_post( $id, true ) : wp_trash_post( $id ); 
  368.  
  369.             if ( ! $result ) 
  370.                 return new WP_Error( "woocommerce_api_cannot_delete_{$resource_name}", sprintf( __( 'This %s cannot be deleted', 'woocommerce' ), $resource_name ), array( 'status' => 500 ) ); 
  371.  
  372.             if ( $force ) { 
  373.                 return array( 'message' => sprintf( __( 'Permanently deleted %s', 'woocommerce' ), $resource_name ) ); 
  374.  
  375.             } else { 
  376.  
  377.                 $this->server->send_status( '202' ); 
  378.  
  379.                 return array( 'message' => sprintf( __( 'Deleted %s', 'woocommerce' ), $resource_name ) ); 
  380.             } 
  381.         } 
  382.     }

  425.     /** 
  426.      * Checks the permissions for the current user given a post and context 
  427.      * 
  428.      * @since 2.1 
  429.      * @param WP_Post|int $post 
  430.      * @param string $context the type of permission to check, either `read`, `write`, or `delete` 
  431.      * @return bool true if the current user has the permissions to perform the context on the post 
  432.      */ 
  433.     private function check_permission( $post, $context ) { 
  434.         $permission = false; 
  435.  
  436.         if ( ! is_a( $post, 'WP_Post' ) ) { 
  437.             $post = get_post( $post ); 
  438.         } 
  439.  
  440.         if ( is_null( $post ) ) { 
  441.             return $permission; 
  442.         } 
  443.  
  444.         $post_type = get_post_type_object( $post->post_type ); 
  445.  
  446.         if ( 'read' === $context ) { 
  447.             $permission = 'revision' !== $post->post_type && current_user_can( $post_type->cap->read_private_posts, $post->ID ); 
  448.         } elseif ( 'edit' === $context ) { 
  449.             $permission = current_user_can( $post_type->cap->edit_post, $post->ID ); 
  450.         } elseif ( 'delete' === $context ) { 
  451.             $permission = current_user_can( $post_type->cap->delete_post, $post->ID ); 
  452.         } 
  453.  
  454.         return apply_filters( 'woocommerce_api_check_permission', $permission, $context, $post, $post_type ); 
  455.     }
