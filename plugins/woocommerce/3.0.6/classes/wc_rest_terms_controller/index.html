  523.         $taxonomy = $this->get_taxonomy( $request ); 
  524.         $force = isset( $request['force'] ) ? (bool) $request['force'] : false; 
  525.  
  526.         // We don't support trashing for this type, error out. 
  527.         if ( ! $force ) { 
  528.             return new WP_Error( 'woocommerce_rest_trash_not_supported', __( 'Resource does not support trashing.', 'woocommerce' ), array( 'status' => 501 ) ); 
  529.         } 
  530.  
  531.         $term = get_term( (int) $request['id'], $taxonomy ); 
  532.         $request->set_param( 'context', 'edit' ); 
  533.         $response = $this->prepare_item_for_response( $term, $request ); 
  534.  
  535.         $retval = wp_delete_term( $term->term_id, $term->taxonomy ); 
  536.         if ( ! $retval ) { 
  537.             return new WP_Error( 'woocommerce_rest_cannot_delete', __( 'The resource cannot be deleted.', 'woocommerce' ), array( 'status' => 500 ) ); 
  538.         }

  547.         do_action( "woocommerce_rest_delete_{$taxonomy}", $term, $response, $request );

  549.         return $response;

  598.     /** 
  599.      * Get the terms attached to a product. 
  600.      * 
  601.      * This is an alternative to `get_terms()` that uses `get_the_terms()` 
  602.      * instead, which hits the object cache. There are a few things not 
  603.      * supported, notably `include`, `exclude`. In `self::get_items()` these 
  604.      * are instead treated as a full query. 
  605.      * 
  606.      * @param array $prepared_args Arguments for `get_terms()`. 
  607.      * @return array List of term objects. (Total count in `$this->total_terms`). 
  608.      */ 
  609.     protected function get_terms_for_product( $prepared_args ) { 
  610.         $taxonomy = $this->get_taxonomy( $request ); 
  611.  
  612.         $query_result = get_the_terms( $prepared_args['product'], $taxonomy ); 
  613.         if ( empty( $query_result ) ) { 
  614.             $this->total_terms = 0; 
  615.             return array(); 
  616.         } 
  617.  
  618.         // get_items() verifies that we don't have `include` set, and default. 
  619.         // ordering is by `name`. 
  620.         if ( ! in_array( $prepared_args['orderby'], array( 'name', 'none', 'include' ) ) ) { 
  621.             switch ( $prepared_args['orderby'] ) { 
  622.                 case 'id' : 
  623.                     $this->sort_column = 'term_id'; 
  624.                     break; 
  625.  
  626.                 case 'slug' : 
  627.                 case 'term_group' : 
  628.                 case 'description' : 
  629.                 case 'count' : 
  630.                     $this->sort_column = $prepared_args['orderby']; 
  631.                     break; 
  632.             } 
  633.             usort( $query_result, array( $this, 'compare_terms' ) ); 
  634.         } 
  635.         if ( strtolower( $prepared_args['order'] ) !== 'asc' ) { 
  636.             $query_result = array_reverse( $query_result ); 
  637.         } 
  638.  
  639.         // Pagination. 
  640.         $this->total_terms = count( $query_result ); 
  641.         $query_result = array_slice( $query_result, $prepared_args['offset'], $prepared_args['number'] ); 
  642.  
  643.         return $query_result; 
  644.     }

  646.     /** 
  647.      * Comparison function for sorting terms by a column. 
  648.      * 
  649.      * Uses `$this->sort_column` to determine field to sort by. 
  650.      * 
  651.      * @param stdClass $left Term object. 
  652.      * @param stdClass $right Term object. 
  653.      * @return int <0 if left is higher "priority" than right, 0 if equal, >0 if right is higher "priority" than left. 
  654.      */ 
  655.     protected function compare_terms( $left, $right ) { 
  656.         $col = $this->sort_column; 
  657.         $left_val = $left->$col; 
  658.         $right_val = $right->$col; 
  659.  
  660.         if ( is_int( $left_val ) && is_int( $right_val ) ) { 
  661.             return $left_val - $right_val; 
  662.         } 
  663.  
  664.         return strcmp( $left_val, $right_val ); 
  665.     }

  770.      * @return int|WP_Error 
  771.      */ 
  772.     protected function get_taxonomy( $request ) { 
  773.         // Check if taxonomy is defined. 
  774.         // Prevents check for attribute taxonomy more than one time for each query. 
  775.         if ( '' !== $this->taxonomy ) {
