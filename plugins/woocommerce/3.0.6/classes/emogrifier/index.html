  332.             foreach ( $matches as $key => $cssRule ) {

  334.                 if ( $cssDeclaration === '' ) { 
  335.                     continue; 
  336.                 }

  339.                 foreach ( $selectors as $selector ) {

  916.     private function sortBySelectorPrecedence( array $a, array $b ) { 
  917.         $precedenceA = $this->getCssSelectorPrecedence($a['selector']); 
  918.         $precedenceB = $this->getCssSelectorPrecedence($b['selector']);

  1116.     /** 
  1117.      * @param string[] $match 
  1118.      * 
  1119.      * @return int[] 
  1120.      */ 
  1121.     private function parseNth( array $match ) { 
  1122.         if ( in_array(strtolower($match[2]), array( 'even', 'odd' ), true) ) { 
  1123.             // we have "even" or "odd" 
  1124.             $index = strtolower($match[2]) === 'even' ? 0 : 1; 
  1125.             return array( self::MULTIPLIER => 2, self::INDEX => $index ); 
  1126.         } 
  1127.         if ( stripos($match[2], 'n') === false ) { 
  1128.             // if there is a multiplier 
  1129.             $index = (int) str_replace(' ', '', $match[2]); 
  1130.             return array( self::INDEX => $index ); 
  1131.         } 
  1132.  
  1133.         if ( isset($match[3]) ) { 
  1134.             $multipleTerm = str_replace($match[3], '', $match[2]); 
  1135.             $index = (int) str_replace(' ', '', $match[3]); 
  1136.         } else { 
  1137.             $multipleTerm = $match[2]; 
  1138.             $index = 0; 
  1139.         } 
  1140.  
  1141.         $multiplier = str_ireplace('n', '', $multipleTerm); 
  1142.  
  1143.         if ( $multiplier === '' ) { 
  1144.             $multiplier = 1; 
  1145.         } elseif ( $multiplier === '0' ) { 
  1146.             return array( self::INDEX => $index ); 
  1147.         } else { 
  1148.             $multiplier = (int) $multiplier; 
  1149.         } 
  1150.  
  1151.         while ( $index < 0 ) { 
  1152.             $index += abs($multiplier); 
  1153.         } 
  1154.  
  1155.         return array( self::MULTIPLIER => $multiplier, self::INDEX => $index ); 
  1156.     }
