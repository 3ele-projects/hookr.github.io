  532.         } 
  533.  
  534.         /**

  538.         if ( version_compare( get_post_meta( $product->get_id(), '_product_version', true ), '2.4.0', '<' ) ) { 
  539.             $parent_attributes = array_filter( (array) get_post_meta( $product->get_id(), '_product_attributes', true ) );

  541.             if ( ! $children ) {

  543.             } 
  544.  
  545.             foreach ( $children as $child_id ) { 
  546.                 $all_meta = get_post_meta( $child_id ); 
  547.  
  548.                 foreach ( $all_meta as $name => $value ) { 
  549.                     if ( 0 !== strpos( $name, 'attribute_' ) ) { 
  550.                         continue; 
  551.                     } 
  552.                     if ( sanitize_title( $value[0] ) === $value[0] ) { 
  553.                         foreach ( $parent_attributes as $attribute ) { 
  554.                             if ( 'attribute_' . sanitize_title( $attribute['name'] ) !== $name ) { 
  555.                                 continue; 
  556.                             } 
  557.                             $text_attributes = wc_get_text_attributes( $attribute['value'] ); 
  558.                             foreach ( $text_attributes as $text_attribute ) { 
  559.                                 if ( sanitize_title( $text_attribute ) === $value[0] ) { 
  560.                                     update_post_meta( $child_id, $name, $text_attribute );
