  642.     /** 
  643.      * Show a single product page. 
  644.      * 
  645.      * @param array $atts 
  646.      * @return string 
  647.      */ 
  648.     public static function product_page( $atts ) { 
  649.         if ( empty( $atts ) ) { 
  650.             return ''; 
  651.         } 
  652.  
  653.         if ( ! isset( $atts['id'] ) && ! isset( $atts['sku'] ) ) { 
  654.             return ''; 
  655.         } 
  656.  
  657.         $args = array( 
  658.             'posts_per_page' => 1,  
  659.             'post_type' => 'product',  
  660.             'post_status' => 'publish',  
  661.             'ignore_sticky_posts' => 1,  
  662.             'no_found_rows' => 1,  
  663.  ); 
  664.  
  665.         if ( isset( $atts['sku'] ) ) { 
  666.             $args['meta_query'][] = array( 
  667.                 'key' => '_sku',  
  668.                 'value' => sanitize_text_field( $atts['sku'] ),  
  669.                 'compare' => '=',  
  670.  ); 
  671.  
  672.             $args['post_type'] = array( 'product', 'product_variation' ); 
  673.         } 
  674.  
  675.         if ( isset( $atts['id'] ) ) { 
  676.             $args['p'] = absint( $atts['id'] ); 
  677.         } 
  678.  
  679.         $single_product = new WP_Query( $args ); 
  680.  
  681.         $preselected_id = '0'; 
  682.  
  683.         // check if sku is a variation 
  684.         if ( isset( $atts['sku'] ) && $single_product->have_posts() && 'product_variation' === $single_product->post->post_type ) { 
  685.  
  686.             $variation = new WC_Product_Variation( $single_product->post->ID ); 
  687.             $attributes = $variation->get_attributes(); 
  688.  
  689.             // set preselected id to be used by JS to provide context 
  690.             $preselected_id = $single_product->post->ID; 
  691.  
  692.             // get the parent product object 
  693.             $args = array( 
  694.                 'posts_per_page' => 1,  
  695.                 'post_type' => 'product',  
  696.                 'post_status' => 'publish',  
  697.                 'ignore_sticky_posts' => 1,  
  698.                 'no_found_rows' => 1,  
  699.                 'p' => $single_product->post->post_parent,  
  700.  ); 
  701.  
  702.             $single_product = new WP_Query( $args ); 
  703.         ?> 
  704.             <script type="text/javascript"> 
  705.                 jQuery( document ).ready( function( $ ) { 
  706.                     var $variations_form = $( '[data-product-page-preselected-id="<?php echo esc_attr( $preselected_id ); ?>"]' ).find( 'form.variations_form' ); 
  707.  
  708.                     <?php foreach ( $attributes as $attr => $value ) { ?> 
  709.                         $variations_form.find( 'select[name="<?php echo esc_attr( $attr ); ?>"]' ).val( '<?php echo esc_js( $value ); ?>' ); 
  710.                     <?php } ?> 
  711.                 }); 
  712.             </script> 
  713.         <?php 
  714.         } 
  715.  
  716.         ob_start(); 
  717.  
  718.         while ( $single_product->have_posts() ) : 
  719.             $single_product->the_post(); 
  720.             wp_enqueue_script( 'wc-single-product' ); 
  721.             ?> 
  722.  
  723.             <div class="single-product" data-product-page-preselected-id="<?php echo esc_attr( $preselected_id ); ?>"> 
  724.  
  725.                 <?php wc_get_template_part( 'content', 'single-product' ); ?> 
  726.  
  727.             </div> 
  728.  
  729.         <?php endwhile; // end of the loop. 
  730.  
  731.         wp_reset_postdata(); 
  732.  
  733.         return '<div class="woocommerce">' . ob_get_clean() . '</div>'; 
  734.     }
