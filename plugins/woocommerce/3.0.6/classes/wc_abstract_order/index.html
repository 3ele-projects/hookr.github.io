  1070.      * Update tax lines for the order based on the line item taxes themselves. 
  1071.      */ 
  1072.     public function update_taxes() { 
  1073.         $cart_taxes = array(); 
  1074.         $shipping_taxes = array(); 
  1075.         $existing_taxes = $this->get_taxes(); 
  1076.         $saved_rate_ids = array(); 
  1077.  
  1078.         foreach ( $this->get_items( array( 'line_item', 'fee' ) ) as $item_id => $item ) { 
  1079.             $taxes = $item->get_taxes(); 
  1080.             foreach ( $taxes['total'] as $tax_rate_id => $tax ) { 
  1081.                 $cart_taxes[ $tax_rate_id ] = isset( $cart_taxes[ $tax_rate_id ] ) ? $cart_taxes[ $tax_rate_id ] + (float) $tax : (float) $tax;

  1083.         } 
  1084.  
  1085.         foreach ( $this->get_shipping_methods() as $item_id => $item ) { 
  1086.             $taxes = $item->get_taxes(); 
  1087.             foreach ( $taxes['total'] as $tax_rate_id => $tax ) { 
  1088.                 $shipping_taxes[ $tax_rate_id ] = isset( $shipping_taxes[ $tax_rate_id ] ) ? $shipping_taxes[ $tax_rate_id ] + (float) $tax : (float) $tax; 
  1089.             } 
  1090.         } 
  1091.  
  1092.         foreach ( $existing_taxes as $tax ) { 
  1093.             // Remove taxes which no longer exist for cart/shipping. 
  1094.             if ( ( ! array_key_exists( $tax->get_rate_id(), $cart_taxes ) && ! array_key_exists( $tax->get_rate_id(), $shipping_taxes ) ) || in_array( $tax->get_rate_id(), $saved_rate_ids ) ) { 
  1095.                 $this->remove_item( $tax->get_id() ); 
  1096.                 continue; 
  1097.             } 
  1098.             $saved_rate_ids[] = $tax->get_rate_id(); 
  1099.             $tax->set_tax_total( isset( $cart_taxes[ $tax->get_rate_id() ] ) ? $cart_taxes[ $tax->get_rate_id() ] : 0 ); 
  1100.             $tax->set_shipping_tax_total( ! empty( $shipping_taxes[ $tax->get_rate_id() ] ) ? $shipping_taxes[ $tax->get_rate_id() ] : 0 ); 
  1101.             $tax->save(); 
  1102.         } 
  1103.  
  1104.         $new_rate_ids = wp_parse_id_list( array_diff( array_keys( $cart_taxes + $shipping_taxes ), $saved_rate_ids ) ); 
  1105.  
  1106.         // New taxes. 
  1107.         foreach ( $new_rate_ids as $tax_rate_id ) { 
  1108.             $item = new WC_Order_Item_Tax(); 
  1109.             $item->set_rate( $tax_rate_id ); 
  1110.             $item->set_tax_total( isset( $cart_taxes[ $tax_rate_id ] ) ? $cart_taxes[ $tax_rate_id ] : 0 ); 
  1111.             $item->set_shipping_tax_total( ! empty( $shipping_taxes[ $tax_rate_id ] ) ? $shipping_taxes[ $tax_rate_id ] : 0 ); 
  1112.             $this->add_item( $item ); 
  1113.         } 
  1114.  
  1115.         // Save tax totals 
  1116.         $this->set_shipping_tax( WC_Tax::round( array_sum( $shipping_taxes ) ) ); 
  1117.         $this->set_cart_tax( WC_Tax::round( array_sum( $cart_taxes ) ) ); 
  1118.         $this->save(); 
  1119.     }
