  206.      * Verify that the consumer-provided request signature matches our generated signature, this ensures the consumer 
  207.      * has a valid key/secret

  210.      * @param array $params the request parameters

  213.     private function check_oauth_signature( $keys, $params ) { 
  214.  
  215.         $http_method = strtoupper( WC()->api->server->method ); 
  216.  
  217.         $base_request_uri = rawurlencode( untrailingslashit( get_woocommerce_api_url( '' ) ) . WC()->api->server->path ); 
  218.  
  219.         // Get the signature provided by the consumer and remove it from the parameters prior to checking the signature 
  220.         $consumer_signature = rawurldecode( str_replace( ' ', '+', $params['oauth_signature'] ) ); 
  221.         unset( $params['oauth_signature'] ); 
  222.  
  223.         // Remove filters and convert them from array to strings to void normalize issues 
  224.         if ( isset( $params['filter'] ) ) { 
  225.             $filters = $params['filter']; 
  226.             unset( $params['filter'] ); 
  227.             foreach ( $filters as $filter => $filter_value ) { 
  228.                 $params[ 'filter[' . $filter . ']' ] = $filter_value; 
  229.             } 
  230.         } 
  231.  
  232.         // Normalize parameter key/values 
  233.         $params = $this->normalize_parameters( $params ); 
  234.  
  235.         // Sort parameters 
  236.         if ( ! uksort( $params, 'strcmp' ) ) { 
  237.             throw new Exception( __( 'Invalid signature - failed to sort parameters.', 'woocommerce' ), 401 ); 
  238.         } 
  239.  
  240.         // Form query string 
  241.         $query_params = array(); 
  242.         foreach ( $params as $param_key => $param_value ) { 
  243.  
  244.             $query_params[] = $param_key . '%3D' . $param_value; // join with equals sign 
  245.         } 
  246.         $query_string = implode( '%26', $query_params ); // join with ampersand 
  247.  
  248.         $string_to_sign = $http_method . '&' . $base_request_uri . '&' . $query_string; 
  249.  
  250.         if ( 'HMAC-SHA1' !== $params['oauth_signature_method'] && 'HMAC-SHA256' !== $params['oauth_signature_method'] ) { 
  251.             throw new Exception( __( 'Invalid signature - signature method is invalid.', 'woocommerce' ), 401 ); 
  252.         } 
  253.  
  254.         $hash_algorithm = strtolower( str_replace( 'HMAC-', '', $params['oauth_signature_method'] ) ); 
  255.  
  256.         $signature = base64_encode( hash_hmac( $hash_algorithm, $string_to_sign, $keys['consumer_secret'], true ) ); 
  257.  
  258.         if ( ! hash_equals( $signature, $consumer_signature ) ) { 
  259.             throw new Exception( __( 'Invalid signature - provided signature does not match.', 'woocommerce' ), 401 ); 
  260.         } 
  261.     }

  263.     /** 
  264.      * Normalize each parameter by assuming each parameter may have already been 
  265.      * encoded, so attempt to decode, and then re-encode according to RFC 3986 
  266.      * 
  267.      * Note both the key and value is normalized so a filter param like: 
  268.      * 
  269.      * 'filter[period]' => 'week' 
  270.      * 
  271.      * is encoded to: 
  272.      * 
  273.      * 'filter%5Bperiod%5D' => 'week' 
  274.      * 
  275.      * This conforms to the OAuth 1.0a spec which indicates the entire query string 
  276.      * should be URL encoded 
  277.      * 
  278.      * @since 2.1 
  279.      * @see rawurlencode() 
  280.      * @param array $parameters un-normalized pararmeters 
  281.      * @return array normalized parameters 
  282.      */ 
  283.     private function normalize_parameters( $parameters ) { 
  284.  
  285.         $normalized_parameters = array(); 
  286.  
  287.         foreach ( $parameters as $key => $value ) { 
  288.  
  289.             // Percent symbols (%) must be double-encoded 
  290.             $key = str_replace( '%', '%25', rawurlencode( rawurldecode( $key ) ) ); 
  291.             $value = str_replace( '%', '%25', rawurlencode( rawurldecode( $value ) ) ); 
  292.  
  293.             $normalized_parameters[ $key ] = $value; 
  294.         } 
  295.  
  296.         return $normalized_parameters; 
  297.     }

  299.     /** 
  300.      * Verify that the timestamp and nonce provided with the request are valid. This prevents replay attacks where 
  301.      * an attacker could attempt to re-send an intercepted request at a later time. 
  302.      * 
  303.      * - A timestamp is valid if it is within 15 minutes of now 
  304.      * - A nonce is valid if it has not been used within the last 15 minutes 
  305.      * 
  306.      * @param array $keys 
  307.      * @param int $timestamp the unix timestamp for when the request was made 
  308.      * @param string $nonce a unique (for the given user) 32 alphanumeric string, consumer-generated 
  309.      * @throws Exception 
  310.      */ 
  311.     private function check_oauth_timestamp_and_nonce( $keys, $timestamp, $nonce ) { 
  312.         global $wpdb; 
  313.  
  314.         $valid_window = 15 * 60; // 15 minute window 
  315.  
  316.         if ( ( $timestamp < time() - $valid_window ) || ( $timestamp > time() + $valid_window ) ) { 
  317.             throw new Exception( __( 'Invalid timestamp.', 'woocommerce' ) ); 
  318.         } 
  319.  
  320.         $used_nonces = maybe_unserialize( $keys['nonces'] ); 
  321.  
  322.         if ( empty( $used_nonces ) ) { 
  323.             $used_nonces = array(); 
  324.         } 
  325.  
  326.         if ( in_array( $nonce, $used_nonces ) ) { 
  327.             throw new Exception( __( 'Invalid nonce - nonce has already been used.', 'woocommerce' ), 401 ); 
  328.         } 
  329.  
  330.         $used_nonces[ $timestamp ] = $nonce; 
  331.  
  332.         // Remove expired nonces 
  333.         foreach ( $used_nonces as $nonce_timestamp => $nonce ) { 
  334.             if ( $nonce_timestamp < ( time() - $valid_window ) ) { 
  335.                 unset( $used_nonces[ $nonce_timestamp ] ); 
  336.             } 
  337.         } 
  338.  
  339.         $used_nonces = maybe_serialize( $used_nonces ); 
  340.  
  341.         $wpdb->update( 
  342.             $wpdb->prefix . 'woocommerce_api_keys',  
  343.             array( 'nonces' => $used_nonces ),  
  344.             array( 'key_id' => $keys['key_id'] ),  
  345.             array( '%s' ),  
  346.             array( '%d' ) 
  347.  ); 
  348.     }

  357.         switch ( WC()->api->server->method ) {

  359.             case 'HEAD':

  371.                     throw new Exception( __( 'The API key provided does not have write permissions.', 'woocommerce' ), 401 ); 
  372.                 } 
  373.                 break; 
  374.         } 
  375.     }

  213.  
  214.         $http_method = strtoupper( WC()->api->server->method ); 
  215.  
  216.         $base_request_uri = rawurlencode( untrailingslashit( get_woocommerce_api_url( '' ) ) . WC()->api->server->path ); 
  217.

  225.             unset( $params['filter'] ); 
  226.             foreach ( $filters as $filter => $filter_value ) {

  236.             throw new Exception( __( 'Invalid signature - failed to sort parameters.', 'woocommerce' ), 401 );

  241.         foreach ( $params as $param_key => $param_value ) {

  244.         }

  246.

  248.  
  249.         if ( 'HMAC-SHA1' !== $params['oauth_signature_method'] && 'HMAC-SHA256' !== $params['oauth_signature_method'] ) {

  254.  
  255.         $signature = base64_encode( hash_hmac( $hash_algorithm, $string_to_sign, $keys['consumer_secret'], true ) );
