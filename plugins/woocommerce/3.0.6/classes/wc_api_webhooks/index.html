  142.     /** 
  143.      * Create an webhook 
  144.      * 
  145.      * @since 2.2 
  146.      * @param array $data parsed webhook data 
  147.      * @return array 
  148.      */ 
  149.     public function create_webhook( $data ) { 
  150.  
  151.         try { 
  152.             if ( ! isset( $data['webhook'] ) ) { 
  153.                 throw new WC_API_Exception( 'woocommerce_api_missing_webhook_data', sprintf( __( 'No %1$s data specified to create %1$s', 'woocommerce' ), 'webhook' ), 400 ); 
  154.             } 
  155.  
  156.             $data = $data['webhook']; 
  157.  
  158.             // permission check 
  159.             if ( ! current_user_can( 'publish_shop_webhooks' ) ) { 
  160.                 throw new WC_API_Exception( 'woocommerce_api_user_cannot_create_webhooks', __( 'You do not have permission to create webhooks.', 'woocommerce' ), 401 ); 
  161.             } 
  162.  
  163.             $data = apply_filters( 'woocommerce_api_create_webhook_data', $data, $this ); 
  164.  
  165.             // validate topic 
  166.             if ( empty( $data['topic'] ) || ! wc_is_webhook_valid_topic( strtolower( $data['topic'] ) ) ) { 
  167.                 throw new WC_API_Exception( 'woocommerce_api_invalid_webhook_topic', __( 'Webhook topic is required and must be valid.', 'woocommerce' ), 400 ); 
  168.             } 
  169.  
  170.             // validate delivery URL 
  171.             if ( empty( $data['delivery_url'] ) || ! wc_is_valid_url( $data['delivery_url'] ) ) { 
  172.                 throw new WC_API_Exception( 'woocommerce_api_invalid_webhook_delivery_url', __( 'Webhook delivery URL must be a valid URL starting with http:// or https://', 'woocommerce' ), 400 ); 
  173.             } 
  174.  
  175.             $webhook_data = apply_filters( 'woocommerce_new_webhook_data', array( 
  176.                 'post_type' => 'shop_webhook',  
  177.                 'post_status' => 'publish',  
  178.                 'ping_status' => 'closed',  
  179.                 'post_author' => get_current_user_id(),  
  180.                 'post_password' => strlen( ( $password = uniqid( 'webhook_' ) ) ) > 20 ? substr( $password, 0, 20 ) : $password,  
  181.                 // @codingStandardsIgnoreStart 
  182.                 'post_title' => ! empty( $data['name'] ) ? $data['name'] : sprintf( __( 'Webhook created on %s', 'woocommerce' ), strftime( _x( '%b %d, %Y @ %I:%M %p', 'Webhook created on date parsed by strftime', 'woocommerce' ) ) ),  
  183.                 // @codingStandardsIgnoreEnd 
  184.  ), $data, $this ); 
  185.  
  186.             $webhook_id = wp_insert_post( $webhook_data ); 
  187.  
  188.             if ( is_wp_error( $webhook_id ) || ! $webhook_id ) { 
  189.                 throw new WC_API_Exception( 'woocommerce_api_cannot_create_webhook', sprintf( __( 'Cannot create webhook: %s', 'woocommerce' ), is_wp_error( $webhook_id ) ? implode( ', ', $webhook_id->get_error_messages() ) : '0' ), 500 ); 
  190.             } 
  191.  
  192.             $webhook = new WC_Webhook( $webhook_id ); 
  193.  
  194.             // set topic, delivery URL, and optional secret 
  195.             $webhook->set_topic( $data['topic'] ); 
  196.             $webhook->set_delivery_url( $data['delivery_url'] ); 
  197.  
  198.             // set secret if provided, defaults to API users consumer secret 
  199.             $webhook->set_secret( ! empty( $data['secret'] ) ? $data['secret'] : '' ); 
  200.  
  201.             // Set API version to legacy v3. 
  202.             $webhook->set_api_version( 'legacy_v3' ); 
  203.  
  204.             // send ping 
  205.             $webhook->deliver_ping(); 
  206.  
  207.             // HTTP 201 Created 
  208.             $this->server->send_status( 201 ); 
  209.  
  210.             do_action( 'woocommerce_api_create_webhook', $webhook->id, $this ); 
  211.  
  212.             delete_transient( 'woocommerce_webhook_ids' ); 
  213.  
  214.             return $this->get_webhook( $webhook->id ); 
  215.  
  216.         } catch ( WC_API_Exception $e ) { 
  217.  
  218.             return new WP_Error( $e->getErrorCode(), $e->getMessage(), array( 'status' => $e->getCode() ) ); 
  219.         } 
  220.     }

  231.  
  232.         try { 
  233.             if ( ! isset( $data['webhook'] ) ) { 
  234.                 throw new WC_API_Exception( 'woocommerce_api_missing_webhook_data', sprintf( __( 'No %1$s data specified to edit %1$s', 'woocommerce' ), 'webhook' ), 400 );

  237.             $data = $data['webhook'];

  249.             // update topic
