  531.         if ( isset( $request['last_name'] ) ) { 
  532.             update_user_meta( $customer->ID, 'last_name', wc_clean( $request['last_name'] ) ); 
  533.         } 
  534.  
  535.         // Customer billing address. 
  536.         if ( isset( $request['billing'] ) ) { 
  537.             foreach ( array_keys( $schema['properties']['billing']['properties'] ) as $address ) { 
  538.                 if ( isset( $request['billing'][ $address ] ) ) { 
  539.                     update_user_meta( $customer->ID, 'billing_' . $address, wc_clean( $request['billing'][ $address ] ) ); 
  540.                 } 
  541.             } 
  542.         } 
  543.  
  544.         // Customer shipping address. 
  545.         if ( isset( $request['shipping'] ) ) {

  550.             } 
  551.         } 
  552.     }
