  122.   public function query_args( $args, $request ) { 
  123.       // Set post_status. 
  124.       $args['post_status'] = $request['status']; 
  125.  
  126.       // Taxonomy query to filter products by type, category,  
  127.       // tag, shipping class, and attribute. 
  128.       $tax_query = array(); 
  129.  
  130.       // Map between taxonomy name and arg's key. 
  131.       $taxonomies = array( 
  132.           'product_cat' => 'category',  
  133.           'product_tag' => 'tag',  
  134.           'product_shipping_class' => 'shipping_class',  
  135.  ); 
  136.  
  137.       // Set tax_query for each passed arg. 
  138.       foreach ( $taxonomies as $taxonomy => $key ) { 
  139.           if ( ! empty( $request[ $key ] ) ) { 
  140.               $terms = explode( ', ', $request[ $key ] ); 
  141.  
  142.               $tax_query[] = array( 
  143.                   'taxonomy' => $taxonomy,  
  144.                   'field' => 'term_id',  
  145.                   'terms' => $terms,  
  146.  ); 
  147.           } 
  148.       } 
  149.  
  150.       // Filter product type by slug. 
  151.       if ( ! empty( $request['type'] ) ) { 
  152.           $terms = explode( ', ', $request['type'] ); 
  153.  
  154.           $tax_query[] = array( 
  155.               'taxonomy' => 'product_type',  
  156.               'field' => 'slug',  
  157.               'terms' => $terms,  
  158.  ); 
  159.       } 
  160.  
  161.       // Filter by attribute and term. 
  162.       if ( ! empty( $request['attribute'] ) && ! empty( $request['attribute_term'] ) ) { 
  163.           if ( in_array( $request['attribute'], wc_get_attribute_taxonomy_names() ) ) { 
  164.               $terms = explode( ', ', $request['attribute_term'] ); 
  165.  
  166.               $tax_query[] = array( 
  167.                   'taxonomy' => $request['attribute'],  
  168.                   'field' => 'term_id',  
  169.                   'terms' => $terms,  
  170.  ); 
  171.           } 
  172.       } 
  173.  
  174.       if ( ! empty( $tax_query ) ) { 
  175.           $args['tax_query'] = $tax_query; 
  176.       } 
  177.  
  178.       // Filter by sku. 
  179.       if ( ! empty( $request['sku'] ) ) { 
  180.           if ( ! empty( $args['meta_query'] ) ) { 
  181.               $args['meta_query'] = array(); 
  182.           } 
  183.  
  184.           $args['meta_query'][] = array( 
  185.               'key' => '_sku',  
  186.               'value' => $request['sku'],  
  187.               'compare' => '=' 
  188.  ); 
  189.  
  190.           $args['post_type'] = array( 'product', 'product_variation' ); 
  191.       } 
  192.  
  193.       return $args; 
  194.   }
