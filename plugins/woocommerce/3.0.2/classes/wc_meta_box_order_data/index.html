  463.                 if ( ! isset( $_POST[ $field['id'] ] ) ) { 
  464.                     continue; 
  465.                 } 
  466.  
  467.                 if ( is_callable( array( $order, 'set_billing_' . $key ) ) ) { 
  468.                     $props[ 'billing_' . $key ] = wc_clean( $_POST[ $field['id'] ] ); 
  469.                 } else { 
  470.                     $order->update_meta_data( $field['id'], wc_clean( $_POST[ $field['id'] ] ) ); 
  471.                 } 
  472.             } 
  473.         } 
  474.  
  475.         // Update shipping fields. 
  476.         if ( ! empty( self::$shipping_fields ) ) { 
  477.             foreach ( self::$shipping_fields as $key => $field ) { 
  478.                 if ( ! isset( $field['id'] ) ) { 
  479.                     $field['id'] = '_shipping_' . $key; 
  480.                 } 
  481.  
  482.                 if ( ! isset( $_POST[ $field['id'] ] ) ) { 
  483.                     continue; 
  484.                 } 
  485.  
  486.                 if ( is_callable( array( $order, 'set_shipping_' . $key ) ) ) { 
  487.                     $props[ 'shipping_' . $key ] = wc_clean( $_POST[ $field['id'] ] ); 
  488.                 } else { 
  489.                     $order->update_meta_data( $field['id'], wc_clean( $_POST[ $field['id'] ] ) ); 
  490.                 } 
  491.             } 
  492.         } 
  493.  
  494.         if ( isset( $_POST['_transaction_id'] ) ) { 
  495.             $props['transaction_id'] = wc_clean( $_POST['_transaction_id'] ); 
  496.         } 
  497.  
  498.         // Payment method handling. 
  499.         if ( $order->get_payment_method() !== wp_unslash( $_POST['_payment_method'] ) ) { 
  500.             $methods = WC()->payment_gateways->payment_gateways(); 
  501.             $payment_method = wc_clean( $_POST['_payment_method'] ); 
  502.             $payment_method_title = $payment_method; 
  503.  
  504.             if ( isset( $methods ) && isset( $methods[ $payment_method ] ) ) { 
  505.                 $payment_method_title = $methods[ $payment_method ]->get_title(); 
  506.             } 
  507.  
  508.             $props['payment_method'] = $payment_method; 
  509.             $props['payment_method_title'] = $payment_method_title; 
  510.         } 
  511.  
  512.         // Update date. 
  513.         if ( empty( $_POST['order_date'] ) ) { 
  514.             $date = current_time( 'timestamp', true ); 
  515.         } else { 
  516.             $date = gmdate( 'Y-m-d H:i:s', strtotime( $_POST['order_date'] . ' ' . (int) $_POST['order_date_hour'] . ':' . (int) $_POST['order_date_minute'] . ':00' ) ); 
  517.         } 
  518.  
  519.         $props['date_created'] = $date; 
  520.  
  521.         // Save order data. 
  522.         $order->set_props( $props ); 
  523.         $order->save(); 
  524.     } 
  525. }
