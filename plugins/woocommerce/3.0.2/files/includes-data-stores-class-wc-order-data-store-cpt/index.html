  175.           '_cart_hash' => 'cart_hash',  
  176.  ); 
  177.  
  178.       $props_to_update = $this->get_props_to_update( $order, $meta_key_to_props ); 
  179.  
  180.       foreach ( $props_to_update as $meta_key => $prop ) { 
  181.           $value = $order->{"get_$prop"}( 'edit' ); 
  182.  
  183.           if ( 'date_paid' === $prop ) { 
  184.               // In 3.0.x we store this as a UTC timestamp. 
  185.               update_post_meta( $id, $meta_key, ! is_null( $value ) ? $value->getTimestamp() : '' ); 
  186.  
  187.               // In 2.6.x date_paid was stored as _paid_date in local mysql format. 
  188.               update_post_meta( $id, '_paid_date', ! is_null( $value ) ? $value->date( 'Y-m-d H:i:s' ) : '' ); 
  189.  
  190.           } elseif ( 'date_completed' === $prop ) { 
  191.               // In 3.0.x we store this as a UTC timestamp. 
  192.               update_post_meta( $id, $meta_key, ! is_null( $value ) ? $value->getTimestamp() : '' ); 
  193.  
  194.               // In 2.6.x date_paid was stored as _paid_date in local mysql format. 
  195.               update_post_meta( $id, '_completed_date', ! is_null( $value ) ? $value->date( 'Y-m-d H:i:s' ) : '' ); 
  196.  
  197.           } else { 
  198.               update_post_meta( $id, $meta_key, $value ); 
  199.           } 
  200.  
  201.           $updated_props[] = $prop; 
  202.       } 
  203.  
  204.       $address_props = array( 
  205.           'billing' => array( 
  206.               '_billing_first_name' => 'billing_first_name',  
  207.               '_billing_last_name' => 'billing_last_name',  
  208.               '_billing_company' => 'billing_company',  
  209.               '_billing_address_1' => 'billing_address_1',  
  210.               '_billing_address_2' => 'billing_address_2',  
  211.               '_billing_city' => 'billing_city',  
  212.               '_billing_state' => 'billing_state',  
  213.               '_billing_postcode' => 'billing_postcode',  
  214.               '_billing_country' => 'billing_country',  
  215.               '_billing_email' => 'billing_email',  
  216.               '_billing_phone' => 'billing_phone',  
  217.  ),  
  218.           'shipping' => array( 
  219.               '_shipping_first_name' => 'shipping_first_name',  
  220.               '_shipping_last_name' => 'shipping_last_name',  
  221.               '_shipping_company' => 'shipping_company',  
  222.               '_shipping_address_1' => 'shipping_address_1',  
  223.               '_shipping_address_2' => 'shipping_address_2',  
  224.               '_shipping_city' => 'shipping_city',  
  225.               '_shipping_state' => 'shipping_state',  
  226.               '_shipping_postcode' => 'shipping_postcode',  
  227.               '_shipping_country' => 'shipping_country',  
  228.  ),  
  229.  ); 
  230.  
  231.       foreach ( $address_props as $props_key => $props ) { 
  232.           $props_to_update = $this->get_props_to_update( $order, $props ); 
  233.           foreach ( $props_to_update as $meta_key => $prop ) { 
  234.               $value = $order->{"get_$prop"}( 'edit' ); 
  235.               update_post_meta( $id, $meta_key, $value ); 
  236.               $updated_props[] = $prop; 
  237.               $updated_props[] = $props_key; 
  238.           } 
  239.       } 
  240.  
  241.       parent::update_post_meta( $order ); 
  242.  
  243.       // If address changed, store concatenated version to make searches faster. 
  244.       if ( in_array( 'billing', $updated_props ) || ! metadata_exists( 'post', $id, '_billing_address_index' ) ) { 
  245.           update_post_meta( $id, '_billing_address_index', implode( ' ', $order->get_address( 'billing' ) ) ); 
  246.       } 
  247.       if ( in_array( 'shipping', $updated_props ) || ! metadata_exists( 'post', $id, '_shipping_address_index' ) ) { 
  248.           update_post_meta( $id, '_shipping_address_index', implode( ' ', $order->get_address( 'shipping' ) ) ); 
  249.       } 
  250.  
  251.       // If customer changed, update any downloadable permissions. 
  252.       if ( in_array( 'customer_user', $updated_props ) || in_array( 'billing_email', $updated_props ) ) { 
  253.           $data_store = WC_Data_Store::load( 'customer-download' ); 
  254.           $data_store->update_user_by_order_id( $id, $order->get_customer_id(), $order->get_billing_email() ); 
  255.       } 
  256.  
  257.       do_action( 'woocommerce_order_object_updated_props', $order, $updated_props ); 
  258.   }

  427.   /** 
  428.    * Generate meta query for wc_get_orders. 
  429.    * 
  430.    * @param  array $values 
  431.    * @param  string $relation 
  432.    * @return array 
  433.    */ 
  434.   private function get_orders_generate_customer_meta_query( $values, $relation = 'or' ) { 
  435.       $meta_query = array( 
  436.           'relation' => strtoupper( $relation ),  
  437.           'customer_emails' => array( 
  438.               'key' => '_billing_email',  
  439.               'value' => array(),  
  440.               'compare' => 'IN',  
  441.  ),  
  442.           'customer_ids' => array( 
  443.               'key' => '_customer_user',  
  444.               'value' => array(),  
  445.               'compare' => 'IN',  
  446.  ),  
  447.  ); 
  448.       foreach ( $values as $value ) { 
  449.           if ( is_array( $value ) ) { 
  450.               $meta_query[] = $this->get_orders_generate_customer_meta_query( $value, 'and' ); 
  451.           } elseif ( is_email( $value ) ) { 
  452.               $meta_query['customer_emails']['value'][] = sanitize_email( $value ); 
  453.           } else { 
  454.               $meta_query['customer_ids']['value'][] = strval( absint( $value ) ); 
  455.           } 
  456.       } 
  457.  
  458.       if ( empty( $meta_query['customer_emails']['value'] ) ) { 
  459.           unset( $meta_query['customer_emails'] ); 
  460.           unset( $meta_query['relation'] ); 
  461.       } 
  462.  
  463.       if ( empty( $meta_query['customer_ids']['value'] ) ) { 
  464.           unset( $meta_query['customer_ids'] ); 
  465.           unset( $meta_query['relation'] ); 
  466.       } 
  467.  
  468.       return $meta_query; 
  469.   }

  483.           WHERE   posts.post_type   IN ('" . implode( "', '", wc_get_order_types() ) . "') 
  484.           AND     posts.post_status = 'wc-pending' 
  485.           AND     posts.post_modified < %s 
  486.       ", date( 'Y-m-d H:i:s', absint( $date ) ) ) ); 
  487.  
  488.       return $unpaid_orders; 
  489.   }

  491.   /** 
  492.    * Search order data for a term and return ids. 
  493.    * 
  494.    * @param  string $term 
  495.    * @return array of ids 
  496.    */ 
  497.   public function search_orders( $term ) { 
  498.       global $wpdb; 
  499.  
  500.       /** 
  501.        * Searches on meta data can be slow - this lets you choose what fields to search. 
  502.        * 3.0.0 added _billing_address and _shipping_address meta which contains all address data to make this faster. 
  503.        * This however won't work on older orders unless updated, so search a few others (expand this using the filter if needed). 
  504.        * @var array 
  505.        */ 
  506.       $search_fields = array_map( 'wc_clean', apply_filters( 'woocommerce_shop_order_search_fields', array( 
  507.           '_billing_address_index',  
  508.           '_shipping_address_index',  
  509.           '_billing_last_name',  
  510.           '_billing_email',  
  511.  ) ) ); 
  512.       $order_ids = array(); 
  513.  
  514.       if ( is_numeric( $term ) ) { 
  515.           $order_ids[] = absint( $term ); 
  516.       } 
  517.  
  518.       if ( ! empty( $search_fields ) ) { 
  519.           $order_ids = array_unique( array_merge( 
  520.               $order_ids,  
  521.               $wpdb->get_col( 
  522.                   $wpdb->prepare( "SELECT DISTINCT p1.post_id FROM {$wpdb->postmeta} p1 WHERE p1.meta_value LIKE '%%%s%%'", $wpdb->esc_like( wc_clean( $term ) ) ) . " AND p1.meta_key IN ('" . implode( "', '", array_map( 'esc_sql', $search_fields ) ) . "')" 
  523.  ),  
  524.               $wpdb->get_col( 
  525.                   $wpdb->prepare( " 
  526.                       SELECT order_id 
  527.                       FROM {$wpdb->prefix}woocommerce_order_items as order_items 
  528.                       WHERE order_item_name LIKE '%%%s%%' 
  529.                       ",  
  530.                       $wpdb->esc_like( wc_clean( $term ) ) 
  531.  ) 
  532.  ) 
  533.  ) ); 
  534.       } 
  535.  
  536.       return $order_ids; 
  537.   }
