  154.   /** 
  155.    * Get line item args for paypal request. 
  156.    * @param  WC_Order $order 
  157.    * @return array 
  158.    */ 
  159.   protected function get_line_item_args( $order ) { 
  160.  
  161.       /** 
  162.        * Try passing a line item per product if supported. 
  163.        */ 
  164.       if ( ( ! wc_tax_enabled() || ! wc_prices_include_tax() ) && $this->prepare_line_items( $order ) ) { 
  165.  
  166.           $line_item_args = array(); 
  167.           $line_item_args['tax_cart'] = $this->number_format( $order->get_total_tax(), $order ); 
  168.  
  169.           if ( $order->get_total_discount() > 0 ) { 
  170.               $line_item_args['discount_amount_cart'] = $this->number_format( $this->round( $order->get_total_discount(), $order ), $order ); 
  171.           } 
  172.  
  173.           // Add shipping costs. Paypal ignores anything over 5 digits (999.99 is the max). 
  174.           // We also check that shipping is not the **only** cost as PayPal won't allow payment 
  175.           // if the items have no cost. 
  176.           if ( $order->get_total_shipping() > 0 && $order->get_total_shipping() < 999.99 && $this->number_format( $order->get_total_shipping() + $order->get_shipping_tax(), $order ) !== $this->number_format( $order->get_total(), $order ) ) { 
  177.               $line_item_args['shipping_1'] = $this->number_format( $order->get_total_shipping(), $order ); 
  178.           } elseif ( $order->get_total_shipping() > 0 ) { 
  179.               $this->add_line_item( sprintf( __( 'Shipping via %s', 'woocommerce' ), $order->get_shipping_method() ), 1, $this->number_format( $order->get_total_shipping(), $order ) ); 
  180.           } 
  181.  
  182.           $line_item_args = array_merge( $line_item_args, $this->get_line_items() ); 
  183.  
  184.       /** 
  185.        * Send order as a single item. 
  186.        * 
  187.        * For shipping, we longer use shipping_1 because paypal ignores it if *any* shipping rules are within paypal, and paypal ignores anything over 5 digits (999.99 is the max). 
  188.        */ 
  189.       } else { 
  190.  
  191.           $this->delete_line_items(); 
  192.  
  193.           $line_item_args = array(); 
  194.           $all_items_name = $this->get_order_item_names( $order ); 
  195.           $this->add_line_item( $all_items_name ? $all_items_name : __( 'Order', 'woocommerce' ), 1, $this->number_format( $order->get_total() - $this->round( $order->get_total_shipping() + $order->get_shipping_tax(), $order ), $order ), $order->get_order_number() ); 
  196.  
  197.           // Add shipping costs. Paypal ignores anything over 5 digits (999.99 is the max). 
  198.           // We also check that shipping is not the **only** cost as PayPal won't allow payment 
  199.           // if the items have no cost. 
  200.           if ( $order->get_total_shipping() > 0 && $order->get_total_shipping() < 999.99 && $this->number_format( $order->get_total_shipping() + $order->get_shipping_tax(), $order ) !== $this->number_format( $order->get_total(), $order ) ) { 
  201.               $line_item_args['shipping_1'] = $this->number_format( $order->get_total_shipping() + $order->get_shipping_tax(), $order ); 
  202.           } elseif ( $order->get_total_shipping() > 0 ) { 
  203.               $this->add_line_item( sprintf( __( 'Shipping via %s', 'woocommerce' ), $order->get_shipping_method() ), 1, $this->number_format( $order->get_total_shipping() + $order->get_shipping_tax(), $order ) ); 
  204.           } 
  205.  
  206.           $line_item_args = array_merge( $line_item_args, $this->get_line_items() ); 
  207.       } 
  208.  
  209.       return $line_item_args; 
  210.   }

  258.   /** 
  259.    * Get line items to send to paypal. 
  260.    * @param  WC_Order $order 
  261.    * @return bool 
  262.    */ 
  263.   protected function prepare_line_items( $order ) { 
  264.       $this->delete_line_items(); 
  265.       $calculated_total = 0; 
  266.  
  267.       // Products 
  268.       foreach ( $order->get_items( array( 'line_item', 'fee' ) ) as $item ) { 
  269.           if ( 'fee' === $item['type'] ) { 
  270.               $item_line_total = $this->number_format( $item['line_total'], $order ); 
  271.               $line_item = $this->add_line_item( $item['name'], 1, $item_line_total ); 
  272.               $calculated_total += $item_line_total; 
  273.           } else { 
  274.               $product = $order->get_product_from_item( $item ); 
  275.               $sku = $product ? $product->get_sku() : ''; 
  276.               $item_line_total = $this->number_format( $order->get_item_subtotal( $item, false ), $order ); 
  277.               $line_item = $this->add_line_item( $this->get_order_item_name( $order, $item ), $item['qty'], $item_line_total, $sku ); 
  278.               $calculated_total += $item_line_total * $item['qty']; 
  279.           } 
  280.  
  281.           if ( ! $line_item ) { 
  282.               return false; 
  283.           } 
  284.       } 
  285.  
  286.       // Check for mismatched totals. 
  287.       if ( $this->number_format( $calculated_total + $order->get_total_tax() + $this->round( $order->get_total_shipping(), $order ) - $this->round( $order->get_total_discount(), $order ), $order ) != $this->number_format( $order->get_total(), $order ) ) { 
  288.           return false; 
  289.       } 
  290.  
  291.       return true; 
  292.   }

  295.    * Add PayPal Line Item. 
  296.    * @param  string  $item_name 
  297.    * @param  int     $quantity 
  298.    * @param  int     $amount 
  299.    * @param  string  $item_number 
  300.    * @return bool successfully added or not

  302.   protected function add_line_item( $item_name, $quantity = 1, $amount = 0, $item_number = '' ) { 
  303.       $index = ( sizeof( $this->line_items ) / 4 ) + 1; 
  304.  
  305.       if ( $amount < 0 || $index > 9 ) { 
  306.           return false; 
  307.       } 
  308.  
  309.       $this->line_items[ 'item_name_' . $index ]   = html_entity_decode( wc_trim_string( $item_name ? $item_name : __( 'Item', 'woocommerce' ), 127 ), ENT_NOQUOTES, 'UTF-8' ); 
  310.       $this->line_items[ 'quantity_' . $index ]    = $quantity; 
  311.       $this->line_items[ 'amount_' . $index ]      = $amount; 
  312.       $this->line_items[ 'item_number_' . $index ] = $item_number; 
  313.  
  314.       return true; 
  315.   }
