  331. /** 
  332.  * Check if the given topic is a valid webhook topic, a topic is valid if: 
  333.  * 
  334.  * + starts with `action.woocommerce_` or `action.wc_`. 
  335.  * + it has a valid resource & event. 
  336.  * 
  337.  * @param  string $topic webhook topic 
  338.  * @return bool true if valid, false otherwise 
  339.  */ 
  340. function wc_is_webhook_valid_topic( $topic ) { 
  341.  
  342.   // Custom topics are prefixed with woocommerce_ or wc_ are valid 
  343.   if ( 0 === strpos( $topic, 'action.woocommerce_' ) || 0 === strpos( $topic, 'action.wc_' ) ) { 
  344.       return true; 
  345.   } 
  346.  
  347.   @list( $resource, $event ) = explode( '.', $topic ); 
  348.  
  349.   if ( ! isset( $resource ) || ! isset( $event ) ) { 
  350.       return false; 
  351.   } 
  352.  
  353.   $valid_resources = apply_filters( 'woocommerce_valid_webhook_resources', array( 'coupon', 'customer', 'order', 'product' ) ); 
  354.   $valid_events = apply_filters( 'woocommerce_valid_webhook_events', array( 'created', 'updated', 'deleted' ) ); 
  355.  
  356.   if ( in_array( $resource, $valid_resources ) && in_array( $event, $valid_events ) ) { 
  357.       return true; 
  358.   } 
  359.  
  360.   return false; 
  361. }

  363. /**

  368.  */ 
  369. function wc_is_valid_url( $url ) { 
  370.  
  371.   // Must start with http:// or https:// 
  372.   if ( 0 !== strpos( $url, 'http://' ) && 0 !== strpos( $url, 'https://' ) ) { 
  373.       return false;

  376.   // Must pass validation 
  377.   if ( ! filter_var( $url, FILTER_VALIDATE_URL ) ) {
