  19.  * 
  20.  * NOTE: At this point in time, ordering by menu_order for example isn't possible with this function. wp_get_post_terms has no. 
  21.  *   filters which we can utilise to modify it's query. https://core.trac.wordpress.org/ticket/19094. 
  22.  *

  28. function wc_get_product_terms( $product_id, $taxonomy, $args = array() ) { 
  29.   if ( ! taxonomy_exists( $taxonomy ) ) { 
  30.       return array(); 
  31.   } 
  32.  
  33.   if ( empty( $args['orderby'] ) && taxonomy_is_product_attribute( $taxonomy ) ) { 
  34.       $args['orderby'] = wc_attribute_orderby( $taxonomy ); 
  35.   } 
  36.  
  37.   // Support ordering by parent 
  38.   if ( ! empty( $args['orderby'] ) && in_array( $args['orderby'], array( 'name_num', 'parent' ) ) ) { 
  39.       $fields = isset( $args['fields'] ) ? $args['fields'] : 'all'; 
  40.       $orderby = $args['orderby']; 
  41.  
  42.       // Unset for wp_get_post_terms 
  43.       unset( $args['orderby'] ); 
  44.       unset( $args['fields'] ); 
  45.  
  46.       $terms = wp_get_post_terms( $product_id, $taxonomy, $args ); 
  47.  
  48.       switch ( $orderby ) { 
  49.           case 'name_num' : 
  50.               usort( $terms, '_wc_get_product_terms_name_num_usort_callback' ); 
  51.           break; 
  52.           case 'parent' : 
  53.               usort( $terms, '_wc_get_product_terms_parent_usort_callback' ); 
  54.           break; 
  55.       } 
  56.  
  57.       switch ( $fields ) { 
  58.           case 'names' : 
  59.               $terms = wp_list_pluck( $terms, 'name' ); 
  60.               break; 
  61.           case 'ids' : 
  62.               $terms = wp_list_pluck( $terms, 'term_id' ); 
  63.               break; 
  64.           case 'slugs' : 
  65.               $terms = wp_list_pluck( $terms, 'slug' ); 
  66.               break; 
  67.       } 
  68.   } elseif ( ! empty( $args['orderby'] ) && $args['orderby'] === 'menu_order' ) { 
  69.       // wp_get_post_terms doesn't let us use custom sort order 
  70.       $args['include'] = wp_get_post_terms( $product_id, $taxonomy, array( 'fields' => 'ids' ) ); 
  71.  
  72.       if ( empty( $args['include'] ) ) { 
  73.           $terms = array(); 
  74.       } else { 
  75.           // This isn't needed for get_terms 
  76.           unset( $args['orderby'] ); 
  77.  
  78.           // Set args for get_terms 
  79.           $args['menu_order'] = isset( $args['order'] ) ? $args['order'] : 'ASC'; 
  80.           $args['hide_empty'] = isset( $args['hide_empty'] ) ? $args['hide_empty'] : 0; 
  81.           $args['fields'] = isset( $args['fields'] ) ? $args['fields'] : 'names'; 
  82.  
  83.           // Ensure slugs is valid for get_terms - slugs isn't supported 
  84.           $args['fields'] = $args['fields'] === 'slugs' ? 'id=>slug' : $args['fields']; 
  85.           $terms = get_terms( $taxonomy, $args ); 
  86.       } 
  87.   } else { 
  88.       $terms = wp_get_post_terms( $product_id, $taxonomy, $args ); 
  89.   } 
  90.  
  91.   return apply_filters( 'woocommerce_get_product_terms' , $terms, $product_id, $taxonomy, $args ); 
  92. }

  318. /** 
  319.  * Move a term before the a    given element of its hierarchy level. 
  320.  * 
  321.  * @param int $the_term 
  322.  * @param int $next_id the id of the next sibling element in save hierarchy level 
  323.  * @param string $taxonomy 
  324.  * @param int $index (default: 0) 
  325.  * @param mixed $terms (default: null) 
  326.  * @return int 
  327.  */ 
  328. function wc_reorder_terms( $the_term, $next_id, $taxonomy, $index = 0, $terms = null ) { 
  329.   if ( ! $terms ) $terms = get_terms( $taxonomy, 'menu_order=ASC&hide_empty=0&parent=0' ); 
  330.   if ( empty( $terms ) ) return $index; 
  331.  
  332.   $id = $the_term->term_id; 
  333.  
  334.   $term_in_level = false; // flag: is our term to order in this level of terms 
  335.  
  336.   foreach ( $terms as $term ) { 
  337.  
  338.       if ( $term->term_id == $id ) { // our term to order, we skip 
  339.           $term_in_level = true; 
  340.           continue; // our term to order, we skip 
  341.       } 
  342.       // the nextid of our term to order, lets move our term here 
  343.       if (null !== $next_id && $term->term_id == $next_id) { 
  344.           $index++; 
  345.           $index = wc_set_term_order($id, $index, $taxonomy, true); 
  346.       } 
  347.  
  348.       // set order 
  349.       $index++; 
  350.       $index = wc_set_term_order($term->term_id, $index, $taxonomy); 
  351.  
  352.       // if that term has children we walk through them 
  353.       $children = get_terms($taxonomy, "parent={$term->term_id}&menu_order=ASC&hide_empty=0"); 
  354.       if ( ! empty( $children ) ) { 
  355.           $index = wc_reorder_terms( $the_term, $next_id, $taxonomy, $index, $children ); 
  356.       } 
  357.   } 
  358.  
  359.   // no nextid meaning our term is in last position 
  360.   if ( $term_in_level && null === $next_id ) { 
  361.       $index = wc_set_term_order( $id, $index + 1, $taxonomy, true ); 
  362.   } 
  363.  
  364.   return $index; 
  365. }

  403. /** 
  404.  * Add term ordering to get_terms. 
  405.  * 
  406.  * It enables the support a 'menu_order' parameter to get_terms for the product_cat taxonomy. 
  407.  * By default it is 'ASC'. It accepts 'DESC' too.

  409.  * To disable it, set it ot false (or 0).

  416. function wc_terms_clauses( $clauses, $taxonomies, $args ) { 
  417.   global $wpdb; 
  418.  
  419.   // No sorting when menu_order is false. 
  420.   if ( isset( $args['menu_order'] ) && $args['menu_order'] == false ) { 
  421.       return $clauses; 
  422.   } 
  423.  
  424.   // No sorting when orderby is non default. 
  425.   if ( isset( $args['orderby'] ) && $args['orderby'] != 'name' ) { 
  426.       return $clauses; 
  427.   } 
  428.  
  429.   // No sorting in admin when sorting by a column. 
  430.   if ( is_admin() && isset( $_GET['orderby'] ) ) { 
  431.       return $clauses; 
  432.   } 
  433.  
  434.   // No need to filter counts 
  435.   if ( strpos( 'COUNT(*)', $clauses['fields'] ) !== false ) { 
  436.       return $clauses; 
  437.   } 
  438.  
  439.   // WordPress should give us the taxonomies asked when calling the get_terms function. Only apply to categories and pa_ attributes. 
  440.   $found = false; 
  441.   foreach ( (array) $taxonomies as $taxonomy ) { 
  442.       if ( taxonomy_is_product_attribute( $taxonomy ) || in_array( $taxonomy, apply_filters( 'woocommerce_sortable_taxonomies', array( 'product_cat' ) ) ) ) { 
  443.           $found = true; 
  444.           break; 
  445.       } 
  446.   } 
  447.   if ( ! $found ) { 
  448.       return $clauses; 
  449.   } 
  450.  
  451.   // Meta name. 
  452.   if ( ! empty( $taxonomies[0] ) && taxonomy_is_product_attribute( $taxonomies[0] ) ) { 
  453.       $meta_name = 'order_' . esc_attr( $taxonomies[0] ); 
  454.   } else { 
  455.       $meta_name = 'order'; 
  456.   } 
  457.  
  458.   // Query fields. 
  459.   $clauses['fields'] = 'DISTINCT ' . $clauses['fields']; 
  460.  
  461.   // Query join. 
  462.   if ( get_option( 'db_version' ) < 34370 ) { 
  463.       $clauses['join'] .= " LEFT JOIN {$wpdb->woocommerce_termmeta} AS tm ON (t.term_id = tm.woocommerce_term_id AND tm.meta_key = '" . esc_sql( $meta_name ) . "') "; 
  464.   } else { 
  465.       $clauses['join'] .= " LEFT JOIN {$wpdb->termmeta} AS tm ON (t.term_id = tm.term_id AND tm.meta_key = '" . esc_sql( $meta_name ) . "') "; 
  466.   } 
  467.  
  468.   // Default to ASC. 
  469.   if ( ! isset( $args['menu_order'] ) || ! in_array( strtoupper( $args['menu_order'] ), array( 'ASC', 'DESC' ) ) ) { 
  470.       $args['menu_order'] = 'ASC'; 
  471.   } 
  472.  
  473.   $order = "ORDER BY tm.meta_value+0 " . $args['menu_order']; 
  474.  
  475.   if ( $clauses['orderby'] ) { 
  476.       $clauses['orderby'] = str_replace( 'ORDER BY', $order . ', ', $clauses['orderby'] ); 
  477.   } else { 
  478.       $clauses['orderby'] = $order; 
  479.   } 
  480.  
  481.   return $clauses; 
  482. }

  495.   global $wpdb; 
  496.  
  497.   // Standard callback 
  498.   if ( $callback ) { 
  499.       _update_post_term_count( $terms, $taxonomy ); 
  500.   } 
  501.  
  502.   // Stock query 
  503.   if ( get_option( 'woocommerce_hide_out_of_stock_items' ) == 'yes' ) { 
  504.       $stock_join = "LEFT JOIN {$wpdb->postmeta} AS meta_stock ON posts.ID = meta_stock.post_id"; 
  505.       $stock_query = " 
  506.       AND meta_stock.meta_key = '_stock_status' 
  507.       AND meta_stock.meta_value = 'instock' 
  508.       "; 
  509.   } else { 
  510.       $stock_query = $stock_join = ''; 
  511.   } 
  512.  
  513.   // Main query 
  514.   $count_query = " 
  515.       SELECT COUNT( DISTINCT posts.ID ) FROM {$wpdb->posts} as posts 
  516.       LEFT JOIN {$wpdb->postmeta} AS meta_visibility ON posts.ID = meta_visibility.post_id 
  517.       LEFT JOIN {$wpdb->term_relationships} AS rel ON posts.ID=rel.object_ID 
  518.       LEFT JOIN {$wpdb->term_taxonomy} AS tax USING( term_taxonomy_id ) 
  519.       LEFT JOIN {$wpdb->terms} AS term USING( term_id ) 
  520.       LEFT JOIN {$wpdb->postmeta} AS postmeta ON posts.ID = postmeta.post_id 
  521.       $stock_join 
  522.       WHERE     post_status = 'publish' 
  523.       AND     post_type = 'product' 
  524.       AND     meta_visibility.meta_key = '_visibility' 
  525.       AND     meta_visibility.meta_value IN ( 'visible', 'catalog' ) 
  526.       $stock_query 
  527.   "; 
  528.  
  529.   // Pre-process term taxonomy ids 
  530.   if ( ! $terms_are_term_taxonomy_ids ) { 
  531.       // We passed in an array of TERMS in format id=>parent 
  532.       $terms = array_filter( (array) array_keys( $terms ) ); 
  533.   } else { 
  534.       // If we have term taxonomy IDs we need to get the term ID 
  535.       $term_taxonomy_ids = $terms; 
  536.       $terms = array(); 
  537.       foreach ( $term_taxonomy_ids as $term_taxonomy_id ) { 
  538.           $term = get_term_by( 'term_taxonomy_id', $term_taxonomy_id, $taxonomy->name ); 
  539.           $terms[] = $term->term_id; 
  540.       } 
  541.   } 
  542.  
  543.   // Exit if we have no terms to count 
  544.   if ( empty( $terms ) ) { 
  545.       return; 
  546.   } 
  547.  
  548.   // Ancestors need counting 
  549.   if ( is_taxonomy_hierarchical( $taxonomy->name ) ) { 
  550.       foreach ( $terms as $term_id ) { 
  551.           $terms = array_merge( $terms, get_ancestors( $term_id, $taxonomy->name ) ); 
  552.       } 
  553.   } 
  554.  
  555.   // Unique terms only 
  556.   $terms = array_unique( $terms ); 
  557.  
  558.   // Count the terms 
  559.   foreach ( $terms as $term_id ) { 
  560.       $terms_to_count = array( absint( $term_id ) ); 
  561.  
  562.       if ( is_taxonomy_hierarchical( $taxonomy->name ) ) { 
  563.           // We need to get the $term's hierarchy so we can count its children too 
  564.           if ( ( $children = get_term_children( $term_id, $taxonomy->name ) ) && ! is_wp_error( $children ) ) { 
  565.               $terms_to_count = array_unique( array_map( 'absint', array_merge( $terms_to_count, $children ) ) ); 
  566.           } 
  567.       } 
  568.  
  569.       // Generate term query 
  570.       $term_query = 'AND term_id IN ( ' . implode( ', ', $terms_to_count ) . ' )'; 
  571.  
  572.       // Get the count 
  573.       $count = $wpdb->get_var( $count_query . $term_query ); 
  574.  
  575.       // Update the count 
  576.       update_woocommerce_term_meta( $term_id, 'product_count_' . $taxonomy->name, absint( $count ) ); 
  577.   } 
  578.  
  579.   delete_transient( 'wc_term_counts' ); 
  580. } 
  581.  
  582. /** 
  583.  * Recount terms after the stock amount changes. 
  584.  * 
  585.  * @param int $product_id 
  586.  */ 
  587. function wc_recount_after_stock_change( $product_id ) { 
  588.   if ( get_option( 'woocommerce_hide_out_of_stock_items' ) != 'yes' ) 
  589.       return; 
  590.  
  591.   $product_terms = get_the_terms( $product_id, 'product_cat' ); 
  592.  
  593.   if ( $product_terms ) { 
  594.       $product_cats = array(); 
  595.  
  596.       foreach ( $product_terms as $term ) { 
  597.           $product_cats[ $term->term_id ] = $term->parent; 
  598.       } 
  599.  
  600.       _wc_term_recount( $product_cats, get_taxonomy( 'product_cat' ), false, false ); 
  601.   } 
  602.  
  603.   $product_terms = get_the_terms( $product_id, 'product_tag' ); 
  604.  
  605.   if ( $product_terms ) { 
  606.       $product_tags = array(); 
  607.  
  608.       foreach ( $product_terms as $term ) { 
  609.           $product_tags[ $term->term_id ] = $term->parent; 
  610.       } 
  611.  
  612.       _wc_term_recount( $product_tags, get_taxonomy( 'product_tag' ), false, false ); 
  613.   } 
  614. } 
  615. add_action( 'woocommerce_product_set_stock_status', 'wc_recount_after_stock_change' );

  618. /** 
  619.  * Overrides the original term count for product categories and tags with the product count. 
  620.  * that takes catalog visibility into account. 
  621.  * 
  622.  * @param array $terms 
  623.  * @param string|array $taxonomies 
  624.  * @return array 
  625.  */ 
  626. function wc_change_term_counts( $terms, $taxonomies ) { 
  627.   if ( is_admin() || is_ajax() ) { 
  628.       return $terms; 
  629.   } 
  630.  
  631.   if ( ! isset( $taxonomies[0] ) || ! in_array( $taxonomies[0], apply_filters( 'woocommerce_change_term_counts', array( 'product_cat', 'product_tag' ) ) ) ) { 
  632.       return $terms; 
  633.   } 
  634.  
  635.   $term_counts = $o_term_counts = get_transient( 'wc_term_counts' ); 
  636.  
  637.   foreach ( $terms as &$term ) { 
  638.       if ( is_object( $term ) ) { 
  639.           $term_counts[ $term->term_id ] = isset( $term_counts[ $term->term_id ] ) ? $term_counts[ $term->term_id ] : get_woocommerce_term_meta( $term->term_id, 'product_count_' . $taxonomies[0] , true ); 
  640.  
  641.           if ( $term_counts[ $term->term_id ] !== '' ) { 
  642.               $term->count = absint( $term_counts[ $term->term_id ] ); 
  643.           } 
  644.       } 
  645.   } 
  646.  
  647.   // Update transient 
  648.   if ( $term_counts != $o_term_counts ) { 
  649.       set_transient( 'wc_term_counts', $term_counts, DAY_IN_SECONDS * 30 ); 
  650.   } 
  651.  
  652.   return $terms; 
  653. } 
  654. add_filter( 'get_terms', 'wc_change_term_counts', 10, 2 );

  657.  * Return products in a given term, and cache value. 
  658.  * 
  659.  * To keep in sync, product_count will be cleared on "set_object_terms". 
  660.  * 
  661.  * @param int $term_id 
  662.  * @param string $taxonomy 
  663.  * @return array 
  664.  */ 
  665. function wc_get_term_product_ids( $term_id, $taxonomy ) { 
  666.   $product_ids = get_woocommerce_term_meta( $term_id, 'product_ids', true ); 
  667.  
  668.   if ( false === $product_ids || ! is_array( $product_ids ) ) { 
  669.       $product_ids = get_objects_in_term( $term_id, $taxonomy ); 
  670.       update_woocommerce_term_meta( $term_id, 'product_ids', $product_ids ); 
  671.   } 
  672.  
  673.   return $product_ids; 
  674. }

  676. /** 
  677.  * When a post is updated and terms recounted (called by _update_post_term_count), clear the ids. 
  678.  * @param int    $object_id  Object ID. 
  679.  * @param array  $terms      An array of object terms. 
  680.  * @param array  $tt_ids     An array of term taxonomy IDs. 
  681.  * @param string $taxonomy   Taxonomy slug. 
  682.  * @param bool   $append     Whether to append new terms to the old terms. 
  683.  * @param array  $old_tt_ids Old array of term taxonomy IDs. 
  684.  */ 
  685. function wc_clear_term_product_ids( $object_id, $terms, $tt_ids, $taxonomy, $append, $old_tt_ids ) { 
  686.   foreach ( $old_tt_ids as $term_id ) { 
  687.       delete_woocommerce_term_meta( $term_id, 'product_ids' ); 
  688.   } 
  689.   foreach ( $tt_ids as $term_id ) { 
  690.       delete_woocommerce_term_meta( $term_id, 'product_ids' ); 
  691.   } 
  692. } 
  693. add_action( 'set_object_terms', 'wc_clear_term_product_ids', 10, 6 );
