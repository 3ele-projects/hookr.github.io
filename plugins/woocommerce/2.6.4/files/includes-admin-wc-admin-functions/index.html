  50. /** 
  51.  * Create a page and store the ID in an option. 
  52.  * 
  53.  * @param mixed $slug Slug for the new page 
  54.  * @param string $option Option name to store the page's ID 
  55.  * @param string $page_title (default: '') Title for the new page 
  56.  * @param string $page_content (default: '') Content for the new page 
  57.  * @param int $post_parent (default: 0) Parent for the new page 
  58.  * @return int page ID

  60. function wc_create_page( $slug, $option = '', $page_title = '', $page_content = '', $post_parent = 0 ) { 
  61.   global $wpdb; 
  62.  
  63.   $option_value = get_option( $option ); 
  64.  
  65.   if ( $option_value > 0 ) { 
  66.       $page_object = get_post( $option_value ); 
  67.  
  68.       if ( 'page' === $page_object->post_type && ! in_array( $page_object->post_status, array( 'pending', 'trash', 'future', 'auto-draft' ) ) ) { 
  69.           // Valid page is already in place 
  70.           return $page_object->ID; 
  71.       } 
  72.   } 
  73.  
  74.   if ( strlen( $page_content ) > 0 ) { 
  75.       // Search for an existing page with the specified page content (typically a shortcode) 
  76.       $valid_page_found = $wpdb->get_var( $wpdb->prepare( "SELECT ID FROM $wpdb->posts WHERE post_type='page' AND post_status NOT IN ( 'pending', 'trash', 'future', 'auto-draft' ) AND post_content LIKE %s LIMIT 1;", "%{$page_content}%" ) ); 
  77.   } else { 
  78.       // Search for an existing page with the specified page slug 
  79.       $valid_page_found = $wpdb->get_var( $wpdb->prepare( "SELECT ID FROM $wpdb->posts WHERE post_type='page' AND post_status NOT IN ( 'pending', 'trash', 'future', 'auto-draft' )  AND post_name = %s LIMIT 1;", $slug ) ); 
  80.   } 
  81.  
  82.   $valid_page_found = apply_filters( 'woocommerce_create_page_id', $valid_page_found, $slug, $page_content ); 
  83.  
  84.   if ( $valid_page_found ) { 
  85.       if ( $option ) { 
  86.           update_option( $option, $valid_page_found ); 
  87.       } 
  88.       return $valid_page_found; 
  89.   } 
  90.  
  91.   // Search for a matching valid trashed page 
  92.   if ( strlen( $page_content ) > 0 ) { 
  93.       // Search for an existing page with the specified page content (typically a shortcode) 
  94.       $trashed_page_found = $wpdb->get_var( $wpdb->prepare( "SELECT ID FROM $wpdb->posts WHERE post_type='page' AND post_status = 'trash' AND post_content LIKE %s LIMIT 1;", "%{$page_content}%" ) ); 
  95.   } else { 
  96.       // Search for an existing page with the specified page slug 
  97.       $trashed_page_found = $wpdb->get_var( $wpdb->prepare( "SELECT ID FROM $wpdb->posts WHERE post_type='page' AND post_status = 'trash' AND post_name = %s LIMIT 1;", $slug ) ); 
  98.   } 
  99.  
  100.   if ( $trashed_page_found ) { 
  101.       $page_id = $trashed_page_found; 
  102.       $page_data = array( 
  103.           'ID' => $page_id,  
  104.           'post_status' => 'publish',  
  105.  ); 
  106.        wp_update_post( $page_data ); 
  107.   } else { 
  108.       $page_data = array( 
  109.           'post_status' => 'publish',  
  110.           'post_type' => 'page',  
  111.           'post_author' => 1,  
  112.           'post_name' => $slug,  
  113.           'post_title' => $page_title,  
  114.           'post_content' => $page_content,  
  115.           'post_parent' => $post_parent,  
  116.           'comment_status' => 'closed' 
  117.  ); 
  118.       $page_id = wp_insert_post( $page_data ); 
  119.   } 
  120.  
  121.   if ( $option ) { 
  122.       update_option( $option, $page_id ); 
  123.   } 
  124.  
  125.   return $page_id; 
  126. }

  174. /** 
  175.  * Save order items. 
  176.  * 
  177.  * @since 2.2 
  178.  * @param int $order_id Order ID 
  179.  * @param array $items Order items to save 
  180.  */ 
  181. function wc_save_order_items( $order_id, $items ) { 
  182.   // Allow other plugins to check change in order items before they are saved 
  183.   do_action( 'woocommerce_before_save_order_items', $order_id, $items ); 
  184.  
  185.   global $wpdb; 
  186.  
  187.   // Order items + fees 
  188.   $subtotal = 0; 
  189.   $total = 0; 
  190.   $subtotal_tax = 0; 
  191.   $total_tax = 0; 
  192.   $taxes = array( 'items' => array(), 'shipping' => array() ); 
  193.  
  194.   if ( isset( $items['order_item_id'] ) ) { 
  195.       $line_total = $line_subtotal = $line_tax = $line_subtotal_tax = array(); 
  196.  
  197.       foreach ( $items['order_item_id'] as $item_id ) { 
  198.  
  199.           $item_id = absint( $item_id ); 
  200.  
  201.           if ( isset( $items['order_item_name'][ $item_id ] ) ) { 
  202.               $wpdb->update( 
  203.                   $wpdb->prefix . 'woocommerce_order_items',  
  204.                   array( 'order_item_name' => wc_clean( wp_unslash( $items['order_item_name'][ $item_id ] ) ) ),  
  205.                   array( 'order_item_id' => $item_id ),  
  206.                   array( '%s' ),  
  207.                   array( '%d' ) 
  208.  ); 
  209.           } 
  210.  
  211.           if ( isset( $items['order_item_qty'][ $item_id ] ) ) { 
  212.               wc_update_order_item_meta( $item_id, '_qty', wc_stock_amount( $items['order_item_qty'][ $item_id ] ) ); 
  213.           } 
  214.  
  215.           if ( isset( $items['order_item_tax_class'][ $item_id ] ) ) { 
  216.               wc_update_order_item_meta( $item_id, '_tax_class', wc_clean( $items['order_item_tax_class'][ $item_id ] ) ); 
  217.           } 
  218.  
  219.           // Get values. Subtotals might not exist, in which case copy value from total field 
  220.           $line_total[ $item_id ]        = isset( $items['line_total'][ $item_id ] ) ? $items['line_total'][ $item_id ] : 0; 
  221.           $line_subtotal[ $item_id ]     = isset( $items['line_subtotal'][ $item_id ] ) ? $items['line_subtotal'][ $item_id ] : $line_total[ $item_id ]; 
  222.           $line_tax[ $item_id ]          = isset( $items['line_tax'][ $item_id ] ) ? $items['line_tax'][ $item_id ] : array(); 
  223.           $line_subtotal_tax[ $item_id ] = isset( $items['line_subtotal_tax'][ $item_id ] ) ? $items['line_subtotal_tax'][ $item_id ] : $line_tax[ $item_id ]; 
  224.  
  225.           // Format taxes 
  226.           $line_taxes = array_map( 'wc_format_decimal', $line_tax[ $item_id ] ); 
  227.           $line_subtotal_taxes = array_map( 'wc_format_decimal', $line_subtotal_tax[ $item_id ] ); 
  228.  
  229.           // Update values 
  230.           wc_update_order_item_meta( $item_id, '_line_subtotal', wc_format_decimal( $line_subtotal[ $item_id ] ) ); 
  231.           wc_update_order_item_meta( $item_id, '_line_total', wc_format_decimal( $line_total[ $item_id ] ) ); 
  232.           wc_update_order_item_meta( $item_id, '_line_subtotal_tax', array_sum( $line_subtotal_taxes ) ); 
  233.           wc_update_order_item_meta( $item_id, '_line_tax', array_sum( $line_taxes ) ); 
  234.  
  235.           // Save line tax data - Since 2.2 
  236.           wc_update_order_item_meta( $item_id, '_line_tax_data', array( 'total' => $line_taxes, 'subtotal' => $line_subtotal_taxes ) ); 
  237.           $taxes['items'][] = $line_taxes; 
  238.  
  239.           // Total up 
  240.           $subtotal     += wc_format_decimal( $line_subtotal[ $item_id ] ); 
  241.           $total        += wc_format_decimal( $line_total[ $item_id ] ); 
  242.           $subtotal_tax += array_sum( $line_subtotal_taxes ); 
  243.           $total_tax    += array_sum( $line_taxes ); 
  244.  
  245.           // Clear meta cache 
  246.           wp_cache_delete( $item_id, 'order_item_meta' ); 
  247.       } 
  248.   } 
  249.  
  250.   // Save meta 
  251.   $meta_keys = isset( $items['meta_key'] ) ? $items['meta_key'] : array(); 
  252.   $meta_values = isset( $items['meta_value'] ) ? $items['meta_value'] : array(); 
  253.  
  254.   foreach ( $meta_keys as $id => $meta_key ) { 
  255.       $meta_value = ( empty( $meta_values[ $id ] ) && ! is_numeric( $meta_values[ $id ] ) ) ? '' : $meta_values[ $id ]; 
  256.  
  257.       // Delele blank item meta entries 
  258.       if ( $meta_key === '' && $meta_value === '' ) { 
  259.           $wpdb->query( $wpdb->prepare( "DELETE FROM {$wpdb->prefix}woocommerce_order_itemmeta WHERE meta_id = %d", $id ) ); 
  260.       } else { 
  261.  
  262.           $wpdb->update( 
  263.               $wpdb->prefix . 'woocommerce_order_itemmeta',  
  264.               array( 
  265.                   'meta_key' => wp_unslash( $meta_key ),  
  266.                   'meta_value' => wp_unslash( $meta_value ) 
  267.  ),  
  268.               array( 'meta_id' => $id ),  
  269.               array( '%s', '%s' ),  
  270.               array( '%d' ) 
  271.  ); 
  272.       } 
  273.   } 
  274.  
  275.   // Shipping Rows 
  276.   $order_shipping = 0; 
  277.  
  278.   if ( isset( $items['shipping_method_id'] ) ) { 
  279.  
  280.       foreach ( $items['shipping_method_id'] as $item_id ) { 
  281.           $item_id = absint( $item_id ); 
  282.           $method_id = isset( $items['shipping_method'][ $item_id ] ) ? wc_clean( $items['shipping_method'][ $item_id ] ) : ''; 
  283.           $method_title = isset( $items['shipping_method_title'][ $item_id ] ) ? wc_clean( wp_unslash( $items['shipping_method_title'][ $item_id ] ) ) : ''; 
  284.           $cost = isset( $items['shipping_cost'][ $item_id ] ) ? wc_format_decimal( $items['shipping_cost'][ $item_id ] ) : ''; 
  285.           $ship_taxes = isset( $items['shipping_taxes'][ $item_id ] ) ? array_map( 'wc_format_decimal', $items['shipping_taxes'][ $item_id ] ) : array(); 
  286.  
  287.           $wpdb->update( 
  288.               $wpdb->prefix . 'woocommerce_order_items',  
  289.               array( 'order_item_name' => $method_title ),  
  290.               array( 'order_item_id' => $item_id ),  
  291.               array( '%s' ),  
  292.               array( '%d' ) 
  293.  ); 
  294.  
  295.           wc_update_order_item_meta( $item_id, 'method_id', $method_id ); 
  296.           wc_update_order_item_meta( $item_id, 'cost', $cost ); 
  297.           wc_update_order_item_meta( $item_id, 'taxes', $ship_taxes ); 
  298.  
  299.           $taxes['shipping'][] = $ship_taxes; 
  300.  
  301.           $order_shipping += $cost; 
  302.       } 
  303.   } 
  304.  
  305.   // Taxes 
  306.   $order_taxes = isset( $items['order_taxes'] ) ? $items['order_taxes'] : array(); 
  307.   $taxes_items = array(); 
  308.   $taxes_shipping = array(); 
  309.   $total_tax = 0; 
  310.   $total_shipping_tax = 0; 
  311.  
  312.   // Sum items taxes 
  313.   foreach ( $taxes['items'] as $rates ) { 
  314.  
  315.       foreach ( $rates as $id => $value ) { 
  316.  
  317.           if ( isset( $taxes_items[ $id ] ) ) { 
  318.               $taxes_items[ $id ] += $value; 
  319.           } else { 
  320.               $taxes_items[ $id ] = $value; 
  321.           } 
  322.       } 
  323.   } 
  324.  
  325.   // Sum shipping taxes 
  326.   foreach ( $taxes['shipping'] as $rates ) { 
  327.  
  328.       foreach ( $rates as $id => $value ) { 
  329.  
  330.           if ( isset( $taxes_shipping[ $id ] ) ) { 
  331.               $taxes_shipping[ $id ] += $value; 
  332.           } else { 
  333.               $taxes_shipping[ $id ] = $value; 
  334.           } 
  335.       } 
  336.   } 
  337.  
  338.   // Update order taxes 
  339.   foreach ( $order_taxes as $item_id => $rate_id ) { 
  340.  
  341.       if ( isset( $taxes_items[ $rate_id ] ) ) { 
  342.           $_total = wc_format_decimal( $taxes_items[ $rate_id ] ); 
  343.           wc_update_order_item_meta( $item_id, 'tax_amount', $_total ); 
  344.  
  345.           $total_tax += $_total; 
  346.       } 
  347.  
  348.       if ( isset( $taxes_shipping[ $rate_id ] ) ) { 
  349.           $_total = wc_format_decimal( $taxes_shipping[ $rate_id ] ); 
  350.           wc_update_order_item_meta( $item_id, 'shipping_tax_amount', $_total ); 
  351.  
  352.           $total_shipping_tax += $_total; 
  353.       } 
  354.   } 
  355.  
  356.   // Update order shipping total 
  357.   update_post_meta( $order_id, '_order_shipping', $order_shipping ); 
  358.  
  359.   // Update cart discount from item totals 
  360.   update_post_meta( $order_id, '_cart_discount', $subtotal - $total ); 
  361.   update_post_meta( $order_id, '_cart_discount_tax', $subtotal_tax - $total_tax ); 
  362.  
  363.   // Update totals 
  364.   update_post_meta( $order_id, '_order_total', wc_format_decimal( $items['_order_total'] ) ); 
  365.  
  366.   // Update tax 
  367.   update_post_meta( $order_id, '_order_tax', wc_format_decimal( $total_tax ) ); 
  368.   update_post_meta( $order_id, '_order_shipping_tax', wc_format_decimal( $total_shipping_tax ) ); 
  369.  
  370.   // Remove old values 
  371.   delete_post_meta( $order_id, '_shipping_method' ); 
  372.   delete_post_meta( $order_id, '_shipping_method_title' ); 
  373.  
  374.   // Set the currency 
  375.   add_post_meta( $order_id, '_order_currency', get_woocommerce_currency(), true ); 
  376.  
  377.   // Update version after saving 
  378.   update_post_meta( $order_id, '_order_version', WC_VERSION ); 
  379.  
  380.   // Inform other plugins that the items have been saved 
  381.   do_action( 'woocommerce_saved_order_items', $order_id, $items ); 
  382. }
