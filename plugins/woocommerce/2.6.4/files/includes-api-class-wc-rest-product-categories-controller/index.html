  113.   protected function update_term_meta_fields( $term, $request ) { 
  114.       $id = (int) $term->term_id; 
  115.  
  116.       update_woocommerce_term_meta( $id, 'display_type', 'default' === $request['display'] ? '' : $request['display'] ); 
  117.       update_woocommerce_term_meta( $id, 'order', $request['menu_order'] ); 
  118.  
  119.       if ( ! empty( $request['image'] ) ) { 
  120.           if ( empty( $request['image']['id'] ) && ! empty( $request['image']['src'] ) ) { 
  121.               $upload = wc_rest_upload_image_from_url( esc_url_raw( $request['image']['src'] ) ); 
  122.  
  123.               if ( is_wp_error( $upload ) ) { 
  124.                   return $upload; 
  125.               } 
  126.  
  127.               $image_id = wc_rest_set_uploaded_image_as_attachment( $upload ); 
  128.           } else { 
  129.               $image_id = absint( $request['image']['id'] ); 
  130.           } 
  131.  
  132.           // Check if image_id is a valid image attachment before updating the term meta. 
  133.           if ( $image_id && wp_attachment_is_image( $image_id ) ) { 
  134.               update_woocommerce_term_meta( $id, 'thumbnail_id', $image_id ); 
  135.  
  136.               // Set the image alt. 
  137.               if ( ! empty( $request['image']['alt'] ) ) { 
  138.                   update_post_meta( $image_id, '_wp_attachment_image_alt', wc_clean( $request['image']['alt'] ) ); 
  139.               } 
  140.  
  141.               // Set the image title. 
  142.               if ( ! empty( $request['image']['title'] ) ) { 
  143.                   wp_update_post( array( 'ID' => $image_id, 'post_title' => wc_clean( $request['image']['title'] ) ) ); 
  144.               } 
  145.           } 
  146.       } 
  147.  
  148.       return true; 
  149.   }
