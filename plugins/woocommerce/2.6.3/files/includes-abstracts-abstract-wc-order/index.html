  1088.   /** 
  1089.    * Update tax lines for the order based on the line item taxes themselves. 
  1090.    */ 
  1091.   public function update_taxes() { 
  1092.       $cart_taxes = array(); 
  1093.       $shipping_taxes = array(); 
  1094.       $existing_taxes = $this->get_taxes(); 
  1095.       $saved_rate_ids = array(); 
  1096.  
  1097.       foreach ( $this->get_items( array( 'line_item', 'fee' ) ) as $item_id => $item ) { 
  1098.           $taxes = $item->get_taxes(); 
  1099.           foreach ( $taxes['total'] as $tax_rate_id => $tax ) { 
  1100.               $cart_taxes[ $tax_rate_id ] = isset( $cart_taxes[ $tax_rate_id ] ) ? $cart_taxes[ $tax_rate_id ] + (float) $tax : (float) $tax; 
  1101.           } 
  1102.       } 
  1103.  
  1104.       foreach ( $this->get_shipping_methods() as $item_id => $item ) { 
  1105.           $taxes = $item->get_taxes(); 
  1106.           foreach ( $taxes['total'] as $tax_rate_id => $tax ) { 
  1107.               $shipping_taxes[ $tax_rate_id ] = isset( $shipping_taxes[ $tax_rate_id ] ) ? $shipping_taxes[ $tax_rate_id ] + (float) $tax : (float) $tax; 
  1108.           } 
  1109.       } 
  1110.  
  1111.       foreach ( $existing_taxes as $tax ) { 
  1112.           // Remove taxes which no longer exist for cart/shipping. 
  1113.           if ( ( ! array_key_exists( $tax->get_rate_id(), $cart_taxes ) && ! array_key_exists( $tax->get_rate_id(), $shipping_taxes ) ) || in_array( $tax->get_rate_id(), $saved_rate_ids ) ) { 
  1114.               $this->remove_item( $tax->get_id() ); 
  1115.               continue; 
  1116.           } 
  1117.           $saved_rate_ids[] = $tax->get_rate_id(); 
  1118.           $tax->set_tax_total( isset( $cart_taxes[ $tax->get_rate_id() ] ) ? $cart_taxes[ $tax->get_rate_id() ] : 0 ); 
  1119.           $tax->set_shipping_tax_total( ! empty( $shipping_taxes[ $tax->get_rate_id() ] ) ? $shipping_taxes[ $tax->get_rate_id() ] : 0 ); 
  1120.           $tax->save(); 
  1121.       } 
  1122.  
  1123.       $new_rate_ids = wp_parse_id_list( array_diff( array_keys( $cart_taxes + $shipping_taxes ), $saved_rate_ids ) ); 
  1124.  
  1125.       // New taxes. 
  1126.       foreach ( $new_rate_ids as $tax_rate_id ) { 
  1127.           $item = new WC_Order_Item_Tax(); 
  1128.           $item->set_rate( $tax_rate_id ); 
  1129.           $item->set_tax_total( isset( $cart_taxes[ $tax_rate_id ] ) ? $cart_taxes[ $tax_rate_id ] : 0 ); 
  1130.           $item->set_shipping_tax_total( ! empty( $shipping_taxes[ $tax_rate_id ] ) ? $shipping_taxes[ $tax_rate_id ] : 0 ); 
  1131.           $this->add_item( $item ); 
  1132.       }
