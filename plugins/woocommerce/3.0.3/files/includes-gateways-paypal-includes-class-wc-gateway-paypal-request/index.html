  168.   protected function get_line_item_args( $order ) {

  174.  
  175.           $line_item_args = array();

  177.  
  178.           if ( $order->get_total_discount() > 0 ) { 
  179.               $line_item_args['discount_amount_cart'] = $this->number_format( $this->round( $order->get_total_discount(), $order ), $order ); 
  180.           } 
  181.  
  182.           // Add shipping costs. Paypal ignores anything over 5 digits (999.99 is the max). 
  183.           // We also check that shipping is not the **only** cost as PayPal won't allow payment 
  184.           // if the items have no cost. 
  185.           if ( $order->get_shipping_total() > 0 && $order->get_shipping_total() < 999.99 && $this->number_format( $order->get_shipping_total() + $order->get_shipping_tax(), $order ) !== $this->number_format( $order->get_total(), $order ) ) { 
  186.               $line_item_args['shipping_1'] = $this->number_format( $order->get_shipping_total(), $order ); 
  187.           } elseif ( $order->get_shipping_total() > 0 ) { 
  188.               $this->add_line_item( sprintf( __( 'Shipping via %s', 'woocommerce' ), $order->get_shipping_method() ), 1, $this->number_format( $order->get_shipping_total(), $order ) ); 
  189.           } 
  190.  
  191.           $line_item_args = array_merge( $line_item_args, $this->get_line_items() ); 
  192.  
  193.       /** 
  194.        * Send order as a single item. 
  195.        * 
  196.        * For shipping, we longer use shipping_1 because paypal ignores it if *any* shipping rules are within paypal, and paypal ignores anything over 5 digits (999.99 is the max). 
  197.        */ 
  198.       } else { 
  199.  
  200.           $this->delete_line_items(); 
  201.  
  202.           $line_item_args = array(); 
  203.           $all_items_name = $this->get_order_item_names( $order ); 
  204.           $this->add_line_item( $all_items_name ? $all_items_name : __( 'Order', 'woocommerce' ), 1, $this->number_format( $order->get_total() - $this->round( $order->get_shipping_total() + $order->get_shipping_tax(), $order ), $order ), $order->get_order_number() ); 
  205.  
  206.           // Add shipping costs. Paypal ignores anything over 5 digits (999.99 is the max). 
  207.           // We also check that shipping is not the **only** cost as PayPal won't allow payment 
  208.           // if the items have no cost. 
  209.           if ( $order->get_shipping_total() > 0 && $order->get_shipping_total() < 999.99 && $this->number_format( $order->get_shipping_total() + $order->get_shipping_tax(), $order ) !== $this->number_format( $order->get_total(), $order ) ) {

  211.           } elseif ( $order->get_shipping_total() > 0 ) { 
  212.               $this->add_line_item( sprintf( __( 'Shipping via %s', 'woocommerce' ), $order->get_shipping_method() ), 1, $this->number_format( $order->get_shipping_total() + $order->get_shipping_tax(), $order ) );

  286.   /** 
  287.    * Get line items to send to paypal. 
  288.    * @param  WC_Order $order 
  289.    * @return bool 
  290.    */ 
  291.   protected function prepare_line_items( $order ) { 
  292.       $this->delete_line_items(); 
  293.       $calculated_total = 0; 
  294.  
  295.       // Products 
  296.       foreach ( $order->get_items( array( 'line_item', 'fee' ) ) as $item ) { 
  297.           if ( 'fee' === $item['type'] ) { 
  298.               $item_line_total = $this->number_format( $item['line_total'], $order ); 
  299.               $line_item = $this->add_line_item( $item->get_name(), 1, $item_line_total ); 
  300.               $calculated_total += $item_line_total; 
  301.           } else { 
  302.               $product = $item->get_product(); 
  303.               $sku = $product ? $product->get_sku() : ''; 
  304.               $item_line_total = $this->number_format( $order->get_item_subtotal( $item, false ), $order ); 
  305.               $line_item = $this->add_line_item( $this->get_order_item_name( $order, $item ), $item->get_quantity(), $item_line_total, $sku ); 
  306.               $calculated_total += $item_line_total * $item->get_quantity(); 
  307.           } 
  308.  
  309.           if ( ! $line_item ) { 
  310.               return false; 
  311.           } 
  312.       } 
  313.  
  314.       // Check for mismatched totals. 
  315.       if ( $this->number_format( $calculated_total + $order->get_total_tax() + $this->round( $order->get_shipping_total(), $order ) - $this->round( $order->get_total_discount(), $order ), $order ) != $this->number_format( $order->get_total(), $order ) ) { 
  316.           return false; 
  317.       } 
  318.  
  319.       return true; 
  320.   }
