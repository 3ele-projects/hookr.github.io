  474.         ", date( 'Y-m-d H:i:s', absint( $date ) ) ) ); 
  475.  
  476.         return $unpaid_orders; 
  477.     }

  479.     /** 
  480.      * Search order data for a term and return ids. 
  481.      * 
  482.      * @param  string $term 
  483.      * @return array of ids 
  484.      */ 
  485.     public function search_orders( $term ) { 
  486.         global $wpdb; 
  487.  
  488.         /** 
  489.          * Searches on meta data can be slow - this lets you choose what fields to search. 
  490.          * 3.0.0 added _billing_address and _shipping_address meta which contains all address data to make this faster. 
  491.          * This however won't work on older orders unless updated, so search a few others (expand this using the filter if needed). 
  492.          * @var array 
  493.          */ 
  494.         $search_fields = array_map( 'wc_clean', apply_filters( 'woocommerce_shop_order_search_fields', array( 
  495.             '_billing_address_index',  
  496.             '_shipping_address_index',  
  497.             '_billing_last_name',  
  498.             '_billing_email',  
  499.  ) ) ); 
  500.         $order_ids = array(); 
  501.  
  502.         if ( is_numeric( $term ) ) { 
  503.             $order_ids[] = absint( $term ); 
  504.         } 
  505.  
  506.         if ( ! empty( $search_fields ) ) { 
  507.             $order_ids = array_unique( array_merge( 
  508.                 $order_ids,  
  509.                 $wpdb->get_col( 
  510.                     $wpdb->prepare( "SELECT DISTINCT p1.post_id FROM {$wpdb->postmeta} p1 WHERE p1.meta_value LIKE '%%%s%%'", $wpdb->esc_like( wc_clean( $term ) ) ) . " AND p1.meta_key IN ('" . implode( "', '", array_map( 'esc_sql', $search_fields ) ) . "')" 
  511.  ),  
  512.                 $wpdb->get_col( 
  513.                     $wpdb->prepare( " 
  514.                         SELECT order_id 
  515.                         FROM {$wpdb->prefix}woocommerce_order_items as order_items 
  516.                         WHERE order_item_name LIKE '%%%s%%' 
  517.                         ",  
  518.                         $wpdb->esc_like( wc_clean( $term ) ) 
  519.  ) 
  520.  ) 
  521.  ) ); 
  522.         } 
  523.  
  524.         return $order_ids; 
  525.     }
