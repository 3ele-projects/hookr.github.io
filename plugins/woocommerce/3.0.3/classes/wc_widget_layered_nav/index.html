  329.         $meta_query = WC_Query::get_main_meta_query();

  331.         if ( 'or' === $query_type ) { 
  332.             foreach ( $tax_query as $key => $query ) { 
  333.                 if ( is_array( $query ) && $taxonomy === $query['taxonomy'] ) { 
  334.                     unset( $tax_query[ $key ] ); 
  335.                 } 
  336.             }

  343.  
  344.         // Generate query 
  345.         $query = array(); 
  346.         $query['select'] = "SELECT COUNT( DISTINCT {$wpdb->posts}.ID ) as term_count, terms.term_id as term_count_id";

  349.             INNER JOIN {$wpdb->term_relationships} AS term_relationships ON {$wpdb->posts}.ID = term_relationships.object_id

  353.  
  354.         $query['where'] = " 
  355.             WHERE {$wpdb->posts}.post_type IN ( 'product' ) 
  356.             AND {$wpdb->posts}.post_status = 'publish' 
  357.             " . $tax_query_sql['where'] . $meta_query_sql['where'] . "

  359.         ";

  361.         if ( $search = WC_Query::get_main_search_query_sql() ) {

  363.         } 
  364.  
  365.         $query['group_by'] = "GROUP BY terms.term_id"; 
  366.         $query = apply_filters( 'woocommerce_get_filtered_term_product_counts_query', $query ); 
  367.         $query = implode( ' ', $query );

  370.         return wp_list_pluck( $results, 'term_count', 'term_count_id' );
