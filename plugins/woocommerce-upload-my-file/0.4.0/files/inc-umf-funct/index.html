  192. function woocommerce_umf_get_product_meta( $product, $field_name ) { 
  193.  
  194. if ( version_compare( WOOCOMMERCE_VERSION, "2.0.0" ) >= 0 ) { 
  195.   // even in WC >= 2.0 product variations still use the product_custom_fields array apparently 
  196.   if ( $product->variation_id && isset( $product->product_custom_fields[ '_' . $field_name ][0] ) && $product->product_custom_fields[ '_' . $field_name ][0] !== '' ) { 
  197.     return $product->product_custom_fields[ '_' . $field_name ][0]; 
  198.   } 
  199.   // use magic __get 
  200.   return maybe_unserialize( $product->$field_name ); 
  201. } else { 
  202.   // use product custom fields array 
  203.  
  204.   // variation support: return the value if it's defined at the variation level 
  205.   if ( isset( $product->variation_id ) && $product->variation_id ) { 
  206.  
  207.     if ( ( $value = get_post_meta( $product->variation_id, '_' . $field_name, true ) ) !== '' ) return $value; 
  208.     // otherwise return the value from the parent 
  209.     return get_post_meta( $product->id, '_' . $field_name, true ); 
  210.   } 
  211.  
  212.   // regular product 
  213.   return isset( $product->product_custom_fields[ '_' . $field_name ][0] ) ? $product->product_custom_fields[ '_' . $field_name ][0] : null;

  260. function upload_files_field( $order_id ) { 
  261.   $order = new WC_Order( $order_id ); 
  262.  
  263.   /** upload handler */ 
  264.   if( isset( $_FILES ) && isset($_POST['upload']) && $_POST['upload']) { 
  265.  
  266.       $upload_dir=wp_upload_dir(); 
  267.       $path = trailingslashit( trailingslashit( $upload_dir['basedir'].'/umf/' ) . $order_id ); 
  268.  
  269.       foreach( $_FILES as $key => $umf_file ) { 
  270.           if( empty( $umf_file['name'] ) ) continue; 
  271.           wp_mkdir_p( $path ); 
  272.           $umf_filepath = $path . $order_id.'_'.$key.'_'.$umf_file['name']; 
  273.           $ext = strtolower( pathinfo( $umf_filepath, PATHINFO_EXTENSION ) ); 
  274.           $doctypes = explode( ', ', str_replace('.', '', get_option( 'woocommerce_umf_allowed_file_types' ) )); 
  275.           foreach($doctypes as $k => $v) { $doctypes[$k] = strtolower( trim( $v ) ); } 
  276.               if( in_array( $ext, $doctypes ) ) $typeallow = true; 
  277.                   else $typeallow = false; 
  278.           if($typeallow==false) { 
  279.               $upload_error.= '<li>' . sprintf( __( 'The %s file type is not allowed.', 'woocommerce-umf'), $ext ) . '</li>'; 
  280.           } 
  281.           $filesize=(int)get_option( 'woocommerce_umf_max_uploadsize' ); 
  282.           $max_upload = (int)(ini_get('upload_max_filesize')); 
  283.           $max_post = (int)(ini_get('post_max_size')); 
  284.           $memory_limit = (int)(ini_get('memory_limit')); 
  285.           $max_mb = min($max_upload, $max_post, $memory_limit); 
  286.           if($filesize > $max_mb) { $filesize=$max_mb;} 
  287.           if($_FILES[$key]["size"]< ($filesize*1024*1024)) 
  288.               {$sizeallow=true; } else {$sizeallow=false; $upload_error.= '<li">' . sprintf( __( 'The file "%s" is to big.', 'woocommerce-umf'), $umf_file['name'] ) . '</li>';} 
  289.  
  290.           if( $typeallow == true && $sizeallow == true ) { 
  291.               if( copy( $umf_file['tmp_name'], $umf_filepath ) ) { 
  292.                   $success=true; 
  293.                   update_post_meta( $order_id, '_woo_umf_uploaded_file_name_' . $key, $umf_file['name'] ); 
  294.                   update_post_meta( $order_id, '_woo_umf_uploaded_file_path_' . $key, $umf_filepath ); 
  295.                   update_post_meta( $order_id, '_woo_umf_uploaded_product_name_' . $key, $_POST['uploaded_product_name'][$key] ); 
  296.               } else { 
  297.                   $upload_error.= '<li>' . __( 'There was an error while uploading your file(s).', 'woocommerce-umf') . '</li>'; 
  298.               } 
  299.           } 
  300.       if($success==true && isset($upload_error) && $upload_error!="") { 
  301.         echo '<p class="woocommerce-info woo-umf-success updated">' . __( 'There was a problem while uploading your files.', 'woocommerce-umf') . '<br>' . __( 'Not all files have been successfully uploaded.', 'woocommerce-umf') . '</p>'; 
  302.       } 
  303.       if($success==true && isset($upload_error) && $upload_error=="") { 
  304.         echo '<p class="woocommerce-message woo-umf-success success">' . __( 'Your file(s) were uploaded successfully.', 'woocommerce-umf') . '</p>'; 
  305.       } 
  306.       if(isset($upload_error) && $upload_error!="") { 
  307.         echo '<div class="woocommerce-error woo-umf-error error"><ul>'; 
  308.         echo $upload_error; 
  309.         echo '</ul></div>'; 
  310.       } 
  311.       } 
  312.   } 
  313.  
  314.   /** wanneer er meer dan 1 bestand geupload kan worden, ga verder */ 
  315.   if(get_max_upload_count($order) > 0) { 
  316.   echo '<h2>' . __( 'Upload files', 'woocommerce-umf' ) . '</h2>'; 
  317.  
  318.   /**begin form*/

  326.           $product=new WC_Product($order_item['product_id']);
