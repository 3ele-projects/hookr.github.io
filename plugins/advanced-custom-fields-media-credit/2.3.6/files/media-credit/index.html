  218. function filter_images_with_caption( $content ) { 
  219.   $captioned_images = array(); 
  220.  
  221.   //find all images w/ caption 
  222.   preg_match_all( '/\[caption(.*?)\](.*?)\[\/caption]/', $content, $captioned_images ); 
  223.   if ( $captioned_images ) { 
  224.       foreach ( $captioned_images[0] as $captioned_image ) { 
  225.           if ( preg_match( '/\[caption(.*?)\](.*?)<img(.*?)class="(.*?)"(.*?)\[\/caption]/', $content ) ) { 
  226.               $pattern = '/\[caption(.*?)\](.*?)<img(.*?)class="(.*?)"(.*?)\[\/caption]/'; 
  227.               //this will add a new shortcode attribute of class that will not be seen in the edit screen 
  228.               //but will be interpreted by our filter_img_caption_shortcode function 
  229.               $new_captioned_image = '[caption${1} class="${4}"]${2}<img${3}class="${4} has-caption-early"${5}[/caption]'; 
  230.           } 
  231.           $content = preg_replace( $pattern, $new_captioned_image, $content ); 
  232.       } 
  233.   } 
  234.   return $content;
