  660.     /** 
  661.      * Detects if API call should use multipart/form-data 
  662.      * 
  663.      * @param string $method The API method to call 
  664.      * 
  665.      * @return bool Whether the method should be sent as multipart 
  666.      */ 
  667.     protected function _detectMultipart($method) 
  668.     { 
  669.         $multiparts = array( 
  670.             // Tweets 
  671.             'statuses/update_with_media',  
  672.  
  673.             // Users 
  674.             'account/update_profile_background_image',

  681.     /** 
  682.      * Detect filenames in upload parameters,  
  683.      * build multipart request from upload params 
  684.      * 
  685.      * @param string $method  The API method to call 
  686.      * @param array  $params  The parameters to send along 
  687.      * 
  688.      * @return void 
  689.      */ 
  690.     protected function _buildMultipart($method, $params) 
  691.     { 
  692.         // well, files will only work in multipart methods 
  693.         if (! $this->_detectMultipart($method)) { 
  694.             return; 
  695.         } 
  696.  
  697.         // only check specific parameters 
  698.         $possible_files = array( 
  699.             // Tweets 
  700.             'statuses/update_with_media' => 'media[]',  
  701.             // Accounts 
  702.             'account/update_profile_background_image' => 'image',  
  703.             'account/update_profile_image' => 'image',  
  704.             'account/update_profile_banner' => 'banner' 
  705.  ); 
  706.         // method might have files? 
  707.         if (! in_array($method, array_keys($possible_files))) { 
  708.             return; 
  709.         } 
  710.  
  711.         $possible_files = explode(' ', $possible_files[$method]); 
  712.  
  713.         $multipart_border = '--------------------' . $this->_nonce(); 
  714.         $multipart_request = ''; 
  715.  
  716.         foreach ($params as $key => $value) { 
  717.             // is it an array? 
  718.             if (is_array($value)) { 
  719.                 throw new \Exception('Using URL-encoded parameters is not supported for uploading media.'); 
  720.                 continue; 
  721.             } 
  722.             $multipart_request .= 
  723.                 '--' . $multipart_border . "\r\n" 
  724.                 . 'Content-Disposition: form-data; name="' . $key . '"'; 
  725.  
  726.             // check for filenames 
  727.             if (in_array($key, $possible_files)) { 
  728.                 if (// is it a file, a readable one? 
  729.                     @file_exists($value) 
  730.                     && @is_readable($value) 
  731.  
  732.                     // is it a valid image? 
  733.                     && $data = @getimagesize($value) 
  734.  ) { 
  735.                     if (// is it a supported image format? 
  736.                         in_array($data[2], $this->_supported_media_files) 
  737.  ) { 
  738.                         // try to read the file 
  739.                         ob_start(); 
  740.                         readfile($value); 
  741.                         $data = ob_get_contents(); 
  742.                         ob_end_clean(); 
  743.                         if (strlen($data) == 0) { 
  744.                             continue; 
  745.                         } 
  746.                         $value = $data; 
  747.                     } 
  748.                 } 
  749.  
  750.                 /** 
  751.                 $multipart_request .= 
  752.                     "\r\nContent-Transfer-Encoding: base64"; 
  753.                 $value = base64_encode($value); 
  754.                 */ 
  755.             } 
  756.  
  757.             $multipart_request .= 
  758.                 "\r\n\r\n" . $value . "\r\n"; 
  759.         } 
  760.         $multipart_request .= '--' . $multipart_border . '--'; 
  761.  
  762.         return $multipart_request; 
  763.     }

  784.     /** 
  785.      * Calls the API using cURL 
  786.      * 
  787.      * @param string          $httpmethod      The HTTP method to use for making the request 
  788.      * @param string          $method          The API method to call 
  789.      * @param string          $method_template The templated API method to call 
  790.      * @param array  optional $params          The parameters to send along 
  791.      * @param bool   optional $multipart       Whether to use multipart/form-data 
  792.      * @param bool   optional $app_only_auth   Whether to use app-only bearer authentication 
  793.      * 
  794.      * @return mixed The API reply, encoded in the set return_format 
  795.      */ 
  796.  
  797.     protected function _callApi($httpmethod, $method, $method_template, $params = array(), $multipart = false, $app_only_auth = false) 
  798.     { 
  799.         if (! function_exists('curl_init')) { 
  800.             throw new \Exception('To make API requests, the PHP curl extension must be available.'); 
  801.         } 
  802.         $url = $this->_getEndpoint($method, $method_template); 
  803.         $ch = false; 
  804.         if ($httpmethod == 'GET') { 
  805.             $url_with_params = $url; 
  806.             if (count($params) > 0) { 
  807.                 $url_with_params .= '?' . http_build_query($params); 
  808.             } 
  809.             $authorization = $this->_sign($httpmethod, $url, $params); 
  810.             $ch = curl_init($url_with_params); 
  811.         } else { 
  812.             if ($multipart) { 
  813.                 $authorization = $this->_sign($httpmethod, $url, array()); 
  814.                 $params = $this->_buildMultipart($method_template, $params); 
  815.             } else { 
  816.                 $authorization = $this->_sign($httpmethod, $url, $params); 
  817.                 $params = http_build_query($params); 
  818.             } 
  819.             $ch = curl_init($url); 
  820.             curl_setopt($ch, CURLOPT_POST, 1); 
  821.             curl_setopt($ch, CURLOPT_POSTFIELDS, $params); 
  822.         } 
  823.         if ($app_only_auth) { 
  824.             if (self::$_oauth_consumer_key == null) { 
  825.                 throw new \Exception('To make an app-only auth API request, the consumer key must be set.'); 
  826.             } 
  827.             // automatically fetch bearer token, if necessary 
  828.             if (self::$_oauth_bearer_token == null) { 
  829.                 $this->oauth2_token(); 
  830.             } 
  831.             $authorization = 'Authorization: Bearer ' . self::$_oauth_bearer_token; 
  832.         } 
  833.         $request_headers = array(); 
  834.         if (isset($authorization)) { 
  835.             $request_headers[] = $authorization; 
  836.             $request_headers[] = 'Expect:'; 
  837.         } 
  838.         if ($multipart) { 
  839.             $first_newline = strpos($params, "\r\n"); 
  840.             $multipart_boundary = substr($params, 2, $first_newline - 2); 
  841.             $request_headers[] = 'Content-Length: ' . strlen($params); 
  842.             $request_headers[] = 'Content-Type: multipart/form-data; boundary=' 
  843.                 . $multipart_boundary; 
  844.         } 
  845.  
  846.         curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1); 
  847.         curl_setopt($ch, CURLOPT_FOLLOWLOCATION, 0); 
  848.         curl_setopt($ch, CURLOPT_HEADER, 1); 
  849.         curl_setopt($ch, CURLOPT_HTTPHEADER, $request_headers); 
  850.         curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, 1); 
  851.         curl_setopt($ch, CURLOPT_SSL_VERIFYHOST, 2); 
  852.         curl_setopt($ch, CURLOPT_CAINFO, __DIR__ . '/cacert.pem'); 
  853.  
  854.         $reply = curl_exec($ch); 
  855.  
  856.         // certificate validation results 
  857.         $validation_result = curl_errno($ch); 
  858.         if (in_array( 
  859.                 $validation_result,  
  860.                 array( 
  861.                     CURLE_SSL_CERTPROBLEM,  
  862.                     CURLE_SSL_CACERT,  
  863.                     CURLE_SSL_CACERT_BADFILE,  
  864.                     CURLE_SSL_CRL_BADFILE,  
  865.                     CURLE_SSL_ISSUER_ERROR 
  866.  ) 
  867.  ) 
  868.  ) { 
  869.             throw new \Exception('Error ' . $validation_result . ' while validating the Twitter API certificate.'); 
  870.         } 
  871.  
  872.         $httpstatus = curl_getinfo($ch, CURLINFO_HTTP_CODE); 
  873.         $reply = $this->_parseApiReply($method_template, $reply); 
  874.         if ($this->_return_format == CODEBIRD_RETURNFORMAT_OBJECT) { 
  875.             $reply->httpstatus = $httpstatus; 
  876.         } elseif ($this->_return_format == CODEBIRD_RETURNFORMAT_ARRAY) { 
  877.             $reply['httpstatus'] = $httpstatus; 
  878.         } 
  879.         return $reply; 
  880.     }

  882.     /** 
  883.      * Parses the API reply to encode it in the set return_format 
  884.      * 
  885.      * @param string $method The method that has been called 
  886.      * @param string $reply  The actual reply, JSON-encoded or URL-encoded 
  887.      * 
  888.      * @return array|object The parsed reply 
  889.      */ 
  890.     protected function _parseApiReply($method, $reply) 
  891.     { 
  892.         // split headers and body 
  893.         $headers = array(); 
  894.         $reply = explode("\r\n\r\n", $reply, 4); 
  895.  
  896.         // check if using proxy 
  897.         if (substr($reply[0], 0, 35) === 'HTTP/1.1 200 Connection Established') { 
  898.             array_shift($reply); 
  899.         } elseif (count($reply) > 2) { 
  900.             $headers = array_shift($reply); 
  901.             $reply = array( 
  902.                 $headers,  
  903.                 implode("\r\n", $reply) 
  904.  ); 
  905.         } 
  906.  
  907.         $headers_array = explode("\r\n", $reply[0]); 
  908.         foreach ($headers_array as $header) { 
  909.             $header_array = explode(': ', $header, 2); 
  910.             $key = $header_array[0]; 
  911.             $value = ''; 
  912.             if (count($header_array) > 1) { 
  913.                 $value = $header_array[1]; 
  914.             } 
  915.             $headers[$key] = $value; 
  916.         } 
  917.         if (count($reply) > 1) { 
  918.             $reply = $reply[1]; 
  919.         } else { 
  920.             $reply = ''; 
  921.         } 
  922.  
  923.         $need_array = $this->_return_format == CODEBIRD_RETURNFORMAT_ARRAY; 
  924.         if ($reply == '[]') { 
  925.             switch ($this->_return_format) { 
  926.                 case CODEBIRD_RETURNFORMAT_ARRAY: 
  927.                     return array(); 
  928.                 case CODEBIRD_RETURNFORMAT_JSON: 
  929.                     return '{}'; 
  930.                 case CODEBIRD_RETURNFORMAT_OBJECT: 
  931.                     return new \stdClass; 
  932.             } 
  933.         } 
  934.         $parsed = array(); 
  935.         if (! $parsed = json_decode($reply, $need_array)) { 
  936.             if ($reply) { 
  937.                 if (stripos($reply, '<' . '?xml version="1.0" encoding="UTF-8"?' . '>') === 0) { 
  938.                     // we received XML... 
  939.                     // since this only happens for errors,  
  940.                     // don't perform a full decoding 
  941.                     preg_match('/<request>(.*)<\/request>/', $reply, $request); 
  942.                     preg_match('/<error>(.*)<\/error>/', $reply, $error); 
  943.                     $parsed['request'] = htmlspecialchars_decode($request[1]); 
  944.                     $parsed['error'] = htmlspecialchars_decode($error[1]); 
  945.                 } else { 
  946.                     // assume query format 
  947.                     $reply = explode('&', $reply); 
  948.                     foreach ($reply as $element) { 
  949.                         if (stristr($element, '=')) { 
  950.                             list($key, $value) = explode('=', $element); 
  951.                             $parsed[$key] = $value; 
  952.                         } else { 
  953.                             $parsed['message'] = $element; 
  954.                         } 
  955.                     } 
  956.                 } 
  957.             } 
  958.             $reply = json_encode($parsed); 
  959.         } 
  960.         switch ($this->_return_format) { 
  961.             case CODEBIRD_RETURNFORMAT_ARRAY: 
  962.                 return $parsed; 
  963.             case CODEBIRD_RETURNFORMAT_JSON: 
  964.                 return $reply; 
  965.             case CODEBIRD_RETURNFORMAT_OBJECT: 
  966.                 return (object) $parsed; 
  967.         } 
  968.         return $parsed; 
  969.     } 
  970.
