  280.   function save_extra_currencies( $value ) { 
  281.  
  282.       $fields = array( 
  283.           'labels',  
  284.           'symbols',  
  285.           'codes' 
  286.  ); 
  287.  
  288.       $error = false; 
  289.  
  290.  
  291.       foreach( $fields as $f ) { 
  292.            
  293.           if( !empty( $_POST[ $f ] ) && !is_array( $_POST[ $f ] ) ) { 
  294.               WC_Admin_Settings::add_error( "Couldn't save. Please try again" ); 
  295.               $error = true; 
  296.               continue; 
  297.           } 
  298.  
  299.       } 
  300.  
  301.       if( $error ) 
  302.           return;

  310.  
  311.           update_option( '_wc_custom_currencies_' . $f, $_POST[ $f ] );

  313.  
  314.
