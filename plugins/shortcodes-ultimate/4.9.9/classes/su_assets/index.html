  206.     /** 
  207.      * Get queried assets 
  208.      */ 
  209.     public static function assets() { 
  210.         // Get assets query 
  211.         $assets = self::$assets; 
  212.         // Apply filters to assets set 
  213.         $assets['css'] = array_unique( (array) apply_filters( 'su/assets/css', (array) array_unique( $assets['css'] ) ) ); 
  214.         $assets['js'] = array_unique( (array) apply_filters( 'su/assets/js', (array) array_unique( $assets['js'] ) ) ); 
  215.         // Return set 
  216.         return $assets; 
  217.     }
