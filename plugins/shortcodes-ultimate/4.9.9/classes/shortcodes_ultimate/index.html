  197.     /** 
  198.      * Register shortcodes 
  199.      */ 
  200.     public static function register() { 
  201.         // Prepare compatibility mode prefix 
  202.         $prefix = su_cmpt(); 
  203.         // Loop through shortcodes 
  204.         foreach ( (array) Su_Data::shortcodes() as $id => $data ) { 
  205.             if ( isset( $data['function'] ) && is_callable( $data['function'] ) ) $func = $data['function']; 
  206.             elseif ( is_callable( array( 'Su_Shortcodes', $id ) ) ) $func = array( 'Su_Shortcodes', $id ); 
  207.             elseif ( is_callable( array( 'Su_Shortcodes', 'su_' . $id ) ) ) $func = array( 'Su_Shortcodes', 'su_' . $id ); 
  208.             else continue; 
  209.             // Register shortcode 
  210.             add_shortcode( $prefix . $id, $func ); 
  211.         } 
  212.         // Register [media] manually // 3.x 
  213.         add_shortcode( $prefix . 'media', array( 'Su_Shortcodes', 'media' ) ); 
  214.     }
