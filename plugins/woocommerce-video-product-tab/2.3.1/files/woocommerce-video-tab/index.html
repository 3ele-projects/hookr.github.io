  182.           private function product_has_video_tabs($product) { 
  183.               if($this->tab_data === false) {

  303.           /** 
  304.            * Saves the data inputed into the product boxes, as post meta data 
  305.            * identified by the name 'woo_video_product_tab' 
  306.            *  
  307.            * @param int $post_id the post (product) identifier 
  308.            * @param stdClass $post the post (product) 
  309.            */ 
  310.           public function product_save_data($post_id, $post) { 
  311.  
  312.               $tab_title = stripslashes($_POST['_tab_video_title']); 
  313.               if($tab_title == '') { 
  314.                   $tab_title = __('Video', 'wc_video_product_tab'); 
  315.               } 
  316.               $hide_title = stripslashes($_POST['_hide_title']); 
  317.               $tab_video = stripslashes($_POST['_tab_video']); 
  318.  
  319.               if(empty($tab_video) && get_post_meta($post_id, 'woo_video_product_tab', true)) { 
  320.                   // clean up if the video tabs are removed 
  321.                   delete_post_meta($post_id, 'woo_video_product_tab'); 
  322.               } 
  323.               elseif(!empty($tab_video)) { 
  324.                   $tab_data = array(); 
  325.  
  326.                   $tab_id = ''; 
  327.                   // convert the tab title into an id string 
  328.                   $tab_id = strtolower($tab_title); 
  329.                   $tab_id = preg_replace("/[^\w\s]/", '', $tab_id); // remove non-alphas, numbers, underscores or whitespace  
  330.                   $tab_id = preg_replace("/_+/", ' ', $tab_id); // replace all underscores with single spaces 
  331.                   $tab_id = preg_replace("/\s+/", '-', $tab_id); // replace all multiple spaces with single dashes 
  332.                   $tab_id = 'tab-'.$tab_id; // prepend with 'tab-' string 
  333.  
  334.                   // save the data to the database 
  335.                   $tab_data[] = array('title' => $tab_title, 'hide_title' => $hide_title, 'id' => $tab_id, 'video' => $tab_video); 
  336.                   update_post_meta($post_id, 'woo_video_product_tab', $tab_data); 
  337.               } 
  338.           } 
  339.       }
