  296.   /** 
  297.    * Processes post meta fields and creates a map, if needed. 
  298.    */ 
  299.   function process_post_meta ($post_id) { 
  300.       if (!$post_id) return false; 
  301.  
  302.       $post = get_post($post_id); 
  303.       if ('publish' != $post->post_status) return false; // Draft, auto-save or something else we don't want 
  304.  
  305.       $map_id = get_post_meta($post_id, 'agm_map_created', true); 
  306.       if ($map_id) return true; // Already have a map, nothing to do 
  307.  
  308.       $opts = get_option('agm_google_maps'); 
  309.       $fields = $opts['custom_fields_map']; 
  310.       $latitude = $longitude = $address = false; 
  311.       if ($fields['latitude_field']) $latitude = get_post_meta($post_id, $fields['latitude_field'], true); 
  312.       if ($fields['longitude_field']) $longitude = get_post_meta($post_id, $fields['longitude_field'], true); 
  313.       if ($fields['address_field']) $address = get_post_meta($post_id, $fields['address_field'], true); 
  314.  
  315.       if (!$latitude && !$longitude && !$address) return false; // Nothing to process 
  316.  
  317.       return $this->model->autocreate_map($post_id, $latitude, $longitude, $address); 
  318.   }
