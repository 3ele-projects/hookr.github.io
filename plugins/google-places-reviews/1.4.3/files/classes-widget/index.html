  389.   /** 
  390.    * cURL (wp_remote_get) the Google Places API 
  391.    * 
  392.    * @description: CURLs the Google Places API with our url parameters and returns JSON response 
  393.    * 
  394.    * @param $url 
  395.    * 
  396.    * @return array|mixed 
  397.    */ 
  398.   function get_reviews( $url ) { 
  399.  
  400.       //Sanitize the URL 
  401.       $url = esc_url_raw( $url ); 
  402.  
  403.       // Send API Call using WP's HTTP API 
  404.       $data = wp_remote_get( $url ); 
  405.  
  406.       if ( is_wp_error( $data ) ) { 
  407.           $error_message = $data->get_error_message(); 
  408.           $this->output_error_message( "Something went wrong: $error_message", 'error' ); 
  409.       } 
  410.  
  411.       //Use curl only if necessary 
  412.       if ( empty( $data['body'] ) ) { 
  413.  
  414.           $ch = curl_init( $url ); 
  415.           curl_setopt( $ch, CURLOPT_RETURNTRANSFER, true ); 
  416.           curl_setopt( $ch, CURLOPT_HEADER, 0 ); 
  417.           $data = curl_exec( $ch ); // Google response 
  418.           curl_close( $ch ); 
  419.           $response = json_decode( $data, true ); 
  420.  
  421.       } else { 
  422.           $response = json_decode( $data['body'], true ); 
  423.       } 
  424.  
  425.       //Get Reviewers Avatars 
  426.       $response = $this->get_reviewers_avatars( $response ); 
  427.  
  428.       //Get Business Avatar 
  429.       $response = $this->get_business_avatar( $response ); 
  430.  
  431.  
  432.       //Google response data in JSON format 
  433.       return $response; 
  434.  
  435.   }

  440.    * Get avatar from Places API response or provide placeholder.

  443.    */ 
  444.   function get_reviewers_avatars( $response ) { 
  445.       // GPR Reviews Array. 
  446.       $gpr_reviews = array(); 
  447.  
  448.       // Includes Avatar image from user. 
  449.       if ( isset( $response['result']['reviews'] ) && ! empty( $response['result']['reviews'] ) ) { 
  450.  
  451.           // Loop Google Places reviews. 
  452.           foreach ( $response['result']['reviews'] as $review ) { 
  453.               // Check to see if image is empty (no broken images). 
  454.               if ( ! empty( $review['profile_photo_url'] ) ) { 
  455.                   $avatar_img = $review['profile_photo_url'] . '?sz=100'; 
  456.               } else { 
  457.                   $avatar_img = GPR_PLUGIN_URL . '/assets/images/mystery-man.png'; 
  458.               } 
  459.  
  460.               // Add array image to review array. 
  461.               $review = array_merge( $review, array( 'avatar' => $avatar_img ) ); 
  462.               // Add full review to $gpr_views. 
  463.               array_push( $gpr_reviews, $review ); 
  464.           } 
  465.  
  466.           // Merge custom reviews array with response. 
  467.           $response = array_merge( $response, array( 'gpr_reviews' => $gpr_reviews ) ); 
  468.       } 
  469.  
  470.       return $response; 
  471.   }

  628.   /** 
  629.    * Time Since 
  630.    * Works out the time since the entry post, takes a an argument in unix time (seconds) 
  631.    */ 
  632.   static public function get_time_since( $date, $granularity = 1 ) { 
  633.       $difference = time() - $date; 
  634.       $retval = ''; 
  635.       $periods = array( 
  636.           'decade' => 315360000,  
  637.           'year' => 31536000,  
  638.           'month' => 2628000,  
  639.           'week' => 604800,  
  640.           'day' => 86400,  
  641.           'hour' => 3600,  
  642.           'minute' => 60,  
  643.           'second' => 1 
  644.  ); 
  645.  
  646.       foreach ( $periods as $key => $value ) { 
  647.           if ( $difference >= $value ) { 
  648.               $time = floor( $difference / $value ); 
  649.               $difference %= $value; 
  650.               $retval .= ( $retval ? ' ' : '' ) . $time . ' '; 
  651.               $retval .= ( ( $time > 1 ) ? $key . 's' : $key ); 
  652.               $granularity --; 
  653.           } 
  654.           if ( $granularity == '0' ) { 
  655.               break; 
  656.           } 
  657.       } 
  658.  
  659.       return ' posted ' . $retval . ' ago'; 
  660.   }
