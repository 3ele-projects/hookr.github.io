  347.     function grp_plugin_curl( $url ) { 
  348.  
  349.         // Send API Call using WP's HTTP API 
  350.         $data = wp_remote_get( $url ); 
  351.  
  352.         if ( is_wp_error( $data ) ) { 
  353.             $error_message = $data->get_error_message(); 
  354.             $this->output_error_message( "Something went wrong: $error_message", 'error' ); 
  355.         } 
  356.  
  357.         //Use curl only if necessary 
  358.         if ( empty( $data['body'] ) ) { 
  359.  
  360.             $ch = curl_init( $url ); 
  361.             curl_setopt( $ch, CURLOPT_RETURNTRANSFER, true ); 
  362.             curl_setopt( $ch, CURLOPT_HEADER, 0 ); 
  363.             $data = curl_exec( $ch ); // Google response 
  364.             curl_close( $ch ); 
  365.             $response = json_decode( $data, true ); 
  366.  
  367.         } else { 
  368.             $response = json_decode( $data['body'], true ); 
  369.         } 
  370.  
  371.         //GPR Reviews Array 
  372.         $gpr_reviews = array(); 
  373.  
  374.         //includes Avatar image from user

  376.         if ( isset( $response['result']['reviews'] ) && ! empty( $response['result']['reviews'] ) ) { 
  377.             //Loop Google Places reviews 
  378.             foreach ( $response['result']['reviews'] as $review ) { 
  379.  
  380.                 $user_id = isset( $review['author_url'] ) ? str_replace( 'https://plus.google.com/', '', $review['author_url'] ) : '';

  383.                 $request_url = add_query_arg( 
  384.                     array( 
  385.                         'alt' => 'json',

  559.     /** 
  560.      * Time Since 
  561.      * Works out the time since the entry post, takes a an argument in unix time (seconds) 
  562.      */ 
  563.     static public function get_time_since( $date, $granularity = 1 ) { 
  564.         $difference = time() - $date; 
  565.         $retval = ''; 
  566.         $periods = array( 
  567.             'decade' => 315360000,  
  568.             'year' => 31536000,  
  569.             'month' => 2628000,  
  570.             'week' => 604800,  
  571.             'day' => 86400,  
  572.             'hour' => 3600,  
  573.             'minute' => 60,  
  574.             'second' => 1 
  575.  ); 
  576.  
  577.         foreach ( $periods as $key => $value ) { 
  578.             if ( $difference >= $value ) { 
  579.                 $time = floor( $difference / $value ); 
  580.                 $difference %= $value; 
  581.                 $retval .= ( $retval ? ' ' : '' ) . $time . ' '; 
  582.                 $retval .= ( ( $time > 1 ) ? $key . 's' : $key ); 
  583.                 $granularity --; 
  584.             } 
  585.             if ( $granularity == '0' ) { 
  586.                 break; 
  587.             } 
  588.         } 
  589.  
  590.         return ' posted ' . $retval . ' ago'; 
  591.     }
