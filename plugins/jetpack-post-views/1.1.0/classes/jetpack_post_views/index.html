  737.     * SCHEDULED UPDATE 
  738.     * Update all posts 'jetpack-post-views' post meta 
  739.     */ 
  740.     function get_post_views() { 
  741.         $post_stats = array(); 
  742.         $postdata = array(); 
  743.  
  744.         // Hack to break cache suggested by Glisse. 
  745.         // Since 'limit' is capped at 500, use it to break cache for each interval. 
  746.         // http://wordpress.org/support/topic/post-views-not-updating 
  747.         $random = rand( 500, 99999 ); 
  748.  
  749.         // Get stats using the stats_get_csv function 
  750.         if ( function_exists('stats_get_csv') && $this->settings['use_stats_get_csv'] ) { 
  751.  
  752.             // Get post views for each interval: Unlimited, Day, Week, Month, Year 
  753.             foreach ($this->interval as $key => $value) { 
  754.                 $args = array( 
  755.                     'days' => $value,  
  756.                     'limit' => $random,  
  757.                     'summarize' => 1 
  758.  ); 
  759.                 $post_stats[$key] = stats_get_csv( 'postviews', $args ); 
  760.             } 
  761.  
  762.             $this->updateMessages[-2] = "Stats being updated using the stats_get_csv function"; 
  763.         } 
  764.         // Get the stats using the blog_uri or blog_id 
  765.         else { 
  766.             $args = $this->apiArgs; 
  767.             $args['api_key'] = $this->settings['api_key']; 
  768.             $args['limit'] = $random; 
  769.  
  770.             // Get the stats using the blog_uri 
  771.             if ( $this->settings['connect_blog_uri'] && $this->settings['use_blog_uri'] ) { 
  772.                 $args['blog_uri'] = $this->settings['blog_uri']; 
  773.                 $this->updateMessages[-2] = "Stats being updated using blog_uri"; 
  774.             } 
  775.             // Get the stats using the blog_id 
  776.             else if ( $this->settings['connect_blog_id'] ) { 
  777.                 $args['blog_id'] = $this->settings['blog_id']; 
  778.                 $this->updateMessages[-2] = "Stats being updated using blog_id"; 
  779.             } 
  780.             else { 
  781.                 return; 
  782.             } 
  783.  
  784.             // Get post views for each interval: Unlimited, Day, Week, Month, Year 
  785.             foreach ($this->interval as $key => $value) { 
  786.                 $args['days'] = $value; 
  787.                 $url = add_query_arg( $args, $this->apiUrl ); 
  788.                 $json = wp_remote_get( $url ); 
  789.                 $data = json_decode( $json['body'], true ); 
  790.                 $post_stats[$key] = $data[0]['postviews']; 
  791.             } 
  792.         } 
  793.  
  794.         $this->updateMessages[-1] = $post_stats; 
  795.         global $post; 
  796.  
  797.         // Create an array indexed by interval and then post_id 
  798.         foreach ( $post_stats as $key => $value ) { 
  799.             $postdata[$key] = array(); 
  800.             foreach ( $value as $postinfo ) { 
  801.                 $postdata[$key][$postinfo['post_id']] = strip_tags( $postinfo['views'] ); 
  802.             } 
  803.         } 
  804.  
  805.         // Get all posts and update them 
  806.         $post_types = get_post_types( array( '_builtin' => false ), 'names' ); 
  807.         $post_types['post'] = 'post'; 
  808.  
  809.         $args = array( 
  810.             'numberposts' => -1,  
  811.             'post_type' => $post_types,  
  812.             'post_status' => 'publish' 
  813.  ); 
  814.         $allposts = get_posts( $args ); 
  815.         foreach( $allposts as $post) { 
  816.             $message = "'".get_the_title( $post->ID )."' updated with stats:"; 
  817.  
  818.             // Cannot save the interval data as an array in a meta key because WP cannot sort post queries by an array. 
  819.             // Have to store each interval data in it's own meta key. 
  820.             // http://wordpress.stackexchange.com/questions/99149/wp-query-meta-query-by-array-key 
  821.             foreach ( $this->interval as $key => $value ) { 
  822.  
  823.                 // Ensure that the $post-ID exists as a key in the array before trying to update post information. 
  824.                 // This prevents the user from pulling data from another website and trying to add it to their own 
  825.                 // posts. 
  826.                 if ( array_key_exists( $post->ID, $postdata[$key] ) ) { 
  827.                     $newViews = intval( $postdata[$key][ $post->ID ] ); 
  828.  
  829.                     // Unlimited meta key different as to not break sites that used this previously 
  830.                     if ( $key == 'Unlimited' ) { 
  831.                         $oldViews = get_post_meta( $post->ID, 'jetpack-post-views', true ); 
  832.  
  833.                         // Only update posts with new stats. Prevents posts from being updated with '0' views when the API service is down. 
  834.                         if ( $newViews < $oldViews ) { 
  835.                             $newViews = $oldViews; 
  836.                         } 
  837.  
  838.                         update_post_meta( $post->ID, 'jetpack-post-views', $newViews ); 
  839.                     } 
  840.                     else { 
  841.                         // Always update non-unlimited intervals since they can fluctuate 
  842.                         update_post_meta( $post->ID, 'jetpack-post-views-'.$key, $newViews ); 
  843.                     } 
  844.  
  845.                     $message .= " $key($newViews)"; 
  846.                 } 
  847.                 else { 
  848.                     $message .= " $key(not updated)"; 
  849.                 } 
  850.             } 
  851.  
  852.             $this->updateMessages[ $post->ID ] = $message; 
  853.         } 
  854.     }

  859.  
  860.         // Delete old options 
  861.         delete_option( 'jetpack-post-views_version' ); 
  862.         delete_option( 'jetpack_post_views_wp_api_key' );

  864.  
  865.         // Add post meta for each post 
  866.         $post_types = get_post_types( array( '_builtin' => false ), 'names' ); 
  867.         $post_types['post'] = 'post'; 
  868.  
  869.         $args = array( 
  870.             'numberposts' => -1,  
  871.             'post_type' => $post_types,  
  872.             'post_status' => 'publish' 
  873.  ); 
  874.         $allposts = get_posts( $args ); 
  875.         foreach( $allposts as $post) { 
  876.             foreach ( $this->interval as $key => $value ) { 
  877.                 $meta_key = ''; 
  878.                 if ( $key == 'Unlimited' ) { 
  879.                     $meta_key = 'jetpack-post-views'; 
  880.                 } 
  881.                 else { 
  882.                     $meta_key = 'jetpack-post-views-'.$key; 
  883.                 } 
  884.  
  885.                 $post_meta = get_post_meta( $post->ID, $meta_key, true ); 
  886.  
  887.                 // Only add post_meta if not already defined 
  888.                 if (empty($post_meta)) { 
  889.                     add_post_meta( $post->ID, $meta_key, 0, true ); 
  890.                 } 
  891.             } 
  892.         } 
  893.  
  894.         $this->get_post_views(); 
  895.  
  896.         update_option(JETPACK_POST_VIEWS_VERSION_KEY, $this->version);
