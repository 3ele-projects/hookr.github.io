  480.         if ( function_exists('stats_get_csv') ) { 
  481.             $post_stats = stats_get_csv('postviews', 'days=-1&limit=-1&summarize'); 
  482.             $this->updateMessages[ -1 ] = "Stats being updated using the stats_get_csv function"; 
  483.             $run_stats = 1; 
  484.         } 
  485.         // Else get the stats using the blog_uri 
  486.         else if ( $this->settings['connect_blog_uri'] ) { 
  487.             $this->updateMessages[ -1 ] = "Stats being updated using blog_uri"; 
  488.             $middle = '?api_key='.$this->settings['api_key'].'&blog_uri='.$this->settings['blog_uri']; 
  489.             $url = $this->apiUrlBefore.$middle.$this->apiUrlAfter; 
  490.             $json = file_get_contents( $url ); 
  491.             $data = json_decode( $json, true ); 
  492.             $post_stats = $data[0]['postviews']; 
  493.             $run_stats = 1; 
  494.         } 
  495.         // Otherwise get the stats using the blog_id 
  496.         else if ( $this->settings['connect_blog_id'] ) { 
  497.             $this->updateMessages[ -1 ] = "Stats being updated using blog_id"; 
  498.             $middle = '?api_key='.$this->settings['api_key'].'&blog_uri='.$this->settings['blog_uri']; 
  499.             $url = $this->apiUrlBefore.$middle.$this->apiUrlAfter; 
  500.             $json = file_get_contents( $url ); 
  501.             $data = json_decode( $json, true ); 
  502.             $post_stats = $data[0]['postviews']; 
  503.             $run_stats = 1; 
  504.         } 
  505.  
  506.         if ( $run_stats ) { 
  507.             global $post; 
  508.  
  509.             // Create a temporary array and save all data 
  510.             foreach ( $post_stats as $postinfo ) { 
  511.                 $postdata[ $postinfo['post_id'] ] = strip_tags( $postinfo['views'] ); 
  512.             } 
  513.  
  514.             // Grab all posts and update them 
  515.             $args = array( 'numberposts' => -1, 'post_type' => 'post', 'post_status' => 'publish'); 
  516.             $allposts = get_posts( $args ); 
  517.             foreach( $allposts as $post) { 
  518.                 // Ensure that the $post-ID exists as a key in the array before trying to update post information. 
  519.                 // This prevents the user from pulling data from another website and trying to add it to their own 
  520.                 // posts.  
  521.                 if ( array_key_exists( $post->ID, $postdata) ) { 
  522.                     $newViews = intval( $postdata[ $post->ID ] ); 
  523.                      $oldViews = get_post_meta( $post->ID, 'jetpack-post-views', true ); 
  524.  
  525.                     // Only update posts with new stats. Prevents posts from being updated with '0' views when 
  526.                     // the API service is down.  
  527.                     if ( $oldViews >= 0 && $newViews > $oldViews ) { 
  528.                         $this->updateMessages[ $post->ID ] = "'".get_the_title( $post->ID )."' succesfully updated with newViews(".$newViews.")"; 
  529.                         update_post_meta( $post->ID, 'jetpack-post-views', $newViews ); 
  530.                     } 
  531.                     else { 
  532.                         if (!$oldViews) { 
  533.                             $this->updateMessages[ $post->ID ] = "'".get_the_title( $post->ID )."' failed to update because oldViews(".$oldViews.") is 0 or does not exist"; 
  534.                         } 
  535.                         else { 
  536.                             $this->updateMessages[ $post->ID ] = "'".get_the_title( $post->ID )."' did not update because newViews(".$newViews.") <= oldViews(".$oldViews.")"; 
  537.                         } 
  538.                         add_post_meta( $post->ID, 'jetpack-post-views', ($newViews ? $newViews: 0), true ); 
  539.                     } 
  540.                 } 
  541.                 else { 
  542.                     $this->updateMessages[ $post->ID ] = "'".get_the_title( $post->ID )."' failed to update because post->ID is not in postData"; 
  543.                     add_post_meta( $post->ID, 'jetpack-post-views', 0, true ); 
  544.                 } 
  545.             } 
  546.         } 
  547.     }
