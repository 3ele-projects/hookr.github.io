  559.             foreach ($rout as $row) {

  587.                 foreach ($cols as $col) {

  589.                         foreach ($rs as $row) { 
  590.                             $nrow[] = $row[$col['id']]; 
  591.                             break; 
  592.                         } 
  593.                     } 
  594.                 }

  597.                     $row = isset($rs[$pivot]) ? $rs[$pivot] : NULL; 
  598.                     foreach ($cols as $col) {

  601.                         } 
  602.                     } 
  603.                 } 
  604.  
  605.                 $nrows[] = $nrow;

  609.             $rout = $nrows; 
  610.         } 
  611.  
  612.         $this->setup_rownums($query, count($rout)); 
  613.  
  614.         $rows = array_slice($rout, $this->first_row, $this->num_rows); 
  615.         return $cols; 
  616.     }
