  185.   public function get_tracktrace_url($order_id) {

  188.  
  189.       $tracktrace = get_post_meta($order_id, '_flespakket_tracktrace', true); 
  190.       $postcode = preg_replace('/\s+/', '', get_post_meta($order_id, '_shipping_postcode', true)); 
  191.       $tracktrace_url = sprintf('https://mijnpakket.postnl.nl/Claim?Barcode=%s&Postalcode=%s', $tracktrace, $postcode);
