  729.     /** 
  730.      * Marks a reply as reported 
  731.      * 
  732.      * @since bbPress (r2740) 
  733.      * 
  734.      * @param int $reply_id Reply id 
  735.      * @uses bbp_get_reply() To get the reply 
  736.      * @uses do_action() Calls 'bbp_rc_reported_reply' with the reply ID 
  737.      * @uses add_post_meta() To add the previous status to a meta 
  738.      * @uses wp_update_post() To insert the updated post 
  739.      * @uses do_action() Calls 'bbp_rc_reported_reply' with the reply ID 
  740.      * @return mixed False or {@link WP_Error} on failure, reply id on success 
  741.      */ 
  742.     function report_reply( $reply_id = 0 ) { 
  743.  
  744.         // Get reply 
  745.         $reply = bbp_get_reply( $reply_id ); 
  746.         if ( empty( $reply ) ) 
  747.             return $reply; 
  748.  
  749.         // Bail if already reported 
  750.         if ( $this->get_reported_status_id() === $reply->post_status ) 
  751.             return false; 
  752.  
  753.         // Execute pre report code 
  754.         do_action( 'bbp_rc_report_reply', $reply_id ); 
  755.  
  756.         // Add the user id of the user who reported 
  757.         update_post_meta( $reply_id, '_bbp_report_user_id', wp_get_current_user()->ID ); 
  758.  
  759.         // Add the original post status as post meta for future restoration 
  760.         add_post_meta( $reply_id, '_bbp_report_meta_status', $reply->post_status ); 
  761.  
  762.         // Set post status to report 
  763.         $reply->post_status = $this->get_reported_status_id(); 
  764.  
  765.         // No revisions 
  766.         remove_action( 'pre_post_update', 'wp_save_post_revision' ); 
  767.  
  768.         // Update the reply 
  769.         $reply_id = wp_update_post( $reply ); 
  770.  
  771.         // Execute post report code 
  772.         do_action( 'bbp_rc_reported_reply', $reply_id ); 
  773.  
  774.         // Return reply_id 
  775.         return $reply_id; 
  776.     }

  792.     function unreport_reply( $reply_id = 0 ) {

  812.  
  813.         // If no previous status, default to publish 
  814.         if ( empty( $reply->post_status ) ) { 
  815.             $reply->post_status = bbp_get_public_status_id(); 
  816.         } 
  817.  
  818.         // Delete pre report meta 
  819.         delete_post_meta( $reply_id, '_bbp_report_meta_status' ); 
  820.  
  821.         // Add the user id of the user who reported 
  822.         delete_post_meta( $reply_id, '_bbp_report_user_id' ); 
  823.  
  824.         // No revisions 
  825.         remove_action( 'pre_post_update', 'wp_save_post_revision' ); 
  826.  
  827.         // Update the reply 
  828.         $reply_id = wp_update_post( $reply ); 
  829.  
  830.         // Execute post unreport code 
  831.         do_action( 'bbp_rc_unreported_reply', $reply_id ); 
  832.  
  833.         // Return reply_id 
  834.         return $reply_id; 
  835.     }
