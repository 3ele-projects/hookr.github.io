  146.               } 
  147.  
  148.               if ( empty( $addons[ $key ]->action ) ) { 
  149.                   $addons[ $key ]->action = array(); 
  150.                   $addons[ $key ]->action[] = array( 
  151.                       'id' => 'ms-toggle-' . $key,  
  152.                       'type' => MS_Helper_Html::INPUT_TYPE_RADIO_SLIDER,  
  153.                       'value' => self::is_enabled( $key ),

  155.                       'ajax_data' => array( 
  156.                           'action' => MS_Controller_Addon::AJAX_ACTION_TOGGLE_ADDON,  
  157.                           'field' => 'active',  
  158.                           'addon' => $key,  
  159.  ),  
  160.  ); 
  161.                   $addons[ $key ]->action[] = MS_Helper_Html::save_text( null, false, true ); 
  162.               }

  165.                * Add custom Actions or remove default actions 
  166.                * 
  167.                * @since  1.0.0 
  168.                */ 
  169.               $addons[ $key ]->action = apply_filters( 
  170.                   'ms_model_addon_action-' . $key,

  172.                   $addons[ $key ]

  211.   static protected function load_core_addons() { 
  212.       $model = MS_Factory::load( 'MS_Model_Addon' ); 
  213.       $content_dir = trailingslashit( dirname( dirname( MS_Plugin::instance()->dir ) ) ); 
  214.       $plugin_dir = substr( MS_Plugin::instance()->dir, strlen( $content_dir ) ); 
  215.  
  216.       $addon_dirs = array( 
  217.           $plugin_dir . 'premium/addon/',  
  218.           $plugin_dir . 'app/addon/',  
  219.  ); 
  220.  
  221.       if ( empty( $model->addon_files ) || self::$_reload_files ) { 
  222.           // In Admin dashboard we always refresh the addon-list... 
  223.           self::$_reload_files = false; 
  224.           $model->addon_files = array(); 
  225.  
  226.           foreach ( $addon_dirs as $addon_dir ) { 
  227.               $mask = $content_dir . $addon_dir . '*/class-ms-addon-*.php'; 
  228.               $addons = glob( $mask ); 
  229.  
  230.               foreach ( $addons as $file ) { 
  231.                   $model->addon_files[] = substr( $file, strlen( $content_dir ) ); 
  232.               } 
  233.           } 
  234.  
  235.           /** 
  236.            * Allow other plugins/themes to register custom addons 
  237.            * 
  238.            * @since  1.0.0 
  239.            * 
  240.            * @var array 
  241.            */ 
  242.           $model->addon_files = apply_filters( 
  243.               'ms_model_addon_files',  
  244.               $model->addon_files 
  245.  ); 
  246.  
  247.           $model->save(); 
  248.       } 
  249.  
  250.       // Loop all recignized Add-ons and initialize them. 
  251.       foreach ( $model->addon_files as $file ) { 
  252.           $addon = $content_dir . $file; 
  253.  
  254.           // Get class-name from file-name 
  255.           $class = basename( $file ); 
  256.           $class = str_replace( '.php', '', $class ); 
  257.           $class = implode( '_', array_map( 'ucfirst', explode( '-', $class ) ) ); 
  258.           $class = substr( $class, 6 ); // remove 'Class_' prefix 
  259.  
  260.           if ( file_exists( $addon ) ) { 
  261.               if ( ! class_exists( $class ) ) { 
  262.                   try { 
  263.                       include_once $addon; 
  264.                   } catch ( Exception $ex ) { 
  265.                   } 
  266.               } 
  267.  
  268.               if ( class_exists( $class ) ) { 
  269.                   MS_Factory::load( $class ); 
  270.               } 
  271.           } 
  272.       } 
  273.  
  274.       /** 
  275.        * Allow custom addon-initialization code to run 
  276.        * 
  277.        * @since  1.0.0 
  278.        */ 
  279.       do_action( 'ms_model_addon_load' ); 
  280.   }
