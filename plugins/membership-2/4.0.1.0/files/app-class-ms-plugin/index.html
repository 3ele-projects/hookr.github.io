  357.   static public function flush_rewrite_rules( $url = false ) { 
  358.       if ( isset( $_GET['ms_flushed'] ) && 'yes' == $_GET['ms_flushed'] ) { 
  359.           $refresh = true; 
  360.       } else { 
  361.           $refresh = lib3()->session->get( 'refresh_url_rules' ); 
  362.       } 
  363.  
  364.       if ( $refresh ) { return; } 
  365.  
  366.       lib3()->session->add( 'refresh_url_rules', true ); 
  367.  
  368.       // The URL param is only to avoid cache. 
  369.       $url = esc_url_raw( 
  370.           add_query_arg( 'ms_ts', time(), $url ) 
  371.  );

  381.   public function maybe_flush_rewrite_rules() {
