  255.   public function simulation_infos() {

  258.  
  259.       /**

  270.

  278.           $view = MS_Factory::create( 'MS_View_Adminbar' ); 
  279.           $view->data = apply_filters( 'ms_view_admin_bar_data', $data );

  281.  
  282.           echo $html; 
  283.       } 
  284.   }
