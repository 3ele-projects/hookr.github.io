  629.    * After the user was successfully created we now have the opportunity to

  635.    * @param  WP_User $user The new user.

  637.   public function save_custom_fields( $user ) { 
  638.       if ( ! bp_is_active( 'xprofile' ) ) { return; } 
  639.  
  640.       // Make sure hidden field is passed and populated 
  641.       if ( isset( $_POST['signup_profile_field_ids'] ) 
  642.           && ! empty( $_POST['signup_profile_field_ids'] ) 
  643.  ) { 
  644.           // Let's compact any profile field info into an array 
  645.           $profile_field_ids = wp_parse_id_list( $_POST['signup_profile_field_ids'] ); 
  646.  
  647.           // Loop through the posted fields formatting any datebox values then add to usermeta 
  648.           foreach ( (array) $profile_field_ids as $field_id ) { 
  649.               $value = ''; 
  650.               $visibility = 'public'; 
  651.  
  652.               if ( ! isset( $_POST['field_' . $field_id] ) ) { 
  653.                   // Build the value of date-fields. 
  654.                   if ( ! empty( $_POST['field_' . $field_id . '_day'] ) 
  655.                       && ! empty( $_POST['field_' . $field_id . '_month'] ) 
  656.                       && ! empty( $_POST['field_' . $field_id . '_year'] ) 
  657.  ) { 
  658.                       // Concatenate the values. 
  659.                       $date_value = 
  660.                           $_POST['field_' . $field_id . '_day'] . ' ' . 
  661.                           $_POST['field_' . $field_id . '_month'] . ' ' . 
  662.                           $_POST['field_' . $field_id . '_year']; 
  663.  
  664.                       // Turn the concatenated value into a timestamp. 
  665.                       $_POST['field_' . $field_id] = date( 'Y-m-d H:i:s', strtotime( $date_value ) ); 
  666.                   } 
  667.               } 
  668.  
  669.               if ( ! empty( $_POST['field_' . $field_id] ) ) { 
  670.                   $value = $_POST['field_' . $field_id]; 
  671.               } 
  672.  
  673.               if ( ! empty( $_POST['field_' . $field_id . '_visibility'] ) ) { 
  674.                   $visibility = $_POST['field_' . $field_id . '_visibility']; 
  675.               } 
  676.  
  677.               xprofile_set_field_visibility_level( $field_id, $user->id, $visibility ); 
  678.               xprofile_set_field_data( $field_id, $user->id, $value, false ); 
  679.           } 
  680.       } 
  681.   }
