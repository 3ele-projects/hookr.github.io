  47.   /** 
  48.    * Adds category__in filter for posts query to remove all posts which not 
  49.    * belong to allowed categories. 
  50.    * 
  51.    * Related Filters: 
  52.    * - pre_get_posts 
  53.    * 
  54.    * @since  1.0.0 
  55.    * 
  56.    * @param WP_Query $query The WP_Query object to filter. 
  57.    */ 
  58.   public function protect_posts( $wp_query ) { 
  59.       // To protect unnecessary protection of other content 
  60.       if ( is_category() || is_home() || is_search() ) { 
  61.           $post_type = self::get_post_type( $wp_query ); 
  62.  
  63.           /** 
  64.            * '' .. when post type is unknown assume 'post' 
  65.            * 'post' .. obviously protect certain posts 
  66.            * 'page' .. when front page is set to static page 
  67.            */ 
  68.           if ( in_array( $post_type, array( 'post', 'page', '' ) ) ) { 
  69.               // This list is already filtered (see the get_terms filter!) 
  70.               $contents = get_categories( 'get=all' ); 
  71.               $categories = array(); 
  72.  
  73.               foreach ( $contents as $content ) { 
  74.                   $categories[] = absint( $content->term_id ); 
  75.               } 
  76.  
  77.               $wp_query->query_vars['category__in'] = $categories; 
  78.           } 
  79.  
  80.           do_action( 
  81.               'ms_rule_category_model_protect_posts',  
  82.               $wp_query,  
  83.               $this 
  84.  ); 
  85.       } 
  86.   }

  88.   /** 
  89.    * Filters categories and removes all not accessible categories. 
  90.    * 
  91.    * @since  1.0.0 
  92.    * 
  93.    * @param array $terms The terms array. 
  94.    * @param array $taxonomies The taxonomies array. 
  95.    * @return array Filtered terms array. 
  96.    */ 
  97.   public function protect_categories( $terms, $taxonomies ) { 
  98.       $new_terms = array(); 
  99.  
  100.       // Bail - not fetching category taxonomy. 
  101.       if ( ! in_array( 'category', $taxonomies ) ) { 
  102.           return $terms; 
  103.       } 
  104.  
  105.       if ( ! is_array( $terms ) ) { 
  106.           $terms = (array) $terms; 
  107.       } 
  108.  
  109.       foreach ( $terms as $key => $term ) { 
  110.           if ( ! empty( $term->taxonomy ) && 'category' === $term->taxonomy ) { 
  111.               $has_access = $this->has_access( $term->term_id ); 
  112.               if( is_null($has_access) ) $has_access = true; 
  113.               if ( $has_access ) { 
  114.                   $new_terms[ $key ] = $term; 
  115.               } 
  116.           } else { 
  117.               // Taxonomy is no category: Add it so custom taxonomies don't break. 
  118.               $new_terms[ $key ] = $term; 
  119.           } 
  120.       } 
  121.  
  122.       return apply_filters( 
  123.           'ms_rule_category_model_protect_categories',  
  124.           $new_terms 
  125.  ); 
  126.   }

  128.   /** 
  129.    * Verify access to the current category or post belonging to a catogory. 
  130.    * 
  131.    * @since  1.0.0 
  132.    * 
  133.    * @param int $id The current post_id. 
  134.    * @return bool|null True if has access, false otherwise. 
  135.    *     Null means: Rule not relevant for current page. 
  136.    */ 
  137.   public function has_access( $id, $admin_has_access = true ) { 
  138.       $has_access = null; 
  139.       $allowed = null; 
  140.       $forbidden = false; 
  141.  
  142.       $taxonomies = get_object_taxonomies( get_post_type() ); 
  143.  
  144.       // Verify post access accordingly to category rules. 
  145.       if ( false === empty( $id ) 
  146.           || ( is_single() && in_array( 'category', $taxonomies ) ) 
  147.  ) { 
  148.           if ( empty( $id ) ) { 
  149.               $id = get_the_ID(); 
  150.           } 
  151.  
  152.           $categories = wp_get_post_categories( $id ); 
  153.           foreach ( $categories as $category_id ) { 
  154.               $has_access = parent::has_access( $category_id, $admin_has_access ); 
  155.  
  156.               if ( $has_access ) { 
  157.                   $allowed = true; 
  158.               } 
  159.               if ( false === $has_access ) { 
  160.                   $forbidden = true; 
  161.               } 
  162.           } 
  163.       } elseif ( is_category() ) { 
  164.           // Category page. 
  165.           $category = get_queried_object_id(); 
  166.           $has_access = parent::has_access( $category, $admin_has_access ); 
  167.       } 
  168.  
  169.       // Prioritize restriction 
  170.       $has_access = $forbidden ? false : $allowed; 
  171.  
  172.       return apply_filters( 
  173.           'ms_rule_category_model_has_access',  
  174.           $has_access,  
  175.           $id,  
  176.           $admin_has_access,  
  177.           $this 
  178.  ); 
  179.   }

  208.                   $this->dripped[ $category->id ]['period_type']; 
  209.               $category->dripped = $this->dripped[ $category->id ]; 
  210.           } else { 
  211.               $category->delayed_period = ''; 
  212.           } 
  213.  
  214.           $cont[ $key ] = $category;
