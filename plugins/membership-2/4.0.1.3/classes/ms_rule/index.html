  1118.     /**

  1123.      * @param  array $args 
  1124.      * @return array { 
  1125.      *     List of post_ids to exclude or include 
  1126.      * 
  1127.      *     array $include 
  1128.      *     array $exclude 
  1129.      * } 
  1130.      */ 
  1131.     public function get_exclude_include( $args ) { 
  1132.         // Filter for Membership and Protection status via 'exclude'/'include' 
  1133.         $include = array(); 
  1134.         $exclude = array(); 
  1135.         $base_rule = $this; 
  1136.         $child_rule = $this; 
  1137.  
  1138.         if ( ! $this->is_base_rule ) { 
  1139.             $base_rule = MS_Model_Membership::get_base()->get_rule( $this->rule_type ); 
  1140.         } 
  1141.         if ( ! empty( $args['membership_id'] ) ) { 
  1142.             $child_membership = MS_Factory::load( 
  1143.                 'MS_Model_Membership',  
  1144.                 $args['membership_id'] 
  1145.  ); 
  1146.             $child_rule = $child_membership->get_rule( $this->rule_type ); 
  1147.         } 
  1148.  
  1149.         $base_items = array_keys( $base_rule->rule_value, true ); 
  1150.         $child_items = array_keys( $child_rule->rule_value, true ); 
  1151.  
  1152.         $status = ! empty( $args['rule_status'] ) ? $args['rule_status'] : null; 
  1153.  
  1154.         switch ( $status ) { 
  1155.             case MS_Model_Rule::FILTER_PROTECTED; 
  1156.                 if ( ! empty( $args['membership_id'] ) ) { 
  1157.                     $include = array_intersect( $child_items, $base_items ); 
  1158.                 } else { 
  1159.                     $include = $child_items; 
  1160.                 } 
  1161.                 if ( empty( $include ) ) { 
  1162.                     $include = array( -1 ); 
  1163.                 } 
  1164.                 break; 
  1165.  
  1166.             case MS_Model_Rule::FILTER_NOT_PROTECTED; 
  1167.                 if ( ! empty( $args['membership_id'] ) ) { 
  1168.                     $include = array_diff( $base_items, $child_items ); 
  1169.                     if ( empty( $include ) && empty( $exclude ) ) { 
  1170.                         $include = array( -1 ); 
  1171.                     } 
  1172.                 } else { 
  1173.                     $exclude = $child_items; 
  1174.                     if ( empty( $include ) && empty( $exclude ) ) { 
  1175.                         $exclude = array( -1 ); 
  1176.                     } 
  1177.                 } 
  1178.                 break; 
  1179.  
  1180.             default: 
  1181.                 // If not visitor membership, just show all Membership2 
  1182.                 if ( ! $child_rule->is_base_rule ) { 
  1183.                     $include = $base_items; 
  1184.                 } 
  1185.                 break; 
  1186.         } 
  1187.  
  1188.         /** 
  1189.          * Allow rules/Add-ons to modify the exclude/include list. 
  1190.          * 
  1191.          * @since  1.0.0 
  1192.          */ 
  1193.         $exclude = array_unique( 
  1194.             apply_filters( 
  1195.                 'ms_rule_exclude_items-' . $this->rule_type,  
  1196.                 $exclude,  
  1197.                 $args
