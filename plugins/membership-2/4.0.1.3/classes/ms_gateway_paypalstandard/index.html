  563.                             $notes_pay .= __( 'Payment registered, though amount differs from invoice.', 'membership2' ); 
  564.                         } 
  565.                         break; 
  566.  
  567.                     case 'pay-free': 
  568.                         if($subscription->is_trial_eligible()) { 
  569.                             $total = $invoice->trial_price; 
  570.                         }else{ 
  571.                             $total = $invoice->total; 
  572.                         } 
  573.                         if ( 0 == $total ) { 
  574.                             $success = true; 
  575.                         } else { 
  576.                             $ignore = true; 
  577.                         } 
  578.                         break; 
  579.  
  580.                     case 'cancel': 
  581.                         $member->cancel_membership( $membership->id ); 
  582.                         $member->save(); 
  583.                         $ignore = true; 
  584.                         break; 
  585.  
  586.                     case 'ignore': 
  587.                         $ignore = true; 
  588.                         break; 
  589.                 } 
  590.  
  591.                 if ( ! empty( $notes_pay ) ) { $invoice->add_notes( $notes_pay ); } 
  592.                 if ( ! empty( $notes_txn ) ) { $invoice->add_notes( $notes_txn ); } 
  593.  
  594.                 $invoice->save(); 
  595.  
  596.                 if ( $success ) { 
  597.                     $invoice->pay_it( self::ID, $external_id ); 
  598.                 } elseif ( ! empty( $status ) ) { 
  599.                     $invoice->status = $status; 
  600.                     $invoice->save(); 
  601.                     $invoice->changed(); 
  602.                 } 
  603.  
  604.                 do_action( 
  605.                     'ms_gateway_paypalstandard_payment_processed_' . $status,  
  606.                     $invoice,  
  607.                     $subscription 
  608.  ); 
  609.  
  610.             } else { 
  611.                 $reason = 'Unexpected transaction response'; 
  612.                 switch ( true ) { 
  613.                     case is_wp_error( $response ): 
  614.                         $reason = 'PayPal did not verify this transaction: Unknown error'; 
  615.                         break; 
  616.  
  617.                     case 200 != $response['response']['code']: 
  618.                         $reason = sprintf( 
  619.                             'PayPal did not verify the transaction: Code %s',  
  620.                             $response['response']['code'] 
  621.  ); 
  622.                         break; 
  623.  
  624.                     case empty( $response['body'] ): 
  625.                         $reason = 'PayPal did not verify this transaction: Empty response'; 
  626.                         break; 
  627.  
  628.                     case 'VERIFIED' != $response['body']: 
  629.                         $reason = sprintf( 
  630.                             'PayPal did not verify this transaction: "%s"',  
  631.                             $response['body'] 
  632.  ); 
  633.                         break; 
  634.  
  635.                     case ! $invoice->id: 
  636.                         $reason = sprintf( 
  637.                             'Specified invoice does not exist: "%s"',  
  638.                             $invoice_id 
  639.  ); 
  640.                         break; 
  641.                 } 
  642.  
  643.                 $notes = 'Response Error: ' . $reason; 
  644.                 $exit = true; 
  645.             } 
  646.         } else { 
  647.             // Did not find expected POST variables. Possible access attempt from a non PayPal site. 
  648.  
  649.             $u_agent = $_SERVER['HTTP_USER_AGENT']; 
  650.             if ( ! $log && false === strpos( $u_agent, 'PayPal' ) ) { 
  651.                 // Very likely someone tried to open the URL manually. Redirect to home page. 
  652.                 if ( ! $notes ) { 
  653.                     $notes = 'Ignored: Missing POST variables. Redirect to Home-URL.'; 
  654.                 } 
  655.                 $redirect = MS_Helper_Utility::home_url( '/' ); 
  656.                 $ignore = true; 
  657.                 $success = false; 
  658.             } elseif ( 'm1' == $ext_type ) { 
  659.                 /** 
  660.                  * The payment belongs to an imported M1 subscription and could 
  661.                  * not be auto-matched. 
  662.                  * Do not return an error code, but also do not modify any 
  663.                  * invoice/subscription. 
  664.                  */ 
  665.                 $notes = 'M1 Payment detected. Manual matching required. ' . $notes_err; 
  666.                 $success = false; 
  667.             } elseif ( 'pay_btn' == $ext_type ) { 
  668.                 /** 
  669.                  * The payment was made by a PayPal Payment button that was 
  670.                  * created in the PayPal account and not by M1/M2. 
  671.                  */ 
  672.                 $notes = 'PayPal Payment button detected. Manual matching required. ' . $notes_err; 
  673.                 $success = false; 
  674.             } else { 
  675.                 // PayPal sent us a IPN notice about a non-Membership payment: 
  676.                 // Ignore it, but add it to the logs. 
  677.  
  678.                 if ( ! empty( $notes_err ) ) { 
  679.                     // Use the notes_err. 
  680.                     $notes = $notes_err; 
  681.                 } if ( ! empty( $notes ) ) { 
  682.                     // We already have an error message, do nothing. 
  683.                 } elseif ( ! $transaction_type ) { 
  684.                     $notes = 'Ignored: txn_type not specified. Cannot process.'; 
  685.                 } elseif ( empty( $_POST['invoice'] ) && empty( $_POST['custom'] ) ) { 
  686.                     $notes = 'Ignored: No invoice or custom data specified.'; 
  687.                 } else { 
  688.                     $notes = 'Ignored: Missing POST variables. Identification is not possible.'; 
  689.                 } 
  690.  
  691.                 $ignore = true; 
  692.                 $success = false; 
  693.             } 
  694.             $exit = true; 
  695.         } 
  696.  
  697.         if ( $ignore && ! $success ) { 
  698.             $success = null; 
  699.             $notes .= ' [Irrelevant IPN call]'; 
  700.         } 
  701.  
  702.         if ( ! $log ) { 
  703.             do_action( 
  704.                 'ms_gateway_transaction_log',  
  705.                 self::ID, // gateway ID 
  706.                 'handle', // request|process|handle 
  707.                 $success, // success flag 
  708.                 $subscription_id, // subscription ID 
  709.                 $invoice_id, // invoice ID 
  710.                 $amount, // charged amount 
  711.                 $notes, // Descriptive text 
  712.                 $external_id // External ID 
  713.  ); 
  714.  
  715.             if ( $redirect ) { 
  716.                 wp_safe_redirect( $redirect ); 
  717.                 exit; 
  718.             } 
  719.             if ( $exit ) { 
  720.                 exit; 
  721.             } 
  722.         } else { 
  723.             $log->invoice_id = $invoice_id; 
  724.             $log->subscription_id = $subscription_id; 
  725.             $log->amount = $amount; 
  726.             $log->description = $notes; 
  727.             $log->external_id = $external_id; 
  728.             if ( $success ) { 
  729.                 $log->manual_state( 'ok' ); 
  730.             } elseif ( $ignore ) { 
  731.                 $log->manual_state( 'ignore' ); 
  732.             } 
  733.             $log->save(); 
  734.         } 
  735.  
  736.         do_action( 
  737.             'ms_gateway_paypalstandard_handle_return_after',  
  738.             $this 
  739.  ); 
  740.  
  741.         if ( $log ) { 
  742.             return $log; 
  743.         } 
  744.     }
