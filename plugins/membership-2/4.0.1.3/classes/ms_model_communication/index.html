  911.  );

  913.                 }

  1088.         $user_id = 0; 
  1089.         $subscription = null; 
  1090.         $member = null; 
  1091.         $sent = false; 
  1092.

  1112.                 $user_id 
  1113.  );

  1119.         // Prevent duplicate messages. 
  1120.         if ( $subscription ) { 
  1121.             $delay = $subscription->seconds_since_last_email( $this->type ); 
  1122.  
  1123.             /** 
  1124.              * Allow other plugins to customize the duration for ignoring 
  1125.              * duplicate emails. This is supposed to be a value in seconds,  
  1126.              * so 60 will prevent duplicate emails for 1 minute. 
  1127.              *
