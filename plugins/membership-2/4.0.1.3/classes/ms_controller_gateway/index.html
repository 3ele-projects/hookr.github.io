  604.     public function process_purchase() { 
  605.         $fields = array( 'gateway', 'ms_relationship_id' ); 
  606.  
  607.         lib3()->array->equip_request( 'gateway', 'ms_relationship_id' ); 
  608.  
  609.         $valid = true; 
  610.         $nonce_name = $_REQUEST['gateway'] . '_' . $_REQUEST['ms_relationship_id']; 
  611.  
  612.         if ( ! self::validate_required( $fields, 'any' ) ) { 
  613.             $valid = false; 
  614.             $err = 'GAT-01 (invalid fields)'; 
  615.         } elseif ( ! MS_Model_Gateway::is_valid_gateway( $_REQUEST['gateway'] ) ) { 
  616.             $valid = false; 
  617.             $err = 'GAT-02 (invalid gateway)'; 
  618.         } elseif ( ! $this->verify_nonce( $nonce_name, 'any' ) ) { 
  619.             $valid = false; 
  620.             $err = 'GAT-03 (invalid nonce)'; 
  621.         } 
  622.  
  623.         if ( $valid ) { 
  624.             $subscription = MS_Factory::load( 
  625.                 'MS_Model_Relationship',  
  626.                 $_REQUEST['ms_relationship_id'] 
  627.  ); 
  628.  
  629.             $gateway_id = $_REQUEST['gateway']; 
  630.             $gateway = MS_Model_Gateway::factory( $gateway_id ); 
  631.  
  632.             try { 
  633.                 $invoice = $gateway->process_purchase( $subscription ); 
  634.  
  635.                 // If invoice is successfully paid, redirect to welcome page. 
  636.                 if ( $invoice->is_paid() 
  637.                     || ( $invoice->uses_trial 
  638.                         && MS_Model_Invoice::STATUS_BILLED == $invoice->status 
  639.  ) 
  640.  ) { 
  641.                     // Make sure to respect the single-membership rule 
  642.                     $this->validate_membership_states( $subscription ); 
  643.  
  644.                     // Redirect user to the Payment-Completed page. 
  645.                     if ( ! defined( 'IS_UNIT_TEST' ) ) { 
  646.                         MS_Model_Pages::redirect_to( 
  647.                             MS_Model_Pages::MS_PAGE_REG_COMPLETE,  
  648.                             array( 'ms_relationship_id' => $subscription->id ) 
  649.  ); 
  650.                     } 
  651.                 } elseif ( MS_Gateway_Manual::ID == $gateway_id ) { 
  652.                     // For manual gateway payments. 
  653.                     $this->add_action( 'the_content', 'purchase_info_content' ); 
  654.                 } else { 
  655.                     // Something went wrong, the payment was not successful. 
  656.                     $this->add_action( 'the_content', 'purchase_error_content' ); 
  657.                 } 
  658.             } catch ( Exception $e ) { 
  659.                 MS_Helper_Debug::log( $e->getMessage() ); 
  660.  
  661.                 switch ( $gateway_id ) { 
  662.                     case MS_Gateway_Authorize::ID: 
  663.                         $_POST['auth_error'] = $e->getMessage(); 
  664.                         // call action to step back 
  665.                         do_action( 'ms_controller_frontend_signup_gateway_form' ); 
  666.                         break; 
  667.  
  668.                     case MS_Gateway_Stripe::ID: 
  669.                     case MS_Gateway_Stripeplan::ID: 
  670.                         $_POST['error'] = sprintf( 
  671.                             __( 'Error: %s', 'membership2' ),  
  672.                             $e->getMessage() 
  673.  ); 
  674.  
  675.                         // Hack to send the error message back to the payment_table. 
  676.                         MS_Plugin::instance()->controller->controllers['frontend']->add_action( 
  677.                             'the_content',  
  678.                             'payment_table', 1 
  679.  ); 
  680.                         break; 
  681.  
  682.                     default: 
  683.                         do_action( 'ms_controller_gateway_form_error', $e ); 
  684.                         $this->add_action( 'the_content', 'purchase_error_content' ); 
  685.                         break; 
  686.                 } 
  687.             } 
  688.         } else { 
  689.             MS_Helper_Debug::log( 'Error Code ' . $err ); 
  690.  
  691.             $this->add_action( 'the_content', 'purchase_error_content' ); 
  692.         } 
  693.  
  694.         // Hack to show signup page in case of errors 
  695.         $ms_page = MS_Model_Pages::get_page( MS_Model_Pages::MS_PAGE_REGISTER ); 
  696.  
  697.         if ( $ms_page ) { 
  698.             // During unit-testing the $ms_page object might be empty. 
  699.             global $wp_query; 
  700.             $wp_query->query_vars['page_id'] = $ms_page->ID; 
  701.             $wp_query->query_vars['post_type'] = 'page'; 
  702.         } 
  703.  
  704.         do_action( 
  705.             'ms_controller_gateway_process_purchase_after',  
  706.             $this 
  707.  ); 
  708.     }

  782.      *

  788.      * @since  1.0.0

  790.      * @param WP_Query $wp_query The WordPress query object 
  791.      */ 
  792.     public function handle_payment_return( $wp_query ) { 
  793.         // Do not check custom loops. 
  794.         if ( ! $wp_query->is_main_query() ) { return; } 
  795.  
  796.         if ( ! empty( $wp_query->query_vars['paymentgateway'] ) ) { 
  797.             $gateway = $wp_query->query_vars['paymentgateway']; 
  798.  
  799.             /** 
  800.              * In 1.1.0 the underscore in payment gateway names was removed.

  802.              * gateway-names. 
  803.              */ 
  804.             switch ( $gateway ) { 
  805.                 case 'paypal_single': $gateway = 'paypalsingle'; break; 
  806.                 case 'paypal_standard': $gateway = 'paypalstandard'; break; 
  807.                 case 'paypal-single': $gateway = 'paypalsingle'; break; 
  808.                 case 'paypal-standard': $gateway = 'paypalstandard'; break; 
  809.                 case 'paypalsolo': $gateway = 'paypalsingle'; break; // M1 
  810.                 case 'paypalexpress': $gateway = 'paypalstandard'; break; //M1 
  811.             } 
  812.  
  813.             if ( MS_Model_Gateway::is_active( $gateway ) ) { 
  814.                 $action = 'ms_gateway_handle_payment_return_' . $gateway; 
  815.                 do_action( $action ); 
  816.             } else { 
  817.                 // Log the payment attempt when the gateway is not active. 
  818.                 if ( MS_Model_Gateway::is_valid_gateway( $gateway ) ) { 
  819.                     $note = __( 'Gateway is inactive', 'membership2' ); 
  820.                 } else {

  822.                         __( 'Unknown Gateway: %s', 'membership2' ),  
  823.                         $gateway 
  824.  ); 
  825.                 } 
  826.  
  827.                 do_action( 
  828.                     'ms_gateway_transaction_log',  
  829.                     $gateway, // gateway ID 
  830.                     'handle', // request|process|handle 
  831.                     false, // success flag 
  832.                     0, // subscription ID 
  833.                     0, // invoice ID 
  834.                     0, // charged amount 
  835.                     $note, // Descriptive text 
  836.                     '' // External ID 
  837.  );
