  590.     /**

  595.      * B) Checks if the permalink structure uses the post-name

  647.     static private function check_migration_handler() {

  686.     /** 
  687.      * Returns a secure token to trigger advanced admin actions like db-reset 
  688.      * or restoring a snapshot. 
  689.      * 
  690.      * - Only one token is valid at any given time. 
  691.      * - Each token has a timeout of max. 120 seconds. 
  692.      * - Each token can be used once only. 
  693.      * 
  694.      * @since  1.0.0 
  695.      * @internal 
  696.      * 
  697.      * @param  string $action Like a nonce, this is the action to execute. 
  698.      * @return array Intended usage: add_query_param( $token, $url ) 
  699.      */ 
  700.     static public function get_token( $action ) { 
  701.         if ( ! is_user_logged_in() ) { return array(); } 
  702.         if ( ! is_admin() ) { return array(); } 
  703.  
  704.         $one_time_key = uniqid(); 
  705.         MS_Factory::set_transient( 'ms_one_time_key-' . $action, $one_time_key, 120 ); 
  706.  
  707.         // Token is valid for 86 seconds because of usage of date('B') 
  708.         $plain = $action . '-' . date( 'B' ) . ':' . get_current_user_id() . '-' . $one_time_key; 
  709.         $token = array( 'ms_token' => wp_create_nonce( $plain ) ); 
  710.         return $token; 
  711.     } 
  712.  
  713.     /** 
  714.      * Verfies the admin token in the $_GET collection 
  715.      * 
  716.      * $_GET['ms_token'] must match the current ms_token 
  717.      * $_POST['confirm'] must have value 'yes' 
  718.      * 
  719.      * @since  1.0.0 
  720.      * @internal 
  721.      * 
  722.      * @param  string $action Like a nonce, this is the action to execute. 
  723.      * @return bool 
  724.      */ 
  725.     static private function verify_token( $action ) { 
  726.         if ( ! self::valid_user() ) { return false; } 
  727.  
  728.         if ( empty( $_GET['ms_token'] ) ) { return false; } 
  729.         $get_token = $_GET['ms_token']; 
  730.  
  731.         if ( empty( $_POST['confirm'] ) ) { return false; } 
  732.         if ( 'yes' != $_POST['confirm'] ) { return false; } 
  733.  
  734.         $one_time_key = MS_Factory::get_transient( 'ms_one_time_key-' . $action ); 
  735.         MS_Factory::delete_transient( 'ms_one_time_key-' . $action ); 
  736.         if ( empty( $one_time_key ) ) { return false; } 
  737.  
  738.         // We verify the current and the previous beat 
  739.         $plain_token_1 = $action . '-' . date( 'B' ) . ':' . get_current_user_id() . '-' . $one_time_key; 
  740.         $plain_token_2 = $action . '-' . ( date( 'B' ) - 1 ) . ':' . get_current_user_id() . '-' . $one_time_key; 
  741.  
  742.         if ( wp_verify_nonce( $get_token, $plain_token_1 ) ) { return true; } 
  743.         if ( wp_verify_nonce( $get_token, $plain_token_2 ) ) { return true; } 
  744.  
  745.         return false; 
  746.     } 
  747.  
  748.     /** 
  749.      * Verifies the following conditions: 
  750.      * - Current user is logged in and has admin permissions 
  751.      * - The request is an wp-admin request 
  752.      * - The request is not an Ajax call 
  753.      * 
  754.      * @since  1.0.0 
  755.      * @return bool True if all conditions are true 
  756.      */ 
  757.     static private function valid_user() { 
  758.         /** 
  759.          * Determine user_id from request cookies. 
  760.          * @see wp-includes/pluggable.php wp_currentuserinfo() 
  761.          */ 
  762.         $user_id = apply_filters( 'determine_current_user', false ); 
  763.  
  764.         if ( ! $user_id ) { return false; } 
  765.         if ( ! is_admin() ) { return false; } 
  766.         if ( defined( 'DOING_AJAX' ) && DOING_AJAX ) { return false; } 
  767.         if ( defined( 'DOING_CRON' ) && DOING_CRON ) { return false; } 
  768.                 /** 
  769.                  * BuddyPress notices hack 
  770.                  */ 
  771.                 $bp = function_exists( 'buddypress' ) ? buddypress() : null; 
  772.                 if( null != $bp ) 
  773.                 { 
  774.                         if ( ! did_action( 'init' ) ) { return false; } 
  775.                 } 
  776.         if ( ! current_user_can( 'manage_options' ) ) { return false; } 
  777.  
  778.         return true; 
  779.     }
