  205.      * Takes Hue/Saturation/Lightness color definition and returns a hex color code 
  206.      * 
  207.      * @since  1.0.0

  210.      * @param  float $l

  213.     static public function hsl2web( $h, $s, $l ) { 
  214.         $r = 0; $g = 0; $b = 0; 
  215.  
  216.         $h %= 360; 
  217.         $c = ( 1 - abs( 2 * $l - 1 ) ) * $s; 
  218.         $x = $c * ( 1 - abs( fmod( ( $h / 60 ), 2 ) - 1 ) ); 
  219.         $m = $l - ( $c / 2 ); 
  220.  
  221.         if ( $h < 60 ) { 
  222.             $r = $c; $g = $x; $b = 0; 
  223.         } else if ( $h < 120 ) { 
  224.             $r = $x; $g = $c; $b = 0; 
  225.         } else if ( $h < 180 ) { 
  226.             $r = 0; $g = $c; $b = $x; 
  227.         } else if ( $h < 240 ) { 
  228.             $r = 0; $g = $x; $b = $c; 
  229.         } else if ( $h < 300 ) { 
  230.             $r = $x; $g = 0; $b = $c; 
  231.         } else { 
  232.             $r = $c; $g = 0; $b = $x; 
  233.         } 
  234.  
  235.         $r = str_pad( dechex( ( $r + $m ) * 255 ), 2, '0', STR_PAD_LEFT ); 
  236.         $g = str_pad( dechex( ( $g + $m ) * 255 ), 2, '0', STR_PAD_LEFT ); 
  237.         $b = str_pad( dechex( ( $b + $m ) * 255 ), 2, '0', STR_PAD_LEFT ); 
  238.  
  239.         return '#' . $r . $g . $b;
