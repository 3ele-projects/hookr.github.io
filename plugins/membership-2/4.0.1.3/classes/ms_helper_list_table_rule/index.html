  607.      * Displayed above the views. 
  608.      * 
  609.      * In the rule list-tables the list-head is used to display a filter for 
  610.      * membership-ID. Combined with the views (below) users can filter all rules 
  611.      * by membership + protection status independantly 
  612.      * 
  613.      * @since  1.0.0 
  614.      */ 
  615.     public function list_head() { 
  616.         $type_name = $this->name['plural']; 
  617.         $membership_name = ''; 
  618.         $membership_color = ''; 
  619.  
  620.         /** 
  621.          * We don't build the title dynamically to make sure translations are 
  622.          * possible and meaningful in the context. 
  623.          * 
  624.          * E.g. "Showing All Pages" in german would typically translate as 
  625.          * "All pages are shown"; also "All" has several translations, depending 
  626.          * on context. 
  627.          */ 
  628.         if ( empty( $_GET['membership_id'] ) ) { 
  629.             if ( empty( $_GET['status'] ) ) { 
  630.                 $title = __( 'Showing <b>All</b> %1$s', 'membership2' ); 
  631.             } elseif ( MS_Model_Rule::FILTER_NOT_PROTECTED == $_GET['status'] ) { 
  632.                 $title = __( 'Showing All <b>Unprotected</b> %1$s', 'membership2' ); 
  633.             } elseif ( MS_Model_Rule::FILTER_PROTECTED == $_GET['status'] ) { 
  634.                 $title = __( 'Showing All <b>Protected</b> %1$s', 'membership2' ); 
  635.             } 
  636.         } else { 
  637.             $membership = MS_Factory::load( 'MS_Model_Membership', $_GET['membership_id'] ); 
  638.  
  639.             if ( empty( $_GET['status'] ) ) { 
  640.                 $title = __( 'Showing <b>All</b> %1$s for %2$s', 'membership2' ); 
  641.             } elseif ( MS_Model_Rule::FILTER_NOT_PROTECTED == $_GET['status'] ) { 
  642.                 $title = __( 'Showing All %1$s that are <b>not protected</b> by %2$s', 'membership2' ); 
  643.             } elseif ( MS_Model_Rule::FILTER_PROTECTED == $_GET['status'] ) { 
  644.                 $title = __( 'Showing All %1$s that are <b>protected</b> by %2$s', 'membership2' ); 
  645.             } 
  646.  
  647.             $membership_name = $membership->name; 
  648.             $membership_color = $membership->get_color(); 
  649.         } 
  650.  
  651.         $title = sprintf( 
  652.             $title,  
  653.             '<b>' . esc_html( $type_name ) . '</b>',  
  654.             sprintf( 
  655.                 '<span class="ms-membership" style="background-color:%2$s">%1$s</span>',  
  656.                 esc_html( $membership_name ),  
  657.                 $membership_color 
  658.  ) 
  659.  ); 
  660.  
  661.         printf( '<h3 class="ms-list-title">%1$s</h3>', $title ); 
  662.     }

  664.     /** 
  665.      * Returns an array that defines possible views. 
  666.      * 
  667.      * In the rule list-tables the views are used to filter by protection status 
  668.      * and not by membership-ID or other factors. 
  669.      * 
  670.      * @since  1.0.0 
  671.      * @return array 
  672.      */ 
  673.     public function get_views() { 
  674.         $count_args = $this->prepared_args; 
  675.         unset( $count_args['rule_status'] ); 
  676.  
  677.         // Count is not working, so we remove it for now 
  678.         //$count = $this->model->count_item_access( $count_args ); 
  679.  
  680.         $url = apply_filters( 
  681.             'ms_helper_listtable_' . $this->id . '_url',  
  682.             esc_url_raw( remove_query_arg( array( 'status', 'paged' ) ) ) 
  683.  ); 
  684.  
  685.         $views = array(); 
  686.  
  687.         $views['all'] = array( 
  688.             'url' => $url,  
  689.             'label' => __( 'All', 'membership2' ),

  691.  );

  694.             add_query_arg(

  697.  ) 
  698.  ); 
  699.         $views['public'] = array( 
  700.             'url' => $public_url,  
  701.             'label' => __( 'Unprotected', 'membership2' ),

  709.  )

  711.         $views['protected'] = array( 
  712.             'url' => $protected_url,  
  713.             'label' => __( 'Protected', 'membership2' ),  
  714.             //'count' => $count['accessible'],  
  715.  ); 
  716.  
  717.         return apply_filters( 
  718.             "ms_helper_listtable_{$this->id}_views",  
  719.             $views
