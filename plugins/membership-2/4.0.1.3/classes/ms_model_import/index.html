  577.     /** 
  578.      * Checks if the specified source/ID need matching. 
  579.      * 
  580.      * If the source or source_id is empty then the return value TRUE means that 
  581.      * there is *any* transaction that needs matching. 
  582.      * 
  583.      * See MS_Helper_Listtable_TransactionMatching for a list of sources. 
  584.      * 
  585.      * @since  1.0.1.2 
  586.      * @param  int    $source_id The M1 sub_id. 
  587.      * @param  string $source The import source. 
  588.      * @return bool True if the transaction details need matching. 
  589.      */ 
  590.     static public function can_match( $source_id = null, $source = null ) { 
  591.         $res = false; 
  592.         $settings = MS_Factory::load( 'MS_Model_Settings' ); 
  593.  
  594.         if ( empty( $source_id ) || empty( $source ) ) { 
  595.             $src = $settings->get_custom_setting( 'import_match' ); 
  596.             $src = lib3()->array->get( $src ); 
  597.  
  598.             foreach ( $src as $lst ) { 
  599.                 if ( is_array( $lst ) ) { 
  600.                     if ( count( $lst ) ) { 
  601.                         $res = true; 
  602.                         break; 
  603.                     } 
  604.                 } 
  605.             } 
  606.         } else { 
  607.             $lst = $settings->get_custom_setting( 'import_match', $source ); 
  608.  
  609.             if ( ! is_array( $lst ) ) { 
  610.                 $lst = array(); 
  611.             } 
  612.  
  613.             $res = in_array( $source_id, $lst ); 
  614.         } 
  615.  
  616.         return $res; 
  617.     }

  619.     /** 
  620.      * Remembers that the source_id needs to be matched with a membership_id to 
  621.      * complete the connected transaction. 
  622.      * 
  623.      * See MS_Helper_Listtable_TransactionMatching for a list of sources. 
  624.      * 
  625.      * @since  1.0.1.2 
  626.      * @param  int    $source_id The M1 sub_id. 
  627.      * @param  string $source The import source. 
  628.      */ 
  629.     static public function need_matching( $source_id, $source ) { 
  630.         $settings = MS_Factory::load( 'MS_Model_Settings' ); 
  631.  
  632.         $lst = $settings->get_custom_setting( 'import_match', $source ); 
  633.  
  634.         if ( ! is_array( $lst ) ) { 
  635.             $lst = array(); 
  636.         } 
  637.  
  638.         if ( ! in_array( $source_id, $lst ) ) { 
  639.             $lst[] = $source_id; 
  640.         } 
  641.  
  642.         $settings->set_custom_setting( 'import_match', $source, $lst ); 
  643.         $settings->save(); 
  644.     }

  646.     /** 
  647.      * Remove the source_id from the missing-matching-list again. 
  648.      * 
  649.      * See MS_Helper_Listtable_TransactionMatching for a list of sources. 
  650.      * 
  651.      * @since  1.0.1.2 
  652.      * @param  int    $source_id The M1 sub_id. 
  653.      * @param  string $source The import source. 
  654.      */ 
  655.     static public function dont_need_matching( $source_id, $source ) { 
  656.         $settings = MS_Factory::load( 'MS_Model_Settings' ); 
  657.  
  658.         $lst = $settings->get_custom_setting( 'import_match', $source ); 
  659.  
  660.         if ( ! is_array( $lst ) ) { 
  661.             $lst = array(); 
  662.         } 
  663.  
  664.         foreach ( $lst as $key => $id ) { 
  665.             if ( $id == $source_id ) { 
  666.                 unset( $lst[ $key ] ); 
  667.             } 
  668.         } 
  669.  
  670.         $settings->set_custom_setting( 'import_match', $source, $lst ); 
  671.         $settings->save(); 
  672.     }

  674.     /** 
  675.      * Save a permanent matching between the specified membership and the 
  676.      * transaction source. 
  677.      * 
  678.      * See MS_Helper_Listtable_TransactionMatching for a list of sources. 
  679.      * 
  680.      * Structure of the custom_data element 'matching': 
  681.      * 
  682.      *   'matching' => array( 
  683.      *     'pay_btn' => array( btn1, btn2, ... ),  
  684.      *     'm1' => array( m1_id1, m1_id2, ... ),  
  685.      * ) 
  686.      * 
  687.      * @since  1.0.1.2 
  688.      * @param  int    $membership_id The M2 membership_id. 
  689.      * @param  string $source_id The matching-ID to identify transactions. 
  690.      * @param  string $source The matching-key to identify transactions. 
  691.      * @return bool True if the matching was saved. 
  692.      */ 
  693.     static public function match_with_source( $membership_id, $source_id, $source ) { 
  694.         $membership = MS_Factory::load( 'MS_Model_Membership', $membership_id ); 
  695.  
  696.         if ( ! $membership || ! $membership->is_valid() ) { 
  697.             return false; 
  698.         } 
  699.  
  700.         // First make sure that no other membership is matched to the source. 
  701.         $memberships = MS_Model_Membership::get_memberships(); 
  702.  
  703.         foreach ( $memberships as $item ) { 
  704.             $data = $item->get_custom_data( 'matching' ); 
  705.             $changed = false; 
  706.  
  707.             if ( ! is_array( $data ) ) { continue; } 
  708.             if ( ! isset( $data[ $source ] ) ) { continue; } 
  709.             if ( ! is_array( $data[ $source ] ) ) { 
  710.                 unset( $data[ $source ] ); 
  711.                 continue; 
  712.             } 
  713.  
  714.             foreach ( $data[ $source ] as $key => $id ) { 
  715.                 if ( $id == $source_id ) { 
  716.                     unset( $data[ $source ][ $key ] ); 
  717.                     $data[ $source ] = array_values( array_unique( $data[ $source ] ) ); 
  718.                     $changed = true; 
  719.                 } 
  720.             } 
  721.             if ( $changed ) { 
  722.                 $item->set_custom_data( 'matching', $data ); 
  723.                 $item->save(); 
  724.             } 
  725.         } 
  726.  
  727.         // Then add the matching to the specified membership. 
  728.         $data = lib3()->array->get( 
  729.             $membership->get_custom_data( 'matching' ) 
  730.  ); 
  731.  
  732.         if ( empty( $data[ $source ] ) || ! is_array( $data[ $source ] ) ) { 
  733.             $data[ $source ] = array(); 
  734.         } 
  735.  
  736.         $data[ $source ][] = $source_id; 
  737.         $data[ $source ] = array_values( array_unique( $data[ $source ] ) ); 
  738.  
  739.         $membership->set_custom_data( 'matching', $data ); 
  740.         $membership->save(); 
  741.  
  742.         self::dont_need_matching( $source_id, $source ); 
  743.  
  744.         return true; 
  745.     } 
  746.  
  747.     /** 
  748.      * Tries to process a single transaction again. 
  749.      * 
  750.      * This function is only useful when the transaction matching was added 
  751.      * before callig it again. 
  752.      * 
  753.      * @since  1.0.1.2 
  754.      * @param  int $transaction_id The ID of the transaction log item. 
  755.      * @return bool True means that the transaction was processed. 
  756.      */ 
  757.     static public function retry_to_process( $transaction_id ) { 
  758.         $res = false; 
  759.         $log = MS_Factory::load( 'MS_Model_Transactionlog', $transaction_id ); 
  760.  
  761.         if ( empty( $log ) || $log->id != $transaction_id ) { 
  762.             // Could not find the requested transaction log item. 
  763.             return $res; 
  764.         } 
  765.  
  766.         if ( 'ok' == $log->state ) { 
  767.             // The transaction was already processed (automatically or manual). 
  768.             return $res; 
  769.         } 
  770.  
  771.         $post_data = $log->post; 
  772.         if ( empty( $post_data ) || ! is_array( $post_data ) ) { 
  773.             // We do not have POST data available for the transaction. 
  774.             // Re-Processing is not possible. 
  775.             return $res; 
  776.         } 
  777.  
  778.         $orig_post = $_POST; 
  779.         $orig_req = $_REQUEST; 
  780.  
  781.         // Set up the PHP environment to process the transaction again. 
  782.         $gateway = MS_Model_Gateway::factory( $log->gateway_id ); 
  783.         $_POST = $post_data; 
  784.         $_REQUEST = $post_data; 
  785.  
  786.         switch ( $log->method ) { 
  787.             case 'request': 
  788.                 // Intentionally not implemented: 
  789.                 // Request payment needs a subscription to work. 
  790.                 break; 
  791.  
  792.             case 'process': 
  793.                 // Intentionally not implemented: 
  794.                 // Request payment needs a subscription to work. 
  795.                 break; 
  796.  
  797.             case 'handle': 
  798.                 $log = $gateway->handle_return( $log ); 
  799.                 break; 
  800.         } 
  801.  
  802.         if ( 'ok' == $log->state ) { 
  803.             $res = true; 
  804.         } 
  805.  
  806.         $_POST = $orig_post; 
  807.         $_REQUEST = $orig_req; 
  808.  
  809.         return $res; 
  810.     }

  812.     /** 
  813.      * Find a M2 membership by a custom matching ID. 
  814.      * 
  815.      * The matching key and matching ID are stored in the memberships custom 
  816.      * data array. 
  817.      * 
  818.      * See MS_Helper_Listtable_TransactionMatching for a list of matching_keys. 
  819.      * 
  820.      * @since  1.0.1.2 
  821.      * @param  int $matching_key The matching key. 
  822.      * @param  int $matching_id The matching ID. 
  823.      * @return MS_Model_Membership|null The M2 membership. 
  824.      */ 
  825.     static public function membership_by_matching( $matching_key, $matching_id ) { 
  826.         $res = null; 
  827.         $args = array( 'include_guest' => 0 ); 
  828.         $memberships = MS_Model_Membership::get_memberships( $args ); 
  829.  
  830.         foreach ( $memberships as $membership ) { 
  831.             $data = $membership->get_custom_data( 'matching' ); 
  832.             if ( empty( $data ) || ! is_array( $data ) ) { continue; } 
  833.             if ( ! isset( $data[ $matching_key ] ) ) { continue; } 
  834.             $ids = lib3()->array->get( $data[ $matching_key ] ); 
  835.  
  836.             foreach ( $ids as $id ) { 
  837.                 if ( $matching_id == $id ) { 
  838.                     $res = $membership; 
  839.                     break 2; 
  840.                 } 
  841.             } 
  842.         } 
  843.  
  844.         return $res; 
  845.     }

  847.     /** 
  848.      * Tries to find a subscription based on the user-ID and M1 sub_id 
  849.      * 
  850.      * Matching values are looked up in the memberships custom data array. 
  851.      * 
  852.      * See MS_Helper_Listtable_TransactionMatching for a list of sources. 
  853.      * 
  854.      * @since  1.0.1.2 
  855.      * @param  int    $user_id The user-ID. 
  856.      * @param  string $matching_id The matching-ID (M1 sub_id, a btn_id, etc). 
  857.      * @param  string $type The matching type to apply. Default is 'm1'. 
  858.      * @param  string $gateway The payment gateway. 
  859.      * @return MS_Model_Relationship|null The subscription object. 
  860.      */ 
  861.     static public function find_subscription( $user_id, $matching_id, $type = 'm1', $gateway = 'admin' ) { 
  862.         $res = null; 
  863.  
  864.         if ( ! is_numeric( $user_id ) ) { 
  865.             // Seems like we got invalid values... 
  866.             return $res; 
  867.         } 
  868.  
  869.         $user_id = intval( $user_id ); 
  870.         $matching_id = trim( $matching_id ); 
  871.  
  872.         if ( $user_id < 1 || empty( $matching_id ) ) { 
  873.             // Seems like user or sub_id are empty or invalid. 
  874.             return $res; 
  875.         } 
  876.  
  877.         $member = MS_Factory::load( 'MS_Model_Member', $user_id ); 
  878.         if ( $user_id != $member->id ) { 
  879.             // The user_id is invalid. 
  880.             return $res; 
  881.         } 
  882.  
  883.         $membership = self::membership_by_matching( $type, $matching_id ); 
  884.  
  885.         if ( ! $membership || ! $membership->is_valid() ) { 
  886.             // The sub_id is invalid. 
  887.             return $res; 
  888.         } 
  889.  
  890.         // Finally we have a member and a membership. Fetch the subscription! 
  891.         $res = $member->get_subscription( $membership->id ); 
  892.         if ( ! $res ) { 
  893.             $res = $member->add_membership( $membership->id, $gateway ); 
  894.         } 
  895.  
  896.         return $res; 
  897.     }
