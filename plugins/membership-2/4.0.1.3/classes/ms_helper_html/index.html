  605.     public static function compact_code( $html ) { 
  606.         $html = str_replace( array( "\r\n", "\r" ), "\n", $html ); 
  607.         $lines = explode( "\n", $html ); 
  608.         $new_lines = array(); 
  609.  
  610.         foreach ( $lines as $i => $line ) {
