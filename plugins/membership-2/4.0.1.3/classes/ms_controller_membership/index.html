  313.     public function autodetect_membership( $preferred = 0, $no_member_check = false, $ignore_system = false ) { 
  314.         $membership_id = 0; 
  315.  
  316.         // Check 1: If the preferred value is correct use it. 
  317.         if ( $preferred ) { 
  318.             $membership = MS_Factory::load( 'MS_Model_Membership', $preferred ); 
  319.  
  320.             // Only use the membership_id if it's valid and not filtered by ignore_system. 
  321.             if ( $membership->is_valid() && $membership->id == $preferred ) { 
  322.                 if ( ! $ignore_system || ! $membership->is_system() ) { 
  323.                     $membership_id = $membership->id; 
  324.                 } 
  325.             } 
  326.         } 
  327.  
  328.         // Check 2: Examine the REQUEST parameters to find a valid ID. 
  329.         if ( ! $membership_id ) { 
  330.             if ( ! $membership_id ) { 
  331.                 if ( isset( $_REQUEST['membership_id'] ) ) { 
  332.                     $membership_id = $_REQUEST['membership_id']; 
  333.                 } elseif ( isset( $_REQUEST['subscription_id'] ) ) { 
  334.                     $sub_id = $_REQUEST['subscription_id']; 
  335.                     $subscription = MS_Factory::load( 'MS_Model_Relationship', $sub_id ); 
  336.                     $membership_id = $subscription->membership_id; 
  337.                 } elseif ( isset( $_REQUEST['ms_relationship_id'] ) ) { 
  338.                     $sub_id = $_REQUEST['ms_relationship_id']; 
  339.                     $subscription = MS_Factory::load( 'MS_Model_Relationship', $sub_id ); 
  340.                     $membership_id = $subscription->membership_id; 
  341.                 } elseif ( isset( $_REQUEST['invoice_id'] ) ) { 
  342.                     $inv_id = $_REQUEST['invoice_id']; 
  343.                     $invoice = MS_Factory::load( 'MS_Model_Invoice', $inv_id ); 
  344.                     $membership_id = $invoice->membership_id; 
  345.                 } 
  346.                 $membership_id = intval( $membership_id ); 
  347.             } 
  348.  
  349.             // Reset the membership_id if it's invalid or filtered by ignore_system. 
  350.             if ( $membership_id ) { 
  351.                 $membership = MS_Factory::load( 'MS_Model_Membership', $membership_id ); 
  352.                 if ( ! $membership->is_valid() ) { 
  353.                     $membership_id = 0; 
  354.                 } elseif ( $membership->id != $membership_id ) { 
  355.                     $membership_id = 0; 
  356.                 } elseif ( $ignore_system && $membership->is_system ) { 
  357.                     $membership_id = 0; 
  358.                 } 
  359.             } 
  360.         } 
  361.  
  362.         // Check 3: Check subscriptions of the current user. 
  363.         if ( ! $no_member_check && ! $membership_id && is_user_logged_in() ) { 
  364.             $member = MS_Model_Member::get_current_member(); 
  365.             $subscription = $member->get_subscription( 'priority' ); 
  366.             if ( $subscription ) { 
  367.                 $membership_id = $subscription->membership_id; 
  368.             } 
  369.  
  370.             // Reset the membership_id if it's invalid or filtered by ignore_system. 
  371.             if ( $membership_id ) { 
  372.                 $membership = MS_Factory::load( 'MS_Model_Membership', $membership_id ); 
  373.                 if ( ! $membership->is_valid() ) { 
  374.                     $membership_id = 0; 
  375.                 } elseif ( $membership->id != $membership_id ) { 
  376.                     $membership_id = 0;

  378.                     $membership_id = 0;
