  557.      * - ms_init_done 
  558.      * 
  559.      * @since  1.0.0 
  560.      * @throws Exception When function is called too early. 
  561.      */ 
  562.     public function setup_admin_protection() { 
  563.         if ( ! is_admin() ) { return; } 
  564.  
  565.         // Make sure we stick to the correct workflow. 
  566.         if ( ! did_action( 'ms_init_done' ) ) { 
  567.             throw new Exception( 'setup_admin_protection() is called too early.', 1 ); 
  568.             return; 
  569.         } 
  570.  
  571.         do_action( 'ms_setup_admin_protection', $this ); 
  572.  
  573.         // Search permissions through all memberships joined. 
  574.         foreach ( $this->member->subscriptions as $subscription ) { 
  575.             // Verify status of the membership. 
  576.             // Only active, trial or canceled (until it expires) status memberships. 
  577.             if ( ! $this->member->has_membership( $subscription->membership_id ) ) { 
  578.                 continue; 
  579.             } 
  580.  
  581.             $membership = $subscription->get_membership(); 
  582.             $membership->initialize( $subscription ); 
  583.  
  584.             // Protection is not applied for Admin users. 
  585.             if ( ! $this->member->is_normal_admin() ) { 
  586.                 $membership->protect_admin_content(); 
  587.             } 
  588.         } 
  589.  
  590.         do_action( 'ms_setup_admin_protection_done', $this ); 
  591.     }

  624.     /** 
  625.      * Runs a single Membership 2 cron service and then re-schedules it. 
  626.      * This function is used to manually trigger the cron services. 
  627.      * 
  628.      * @since  1.0.0 
  629.      * @param  string $hook Cron-Job to run on next page load. 
  630.      */ 
  631.     public function run_cron_services( $hook ) { 
  632.         wp_clear_scheduled_hook( $hook ); 
  633.         $this->setup_cron_services( $hook ); 
  634.  
  635.         // Note that we only remove the cron job and add it again. 
  636.         // As a result the job is re-scheduled with current timestamp and 
  637.         // therefore it will be executed instantly. 
  638.     } 
  639.  
  640.     /** 
  641.      * Setup cron plugin services. 
  642.      * 
  643.      * Setup cron to call actions. 
  644.      * The action-hook is called via the WordPress Cron implementation on a 
  645.      * regular basis - this hooks are set up only once. 
  646.      * 
  647.      * The Cron jobs can be manually executed by opening the admin page 
  648.      * "Membership2 > Settings" and adding URL param "&run_cron=1" 
  649.      * 
  650.      * @since  1.0.0 
  651.      * @param  string $reschedule Optional. Hook to re-schedule. 
  652.      */ 
  653.     public function setup_cron_services( $reschedule = null ) { 
  654.         do_action( 'ms_model_plugin_setup_cron_services_before', $this ); 
  655.  
  656.         $jobs = array( 
  657.             'ms_cron_check_membership_status' => '6hours',  
  658.             'ms_cron_process_communications' => 'hourly',  
  659.  ); 
  660.  
  661.         foreach ( $jobs as $hook => $interval ) { 
  662.             if ( ! wp_next_scheduled( $hook ) || $hook == $reschedule ) { 
  663.                 wp_schedule_event( time(), $interval, $hook ); 
  664.             } 
  665.         } 
  666.  
  667.         do_action( 'ms_model_plugin_setup_cron_services_after', $this ); 
  668.     } 
  669.  
  670.     /** 
  671.      * Check membership status. 
  672.      * 
  673.      * Execute actions when time/period condition are met. 
  674.      * E.g. change membership status, add communication to queue, create invoices. 
  675.      * 
  676.      * @since  1.0.0 
  677.      */ 
  678.     public function check_membership_status() { 
  679.         do_action( 'ms_model_plugin_check_membership_status_before', $this ); 
  680.  
  681.         if ( $this->member->is_simulated_user() ) { 
  682.             return; 
  683.         } 
  684.  
  685.         /** 
  686.          * For performance reasons we only process a small batch at once. 
  687.          * Here we find out, which subscriptions should be processed during 
  688.          * the current request. 
  689.          */ 
  690.         $offset = (int) MS_Factory::get_option( 'ms_batch_check_offset_flag' ); 
  691.  
  692.         // Find the next X subscriptions from DB. 
  693.         $args = apply_filters( 
  694.             'ms_model_plugin_check_membership_status_get_subscription_args',  
  695.             array( 
  696.                 'status' => 'valid',  
  697.                 'orderby' => 'ID',  
  698.                 'posts_per_page' => $this->_process_per_batch,  
  699.                 'offset' => $offset,  
  700.                 'nopaging' => false,  
  701.  ) 
  702.  ); 
  703.         $subscriptions = MS_Model_Relationship::get_subscriptions( $args ); 
  704.  
  705.         if ( count( $subscriptions ) < $this->_process_per_batch ) { 
  706.             // We processed all subscriptions. Clean up. 
  707.             MS_Factory::delete_option( 'ms_batch_check_offset_flag' ); 
  708.         } else { 
  709.             // We did not process all subscriptions. Remember where to continue. 
  710.             MS_Factory::update_option( 
  711.                 'ms_batch_check_offset_flag',  
  712.                 $offset + $this->_process_per_batch 
  713.  ); 
  714.  
  715.             // Re-scheduling the cron job will run it again on next page load. 
  716.             $hook = 'ms_cron_check_membership_status'; 
  717.             wp_clear_scheduled_hook( $hook ); 
  718.             $this->setup_cron_services( $hook ); 
  719.         } 
  720.                  
  721.                 $_SESSION['m2_status_check'] = 'inv'; 
  722.  
  723.         // Perform the actual status checks! 
  724.         foreach ( $subscriptions as $subscription ) { 
  725.             $subscription->check_membership_status(); 
  726.         } 
  727.                  
  728.         do_action( 'ms_model_plugin_check_membership_status_after', $this ); 
  729.     }

  731.     /** 
  732.      * Copies the full WordPress Admin menu before any restriction is applied 
  733.      * by WordPress or an Plugin. This menu-information is used on the 
  734.      * Membership2/Accessible Content settings pages 
  735.      * 
  736.      * @since  1.0.0 
  737.      * @global array $menu 
  738.      * @global array $submenu 
  739.      */ 
  740.     public function store_admin_menu() { 
  741.         global $menu, $submenu; 
  742.  
  743.         if ( ! isset( $this->admin_menu['main'] ) ) { 
  744.             $this->admin_menu = array( 
  745.                 'main' => $menu,  
  746.                 'sub' => $submenu,  
  747.  ); 
  748.         } else { 
  749.             foreach ( $menu as $pos => $item ) { 
  750.                 $this->admin_menu['main'][ $pos ] = $item; 
  751.             } 
  752.             foreach ( $submenu as $parent => $item ) { 
  753.                 $this->admin_menu['sub'][ $parent ] = $item; 
  754.             } 
  755.             ksort( $this->admin_menu['main'] ); 
  756.         } 
  757.     }
