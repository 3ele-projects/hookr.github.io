  400.                       $invoice 
  401.  );

  403.           }

  406.       return apply_filters( 
  407.           'ms_controller_billing_billing_do_action',

  422.   private function save_invoice( $fields ) {

  424.

  431.           $membership_id = $fields['membership_id'];

  441.                   $membership_id,  
  442.                   $member->id,  
  443.                   $gateway_id 
  444.  ); 
  445.           } else {

  447.               $subscription->save(); 
  448.           } 
  449.  
  450.           $invoice_id = intval( $fields['invoice_id'] ); 
  451.           $invoice = MS_Factory::load( 'MS_Model_Invoice', $invoice_id );

  454.               $msg = MS_Helper_Billing::BILLING_MSG_ADDED; 
  455.           } else {

  457.           } 
  458.  
  459.           foreach ( $fields as $field => $value ) {

  470.       return apply_filters( 
  471.           'ms_controller_billing_save_invoice',

  474.           $this
