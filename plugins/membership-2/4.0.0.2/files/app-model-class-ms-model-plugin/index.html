  139.               $this->member->add_membership( $ms_guest->id );

  142.           // Apply User-Membership to logged-in users without subscriptions. 
  143.           $ms_user = MS_Model_Membership::get_user();

  145.               $this->member->add_membership( $ms_user->id ); 
  146.           } 
  147.       } 
  148.  
  149.       // No subscription: Assign the base membership, which only denies access. 
  150.       if ( ! $this->member->has_membership() ) { 
  151.           $this->member->add_membership( 
  152.               MS_Model_Membership::get_base()->id 
  153.  ); 
  154.       }

  177.   /** 
  178.    * Returns an array with access-information on the current page/user 
  179.    * 
  180.    * @since  1.0.2 
  181.    * 
  182.    * @return array { 
  183.    *     Access information 
  184.    * 
  185.    *     @type bool $has_access If the current user can view the current page. 
  186.    *     @type array $memberships List of active membership-IDs the user has 
  187.    *         registered to. 
  188.    * } 
  189.    */ 
  190.   public function get_access_info() { 
  191.       static $Info = null; 
  192.  
  193.       if ( null === $Info ) { 
  194.           $Info = array( 
  195.               'has_access' => null,  
  196.               'is_admin' => false,  
  197.               'memberships' => array(),  
  198.               'url' => MS_Helper_Utility::get_current_url(),  
  199.  ); 
  200.  
  201.           // The ID of the main system membership. 
  202.           $base_id = MS_Model_Membership::get_base()->id; 
  203.  
  204.           $simulation = $this->member->is_simulated_user() || isset( $_GET['explain'] ) && 'access' == $_GET['explain']; 
  205.           if ( $simulation ) { $Info['reason'] = array(); } 
  206.  
  207.           if ( $this->member->is_normal_admin() ) { 
  208.               // Admins have access to ALL memberships. 
  209.               $Info['is_admin'] = true; 
  210.               $Info['has_access'] = true; 
  211.  
  212.               if ( $simulation ) { 
  213.                   $Info['reason'][] = __( 'Allow: Admin-User always has access', MS_TEXT_DOMAIN ); 
  214.               } 
  215.  
  216.               $memberships = MS_Model_Membership::get_memberships(); 
  217.               foreach ( $memberships as $membership ) { 
  218.                   $Info['memberships'][] = $membership->id; 
  219.               } 
  220.           } else { 
  221.               /** 
  222.                * A non-admin visitor is only guaranteed access to special 
  223.                * Membership2 pages: 
  224.                * Registration, Login, etc. 
  225.                */ 
  226.               $ms_page = MS_Model_Pages::current_page(); 
  227.               if ( $ms_page ) { 
  228.                   $Info['has_access'] = true; 
  229.  
  230.                   if ( $simulation ) { 
  231.                       $Info['reason'][] = __( 'Allow: This is a Membership Page', MS_TEXT_DOMAIN ); 
  232.                   } 
  233.               } 
  234.  
  235.               // Build a list of memberships the user belongs to and check permission. 
  236.               foreach ( $this->member->subscriptions as $subscription ) { 
  237.                   // Verify status of the membership. 
  238.                   // Only active, trial or canceled (until it expires) status memberships. 
  239.                   if ( ! $this->member->has_membership( $subscription->membership_id ) ) { 
  240.                       if ( $simulation ) { 
  241.                           $Info['reason'][] = sprintf( 
  242.                               __( 'Skipped: Not a member of "%s"', MS_TEXT_DOMAIN ),  
  243.                               $subscription->get_membership()->name 
  244.  ); 
  245.                       } 
  246.  
  247.                       continue; 
  248.                   } 
  249.  
  250.                   if ( $base_id !== $subscription->membership_id ) { 
  251.                       $Info['memberships'][] = $subscription->membership_id; 
  252.                   } 
  253.  
  254.                   // If permission is not clear yet then check current membership... 
  255.                   if ( true !== $Info['has_access'] ) { 
  256.                       $membership = $subscription->get_membership(); 
  257.                       $access = $membership->has_access_to_current_page(); 
  258.  
  259.                       if ( null === $access ) { 
  260.                           if ( $simulation ) { 
  261.                               $Info['reason'][] = sprintf( 
  262.                                   __( 'Ignored: Membership "%s"', MS_TEXT_DOMAIN ),  
  263.                                   $membership->name 
  264.  ); 
  265.                               $Info['reason'][] = $membership->_access_reason; 
  266.                           } 
  267.                           continue; 
  268.                       } 
  269.  
  270.                       if ( $simulation ) { 
  271.                           $Info['reason'][] = sprintf( 
  272.                               __( '%s: Membership "%s"', MS_TEXT_DOMAIN ),  
  273.                               $access ? __( 'Allow', MS_TEXT_DOMAIN ) : __( 'Deny', MS_TEXT_DOMAIN ),  
  274.                               $membership->name 
  275.  ); 
  276.  
  277.                           $Info['deciding_membership'] = $membership->id; 
  278.                           if ( $access ) { 
  279.                               $Info['deciding_rule'] = $membership->_allow_rule; 
  280.                           } else { 
  281.                               $Info['deciding_rule'] = $membership->_deny_rule; 
  282.                           } 
  283.                           $Info['reason'][] = $membership->_access_reason; 
  284.                       } 
  285.  
  286.                       $Info['has_access'] = $access; 
  287.                   } 
  288.               } 
  289.  
  290.               if ( null === $Info['has_access'] ) { 
  291.                   $Info['has_access'] = true; 
  292.  
  293.                   if ( $simulation ) { 
  294.                       $Info['reason'][] = __( 'Allow: Page is not protected', MS_TEXT_DOMAIN ); 
  295.                   } 
  296.               } 
  297.  
  298.               // "membership-id: 0" means: User does not belong to any membership. 
  299.               if ( ! count( $Info['memberships'] ) ) { 
  300.                   $Info['memberships'][] = 0; 
  301.               } 
  302.           } 
  303.  
  304.           $Info = apply_filters( 'ms_model_plugin_get_access_info', $Info ); 
  305.  
  306.           if ( $simulation ) { 
  307.               $access = lib2()->session->get_clear( 'ms-access' ); 
  308.               lib2()->session->add( 'ms-access', $Info ); 
  309.               for ( $i = 0; $i < 9; $i += 1 ) { 
  310.                   if ( isset( $access[$i] ) ) { 
  311.                       lib2()->session->add( 'ms-access', $access[$i] ); 
  312.                   } 
  313.               } 
  314.  
  315.               if ( WP_DEBUG && isset( $_GET['explain'] ) && 'access' == $_GET['explain'] ) { 
  316.                   echo '<style>code{background:#EEE;background:rgba(0, 0, 0, 0.1);padding:1px 4px;}</style>'; 
  317.                   echo '<h3>Note</h3>'; 
  318.                   echo '<p>To disable the URL param <code>?explain=access</code> you have to set <code>WP_DEBUG</code> to false.</p>'; 
  319.                   echo '<hr><h3>Recent Access checks</h3>'; 
  320.  
  321.                   lib2()->debug->stacktrace_off(); 
  322.                   foreach ( $access as $item ) { 
  323.                       printf( 
  324.                           '<a href="%1$s">%1$s</a>: <strong>%2$s</strong>',  
  325.                           $item['url'],  
  326.                           $item['has_access'] ? __( 'Allow', MS_TEXT_DOMAIN ) : __( 'Deny', MS_TEXT_DOMAIN ) 
  327.  ); 
  328.                       // Intended debug output, leave it here. 
  329.                       lib2()->debug->dump( $item ); 
  330.                   } 
  331.                   wp_die( '' ); 
  332.               } 
  333.           } 
  334.       } 
  335.  
  336.       return $Info; 
  337.   }

  356.  
  357.       if ( ! $access['has_access'] ) { 
  358.           MS_Model_Pages::create_missing_pages(); 
  359.           $no_access_page_url = MS_Model_Pages::get_page_url( 
  360.               MS_Model_Pages::MS_PAGE_PROTECTED_CONTENT,  
  361.               false 
  362.  ); 
  363.           $current_page_url = MS_Helper_Utility::get_current_url(); 
  364.  
  365.           // Don't (re-)redirect the protection page. 
  366.           if ( ! MS_Model_Pages::is_membership_page( null, MS_Model_Pages::MS_PAGE_PROTECTED_CONTENT ) ) { 
  367.               $no_access_page_url = esc_url_raw( 
  368.                   add_query_arg( 
  369.                       array( 'redirect_to' => $current_page_url ),  
  370.                       $no_access_page_url 
  371.  ) 
  372.  ); 
  373.  
  374.               $no_access_page_url = apply_filters( 
  375.                   'ms_model_plugin_protected_content_page',  
  376.                   $no_access_page_url 
  377.  ); 
  378.               wp_safe_redirect( $no_access_page_url ); 
  379.  
  380.               exit; 
  381.           } 
  382.       } 
  383.  
  384.       do_action( 'ms_model_plugin_protect_current_page_after', $this ); 
  385.   }

  581.    * 
  582.    * @since 1.0.0 
  583.    */ 
  584.   public function setup_cron_services( $reschedule = null ) { 
  585.       do_action( 'ms_model_plugin_setup_cron_services_before', $this ); 
  586.  
  587.       $jobs = array( 
  588.           'ms_cron_check_membership_status' => '6hours',  
  589.           'ms_cron_process_communications' => 'hourly',  
  590.  ); 
  591.  
  592.       foreach ( $jobs as $hook => $interval ) { 
  593.           if ( ! wp_next_scheduled( $hook ) || $hook == $reschedule ) { 
  594.               wp_schedule_event( time(), $interval, $hook ); 
  595.           } 
  596.       } 
  597.  
  598.       do_action( 'ms_model_plugin_setup_cron_services_after', $this ); 
  599.   }
