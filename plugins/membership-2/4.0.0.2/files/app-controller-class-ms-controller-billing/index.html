  67.   /** 
  68.    * Manages billing actions. 
  69.    * 
  70.    * Verifies GET and POST requests to manage billing. 
  71.    * 
  72.    * @since 1.0.0 
  73.    */ 
  74.   public function admin_billing_manager() { 
  75.       $this->print_admin_message(); 
  76.       $msg = 0; 
  77.       $redirect = false; 
  78.  
  79.       if ( ! $this->is_admin_user() ) { 
  80.           return; 
  81.       } 
  82.  
  83.       $fields = array( 'user_id', 'membership_id' ); 
  84.  
  85.       if ( self::validate_required( $fields ) 
  86.           && $this->verify_nonce() 
  87.  ) { 
  88.           // Save billing add/edit 
  89.           $msg = $this->save_invoice( $_POST ); 
  90.  
  91.           $redirect = esc_url_raw( 
  92.               add_query_arg( 
  93.                   array( 'msg' => $msg ),  
  94.                   remove_query_arg( array( 'invoice_id') ) 
  95.  ) 
  96.  ); 
  97.       } elseif ( self::validate_required( array( 'invoice_id' ) ) 
  98.           && $this->verify_nonce( 'bulk' ) 
  99.  ) { 
  100.           // Execute bulk actions. 
  101.           $action = $_POST['action'] != -1 ? $_POST['action'] : $_POST['action2']; 
  102.           $msg = $this->billing_do_action( $action, $_POST['invoice_id'] ); 
  103.           $redirect = esc_url_raw( add_query_arg( array( 'msg' => $msg ) ) ); 
  104.       } 
  105.  
  106.       if ( $redirect ) { 
  107.           wp_safe_redirect( $redirect ); 
  108.           exit; 
  109.       } 
  110.   }

  117.   public function admin_billing() {

  137.   /** 
  138.    * Perform actions for each invoice. 
  139.    * 
  140.    * @since 1.0.0 
  141.    * @param string $action The action to perform on selected invoices. 
  142.    * @param int[] $invoice_ids The list of invoices ids to process. 
  143.    */ 
  144.   public function billing_do_action( $action, $invoice_ids ) { 
  145.       $msg = MS_Helper_Billing::BILLING_MSG_NOT_UPDATED; 
  146.  
  147.       if ( $this->is_admin_user() && is_array( $invoice_ids ) ) { 
  148.           foreach ( $invoice_ids as $invoice_id ) { 
  149.               switch ( $action ) { 
  150.                   case 'delete': 
  151.                       $invoice = MS_Factory::load( 'MS_Model_Invoice', $invoice_id ); 
  152.                       $invoice->delete(); 
  153.                       $msg = MS_Helper_Billing::BILLING_MSG_DELETED; 
  154.                       break; 
  155.  
  156.                   default: 
  157.                       do_action( 'ms_controller_billing_do_action_' . $action, $invoice_ids ); 
  158.                       break; 
  159.               } 
  160.           } 
  161.       } 
  162.  
  163.       return apply_filters( 
  164.           'ms_controller_billing_billing_do_action',  
  165.           $msg,  
  166.           $action,  
  167.           $invoice_ids,  
  168.           $this 
  169.  ); 
  170.   }

  172.   /** 
  173.    * Save invoices using the invoices model. 
  174.    * 
  175.    * @since 1.0.0 
  176.    * 
  177.    * @param mixed $fields Transaction fields 
  178.    */ 
  179.   private function save_invoice( $fields ) { 
  180.       $msg = MS_Helper_Billing::BILLING_MSG_NOT_UPDATED; 
  181.  
  182.       if ( $this->is_admin_user() 
  183.           && is_array( $fields ) 
  184.           && ! empty( $fields['user_id'] ) 
  185.           && ! empty( $fields['membership_id'] ) 
  186.  ) { 
  187.           $member = MS_Factory::load( 'MS_Model_Member', $fields['user_id'] ); 
  188.           $membership_id = $fields['membership_id']; 
  189.           $gateway_id = 'admin'; 
  190.  
  191.           $subscription = MS_Model_Relationship::get_subscription( $member->id, $membership_id ); 
  192.           if ( empty( $subscription ) ) { 
  193.               $subscription = MS_Model_Relationship::create_ms_relationship( 
  194.                   $membership_id,  
  195.                   $member->id,  
  196.                   $gateway_id 
  197.  ); 
  198.           } else { 
  199.               $subscription->gateway_id = $gateway_id; 
  200.               $subscription->save(); 
  201.           } 
  202.  
  203.           $invoice = MS_Factory::load( 'MS_Model_Invoice', $fields['invoice_id'] ); 
  204.           if ( ! $invoice->is_valid() ) { 
  205.               $invoice = $subscription->get_current_invoice(); 
  206.               $msg = MS_Helper_Billing::BILLING_MSG_ADDED; 
  207.           } else { 
  208.               $msg = MS_Helper_Billing::BILLING_MSG_UPDATED; 
  209.           } 
  210.  
  211.           foreach ( $fields as $field => $value ) { 
  212.               $invoice->$field = $value; 
  213.           } 
  214.  
  215.           $invoice->save(); 
  216.  
  217.           if ( ! empty( $fields['execute'] ) ) { 
  218.               $invoice->changed(); 
  219.           } 
  220.       } 
  221.  
  222.       return apply_filters( 
  223.           'ms_controller_billing_save_invoice',  
  224.           $msg,  
  225.           $fields,  
  226.           $this 
  227.  ); 
  228.   }
