  534.   /** 
  535.    * Class autoloading callback function. 
  536.    * 
  537.    * Uses the **MS_** namespace to autoload classes when called. 
  538.    * Avoids creating include functions for each file in the MVC structure. 
  539.    * **MS_** namespace ONLY will be based on folder structure in /app/ 
  540.    * 
  541.    * @since 1.0.0 
  542.    * 
  543.    * @param  string $class Uses PHP autoloader function. 
  544.    * @return boolean 
  545.    */ 
  546.   private function class_loader( $class ) { 
  547.       static $Path_overrides = null; 
  548.  
  549.       /** 
  550.        * Actions to execute before the autoloader loads a class. 
  551.        * 
  552.        * @since 1.0.0 
  553.        * @param object $this The MS_Plugin object. 
  554.        */ 
  555.       do_action( 'ms_plugin_class_loader_pre_processing', $this ); 
  556.  
  557.       $basedir = dirname( __FILE__ ); 
  558.       $class = trim( $class ); 
  559.  
  560.       if ( null === $Path_overrides ) { 
  561.           /** 
  562.            * Adds and Filters class path overrides. 
  563.            * 
  564.            * @since 1.0.0 
  565.            * @param object $this The MS_Plugin object. 
  566.            */ 
  567.           $Path_overrides = apply_filters( 'ms_class_path_overrides', array(), $this ); 
  568.       } 
  569.  
  570.       if ( array_key_exists( $class, $Path_overrides ) ) { 
  571.           /** 
  572.            * Case 1: The class-path is explicitly defined in $Path_overrides. 
  573.            * Simply use the defined path to load the class. 
  574.            */ 
  575.           $file_path = $basedir . '/' . $Path_overrides[ $class ]; 
  576.  
  577.           /** 
  578.            * Overrides the filename and path. 
  579.            * 
  580.            * @since 1.0.0 
  581.            * @param object $this The MS_Plugin object. 
  582.            */ 
  583.           $file_path = apply_filters( 'ms_class_file_override', $file_path, $this ); 
  584.  
  585.           if ( is_file( $file_path ) ) { 
  586.               include_once $file_path; 
  587.           } 
  588.  
  589.           return true; 
  590.       } elseif ( 'MS_' == substr( $class, 0, 3 ) ) { 
  591.           /** 
  592.            * Case 2: The class-path is not explicitely defined in $Path_overrides. 
  593.            * Use /app/ path and class-name to build the file-name. 
  594.            */ 
  595.  
  596.           $path_array = explode( '_', $class ); 
  597.           array_shift( $path_array ); 
  598.           $alt_dir = array_pop( $path_array ); 
  599.           $sub_path = implode( '/', $path_array ); 
  600.  
  601.           $filename = str_replace( '_', '-', 'class-' . $class . '.php' ); 
  602.           $file_path = $basedir . '/app/' . strtolower( $sub_path . '/' . $filename ); 
  603.           $file_path_alt = $basedir . '/app/' . strtolower( $sub_path . '/' . $alt_dir . '/' . $filename ); 
  604.  
  605.           /** 
  606.            * Overrides the filename and path. 
  607.            * 
  608.            * @since 1.0.0 
  609.            * @param object $this The MS_Plugin object. 
  610.            */ 
  611.           $file_path = apply_filters( 'ms_class_file_override', $file_path, $this ); 
  612.           $file_path_alt = apply_filters( 'ms_class_file_override', $file_path_alt, $this ); 
  613.  
  614.           if ( is_file( $file_path ) ) { 
  615.               include_once $file_path;
