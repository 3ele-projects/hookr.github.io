  291.   static public function get_page_by( $field, $value ) { 
  292.       static $Page_list = array();

  295.           $Page_list[$field] = array();

  297.  
  298.       if ( ! isset( $Page_list[$field][ $value ] ) ) { 
  299.           $page_found = null; 
  300.  
  301.           switch ( $field ) { 
  302.               case 'id': $value = absint( $value ); break; 
  303.           }

  306.           $found = false; 
  307.  
  308.           foreach ( $ms_pages as $type => $page ) {

  310.                   case 'id':   $found = ($value === absint( $page->ID ) ); break; 
  311.                   case 'slug': $found = ($value === $page->post_name ); break; 
  312.               } 
  313.  
  314.               if ( $found ) { 
  315.                   $page_found = $page; 
  316.                   break; 
  317.               } 
  318.           } 
  319.  
  320.           $Page_list[$field][ $value ] = apply_filters( 
  321.               'ms_model_pages_get_page_by_id',  
  322.               $page_found,  
  323.               $field,  
  324.               $value 
  325.  ); 
  326.       } 
  327.  
  328.       return $Page_list[$field][ $value ]; 
  329.   }

  411.               if ( ! empty( $page_type ) ) { 
  412.                   /**

  419.                       $this_page = $expected_page; 
  420.                   } 
  421.               } else {

  435.                           $this_page = self::get_page_by( 'id', $page_id );

  438.                       } 
  439.                   } 
  440.               } 
  441.           } 
  442.  
  443.           $Res[$key] = apply_filters( 
  444.               'ms_model_pages_current_page',  
  445.               $this_page 
  446.  );

  448.  
  449.       return $Res[$key];

  715.                    * @since 1.1.0

  718.                       'ms_model_pages_create_wp_page',  
  719.                       $new_id 
  720.  ); 
  721.               } 
  722.           } 
  723.       } 
  724.       MS_Factory::revert_blog(); 
  725.  
  726.       return apply_filters( 
  727.           'ms_model_pages_create_missing_page',  
  728.           $res
