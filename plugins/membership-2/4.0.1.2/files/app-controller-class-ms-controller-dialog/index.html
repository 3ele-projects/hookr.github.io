  91.   public function ajax_login() { 
  92.       $resp = array(); 
  93.  
  94.       // First check the nonce, if it fails the function will break 
  95.       check_ajax_referer( 'ms-ajax-login' ); 
  96.  
  97.       /** 
  98.        * The login fields have alternative names: 
  99.        * - username or log 
  100.        * - password or pwd 
  101.        * - remember or rememberme 
  102.        */ 
  103.       lib3()->array->equip_post( 
  104.           'username',  
  105.           'password',  
  106.           'remember',  
  107.           'log',  
  108.           'pwd',  
  109.           'rememberme' 
  110.  ); 
  111.  
  112.       if ( empty( $_POST['username'] ) && ! empty( $_POST['log'] ) ) { 
  113.           $_POST['username'] = $_POST['log']; 
  114.       } 
  115.       if ( empty( $_POST['password'] ) && ! empty( $_POST['pwd'] ) ) { 
  116.           $_POST['password'] = $_POST['pwd']; 
  117.       } 
  118.       if ( empty( $_POST['remember'] ) && ! empty( $_POST['rememberme'] ) ) { 
  119.           $_POST['remember'] = $_POST['rememberme']; 
  120.       } 
  121.  
  122.       lib3()->array->equip_post( 'username', 'password', 'remember' ); 
  123.       lib3()->array->strip_slashes( $_POST, 'password' ); 
  124.  
  125.       // Nonce is checked, get the POST data and sign user on 
  126.       $info = array( 
  127.           'user_login' => $_POST['username'],  
  128.           'user_password' => $_POST['password'],  
  129.           'remember' => (bool) $_POST['remember'],  
  130.  ); 
  131.  
  132.       $user_signon = wp_signon( $info, false ); 
  133.       if ( is_wp_error( $user_signon ) ) { 
  134.           $resp['error'] = __( 'Wrong username or password', 'membership2' ); 
  135.       } else { 
  136.           $member = MS_Factory::load( 'MS_Model_Member', $user_signon->ID ); 
  137.  
  138.           // Also used in class-ms-model-member.php (signon_user) 
  139.           wp_set_current_user( $member->id ); 
  140.           wp_set_auth_cookie( $member->id ); 
  141.           do_action( 'wp_login', $member->username, $user_signon ); 
  142.           do_action( 'ms_model_member_signon_user', $user_signon, $member ); 
  143.  
  144.           $resp['loggedin'] = true; 
  145.           $resp['success'] = __( 'Logging in...', 'membership2' ); 
  146.  
  147.           /** 
  148.            * Allows a custom redirection after login. 
  149.            * Empty value will use the default redirect option of the login form. 
  150.            * 
  151.            * @since  1.0.0 
  152.            */ 
  153.                       $enforce = false; 
  154.                       if( isset( $_POST['redirect_to'] ) ) { 
  155.                           $resp['redirect'] = apply_filters( 'ms-ajax-login-redirect', $_POST['redirect_to'], $member ); 
  156.                       }else{ 
  157.                           $resp['redirect'] = apply_filters( 
  158.               'ms_url_after_login',  
  159.               $_POST['redirect_to'],  
  160.               $enforce 
  161.  ); 
  162.                       } 
  163.            
  164.       } 
  165.  
  166.       $this->respond( $resp ); 
  167.   }

  173.    * @internal 
  174.    */ 
  175.   public function ajax_lostpass() { 
  176.       $resp = array(); 
  177.  
  178.       // First check the nonce, if it fails the function will break 
  179.       check_ajax_referer( 'ms-ajax-lostpass' ); 
  180.  
  181.       // Nonce is checked, get the POST data and sign user on 
  182.       $errors = new WP_Error(); 
  183.  
  184.       if ( empty( $_POST['user_login'] ) ) { 
  185.           $resp['error'] = __( 'Enter a username or e-mail address.', 'membership2' ); 
  186.       } else if ( strpos( $_POST['user_login'], '@' ) ) { 
  187.           $user_data = get_user_by( 'email', trim( $_POST['user_login'] ) ); 
  188.           if ( empty( $user_data ) ) { 
  189.               $resp['error'] = __( 'There is no user registered with that email address.', 'membership2' ); 
  190.           } 
  191.       } else { 
  192.           $login = trim( $_POST['user_login'] ); 
  193.           $user_data = get_user_by( 'login', $login ); 
  194.       } 
  195.  
  196.       do_action( 'lostpassword_post' ); 
  197.  
  198.       if ( ! empty( $resp['error'] ) ) { 
  199.           $this->respond( $resp ); 
  200.       } 
  201.  
  202.       if ( ! $user_data ) { 
  203.           $resp['error'] = __( 'Invalid username or e-mail.', 'membership2' ); 
  204.           $this->respond( $resp ); 
  205.       } 
  206.  
  207.       // Redefining user_login ensures we return the right case in the email. 
  208.       $user_login = $user_data->user_login; 
  209.       $user_email = $user_data->user_email; 
  210.  
  211.       do_action( 'retreive_password', $user_login ); // Legacy (misspelled) 
  212.       do_action( 'retrieve_password', $user_login ); 
  213.  
  214.       $allow = apply_filters( 'allow_password_reset', true, $user_data->ID ); 
  215.  
  216.       if ( ! $allow ) { 
  217.           $resp['error'] = __( 'Password reset is not allowed for this user', 'membership2' ); 
  218.           $this->respond( $resp ); 
  219.       } elseif ( is_wp_error( $allow ) ) { 
  220.           return $allow; 
  221.       } 
  222.  
  223.       // Save an event about the password reset; also send the email template. 
  224.       $member = MS_Factory::load( 'MS_Model_Member', $user_data->ID ); 
  225.       MS_Model_Event::save_event( MS_Model_Event::TYPE_MS_RESETPASSWORD, $member ); 
  226.  
  227.       // Send our default email if the user does not have a custom email template in place. 
  228.       if ( ! apply_filters( 'ms_sent_reset_password_email', false ) ) { 
  229.           // Get a new reset-key. 
  230.           $reset = $member->new_password_reset_key(); 
  231.  
  232.           $schema = is_ssl() ? 'https' : 'http'; 
  233.  
  234.           $message = sprintf( 
  235.               __( 'Someone has requested a password reset for the following account: %sIf this was a mistake, just ignore this email and nothing will happen.%s %s', 'membership2' ),  
  236.               "\r\n\r\n" . network_home_url( '/', $schema ) . "\r\n\r\n" . 
  237.               sprintf( __( 'Username: %s', 'membership2' ), $user_login ) . "\r\n\r\n",  
  238.                               "\r\n\r\n" . __( 'To reset your password, visit the following address:', 'membership2' ) . "\r\n",  
  239.               "\r\n<" . $reset->url . ">\r\n" 
  240.  ); 
  241.  
  242.           if ( is_multisite() ) { 
  243.               $blogname = $GLOBALS['current_site']->site_name; 
  244.           } else { 
  245.               $blogname = wp_specialchars_decode( get_option( 'blogname' ), ENT_QUOTES ); 
  246.           } 
  247.  
  248.           $title = sprintf( __( '[%s] Password Reset' ), $blogname ); 
  249.  
  250.           $title = apply_filters( 'retrieve_password_title', $title ); 
  251.           $message = apply_filters( 'retrieve_password_message', $message, $reset->key, $reset->url ); 
  252.  
  253.           if ( $message && ! wp_mail( $user_email, wp_specialchars_decode( $title ), $message ) ) { 
  254.               $resp['error'] = __( 'The e-mail could not be sent.' ) . '<br />' . 
  255.                   __( 'Possible reason: your host may have disabled the mail() function.' ); 
  256.           } else { 
  257.               $resp['success'] = __( 'Check your e-mail for the confirmation link.', 'membership2' ); 
  258.           } 
  259.       } else { 
  260.           $resp['success'] = __( 'Check your e-mail for the confirmation link.', 'membership2' ); 
  261.       } 
  262.  
  263.       $this->respond( $resp ); 
  264.   }
