  211.   public function __construct() { 
  212.       add_filter( 
  213.           'ms_class_path_overrides',  
  214.           array( $this, 'ms_class_path_overrides' )

  222.   /** 
  223.    * Hooks 'ms_class_path_overrides'. 
  224.    * 
  225.    * Overrides plugin class paths to adhere to naming conventions 
  226.    * where object names are separated by underscores or for special cases. 
  227.    * 
  228.    * @since  1.0.0 
  229.    * 
  230.    * @param  array $overrides Array passed in by filter. 
  231.    * @return array(class=>path) Classes with new file paths. 
  232.    */ 
  233.   public function ms_class_path_overrides( $overrides ) { 
  234.       $models_base = 'app/model/'; 
  235.       $models = array( 
  236.           'MS_Model_Communication_After_Finishes' => 'communication/class-ms-model-communication-after-finishes.php',  
  237.           'MS_Model_Communication_After_Payment_Due' => 'communication/class-ms-model-communication-after-payment-due.php',  
  238.           'MS_Model_Communication_Before_Finishes' => 'communication/class-ms-model-communication-before-finishes.php',  
  239.           'MS_Model_Communication_Before_Payment_Due' => 'communication/class-ms-model-communication-before-payment-due.php',  
  240.           'MS_Model_Communication_Before_Trial_Finishes' => 'communication/class-ms-model-communication-before-trial-finishes.php',  
  241.           'MS_Model_Communication_Credit_Card_Expire' => 'communication/class-ms-model-communication-credit-card-expire.php',  
  242.           'MS_Model_Communication_Failed_Payment' => 'communication/class-ms-model-communication-failed-payment.php',  
  243.           'MS_Model_Communication_Info_Update' => 'communication/class-ms-model-communication-info-update.php',  
  244.           'MS_Model_Communication_Registration_Free' => 'communication/class-ms-model-communication-registration-free.php',  
  245.  ); 
  246.  
  247.       foreach ( $models as $key => $path ) { 
  248.           $overrides[ $key ] = $models_base . $path; 
  249.       } 
  250.  
  251.       return $overrides; 
  252.   } 
  253.  
  254.   /** 
  255.    * Class autoloading callback function. 
  256.    * 
  257.    * Uses the **MS_** namespace to autoload classes when called. 
  258.    * Avoids creating include functions for each file in the MVC structure. 
  259.    * **MS_** namespace ONLY will be based on folder structure in /app/ 
  260.    * 
  261.    * @since  1.0.0 
  262.    * 
  263.    * @param  string $class Uses PHP autoloader function. 
  264.    * @return boolean 
  265.    */ 
  266.   private function class_loader( $class ) { 
  267.       static $Path_overrides = null; 
  268.  
  269.       /** 
  270.        * Actions to execute before the autoloader loads a class. 
  271.        * 
  272.        * @since  1.0.0 
  273.        * @param object $this The MS_Plugin object. 
  274.        */ 
  275.       do_action( 'ms_plugin_class_loader_pre_processing', $this ); 
  276.  
  277.       $basedir = dirname( __FILE__ ); 
  278.       $class = trim( $class ); 
  279.  
  280.       if ( null === $Path_overrides ) { 
  281.           /** 
  282.            * Adds and Filters class path overrides. 
  283.            * 
  284.            * @since  1.0.0 
  285.            * @param object $this The MS_Plugin object. 
  286.            */ 
  287.           $Path_overrides = apply_filters( 'ms_class_path_overrides', array(), $this ); 
  288.       } 
  289.  
  290.       if ( array_key_exists( $class, $Path_overrides ) ) { 
  291.           /** 
  292.            * Case 1: The class-path is explicitly defined in $Path_overrides. 
  293.            * Simply use the defined path to load the class. 
  294.            */ 
  295.           $file_path = $basedir . '/' . $Path_overrides[ $class ]; 
  296.  
  297.           /** 
  298.            * Overrides the filename and path. 
  299.            * 
  300.            * @since  1.0.0 
  301.            * @param object $this The MS_Plugin object. 
  302.            */ 
  303.           $file_path = apply_filters( 'ms_class_file_override', $file_path, $this ); 
  304.  
  305.           if ( is_file( $file_path ) ) { 
  306.               include_once $file_path; 
  307.           } 
  308.  
  309.           return true; 
  310.       } elseif ( 'MS_' == substr( $class, 0, 3 ) ) { 
  311.           /** 
  312.            * Case 2: The class-path is not explicitely defined in $Path_overrides. 
  313.            * Use /app/ path and class-name to build the file-name. 
  314.            */ 
  315.  
  316.           $path_array = explode( '_', $class ); 
  317.           array_shift( $path_array ); // Remove the 'MS' prefix from path. 
  318.           $alt_dir = array_pop( $path_array ); 
  319.           $sub_path = implode( '/', $path_array ); 
  320.  
  321.           $filename = str_replace( '_', '-', 'class-' . $class . '.php' ); 
  322.           $file_path = trim( strtolower( $sub_path . '/' . $filename ), '/' ); 
  323.           $file_path_alt = trim( strtolower( $sub_path . '/' . $alt_dir . '/' . $filename ), '/' ); 
  324.           $candidates = array(); 
  325.  
  326.            
  327.  
  328.           // If no premium class is found check for default app class. 
  329.           $candidates[] = $basedir . '/app/' . $file_path; 
  330.           $candidates[] = $basedir . '/app/' . $file_path_alt; 
  331.  
  332.           foreach ( $candidates as $path ) { 
  333.               if ( is_file( $path ) ) { 
  334.                   include_once $path; 
  335.                   return true; 
  336.               } 
  337.           } 
  338.       } 
  339.  
  340.       return false; 
  341.   } 
  342. };

  357.       function ms_ajax_login() { 
  358.           $resp = array(); 
  359.           check_ajax_referer( 'ms-ajax-login' ); 
  360.  
  361.           if ( empty( $_POST['username'] ) && ! empty( $_POST['log'] ) ) { 
  362.               $_POST['username'] = $_POST['log']; 
  363.           } 
  364.           if ( empty( $_POST['password'] ) && ! empty( $_POST['pwd'] ) ) { 
  365.               $_POST['password'] = $_POST['pwd']; 
  366.           } 
  367.           if ( empty( $_POST['remember'] ) && ! empty( $_POST['rememberme'] ) ) { 
  368.               $_POST['remember'] = $_POST['rememberme']; 
  369.           } 
  370.  
  371.           // Nonce is checked, get the POST data and sign user on 
  372.           $info = array( 
  373.               'user_login' => $_POST['username'],  
  374.               'user_password' => $_POST['password'],  
  375.               'remember' => (bool) isset( $_POST['remember'] ) ? $_POST['remember'] : false,  
  376.  ); 
  377.  
  378.           $user_signon = wp_signon( $info, false ); 
  379.  
  380.           if ( is_wp_error( $user_signon ) ) { 
  381.               $resp['error'] = __( 'Wrong username or password', 'membership2' ); 
  382.           } else { 
  383.               $resp['loggedin'] = true; 
  384.               $resp['success'] = __( 'Logging in...', 'membership2' ); 
  385.  
  386.               /** 
  387.                * Allows a custom redirection after login. 
  388.                * Empty value will use the default redirect option of the login form. 
  389.                */ 
  390.  
  391.               // TODO: These filters are never called! 
  392.               //       This code is too early to allow any other plugin to register a filter handler... 
  393.               $enforce = false; 
  394.               if ( isset( $_POST['redirect_to'] ) ) { 
  395.                   $resp['redirect'] = apply_filters( 
  396.                       'ms-ajax-login-redirect',  
  397.                       $_POST['redirect_to'],  
  398.                       $user_signon->ID 
  399.  ); 
  400.               } else { 
  401.                   $resp['redirect'] = apply_filters( 
  402.                       'ms_url_after_login',  
  403.                       $_POST['redirect_to'],  
  404.                       $enforce 
  405.  ); 
  406.               } 
  407.           } 
  408.  
  409.           echo json_encode( $resp ); 
  410.           exit(); 
  411.       } 
  412.   }
