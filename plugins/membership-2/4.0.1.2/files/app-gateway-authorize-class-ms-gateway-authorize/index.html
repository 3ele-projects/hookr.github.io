  253.                       __( 'Payment Failed: code %s, subcode %s, reason code %, reason %s', 'membership2' ),

  255.                       $transaction_response->response_subcode,  
  256.                       $transaction_response->response_reason_code,  
  257.                       $transaction_response->response_reason 
  258.  ); 
  259.               } 
  260.           } else { 
  261.               $notes = __( 'Payment Failed: ', 'membership2' ) . $response->getMessageText(); 
  262.           } 
  263.       } elseif ( $need_code && ! $have_code ) { 
  264.           $notes = __( 'Secure payment failed: No Card Code found', 'membership2' ); 
  265.       } 
  266.  
  267.       // Mask the credit card number before logging it to database. 
  268.       $card_num = '';

  278.               STR_PAD_LEFT

  281.       if ( isset( $_POST['card_code'] ) ) {

  288.           self::ID, // gateway ID

  290.           $success, // success flag 
  291.           $subscription->id, // subscription ID 
  292.           $invoice->id, // invoice ID 
  293.           $amount, // charged amount 
  294.           $notes, // Descriptive text 
  295.           $external_id // External ID 
  296.  ); 
  297.  
  298.       // Restore the POST data in case it's used elsewhere. 
  299.       $_POST['card_num'] = $card_num; 
  300.       $_POST['card_code'] = $card_code;

  302.       unset( $_POST['API Out: PaymentProfileID'] );
