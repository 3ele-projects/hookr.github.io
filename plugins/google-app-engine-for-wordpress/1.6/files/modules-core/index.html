  29. /** 
  30.  * Core functionality for WordPress on App Engine 
  31.  * 
  32.  * This includes setting some sensible defaults (e.g. rewrites, SSL admin, file 
  33.  * editing), as well as infrastructure for the other modules. 
  34.  * 
  35.  * @package WordPress 
  36.  * @subpackage Mail 
  37.  */ 
  38. class Core { 
  39.   /** 
  40.    * Normal priority for filters 
  41.    */ 
  42.   const NORMAL_PRIORITY = 10; 
  43.  
  44.   /** 
  45.    * Low priority for filters 
  46.    */ 
  47.   const LOW_PRIORITY = 100; 
  48.  
  49.   /** 
  50.    * When was the admin.css file last updated? 
  51.    */ 
  52.   const CSS_VERSION = '201305100635'; 
  53.  
  54.   /** 
  55.    * Set required settings and register our actions 
  56.    */ 
  57.   public static function bootstrap() { 
  58.       global $PHP_SELF; 
  59.       $_SERVER['PHP_SELF'] = $PHP_SELF = preg_replace( '/(\?.*)?$/', '', $_SERVER['REQUEST_URI'] ); 
  60.  
  61.       add_filter( 'got_rewrite', '__return_true', self::LOW_PRIORITY ); 
  62.       if( is_production() ) { 
  63.           add_filter( 'secure_auth_redirect', '__return_true' ); 
  64.           force_ssl_admin( true ); 
  65.            
  66.           defined( 'DISALLOW_FILE_EDIT' ) or define( 'DISALLOW_FILE_EDIT', true ); 
  67.           defined( 'DISALLOW_FILE_MODS' ) or define( 'DISALLOW_FILE_MODS', true ); 
  68.       } 
  69.       defined( 'DISABLE_WP_CRON' ) or define( 'DISABLE_WP_CRON', true ); 
  70.  
  71.   // We don't want to use fsockopen as on App Engine it's not efficient 
  72.   add_filter( 'use_fsockopen_transport', '__return_false' ); 
  73.  
  74.       // ::settings_link() takes 2 parameters 
  75.       add_filter( 'plugin_action_links', __CLASS__ . '::settings_link', self::NORMAL_PRIORITY, 2 ); 
  76.       add_action( 'admin_enqueue_scripts', __CLASS__ . '::register_styles' ); 
  77.       add_action( 'admin_menu', __CLASS__ . '::register_settings_page' ); 
  78.       add_action( 'admin_init', __CLASS__ . '::register_settings' ); 
  79.       add_action( 'init', __CLASS__ . '::load_textdomain' ); 
  80.  
  81.   } 
  82.  
  83.   /** 
  84.    * Load the translation text domain for App Engine 
  85.    * 
  86.    * @wp-action init 
  87.    */ 
  88.   public static function load_textdomain() { 
  89.       load_plugin_textdomain( 'appengine', false, dirname( plugin_basename( PLUGIN_PATH ) ) . '/languages/' ); 
  90.   } 
  91.  
  92.   /** 
  93.    * Add a settings link to the plugin 
  94.    * 
  95.    * @wp-action plugin_action_links 
  96.    */ 
  97.   public static function settings_link( $links, $file ) { 
  98.       if ( $file == plugin_basename( PLUGIN_PATH ) ) { 
  99.           $links[] = '<a href="' . admin_url( 'options-general.php?page=appengine' ) . '">' 
  100.               . __( 'Settings', 'appengine' ) . '</a>'; 
  101.       } 
  102.  
  103.       return $links; 
  104.   } 
  105.  
  106.   /** 
  107.    * Register the App Engine settings page 
  108.    * 
  109.    * @wp-action admin_menu 
  110.    */ 
  111.   public static function register_settings_page() { 
  112.       add_options_page( 
  113.           __( 'App Engine Options', 'appengine' ),  
  114.           __( 'App Engine', 'appengine' ),  
  115.           'manage_options',  
  116.           'appengine',  
  117.           __CLASS__ . '::settings_view' 
  118.  ); 
  119.   } 
  120.  
  121.   /** 
  122.    * Register the styles for the App Engine administration UI 
  123.    * 
  124.    * @wp-action admin_enqueue_scripts 
  125.    */ 
  126.   public static function register_styles() { 
  127.       wp_enqueue_style( 'appengine-admin', plugins_url( 'static/admin.css', PLUGIN_PATH ), array(), self::CSS_VERSION, 'all' ); 
  128.   } 
  129.  
  130.   /** 
  131.    * Display the App Engine options page 
  132.    * 
  133.    * This is registered in {@see self::register_settings_page()} 
  134.    */ 
  135.   public static function settings_view() { 
  136. ?> 
  137.       <div class="wrap"> 
  138.           <?php screen_icon( 'appengine' ); ?> 
  139.           <h2><?php _e( 'Google App Engine Options', 'appengine' ) ?></h2> 
  140.           <form action="options.php" method="POST"> 
  141.               <?php 
  142.                   settings_fields( 'appengine_settings' ); 
  143.                   do_settings_sections( 'appengine' ); 
  144.                   submit_button(); 
  145.               ?> 
  146.           </form> 
  147.       </div> 
  148. <?php 
  149.   } 
  150.  
  151.   /** 
  152.    * Register the App Engine settings 
  153.    * 
  154.    * This is a better hook for the modules to use, as it's much more 
  155.    * descriptive, plus using it ensures that the Core module has loaded. 
  156.    * 
  157.    * @wp-action admin_init 
  158.    */ 
  159.   public static function register_settings() { 
  160.       do_action( 'appengine_register_settings' ); 
  161.   } 
  162. }
