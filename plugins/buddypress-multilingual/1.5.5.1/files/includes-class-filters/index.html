  63.    * Appends BP URI components to language switcher base URLs

  66.   public function icl_ls_languages_filter( $languages ) { 
  67.  
  68.       if ( !bp_is_blog_page() ) { 
  69.  
  70.           if ( !is_null( $this->_icl_ls_languages ) ) { 
  71.               return $this->_icl_ls_languages; 
  72.           } 
  73.  
  74.           global $sitepress, $bp, $wp_query; 
  75.  
  76.           // Set page 
  77.           if ( !empty( $wp_query->queried_object_id ) 
  78.                   && get_post_type( $wp_query->queried_object_id ) == 'page' ) { 
  79.               $page = get_post( $wp_query->queried_object_id ); 
  80.           } else {

  83.                   'name' => $pagename,  
  84.                   'post_type' => 'page',  
  85.                   'posts_per_page' => 1 
  86.  ); 
  87.               $posts = get_posts( $args ); 
  88.               $page = array_shift( $posts ); 
  89.           } 
  90.  
  91.           if ( !empty( $page->ID ) ) { 
  92.               /** 
  93.                * If languages are empty (WPML failed in setting language switcher data) 
  94.                * re-create language switcher data. 
  95.                *  
  96.                * Only case so far known is when WP_Query queried_object is messed up by BP. 
  97.                * BP sets queried object to be BP content type, but it's fake WP_Post without ID. 
  98.                */ 
  99.               // @todo Add persistent message for admin to report and mark as deprecated 
  100.               if ( empty( $languages ) 
  101.                       && method_exists( $sitepress, 'set_wp_query' ) 
  102.                       && method_exists( $sitepress, 'get_ls_languages' ) ) { 
  103.  
  104.                   // Clone original $wp_query 
  105.                   $_wp_query = clone $wp_query; 
  106.                   // Fix query 
  107.                   $wp_query->queried_object_id = $page->ID; 
  108.                   $wp_query->queried_object = $page; 
  109.                   $sitepress->set_wp_query(); 
  110.                   remove_filter( 'icl_ls_languages', array($this, 'icl_ls_languages_filter') ); 
  111.                   // Re-create language switcher data 
  112.                   $languages = $sitepress->get_ls_languages(); 
  113.                   add_filter( 'icl_ls_languages', array($this, 'icl_ls_languages_filter') ); 
  114.                   // Restore $wp_query 
  115.                   unset( $wp_query ); 
  116.                   global $wp_query; 
  117.                   $wp_query = clone $_wp_query; 
  118.                   unset( $_wp_query ); 
  119.                   $sitepress->set_wp_query(); 
  120.               } 
  121.               /** 
  122.                * Append all URI components after base component. 
  123.                * For example member screen: 
  124.                * {http://localhost/es/miembros}/{keir/profile/view/} 
  125.                */ 
  126.               if ( is_array( $languages ) && get_option( 'permalink_structure' ) != '' ) { 
  127.                   $unfiltered_uri = $bp->unfiltered_uri; 
  128.                   $offset = intval( $bp->unfiltered_uri_offset ) + 1; 
  129.                   $append_array = array_slice( $unfiltered_uri, $offset ); 
  130.                   $append = implode( '/', $append_array ); 
  131.                   $current_language = apply_filters( 'wpml_current_language', null ); 
  132.                   foreach ( $languages as $code => &$language ) { 
  133.                       $translated_page_id = apply_filters( 'wpml_object_id', $page->ID, 'page', false, $code ); 
  134.                       if ( $translated_page_id ) { 
  135.                           do_action( 'wpml_switch_language', $code ); 
  136.                           $page_permalink = untrailingslashit( get_permalink( $translated_page_id ) ); 
  137.                           do_action( 'wpml_switch_language', $current_language ); 
  138.                           $language['url'] = "{$page_permalink}/{$append}"; 
  139.                       } 
  140.                   } 
  141.               } 
  142.           } 
  143.           $this->_icl_ls_languages = $languages; 
  144.       } 
  145.  
  146.       return $languages; 
  147.   }
