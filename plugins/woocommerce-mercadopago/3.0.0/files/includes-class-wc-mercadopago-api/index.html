  201.       $args = array( 
  202.           'back_urls' => array( 
  203.               'success' => esc_url( $this->gateway->get_return_url( $order ) ),  
  204.               'failure' => str_replace( '&', '&', $order->get_cancel_order_url() ),

  206.  ),  
  207.           'auto_return' => 'approved',  
  208.           'payer' => array( 
  209.               'name' => $order->billing_first_name,  
  210.               'surname' => $order->billing_last_name,  
  211.               'email' => $order->billing_email,

  213.           'external_reference' => $this->gateway->invoice_prefix . $order->id,  
  214.           'items' => array( 
  215.               array(

  217.                   'unit_price' => $order->get_total(),  
  218.                   'currency_id' => $order->get_order_currency(),  
  219.                   'category_id' => 'others', // Generic category ID.

  240.           } else { 
  241.               $shipping_method = ucwords( $order->shipping_method_title ); 
  242.           } 
  243.  
  244.           $args['items'][0]['title'] .= ', ' . __( 'Shipping via', 'woocommerce-mercadopago' ) . ' ' . $shipping_method;

  247.       // Deprecated since 3.0.0. 
  248.       $args = apply_filters( 'woocommerce_mercadopago_args', $args, $order );

  308.   /** 
  309.    * Get cliente token. 
  310.    * 
  311.    * @return mixed Sucesse return the token and error return null. 
  312.    */ 
  313.   protected function get_client_credentials() { 
  314.       if ( 'yes' == $this->gateway->debug ) { 
  315.           $this->gateway->log->add( $this->gateway->id, 'Getting client credentials...' ); 
  316.       } 
  317.  
  318.       // Set data. 
  319.       $data = build_query( array( 
  320.           'client_id' => $this->gateway->client_id,  
  321.           'client_secret' => $this->gateway->client_secret,  
  322.           'grant_type' => 'client_credentials',  
  323.  ) ); 
  324.       $headers = array( 
  325.           'Content-Type' => 'application/x-www-form-urlencoded',  
  326.  ); 
  327.  
  328.       $response = $this->do_request( $this->get_oauth_token_url(), 'POST', $data, $headers ); 
  329.  
  330.       // Check if the request was valid and return the token. 
  331.       if ( ! is_wp_error( $response ) && $response['response']['code'] >= 200 && $response['response']['code'] < 300 && ( strcmp( $response['response']['message'], 'OK' ) == 0 ) ) { 
  332.  
  333.           $token = json_decode( $response['body'] ); 
  334.  
  335.           if ( 'yes' == $this->gateway->debug ) { 
  336.               $this->gateway->log->add( $this->gateway->id, 'Received valid response from MercadoPago' ); 
  337.           } 
  338.  
  339.           return $token->access_token; 
  340.       } else { 
  341.           if ( 'yes' == $this->gateway->debug ) { 
  342.               $this->gateway->log->add( $this->gateway->id, 'Received invalid response from MercadoPago. Error response: ' . print_r( $response, true ) ); 
  343.           } 
  344.       } 
  345.  
  346.       return null; 
  347.   }

  354.    * 
  355.    * @return string

  357.   public function get_user_payment_url( $order, $subscription = false ) { 
  358.       $data = json_encode( $this->get_payment_args( $order, $subscription ) ); 
  359.  
  360.       if ( empty( $data ) ) { 
  361.           return ''; 
  362.       } 
  363.  
  364.       if ( 'yes' == $this->gateway->debug ) { 
  365.           $this->gateway->log->add( $this->gateway->id, 'Payment arguments for order ' . $order->get_order_number() . ': ' . print_r( $data, true ) ); 
  366.       } 
  367.  
  368.       $credentials = $this->get_client_credentials(); 
  369.       $url = $this->get_checkout_url( $credentials, $subscription ); 
  370.       $response = $this->do_request( $url, 'POST', $data ); 
  371.  
  372.       if ( ! is_wp_error( $response ) && 201 == $response['response']['code'] && ( 0 == strcmp( $response['response']['message'], 'Created' ) ) ) { 
  373.           $payment_data = json_decode( $response['body'] ); 
  374.  
  375.           if ( 'yes' == $this->gateway->debug ) { 
  376.               $this->gateway->log->add( $this->gateway->id, 'Payment link generated from MercadoPago successfully!' ); 
  377.           } 
  378.  
  379.           add_post_meta( $order->id, '_mercadopago_payment_id', sanitize_text_field( $payment_data->id ), true ); 
  380.  
  381.           if ( 'yes' == $this->gateway->sandbox ) { 
  382.               return $payment_data->sandbox_init_point; 
  383.           } else { 
  384.               return $payment_data->init_point; 
  385.           } 
  386.       } else { 
  387.           if ( 'yes' == $this->gateway->debug ) { 
  388.               $this->gateway->log->add( $this->gateway->id, 'Generate payment error response: ' . print_r( $response, true ) ); 
  389.           } 
  390.       } 
  391.  
  392.       wc_add_notice( __( 'An error has occurred while processing your order, please try again. Or contact us for assistance.', 'woocommerce-mercadopago' ), 'error' ); 
  393.  
  394.       return ''; 
  395.   }
