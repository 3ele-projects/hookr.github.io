  54.

  58.           $item->title = '<form action="'.esc_url( home_url( '/' ) ).'" class="woocommerce-product-search" method="get" role="search"> 
  59.               <input type="search" title="'.esc_attr_x( 'Search for:', 'label' ).'" name="s" value="'.get_search_query().'" placeholder="'.esc_html( isset( $titles[0] ) ? $titles[0] : $item->title ).'" class="search-field">

  62.           </form>'; break;

  223. function aiwoo_shortcode_productcat( $atts, $content = null ) { 
  224.   global $woocommerce, $wp_query, $post;     
  225.   $atts = shortcode_atts(array( 
  226.       "show_count" => 0,  
  227.       "hierarchical" => 0,  
  228.       "show_children_only" => 0,  
  229.       "dropdown" => 0,  
  230.       "hide_empty" => 0,  
  231.       "orderby" => "order" 
  232.  ), $atts ); 
  233.  
  234.   $c = $atts['show_count']; 
  235.   $h = $atts['hierarchical'];     
  236.   $s = $atts['show_children_only']; 
  237.   $d = $atts['dropdown']; 
  238.   $e = $atts['hide_empty']; 
  239.   $o = $atts['orderby']; 
  240.   $dropdown_args = array( 'hide_empty' => $e ); 
  241.   $list_args = array( 'show_count' => $c, 'hierarchical' => $h, 'taxonomy' => 'product_cat', 'hide_empty' => $e ); 
  242.  
  243.   // Menu Order 
  244.   $list_args['menu_order'] = false; 
  245.   if ( $o == 'order' ) { 
  246.       $list_args['menu_order'] = 'asc'; 
  247.   } else { 
  248.       $list_args['orderby'] = 'title'; 
  249.   } 
  250.  
  251.   // Setup Current Category 
  252.   $current_cat = false; 
  253.   $cat_ancestors = array(); 
  254.   if ( is_tax( 'product_cat' ) ) { 
  255.       $current_cat = $wp_query->queried_object; 
  256.       $cat_ancestors = get_ancestors( $current_cat->term_id, 'product_cat' ); 
  257.   } elseif ( is_singular( 'product' ) ) { 
  258.       $product_category = wc_get_product_terms( $post->ID, 'product_cat', array( 'orderby' => 'parent' ) ); 
  259.       if ( $product_category ) { 
  260.           $current_cat = end( $product_category ); 
  261.           $cat_ancestors = get_ancestors( $current_cat->term_id, 'product_cat' ); 
  262.       } 
  263.   } 
  264.  
  265.   // Show Siblings and Children Only 
  266.   if ( $s && $current_cat ) { 
  267.       // Top level is needed 
  268.       $top_level = get_terms( 
  269.           'product_cat',  
  270.           array( 
  271.               'fields' => 'ids',  
  272.               'parent' => 0,  
  273.               'hierarchical' => true,  
  274.               'hide_empty' => false 
  275.  ) 
  276.  ); 
  277.  
  278.       // Direct children are wanted 
  279.       $direct_children = get_terms( 
  280.           'product_cat',  
  281.           array( 
  282.               'fields' => 'ids',  
  283.               'parent' => $current_cat->term_id,  
  284.               'hierarchical' => true,  
  285.               'hide_empty' => false 
  286.  ) 
  287.  ); 
  288.  
  289.       // Gather siblings of ancestors 
  290.       $siblings = array(); 
  291.       if ( $cat_ancestors ) { 
  292.           foreach ( $cat_ancestors as $ancestor ) { 
  293.               $ancestor_siblings = get_terms( 
  294.                   'product_cat',  
  295.                   array( 
  296.                       'fields' => 'ids',  
  297.                       'parent' => $ancestor,  
  298.                       'hierarchical' => false,  
  299.                       'hide_empty' => false 
  300.  ) 
  301.  ); 
  302.               $siblings = array_merge( $siblings, $ancestor_siblings ); 
  303.           } 
  304.       } 
  305.       if ( $h ) { 
  306.           $include = array_merge( $top_level, $cat_ancestors, $siblings, $direct_children, array( $current_cat->term_id ) ); 
  307.       } else { 
  308.           $include = array_merge( $direct_children ); 
  309.       } 
  310.       $dropdown_args['include'] = implode( ', ', $include ); 
  311.       $list_args['include'] = implode( ', ', $include ); 
  312.  
  313.       if ( empty( $include ) ) { 
  314.           return; 
  315.       } 
  316.   } elseif ( $s ) { 
  317.       $dropdown_args['depth'] = 1; 
  318.       $dropdown_args['child_of'] = 0; 
  319.       $dropdown_args['hierarchical'] = 1; 
  320.       $list_args['depth'] = 1; 
  321.       $list_args['child_of'] = 0; 
  322.       $list_args['hierarchical'] = 1; 
  323.   } 
  324.  
  325.   // Dropdown

  332.           'selected' => $current_cat ? $current_cat->slug : '' 
  333.  ); 
  334.       $dropdown_args = wp_parse_args( $dropdown_args, $dropdown_defaults ); 
  335.       //wc_product_dropdown_categories( apply_filters( 'woocommerce_product_categories_widget_dropdown_args', $dropdown_args ) ); 
  336.

  340.         
  341.       if ( ! $terms ) { 
  342.           return; 
  343.       } 
  344.        
  345.       $content = $content != '' ? $content : __( 'Select a category' ); 
  346.       $cat_drop = "<select name='product_cat' class='dropdown_product_cat'>";

  361.   // List

  378. }
