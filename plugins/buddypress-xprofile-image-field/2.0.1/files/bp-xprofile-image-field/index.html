  340.       protected function bpxp_image_field_save($posted_field_ids) { 
  341.            
  342.           $post_action_found = false; 
  343.           $post_action = ''; 
  344.           if (isset($_POST['action'])) { 
  345.               $post_action_found = true; 
  346.  
  347.               $post_action = apply_filters('bpxp_image_field_preserve_post_action', $_POST['action']); 
  348.  
  349.           } 
  350.  
  351.           foreach ( (array)$posted_field_ids as $field_id ) { 
  352.               $field_name = 'field_' . $field_id; 
  353.  
  354.               if ( isset( $_FILES[$field_name] ) ) { 
  355.                   if($_FILES[$field_name]['size'] > 0) { 
  356.                       require_once( ABSPATH . '/wp-admin/includes/file.php' ); 
  357.  
  358.                       $uploaded_file = $_FILES[$field_name]['tmp_name']; 
  359.  
  360.                       // Filter the upload location 
  361.                       add_filter( 'upload_dir', array( $this, 'bpxp_image_field_profile_upload_dir'), 10, 1 ); 
  362.  
  363.                       //ensure WP accepts the upload job 
  364.                       $_POST['action'] = 'wp_handle_upload'; 
  365.  
  366.                       $wp_uploaded_file = wp_handle_upload( $_FILES[$field_name] ); 
  367.  
  368.                       $db_uploaded_file = str_replace(WP_CONTENT_URL, '', $wp_uploaded_file['url']) ; 
  369.  
  370.                       $uploaded_file = apply_filters('bpxp_image_field_image_uploaded', $db_uploaded_file, $wp_uploaded_file); 
  371.  
  372.                       $_POST[$field_name] = $uploaded_file; 
  373.                   } 
  374.                   else{ 
  375.                       $field_name_hidden = 'field_' . $field_id . '_hidden'; 
  376.                       if ( isset( $_POST[$field_name_hidden] ) ) { 
  377.                           $field_name_delete = 'field_' . $field_id . '_delete'; 
  378.                           if ( isset( $_POST[$field_name_delete] ) && $_POST[$field_name_delete] == 'deleted') { 
  379.                               $_POST[$field_name] = ''; 
  380.                           } 
  381.                           else { 
  382.                               $_POST[$field_name] = $_POST[$field_name_hidden]; 
  383.                           } 
  384.                       } 
  385.                   } 
  386.  
  387.               } 
  388.  
  389.           } 
  390.  
  391.           if($post_action_found) { 
  392.               $_POST['action'] = apply_filters('bpxp_image_field_restore_post_action', $post_action); 
  393.           } 
  394.           else{ 
  395.               unset($_POST['action']); 
  396.           } 
  397.       }
