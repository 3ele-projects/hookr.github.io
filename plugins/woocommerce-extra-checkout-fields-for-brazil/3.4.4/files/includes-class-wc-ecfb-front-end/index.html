  400.   public function valid_checkout_fields() { 
  401.       if ( apply_filters( 'wcbcf_disable_checkout_validation', false ) ) { 
  402.           return; 
  403.       } 
  404.  
  405.       // Get plugin settings. 
  406.       $settings = get_option( 'wcbcf_settings' ); 
  407.       $only_brazil = isset( $settings['only_brazil'] ) ? true : false; 
  408.       $billing_persontype = isset( $_POST['billing_persontype'] ) ? $_POST['billing_persontype'] : 0; 
  409.  
  410.       if ( $only_brazil && 'BR' != $_POST['billing_country'] || 0 == $settings['person_type'] ) { 
  411.           return; 
  412.       } 
  413.  
  414.       if ( 0 == $billing_persontype && 1 == $settings['person_type'] ) { 
  415.           wc_add_notice( sprintf( '<strong>%s</strong> %s.', __( 'Person type', 'woocommerce-extra-checkout-fields-for-brazil' ), __( 'is a required field', 'woocommerce-extra-checkout-fields-for-brazil' ) ), 'error' ); 
  416.       } else { 
  417.  
  418.           // Check CPF. 
  419.           if ( ( 1 == $settings['person_type'] && 1 == $billing_persontype ) || 2 == $settings['person_type'] ) { 
  420.               if ( empty( $_POST['billing_cpf'] ) ) { 
  421.                   wc_add_notice( sprintf( '<strong>%s</strong> %s.', __( 'CPF', 'woocommerce-extra-checkout-fields-for-brazil' ), __( 'is a required field', 'woocommerce-extra-checkout-fields-for-brazil' ) ), 'error' );
