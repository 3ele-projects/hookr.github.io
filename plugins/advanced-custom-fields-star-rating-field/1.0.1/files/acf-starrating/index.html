  74. function acf_srf_activation() { 
  75.   
  76.   global $wpdb; 
  77.  
  78.   $charset_collate = ''; 
  79.   if ( $wpdb->has_cap( 'collation' ) ) { 
  80.       if ( ! empty($wpdb->charset) ) { 
  81.           $charset_collate .= " DEFAULT CHARACTER SET $wpdb->charset"; 
  82.       } 
  83.       if ( ! empty($wpdb->collate) ) { 
  84.           $charset_collate .= " COLLATE $wpdb->collate"; 
  85.       } 
  86.   } 
  87.  
  88.   // need drop table? 
  89.   // $wpdb->get_var("DROP TABLE $wpdb->acfsrf"); 
  90.   if ( $wpdb->get_var("SHOW tables LIKE '$wpdb->acfsrf'") != $wpdb->acfsrf ) { 
  91.  
  92.       $sql = "CREATE TABLE " . $wpdb->acfsrf . " (". 
  93.           "acfsrf_id  bigint(20)      NOT NULL AUTO_INCREMENT, ". 
  94.           "field_key  VARCHAR(20)     NOT NULL, ". 
  95.           "meta_id    VARCHAR(256)    NOT NULL, ". 
  96.           "rating     INT(4)          NOT NULL, ". 
  97.           "timestamp  TIMESTAMP, ". 
  98.           "user_id    INT(10)         NOT NULL default '0', ". 
  99.           "user_ip    VARCHAR(40)     NOT NULL default '0', ". 
  100.           "user_cookie VARCHAR(200)   NOT NULL default '0', ". 
  101.           "UNIQUE KEY (acfsrf_id)) $charset_collate"; 
  102.  
  103.       require_once(ABSPATH . 'wp-admin/includes/upgrade.php'); 
  104.       dbDelta($sql); 
  105.        
  106.   } 
  107.    
  108. } 
  109. register_activation_hook(__FILE__, 'acf_srf_activation');

  158. /** 
  159.  * Deleting vote statistics when deleting an object (post, user, etc.).

  161.  
  162. function acf_srf_delete_userlog( $user_id ) { 
  163.   global $wpdb; 
  164.   $meta = 'user_' . $user_id; 
  165.   $result = $wpdb->get_var( $wpdb->prepare("DELETE FROM $wpdb->acfsrf WHERE meta_id = '%s'", $meta) );        
  166. } 
  167. add_action( 'delete_user', 'acf_srf_delete_userlog' ); 
  168.  
  169. function acf_srf_delete_postlog( $post_id ) {

  171.   $meta = $post_id; 
  172.   $result = $wpdb->get_var( $wpdb->prepare("DELETE FROM $wpdb->acfsrf WHERE meta_id = '%s'", $meta) );        
  173. } 
  174. add_action( 'delete_post', 'acf_srf_delete_postlog' ); 
  175. add_action( 'delete_attachment', 'acf_srf_delete_postlog' );
