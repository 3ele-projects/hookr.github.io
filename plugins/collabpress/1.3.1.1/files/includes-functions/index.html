  711. function is_collabpress_page( $slug = '' ) { 
  712.  
  713.   global $post; 
  714.   // If the page is not set 
  715.   if ( 
  716.       ( empty( $_REQUEST['page'] ) || $_REQUEST['page'] != 'collabpress-dashboard' ) 
  717.       && 
  718.       ( 
  719.           ( ! empty( $post->post_content ) && strpos( $post->post_content, '[collabpress]' ) === FALSE ) 
  720.           || empty( $post->post_content ) 
  721.  ) 
  722.  ) 
  723.       return apply_filters( 'is_collabpress_page', false ); 
  724.  
  725.   // Default, if we're just checking that we're on a CollabPress page 
  726.   if ( ! $slug ) 
  727.       return apply_filters( 'is_collabpress_page', true ); 
  728.   $return = false; 
  729.   if ( ! empty( $_REQUEST['project'] ) ) { 
  730.       if ( ! empty( $_REQUEST['view'] ) ) { 
  731.           if ( $slug == 'project-calendar' 
  732.             && $_REQUEST['view'] == 'calendar' ) 
  733.               $return = true; 
  734.           if ( $slug == 'project-tasks' 
  735.             && $_REQUEST['view'] == 'tasks' ) 
  736.               $return = true; 
  737.           if ( $slug == 'project-files' 
  738.             && $_REQUEST['view'] == 'files' ) 
  739.               $return = true; 
  740.           if ( $slug == 'project-users' 
  741.             && $_REQUEST['view'] == 'users' ) 
  742.               $return = true; 
  743.       } else { 
  744.           if ( $slug == 'project-overview' ) 
  745.               $return = true; 
  746.       } 
  747.   } else if ( ! empty( $_REQUEST['task'] ) ) { 
  748.       if ( $slug == 'task' ) 
  749.           $return = true; 
  750.   } else { 
  751.       if ( ! empty( $_REQUEST['view'] ) ) { 
  752.           if ( $slug == 'calendar' && $_REQUEST['view'] == 'calendar' ) 
  753.               $return = true; 
  754.           if ( $slug == 'activity' && $_REQUEST['view'] == 'activity' ) 
  755.               $return = true;

  758.               $return = true;

  760.   } 
  761.  
  762.   if ( ! isset( $return ) )
