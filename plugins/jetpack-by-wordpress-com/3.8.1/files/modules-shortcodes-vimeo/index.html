  12. function jetpack_shortcode_get_vimeo_id( $atts ) { 
  13.   if ( isset( $atts[0] ) ) { 
  14.       $atts[0] = trim( $atts[0] , '=' ); 
  15.       $id = false; 
  16.       if ( is_numeric( $atts[0] ) ) 
  17.           $id = (int) $atts[0]; 
  18.       elseif ( preg_match( '|vimeo\.com/(\d+)/?$|i', $atts[0], $match ) ) 
  19.           $id = (int) $match[1]; 
  20.       elseif ( preg_match( '|player\.vimeo\.com/video/(\d+)/?$|i', $atts[0], $match ) )

  33. function vimeo_shortcode( $atts ) { 
  34.   global $content_width; 
  35.  
  36.   extract( array_map( 'intval', shortcode_atts( array( 
  37.       'id' => 0,  
  38.       'width' => 400,  
  39.       'height' => 300,  
  40.       'autoplay' => 0,  
  41.       'loop' => 0,  
  42.  ), $atts, 'vimeo' ) ) ); 
  43.  
  44.   if ( isset( $atts[0] ) ) { 
  45.       $id = jetpack_shortcode_get_vimeo_id( $atts ); 
  46.   }

  50.   // [vimeo 141358 h=500&w=350] 
  51.   $params = shortcode_new_to_old_params( $atts ); // h=500&w=350

  65.   if ( isset( $args['h'] ) ) { 
  66.       $height = (int) $args['h']; 
  67.  
  68.       if ( ! isset( $args['w'] ) ) { 
  69.           $width = round( ( $height / 360 ) * 640 ); 
  70.       } 
  71.   }

  73.   if ( ! $width ) { 
  74.       $width = absint( $content_width ); 
  75.   } 
  76.  
  77.   if ( ! $height ) { 
  78.       $height = round( ( $width / 640 ) * 360 ); 
  79.   } 
  80.

  105.   // $args['autoplay'] is parsed from the embedded url. 
  106.   // $autoplay is parsed from shortcode arguments.

  108.   if ( ! empty( $args['autoplay'] ) || ! empty( $autoplay ) || in_array( 'autoplay', $atts ) ) {

  113.       $url = add_query_arg( 'loop', 1, $url ); 
  114.   }

  116.   $html = sprintf( '<div class="embed-vimeo" style="text-align:center;"><iframe src="%1$s" width="%2$u" height="%3$u" frameborder="0" webkitallowfullscreen mozallowfullscreen allowfullscreen></iframe></div>', esc_url( $url ), $width, $height );

  134. function vimeo_embed_to_shortcode( $content ) { 
  135.   if ( false === stripos( $content, 'player.vimeo.com/video/' ) ) 
  136.       return $content; 
  137.  
  138.   $regexp = '!<iframe\s+src=[\'"](https?:)?//player\.vimeo\.com/video/(\d+)[\w=&;?]*[\'"]((?:\s+\w+=[\'"][^\'"]*[\'"])*)((?:[\s\w]*))></iframe>!i'; 
  139.   $regexp_ent = str_replace( '&#0*58;', '&#0*58;|�*58;', htmlspecialchars( $regexp, ENT_NOQUOTES ) ); 
  140.  
  141.   foreach ( array( 'regexp', 'regexp_ent' ) as $reg ) { 
  142.       if ( !preg_match_all( $$reg, $content, $matches, PREG_SET_ORDER ) ) 
  143.           continue; 
  144.  
  145.       foreach ( $matches as $match ) { 
  146.           $id = (int) $match[2]; 
  147.  
  148.           $params = $match[3]; 
  149.  
  150.           if ( 'regexp_ent' == $reg ) 
  151.               $params = html_entity_decode( $params ); 
  152.  
  153.           $params = wp_kses_hair( $params, array( 'http' ) ); 
  154.  
  155.           $width = isset( $params['width'] ) ? (int) $params['width']['value'] : 0; 
  156.           $height = isset( $params['height'] ) ? (int) $params['height']['value'] : 0; 
  157.  
  158.           $wh = ''; 
  159.           if ( $width && $height ) 
  160.               $wh = ' w=' . $width . ' h=' . $height; 
  161.  
  162.           $shortcode = '[vimeo ' . $id . $wh . ']'; 
  163.           $content = str_replace( $match[0], $shortcode, $content ); 
  164.       } 
  165.   } 
  166.  
  167.   return $content; 
  168. }
