  316.          */ 
  317.         $days = (int) apply_filters( 'jetpack_top_posts_days', 2 ); 
  318.  
  319.         if ( $days < 1 ) {

  323.         if ( $days > 10 ) {

  325.         } 
  326.  
  327.         $post_view_posts = stats_get_csv( 'postviews', array( 'days' => absint( $days ), 'limit' => 11 ) );

  330.         } 
  331.  
  332.         $post_view_ids = array_filter( wp_list_pluck( $post_view_posts, 'post_id' ) ); 
  333.         if ( ! $post_view_ids ) {

  336.
