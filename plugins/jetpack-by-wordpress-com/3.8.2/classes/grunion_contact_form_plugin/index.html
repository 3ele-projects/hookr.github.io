  503.         check_admin_referer( 'feedback_export', 'feedback_export_nonce' ); 
  504.  
  505.         if ( ! current_user_can( 'export' ) ) { 
  506.             return; 
  507.         } 
  508.  
  509.         $args = array( 
  510.             'posts_per_page' => -1,  
  511.             'post_type' => 'feedback',  
  512.             'post_status' => 'publish',  
  513.             'order' => 'ASC',  
  514.             'fields' => 'ids',  
  515.             'suppress_filters' => false,  
  516.  ); 
  517.  
  518.         $filename = date( "Y-m-d" ) . '-feedback-export.csv'; 
  519.  
  520.         // Check if we want to download all the feedbacks or just a certain contact form 
  521.         if ( ! empty( $_POST['post'] ) && $_POST['post'] !== 'all' ) { 
  522.             $args['post_parent'] = (int) $_POST['post']; 
  523.             $filename = date( "Y-m-d" ) . '-' . str_replace( ' ', '-', get_the_title( (int) $_POST['post'] ) ) . '.csv'; 
  524.         } 
  525.  
  526.         $feedbacks = get_posts( $args ); 
  527.         $filename = sanitize_file_name( $filename ); 
  528.         $fields = $this->get_field_names( $feedbacks ); 
  529.  
  530.         array_unshift( $fields, __( 'Contact Form', 'jetpack' ) ); 
  531.  
  532.         if ( empty( $feedbacks ) ) 
  533.             return; 
  534.  
  535.         // Forces the download of the CSV instead of echoing 
  536.         header( 'Content-Disposition: attachment; filename=' . $filename ); 
  537.         header( 'Pragma: no-cache' ); 
  538.         header( 'Expires: 0' ); 
  539.         header( 'Content-Type: text/csv; charset=utf-8' ); 
  540.  
  541.         $output = fopen( 'php://output', 'w' ); 
  542.  
  543.         // Prints the header 
  544.         fputcsv( $output, $fields ); 
  545.  
  546.         // Create the csv string from the array of post ids 
  547.         foreach ( $feedbacks as $feedback ) { 
  548.             fputcsv( $output, self::make_csv_row_from_feedback( $feedback, $fields ) ); 
  549.         } 
  550.  
  551.         fclose( $output ); 
  552.     }

  559.     protected function get_feedbacks_as_options() { 
  560.         $options = ''; 
  561.  
  562.         // Get the feedbacks' parents' post IDs 
  563.         $feedbacks = get_posts( array(

  588.     /** 
  589.      * Get the names of all the form's fields 
  590.      * 
  591.      * @param  array|int $posts the post we want the fields of 
  592.      * @return array     the array of fields 
  593.      */ 
  594.     protected function get_field_names( $posts ) { 
  595.         $posts = (array) $posts; 
  596.         $all_fields = array(); 
  597.  
  598.         foreach ( $posts as $post ) { 
  599.             $fields = self::parse_fields_from_content( $post ); 
  600.  
  601.             if ( isset( $fields['_feedback_all_fields'] ) ) { 
  602.                 $extra_fields = array_keys( $fields['_feedback_all_fields'] ); 
  603.                 $all_fields = array_merge( $all_fields, $extra_fields ); 
  604.             } 
  605.         } 
  606.  
  607.         $all_fields = array_unique( $all_fields ); 
  608.         return $all_fields; 
  609.     }

  611.     public static function parse_fields_from_content( $post_id ) { 
  612.         static $post_fields; 
  613.  
  614.         if ( !is_array( $post_fields ) ) 
  615.             $post_fields = array(); 
  616.  
  617.         if ( isset( $post_fields[$post_id] ) ) 
  618.             return $post_fields[$post_id]; 
  619.  
  620.         $all_values = array(); 
  621.         $post_content = get_post_field( 'post_content', $post_id ); 
  622.         $content = explode( '<!--more-->', $post_content ); 
  623.         $lines = array(); 
  624.  
  625.         if ( count( $content ) > 1 ) { 
  626.             $content = str_ireplace( array( '<br />', ')</p>' ), '', $content[1] ); 
  627.             $one_line = preg_replace( '/\s+/', ' ', $content ); 
  628.             $one_line = preg_replace( '/.*Array \( (.*)\)/', '$1', $one_line ); 
  629.  
  630.             preg_match_all( '/\[([^\]]+)\] =\>\; ([^\[]+)/', $one_line, $matches ); 
  631.  
  632.             if ( count( $matches ) > 1 ) 
  633.                 $all_values = array_combine( array_map('trim', $matches[1]), array_map('trim', $matches[2]) ); 
  634.  
  635.             $lines = array_filter( explode( "\n", $content ) ); 
  636.         } 
  637.  
  638.         $var_map = array( 
  639.             'AUTHOR' => '_feedback_author',  
  640.             'AUTHOR EMAIL' => '_feedback_author_email',  
  641.             'AUTHOR URL' => '_feedback_author_url',  
  642.             'SUBJECT' => '_feedback_subject',  
  643.             'IP' => '_feedback_ip' 
  644.  ); 
  645.  
  646.         $fields = array(); 
  647.  
  648.         foreach( $lines as $line ) { 
  649.             $vars = explode( ': ', $line, 2 ); 
  650.             if ( !empty( $vars ) ) { 
  651.                 if ( isset( $var_map[$vars[0]] ) ) { 
  652.                     $fields[$var_map[$vars[0]]] = self::strip_tags( trim( $vars[1] ) ); 
  653.                 } 
  654.             } 
  655.         } 
  656.  
  657.         $fields['_feedback_all_fields'] = $all_values; 
  658.  
  659.         $post_fields[$post_id] = $fields; 
  660.  
  661.         return $fields; 
  662.     }

  664.     /** 
  665.      * Creates a valid csv row from a post id 
  666.      * 
  667.      * @param  int    $post_id The id of the post 
  668.      * @param  array  $fields  An array containing the names of all the fields of the csv 
  669.      * @return String The csv row 
  670.      */ 
  671.     protected static function make_csv_row_from_feedback( $post_id, $fields ) { 
  672.         $content_fields = self::parse_fields_from_content( $post_id ); 
  673.         $all_fields = array(); 
  674.  
  675.         if ( isset( $content_fields['_feedback_all_fields'] ) ) 
  676.             $all_fields = $content_fields['_feedback_all_fields']; 
  677.  
  678.         // Overwrite the parsed content with the content we stored in post_meta in a better format. 
  679.         $extra_fields = get_post_meta( $post_id, '_feedback_extra_fields', true ); 
  680.         foreach ( $extra_fields as $extra_field => $extra_value ) { 
  681.             $all_fields[$extra_field] = $extra_value; 
  682.         } 
  683.  
  684.         // The first element in all of the exports will be the subject 
  685.         $row_items[] = $content_fields['_feedback_subject']; 
  686.  
  687.         // Loop the fields array in order to fill the $row_items array correctly 
  688.         foreach ( $fields as $field ) { 
  689.             if ( $field === __( 'Contact Form', 'jetpack' ) ) // the first field will ever be the contact form, so we can continue 
  690.                 continue; 
  691.             elseif ( array_key_exists( $field, $all_fields ) ) 
  692.                 $row_items[] = $all_fields[$field]; 
  693.             else 
  694.                 $row_items[] = ''; 
  695.         } 
  696.  
  697.         return $row_items; 
  698.     }
