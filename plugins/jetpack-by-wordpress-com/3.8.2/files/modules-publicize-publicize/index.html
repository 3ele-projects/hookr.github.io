  136.   function get_profile_link( $service_name, $c ) { 
  137.       $cmeta = $this->get_connection_meta( $c ); 
  138.  
  139.       if ( isset( $cmeta['connection_data']['meta']['link'] ) ) { 
  140.           if ( 'facebook' == $service_name && 0 === strpos( parse_url( $cmeta['connection_data']['meta']['link'], PHP_URL_PATH ), '/app_scoped_user_id/' ) ) { 
  141.               // App-scoped Facebook user IDs are not usable profile links 
  142.               return false; 
  143.           } 
  144.  
  145.           return $cmeta['connection_data']['meta']['link']; 
  146.       } elseif ( 'facebook' == $service_name && isset( $cmeta['connection_data']['meta']['facebook_page'] ) ) { 
  147.           return 'https://facebook.com/' . $cmeta['connection_data']['meta']['facebook_page']; 
  148.       } elseif ( 'tumblr' == $service_name && isset( $cmeta['connection_data']['meta']['tumblr_base_hostname'] ) ) { 
  149.            return 'http://' . $cmeta['connection_data']['meta']['tumblr_base_hostname']; 
  150.       } elseif ( 'twitter' == $service_name ) { 
  151.           return 'https://twitter.com/' . substr( $cmeta['external_display'], 1 ); // Has a leading '@' 
  152.       } elseif ( 'google_plus' == $service_name && isset( $cmeta['connection_data']['meta']['google_plus_page'] ) ) { 
  153.           return 'https://plus.google.com/' . $cmeta['connection_data']['meta']['google_plus_page']; 
  154.       } elseif ( 'google_plus' == $service_name ) { 
  155.           return 'https://plus.google.com/' . $cmeta['external_id']; 
  156.       } else if ( 'linkedin' == $service_name ) { 
  157.           if ( !isset( $cmeta['connection_data']['meta']['profile_url'] ) ) { 
  158.               return false; 
  159.           } 
  160.  
  161.           $profile_url_query = parse_url( $cmeta['connection_data']['meta']['profile_url'], PHP_URL_QUERY ); 
  162.           wp_parse_str( $profile_url_query, $profile_url_query_args ); 
  163.           if ( isset( $profile_url_query_args['key'] ) ) { 
  164.               $id = $profile_url_query_args['key']; 
  165.           } elseif ( isset( $profile_url_query_args['id'] ) ) { 
  166.               $id = $profile_url_query_args['id']; 
  167.           } else { 
  168.               return false; 
  169.           } 
  170.  
  171.           return esc_url_raw( add_query_arg( 'id', urlencode( $id ), 'http://www.linkedin.com/profile/view' ) ); 
  172.       } else { 
  173.           return false; // no fallback. we just won't link it 
  174.       } 
  175.   }

  262.   * Fires when a post is saved, checks conditions and saves state in postmeta so that it 
  263.   * can be picked up later by @see ::publicize_post() 
  264.   */ 
  265.   function save_meta( $post_id, $post ) { 
  266.       $cron_user = null; 
  267.       $submit_post = true; 
  268.  
  269.       if ( ! $this->post_type_is_publicizeable( $post->post_type ) ) 
  270.           return; 
  271.  
  272.       // Don't Publicize during certain contexts: 
  273.  
  274.       // - import 
  275.       if ( defined( 'WP_IMPORTING' ) && WP_IMPORTING ) { 
  276.           $submit_post = false; 
  277.       } 
  278.  
  279.       // - on quick edit, autosave, etc but do fire on p2, quickpress, and instapost ajax 
  280.       if ( 
  281.           defined( 'DOING_AJAX' ) 
  282.       && 
  283.           DOING_AJAX 
  284.       && 
  285.           !did_action( 'p2_ajax' ) 
  286.       && 
  287.           !did_action( 'wp_ajax_json_quickpress_post' ) 
  288.       && 
  289.           !did_action( 'wp_ajax_instapost_publish' ) 
  290.       && 
  291.           !did_action( 'wp_ajax_post_reblog' ) 
  292.  ) { 
  293.           $submit_post = false; 
  294.       } 
  295.  
  296.       // - bulk edit 
  297.       if ( isset( $_GET['bulk_edit'] ) ) { 
  298.           $submit_post = false; 
  299.       } 
  300.  
  301.       // - API/XML-RPC Test Posts 
  302.       if ( 
  303.           ( 
  304.               defined( 'XMLRPC_REQUEST' ) 
  305.           && 
  306.               XMLRPC_REQUEST 
  307.           || 
  308.               defined( 'APP_REQUEST' ) 
  309.           && 
  310.               APP_REQUEST 
  311.  ) 
  312.       && 
  313.           0 === strpos( $post->post_title, 'Temporary Post Used For Theme Detection' ) 
  314.  ) { 
  315.           $submit_post = false; 
  316.       } 
  317.  
  318.       // only work with certain statuses (avoids inherits, auto drafts etc) 
  319.       if ( !in_array( $post->post_status, array( 'publish', 'draft', 'future' ) ) ) { 
  320.           $submit_post = false; 
  321.       } 
  322.  
  323.       // don't publish password protected posts 
  324.       if ( '' !== $post->post_password ) { 
  325.           $submit_post = false; 
  326.       } 
  327.  
  328.       // Did this request happen via wp-admin? 
  329.       $from_web = 'post' == strtolower( $_SERVER['REQUEST_METHOD'] ) && isset( $_POST[$this->ADMIN_PAGE] ); 
  330.  
  331.       if ( ( $from_web || defined( 'POST_BY_EMAIL' ) ) && isset( $_POST['wpas_title'] ) ) { 
  332.           if ( empty( $_POST['wpas_title'] ) ) { 
  333.               delete_post_meta( $post_id, $this->POST_MESS ); 
  334.           } else { 
  335.               update_post_meta( $post_id, $this->POST_MESS, trim( stripslashes( $_POST['wpas_title'] ) ) ); 
  336.           } 
  337.       } 
  338.  
  339.       // change current user to provide context for get_services() if we're running during cron 
  340.       if ( defined( 'DOING_CRON' ) && DOING_CRON ) { 
  341.           $cron_user = (int) $GLOBALS['user_ID']; 
  342.           wp_set_current_user( $post->post_author ); 
  343.       } 
  344.  
  345.       /** 
  346.        * In this phase, we mark connections that we want to SKIP. When Publicize is actually triggered,  
  347.        * it will Publicize to everything *except* those marked for skipping. 
  348.        */ 
  349.       foreach ( (array) $this->get_services( 'connected' ) as $service_name => $connections ) { 
  350.           foreach ( $connections as $connection ) { 
  351.               $connection_data = ''; 
  352.               if ( method_exists( $connection, 'get_meta' ) ) 
  353.                   $connection_data = $connection->get_meta( 'connection_data' ); 
  354.               elseif ( ! empty( $connection['connection_data'] ) ) 
  355.                   $connection_data = $connection['connection_data']; 
  356.  
  357.               /** This action is documented in modules/publicize/ui.php */ 
  358.               if ( false == apply_filters( 'wpas_submit_post?', $submit_post, $post_id, $service_name, $connection_data ) ) { 
  359.                   delete_post_meta( $post_id, $this->PENDING ); 
  360.                   continue; 
  361.               } 
  362.  
  363.               if ( !empty( $connection->unique_id ) ) 
  364.                   $unique_id = $connection->unique_id; 
  365.               else if ( !empty( $connection['connection_data']['token_id'] ) ) 
  366.                   $unique_id = $connection['connection_data']['token_id']; 
  367.  
  368.               // This was a wp-admin request, so we need to check the state of checkboxes 
  369.               if ( $from_web ) { 
  370.                   // delete stray service-based post meta 
  371.                   delete_post_meta( $post_id, $this->POST_SKIP . $service_name ); 
  372.  
  373.                   // We *unchecked* this stream from the admin page, or it's set to readonly, or it's a new addition 
  374.                   if ( empty( $_POST[$this->ADMIN_PAGE]['submit'][$unique_id] ) ) { 
  375.                       // Also make sure that the service-specific input isn't there. 
  376.                       // If the user connected to a new service 'in-page' then a hidden field with the service 
  377.                       // name is added, so we just assume they wanted to Publicize to that service. 
  378.                       if ( empty( $_POST[$this->ADMIN_PAGE]['submit'][$service_name] ) ) { 
  379.                           // Nothing seems to be checked, so we're going to mark this one to be skipped 
  380.                           update_post_meta( $post_id, $this->POST_SKIP . $unique_id, 1 ); 
  381.                           continue; 
  382.                       } else { 
  383.                           // clean up any stray post meta 
  384.                           delete_post_meta( $post_id, $this->POST_SKIP . $unique_id ); 
  385.                       } 
  386.                   } else { 
  387.                       // The checkbox for this connection is explicitly checked -- make sure we DON'T skip it 
  388.                       delete_post_meta( $post_id, $this->POST_SKIP . $unique_id ); 
  389.                   } 
  390.               } 
  391.  
  392.               /** 
  393.                * Fires right before the post is processed for Publicize. 
  394.                * Users may hook in here and do anything else they need to after meta is written,  
  395.                * and before the post is processed for Publicize. 
  396.                * 
  397.                * @since 2.1.2 
  398.                * 
  399.                * @param bool $submit_post Should the post be publicized. 
  400.                * @param int $post->ID Post ID. 
  401.                * @param string $service_name Service name. 
  402.                * @param array $connection Array of connection details. 
  403.                */ 
  404.               do_action( 'publicize_save_meta', $submit_post, $post_id, $service_name, $connection ); 
  405.           } 
  406.       } 
  407.  
  408.       if ( defined( 'DOING_CRON' ) && DOING_CRON ) { 
  409.           wp_set_current_user( $cron_user ); 
  410.       } 
  411.  
  412.       // Next up will be ::publicize_post() 
  413.   }
