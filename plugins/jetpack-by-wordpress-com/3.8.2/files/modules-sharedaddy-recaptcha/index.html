  118.               $error_message = $this->error_codes[ $resp_decoded['error-codes'][0] ]; 
  119.               $error_code = $resp_decoded['error-codes'][0]; 
  120.           } 
  121.       } 
  122.  
  123.       if ( ! isset( $resp_decoded['success'] ) ) { 
  124.           return new WP_Error( $error_code, $error_message ); 
  125.       } 
  126.  
  127.       if ( true !== $resp_decoded['success'] ) { 
  128.           return new WP_Error( $error_code, $error_message ); 
  129.       } 
  130.  
  131.       return true; 
  132.   }
