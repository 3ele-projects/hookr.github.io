  90. function minileven_fonts() {

  92.   /**    translators: If there are characters in your language that are not supported 
  93.       by Open Sans, translate this to 'off'. Do not translate into your own language. */ 
  94.  
  95.   if ( 'off' !== _x( 'on', 'Open Sans font: on or off', 'jetpack' ) ) {

  98.  
  99.       /** translators: To add an additional Open Sans character subset specific to your language, translate

  102.  
  103.       if ( 'cyrillic' == $opensans_subset ) 
  104.           $opensans_subsets .= ', cyrillic, cyrillic-ext'; 
  105.       elseif ( 'greek' == $opensans_subset ) 
  106.           $opensans_subsets .= ', greek, greek-ext'; 
  107.       elseif ( 'vietnamese' == $opensans_subset ) 
  108.           $opensans_subsets .= ', vietnamese'; 
  109.  
  110.       $opensans_query_args = array( 
  111.           'family' => 'Open+Sans:200, 200italic, 300, 300italic, 400, 400italic, 600, 600italic, 700, 700italic',  
  112.           'subset' => $opensans_subsets,  
  113.  ); 
  114.       wp_register_style( 'minileven-open-sans', add_query_arg( $opensans_query_args, "//fonts.googleapis.com/css" ), array(), null ); 
  115.   }

  201. /** 
  202.  * Retrieves the IDs for images in a gallery. 
  203.  * 
  204.  * @uses get_post_galleries() first, if available. Falls back to shortcode parsing,  
  205.  * then as last option uses a get_posts() call. 
  206.  * 
  207.  * @return array List of image IDs from the post gallery. 
  208.  */ 
  209. function minileven_get_gallery_images() { 
  210.   $images = array(); 
  211.  
  212.   if ( function_exists( 'get_post_galleries' ) ) { 
  213.       $galleries = get_post_galleries( get_the_ID(), false ); 
  214.       if ( isset( $galleries[0]['ids'] ) ) 
  215.            $images = explode( ', ', $galleries[0]['ids'] ); 
  216.   } else { 
  217.       $pattern = get_shortcode_regex(); 
  218.       preg_match( "/$pattern/s", get_the_content(), $match ); 
  219.       $atts = shortcode_parse_atts( $match[3] ); 
  220.       if ( isset( $atts['ids'] ) ) 
  221.           $images = explode( ', ', $atts['ids'] ); 
  222.   } 
  223.  
  224.   if ( ! $images ) { 
  225.       $images = get_posts( array( 
  226.           'fields' => 'ids',  
  227.           'numberposts' => 999,  
  228.           'order' => 'ASC',  
  229.           'orderby' => 'menu_order',  
  230.           'post_mime_type' => 'image',  
  231.           'post_parent' => get_the_ID(),  
  232.           'post_type' => 'attachment',  
  233.  ) ); 
  234.   } 
  235.  
  236.   return $images; 
  237. }
