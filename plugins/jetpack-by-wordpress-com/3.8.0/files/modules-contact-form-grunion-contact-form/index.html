  865.               $value = "'$value'"; 
  866.           } elseif ( false === strpos( $value, '"' ) ) { 
  867.               $value = '"' . $value . '"'; 
  868.           } else { 
  869.               // Shortcodes can't contain both '"' and "'".  Strip one. 
  870.               $value = str_replace( "'", '', $value ); 
  871.               $value = "'$value'"; 
  872.           } 
  873.  
  874.           $r .= "{$key}={$value} "; 
  875.       } 
  876.

  878.  
  879.       if ( $this->fields ) { 
  880.           $r .= ']'; 
  881.  
  882.           foreach ( $this->fields as $field ) { 
  883.               $r .= (string) $field; 
  884.           } 
  885.  
  886.           $r .= "[/{$this->shortcode_name}]"; 
  887.       } else { 
  888.           $r .= '/]'; 
  889.       } 
  890.
