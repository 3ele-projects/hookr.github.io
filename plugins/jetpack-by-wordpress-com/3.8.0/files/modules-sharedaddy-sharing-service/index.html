  559.       wp_enqueue_style( 'genericons' ); 
  560.   } 
  561.  
  562. }

  570.       $sharer = new Sharing_Service();

  573.       if ( $service ) { 
  574.           $service->process_request( $post, $_POST ); 
  575.       }

  580. function sharing_display( $text = '', $echo = false ) { 
  581.   global $post, $wp_current_filter; 
  582.  
  583.   if ( empty( $post ) ) 
  584.       return $text; 
  585.  
  586.   if ( ( is_preview() || is_admin() ) && ! ( defined( 'DOING_AJAX' ) && DOING_AJAX ) ) { 
  587.       return $text; 
  588.   } 
  589.  
  590.   // Don't output flair on excerpts 
  591.   if ( in_array( 'get_the_excerpt', (array) $wp_current_filter ) ) { 
  592.       return $text; 
  593.   } 
  594.  
  595.   // Don't allow flair to be added to the_content more than once (prevent infinite loops) 
  596.   $done = false; 
  597.   foreach ( $wp_current_filter as $filter ) { 
  598.       if ( 'the_content' == $filter ) { 
  599.           if ( $done ) 
  600.               return $text; 
  601.           else 
  602.               $done = true; 
  603.       } 
  604.   } 
  605.  
  606.   // check whether we are viewing the front page and whether the front page option is checked 
  607.   $options = get_option( 'sharing-options' ); 
  608.   $display_options = $options['global']['show']; 
  609.  
  610.   if ( is_front_page() && ( is_array( $display_options ) && ! in_array( 'index', $display_options ) ) ) 
  611.       return $text; 
  612.  
  613.   if ( is_attachment() && in_array( 'the_excerpt', (array) $wp_current_filter ) ) { 
  614.       // Many themes run the_excerpt() conditionally on an attachment page, then run the_content(). 
  615.       // We only want to output the sharing buttons once.  Let's stick with the_content(). 
  616.       return $text; 
  617.   } 
  618.  
  619.   $sharer = new Sharing_Service(); 
  620.   $global = $sharer->get_global_options(); 
  621.  
  622.   $show = false; 
  623.   if ( !is_feed() ) { 
  624.       if ( is_singular() && in_array( get_post_type(), $global['show'] ) ) { 
  625.           $show = true; 
  626.       } elseif ( in_array( 'index', $global['show'] ) && ( is_home() || is_front_page() || is_archive() || is_search() || in_array( get_post_type(), $global['show'] ) ) ) { 
  627.           $show = true; 
  628.       } 
  629.   } 
  630.  
  631.   /** 
  632.    * Filter to decide if sharing buttons should be displayed. 
  633.    * 
  634.    * @module sharedaddy 
  635.    * 
  636.    * @since 1.1.0 
  637.    * 
  638.    * @param bool $show Should the sharing buttons be displayed. 
  639.    * @param WP_Post $post The post to share. 
  640.    */ 
  641.   $show = apply_filters( 'sharing_show', $show, $post ); 
  642.  
  643.   // Disabled for this post? 
  644.   $switched_status = get_post_meta( $post->ID, 'sharing_disabled', false ); 
  645.  
  646.   if ( !empty( $switched_status ) ) 
  647.       $show = false; 
  648.  
  649.   // Private post? 
  650.   $post_status = get_post_status( $post->ID ); 
  651.  
  652.   if ( 'private' === $post_status ) { 
  653.       $show = false; 
  654.   } 
  655.  
  656.   // Allow to be used on P2 ajax requests for latest posts. 
  657.   if ( defined( 'DOING_AJAX' ) && DOING_AJAX && isset( $_REQUEST['action'] ) && 'get_latest_posts' == $_REQUEST['action'] ) 
  658.       $show = true; 
  659.  
  660.   $sharing_content = ''; 
  661.  
  662.   if ( $show ) { 
  663.       /** 
  664.        * Filters the list of enabled Sharing Services. 
  665.        * 
  666.        * @module sharedaddy 
  667.        * 
  668.        * @since 2.2.3 
  669.        * 
  670.        * @param array $sharer->get_blog_services() Array of Sharing Services currently enabled. 
  671.        */ 
  672.       $enabled = apply_filters( 'sharing_enabled', $sharer->get_blog_services() ); 
  673.  
  674.       if ( count( $enabled['all'] ) > 0 ) { 
  675.           global $post; 
  676.  
  677.           $dir = get_option( 'text_direction' ); 
  678.  
  679.           // Wrapper 
  680.           $sharing_content .= '<div class="sharedaddy sd-sharing-enabled"><div class="robots-nocontent sd-block sd-social sd-social-' . $global['button_style'] . ' sd-sharing">'; 
  681.           if ( $global['sharing_label'] != '' ) 
  682.               $sharing_content .= '<h3 class="sd-title">' . $global['sharing_label'] . '</h3>'; 
  683.           $sharing_content .= '<div class="sd-content"><ul>'; 
  684.  
  685.           // Visible items 
  686.           $visible = ''; 
  687.           foreach ( $enabled['visible'] as $id => $service ) { 
  688.               // Individual HTML for sharing service 
  689.               $visible .= '<li class="share-' . $service->get_class() . '">' . $service->get_display( $post ) . '</li>'; 
  690.           } 
  691.  
  692.           $parts = array(); 
  693.           $parts[] = $visible; 
  694.           if ( count( $enabled['hidden'] ) > 0 ) { 
  695.               if ( count( $enabled['visible'] ) > 0 ) 
  696.                   $expand = __( 'More', 'jetpack' ); 
  697.               else 
  698.                   $expand = __( 'Share', 'jetpack' ); 
  699.               $parts[] = '<li><a href="#" class="sharing-anchor sd-button share-more"><span>'.$expand.'</span></a></li>'; 
  700.           } 
  701.  
  702.           if ( $dir == 'rtl' ) 
  703.               $parts = array_reverse( $parts ); 
  704.  
  705.           $sharing_content .= implode( '', $parts ); 
  706.           $sharing_content .= '<li class="share-end"></li></ul>'; 
  707.  
  708.           if ( count( $enabled['hidden'] ) > 0 ) { 
  709.               $sharing_content .= '<div class="sharing-hidden"><div class="inner" style="display: none;'; 
  710.  
  711.               if ( count( $enabled['hidden'] ) == 1 ) 
  712.                   $sharing_content .= 'width:150px;'; 
  713.  
  714.               $sharing_content .= '">'; 
  715.  
  716.               if ( count( $enabled['hidden'] ) == 1 ) 
  717.                   $sharing_content .= '<ul style="background-image:none;">'; 
  718.               else 
  719.                   $sharing_content .= '<ul>'; 
  720.  
  721.               $count = 1; 
  722.               foreach ( $enabled['hidden'] as $id => $service ) { 
  723.                   // Individual HTML for sharing service 
  724.                   $sharing_content .= '<li class="share-'.$service->get_class().'">'; 
  725.                   $sharing_content .= $service->get_display( $post ); 
  726.                   $sharing_content .= '</li>'; 
  727.  
  728.                   if ( ( $count % 2 ) == 0 ) 
  729.                       $sharing_content .= '<li class="share-end"></li>'; 
  730.  
  731.                   $count ++; 
  732.               } 
  733.  
  734.               // End of wrapper 
  735.               $sharing_content .= '<li class="share-end"></li></ul></div></div>'; 
  736.           } 
  737.  
  738.           $sharing_content .= '</div></div></div>'; 
  739.  
  740.           // Register our JS 
  741.           if ( defined( 'JETPACK__VERSION' ) ) { 
  742.               $ver = JETPACK__VERSION; 
  743.           } else { 
  744.               $ver = '20141212'; 
  745.           } 
  746.           wp_register_script( 'sharing-js', plugin_dir_url( __FILE__ ).'sharing.js', array( 'jquery' ), $ver ); 
  747.           add_action( 'wp_footer', 'sharing_add_footer' ); 
  748.       } 
  749.   } 
  750.  
  751.   /** 
  752.    * Filters the content markup of the Jetpack sharing links 
  753.    * 
  754.    * @module sharedaddy 
  755.    * 
  756.    * @since 3.8.0 
  757.    * 
  758.    * @param string $sharing_content Content markup of the Jetpack sharing links 
  759.    */ 
  760.   $sharing_markup = apply_filters( 'jetpack_sharing_display_markup', $sharing_content ); 
  761.  
  762.   if ( $echo ) 
  763.       echo $text . $sharing_markup; 
  764.   else 
  765.       return $text . $sharing_markup; 
  766. }
