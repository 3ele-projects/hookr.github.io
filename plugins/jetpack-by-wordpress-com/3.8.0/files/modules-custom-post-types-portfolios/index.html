  391.   /** 
  392.    * Our [portfolio] shortcode. 
  393.    * Prints Portfolio data styled to look good on *any* theme. 
  394.    * 
  395.    * @return portfolio_shortcode_html 
  396.    */ 
  397.   static function portfolio_shortcode( $atts ) { 
  398.       // Default attributes 
  399.       $atts = shortcode_atts( array( 
  400.           'display_types' => true,  
  401.           'display_tags' => true,  
  402.           'display_content' => true,  
  403.           'show_filter' => false,  
  404.           'include_type' => false,  
  405.           'include_tag' => false,  
  406.           'columns' => 2,  
  407.           'showposts' => -1,  
  408.           'order' => 'asc',  
  409.           'orderby' => 'date',  
  410.  ), $atts, 'portfolio' ); 
  411.  
  412.       // A little sanitization 
  413.       if ( $atts['display_types'] && 'true' != $atts['display_types'] ) { 
  414.           $atts['display_types'] = false; 
  415.       } 
  416.  
  417.       if ( $atts['display_tags'] && 'true' != $atts['display_tags'] ) { 
  418.           $atts['display_tags'] = false; 
  419.       } 
  420.  
  421.       if ( $atts['display_content'] && 'true' != $atts['display_content'] && 'full' != $atts['display_content'] ) { 
  422.           $atts['display_content'] = false; 
  423.       } 
  424.  
  425.       if ( $atts['include_type'] ) { 
  426.           $atts['include_type'] = explode( ', ', str_replace( ' ', '', $atts['include_type'] ) ); 
  427.       } 
  428.  
  429.       if ( $atts['include_tag'] ) { 
  430.           $atts['include_tag'] = explode( ', ', str_replace( ' ', '', $atts['include_tag'] ) ); 
  431.       } 
  432.  
  433.       $atts['columns'] = absint( $atts['columns'] ); 
  434.  
  435.       $atts['showposts'] = intval( $atts['showposts'] ); 
  436.  
  437.  
  438.       if ( $atts['order'] ) { 
  439.           $atts['order'] = urldecode( $atts['order'] ); 
  440.           $atts['order'] = strtoupper( $atts['order'] ); 
  441.           if ( 'DESC' != $atts['order'] ) { 
  442.               $atts['order'] = 'ASC'; 
  443.           } 
  444.       } 
  445.  
  446.       if ( $atts['orderby'] ) { 
  447.           $atts['orderby'] = urldecode( $atts['orderby'] ); 
  448.           $atts['orderby'] = strtolower( $atts['orderby'] ); 
  449.           $allowed_keys = array( 'author', 'date', 'title', 'rand' ); 
  450.  
  451.           $parsed = array(); 
  452.           foreach ( explode( ', ', $atts['orderby'] ) as $portfolio_index_number => $orderby ) { 
  453.               if ( ! in_array( $orderby, $allowed_keys ) ) { 
  454.                   continue; 
  455.               } 
  456.               $parsed[] = $orderby; 
  457.           } 
  458.  
  459.           if ( empty( $parsed ) ) { 
  460.               unset( $atts['orderby'] ); 
  461.           } else { 
  462.               $atts['orderby'] = implode( ' ', $parsed ); 
  463.           } 
  464.       } 
  465.  
  466.       // enqueue shortcode styles when shortcode is used 
  467.       wp_enqueue_style( 'jetpack-portfolio-style', plugins_url( 'css/portfolio-shortcode.css', __FILE__ ), array(), '20140326' ); 
  468.  
  469.       return self::portfolio_shortcode_html( $atts ); 
  470.   }

  472.   /** 
  473.    * Query to retrieve entries from the Portfolio post_type. 
  474.    * 
  475.    * @return object 
  476.    */ 
  477.   static function portfolio_query( $atts ) { 
  478.       // Default query arguments 
  479.       $default = array( 
  480.           'order' => $atts['order'],  
  481.           'orderby' => $atts['orderby'],  
  482.           'posts_per_page' => $atts['showposts'],  
  483.  ); 
  484.  
  485.       $args = wp_parse_args( $atts, $default ); 
  486.       $args['post_type'] = self::CUSTOM_POST_TYPE; // Force this post type 
  487.  
  488.       if ( false != $atts['include_type'] || false != $atts['include_tag'] ) { 
  489.           $args['tax_query'] = array(); 
  490.       } 
  491.  
  492.       // If 'include_type' has been set use it on the main query 
  493.       if ( false != $atts['include_type'] ) { 
  494.           array_push( $args['tax_query'], array( 
  495.               'taxonomy' => self::CUSTOM_TAXONOMY_TYPE,  
  496.               'field' => 'slug',  
  497.               'terms' => $atts['include_type'],  
  498.  ) ); 
  499.       } 
  500.  
  501.       // If 'include_tag' has been set use it on the main query 
  502.       if ( false != $atts['include_tag'] ) { 
  503.           array_push( $args['tax_query'], array( 
  504.               'taxonomy' => self::CUSTOM_TAXONOMY_TAG,  
  505.               'field' => 'slug',  
  506.               'terms' => $atts['include_tag'],  
  507.  ) ); 
  508.       } 
  509.  
  510.       if ( false != $atts['include_type'] && false != $atts['include_tag'] ) { 
  511.           $args['tax_query']['relation'] = 'AND'; 
  512.       } 
  513.  
  514.       // Run the query and return 
  515.       $query = new WP_Query( $args ); 
  516.       return $query; 
  517.   }

  522.    * @todo add theme color styles 
  523.    * @return html 
  524.    */ 
  525.   static function portfolio_shortcode_html( $atts ) { 
  526.  
  527.       $query = self::portfolio_query( $atts ); 
  528.       $portfolio_index_number = 0; 
  529.  
  530.       ob_start(); 
  531.  
  532.       // If we have posts, create the html 
  533.       // with hportfolio markup 
  534.       if ( $query->have_posts() ) { 
  535.  
  536.           // Render styles 
  537.           //self::themecolor_styles();

  541.           <?php  // open .jetpack-portfolio 
  542.  
  543.           // Construct the loop... 
  544.           while ( $query->have_posts() ) { 
  545.               $query->the_post();

  556.  
  557.                       <div class="portfolio-entry-meta"> 
  558.                       <?php 
  559.                       if ( false != $atts['display_types'] ) { 
  560.                           echo self::get_project_type( $post_id ); 
  561.                       } 
  562.  
  563.                       if ( false != $atts['display_tags'] ) { 
  564.                           echo self::get_project_tags( $post_id ); 
  565.                       } 
  566.                       ?> 
  567.                       </div> 
  568.  
  569.                   </header> 
  570.  
  571.               <?php 
  572.               // The content 
  573.               if ( false !== $atts['display_content'] ) { 
  574.                   if ( 'full' === $atts['display_content'] ) { 
  575.                       echo '<div class="portfolio-entry-content">' . the_content() . '</div>'; 
  576.                   } else { 
  577.                       echo '<div class="portfolio-entry-content">' . the_excerpt() . '</div>'; 
  578.                   } 
  579.               } ?> 
  580.               </div><!-- close .portfolio-entry --> 
  581.               <?php $portfolio_index_number++; 
  582.           } // end of while loop 
  583.  
  584.           wp_reset_postdata(); 
  585.           ?> 
  586.           </div><!-- close .jetpack-portfolio --> 
  587.       <?php 
  588.       } else { ?> 
  589.           <p><em><?php _e( 'Your Portfolio Archive currently has no entries. You can start creating them on your dashboard.', 'jetpack' ); ?></p></em> 
  590.       <?php 
  591.       } 
  592.       $html = ob_get_clean(); 
  593.  
  594.       // If there is a [portfolio] within a [portfolio], remove the shortcode 
  595.       if ( has_shortcode( $html, 'portfolio' ) ) { 
  596.           remove_shortcode( 'portfolio' ); 
  597.       } 
  598.  
  599.       // Return the HTML block 
  600.       return $html; 
  601.   }

  604.    * Individual project class 
  605.    * 
  606.    * @return string 
  607.    */ 
  608.   static function get_project_class( $portfolio_index_number, $columns ) { 
  609.       $project_types = wp_get_object_terms( get_the_ID(), self::CUSTOM_TAXONOMY_TYPE, array( 'fields' => 'slugs' ) ); 
  610.       $class = array(); 
  611.  
  612.       $class[] = 'portfolio-entry-column-'.$columns; 
  613.       // add a type- class for each project type 
  614.       foreach ( $project_types as $project_type ) { 
  615.           $class[] = 'type-' . esc_html( $project_type ); 
  616.       } 
  617.       if( $columns > 1) { 
  618.           if ( ( $portfolio_index_number % 2 ) == 0 ) { 
  619.               $class[] = 'portfolio-entry-mobile-first-item-row'; 
  620.           } else { 
  621.               $class[] = 'portfolio-entry-mobile-last-item-row'; 
  622.           } 
  623.       } 
  624.  
  625.       // add first and last classes to first and last items in a row 
  626.       if ( ( $portfolio_index_number % $columns ) == 0 ) { 
  627.           $class[] = 'portfolio-entry-first-item-row'; 
  628.       } elseif ( ( $portfolio_index_number % $columns ) == ( $columns - 1 ) ) { 
  629.           $class[] = 'portfolio-entry-last-item-row'; 
  630.       } 
  631.  
  632.  
  633.       /** 
  634.        * Filter the class applied to project div in the portfolio 
  635.        * 
  636.        * @module custom-content-types 
  637.        * 
  638.        * @since 3.1.0 
  639.        * 
  640.        * @param string $class class name of the div. 
  641.        * @param int $portfolio_index_number iterator count the number of columns up starting from 0. 
  642.        * @param int $columns number of columns to display the content in. 
  643.        * 
  644.        */ 
  645.       return apply_filters( 'portfolio-project-post-class', implode( " ", $class ) , $portfolio_index_number, $columns ); 
  646.   }

  655.

  665.           $project_type_link = get_term_link( $project_type, self::CUSTOM_TAXONOMY_TYPE ); 
  666.  
  667.           if ( is_wp_error( $project_type_link ) ) { 
  668.               return $project_type_link; 
  669.           } 
  670.  
  671.           $types[] = '<a href="' . esc_url( $project_type_link ) . '" rel="tag">' . esc_html( $project_type->name ) . '</a>'; 
  672.       } 
  673.       $html .= ' '.implode( ', ', $types ); 
  674.       $html .= '</div>'; 
  675.
