  32.   if ( false !== strpos( $url, 'video.php' ) || false !== strpos( $url, '/videos/' ) ) {

  34.   } else {

  36.   }

  38.   // since Facebook is a faux embed, we need to load the JS SDK in the wpview embed iframe

  41.   } else { 
  42.       wp_enqueue_script( 'jetpack-facebook-embed', plugins_url( 'js/facebook.js', __FILE__ ), array( 'jquery' ), null, true );

  46.           $fb_app_id = ''; 
  47.       }

  49.           'jetpack-facebook-embed',
