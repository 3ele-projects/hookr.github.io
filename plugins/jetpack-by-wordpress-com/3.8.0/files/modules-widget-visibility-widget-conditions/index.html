  461.                           break; 
  462.                           case 'month': 
  463.                               $condition_result = is_month(); 
  464.                           break; 
  465.                           case 'day': 
  466.                               $condition_result = is_day(); 
  467.                           break; 
  468.                           case 'year': 
  469.                               $condition_result = is_year(); 
  470.                           break; 
  471.                       } 
  472.                   break; 
  473.                   case 'page': 
  474.                       // Previously hardcoded post type options. 
  475.                       if ( 'post' == $rule['minor'] ) 
  476.                           $rule['minor'] = 'post_type-post'; 
  477.                       else if ( ! $rule['minor'] ) 
  478.                           $rule['minor'] = 'post_type-page'; 
  479.  
  480.                       switch ( $rule['minor'] ) { 
  481.                           case '404': 
  482.                               $condition_result = is_404(); 
  483.                           break; 
  484.                           case 'search': 
  485.                               $condition_result = is_search(); 
  486.                           break; 
  487.                           case 'archive': 
  488.                               $condition_result = is_archive(); 
  489.                           break; 
  490.                           case 'posts': 
  491.                               $condition_result = $wp_query->is_posts_page; 
  492.                           break; 
  493.                           case 'home': 
  494.                               $condition_result = is_home(); 
  495.                           break; 
  496.                           case 'front': 
  497.                               if ( current_theme_supports( 'infinite-scroll' ) ) 
  498.                                   $condition_result = is_front_page(); 
  499.                               else { 
  500.                                   $condition_result = is_front_page() && !is_paged(); 
  501.                               } 
  502.                           break; 
  503.                           default: 
  504.                               if ( substr( $rule['minor'], 0, 10 ) == 'post_type-' ) { 
  505.                                   $condition_result = is_singular( substr( $rule['minor'], 10 ) ); 
  506.                               } elseif ( $rule['minor'] == get_option( 'page_for_posts' ) ) { 
  507.                                   // If $rule['minor'] is a page ID which is also the posts page 
  508.                                   $condition_result = $wp_query->is_posts_page; 
  509.                               } else { 
  510.                                   // $rule['minor'] is a page ID 
  511.                                   $condition_result = is_page( $rule['minor'] ); 
  512.                                   // Check if $rule['minor'] is parent of page ID 
  513.                                   if ( ! $condition_result && isset( $rule['has_children'] ) && $rule['has_children'] ) 
  514.                                       $condition_result = wp_get_post_parent_id( get_the_ID() ) == $rule['minor']; 
  515.                               } 
  516.                           break; 
  517.                       } 
  518.                   break; 
  519.                   case 'tag': 
  520.                       if ( ! $rule['minor'] && is_tag() ) { 
  521.                           $condition_result = true; 
  522.                       } else { 
  523.                           $rule['minor'] = self::maybe_get_split_term( $rule['minor'], $rule['major'] ); 
  524.                           if ( is_singular() && $rule['minor'] && has_tag( $rule['minor'] ) ) { 
  525.                               $condition_result = true; 
  526.                           } else { 
  527.                               $tag = get_tag( $rule['minor'] ); 
  528.                               if ( $tag && is_tag( $tag->slug ) ) { 
  529.                                   $condition_result = true; 
  530.                               } 
  531.                           } 
  532.                       } 
  533.                   break; 
  534.                   case 'category': 
  535.                       if ( ! $rule['minor'] && is_category() ) { 
  536.                           $condition_result = true; 
  537.                       } else { 
  538.                           $rule['minor'] = self::maybe_get_split_term( $rule['minor'], $rule['major'] ); 
  539.                           if ( is_category( $rule['minor'] ) ) { 
  540.                               $condition_result = true; 
  541.                           } else if ( is_singular() && $rule['minor'] && in_array( 'category', get_post_taxonomies() ) &&  has_category( $rule['minor'] ) ) 
  542.                               $condition_result = true; 
  543.                       } 
  544.                   break; 
  545.                   case 'loggedin': 
  546.                       $condition_result = is_user_logged_in(); 
  547.                       if ( 'loggedin' !== $rule['minor'] ) { 
  548.                           $condition_result = ! $condition_result; 
  549.                       } 
  550.                   break; 
  551.                   case 'author': 
  552.                       $post = get_post(); 
  553.                       if ( ! $rule['minor'] && is_author() ) 
  554.                           $condition_result = true; 
  555.                       else if ( $rule['minor'] && is_author( $rule['minor'] ) ) 
  556.                           $condition_result = true; 
  557.                       else if ( is_singular() && $rule['minor'] && $rule['minor'] == $post->post_author ) 
  558.                           $condition_result = true; 
  559.                   break; 
  560.                   case 'role': 
  561.                       if( is_user_logged_in() ) { 
  562.                           global $current_user; 
  563.                           get_currentuserinfo(); 
  564.  
  565.                           $user_roles = $current_user->roles; 
  566.  
  567.                           if( in_array( $rule['minor'], $user_roles ) ) { 
  568.                               $condition_result = true; 
  569.                           } else { 
  570.                               $condition_result = false; 
  571.                           } 
  572.  
  573.                       } else { 
  574.                           $condition_result = false; 
  575.                       } 
  576.                   break; 
  577.                   case 'taxonomy': 
  578.                       $term = explode( '_tax_', $rule['minor'] ); // $term[0] = taxonomy name; $term[1] = term id 
  579.                       if ( isset( $term[0] ) && isset( $term[1] ) ) { 
  580.                           $term[1] = self::maybe_get_split_term( $term[1], $term[0] ); 
  581.                       } 
  582.                       if ( isset( $term[1] ) && is_tax( $term[0], $term[1] ) ) 
  583.                           $condition_result = true; 
  584.                       else if ( isset( $term[1] ) && is_singular() && $term[1] && has_term( $term[1], $term[0] ) ) 
  585.                           $condition_result = true; 
  586.                       else if ( is_singular() && $post_id = get_the_ID() ) { 
  587.                           $terms = get_the_terms( $post_id, $rule['minor'] ); // Does post have terms in taxonomy? 
  588.                           if( $terms && ! is_wp_error( $terms ) ) { 
  589.                               $condition_result = true; 
  590.                           } 
  591.                       } 
  592.                   break; 
  593.               } 
  594.  
  595.               if ( $condition_result || self::$passed_template_redirect ) { 
  596.                   // Some of the conditions will return false when checked before the template_redirect 
  597.                   // action has been called, like is_page(). Only store positive lookup results, which 
  598.                   // won't be false positives, before template_redirect, and everything after. 
  599.                   $condition_result_cache[ $condition_key ] = $condition_result; 
  600.               } 
  601.           } 
  602.  
  603.           if ( $condition_result ) 
  604.               break; 
  605.       } 
  606.  
  607.       if ( ( 'show' == $instance['conditions']['action'] && ! $condition_result ) || ( 'hide' == $instance['conditions']['action'] && $condition_result ) ) 
  608.           return false; 
  609.  
  610.       return $instance; 
  611.   }

  613.   public static function strcasecmp_name( $a, $b ) { 
  614.       return strcasecmp( $a->name, $b->name ); 
  615.   }

  617.   public static function maybe_get_split_term( $old_term_id = '', $taxonomy = '' ) { 
  618.       $term_id = $old_term_id; 
  619.  
  620.       if ( 'tag' == $taxonomy ) { 
  621.           $taxonomy = 'post_tag'; 
  622.       } 
  623.  
  624.       if ( function_exists( 'wp_get_split_term' ) && $new_term_id = wp_get_split_term( $old_term_id, $taxonomy ) ) { 
  625.           $term_id = $new_term_id; 
  626.       } 
  627.  
  628.       return $term_id; 
  629.   }
