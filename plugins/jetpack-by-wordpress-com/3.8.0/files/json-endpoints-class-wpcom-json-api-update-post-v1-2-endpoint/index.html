  467.               if ( !in_array( $name, $publicize ) && !array_key_exists( $name, $publicize ) ) { 
  468.                   // Skip the whole service by adding each connection ID 
  469.                   foreach ( $service_connections as $service_connection ) { 
  470.                       update_post_meta( $post_id, $GLOBALS['publicize_ui']->publicize->POST_SKIP . $service_connection->unique_id, 1 ); 
  471.                   } 
  472.               } else if ( !empty( $publicize[ $name ] ) ) { 
  473.                   // Seems we're being asked to only push to [a] specific connection[s]. 
  474.                   // Explode the list on commas, which will also support a single passed ID 
  475.                   $requested_connections = explode( ', ', ( preg_replace( '/[\s]*/', '', $publicize[ $name ] ) ) ); 
  476.  
  477.                   // Flag the connections we can't match with the requested list to be skipped. 
  478.                   foreach ( $service_connections as $service_connection ) { 
  479.                       if ( !in_array( $service_connection->meta['connection_data']->id, $requested_connections ) ) { 
  480.                           update_post_meta( $post_id, $GLOBALS['publicize_ui']->publicize->POST_SKIP . $service_connection->unique_id, 1 ); 
  481.                       } else { 
  482.                           delete_post_meta( $post_id, $GLOBALS['publicize_ui']->publicize->POST_SKIP . $service_connection->unique_id ); 
  483.                       } 
  484.                   } 
  485.               } else { 
  486.                   // delete all SKIP values; it's okay to publish to all connected IDs for this service 
  487.                   foreach ( $service_connections as $service_connection ) { 
  488.                       delete_post_meta( $post_id, $GLOBALS['publicize_ui']->publicize->POST_SKIP . $service_connection->unique_id ); 
  489.                   } 
  490.               } 
  491.           } 
  492.       } 
  493.  
  494.       if ( ! is_null( $publicize_custom_message ) ) { 
  495.           if ( empty( $publicize_custom_message ) ) { 
  496.               delete_post_meta( $post_id, $GLOBALS['publicize_ui']->publicize->POST_MESS ); 
  497.           } else { 
  498.               update_post_meta( $post_id, $GLOBALS['publicize_ui']->publicize->POST_MESS, trim( $publicize_custom_message ) ); 
  499.           } 
  500.       } 
  501.  
  502.       if ( ! empty( $insert['post_format'] ) ) { 
  503.           if ( 'default' !== strtolower( $insert['post_format'] ) ) { 
  504.               set_post_format( $post_id, $insert['post_format'] ); 
  505.           } 
  506.           else { 
  507.               set_post_format( $post_id, get_option( 'default_post_format' ) ); 
  508.           } 
  509.       } 
  510.  
  511.       if ( isset( $featured_image ) ) { 
  512.           parent::parse_and_set_featured_image( $post_id, $delete_featured_image, $featured_image ); 
  513.       } 
  514.  
  515.       if ( ! empty( $metadata ) ) { 
  516.           foreach ( (array) $metadata as $meta ) { 
  517.  
  518.               $meta = (object) $meta; 
  519.  
  520.               $existing_meta_item = new stdClass; 
  521.  
  522.               if ( empty( $meta->operation ) ) 
  523.                   $meta->operation = 'update'; 
  524.  
  525.               if ( ! empty( $meta->value ) ) { 
  526.                   if ( 'true' == $meta->value ) 
  527.                       $meta->value = true; 
  528.                   if ( 'false' == $meta->value ) 
  529.                       $meta->value = false; 
  530.               } 
  531.  
  532.               if ( ! empty( $meta->id ) ) { 
  533.                   $meta->id = absint( $meta->id ); 
  534.                   $existing_meta_item = get_metadata_by_mid( 'post', $meta->id ); 
  535.               } 
  536.  
  537.               $unslashed_meta_key = wp_unslash( $meta->key ); // should match what the final key will be 
  538.               $meta->key = wp_slash( $meta->key ); 
  539.               $unslashed_existing_meta_key = wp_unslash( $existing_meta_item->meta_key ); 
  540.               $existing_meta_item->meta_key = wp_slash( $existing_meta_item->meta_key ); 
  541.  
  542.               // make sure that the meta id passed matches the existing meta key 
  543.               if ( ! empty( $meta->id ) && ! empty( $meta->key ) ) { 
  544.                   $meta_by_id = get_metadata_by_mid( 'post', $meta->id ); 
  545.                   if ( $meta_by_id->meta_key !== $meta->key ) { 
  546.                       continue; // skip this meta 
  547.                   } 
  548.               } 
  549.  
  550.               switch ( $meta->operation ) { 
  551.                   case 'delete': 
  552.  
  553.                       if ( ! empty( $meta->id ) && ! empty( $existing_meta_item->meta_key ) && current_user_can( 'delete_post_meta', $post_id, $unslashed_existing_meta_key ) ) { 
  554.                           delete_metadata_by_mid( 'post', $meta->id ); 
  555.                       } elseif ( ! empty( $meta->key ) && ! empty( $meta->previous_value ) && current_user_can( 'delete_post_meta', $post_id, $unslashed_meta_key ) ) { 
  556.                           delete_post_meta( $post_id, $meta->key, $meta->previous_value ); 
  557.                       } elseif ( ! empty( $meta->key ) && current_user_can( 'delete_post_meta', $post_id, $unslashed_meta_key ) ) { 
  558.                           delete_post_meta( $post_id, $meta->key ); 
  559.                       } 
  560.  
  561.                       break; 
  562.                   case 'add': 
  563.  
  564.                       if ( ! empty( $meta->id ) || ! empty( $meta->previous_value ) ) { 
  565.                           continue; 
  566.                       } elseif ( ! empty( $meta->key ) && ! empty( $meta->value ) && ( current_user_can( 'add_post_meta', $post_id, $unslashed_meta_key ) ) || $this->is_metadata_public( $meta->key ) ) { 
  567.                           add_post_meta( $post_id, $meta->key, $meta->value ); 
  568.                       } 
  569.  
  570.                       break; 
  571.                   case 'update': 
  572.  
  573.                       if ( ! isset( $meta->value ) ) { 
  574.                           continue; 
  575.                       } elseif ( ! empty( $meta->id ) && ! empty( $existing_meta_item->meta_key ) && ( current_user_can( 'edit_post_meta', $post_id, $unslashed_existing_meta_key ) || $this->is_metadata_public( $meta->key ) ) ) { 
  576.                           update_metadata_by_mid( 'post', $meta->id, $meta->value ); 
  577.                       } elseif ( ! empty( $meta->key ) && ! empty( $meta->previous_value ) && ( current_user_can( 'edit_post_meta', $post_id, $unslashed_meta_key ) || $this->is_metadata_public( $meta->key ) ) ) { 
  578.                           update_post_meta( $post_id, $meta->key, $meta->value, $meta->previous_value ); 
  579.                       } elseif ( ! empty( $meta->key ) && ( current_user_can( 'edit_post_meta', $post_id, $unslashed_meta_key ) || $this->is_metadata_public( $meta->key ) ) ) { 
  580.                           update_post_meta( $post_id, $meta->key, $meta->value ); 
  581.                       } 
  582.  
  583.                       break; 
  584.               } 
  585.  
  586.           } 
  587.       } 
  588.  
  589.       /** This action is documented in json-endpoints/class.wpcom-json-api-update-post-endpoint.php */ 
  590.       do_action( 'rest_api_inserted_post', $post_id, $insert, $new ); 
  591.  
  592.       $return = $this->get_post_by( 'ID', $post_id, $args['context'] ); 
  593.       if ( !$return || is_wp_error( $return ) ) { 
  594.           return $return; 
  595.       } 
  596.  
  597.       if ( isset( $input['type'] ) && 'revision' === $input['type'] ) { 
  598.           $return['preview_nonce'] = wp_create_nonce( 'post_preview_' . $input['parent'] ); 
  599.       } 
  600.  
  601.       if ( isset( $sticky ) ) { 
  602.           // workaround for sticky test occasionally failing, maybe a race condition with stick_post() above 
  603.           $return['sticky'] = ( true === $sticky ); 
  604.       } 
  605.  
  606.       if ( ! empty( $media_results['errors'] ) ) 
  607.           $return['media_errors'] = $media_results['errors']; 
  608.  
  609.       if ( 'publish' !== $post->post_status && isset( $input['title'] )) { 
  610.           $return['other_URLs'] = (object) $this->get_post_permalink_suggestions( $post_id, $input['title'] ); 
  611.       } 
  612.  
  613.       /** This action is documented in json-endpoints/class.wpcom-json-api-site-settings-endpoint.php */ 
  614.       do_action( 'wpcom_json_api_objects', 'posts' ); 
  615.  
  616.       return $return; 
  617.   }
