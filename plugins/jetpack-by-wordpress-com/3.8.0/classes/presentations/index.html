  272.     /** 
  273.      * Determines the position of the next slide based on the position and scaling of the previous slide. 
  274.      * 
  275.      * @param array $args: an array with the following key-value pairs 
  276.      *           string $transition: the transition name, "up", "down", "left", or "right" 
  277.      *           float $scale: the scale of the next slide (used to determine the position of the slide after that) 
  278.      * 
  279.      * @return array with the 'x' and 'y' coordinates of the slide 
  280.      */ 
  281.     function get_coords( $args ) { 
  282.         if ( 0 == $args['scale'] ) 
  283.             $args['scale'] = 1; 
  284.  
  285.         $width = $this->presentation_settings['width']; 
  286.         $height = $this->presentation_settings['height']; 
  287.         $last = $this->presentation_settings['last']; 
  288.         $scale = $last['scale']; 
  289.  
  290.         $next = array( 
  291.             'x' => $last['x'],  
  292.             'y' => $last['y'],  
  293.             'scale' => $args['scale'],  
  294.             'rotate' => $args['rotate'],  
  295.  ); 
  296.  
  297.         // All angles are measured from the vertical axis, so everything is backwards! 
  298.         $diagAngle = atan2( $width, $height ); 
  299.         $diagonal = sqrt( pow( $width, 2 ) + pow( $height, 2 ) ); 
  300.  
  301.         // We offset the angles by the angle formed by the diagonal so that 
  302.         // we can multiply the sines directly against the diagonal length 
  303.         $theta = deg2rad( $last['rotate'] ) - $diagAngle; 
  304.         $phi = deg2rad( $next['rotate'] ) - $diagAngle; 
  305.  
  306.         // We start by displacing by the slide dimensions 
  307.         $totalHorizDisp = $width  * $scale; 
  308.         $totalVertDisp = $height * $scale; 
  309.  
  310.         // If the previous slide was rotated, we add the incremental offset from the rotation 
  311.         // Namely the difference between the regular dimension (no rotation) and the component 
  312.         // of the diagonal for that angle 
  313.         $totalHorizDisp += ( ( ( abs( sin( $theta ) ) * $diagonal) - $width ) / 2) * $scale; 
  314.         $totalVertDisp  += ( ( ( abs( cos( $theta ) ) * $diagonal) - $height) / 2) * $scale; 
  315.  
  316.         // Similarly, we check if the current slide has been rotated and add whatever additional 
  317.         // offset has been added. This is so that two rotated corners don't clash with each other. 
  318.         // Note: we are checking the raw angle relative to the vertical axis, NOT the diagonal angle. 
  319.         if ( $next['rotate'] % 180 != 0 ) { 
  320.             $totalHorizDisp += ( abs( ( sin( $phi ) * $diagonal ) - $width ) / 2) * $next['scale']; 
  321.             $totalVertDisp  += ( abs( ( cos( $phi ) * $diagonal ) - $height ) / 2) * $next['scale']; 
  322.         } 
  323.  
  324.         switch ( trim( $args['transition'] ) ) { 
  325.             case 'none': 
  326.                 break; 
  327.  
  328.             case 'left': 
  329.                 $next['x'] -= $totalHorizDisp; 
  330.                 break; 
  331.  
  332.             case 'right': 
  333.                 $next['x'] += $totalHorizDisp; 
  334.                 break; 
  335.  
  336.             case 'up': 
  337.                 $next['y'] -= $totalVertDisp; 
  338.                 break; 
  339.  
  340.             case 'down': 
  341.             default: 
  342.                 $next['y'] += $totalVertDisp; 
  343.                 break; 
  344.         } 
  345.  
  346.         $this->presentation_settings['last'] = $next; 
  347.         return $next; 
  348.     }
