  207.     function slideshow_js( $attr ) { 
  208.         // Enqueue scripts 
  209.         $this->enqueue_scripts(); 
  210.  
  211.         $output = ''; 
  212.  
  213.         if ( defined( 'JSON_HEX_AMP' ) ) { 
  214.             // This is nice to have, but not strictly necessary since we use _wp_specialchars() below 
  215.             $gallery = json_encode( $attr['gallery'], JSON_HEX_TAG | JSON_HEX_AMP | JSON_HEX_APOS | JSON_HEX_QUOT ); 
  216.         } else { 
  217.             $gallery = json_encode( $attr['gallery'] ); 
  218.         } 
  219.  
  220.         $output .= '<p class="jetpack-slideshow-noscript robots-nocontent">' . esc_html__( 'This slideshow requires JavaScript.', 'jetpack' ) . '</p>'; 
  221.         $output .= sprintf( '<div id="%s" class="slideshow-window jetpack-slideshow slideshow-%s" data-trans="%s" data-autostart="%s" data-gallery="%s"></div>',  
  222.             esc_attr( $attr['selector'] . '-slideshow' ),  
  223.             esc_attr( $attr['color'] ),  
  224.             esc_attr( $attr['trans'] ),  
  225.             esc_attr( $attr['autostart'] ),  
  226.             /** 
  227.              * The input to json_encode() above can contain '"'. 
  228.              * 
  229.              * For calls to json_encode() lacking the JSON_HEX_AMP option,  
  230.              * that '"' is left unaltered.  Running '"' through esc_attr() 
  231.              * also leaves it unaltered since esc_attr() does not double-encode. 
  232.              * 
  233.              * This means we end up with an attribute like 
  234.              * `data-gallery="{"foo":"""}"`,  
  235.              * which is interpreted by the browser as `{"foo":"""}`,  
  236.              * which cannot be JSON decoded. 
  237.              * 
  238.              * The preferred workaround is to include the JSON_HEX_AMP (and friends) 
  239.              * options, but these are not available until 5.3.0. 
  240.              * Alternatively, we can use _wp_specialchars( , , , true ) instead of 
  241.              * esc_attr(), which will double-encode. 
  242.              * 
  243.              * Since we can't rely on JSON_HEX_AMP, we do both. 
  244.              */ 
  245.             _wp_specialchars( wp_check_invalid_utf8( $gallery ), ENT_QUOTES, false, true ) 
  246.  ); 
  247.  
  248.         return $output;
