  521.                     /** This filter is already documented in core/wp-includes/comment-functions.php */ 
  522.                     apply_filters( 'pre_comment_author_name', addslashes( $field->value ) ) 
  523.  ) 
  524.  ); 
  525.             $comment_author_label = Grunion_Contact_Form_Plugin::strip_tags( $field->get_attribute( 'label' ) ); 
  526.         } 
  527.  
  528.         if ( isset( $field_ids['email'] ) ) { 
  529.             $field = $this->fields[$field_ids['email']]; 
  530.             $comment_author_email = Grunion_Contact_Form_Plugin::strip_tags( 
  531.                 stripslashes( 
  532.                     /** This filter is already documented in core/wp-includes/comment-functions.php */ 
  533.                     apply_filters( 'pre_comment_author_email', addslashes( $field->value ) ) 
  534.  ) 
  535.  ); 
  536.             $comment_author_email_label = Grunion_Contact_Form_Plugin::strip_tags( $field->get_attribute( 'label' ) ); 
  537.         } 
  538.  
  539.         if ( isset( $field_ids['url'] ) ) { 
  540.             $field = $this->fields[$field_ids['url']]; 
  541.             $comment_author_url = Grunion_Contact_Form_Plugin::strip_tags( 
  542.                 stripslashes( 
  543.                     /** This filter is already documented in core/wp-includes/comment-functions.php */ 
  544.                     apply_filters( 'pre_comment_author_url', addslashes( $field->value ) ) 
  545.  ) 
  546.  ); 
  547.             if ( 'http://' == $comment_author_url ) { 
  548.                 $comment_author_url = ''; 
  549.             } 
  550.             $comment_author_url_label = Grunion_Contact_Form_Plugin::strip_tags( $field->get_attribute( 'label' ) ); 
  551.         } 
  552.  
  553.         if ( isset( $field_ids['textarea'] ) ) { 
  554.             $field = $this->fields[$field_ids['textarea']]; 
  555.             $comment_content = trim( Grunion_Contact_Form_Plugin::strip_tags( $field->value ) ); 
  556.             $comment_content_label = Grunion_Contact_Form_Plugin::strip_tags( $field->get_attribute( 'label' ) ); 
  557.         } 
  558.  
  559.         if ( isset( $field_ids['subject'] ) ) { 
  560.             $field = $this->fields[$field_ids['subject']]; 
  561.             if ( $field->value ) { 
  562.                 $contact_form_subject = Grunion_Contact_Form_Plugin::strip_tags( $field->value ); 
  563.             } 
  564.         } 
  565.  
  566.         $all_values = $extra_values = array(); 
  567.         $i = 1; // Prefix counter for stored metadata 
  568.  
  569.         // For all fields, grab label and value 
  570.         foreach ( $field_ids['all'] as $field_id ) { 
  571.             $field = $this->fields[$field_id]; 
  572.             $label = $i . '_' . $field->get_attribute( 'label' ); 
  573.             $value = $field->value; 
  574.  
  575.             $all_values[$label] = $value; 
  576.             $i++; // Increment prefix counter for the next field 
  577.         } 
  578.  
  579.         // For the "non-standard" fields, grab label and value 
  580.         // Extra fields have their prefix starting from count( $all_values ) + 1 
  581.         foreach ( $field_ids['extra'] as $field_id ) { 
  582.             $field = $this->fields[$field_id]; 
  583.             $label = $i . '_' . $field->get_attribute( 'label' ); 
  584.             $value = $field->value; 
  585.  
  586.             if ( is_array( $value ) ) { 
  587.                 $value = implode( ', ', $value ); 
  588.             } 
  589.  
  590.             $extra_values[$label] = $value; 
  591.             $i++; // Increment prefix counter for the next extra field 
  592.         } 
  593.  
  594.         $contact_form_subject = trim( $contact_form_subject ); 
  595.  
  596.         $comment_author_IP = Grunion_Contact_Form_Plugin::get_ip_address(); 
  597.  
  598.         $vars = array( 'comment_author', 'comment_author_email', 'comment_author_url', 'contact_form_subject', 'comment_author_IP' ); 
  599.         foreach ( $vars as $var ) 
  600.             $$var = str_replace( array( "\n", "\r" ), '', $$var ); 
  601.  
  602.         // Ensure that Akismet gets all of the relevant information from the contact form,  
  603.         // not just the textarea field and predetermined subject. 
  604.         $akismet_vars = compact( $vars ); 
  605.         $akismet_vars['comment_content'] = $comment_content; 
  606.  
  607.         foreach ( array_merge( $field_ids['all'], $field_ids['extra'] ) as $field_id ) { 
  608.             $field = $this->fields[$field_id]; 
  609.  
  610.             // Normalize the label into a slug. 
  611.             $field_slug = trim( // Strip all leading/trailing dashes. 
  612.                 preg_replace(   // Normalize everything to a-z0-9_- 
  613.                     '/[^a-z0-9_]+/',  
  614.                     '-',  
  615.                     strtolower( $field->get_attribute( 'label' ) ) // Lowercase 
  616.  ),  
  617.                 '-' 
  618.  ); 
  619.  
  620.             $field_value = ( is_array( $field->value ) ) ? trim( implode( ', ', $field->value ) ) : trim( $field->value ); 
  621.  
  622.             // Skip any values that are already in the array we're sending. 
  623.             if ( $field_value && in_array( $field_value, $akismet_vars ) ) { 
  624.                 continue; 
  625.             } 
  626.  
  627.             $akismet_vars[ 'contact_form_field_' . $field_slug ] = $field_value; 
  628.         } 
  629.  
  630.         $spam = ''; 
  631.         $akismet_values = $plugin->prepare_for_akismet( $akismet_vars ); 
  632.  
  633.         // Is it spam? 
  634.         /** This filter is already documented in modules/contact-form/admin.php */ 
  635.         $is_spam = apply_filters( 'jetpack_contact_form_is_spam', false, $akismet_values ); 
  636.         if ( is_wp_error( $is_spam ) ) // WP_Error to abort 
  637.             return $is_spam; // abort 
  638.         elseif ( $is_spam === TRUE )  // TRUE to flag a spam 
  639.             $spam = '***SPAM*** '; 
  640.  
  641.         if ( !$comment_author ) 
  642.             $comment_author = $comment_author_email; 
  643.  
  644.         /** 
  645.          * Filter the email where a submitted feedback is sent. 
  646.          * 
  647.          * @module contact-form 
  648.          * 
  649.          * @since 1.3.1 
  650.          * 
  651.          * @param string|array $to Array of valid email addresses, or single email address. 
  652.          */ 
  653.         $to = (array) apply_filters( 'contact_form_to', $to ); 
  654.         foreach ( $to as $to_key => $to_value ) { 
  655.             $to[$to_key] = Grunion_Contact_Form_Plugin::strip_tags( $to_value ); 
  656.         } 
  657.  
  658.         $blog_url = parse_url( site_url() ); 
  659.         $from_email_addr = 'wordpress@' . $blog_url['host']; 
  660.  
  661.         $reply_to_addr = $to[0]; 
  662.         if ( ! empty( $comment_author_email ) ) { 
  663.             $reply_to_addr = $comment_author_email; 
  664.         } 
  665.  
  666.         $headers =  'From: "' . $comment_author  .'" <' . $from_email_addr  . ">\r\n" . 
  667.                     'Reply-To: "' . $comment_author . '" <' . $reply_to_addr  . ">\r\n" . 
  668.                     "Content-Type: text/html; charset=\"" . get_option('blog_charset') . "\""; 
  669.  
  670.         /** This filter is already documented in modules/contact-form/admin.php */ 
  671.         $subject = apply_filters( 'contact_form_subject', $contact_form_subject, $all_values ); 
  672.         $url = $widget ? home_url( '/' ) : get_permalink( $post->ID ); 
  673.  
  674.         $date_time_format = _x( '%1$s \a\t %2$s', '{$date_format} \a\t {$time_format}', 'jetpack' ); 
  675.         $date_time_format = sprintf( $date_time_format, get_option( 'date_format' ), get_option( 'time_format' ) ); 
  676.         $time = date_i18n( $date_time_format, current_time( 'timestamp' ) ); 
  677.  
  678.         // keep a copy of the feedback as a custom post type 
  679.         $feedback_time = current_time( 'mysql' ); 
  680.         $feedback_title = "{$comment_author} - {$feedback_time}"; 
  681.         $feedback_status = $is_spam === TRUE ? 'spam' : 'publish'; 
  682.  
  683.         foreach ( (array) $akismet_values as $av_key => $av_value ) { 
  684.             $akismet_values[$av_key] = Grunion_Contact_Form_Plugin::strip_tags( $av_value ); 
  685.         } 
  686.  
  687.         foreach ( (array) $all_values as $all_key => $all_value ) { 
  688.             $all_values[$all_key] = Grunion_Contact_Form_Plugin::strip_tags( $all_value ); 
  689.         } 
  690.  
  691.         foreach ( (array) $extra_values as $ev_key => $ev_value ) { 
  692.             $extra_values[$ev_key] = Grunion_Contact_Form_Plugin::strip_tags( $ev_value ); 
  693.         } 
  694.  
  695.         /** We need to make sure that the post author is always zero for contact 
  696.          * form submissions.  This prevents export/import from trying to create 
  697.          * new users based on form submissions from people who were logged in 
  698.          * at the time. 
  699.          * 
  700.          * Unfortunately wp_insert_post() tries very hard to make sure the post 
  701.          * author gets the currently logged in user id.  That is how we ended up 
  702.          * with this work around. */ 
  703.         add_filter( 'wp_insert_post_data', array( $plugin, 'insert_feedback_filter' ), 10, 2 ); 
  704.  
  705.         $post_id = wp_insert_post( array( 
  706.             'post_date' => addslashes( $feedback_time ),  
  707.             'post_type' => 'feedback',  
  708.             'post_status' => addslashes( $feedback_status ),  
  709.             'post_parent' => (int) $post->ID,  
  710.             'post_title' => addslashes( wp_kses( $feedback_title, array() ) ),  
  711.             'post_content' => addslashes( wp_kses( $comment_content . "\n<!--more-->\n" . "AUTHOR: {$comment_author}\nAUTHOR EMAIL: {$comment_author_email}\nAUTHOR URL: {$comment_author_url}\nSUBJECT: {$subject}\nIP: {$comment_author_IP}\n" . print_r( $all_values, TRUE ), array() ) ), // so that search will pick up this data 
  712.             'post_name' => md5( $feedback_title ),  
  713.  ) ); 
  714.  
  715.         // once insert has finished we don't need this filter any more 
  716.         remove_filter( 'wp_insert_post_data', array( $plugin, 'insert_feedback_filter' ), 10, 2 ); 
  717.  
  718.         update_post_meta( $post_id, '_feedback_extra_fields', $this->addslashes_deep( $extra_values ) ); 
  719.         update_post_meta( $post_id, '_feedback_akismet_values', $this->addslashes_deep( $akismet_values ) ); 
  720.  
  721.         $message = self::get_compiled_form( $post_id, $this ); 
  722.  
  723.         array_push( 
  724.             $message,  
  725.             "", // Empty line left intentionally 
  726.             __( 'Time:', 'jetpack' ) . ' ' . $time . '<br />',  
  727.             __( 'IP Address:', 'jetpack' ) . ' ' . $comment_author_IP . '<br />',  
  728.             __( 'Contact Form URL:', 'jetpack' ) . " " . $url . '<br />' 
  729.  ); 
  730.  
  731.         if ( is_user_logged_in() ) { 
  732.             array_push( 
  733.                 $message,  
  734.                 "",  
  735.                 sprintf( 
  736.                     __( 'Sent by a verified %s user.', 'jetpack' ),  
  737.                     isset( $GLOBALS['current_site']->site_name ) && $GLOBALS['current_site']->site_name ? 
  738.                         $GLOBALS['current_site']->site_name : '"' . get_option( 'blogname' ) . '"' 
  739.  ) 
  740.  ); 
  741.         } else { 
  742.             array_push( $message, __( 'Sent by an unverified visitor to your site.', 'jetpack' ) ); 
  743.         } 
  744.  
  745.         $message = join( $message, "" ); 
  746.         /** 
  747.          * Filters the message sent via email after a successfull form submission. 
  748.          * 
  749.          * @module contact-form 
  750.          * 
  751.          * @since 1.3.1 
  752.          * 
  753.          * @param string $message Feedback email message. 
  754.          */ 
  755.         $message = apply_filters( 'contact_form_message', $message ); 
  756.  
  757.         update_post_meta( $post_id, '_feedback_email', $this->addslashes_deep( compact( 'to', 'message' ) ) ); 
  758.  
  759.         /** 
  760.          * Fires right before the contact form message is sent via email to 
  761.          * the recipient specified in the contact form. 
  762.          * 
  763.          * @module contact-form 
  764.          * 
  765.          * @since 1.3.1 
  766.          * 
  767.          * @param integer $post_id Post contact form lives on 
  768.          * @param array $all_values Contact form fields 
  769.          * @param array $extra_values Contact form fields not included in $all_values 
  770.          */ 
  771.         do_action( 'grunion_pre_message_sent', $post_id, $all_values, $extra_values ); 
  772.  
  773.         // schedule deletes of old spam feedbacks 
  774.         if ( ! wp_next_scheduled( 'grunion_scheduled_delete' ) ) { 
  775.             wp_schedule_event( time() + 250, 'daily', 'grunion_scheduled_delete' ); 
  776.         } 
  777.  
  778.         if ( 
  779.             $is_spam !== TRUE && 
  780.             /** 
  781.              * Filter to choose whether an email should be sent after each successfull contact form submission. 
  782.              * 
  783.              * @module contact-form 
  784.              * 
  785.              * @since 2.6.0 
  786.              * 
  787.              * @param bool true Should an email be sent after a form submission. Default to true. 
  788.              * @param int $post_id Post ID. 
  789.              */ 
  790.             true === apply_filters( 'grunion_should_send_email', true, $post_id ) 
  791.  ) { 
  792.             wp_mail( $to, "{$spam}{$subject}", $message, $headers ); 
  793.         } elseif ( 
  794.             true === $is_spam && 
  795.             /** 
  796.              * Choose whether an email should be sent for each spam contact form submission. 
  797.              * 
  798.              * @module contact-form 
  799.              * 
  800.              * @since 1.3.1 
  801.              * 
  802.              * @param bool false Should an email be sent after a spam form submission. Default to false. 
  803.              */ 
  804.             apply_filters( 'grunion_still_email_spam', FALSE ) == TRUE 
  805.  ) { // don't send spam by default.  Filterable. 
  806.             wp_mail( $to, "{$spam}{$subject}", $message, $headers ); 
  807.         } 
  808.  
  809.         if ( defined( 'DOING_AJAX' ) && DOING_AJAX ) { 
  810.             return self::success_message( $post_id, $this ); 
  811.         } 
  812.  
  813.         $redirect = wp_get_referer(); 
  814.         if ( !$redirect ) { // wp_get_referer() returns false if the referer is the same as the current page 
  815.             $redirect = $_SERVER['REQUEST_URI']; 
  816.         } 
  817.  
  818.         $redirect = add_query_arg( urlencode_deep( array( 
  819.             'contact-form-id' => $id,  
  820.             'contact-form-sent' => $post_id,  
  821.             '_wpnonce' => wp_create_nonce( "contact-form-sent-{$post_id}" ), // wp_nonce_url HTMLencodes :( 
  822.  ) ), $redirect ); 
  823.  
  824.         /** 
  825.          * Filter the URL where the reader is redirected after submitting a form. 
  826.          * 
  827.          * @module contact-form 
  828.          * 
  829.          * @since 1.9.0 
  830.          * 
  831.          * @param string $redirect Post submission URL. 
  832.          * @param int $id Contact Form ID. 
  833.          * @param int $post_id Post ID. 
  834.          */ 
  835.         $redirect = apply_filters( 'grunion_contact_form_redirect_url', $redirect, $id, $post_id ); 
  836.  
  837.         wp_safe_redirect( $redirect ); 
  838.         exit; 
  839.     }

  841.     function addslashes_deep( $value ) { 
  842.         if ( is_array( $value ) ) { 
  843.             return array_map( array( $this, 'addslashes_deep' ), $value ); 
  844.         } elseif ( is_object( $value ) ) { 
  845.             $vars = get_object_vars( $value ); 
  846.             foreach ( $vars as $key => $data ) { 
  847.                 $value->{$key} = $this->addslashes_deep( $data ); 
  848.             } 
  849.             return $value; 
  850.         } 
  851.  
  852.         return addslashes( $value ); 
  853.     }
