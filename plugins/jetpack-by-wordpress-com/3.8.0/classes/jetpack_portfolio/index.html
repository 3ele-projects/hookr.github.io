  452.                     continue; 
  453.                 } 
  454.                 $parsed[] = $orderby; 
  455.             } 
  456.  
  457.             if ( empty( $parsed ) ) { 
  458.                 unset( $atts['orderby'] ); 
  459.             } else { 
  460.                 $atts['orderby'] = implode( ' ', $parsed ); 
  461.             } 
  462.         } 
  463.  
  464.         // enqueue shortcode styles when shortcode is used 
  465.         wp_enqueue_style( 'jetpack-portfolio-style', plugins_url( 'css/portfolio-shortcode.css', __FILE__ ), array(), '20140326' ); 
  466.  
  467.         return self::portfolio_shortcode_html( $atts ); 
  468.     }

  470.     /** 
  471.      * Query to retrieve entries from the Portfolio post_type. 
  472.      * 
  473.      * @return object 
  474.      */ 
  475.     static function portfolio_query( $atts ) { 
  476.         // Default query arguments 
  477.         $default = array( 
  478.             'order' => $atts['order'],  
  479.             'orderby' => $atts['orderby'],  
  480.             'posts_per_page' => $atts['showposts'],  
  481.  ); 
  482.  
  483.         $args = wp_parse_args( $atts, $default ); 
  484.         $args['post_type'] = self::CUSTOM_POST_TYPE; // Force this post type 
  485.  
  486.         if ( false != $atts['include_type'] || false != $atts['include_tag'] ) { 
  487.             $args['tax_query'] = array(); 
  488.         } 
  489.  
  490.         // If 'include_type' has been set use it on the main query 
  491.         if ( false != $atts['include_type'] ) { 
  492.             array_push( $args['tax_query'], array( 
  493.                 'taxonomy' => self::CUSTOM_TAXONOMY_TYPE,  
  494.                 'field' => 'slug',  
  495.                 'terms' => $atts['include_type'],  
  496.  ) ); 
  497.         } 
  498.  
  499.         // If 'include_tag' has been set use it on the main query 
  500.         if ( false != $atts['include_tag'] ) { 
  501.             array_push( $args['tax_query'], array( 
  502.                 'taxonomy' => self::CUSTOM_TAXONOMY_TAG,  
  503.                 'field' => 'slug',  
  504.                 'terms' => $atts['include_tag'],  
  505.  ) ); 
  506.         } 
  507.  
  508.         if ( false != $atts['include_type'] && false != $atts['include_tag'] ) { 
  509.             $args['tax_query']['relation'] = 'AND'; 
  510.         } 
  511.  
  512.         // Run the query and return 
  513.         $query = new WP_Query( $args ); 
  514.         return $query; 
  515.     }

  519.      * 
  520.      * @todo add theme color styles 
  521.      * @return html 
  522.      */ 
  523.     static function portfolio_shortcode_html( $atts ) { 
  524.  
  525.         $query = self::portfolio_query( $atts ); 
  526.         $portfolio_index_number = 0; 
  527.  
  528.         ob_start(); 
  529.  
  530.         // If we have posts, create the html 
  531.         // with hportfolio markup 
  532.         if ( $query->have_posts() ) { 
  533.  
  534.             // Render styles 
  535.             //self::themecolor_styles(); 
  536.  
  537.         ?> 
  538.             <div class="jetpack-portfolio-shortcode column-<?php echo esc_attr( $atts['columns'] ); ?>"> 
  539.             <?php  // open .jetpack-portfolio 
  540.  
  541.             // Construct the loop... 
  542.             while ( $query->have_posts() ) { 
  543.                 $query->the_post(); 
  544.                 $post_id = get_the_ID(); 
  545.                 ?> 
  546.                 <div class="portfolio-entry <?php echo esc_attr( self::get_project_class( $portfolio_index_number, $atts['columns'] ) ); ?>"> 
  547.                     <header class="portfolio-entry-header">

  549.                     // Featured image 
  550.                     echo self::get_portfolio_thumbnail_link( $post_id ); 
  551.                     ?> 
  552.  
  553.                     <h2 class="portfolio-entry-title"><a href="<?php echo esc_url( get_permalink() ); ?>" title="<?php echo esc_attr( the_title_attribute( ) ); ?>"><?php the_title(); ?></a></h2> 
  554.  
  555.                         <div class="portfolio-entry-meta"> 
  556.                         <?php 
  557.                         if ( false != $atts['display_types'] ) { 
  558.                             echo self::get_project_type( $post_id ); 
  559.                         } 
  560.  
  561.                         if ( false != $atts['display_tags'] ) { 
  562.                             echo self::get_project_tags( $post_id ); 
  563.                         } 
  564.                         ?> 
  565.                         </div> 
  566.  
  567.                     </header> 
  568.  
  569.                 <?php 
  570.                 // The content 
  571.                 if ( false !== $atts['display_content'] ) { 
  572.                     if ( 'full' === $atts['display_content'] ) { 
  573.                         echo '<div class="portfolio-entry-content">' . the_content() . '</div>'; 
  574.                     } else { 
  575.                         echo '<div class="portfolio-entry-content">' . the_excerpt() . '</div>'; 
  576.                     } 
  577.                 } ?> 
  578.                 </div><!-- close .portfolio-entry --> 
  579.                 <?php $portfolio_index_number++; 
  580.             } // end of while loop 
  581.  
  582.             wp_reset_postdata(); 
  583.             ?> 
  584.             </div><!-- close .jetpack-portfolio --> 
  585.         <?php 
  586.         } else { ?> 
  587.             <p><em><?php _e( 'Your Portfolio Archive currently has no entries. You can start creating them on your dashboard.', 'jetpack' ); ?></p></em> 
  588.         <?php 
  589.         } 
  590.         $html = ob_get_clean(); 
  591.  
  592.         // If there is a [portfolio] within a [portfolio], remove the shortcode 
  593.         if ( has_shortcode( $html, 'portfolio' ) ) { 
  594.             remove_shortcode( 'portfolio' ); 
  595.         } 
  596.  
  597.         // Return the HTML block 
  598.         return $html; 
  599.     }

  601.     /** 
  602.      * Individual project class 
  603.      * 
  604.      * @return string 
  605.      */ 
  606.     static function get_project_class( $portfolio_index_number, $columns ) { 
  607.         $project_types = wp_get_object_terms( get_the_ID(), self::CUSTOM_TAXONOMY_TYPE, array( 'fields' => 'slugs' ) ); 
  608.         $class = array(); 
  609.  
  610.         $class[] = 'portfolio-entry-column-'.$columns; 
  611.         // add a type- class for each project type 
  612.         foreach ( $project_types as $project_type ) { 
  613.             $class[] = 'type-' . esc_html( $project_type ); 
  614.         } 
  615.         if( $columns > 1) { 
  616.             if ( ( $portfolio_index_number % 2 ) == 0 ) { 
  617.                 $class[] = 'portfolio-entry-mobile-first-item-row'; 
  618.             } else { 
  619.                 $class[] = 'portfolio-entry-mobile-last-item-row'; 
  620.             } 
  621.         } 
  622.  
  623.         // add first and last classes to first and last items in a row 
  624.         if ( ( $portfolio_index_number % $columns ) == 0 ) { 
  625.             $class[] = 'portfolio-entry-first-item-row'; 
  626.         } elseif ( ( $portfolio_index_number % $columns ) == ( $columns - 1 ) ) { 
  627.             $class[] = 'portfolio-entry-last-item-row'; 
  628.         } 
  629.  
  630.  
  631.         /** 
  632.          * Filter the class applied to project div in the portfolio 
  633.          * 
  634.          * @module custom-content-types 
  635.          * 
  636.          * @since 3.1.0 
  637.          * 
  638.          * @param string $class class name of the div. 
  639.          * @param int $portfolio_index_number iterator count the number of columns up starting from 0. 
  640.          * @param int $columns number of columns to display the content in. 
  641.          * 
  642.          */ 
  643.         return apply_filters( 'portfolio-project-post-class', implode( " ", $class ) , $portfolio_index_number, $columns ); 
  644.     }

  651.     static function get_project_type( $post_id ) { 
  652.         $project_types = get_the_terms( $post_id, self::CUSTOM_TAXONOMY_TYPE ); 
  653.  
  654.         // If no types, return empty string

  656.             return; 
  657.         } 
  658.  
  659.         $html = '<div class="project-types"><span>' . __( 'Types', 'jetpack' ) . ':</span>'; 
  660.         $types = array(); 
  661.         // Loop thorugh all the types 
  662.         foreach ( $project_types as $project_type ) { 
  663.             $project_type_link = get_term_link( $project_type, self::CUSTOM_TAXONOMY_TYPE ); 
  664.  
  665.             if ( is_wp_error( $project_type_link ) ) { 
  666.                 return $project_type_link; 
  667.             } 
  668.  
  669.             $types[] = '<a href="' . esc_url( $project_type_link ) . '" rel="tag">' . esc_html( $project_type->name ) . '</a>'; 
  670.         } 
  671.         $html .= ' '.implode( ', ', $types ); 
  672.         $html .= '</div>'; 
  673.  
  674.         return $html;
