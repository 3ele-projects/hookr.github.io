  472.  ) ); 
  473.         $wp_customize->add_control( new WP_Customize_Image_Control( $wp_customize, 'jetpack_testimonials[featured-image]', array(

  485.     public function coerce_testimonial_image_to_url( $opt ) { 
  486.         if ( ! $opt || ! is_array( $opt ) ) { 
  487.             return $opt; 
  488.         }

  492.         $url = wp_get_attachment_url( $opt['featured-image'] );

  499.     /** 
  500.      * Our [testimonial] shortcode. 
  501.      * Prints Testimonial data styled to look good on *any* theme. 
  502.      * 
  503.      * @return jetpack_testimonial_shortcode_html 
  504.      */ 
  505.     static function jetpack_testimonial_shortcode( $atts ) { 
  506.         // Default attributes 
  507.         $atts = shortcode_atts( array( 
  508.             'display_content' => true,  
  509.             'image' => true,  
  510.             'columns' => 1,  
  511.             'showposts' => -1,  
  512.             'order' => 'asc',  
  513.             'orderby' => 'date',  
  514.  ), $atts, 'testimonial' ); 
  515.  
  516.         // A little sanitization 
  517.         if ( $atts['display_content'] && 'true' != $atts['display_content'] && 'full' != $atts['display_content'] ) { 
  518.             $atts['display_content'] = false; 
  519.         } 
  520.  
  521.         if ( $atts['image'] && 'true' != $atts['image'] ) { 
  522.             $atts['image'] = false; 
  523.         } 
  524.  
  525.         $atts['columns'] = absint( $atts['columns'] ); 
  526.  
  527.         $atts['showposts'] = intval( $atts['showposts'] ); 
  528.  
  529.         if ( $atts['order'] ) { 
  530.             $atts['order'] = urldecode( $atts['order'] ); 
  531.             $atts['order'] = strtoupper( $atts['order'] ); 
  532.             if ( 'DESC' != $atts['order'] ) { 
  533.                 $atts['order'] = 'ASC'; 
  534.             } 
  535.         } 
  536.  
  537.         if ( $atts['orderby'] ) { 
  538.             $atts['orderby'] = urldecode( $atts['orderby'] ); 
  539.             $atts['orderby'] = strtolower( $atts['orderby'] ); 
  540.             $allowed_keys = array('author', 'date', 'title', 'rand'); 
  541.  
  542.             $parsed = array(); 
  543.             foreach ( explode( ', ', $atts['orderby'] ) as $testimonial_index_number => $orderby ) { 
  544.                 if ( ! in_array( $orderby, $allowed_keys ) ) { 
  545.                     continue; 
  546.                 } 
  547.                 $parsed[] = $orderby; 
  548.             } 
  549.  
  550.             if ( empty( $parsed ) ) { 
  551.                 unset($atts['orderby']); 
  552.             } else { 
  553.                 $atts['orderby'] = implode( ' ', $parsed ); 
  554.             } 
  555.         } 
  556.  
  557.         // enqueue shortcode styles when shortcode is used 
  558.         wp_enqueue_style( 'jetpack-testimonial-style', plugins_url( 'css/testimonial-shortcode.css', __FILE__ ), array(), '20140326' ); 
  559.  
  560.         return self::jetpack_testimonial_shortcode_html( $atts ); 
  561.     }

  563.     /** 
  564.      * The Testimonial shortcode loop. 
  565.      * 
  566.      * @return html 
  567.      */ 
  568.     static function jetpack_testimonial_shortcode_html( $atts ) { 
  569.         // Default query arguments 
  570.         $defaults = array( 
  571.             'order' => $atts['order'],  
  572.             'orderby' => $atts['orderby'],  
  573.             'posts_per_page' => $atts['showposts'],  
  574.  ); 
  575.  
  576.         $args = wp_parse_args( $atts, $defaults ); 
  577.         $args['post_type'] = self::CUSTOM_POST_TYPE; // Force this post type 
  578.         $query = new WP_Query( $args ); 
  579.  
  580.         $testimonial_index_number = 0; 
  581.  
  582.         ob_start(); 
  583.  
  584.         // If we have testimonials, create the html 
  585.         if ( $query->have_posts() ) { 
  586.  
  587.             ?> 
  588.             <div class="jetpack-testimonial-shortcode column-<?php echo esc_attr( $atts['columns'] ); ?>"> 
  589.                 <?php  // open .jetpack-testimonial-shortcode 
  590.  
  591.                 // Construct the loop... 
  592.                 while ( $query->have_posts() ) { 
  593.                     $query->the_post(); 
  594.                     $post_id = get_the_ID(); 
  595.                     ?> 
  596.                     <div class="testimonial-entry <?php echo esc_attr( self::get_testimonial_class( $testimonial_index_number, $atts['columns'] ) ); ?>"> 
  597.                         <?php 
  598.                         // The content 
  599.                         if ( false !== $atts['display_content'] ) { 
  600.                             if ( 'full' === $atts['display_content'] ) { 
  601.                                 echo '<div class="testimonial-entry-content">' . the_content() . '</div>'; 
  602.                             } else { 
  603.                                 echo '<div class="testimonial-entry-content">' . the_excerpt() . '</div>'; 
  604.                             } 
  605.                         } 
  606.                         ?> 
  607.                         <span class="testimonial-entry-title">― <a href="<?php echo esc_url( get_permalink() ); ?>" title="<?php echo esc_attr( the_title_attribute( ) ); ?>"><?php the_title(); ?></a></span> 
  608.                         <?php 
  609.                         // Featured image 
  610.                         if ( false !== $atts['image'] ) : 
  611.                             echo self::get_testimonial_thumbnail_link( $post_id ); 
  612.                         endif; 
  613.                         ?> 
  614.                     </div><!-- close .testimonial-entry --> 
  615.                     <?php 
  616.                     $testimonial_index_number++; 
  617.                 } // end of while loop 
  618.  
  619.                 wp_reset_postdata(); 
  620.                 ?> 
  621.             </div><!-- close .jetpack-testimonial-shortcode --> 
  622.         <?php 
  623.         } else { ?> 
  624.             <p><em><?php _e( 'Your Testimonial Archive currently has no entries. You can start creating them on your dashboard.', 'jetpack' ); ?></p></em> 
  625.         <?php 
  626.         } 
  627.         $html = ob_get_clean(); 
  628.  
  629.         // Return the HTML block 
  630.         return $html; 
  631.     }

  633.     /** 
  634.      * Individual testimonial class 
  635.      * 
  636.      * @return string 
  637.      */ 
  638.     static function get_testimonial_class( $testimonial_index_number, $columns ) { 
  639.         $class = array(); 
  640.  
  641.         $class[] = 'testimonial-entry-column-'.$columns; 
  642.  
  643.         if( $columns > 1) { 
  644.             if ( ( $testimonial_index_number % 2 ) == 0 ) { 
  645.                 $class[] = 'testimonial-entry-mobile-first-item-row'; 
  646.             } else { 
  647.                 $class[] = 'testimonial-entry-mobile-last-item-row'; 
  648.             } 
  649.         } 
  650.  
  651.         // add first and last classes to first and last items in a row 
  652.         if ( ( $testimonial_index_number % $columns ) == 0 ) { 
  653.             $class[] = 'testimonial-entry-first-item-row'; 
  654.         } elseif ( ( $testimonial_index_number % $columns ) == ( $columns - 1 ) ) { 
  655.             $class[] = 'testimonial-entry-last-item-row'; 
  656.         } 
  657.  
  658.  
  659.         /** 
  660.          * Filter the class applied to testimonial div in the testimonial 
  661.          * 
  662.          * @module custom-content-types 
  663.          * 
  664.          * @since 3.4.0 
  665.          * 
  666.          * @param string $class class name of the div. 
  667.          * @param int $testimonial_index_number iterator count the number of columns up starting from 0. 
  668.          * @param int $columns number of columns to display the content in. 
  669.          * 
  670.          */ 
  671.         return apply_filters( 'testimonial-entry-post-class', implode( " ", $class ) , $testimonial_index_number, $columns ); 
  672.     }
