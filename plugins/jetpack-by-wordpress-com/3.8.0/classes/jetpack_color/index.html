  594.         // if current max contrast is less than the target contrast, we had wishful thinking. 
  595.         // still, go max 
  596.         if ( $max_contrast <= $target_contrast ) { 
  597.             return $max_contrast_color; 
  598.         } 
  599.         // or, we might already have sufficient contrast 
  600.         if ( $contrast >= $target_contrast ) { 
  601.             return $this; 
  602.         } 
  603.  
  604.         $incr = ( 0 === $max_contrast_color->toInt() ) ? -1 : 1; 
  605.         while ( $contrast < $target_contrast ) { 
  606.             $this->incrementLightness( $incr ); 
  607.             $contrast = $bg_color->getDistanceLuminosityFrom( $this ); 
  608.             // infininite loop prevention: you never know. 
  609.             if ( $this->color === 0 || $this->color === 16777215 ) { 
  610.                 break; 
  611.             } 
  612.         } 
  613.  
  614.         return $this; 
  615.     }
