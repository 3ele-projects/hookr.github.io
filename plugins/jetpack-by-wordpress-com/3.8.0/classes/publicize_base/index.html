  153.             return 'https://plus.google.com/' . $cmeta['external_id']; 
  154.         } else if ( 'linkedin' == $service_name ) {

  264.         $cron_user = null;

  271.

  275.         } 
  276.  
  277.         // - on quick edit, autosave, etc but do fire on p2, quickpress, and instapost ajax

  279.             defined( 'DOING_AJAX' )

  281.             DOING_AJAX

  294.         // - bulk edit

  297.         } 
  298.  
  299.         // - API/XML-RPC Test Posts 
  300.         if ( 
  301.             ( 
  302.                 defined( 'XMLRPC_REQUEST' ) 
  303.             && 
  304.                 XMLRPC_REQUEST 
  305.             || 
  306.                 defined( 'APP_REQUEST' )

  308.                 APP_REQUEST

  312.  ) { 
  313.             $submit_post = false; 
  314.         } 
  315.  
  316.         // only work with certain statuses (avoids inherits, auto drafts etc) 
  317.         if ( !in_array( $post->post_status, array( 'publish', 'draft', 'future' ) ) ) { 
  318.             $submit_post = false; 
  319.         } 
  320.  
  321.         // don't publish password protected posts 
  322.         if ( '' !== $post->post_password ) { 
  323.             $submit_post = false; 
  324.         } 
  325.  
  326.         // Did this request happen via wp-admin?

  328.  
  329.         if ( ( $from_web || defined( 'POST_BY_EMAIL' ) ) && isset( $_POST['wpas_title'] ) ) { 
  330.             if ( empty( $_POST['wpas_title'] ) ) { 
  331.                 delete_post_meta( $post_id, $this->POST_MESS ); 
  332.             } else { 
  333.                 update_post_meta( $post_id, $this->POST_MESS, trim( stripslashes( $_POST['wpas_title'] ) ) ); 
  334.             } 
  335.         } 
  336.  
  337.         // change current user to provide context for get_services() if we're running during cron 
  338.         if ( defined( 'DOING_CRON' ) && DOING_CRON ) { 
  339.             $cron_user = (int) $GLOBALS['user_ID']; 
  340.             wp_set_current_user( $post->post_author ); 
  341.         }

  343.         /** 
  344.          * In this phase, we mark connections that we want to SKIP. When Publicize is actually triggered,  
  345.          * it will Publicize to everything *except* those marked for skipping. 
  346.          */ 
  347.         foreach ( (array) $this->get_services( 'connected' ) as $service_name => $connections ) { 
  348.             foreach ( $connections as $connection ) {

  359.                 } 
  360.  
  361.                 if ( !empty( $connection->unique_id ) )

  363.                 else if ( !empty( $connection['connection_data']['token_id'] ) ) 
  364.                     $unique_id = $connection['connection_data']['token_id'];
