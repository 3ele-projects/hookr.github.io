  490.  
  491.         // Let's print_r the option if it's an array 
  492.         // Used in the 'get' and 'list' actions 
  493.         $option = is_array( $option ) ? print_r( $option ) : $option; 
  494.  
  495.         switch ( $action ) { 
  496.             case 'get': 
  497.                 WP_CLI::success( "\t" . $option ); 
  498.                 break; 
  499.             case 'delete': 
  500.                 jetpack_cli_are_you_sure( $flagged ); 
  501.  
  502.                 Jetpack_Options::delete_option( $args[1] ); 
  503.                 WP_CLI::success( sprintf( __( 'Deleted option: %s', 'jetpack' ), $args[1] ) ); 
  504.                 break; 
  505.             case 'update': 
  506.                 jetpack_cli_are_you_sure( $flagged ); 
  507.  
  508.                 // Updating arrays would get pretty tricky... 
  509.                 $value = Jetpack_Options::get_option( $args[1] ); 
  510.                 if ( $value && is_array( $value ) ) { 
  511.                     WP_CLI::error( __( 'Sorry, no updating arrays at this time', 'jetpack' ) ); 
  512.                 } 
  513.  
  514.                 Jetpack_Options::update_option( $args[1], $args[2] ); 
  515.                 WP_CLI::success( sprintf( _x( 'Updated option: %s to "%s"', 'Updating an option from "this" to "that".', 'jetpack' ), $args[1], $args[2] ) ); 
  516.                 break; 
  517.             case 'list': 
  518.                 $options_compact = Jetpack_Options::get_option_names(); 
  519.                 $options_non_compact = Jetpack_Options::get_option_names( 'non_compact' ); 
  520.                 $options_private = Jetpack_Options::get_option_names( 'private' ); 
  521.                 $options = array_merge( $options_compact, $options_non_compact, $options_private ); 
  522.  
  523.                 // Table headers 
  524.                 WP_CLI::line( "\t" . str_pad( __( 'Option', 'jetpack' ), 30 ) . __( 'Value', 'jetpack' ) ); 
  525.  
  526.                 // List out the options and their values 
  527.                 // Tell them if the value is empty or not 
  528.                 // Tell them if it's an array 
  529.                 foreach ( $options as $option ) { 
  530.                     $value = Jetpack_Options::get_option( $option ); 
  531.                     if ( ! $value ) { 
  532.                         WP_CLI::line( "\t" . str_pad( $option, 30 ) . 'Empty' ); 
  533.                         continue; 
  534.                     } 
  535.  
  536.                     if ( ! is_array( $value ) ) { 
  537.                         WP_CLI::line( "\t" . str_pad( $option, 30 ) . $value ); 
  538.                     } else if ( is_array( $value ) ) { 
  539.                         WP_CLI::line( "\t" . str_pad( $option, 30 ) . 'Array - Use "get <option>" to read option array.' ); 
  540.                     } 
  541.                 } 
  542.                 $option_text = '{' . _x( 'option', 'a variable command that a user can write, provided in the printed instructions', 'jetpack' ) . '}'; 
  543.                 $value_text = '{' . _x( 'value', 'the value that they want to update the option to', 'jetpack' ) . '}'; 
  544.  
  545.                 WP_CLI::success( 
  546.                     _x( "Above are your options. You may 'get', 'delete', and 'update' them.", "'get', 'delete', and 'update' are commands - do not translate.", 'jetpack' ) . "\n" . 
  547.                     str_pad( 'wp jetpack options get', 26 )    . $option_text . "\n" . 
  548.                     str_pad( 'wp jetpack options delete', 26 ) . $option_text . "\n" . 
  549.                     str_pad( 'wp jetpack options update', 26 ) . "$option_text $value_text" . "\n" . 
  550.                     _x( "Type 'wp jetpack options' for more info.", "'wp jetpack options' is a command - do not translate.", 'jetpack' ) . "\n" 
  551.  ); 
  552.                 break; 
  553.         } 
  554.     } 
  555. }
