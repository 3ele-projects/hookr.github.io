  314.       public function bxcft_signup_validate() 
  315.       { 
  316.           global $bp; 
  317.           if ( bp_is_active( 'xprofile' ) ) 
  318.           { 
  319.               if ( isset( $_POST['signup_profile_field_ids'] ) && !empty( $_POST['signup_profile_field_ids'] ) ) 
  320.               { 
  321.                   $profile_field_ids = explode(', ', $_POST['signup_profile_field_ids']); 
  322.                   foreach ($profile_field_ids as $field_id) 
  323.                   { 
  324.                       $field = new BP_XProfile_Field($field_id); 
  325.                       if ( ($field->type == 'image' || $field->type == 'file') && 
  326.                               isset($_FILES['field_'.$field_id])) { 
  327.                           // Delete required field error. 
  328.                           unset($bp->signup->errors['field_'.$field_id]); 
  329.  
  330.                           $filesize = round($_FILES['field_'.$field_id]['size'] / (1024 * 1024), 2); 
  331.                           if ($field->is_required && $filesize <= 0) { 
  332.                               $bp->signup->errors['field_' . $field_id] = __( 'This is a required field', 'buddypress' ); 
  333.                           } elseif ($filesize > 0) { 
  334.                               // Check extensions. 
  335.                               $ext = strtolower(substr($_FILES['field_'.$field_id]['name'], strrpos($_FILES['field_'.$field_id]['name'], '.')+1)); 
  336.                               if ($field->type == 'image') { 
  337.                                   if (!in_array($ext, $this->images_ext_allowed)) { 
  338.                                       $bp->signup->errors['field_'.$field_id] = sprintf(__('Image type not allowed: (%s).', 'bxcft'), implode(', ', $this->images_ext_allowed)); 
  339.                                   } 
  340.                                   elseif ($filesize > $this->images_max_filesize) { 
  341.                                       $bp->signup->errors['field_'.$field_id] = sprintf(__('Max image upload size: %s MB.', 'bxcft'), $this->images_max_filesize); 
  342.                                   } 
  343.                               } elseif ($field->type == 'file') { 
  344.                                   if (!in_array($ext, $this->files_ext_allowed)) { 
  345.                                       $bp->signup->errors['field_'.$field_id] = sprintf(__('File type not allowed: (%s).', 'bxcft'), implode(', ', $this->files_ext_allowed)); 
  346.                                   } 
  347.                                   elseif ($filesize > $this->files_max_filesize) { 
  348.                                       $bp->signup->errors['field_'.$field_id] = sprintf(__('Max file upload size: %s MB.', 'bxcft'), $this->files_max_filesize); 
  349.                                   } 
  350.                               } 
  351.                           } 
  352.                       } 
  353.                       elseif ($field->type == 'checkbox_acceptance' && $field->is_required) { 
  354.                           if (isset($_POST['field_' . $field_id]) 
  355.                                   && $_POST['field_' . $field_id] != 1) { 
  356.                               $bp->signup->errors['field_' . $field_id] = __( 'This is a required field', 'buddypress' ); 
  357.                           } 
  358.                       } 
  359.                       elseif ($field->type == 'birthdate') { 
  360.                           $max_age = 0; 
  361.                           $options = $field->get_children(); 
  362.                           foreach ($options as $o) { 
  363.                               if ($o->name != 'show_age' && $o->name != 'show_birthdate') { 
  364.                                   $max_age = (int)$o->name; 
  365.                                   break; 
  366.                               } 
  367.                           } 
  368.  
  369.                           if ($max_age > 0) { 
  370.                               // Check birthdate. 
  371.                               if (class_exists(DateTime)) { 
  372.                                   $now = new DateTime(); 
  373.                                   $birthdate = new DateTime(sprintf("%s-%s-%s",  
  374.                                       $_POST['field_'.$field_id.'_year'],  
  375.                                       $_POST['field_'.$field_id.'_month'],  
  376.                                       $_POST['field_'.$field_id.'_day'])); 
  377.                                   $age = $now->diff($birthdate); 
  378.                                   if ($age->y < $max_age) { 
  379.                                       $bp->signup->errors['field_' . $field_id] = sprintf(__( 'You have to be at least %s years old.', 'bxcft' ), $max_age); 
  380.                                   } 
  381.                               } 
  382.                           } 
  383.                       } 
  384.                   } // End foreach... 
  385.               } // End if ( isset... 
  386.           } // End if ( bp_is_active(... 
  387.       }

  389.       function bxcft_xprofile_data_before_save($data) 
  390.       { 
  391.           global $bp; 
  392.  
  393.           $field_id = $data->field_id; 
  394.           $field = new BP_XProfile_Field($field_id); 
  395.  
  396.           if ($field->type == 'image' || $field->type == 'file' && isset($_FILES['field_'.$field_id])) 
  397.           { 
  398.               $uploads = wp_upload_dir(); 
  399.               $filesize = round($_FILES['field_'.$field_id]['size'] / (1024 * 1024), 2); 
  400.               if (isset($_FILES['field_'.$field_id]) && $filesize > 0) 
  401.               { 
  402.                   $ext = strtolower(substr($_FILES['field_'.$field_id]['name'], strrpos($_FILES['field_'.$field_id]['name'], '.')+1)); 
  403.                   if ($field->type == 'image') 
  404.                   { 
  405.                       $ext_allowed = $this->images_ext_allowed; 
  406.                       if (!in_array($ext, $ext_allowed)) 
  407.                       { 
  408.                           bp_core_add_message( sprintf(__('Image type not allowed: (%s).', 'bxcft'), implode(', ', $this->images_ext_allowed)), 'error' ); 
  409.                           bp_core_redirect( trailingslashit( bp_displayed_user_domain() . $bp->profile->slug . '/edit/group/' . bp_action_variable( 1 ) ) ); 
  410.                       } 
  411.                       elseif ($filesize > $this->images_max_filesize) { 
  412.                           bp_core_add_message( sprintf(__('Max image upload size: %s MB.', 'bxcft'), $this->images_max_filesize), 'error' ); 
  413.                           bp_core_redirect( trailingslashit( bp_displayed_user_domain() . $bp->profile->slug . '/edit/group/' . bp_action_variable( 1 ) ) ); 
  414.                       } else { 
  415.                           // Delete previous image. 
  416.                           if (isset($_POST['field_'.$field_id.'_hiddenimg'])      && 
  417.                               !empty($_POST['field_'.$field_id.'_hiddenimg'])     && 
  418.                               file_exists($uploads['basedir'] . $_POST['field_'.$field_id.'_hiddenimg'])) 
  419.                           { 
  420.                               unlink($uploads['basedir'] . $_POST['field_'.$field_id.'_hiddenimg']); 
  421.                           } 
  422.                       } 
  423.                   }
