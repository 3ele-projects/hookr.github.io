  315.   function sort_menu_item_queries_by_menu_order( $query ) { 
  316.       if ( ! $this->is_menu_item_query( $query ) ) { 
  317.           return; 
  318.       } 
  319.  
  320.       $query->query_vars['orderby'] = 'menu_order'; 
  321.       $query->query_vars['order'] = 'ASC'; 
  322.  
  323.       // For now, just turn off paging so we can sort by taxonmy later 
  324.       // If we want paging in the future, we'll need to add the taxonomy sort here (or at least before the DB query is made) 
  325.       $query->query_vars['posts_per_page'] = -1; 
  326.   }

  328.   function sort_menu_item_queries_by_menu_taxonomy( $posts, $query ) { 
  329.       if ( !$posts ) { 
  330.           return $posts; 
  331.       } 
  332.  
  333.       if ( !$this->is_menu_item_query( $query ) ) { 
  334.           return $posts; 
  335.       } 
  336.  
  337.       $grouped_by_term = array(); 
  338.  
  339.       foreach ( $posts as $post ) { 
  340.           $term = $this->get_menu_item_menu_leaf( $post->ID ); 
  341.           if ( !$term || is_wp_error( $term ) ) { 
  342.               $term_id = 0; 
  343.           } else { 
  344.               $term_id = $term->term_id; 
  345.           } 
  346.  
  347.           if ( !isset( $grouped_by_term["$term_id"] ) ) { 
  348.               $grouped_by_term["$term_id"] = array(); 
  349.           } 
  350.  
  351.           $grouped_by_term["$term_id"][] = $post; 
  352.       } 
  353.  
  354.       $term_order = get_option( 'nova_menu_order', array() ); 
  355.  
  356.       $return = array(); 
  357.       foreach ( $term_order as $term_id ) { 
  358.           if ( isset( $grouped_by_term["$term_id"] ) ) { 
  359.               $return = array_merge( $return, $grouped_by_term["$term_id"] ); 
  360.               unset( $grouped_by_term["$term_id"] ); 
  361.           } 
  362.       } 
  363.  
  364.       foreach ( $grouped_by_term as $term_id => $posts ) { 
  365.           $return = array_merge( $return, $posts ); 
  366.       } 
  367.  
  368.       return $return; 
  369.   }

  1050.    * If the Query is a Menu Item Query, start outputing the Menu Item Loop Marku

  1053.    * @param WP_Query

  1055.   function start_menu_item_loop( $query ) { 
  1056.       if ( !$this->is_menu_item_query( $query ) ) { 
  1057.           return; 
  1058.       } 
  1059.  
  1060.       $this->menu_item_loop_last_term_id = false; 
  1061.       $this->menu_item_loop_current_term = false; 
  1062.  
  1063.       add_action( 'the_post', array( $this, 'menu_item_loop_each_post' ) );

  1067.   /** 
  1068.    * Outputs the Menu Item Loop Marku 
  1069.    * Attached to the 'the_post' action. 
  1070.    * 
  1071.    * @param WP_Post 
  1072.    */ 
  1073.   function menu_item_loop_each_post( $post ) { 
  1074.       $this->menu_item_loop_current_term = $this->get_menu_item_menu_leaf( $post->ID ); 
  1075.  
  1076.       if ( false === $this->menu_item_loop_last_term_id ) { 
  1077.           // We're at the very beginning of the loop 
  1078.  
  1079.           $this->menu_item_loop_open_element( 'menu' ); // Start a new menu section 
  1080.           $this->menu_item_loop_header(); // Output the menu's header 
  1081.       } elseif ( $this->menu_item_loop_last_term_id != $this->menu_item_loop_current_term->term_id ) { 
  1082.           // We're not at the very beginning but still need to start a new menu section.  End the previous menu section first. 
  1083.  
  1084.           $this->menu_item_loop_close_element( 'menu' ); // End the previous menu section 
  1085.           $this->menu_item_loop_open_element( 'menu' ); // Start a new menu section 
  1086.           $this->menu_item_loop_header(); // Output the menu's header 
  1087.       } 
  1088.  
  1089.       $this->menu_item_loop_last_term_id = $this->menu_item_loop_current_term->term_id; 
  1090.   }

  1098.   function stop_menu_item_loop( $query ) {

  1100.           return;

  1102.
