  256.   function add_data_to_images( $attr, $attachment = null ) { 
  257.  
  258.       // not in a gallery? 
  259.       if ( ! $this->in_gallery ) { 
  260.           return $attr; 
  261.       } 
  262.  
  263.       $attachment_id = intval( $attachment->ID ); 
  264.       $orig_file = wp_get_attachment_image_src( $attachment_id, 'full' ); 
  265.       $orig_file = isset( $orig_file[0] ) ? $orig_file[0] : wp_get_attachment_url( $attachment_id ); 
  266.       $meta = wp_get_attachment_metadata( $attachment_id ); 
  267.       $size = isset( $meta['width'] ) ? intval( $meta['width'] ) . ', ' . intval( $meta['height'] ) : ''; 
  268.       $img_meta = ( ! empty( $meta['image_meta'] ) ) ? (array) $meta['image_meta'] : array(); 
  269.       $comments_opened = intval( comments_open( $attachment_id ) ); 
  270.  
  271.        /** 
  272.        * Note: Cannot generate a filename from the width and height wp_get_attachment_image_src() returns because 
  273.        * it takes the $content_width global variable themes can set in consideration, therefore returning sizes 
  274.        * which when used to generate a filename will likely result in a 404 on the image. 
  275.        * $content_width has no filter we could temporarily de-register, run wp_get_attachment_image_src(), then 
  276.        * re-register. So using returned file URL instead, which we can define the sizes from through filename 
  277.        * parsing in the JS, as this is a failsafe file reference. 
  278.        * 
  279.        * EG with Twenty Eleven activated: 
  280.        * array(4) { [0]=> string(82) "http://vanillawpinstall.blah/wp-content/uploads/2012/06/IMG_3534-1024x764.jpg" [1]=> int(584) [2]=> int(435) [3]=> bool(true) } 
  281.        * 
  282.        * EG with Twenty Ten activated: 
  283.        * array(4) { [0]=> string(82) "http://vanillawpinstall.blah/wp-content/uploads/2012/06/IMG_3534-1024x764.jpg" [1]=> int(640) [2]=> int(477) [3]=> bool(true) } 
  284.        */ 
  285.  
  286.       $medium_file_info = wp_get_attachment_image_src( $attachment_id, 'medium' ); 
  287.       $medium_file = isset( $medium_file_info[0] ) ? $medium_file_info[0] : ''; 
  288.  
  289.       $large_file_info = wp_get_attachment_image_src( $attachment_id, 'large' ); 
  290.       $large_file = isset( $large_file_info[0] ) ? $large_file_info[0] : ''; 
  291.  
  292.       $attachment = get_post( $attachment_id ); 
  293.       $attachment_title = wptexturize( $attachment->post_title ); 
  294.       $attachment_desc = wpautop( wptexturize( $attachment->post_content ) ); 
  295.  
  296.       // Not yet providing geo-data, need to "fuzzify" for privacy 
  297.       if ( ! empty( $img_meta ) ) { 
  298.           foreach ( $img_meta as $k => $v ) { 
  299.               if ( 'latitude' == $k || 'longitude' == $k ) 
  300.                   unset( $img_meta[$k] ); 
  301.           } 
  302.       } 
  303.  
  304.       // See https://github.com/Automattic/jetpack/issues/2765 
  305.       if ( isset( $img_meta['keywords'] ) ) { 
  306.           unset( $img_meta['keywords'] ); 
  307.       } 
  308.  
  309.       $img_meta = json_encode( array_map( 'strval', $img_meta ) ); 
  310.  
  311.       $attr['data-attachment-id'] = $attachment_id; 
  312.       $attr['data-orig-file'] = esc_attr( $orig_file ); 
  313.       $attr['data-orig-size'] = $size; 
  314.       $attr['data-comments-opened'] = $comments_opened; 
  315.       $attr['data-image-meta'] = esc_attr( $img_meta ); 
  316.       $attr['data-image-title'] = esc_attr( $attachment_title ); 
  317.       $attr['data-image-description'] = esc_attr( $attachment_desc ); 
  318.       $attr['data-medium-file'] = esc_attr( $medium_file ); 
  319.       $attr['data-large-file'] = esc_attr( $large_file ); 
  320.  
  321.       return $attr; 
  322.   }

  362.   function get_attachment_comments() { 
  363.       if ( ! headers_sent() ) 
  364.           header('Content-type: text/javascript'); 
  365.  
  366.       /** 
  367.        * Allows for the checking of privileges of the blog user before comments 
  368.        * are packaged as JSON and sent back from the get_attachment_comments 
  369.        * AJAX endpoint 
  370.        * 
  371.        * @module carousel 
  372.        * 
  373.        * @since 1.6.0 
  374.        */ 
  375.       do_action('jp_carousel_check_blog_user_privileges'); 
  376.  
  377.       $attachment_id = ( isset( $_REQUEST['id'] ) ) ? (int) $_REQUEST['id'] : 0; 
  378.       $offset = ( isset( $_REQUEST['offset'] ) ) ? (int) $_REQUEST['offset'] : 0; 
  379.  
  380.       if ( ! $attachment_id ) { 
  381.           echo json_encode( __( 'Missing attachment ID.', 'jetpack' ) ); 
  382.           die(); 
  383.       } 
  384.  
  385.       if ( $offset < 1 ) 
  386.           $offset = 0; 
  387.  
  388.       $comments = get_comments( array( 
  389.           'status' => 'approve',  
  390.           'order' => ( 'asc' == get_option('comment_order') ) ? 'ASC' : 'DESC',  
  391.           'number' => 10,  
  392.           'offset' => $offset,  
  393.           'post_id' => $attachment_id,  
  394.  ) ); 
  395.  
  396.       $out = array(); 
  397.  
  398.       // Can't just send the results, they contain the commenter's email address. 
  399.       foreach ( $comments as $comment ) { 
  400.           $avatar = get_avatar( $comment->comment_author_email, 64 ); 
  401.           if( ! $avatar ) 
  402.               $avatar = ''; 
  403.           $out[] = array( 
  404.               'id' => $comment->comment_ID,  
  405.               'parent_id' => $comment->comment_parent,  
  406.               'author_markup' => get_comment_author_link( $comment->comment_ID ),  
  407.               'gravatar_markup' => $avatar,  
  408.               'date_gmt' => $comment->comment_date_gmt,  
  409.               'content' => wpautop($comment->comment_content),  
  410.  ); 
  411.       } 
  412.  
  413.       die( json_encode( $out ) ); 
  414.   }

  416.   function post_attachment_comment() { 
  417.       if ( ! headers_sent() ) 
  418.           header('Content-type: text/javascript'); 
  419.  
  420.       if ( empty( $_POST['nonce'] ) || ! wp_verify_nonce($_POST['nonce'], 'carousel_nonce') ) 
  421.           die( json_encode( array( 'error' => __( 'Nonce verification failed.', 'jetpack' ) ) ) ); 
  422.  
  423.       $_blog_id = (int) $_POST['blog_id']; 
  424.       $_post_id = (int) $_POST['id']; 
  425.       $comment = $_POST['comment']; 
  426.  
  427.       if ( empty( $_blog_id ) ) 
  428.           die( json_encode( array( 'error' => __( 'Missing target blog ID.', 'jetpack' ) ) ) ); 
  429.  
  430.       if ( empty( $_post_id ) ) 
  431.           die( json_encode( array( 'error' => __( 'Missing target post ID.', 'jetpack' ) ) ) ); 
  432.  
  433.       if ( empty( $comment ) ) 
  434.           die( json_encode( array( 'error' => __( 'No comment text was submitted.', 'jetpack' ) ) ) ); 
  435.  
  436.       // Used in context like NewDash 
  437.       $switched = false; 
  438.       if ( is_multisite() && $_blog_id != get_current_blog_id() ) { 
  439.           switch_to_blog( $_blog_id ); 
  440.           $switched = true; 
  441.       } 
  442.  
  443.       /** This action is documented in modules/carousel/jetpack-carousel.php */ 
  444.       do_action('jp_carousel_check_blog_user_privileges'); 
  445.  
  446.       if ( ! comments_open( $_post_id ) ) 
  447.           die( json_encode( array( 'error' => __( 'Comments on this post are closed.', 'jetpack' ) ) ) ); 
  448.  
  449.       if ( is_user_logged_in() ) { 
  450.           $user = wp_get_current_user(); 
  451.           $user_id = $user->ID; 
  452.           $display_name = $user->display_name; 
  453.           $email = $user->user_email; 
  454.           $url = $user->user_url; 
  455.  
  456.           if ( empty( $user_id ) ) 
  457.               die( json_encode( array( 'error' => __( 'Sorry, but we could not authenticate your request.', 'jetpack' ) ) ) ); 
  458.       } else { 
  459.           $user_id = 0; 
  460.           $display_name = $_POST['author']; 
  461.           $email = $_POST['email']; 
  462.           $url = $_POST['url']; 
  463.  
  464.           if ( get_option( 'require_name_email' ) ) { 
  465.               if ( empty( $display_name ) ) 
  466.                   die( json_encode( array( 'error' => __( 'Please provide your name.', 'jetpack' ) ) ) ); 
  467.  
  468.               if ( empty( $email ) ) 
  469.                   die( json_encode( array( 'error' => __( 'Please provide an email address.', 'jetpack' ) ) ) ); 
  470.  
  471.               if ( ! is_email( $email ) ) 
  472.                   die( json_encode( array( 'error' => __( 'Please provide a valid email address.', 'jetpack' ) ) ) ); 
  473.           } 
  474.       } 
  475.  
  476.       $comment_data =  array( 
  477.           'comment_content' => $comment,  
  478.           'comment_post_ID' => $_post_id,  
  479.           'comment_author' => $display_name,  
  480.           'comment_author_email' => $email,  
  481.           'comment_author_url' => $url,  
  482.           'comment_approved' => 0,  
  483.           'comment_type' => '',  
  484.  ); 
  485.  
  486.       if ( ! empty( $user_id ) ) 
  487.           $comment_data['user_id'] = $user_id; 
  488.  
  489.       // Note: wp_new_comment() sanitizes and validates the values (too). 
  490.       $comment_id = wp_new_comment( $comment_data ); 
  491.  
  492.       /** 
  493.        * Fires before adding a new comment to the database via the get_attachment_comments ajax endpoint. 
  494.        * 
  495.        * @module carousel 
  496.        * 
  497.        * @since 1.6.0 
  498.        */ 
  499.       do_action( 'jp_carousel_post_attachment_comment' ); 
  500.       $comment_status = wp_get_comment_status( $comment_id ); 
  501.  
  502.       if ( true == $switched ) 
  503.           restore_current_blog(); 
  504.  
  505.       die( json_encode( array( 'comment_id' => $comment_id, 'comment_status' => $comment_status ) ) ); 
  506.   }

  532.   function test_1or0_option( $value, $default_to_1 = true ) { 
  533.       if ( true == $default_to_1 ) { 
  534.           // Binary false (===) of $value means it has not yet been set, in which case we do want to default sites to 1 
  535.           if ( false === $value ) 
  536.               $value = 1; 
  537.       } 
  538.       return ( 1 == $value ) ? 1 : 0; 
  539.   }

  542.       return ( 1 == $value ) ? 1 : 0;

  546.       if ( empty( $name ) )

  558.   function settings_select($name, $values, $extra_text = '') { 
  559.       if ( empty( $name ) || ! is_array( $values ) || empty( $values ) ) 
  560.           return; 
  561.       $option = get_option( $name ); 
  562.       echo '<fieldset>'; 
  563.       echo '<select name="'.esc_attr($name).'" id="'.esc_attr($name).'">'; 
  564.       foreach( $values as $key => $value ) {

  566.           selected( $key, $option );

  568.       } 
  569.       echo '</select>'; 
  570.       if ( ! empty( $extra_text ) )
