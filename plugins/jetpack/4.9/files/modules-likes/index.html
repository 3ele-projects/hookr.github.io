  1023.  
  1024.       global $post, $wp_current_filter;              // Used to apply 'sharing_show' filter

  1027.       if ( is_feed() || is_preview() || is_comments_popup() ) { 
  1028.           $enabled = false;

  1030.       // Not a feed or preview, so what is it? 
  1031.       } else { 
  1032.

  1034.               // If in the loop, check if the current post is likeable 
  1035.               $enabled = $this->is_post_likeable();

  1037.               // Otherwise, check and see if likes are enabled sitewide 
  1038.               $enabled = $this->is_enabled_sitewide();

  1041.           if ( post_password_required() ) 
  1042.               $enabled = false; 
  1043.  
  1044.           if ( in_array( 'get_the_excerpt', (array) $wp_current_filter ) ) { 
  1045.               $enabled = false; 
  1046.           } 
  1047.  
  1048.           // Sharing Setting Overrides **************************************** 
  1049.  
  1050.           // Single post including custom post types 
  1051.           if ( is_single() ) {

  1053.                   $enabled = false;

  1059.                   $enabled = false; 
  1060.               } 
  1061.  
  1062.           // Attachment

  1064.               if ( ! $this->is_attachment_enabled() ) { 
  1065.                   $enabled = false; 
  1066.               } 
  1067.  
  1068.           // All other loops 
  1069.           } elseif ( ! $this->is_index_enabled() ) { 
  1070.               $enabled = false; 
  1071.           } 
  1072.       } 
  1073.  
  1074.       if( is_object( $post ) ) { 
  1075.           // Check that the post is a public, published post. 
  1076.           if ( 'attachment' == $post->post_type ) { 
  1077.               $post_status = get_post_status( $post->post_parent ); 
  1078.           } else { 
  1079.               $post_status = $post->post_status; 
  1080.           } 
  1081.           if ( 'publish' != $post_status ) { 
  1082.               $enabled = false; 
  1083.           } 
  1084.       } 
  1085.  
  1086.       // Run through the sharing filters 
  1087.       /** This filter is documented in modules/sharedaddy/sharing-service.php */ 
  1088.       $enabled = apply_filters( 'sharing_show', $enabled, $post );
