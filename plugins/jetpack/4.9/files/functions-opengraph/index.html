  31.  
  32.   $og_output = "\n<!-- Jetpack Open Graph Tags -->\n";

  35.   /** 
  36.    * Filter the minimum width of the images used in Jetpack Open Graph Meta Tags.

  38.    * @module sharedaddy, publicize 
  39.    * 
  40.    * @since 2.0.0

  52.    * @param int 200 Minimum image height used in Jetpack Open Graph Meta Tags.

  59.       $tags['og:type'] = ! empty( $site_type ) ? $site_type : 'website';

  70.       $facebook_admins = get_option( 'facebook_admins' ); 
  71.       if ( ! empty( $facebook_admins ) )

  76.  
  77.       $author = get_queried_object(); 
  78.

  80.       if ( ! empty( $author->user_url ) ) { 
  81.           $tags['og:url'] = $author->user_url; 
  82.       } else { 
  83.           $tags['og:url'] = get_author_posts_url( $author->ID ); 
  84.       } 
  85.       $tags['og:description'] = $author->description; 
  86.       $tags['profile:first_name'] = get_the_author_meta( 'first_name', $author->ID );

  88.  
  89.   } else if ( is_singular() ) { 
  90.       global $post; 
  91.       $data = $post; // so that we don't accidentally explode the global 
  92.  
  93.       $tags['og:type'] = 'article'; 
  94.       if ( empty( $data->post_title ) ) { 
  95.           $tags['og:title'] = ' '; 
  96.       } else { 
  97.           /** This filter is documented in core/src/wp-includes/post-template.php */ 
  98.           $tags['og:title'] = wp_kses( apply_filters( 'the_title', $data->post_title, $data->ID ), array() ); 
  99.       } 
  100.  
  101.       $tags['og:url'] = get_permalink( $data->ID ); 
  102.       if ( ! post_password_required() ) { 
  103.           if ( ! empty( $data->post_excerpt ) ) { 
  104.               $tags['og:description'] = preg_replace( '@https?://[\S]+@', '', strip_shortcodes( wp_kses( $data->post_excerpt, array() ) ) ); 
  105.           } else { 
  106.               $exploded_content_on_more_tag = explode( '<!--more-->', $data->post_content ); 
  107.               $tags['og:description'] = wp_trim_words( preg_replace( '@https?://[\S]+@', '', strip_shortcodes( wp_kses( $exploded_content_on_more_tag[0], array() ) ) ) ); 
  108.           } 
  109.       } 
  110.       if ( empty( $tags['og:description'] ) ) { 
  111.           $tags['og:description'] = __('Visit the post for more.', 'jetpack'); 
  112.       } else { 
  113.           // Intentionally not using a filter to prevent pollution. @see https://github.com/Automattic/jetpack/pull/2899#issuecomment-151957382 
  114.           $tags['og:description'] = wp_kses( trim( convert_chars( wptexturize( $tags['og:description'] ) ) ), array() ); 
  115.       } 
  116.  
  117.       $tags['article:published_time'] = date( 'c', strtotime( $data->post_date_gmt ) ); 
  118.       $tags['article:modified_time'] = date( 'c', strtotime( $data->post_modified_gmt ) ); 
  119.       if ( post_type_supports( get_post_type( $data ), 'author' ) && isset( $data->post_author ) ) { 
  120.           $publicize_facebook_user = get_post_meta( $data->ID, '_publicize_facebook_user', true ); 
  121.           if ( ! empty( $publicize_facebook_user ) ) { 
  122.               $tags['article:author'] = esc_url( $publicize_facebook_user ); 
  123.           } 
  124.       } 
  125.   }

  127.   /** 
  128.    * Allow plugins to inject additional template-specific Open Graph tags. 
  129.    * 
  130.    * @module sharedaddy, publicize 
  131.    * 
  132.    * @since 3.0.0 
  133.    * 
  134.    * @param array $tags Array of Open Graph Meta tags. 
  135.    * @param array $args Array of image size parameters. 
  136.    */ 
  137.   $tags = apply_filters( 'jetpack_open_graph_base_tags', $tags, compact( 'image_width', 'image_height' ) ); 
  138.  
  139.   // Re-enable widont if we had disabled it 
  140.   if ( $disable_widont ) 
  141.       add_filter( 'the_title', 'widont' ); 
  142.  
  143.   /** 
  144.    * Do not return any Open Graph Meta tags if we don't have any info about a post. 
  145.    * 
  146.    * @module sharedaddy, publicize 
  147.    * 
  148.    * @since 3.0.0 
  149.    * 
  150.    * @param bool true Do not return any Open Graph Meta tags if we don't have any info about a post. 
  151.    */ 
  152.   if ( empty( $tags ) && apply_filters( 'jetpack_open_graph_return_if_empty', true ) ) 
  153.       return; 
  154.  
  155.   $tags['og:site_name'] = get_bloginfo( 'name' ); 
  156.  
  157.   // Get image info and build tags 
  158.   if ( ! post_password_required() ) { 
  159.       $image_info = jetpack_og_get_image( $image_width, $image_height ); 
  160.       $tags['og:image'] = $image_info['src']; 
  161.  
  162.       if ( ! empty( $image_info['width'] ) ) { 
  163.           $tags['og:image:width'] = $image_info['width']; 
  164.       } 
  165.       if ( ! empty( $image_info['height'] ) ) { 
  166.           $tags['og:image:height'] = $image_info['height']; 
  167.       } 
  168.   } 
  169.  
  170.   // Facebook whines if you give it an empty title 
  171.   if ( empty( $tags['og:title'] ) ) 
  172.       $tags['og:title'] = __( '(no title)', 'jetpack' ); 
  173.  
  174.   // Shorten the description if it's too long 
  175.   if ( isset( $tags['og:description'] ) ) { 
  176.       $tags['og:description'] = strlen( $tags['og:description'] ) > $description_length ? mb_substr( $tags['og:description'], 0, $description_length ) . '*' : $tags['og:description']; 
  177.   } 
  178.  
  179.   // Try to add OG locale tag if the WP->FB data mapping exists 
  180.   if ( defined( 'JETPACK__GLOTPRESS_LOCALES_PATH' ) && file_exists( JETPACK__GLOTPRESS_LOCALES_PATH ) ) { 
  181.       require_once JETPACK__GLOTPRESS_LOCALES_PATH; 
  182.       $_locale = get_locale(); 
  183.  
  184.       // We have to account for WP.org vs WP.com locale divergence 
  185.       if ( defined( 'IS_WPCOM' ) && IS_WPCOM ) { 
  186.           $gp_locale = GP_Locales::by_field( 'slug', $_locale ); 
  187.       } else { 
  188.           $gp_locale = GP_Locales::by_field( 'wp_locale', $_locale ); 
  189.       } 
  190.   } 
  191.  
  192.   if ( isset( $gp_locale->facebook_locale ) && ! empty( $gp_locale->facebook_locale ) ) { 
  193.       $tags['og:locale'] = $gp_locale->facebook_locale; 
  194.   } 
  195.  
  196.   /** 
  197.    * Allow the addition of additional Open Graph Meta tags, or modify the existing tags. 
  198.    * 
  199.    * @module sharedaddy, publicize 
  200.    * 
  201.    * @since 2.0.0 
  202.    * 
  203.    * @param array $tags Array of Open Graph Meta tags. 
  204.    * @param array $args Array of image size parameters. 
  205.    */ 
  206.   $tags = apply_filters( 'jetpack_open_graph_tags', $tags, compact( 'image_width', 'image_height' ) ); 
  207.  
  208.   // secure_urls need to go right after each og:image to work properly so we will abstract them here 
  209.   $secure = $tags['og:image:secure_url'] = ( empty( $tags['og:image:secure_url'] ) ) ? '' : $tags['og:image:secure_url']; 
  210.   unset( $tags['og:image:secure_url'] ); 
  211.   $secure_image_num = 0; 
  212.  
  213.   foreach ( (array) $tags as $tag_property => $tag_content ) { 
  214.       // to accommodate multiple images 
  215.       $tag_content = (array) $tag_content; 
  216.       $tag_content = array_unique( $tag_content ); 
  217.  
  218.       foreach ( $tag_content as $tag_content_single ) { 
  219.           if ( empty( $tag_content_single ) ) 
  220.               continue; // Don't ever output empty tags 
  221.           $og_tag = sprintf( '<meta property="%s" content="%s" />', esc_attr( $tag_property ), esc_attr( $tag_content_single ) ); 
  222.           /** 
  223.            * Filter the HTML Output of each Open Graph Meta tag. 
  224.            * 
  225.            * @module sharedaddy, publicize 
  226.            * 
  227.            * @since 2.0.0 
  228.            * 
  229.            * @param string $og_tag HTML HTML Output of each Open Graph Meta tag. 
  230.            */ 
  231.           $og_output .= apply_filters( 'jetpack_open_graph_output', $og_tag ); 
  232.           $og_output .= "\n"; 
  233.  
  234.           if ( 'og:image' == $tag_property ) { 
  235.               if ( is_array( $secure ) && !empty( $secure[$secure_image_num] ) ) { 
  236.                   $og_tag = sprintf( '<meta property="og:image:secure_url" content="%s" />', esc_url( $secure[ $secure_image_num ] ) ); 
  237.                   /** This filter is documented in functions.opengraph.php */ 
  238.                   $og_output .= apply_filters( 'jetpack_open_graph_output', $og_tag ); 
  239.                   $og_output .= "\n"; 
  240.               } else if ( !is_array( $secure ) && !empty( $secure ) ) { 
  241.                   $og_tag = sprintf( '<meta property="og:image:secure_url" content="%s" />', esc_url( $secure ) ); 
  242.                   /** This filter is documented in functions.opengraph.php */ 
  243.                   $og_output .= apply_filters( 'jetpack_open_graph_output', $og_tag ); 
  244.                   $og_output .= "\n"; 
  245.               } 
  246.               $secure_image_num++; 
  247.           } 
  248.       } 
  249.   } 
  250.   echo $og_output; 
  251. }

  253. function jetpack_og_get_image( $width = 200, $height = 200, $max_images = 4 ) { // Facebook requires thumbnails to be a minimum of 200x200 
  254.   $image = ''; 
  255.  
  256.   if ( is_singular() && ! is_home() ) { 
  257.       global $post; 
  258.       $image = ''; 
  259.  
  260.       // Grab obvious image if $post is an attachment page for an image 
  261.       if ( is_attachment( $post->ID ) && 'image' == substr( $post->post_mime_type, 0, 5 ) ) { 
  262.           $image = wp_get_attachment_url( $post->ID ); 
  263.       } 
  264.  
  265.       // Attempt to find something good for this post using our generalized PostImages code 
  266.       if ( ! $image && class_exists( 'Jetpack_PostImages' ) ) { 
  267.           $post_images = Jetpack_PostImages::get_images( $post->ID, array( 'width' => $width, 'height' => $height ) ); 
  268.           if ( $post_images && ! is_wp_error( $post_images ) ) { 
  269.               $image = array(); 
  270.               foreach ( (array) $post_images as $post_image ) { 
  271.                   $image['src'] = $post_image['src']; 
  272.                   if ( isset( $post_image['src_width'], $post_image['src_height'] ) ) { 
  273.                       $image['width'] = $post_image['src_width']; 
  274.                       $image['height'] = $post_image['src_height']; 
  275.                   } 
  276.               } 
  277.           } 
  278.       } 
  279.   } else if ( is_author() ) { 
  280.       $author = get_queried_object(); 
  281.       if ( function_exists( 'get_avatar_url' ) ) { 
  282.           // Prefer the core function get_avatar_url() if available, WP 4.2+ 
  283.           $image['src'] = get_avatar_url( $author->user_email, array( 'size' => $width ) ); 
  284.       } 
  285.       else { 
  286.           $has_filter = has_filter( 'pre_option_show_avatars', '__return_true' ); 
  287.           if ( ! $has_filter ) { 
  288.               add_filter( 'pre_option_show_avatars', '__return_true' ); 
  289.           } 
  290.           $avatar = get_avatar( $author->user_email, $width ); 
  291.           if ( ! $has_filter ) { 
  292.               remove_filter( 'pre_option_show_avatars', '__return_true' ); 
  293.           } 
  294.  
  295.           if ( ! empty( $avatar ) && ! is_wp_error( $avatar ) ) { 
  296.               if ( preg_match( '/src=["\']([^"\']+)["\']/', $avatar, $matches ) ); 
  297.                   $image['src'] = wp_specialchars_decode( $matches[1], ENT_QUOTES ); 
  298.           } 
  299.       } 
  300.   } 
  301.  
  302.   if ( empty( $image ) ) { 
  303.       $image = array(); 
  304.   } else if ( ! is_array( $image ) ) { 
  305.       $image = array( 
  306.           'src' => $image 
  307.  ); 
  308.   } 
  309.  
  310.   // First fall back, blavatar 
  311.   if ( empty( $image ) && function_exists( 'blavatar_domain' ) ) { 
  312.       $blavatar_domain = blavatar_domain( site_url() ); 
  313.       if ( blavatar_exists( $blavatar_domain ) ) { 
  314.           $image['src'] = blavatar_url( $blavatar_domain, 'img', $width, false, true ); 
  315.           $image['width'] = $width; 
  316.           $image['height'] = $height; 
  317.       } 
  318.   } 
  319.  
  320.   // Second fall back, Site Logo 
  321.   if ( empty( $image ) && ( function_exists( 'jetpack_has_site_logo' ) && jetpack_has_site_logo() ) ) { 
  322.       $image['src'] = jetpack_get_site_logo( 'url' ); 
  323.       $image_dimensions = jetpack_get_site_logo_dimensions(); 
  324.       if ( ! empty( $image_dimensions ) ) { 
  325.           $image['width'] = $image_dimensions['width']; 
  326.           $image['height'] = $image_dimensions['height']; 
  327.       } 
  328.   } 
  329.  
  330.   // Third fall back, Site Icon 
  331.   if ( empty( $image ) && ( function_exists( 'jetpack_has_site_icon' ) && jetpack_has_site_icon() ) ) { 
  332.       $image['src'] = jetpack_site_icon_url( null, '512' ); 
  333.       $image['width'] = '512'; 
  334.       $image['height'] = '512'; 
  335.   } 
  336.  
  337.   // Fourth fall back, Core Site Icon. Added in WP 4.3. 
  338.   if ( empty( $image ) && ( function_exists( 'has_site_icon') && has_site_icon() ) ) { 
  339.       $image['src'] = get_site_icon_url( null, '512' ); 
  340.   } 
  341.  
  342.   // Finally fall back, blank image 
  343.   if ( empty( $image ) ) { 
  344.       /** 
  345.        * Filter the default Open Graph Image tag, used when no Image can be found in a post. 
  346.        * 
  347.        * @since 3.0.0 
  348.        * 
  349.        * @param string $str Default Image URL. 
  350.        */ 
  351.       $image['src'] = apply_filters( 'jetpack_open_graph_image_default', 'https://s0.wp.com/i/blank.jpg' ); 
  352.   } 
  353.  
  354.   return $image; 
  355. }

  372.   } else { 
  373.       $has_filter = has_filter( 'pre_option_show_avatars', '__return_true' ); 
  374.       if ( !$has_filter ) {

  376.       } 
  377.       $avatar = get_avatar( $email, $width ); 
  378.  
  379.       if ( !$has_filter ) { 
  380.           remove_filter( 'pre_option_show_avatars', '__return_true' );

  383.       if ( !empty( $avatar ) && !is_wp_error( $avatar ) ) {
