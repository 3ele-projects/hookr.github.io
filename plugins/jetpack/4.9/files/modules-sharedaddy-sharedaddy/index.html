  24. function sharing_email_check_for_spam_via_akismet( $data ) { 
  25.  
  26.   if ( ! function_exists( 'akismet_http_post' ) && ! method_exists( 'Akismet', 'http_post' ) ) 
  27.       return $data; 
  28.  
  29.   // Prepare the body_request for akismet 
  30.   $body_request = array( 
  31.       'blog' => get_option( 'home' ),  
  32.       'permalink' => get_permalink( $data['post']->ID ),  
  33.       'comment_type' => 'share',  
  34.       'comment_author' => $data['name'],  
  35.       'comment_author_email' => $data['source'],  
  36.       'comment_content' => sharing_email_send_post_content( $data ),  
  37.       'user_agent' => ( isset( $_SERVER['HTTP_USER_AGENT'] ) ? $_SERVER['HTTP_USER_AGENT'] : null ),  
  38.  ); 
  39.  
  40.   if ( method_exists( 'Akismet', 'http_post' ) ) { 
  41.       $body_request['user_ip'] = Akismet::get_ip_address(); 
  42.       $response = Akismet::http_post( build_query( $body_request ), 'comment-check' ); 
  43.   } else { 
  44.       global $akismet_api_host, $akismet_api_port; 
  45.       $body_request['user_ip'] = ( isset( $_SERVER['REMOTE_ADDR'] ) ? $_SERVER['REMOTE_ADDR'] : null ); 
  46.       $response = akismet_http_post( build_query( $body_request ), $akismet_api_host, '/1.1/comment-check', $akismet_api_port ); 
  47.   } 
  48.  
  49.   // The Response is spam lets not send the email. 
  50.   if ( ! empty( $response ) && isset( $response[1] ) && 'true' == trim( $response[1] ) ) { // 'true' is spam 
  51.       return false; // don't send the email 
  52.   } 
  53.   return $data; 
  54. }
