  143.   public function header_text_classes() {

  145.  
  146.       if ( isset( $args[0][ 'header-text' ] ) ) { 
  147.           // Use any classes defined in add_theme_support(). 
  148.           $classes = $args[0][ 'header-text' ]; 
  149.       } else { 
  150.           // Otherwise, use these defaults, which will work with any Underscores-based theme. 
  151.           $classes = array( 
  152.               'site-title',  
  153.               'site-description',  
  154.  ); 
  155.       } 
  156.  
  157.       // If we've got an array, reduce them to a string for output 
  158.       if ( is_array( $classes ) ) { 
  159.           $classes = (string) '.' . implode( ', .', $classes ); 
  160.       } else { 
  161.           $classes = (string) '.' . $classes; 
  162.       } 
  163.  
  164.       return $classes;
