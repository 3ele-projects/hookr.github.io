  501.   /** 
  502.    * Our [testimonial] shortcode. 
  503.    * Prints Testimonial data styled to look good on *any* theme. 
  504.    * 
  505.    * @return jetpack_testimonial_shortcode_html 
  506.    */ 
  507.   static function jetpack_testimonial_shortcode( $atts ) { 
  508.       // Default attributes 
  509.       $atts = shortcode_atts( array( 
  510.           'display_content' => true,  
  511.           'image' => true,  
  512.           'columns' => 1,  
  513.           'showposts' => -1,  
  514.           'order' => 'asc',  
  515.           'orderby' => 'date',  
  516.  ), $atts, 'testimonial' ); 
  517.  
  518.       // A little sanitization 
  519.       if ( $atts['display_content'] && 'true' != $atts['display_content'] && 'full' != $atts['display_content'] ) { 
  520.           $atts['display_content'] = false; 
  521.       } 
  522.  
  523.       if ( $atts['image'] && 'true' != $atts['image'] ) { 
  524.           $atts['image'] = false; 
  525.       } 
  526.  
  527.       $atts['columns'] = absint( $atts['columns'] ); 
  528.  
  529.       $atts['showposts'] = intval( $atts['showposts'] ); 
  530.  
  531.       if ( $atts['order'] ) { 
  532.           $atts['order'] = urldecode( $atts['order'] ); 
  533.           $atts['order'] = strtoupper( $atts['order'] ); 
  534.           if ( 'DESC' != $atts['order'] ) { 
  535.               $atts['order'] = 'ASC'; 
  536.           } 
  537.       } 
  538.  
  539.       if ( $atts['orderby'] ) { 
  540.           $atts['orderby'] = urldecode( $atts['orderby'] ); 
  541.           $atts['orderby'] = strtolower( $atts['orderby'] ); 
  542.           $allowed_keys = array('author', 'date', 'title', 'rand'); 
  543.  
  544.           $parsed = array(); 
  545.           foreach ( explode( ', ', $atts['orderby'] ) as $testimonial_index_number => $orderby ) { 
  546.               if ( ! in_array( $orderby, $allowed_keys ) ) { 
  547.                   continue; 
  548.               } 
  549.               $parsed[] = $orderby; 
  550.           } 
  551.  
  552.           if ( empty( $parsed ) ) { 
  553.               unset($atts['orderby']); 
  554.           } else { 
  555.               $atts['orderby'] = implode( ' ', $parsed ); 
  556.           } 
  557.       } 
  558.  
  559.       // enqueue shortcode styles when shortcode is used 
  560.       wp_enqueue_style( 'jetpack-testimonial-style', plugins_url( 'css/testimonial-shortcode.css', __FILE__ ), array(), '20140326' ); 
  561.  
  562.       return self::jetpack_testimonial_shortcode_html( $atts ); 
  563.   }

  565.   /** 
  566.    * The Testimonial shortcode loop. 
  567.    * 
  568.    * @return html 
  569.    */ 
  570.   static function jetpack_testimonial_shortcode_html( $atts ) { 
  571.       // Default query arguments 
  572.       $defaults = array( 
  573.           'order' => $atts['order'],  
  574.           'orderby' => $atts['orderby'],  
  575.           'posts_per_page' => $atts['showposts'],  
  576.  ); 
  577.  
  578.       $args = wp_parse_args( $atts, $defaults ); 
  579.       $args['post_type'] = self::CUSTOM_POST_TYPE; // Force this post type 
  580.       $query = new WP_Query( $args ); 
  581.  
  582.       $testimonial_index_number = 0; 
  583.  
  584.       ob_start(); 
  585.  
  586.       // If we have testimonials, create the html 
  587.       if ( $query->have_posts() ) { 
  588.  
  589.           ?> 
  590.           <div class="jetpack-testimonial-shortcode column-<?php echo esc_attr( $atts['columns'] ); ?>"> 
  591.               <?php  // open .jetpack-testimonial-shortcode 
  592.  
  593.               // Construct the loop... 
  594.               while ( $query->have_posts() ) { 
  595.                   $query->the_post(); 
  596.                   $post_id = get_the_ID(); 
  597.                   ?> 
  598.                   <div class="testimonial-entry <?php echo esc_attr( self::get_testimonial_class( $testimonial_index_number, $atts['columns'] ) ); ?>"> 
  599.                       <?php 
  600.                       // The content 
  601.                       if ( false !== $atts['display_content'] ) { 
  602.                           if ( 'full' === $atts['display_content'] ) { 
  603.                               echo '<div class="testimonial-entry-content">' . the_content() . '</div>'; 
  604.                           } else { 
  605.                               echo '<div class="testimonial-entry-content">' . the_excerpt() . '</div>'; 
  606.                           } 
  607.                       } 
  608.                       ?> 
  609.                       <span class="testimonial-entry-title">― <a href="<?php echo esc_url( get_permalink() ); ?>" title="<?php echo esc_attr( the_title_attribute( ) ); ?>"><?php the_title(); ?></a></span> 
  610.                       <?php 
  611.                       // Featured image 
  612.                       if ( false !== $atts['image'] ) : 
  613.                           echo self::get_testimonial_thumbnail_link( $post_id ); 
  614.                       endif; 
  615.                       ?> 
  616.                   </div><!-- close .testimonial-entry --> 
  617.                   <?php 
  618.                   $testimonial_index_number++; 
  619.               } // end of while loop 
  620.  
  621.               wp_reset_postdata(); 
  622.               ?> 
  623.           </div><!-- close .jetpack-testimonial-shortcode --> 
  624.       <?php 
  625.       } else { ?> 
  626.           <p><em><?php _e( 'Your Testimonial Archive currently has no entries. You can start creating them on your dashboard.', 'jetpack' ); ?></p></em> 
  627.       <?php 
  628.       } 
  629.       $html = ob_get_clean(); 
  630.  
  631.       // Return the HTML block 
  632.       return $html; 
  633.   }

  636.    * Individual testimonial class 
  637.    * 
  638.    * @return string 
  639.    */ 
  640.   static function get_testimonial_class( $testimonial_index_number, $columns ) { 
  641.       $class = array(); 
  642.  
  643.       $class[] = 'testimonial-entry-column-'.$columns; 
  644.  
  645.       if( $columns > 1) { 
  646.           if ( ( $testimonial_index_number % 2 ) == 0 ) { 
  647.               $class[] = 'testimonial-entry-mobile-first-item-row'; 
  648.           } else { 
  649.               $class[] = 'testimonial-entry-mobile-last-item-row'; 
  650.           } 
  651.       } 
  652.  
  653.       // add first and last classes to first and last items in a row 
  654.       if ( ( $testimonial_index_number % $columns ) == 0 ) { 
  655.           $class[] = 'testimonial-entry-first-item-row'; 
  656.       } elseif ( ( $testimonial_index_number % $columns ) == ( $columns - 1 ) ) { 
  657.           $class[] = 'testimonial-entry-last-item-row'; 
  658.       } 
  659.  
  660.  
  661.       /** 
  662.        * Filter the class applied to testimonial div in the testimonial 
  663.        * 
  664.        * @module custom-content-types 
  665.        * 
  666.        * @since 3.4.0 
  667.        * 
  668.        * @param string $class class name of the div. 
  669.        * @param int $testimonial_index_number iterator count the number of columns up starting from 0. 
  670.        * @param int $columns number of columns to display the content in. 
  671.        * 
  672.        */ 
  673.       return apply_filters( 'testimonial-entry-post-class', implode( " ", $class ) , $testimonial_index_number, $columns ); 
  674.   }
