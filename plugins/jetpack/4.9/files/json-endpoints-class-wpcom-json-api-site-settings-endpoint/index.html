  201.    * Updates site settings for authorized users

  205.   public function update_settings() { 
  206.  
  207.       // $this->input() retrieves posted arguments whitelisted and casted to the $request_format 
  208.       // specs that get passed in when this class is instantiated 
  209.       /** 
  210.        * Filters the settings to be updated on the site. 
  211.        * 
  212.        * @module json-api 
  213.        * 
  214.        * @since 3.6.0 
  215.        * 
  216.        * @param array $input Associative array of site settings to be updated. 
  217.        */ 
  218.       $input = apply_filters( 'rest_api_update_site_settings', $this->input() ); 
  219.  
  220.       $jetpack_relatedposts_options = array(); 
  221.       $sharing_options = array(); 
  222.       $updated = array(); 
  223.  
  224.       foreach ( $input as $key => $value ) { 
  225.  
  226.           if ( ! is_array( $value ) ) { 
  227.               $value = trim( $value ); 
  228.           } 
  229.           $value = wp_unslash( $value ); 
  230.  
  231.           switch ( $key ) { 
  232.  
  233.               case 'default_ping_status': 
  234.               case 'default_comment_status': 
  235.                   // settings are stored as closed|open 
  236.                   $coerce_value = ( $value ) ? 'open' : 'closed'; 
  237.                   if ( update_option( $key, $coerce_value ) ) { 
  238.                       $updated[ $key ] = $value; 
  239.                   }; 
  240.                   break; 
  241.               case 'jetpack_protect_whitelist': 
  242.                   if ( function_exists( 'jetpack_protect_save_whitelist' ) ) { 
  243.                       $result = jetpack_protect_save_whitelist( $value ); 
  244.                       if ( is_wp_error( $result ) ) { 
  245.                           return $result; 
  246.                       } 
  247.                       $updated[ $key ] = jetpack_protect_format_whitelist(); 
  248.                   } 
  249.                   break; 
  250.               case 'jetpack_sync_non_public_post_stati': 
  251.                   Jetpack_Options::update_option( 'sync_non_public_post_stati', $value ); 
  252.                   break; 
  253.               case 'jetpack_relatedposts_enabled': 
  254.               case 'jetpack_relatedposts_show_thumbnails': 
  255.               case 'jetpack_relatedposts_show_headline': 
  256.                   if ( ! $this->jetpack_relatedposts_supported() ) { 
  257.                       break; 
  258.                   } 
  259.                   if ( 'jetpack_relatedposts_enabled' === $key && method_exists( 'Jetpack', 'is_module_active' ) && $this->jetpack_relatedposts_supported() ) { 
  260.                       $before_action = Jetpack::is_module_active('related-posts'); 
  261.                       if ( $value ) { 
  262.                           Jetpack::activate_module( 'related-posts', false, false ); 
  263.                       } else { 
  264.                           Jetpack::deactivate_module( 'related-posts' ); 
  265.                       } 
  266.                       $after_action = Jetpack::is_module_active('related-posts');

  268.                           break; 
  269.                       } 
  270.                   } 
  271.                   $just_the_key = substr( $key, 21 ); 
  272.                   $jetpack_relatedposts_options[ $just_the_key ] = $value; 
  273.               break; 
  274.  
  275.               case 'social_notifications_like': 
  276.               case 'social_notifications_reblog': 
  277.               case 'social_notifications_subscribe': 
  278.                   // settings are stored as on|off 
  279.                   $coerce_value = ( $value ) ? 'on' : 'off'; 
  280.                   if ( update_option( $key, $coerce_value ) ) { 
  281.                       $updated[ $key ] = $value; 
  282.                   } 
  283.                   break; 
  284.               case 'wga': 
  285.                   if ( ! isset( $value['code'] ) || ! preg_match( '/^$|^UA-[\d-]+$/i', $value['code'] ) ) { 
  286.                       return new WP_Error( 'invalid_code', 'Invalid UA ID' ); 
  287.                   }

  290.                   if ( update_option( 'wga', $wga ) ) { 
  291.                       $updated[ $key ] = $value; 
  292.                   }

  321.               // Keyring token option

  331.                       } 
  332.                   }

  335.               // no worries, we've already whitelisted and casted arguments above

  341.           }

  343.
