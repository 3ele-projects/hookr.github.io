  148.   public function filter_infinite_scroll_js_settings( $settings ) { 
  149.       // Provide WP Stats info for tracking Infinite Scroll loads 
  150.       // Abort if Stats module isn't active 
  151.       if ( in_array( 'stats', Jetpack::get_active_modules() ) ) { 
  152.           // Abort if user is logged in but logged-in users shouldn't be tracked. 
  153.           if ( is_user_logged_in() && function_exists( 'stats_get_options' ) ) { 
  154.               $stats_options = stats_get_options(); 
  155.               $track_loggedin_users = isset( $stats_options['reg_users'] ) ? (bool) $stats_options['reg_users'] : false; 
  156.  
  157.               if ( ! $track_loggedin_users ) 
  158.                   return $settings; 
  159.           } 
  160.  
  161.           // We made it this far, so gather the data needed to track IS views 
  162.           $settings['stats'] = 'blog=' . Jetpack_Options::get_option( 'id' ) . '&host=' . parse_url( get_option( 'home' ), PHP_URL_HOST ) . '&v=ext&j=' . JETPACK__API_VERSION . ':' . JETPACK__VERSION; 
  163.  
  164.           // Pagetype parameter 
  165.           $settings['stats'] .= '&x_pagetype=infinite'; 
  166.           if ( 'click' == $settings['type'] ) 
  167.               $settings['stats'] .= '-click'; 
  168.  
  169.           $settings['stats'] .= '-jetpack'; 
  170.       } 
  171.  
  172.       // Check if Google Analytics tracking is requested 
  173.       $settings['google_analytics'] = (bool) get_option( $this->option_name_google_analytics ); 
  174.  
  175.       return $settings; 
  176.   }
