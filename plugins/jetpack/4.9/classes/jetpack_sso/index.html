  547.  
  548.         if ( $xml->isError() ) { 
  549.             wp_die( sprintf( '%s: %s', $xml->getErrorCode(), $xml->getErrorMessage() ) ); 
  550.         } 
  551.  
  552.         $user_data = $xml->getResponse(); 
  553.  
  554.         if ( empty( $user_data ) ) { 
  555.             wp_die( __( 'Error, invalid response data.', 'jetpack' ) ); 
  556.         } 
  557.  
  558.         $user_data = (object) $user_data; 
  559.         $user = null; 
  560.  
  561.         /** 
  562.          * Fires before Jetpack's SSO modifies the log in form. 
  563.          * 
  564.          * @module sso 
  565.          * 
  566.          * @since 2.6.0 
  567.          * 
  568.          * @param object $user_data User login information. 
  569.          */ 
  570.         do_action( 'jetpack_sso_pre_handle_login', $user_data ); 
  571.  
  572.         /** 
  573.          * Is it required to have 2-step authentication enabled on WordPress.com to use SSO? 
  574.          * 
  575.          * @module sso 
  576.          * 
  577.          * @since 2.8.0 
  578.          * 
  579.          * @param bool get_option( 'jetpack_sso_require_two_step' ) Does SSO require 2-step authentication? 
  580.          */ 
  581.         $require_two_step = apply_filters( 'jetpack_sso_require_two_step', get_option( 'jetpack_sso_require_two_step' ) ); 
  582.         if( $require_two_step && 0 == (int) $user_data->two_step_enabled ) { 
  583.             $this->user_data = $user_data; 
  584.             /** This filter is documented in core/src/wp-includes/pluggable.php */ 
  585.             do_action( 'wp_login_failed', $user_data->login ); 
  586.             add_action( 'login_message', array( $this, 'error_msg_enable_two_step' ) ); 
  587.             return; 
  588.         } 
  589.  
  590.         if ( isset( $_GET['state'] ) && ( 0 < strpos( $_GET['state'], '|' ) ) ) { 
  591.             list( $state, $nonce ) = explode( '|', $_GET['state'] ); 
  592.  
  593.             if ( wp_verify_nonce( $nonce, $state ) ) { 
  594.                 if ( 'sso-link-user' == $state ) { 
  595.                     $user = wp_get_current_user(); 
  596.                     update_user_meta( $user->ID, 'wpcom_user_id', $user_data->ID ); 
  597.                     add_filter( 'login_redirect', array( __CLASS__, 'profile_page_url' ) ); 
  598.                 } 
  599.             } else wp_nonce_ays(); 
  600.         } 
  601.  
  602.         if ( empty( $user ) ) { 
  603.             $user = $this->get_user_by_wpcom_id( $user_data->ID ); 
  604.         } 
  605.  
  606.         // If we don't have one by wpcom_user_id, try by the email? 
  607.         if ( empty( $user ) && self::match_by_email() ) { 
  608.             $user = get_user_by( 'email', $user_data->email ); 
  609.             if ( $user ) { 
  610.                 update_user_meta( $user->ID, 'wpcom_user_id', $user_data->ID ); 
  611.             } 
  612.         } 
  613.  
  614.         // If we've still got nothing, create the user. 
  615.         if ( empty( $user ) && ( get_option( 'users_can_register' ) || self::new_user_override() ) ) { 
  616.             // If not matching by email we still need to verify the email does not exist 
  617.             // or this blows up 
  618.             /** 
  619.              * If match_by_email is true, we know the email doesn't exist, as it would have 
  620.              * been found in the first pass.  If get_user_by( 'email' ) doesn't find the 
  621.              * user, then we know that email is unused, so it's safe to add. 
  622.              */ 
  623.             if ( self::match_by_email() || ! get_user_by( 'email', $user_data->email ) ) { 
  624.                 $username = $user_data->login; 
  625.  
  626.                 if ( username_exists( $username ) ) { 
  627.                     $username = $user_data->login . '_' . $user_data->ID; 
  628.                 } 
  629.  
  630.                 $tries = 0; 
  631.                 while ( username_exists( $username ) ) { 
  632.                     $username = $user_data->login . '_' . $user_data->ID . '_' . mt_rand(); 
  633.                     if ( $tries++ >= 5 ) { 
  634.                         wp_die( __( "Error: Couldn't create suitable username.", 'jetpack' ) ); 
  635.                     } 
  636.                 } 
  637.  
  638.                 $password = wp_generate_password( 20 ); 
  639.                 $user_id = wp_create_user( $username, $password, $user_data->email ); 
  640.                 $user = get_userdata( $user_id ); 
  641.  
  642.                 $user->display_name = $user_data->display_name; 
  643.                 $user->first_name = $user_data->first_name; 
  644.                 $user->last_name = $user_data->last_name; 
  645.                 $user->url = $user_data->url; 
  646.                 $user->description = $user_data->description; 
  647.                 wp_update_user( $user ); 
  648.  
  649.                 update_user_meta( $user->ID, 'wpcom_user_id', $user_data->ID ); 
  650.             } else { 
  651.                 $this->user_data = $user_data; 
  652.                 // do_action( 'wp_login_failed', $user_data->login ); 
  653.                 add_action( 'login_message', array( $this, 'error_msg_email_already_exists' ) ); 
  654.                 return; 
  655.             } 
  656.         } 
  657.  
  658.         /** 
  659.          * Fires after we got login information from WordPress.com. 
  660.          * 
  661.          * @module sso 
  662.          * 
  663.          * @since 2.6.0 
  664.          * 
  665.          * @param array $user WordPress.com User information. 
  666.          * @param object $user_data User Login information. 
  667.          */ 
  668.         do_action( 'jetpack_sso_handle_login', $user, $user_data ); 
  669.  
  670.         if ( $user ) { 
  671.             // Cache the user's details, so we can present it back to them on their user screen. 
  672.             update_user_meta( $user->ID, 'wpcom_user_data', $user_data ); 
  673.  
  674.             $remember = false; 
  675.             if ( ! empty( $_COOKIE['jetpack_sso_remember_me'] ) ) { 
  676.                 $remember = true; 
  677.                 // And then purge it 
  678.                 setcookie( 'jetpack_sso_remember_me', ' ', time() - YEAR_IN_SECONDS, COOKIEPATH, COOKIE_DOMAIN ); 
  679.             } 
  680.             /** 
  681.              * Filter the remember me value. 
  682.              * 
  683.              * @module sso 
  684.              * 
  685.              * @since 2.8.0 
  686.              * 
  687.              * @param bool $remember Is the remember me option checked? 
  688.              */ 
  689.             $remember = apply_filters( 'jetpack_remember_login', $remember ); 
  690.             wp_set_auth_cookie( $user->ID, $remember ); 
  691.  
  692.             /** This filter is documented in core/src/wp-includes/user.php */ 
  693.             do_action( 'wp_login', $user->user_login, $user ); 
  694.  
  695.             $_request_redirect_to = isset( $_REQUEST['redirect_to'] ) ? $_REQUEST['redirect_to'] : ''; 
  696.             $redirect_to = user_can( $user, 'edit_posts' ) ? admin_url() : self::profile_page_url(); 
  697.  
  698.             // If we have a saved redirect to request in a cookie 
  699.             if ( ! empty( $_COOKIE['jetpack_sso_redirect_to'] ) ) { 
  700.                 // Set that as the requested redirect to 
  701.                 $redirect_to = $_request_redirect_to = esc_url_raw( $_COOKIE['jetpack_sso_redirect_to'] ); 
  702.                 // And then purge it 
  703.                 setcookie( 'jetpack_sso_redirect_to', ' ', time() - YEAR_IN_SECONDS, COOKIEPATH, COOKIE_DOMAIN ); 
  704.             } 
  705.  
  706.             wp_safe_redirect( 
  707.                 /** This filter is documented in core/src/wp-login.php */ 
  708.                 apply_filters( 'login_redirect', $redirect_to, $_request_redirect_to, $user ) 
  709.  ); 
  710.             exit; 
  711.         } 
  712.  
  713.         $this->user_data = $user_data; 
  714.         /** This filter is documented in core/src/wp-includes/pluggable.php */ 
  715.         do_action( 'wp_login_failed', $user_data->login ); 
  716.         add_action( 'login_message', array( $this, 'cant_find_user' ) ); 
  717.     }
