  395.         // bail early if markdown is disabled or this post type is unsupported. 
  396.         if ( ! $this->is_posting_enabled() || ! post_type_supports( $post_data['post_type'], self::POST_TYPE_SUPPORT ) ) {

  492.             'decode_code_blocks' => ! $this->get_parser()->use_code_shortcode 
  493.  ) ); 
  494.         // probably need to unslash 
  495.         if ( $args['unslash'] ) 
  496.             $text = wp_unslash( $text ); 
  497.  
  498.         /** 
  499.          * Filter the content to be run through Markdown, before it's transformed by Markdown. 
  500.          * 
  501.          * @module markdown 
  502.          * 
  503.          * @since 2.8.0 
  504.          * 
  505.          * @param string $text Content to be run through Markdown 
  506.          * @param array $args Array of Markdown options. 
  507.          */ 
  508.         $text = apply_filters( 'wpcom_markdown_transform_pre', $text, $args ); 
  509.         // ensure our paragraphs are separated 
  510.         $text = str_replace( array( '</p><p>', "</p>\n<p>" ), "</p>\n\n<p>", $text ); 
  511.         // visual editor likes to add <p>s. Buh-bye. 
  512.         $text = $this->get_parser()->unp( $text ); 
  513.         // sometimes we get an encoded > at start of line, breaking blockquotes 
  514.         $text = preg_replace( '/^>/m', '>', $text ); 
  515.         // prefixes are because we need to namespace footnotes by post_id 
  516.         $this->get_parser()->fn_id_prefix = $args['id'] ? $args['id'] . '-' : ''; 
  517.         // If we're not using the code shortcode, prevent over-encoding. 
  518.         if ( $args['decode_code_blocks'] ) { 
  519.             $text = $this->get_parser()->codeblock_restore( $text ); 
  520.         } 
  521.         // Transform it! 
  522.         $text = $this->get_parser()->transform( $text ); 
  523.         // Fix footnotes - kses doesn't like the : IDs it supplies 
  524.         $text = preg_replace( '/((id|href)="#?fn(ref)?):/', "$1-", $text ); 
  525.         // Markdown inserts extra spaces to make itself work. Buh-bye. 
  526.         $text = rtrim( $text ); 
  527.         /** 
  528.          * Filter the content to be run through Markdown, after it was transformed by Markdown. 
  529.          * 
  530.          * @module markdown 
  531.          * 
  532.          * @since 2.8.0 
  533.          * 
  534.          * @param string $text Content to be run through Markdown 
  535.          * @param array $args Array of Markdown options. 
  536.          */ 
  537.         $text = apply_filters( 'wpcom_markdown_transform_post', $text, $args ); 
  538.  
  539.         // probably need to re-slash 
  540.         if ( $args['unslash'] ) 
  541.             $text = wp_slash( $text ); 
  542.  
  543.         return $text; 
  544.     }

  546.     /** 
  547.      * Shows Markdown in the Revisions screen, and ensures that post_content_filtered 
  548.      * is maintained on revisions 
  549.      * @param  array $fields  Post fields pertinent to revisions 
  550.      * @return array          Modified array to include post_content_filtered 
  551.      */ 
  552.     public function _wp_post_revision_fields( $fields ) { 
  553.         $fields['post_content_filtered'] = __( 'Markdown content', 'jetpack' ); 
  554.         return $fields; 
  555.     } 
  556.  
  557.     /** 
  558.      * Do some song and dance to keep all post_content and post_content_filtered content 
  559.      * in the expected place when a post revision is restored. 
  560.      * @param  int $post_id        The post ID have a restore done to it 
  561.      * @param  int $revision_id    The revision ID being restored 
  562.      * @return null 
  563.      */ 
  564.     public function wp_restore_post_revision( $post_id, $revision_id ) { 
  565.         if ( $this->is_markdown( $revision_id ) ) { 
  566.             $revision = get_post( $revision_id, ARRAY_A ); 
  567.             $post = get_post( $post_id, ARRAY_A ); 
  568.             $post['post_content'] = $revision['post_content_filtered']; // Yes, we put it in post_content, because our wp_insert_post_data() expects that 
  569.             // set this flag so we can restore the post_content_filtered on the last revision later 
  570.             $this->monitoring['restore'] = true; 
  571.             // let's not make a revision of our fixing update 
  572.             add_filter( 'wp_revisions_to_keep', '__return_false', 99 ); 
  573.             wp_update_post( $post ); 
  574.             $this->fix_latest_revision_on_restore( $post_id ); 
  575.             remove_filter( 'wp_revisions_to_keep', '__return_false', 99 ); 
  576.         } 
  577.     } 
  578.  
  579.     /** 
  580.      * We need to ensure the last revision has Markdown, not HTML in its post_content_filtered 
  581.      * column after a restore. 
  582.      * @param  int $post_id The post ID that was just restored. 
  583.      * @return null 
  584.      */ 
  585.     protected function fix_latest_revision_on_restore( $post_id ) { 
  586.         global $wpdb; 
  587.         $post = get_post( $post_id ); 
  588.         $last_revision = $wpdb->get_row( $wpdb->prepare( "SELECT * FROM $wpdb->posts WHERE post_type = 'revision' AND post_parent = %d ORDER BY ID DESC", $post->ID ) ); 
  589.         $last_revision->post_content_filtered = $post->post_content_filtered; 
  590.         wp_insert_post( (array) $last_revision ); 
  591.     }

  593.     /** 
  594.      * Kicks off magic for an XML-RPC session. We want to keep editing Markdown 
  595.      * and publishing HTML. 
  596.      * @param  string $xmlrpc_method The current XML-RPC method 
  597.      * @return null 
  598.      */ 
  599.     public function xmlrpc_actions( $xmlrpc_method ) { 
  600.         switch ( $xmlrpc_method ) { 
  601.             case 'metaWeblog.getRecentPosts': 
  602.             case 'wp.getPosts': 
  603.             case 'wp.getPages': 
  604.                 add_action( 'parse_query', array( $this, 'make_filterable' ), 10, 1 ); 
  605.                 break; 
  606.             case 'wp.getPost': 
  607.                 $this->prime_post_cache(); 
  608.                 break; 
  609.         } 
  610.     } 
  611.  
  612.     /** 
  613.      * metaWeblog.getPost and wp.getPage fire xmlrpc_call action *after* get_post() is called. 
  614.      * So, we have to detect those methods and prime the post cache early. 
  615.      * @return null 
  616.      */ 
  617.     protected function check_for_early_methods() { 
  618.         global $HTTP_RAW_POST_DATA; 
  619.         if ( false === strpos( $HTTP_RAW_POST_DATA, 'metaWeblog.getPost' ) 
  620.             && false === strpos( $HTTP_RAW_POST_DATA, 'wp.getPage' ) ) { 
  621.             return; 
  622.         } 
  623.         include_once( ABSPATH . WPINC . '/class-IXR.php' ); 
  624.         $message = new IXR_Message( $HTTP_RAW_POST_DATA ); 
  625.         $message->parse(); 
  626.         $post_id_position = 'metaWeblog.getPost' === $message->methodName ?  0 : 1; 
  627.         $this->prime_post_cache( $message->params[ $post_id_position ] ); 
  628.     } 
  629.  
  630.     /** 
  631.      * Prime the post cache with swapped post_content. This is a sneaky way of getting around 
  632.      * the fact that there are no good hooks to call on the *.getPost xmlrpc methods. 
  633.      * 
  634.      * @return null 
  635.      */ 
  636.     private function prime_post_cache( $post_id = false ) { 
  637.         global $wp_xmlrpc_server; 
  638.         if ( ! $post_id ) { 
  639.             $post_id = $wp_xmlrpc_server->message->params[3]; 
  640.         } 
  641.  
  642.         // prime the post cache 
  643.         if ( $this->is_markdown( $post_id ) ) { 
  644.             $post = get_post( $post_id ); 
  645.             if ( ! empty( $post->post_content_filtered ) ) { 
  646.                 wp_cache_delete( $post->ID, 'posts' ); 
  647.                 $post = $this->swap_for_editing( $post ); 
  648.                 wp_cache_add( $post->ID, $post, 'posts' ); 
  649.                 $this->posts_to_uncache[] = $post_id; 
  650.             } 
  651.         } 
  652.         // uncache munged posts if using a persistent object cache 
  653.         if ( wp_using_ext_object_cache() ) { 
  654.             add_action( 'shutdown', array( $this, 'uncache_munged_posts' ) ); 
  655.         } 
  656.     } 
  657.  
  658.     /** 
  659.      * Swaps `post_content_filtered` back to `post_content` for editing purposes. 
  660.      * @param  object $post WP_Post object 
  661.      * @return object       WP_Post object with swapped `post_content_filtered` and `post_content` 
  662.      */ 
  663.     protected function swap_for_editing( $post ) { 
  664.         $markdown = $post->post_content_filtered; 
  665.         // unencode encoded code blocks 
  666.         $markdown = $this->get_parser()->codeblock_restore( $markdown ); 
  667.         // restore beginning of line blockquotes 
  668.         $markdown = preg_replace( '/^> /m', '> ', $markdown ); 
  669.         $post->post_content_filtered = $post->post_content; 
  670.         $post->post_content = $markdown; 
  671.         return $post; 
  672.     } 
  673.  
  674.  
  675.     /** 
  676.      * We munge the post cache to serve proper markdown content to XML-RPC clients. 
  677.      * Uncache these after the XML-RPC session ends. 
  678.      * @return null 
  679.      */ 
  680.     public function uncache_munged_posts() { 
  681.         // $this context gets lost in testing sometimes. Weird. 
  682.         foreach( WPCom_Markdown::get_instance()->posts_to_uncache as $post_id ) { 
  683.             wp_cache_delete( $post_id, 'posts' ); 
  684.         } 
  685.     }

  687.     /** 
  688.      * Since *.(get)?[Rr]ecentPosts calls get_posts with suppress filters on, we need to 
  689.      * turn them back on so that we can swap things for editing. 
  690.      * @param  object $wp_query WP_Query object 
  691.      * @return null 
  692.      */ 
  693.     public function make_filterable( $wp_query ) { 
  694.         $wp_query->set( 'suppress_filters', false ); 
  695.         add_action( 'the_posts', array( $this, 'the_posts' ), 10, 2 ); 
  696.     } 
  697.  
  698.     /** 
  699.      * Swaps post_content and post_content_filtered for editing. 
  700.      * @param  array  $posts     Posts returned by the just-completed query 
  701.      * @param  object $wp_query  Current WP_Query object 
  702.      * @return array             Modified $posts 
  703.      */ 
  704.     public function the_posts( $posts, $wp_query ) { 
  705.         foreach ( $posts as $key => $post ) { 
  706.             if ( $this->is_markdown( $post->ID ) && ! empty( $posts[ $key ]->post_content_filtered ) ) { 
  707.                 $markdown = $posts[ $key ]->post_content_filtered; 
  708.                 $posts[ $key ]->post_content_filtered = $posts[ $key ]->post_content; 
  709.                 $posts[ $key ]->post_content = $markdown; 
  710.             } 
  711.         } 
  712.         return $posts; 
  713.     }
