  418.   /** 
  419.    * Magic happens here. Markdown is converted and stored on post_content. Original Markdown is stored 
  420.    * in post_content_filtered so that we can continue editing as Markdown. 
  421.    * @param  array $post_data  The post data that will be inserted into the DB. Slashed. 
  422.    * @param  array $postarr    All the stuff that was in $_POST. 
  423.    * @return array             $post_data with post_content and post_content_filtered modified 
  424.    */ 
  425.   public function wp_insert_post_data( $post_data, $postarr ) { 
  426.       // $post_data array is slashed! 
  427.       $post_id = isset( $postarr['ID'] ) ? $postarr['ID'] : false; 
  428.       // bail early if markdown is disabled or this post type is unsupported. 
  429.       if ( ! $this->is_posting_enabled() || ! post_type_supports( $post_data['post_type'], self::POST_TYPE_SUPPORT ) ) { 
  430.           // it's disabled, but maybe this *was* a markdown post before. 
  431.           if ( $this->is_markdown( $post_id ) && ! empty( $post_data['post_content_filtered'] ) ) { 
  432.               $post_data['post_content_filtered'] = ''; 
  433.           } 
  434.           // we have no context to determine supported post types in the `post_content_pre` hook,  
  435.           // which already ran to sanitize code blocks. Undo that. 
  436.           $post_data['post_content'] = $this->get_parser()->codeblock_restore( $post_data['post_content'] ); 
  437.           return $post_data; 
  438.       } 
  439.       // rejigger post_content and post_content_filtered 
  440.       // revisions are already in the right place, except when we're restoring, but that's taken care of elsewhere 
  441.       if ( 'revision' !== $post_data['post_type'] ) { 
  442.           /** 
  443.            * Filter the original post content passed to Markdown. 
  444.            * 
  445.            * @module markdown 
  446.            * 
  447.            * @since 2.8.0 
  448.            * 
  449.            * @param string $post_data['post_content'] Untransformed post content. 
  450.            */ 
  451.           $post_data['post_content_filtered'] = apply_filters( 'wpcom_untransformed_content', $post_data['post_content'] ); 
  452.           $post_data['post_content'] = $this->transform( $post_data['post_content'], array( 'id' => $post_id ) ); 
  453.           /** This filter is already documented in core/wp-includes/default-filters.php */ 
  454.           $post_data['post_content'] = apply_filters( 'content_save_pre', $post_data['post_content'] ); 
  455.       } elseif ( 0 === strpos( $post_data['post_name'], $post_data['post_parent'] . '-autosave' ) ) { 
  456.           // autosaves for previews are weird 
  457.           $post_data['post_content'] = $this->transform( $post_data['post_content'], array( 'id' => $post_data['post_parent'] ) ); 
  458.           /** This filter is already documented in core/wp-includes/default-filters.php */ 
  459.           $post_data['post_content'] = apply_filters( 'content_save_pre', $post_data['post_content'] ); 
  460.       } 
  461.  
  462.       // set as markdown on the wp_insert_post hook later 
  463.       if ( $post_id ) 
  464.           $this->monitoring['post'][ $post_id ] = true; 
  465.       else 
  466.           $this->monitoring['content'] = wp_unslash( $post_data['post_content'] ); 
  467.       if ( 'revision' === $postarr['post_type'] && $this->is_markdown( $postarr['post_parent'] ) ) 
  468.           $this->monitoring['parent'][ $postarr['post_parent'] ] = true; 
  469.  
  470.       return $post_data; 
  471.   } 
  472.  
  473.   /** 
  474.    * Calls on wp_insert_post action, after wp_insert_post_data. This way we can 
  475.    * still set postmeta on our revisions after it's all been deleted. 
  476.    * @param  int $post_id The post ID that has just been added/updated 
  477.    * @return null 
  478.    */ 
  479.   public function wp_insert_post( $post_id ) { 
  480.       $post_parent = get_post_field( 'post_parent', $post_id ); 
  481.       // this didn't have an ID yet. Compare the content that was just saved. 
  482.       if ( isset( $this->monitoring['content'] ) && $this->monitoring['content'] === get_post_field( 'post_content', $post_id ) ) { 
  483.           unset( $this->monitoring['content'] ); 
  484.           $this->set_as_markdown( $post_id ); 
  485.       } 
  486.       if ( isset( $this->monitoring['post'][$post_id] ) ) { 
  487.           unset( $this->monitoring['post'][$post_id] ); 
  488.           $this->set_as_markdown( $post_id ); 
  489.       } elseif ( isset( $this->monitoring['parent'][$post_parent] ) ) { 
  490.           unset( $this->monitoring['parent'][$post_parent] ); 
  491.           $this->set_as_markdown( $post_id ); 
  492.       } 
  493.   }

  495.   /** 
  496.    * Run a comment through Markdown. Easy peasy. 
  497.    * @param  string $content 
  498.    * @return string 
  499.    */ 
  500.   public function pre_comment_content( $content ) { 
  501.       return $this->transform( $content, array( 
  502.           'id' => $this->comment_hash( $content ),  
  503.  ) ); 
  504.   } 
  505.  
  506.   protected function comment_hash( $content ) { 
  507.       return 'c-' . substr( md5( $content ), 0, 8 ); 
  508.   }

  510.   /** 
  511.    * Markdown conversion. Some DRYness for repetitive tasks. 
  512.    * @param  string $text  Content to be run through Markdown 
  513.    * @param  array  $args  Arguments, with keys: 
  514.    *                       id: provide a string to prefix footnotes with a unique identifier 
  515.    *                       unslash: when true, expects and returns slashed data 
  516.    *                       decode_code_blocks: when true, assume that text in fenced code blocks is already 
  517.    *                         HTML encoded and should be decoded before being passed to Markdown, which does 
  518.    *                         its own encoding. 
  519.    * @return string        Markdown-processed content 
  520.    */ 
  521.   public function transform( $text, $args = array() ) { 
  522.       $args = wp_parse_args( $args, array( 
  523.           'id' => false,  
  524.           'unslash' => true,  
  525.           'decode_code_blocks' => ! $this->get_parser()->use_code_shortcode 
  526.  ) ); 
  527.       // probably need to unslash 
  528.       if ( $args['unslash'] ) 
  529.           $text = wp_unslash( $text ); 
  530.  
  531.       /** 
  532.        * Filter the content to be run through Markdown, before it's transformed by Markdown. 
  533.        * 
  534.        * @module markdown 
  535.        * 
  536.        * @since 2.8.0 
  537.        * 
  538.        * @param string $text Content to be run through Markdown 
  539.        * @param array $args Array of Markdown options. 
  540.        */ 
  541.       $text = apply_filters( 'wpcom_markdown_transform_pre', $text, $args ); 
  542.       // ensure our paragraphs are separated 
  543.       $text = str_replace( array( '</p><p>', "</p>\n<p>" ), "</p>\n\n<p>", $text ); 
  544.       // visual editor likes to add <p>s. Buh-bye. 
  545.       $text = $this->get_parser()->unp( $text ); 
  546.       // sometimes we get an encoded > at start of line, breaking blockquotes 
  547.       $text = preg_replace( '/^>/m', '>', $text ); 
  548.       // prefixes are because we need to namespace footnotes by post_id 
  549.       $this->get_parser()->fn_id_prefix = $args['id'] ? $args['id'] . '-' : ''; 
  550.       // If we're not using the code shortcode, prevent over-encoding. 
  551.       if ( $args['decode_code_blocks'] ) { 
  552.           $text = $this->get_parser()->codeblock_restore( $text ); 
  553.       } 
  554.       // Transform it! 
  555.       $text = $this->get_parser()->transform( $text ); 
  556.       // Fix footnotes - kses doesn't like the : IDs it supplies 
  557.       $text = preg_replace( '/((id|href)="#?fn(ref)?):/', "$1-", $text ); 
  558.       // Markdown inserts extra spaces to make itself work. Buh-bye. 
  559.       $text = rtrim( $text ); 
  560.       /** 
  561.        * Filter the content to be run through Markdown, after it was transformed by Markdown. 
  562.        * 
  563.        * @module markdown 
  564.        * 
  565.        * @since 2.8.0 
  566.        * 
  567.        * @param string $text Content to be run through Markdown 
  568.        * @param array $args Array of Markdown options. 
  569.        */ 
  570.       $text = apply_filters( 'wpcom_markdown_transform_post', $text, $args ); 
  571.  
  572.       // probably need to re-slash 
  573.       if ( $args['unslash'] ) 
  574.           $text = wp_slash( $text ); 
  575.  
  576.       return $text; 
  577.   }

  580.    * Shows Markdown in the Revisions screen, and ensures that post_content_filtered 
  581.    * is maintained on revisions 
  582.    * @param  array $fields  Post fields pertinent to revisions 
  583.    * @return array          Modified array to include post_content_filtered 
  584.    */

  587.       return $fields;

  590.   /** 
  591.    * Do some song and dance to keep all post_content and post_content_filtered content 
  592.    * in the expected place when a post revision is restored. 
  593.    * @param  int $post_id        The post ID have a restore done to it 
  594.    * @param  int $revision_id    The revision ID being restored 
  595.    * @return null 
  596.    */ 
  597.   public function wp_restore_post_revision( $post_id, $revision_id ) { 
  598.       if ( $this->is_markdown( $revision_id ) ) { 
  599.           $revision = get_post( $revision_id, ARRAY_A ); 
  600.           $post = get_post( $post_id, ARRAY_A ); 
  601.           $post['post_content'] = $revision['post_content_filtered']; // Yes, we put it in post_content, because our wp_insert_post_data() expects that 
  602.           // set this flag so we can restore the post_content_filtered on the last revision later 
  603.           $this->monitoring['restore'] = true; 
  604.           // let's not make a revision of our fixing update 
  605.           add_filter( 'wp_revisions_to_keep', '__return_false', 99 ); 
  606.           wp_update_post( $post ); 
  607.           $this->fix_latest_revision_on_restore( $post_id ); 
  608.           remove_filter( 'wp_revisions_to_keep', '__return_false', 99 ); 
  609.       } 
  610.   } 
  611.  
  612.   /** 
  613.    * We need to ensure the last revision has Markdown, not HTML in its post_content_filtered 
  614.    * column after a restore. 
  615.    * @param  int $post_id The post ID that was just restored. 
  616.    * @return null 
  617.    */ 
  618.   protected function fix_latest_revision_on_restore( $post_id ) { 
  619.       global $wpdb; 
  620.       $post = get_post( $post_id ); 
  621.       $last_revision = $wpdb->get_row( $wpdb->prepare( "SELECT * FROM $wpdb->posts WHERE post_type = 'revision' AND post_parent = %d ORDER BY ID DESC", $post->ID ) ); 
  622.       $last_revision->post_content_filtered = $post->post_content_filtered; 
  623.       wp_insert_post( (array) $last_revision ); 
  624.   }

  626.   /** 
  627.    * Kicks off magic for an XML-RPC session. We want to keep editing Markdown 
  628.    * and publishing HTML. 
  629.    * @param  string $xmlrpc_method The current XML-RPC method 
  630.    * @return null 
  631.    */ 
  632.   public function xmlrpc_actions( $xmlrpc_method ) { 
  633.       switch ( $xmlrpc_method ) { 
  634.           case 'metaWeblog.getRecentPosts': 
  635.           case 'wp.getPosts': 
  636.           case 'wp.getPages': 
  637.               add_action( 'parse_query', array( $this, 'make_filterable' ), 10, 1 ); 
  638.               break; 
  639.           case 'wp.getPost': 
  640.               $this->prime_post_cache(); 
  641.               break; 
  642.       } 
  643.   } 
  644.  
  645.   /** 
  646.    * metaWeblog.getPost and wp.getPage fire xmlrpc_call action *after* get_post() is called. 
  647.    * So, we have to detect those methods and prime the post cache early. 
  648.    * @return null 
  649.    */ 
  650.   protected function check_for_early_methods() { 
  651.       global $HTTP_RAW_POST_DATA; 
  652.       if ( false === strpos( $HTTP_RAW_POST_DATA, 'metaWeblog.getPost' ) 
  653.           && false === strpos( $HTTP_RAW_POST_DATA, 'wp.getPage' ) ) { 
  654.           return; 
  655.       } 
  656.       include_once( ABSPATH . WPINC . '/class-IXR.php' ); 
  657.       $message = new IXR_Message( $HTTP_RAW_POST_DATA ); 
  658.       $message->parse(); 
  659.       $post_id_position = 'metaWeblog.getPost' === $message->methodName ?  0 : 1; 
  660.       $this->prime_post_cache( $message->params[ $post_id_position ] ); 
  661.   } 
  662.  
  663.   /** 
  664.    * Prime the post cache with swapped post_content. This is a sneaky way of getting around 
  665.    * the fact that there are no good hooks to call on the *.getPost xmlrpc methods. 
  666.    * 
  667.    * @return null 
  668.    */ 
  669.   private function prime_post_cache( $post_id = false ) { 
  670.       global $wp_xmlrpc_server; 
  671.       if ( ! $post_id ) { 
  672.           $post_id = $wp_xmlrpc_server->message->params[3]; 
  673.       } 
  674.  
  675.       // prime the post cache 
  676.       if ( $this->is_markdown( $post_id ) ) { 
  677.           $post = get_post( $post_id ); 
  678.           if ( ! empty( $post->post_content_filtered ) ) { 
  679.               wp_cache_delete( $post->ID, 'posts' ); 
  680.               $post = $this->swap_for_editing( $post ); 
  681.               wp_cache_add( $post->ID, $post, 'posts' ); 
  682.               $this->posts_to_uncache[] = $post_id; 
  683.           } 
  684.       } 
  685.       // uncache munged posts if using a persistent object cache 
  686.       if ( wp_using_ext_object_cache() ) { 
  687.           add_action( 'shutdown', array( $this, 'uncache_munged_posts' ) ); 
  688.       } 
  689.   }

  691.   /** 
  692.    * Swaps `post_content_filtered` back to `post_content` for editing purposes. 
  693.    * @param  object $post WP_Post object 
  694.    * @return object       WP_Post object with swapped `post_content_filtered` and `post_content` 
  695.    */ 
  696.   protected function swap_for_editing( $post ) { 
  697.       $markdown = $post->post_content_filtered; 
  698.       // unencode encoded code blocks 
  699.       $markdown = $this->get_parser()->codeblock_restore( $markdown ); 
  700.       // restore beginning of line blockquotes 
  701.       $markdown = preg_replace( '/^> /m', '> ', $markdown ); 
  702.       $post->post_content_filtered = $post->post_content; 
  703.       $post->post_content = $markdown; 
  704.       return $post; 
  705.   }

  709.    * We munge the post cache to serve proper markdown content to XML-RPC clients. 
  710.    * Uncache these after the XML-RPC session ends. 
  711.    * @return null 
  712.    */ 
  713.   public function uncache_munged_posts() { 
  714.       // $this context gets lost in testing sometimes. Weird. 
  715.       foreach( WPCom_Markdown::get_instance()->posts_to_uncache as $post_id ) { 
  716.           wp_cache_delete( $post_id, 'posts' ); 
  717.       } 
  718.   }

  721.    * Since *.(get)?[Rr]ecentPosts calls get_posts with suppress filters on, we need to

  723.    * @param  object $wp_query WP_Query object

  726.   public function make_filterable( $wp_query ) {

  728.       add_action( 'the_posts', array( $this, 'the_posts' ), 10, 2 ); 
  729.   }

  731.   /** 
  732.    * Swaps post_content and post_content_filtered for editing. 
  733.    * @param  array  $posts     Posts returned by the just-completed query 
  734.    * @param  object $wp_query  Current WP_Query object 
  735.    * @return array             Modified $posts 
  736.    */ 
  737.   public function the_posts( $posts, $wp_query ) { 
  738.       foreach ( $posts as $key => $post ) { 
  739.           if ( $this->is_markdown( $post->ID ) && ! empty( $posts[ $key ]->post_content_filtered ) ) { 
  740.               $markdown = $posts[ $key ]->post_content_filtered; 
  741.               $posts[ $key ]->post_content_filtered = $posts[ $key ]->post_content; 
  742.               $posts[ $key ]->post_content = $markdown; 
  743.           } 
  744.       } 
  745.       return $posts; 
  746.   }
