  344.   function get_by_views( $count ) { 
  345.       /** 
  346.        * Filter the number of days used to calculate Top Posts for the Top Posts widget. 
  347.        * 
  348.        * @module widgets 
  349.        * 
  350.        * @since 2.8.0 
  351.        * 
  352.        * @param int 2 Number of days. Default is 2. 
  353.        */ 
  354.       $days = (int) apply_filters( 'jetpack_top_posts_days', 2 ); 
  355.  
  356.       if ( $days < 1 ) { 
  357.           $days = 2; 
  358.       } 
  359.  
  360.       if ( $days > 10 ) { 
  361.           $days = 10; 
  362.       } 
  363.  
  364.       $post_view_posts = stats_get_csv( 'postviews', array( 'days' => absint( $days ), 'limit' => 11 ) ); 
  365.       if ( ! $post_view_posts ) { 
  366.           return array(); 
  367.       } 
  368.  
  369.       $post_view_ids = array_filter( wp_list_pluck( $post_view_posts, 'post_id' ) ); 
  370.       if ( ! $post_view_ids ) { 
  371.           return array(); 
  372.       } 
  373.  
  374.       return $this->get_posts( $post_view_ids, $count ); 
  375.   }

  377.   function get_fallback_posts() { 
  378.       if ( current_user_can( 'edit_theme_options' ) ) { 
  379.           return array(); 
  380.       } 
  381.  
  382.       $post_query = new WP_Query; 
  383.  
  384.       $posts = $post_query->query( array( 
  385.           'posts_per_page' => 1,  
  386.           'post_status' => 'publish',  
  387.           'post_type' => array( 'post', 'page' ),  
  388.           'no_found_rows' => true,  
  389.  ) ); 
  390.  
  391.       if ( ! $posts ) { 
  392.           return array(); 
  393.       } 
  394.  
  395.       $post = array_pop( $posts ); 
  396.  
  397.       return $this->get_posts( $post->ID, 1 ); 
  398.   }

  400.   function get_posts( $post_ids, $count ) { 
  401.       $counter = 0; 
  402.  
  403.       $posts = array(); 
  404.       foreach ( (array) $post_ids as $post_id ) { 
  405.           $post = get_post( $post_id ); 
  406.  
  407.           if ( ! $post ) 
  408.               continue; 
  409.  
  410.           // hide private and password protected posts 
  411.           if ( 'publish' != $post->post_status || ! empty( $post->post_password ) || empty( $post->ID ) ) 
  412.               continue; 
  413.  
  414.           // Both get HTML stripped etc on display 
  415.           if ( empty( $post->post_title ) ) { 
  416.               $title_source = $post->post_content; 
  417.               $title = wp_html_excerpt( $title_source, 50 ); 
  418.               $title .= '…'; 
  419.           } else { 
  420.               $title = $post->post_title; 
  421.           } 
  422.  
  423.           $permalink = get_permalink( $post->ID ); 
  424.  
  425.           $post_type = $post->post_type; 
  426.  
  427.           $posts[] = compact( 'title', 'permalink', 'post_id', 'post_type' ); 
  428.           $counter++; 
  429.  
  430.           if ( $counter == $count ) { 
  431.               break; // only need to load and show x number of likes 
  432.           } 
  433.       } 
  434.  
  435.       /** 
  436.        * Filter the Top Posts and Pages. 
  437.        * 
  438.        * @module widgets 
  439.        * 
  440.        * @since 3.0.0 
  441.        * 
  442.        * @param array $posts Array of the most popular posts. 
  443.        * @param array $post_ids Array of Post IDs. 
  444.        * @param string $count Number of Top Posts we want to display. 
  445.        */ 
  446.       return apply_filters( 'jetpack_widget_get_top_posts', $posts, $post_ids, $count ); 
  447.   }
