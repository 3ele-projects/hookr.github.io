  62. function jetpack_instagram_handler( $matches, $atts, $url ) { 
  63.   global $content_width; 
  64.  
  65.   // keep a copy of the passed-in URL since it's modified below 
  66.   $passed_url = $url; 
  67.  
  68.   $max_width = 698; 
  69.   $min_width = 320; 
  70.  
  71.   if ( is_feed() ) { 
  72.       $media_url = sprintf( 'http://instagr.am/p/%s/media/?size=l', $matches[3] ); 
  73.       return sprintf( '<a href="%s" title="%s"><img src="%s" alt="Instagram Photo" /></a>', esc_url( $url ), esc_attr__( 'View on Instagram', 'jetpack' ), esc_url( $media_url ) ); 
  74.   } 
  75.  
  76.   $atts = shortcode_atts( array( 
  77.       'width' => isset( $content_width ) ? $content_width : $max_width,  
  78.       'hidecaption' => false,  
  79.  ), $atts ); 
  80.  
  81.   $atts['width'] = absint( $atts['width'] ); 
  82.   if ( $atts['width'] > $max_width || $min_width > $atts['width'] ) 
  83.       $atts['width'] = $max_width; 
  84.  
  85.  
  86.   // remove the modal param from the URL 
  87.   $url = remove_query_arg( 'modal', $url ); 
  88.  
  89.   // force .com instead of .am for https support 
  90.   $url = str_replace( 'instagr.am', 'instagram.com', $url ); 
  91.  
  92.   // The oembed endpoint expects HTTP, but HTTP requests 301 to HTTPS 
  93.   $instagram_http_url = str_replace( 'https://', 'http://', $url ); 
  94.   $instagram_https_url = str_replace( 'http://', 'https://', $url ); 
  95.  
  96.   $url_args = array( 
  97.       'url' => $instagram_http_url,  
  98.       'maxwidth' => $atts['width'],  
  99.  ); 
  100.  
  101.   if ( $atts['hidecaption'] ) { 
  102.       $url_args['hidecaption'] = 'true'; 
  103.   } 
  104.  
  105.   $url = esc_url_raw( add_query_arg( $url_args, 'https://api.instagram.com/oembed/' ) ); 
  106.  
  107.   /** 
  108.    * Filter Object Caching for response from Instagram. 
  109.    * 
  110.    * Allow enabling of object caching for the response sent by Instagram when querying for Instagram image HTML. 
  111.    * 
  112.    * @module shortcodes 
  113.    * 
  114.    * @since 3.3.0 
  115.    * 
  116.    * @param bool false Object caching is off by default. 
  117.    * @param array $matches Array of Instagram URLs found in the post. 
  118.    * @param array $atts Instagram Shortcode attributes. 
  119.    * @param string $passed_url Instagram API URL. 
  120.    */ 
  121.   $response_body_use_cache = apply_filters( 'instagram_cache_oembed_api_response_body', false, $matches, $atts, $passed_url ); 
  122.   $response_body = false; 
  123.   if ( $response_body_use_cache ) { 
  124.       $cache_key = 'oembed_response_body_' . md5( $url ); 
  125.       $response_body = wp_cache_get( $cache_key, 'instagram_embeds' ); 
  126.   } 
  127.  
  128.   if ( ! $response_body ) { 
  129.       // Not using cache (default case) or cache miss 
  130.       $instagram_response = wp_remote_get( $url, array( 'redirection' => 0 ) ); 
  131.       if ( is_wp_error( $instagram_response ) || 200 != $instagram_response['response']['code'] || empty( $instagram_response['body'] ) ) { 
  132.           return "<!-- instagram error: invalid instagram resource -->"; 
  133.       } 
  134.  
  135.       $response_body = json_decode( $instagram_response['body'] ); 
  136.       if ( $response_body_use_cache ) { 
  137.           // if caching it is short-lived since this is a "Cache-Control: no-cache" resource 
  138.           wp_cache_set( $cache_key, $response_body, 'instagram_embeds', HOUR_IN_SECONDS + mt_rand( 0, HOUR_IN_SECONDS ) ); 
  139.       } 
  140.   } 
  141.  
  142.   if ( ! empty( $response_body->html ) ) { 
  143.       wp_enqueue_script( 'jetpack-instagram-embed', plugins_url( 'js/instagram.js', __FILE__ ), array( 'jquery' ), false, true ); 
  144.       // there's a script in the response, which we strip on purpose since it's added by this ^ script 
  145.       $ig_embed = preg_replace( '@<(script)[^>]*?>.*?</\\1>@si', '', $response_body->html ); 
  146.       return $ig_embed; 
  147.   } 
  148.  
  149.   return '<!-- instagram error: no embed found -->'; 
  150. }
