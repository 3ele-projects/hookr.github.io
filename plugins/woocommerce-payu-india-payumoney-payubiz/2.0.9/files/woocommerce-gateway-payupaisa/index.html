  312.       function check_payuindia_response() { 
  313.           global $woocommerce; 
  314.  
  315.           if( isset($_REQUEST['txnid']) && isset($_REQUEST['mihpayid']) ) { 
  316.               $order_id = $_REQUEST['udf1']; 
  317.               if($order_id != '') { 
  318.                   try{ 
  319.                       $order = new WC_Order( $order_id ); 
  320.                       $hash = $_REQUEST['hash']; 
  321.                       $status = $_REQUEST['status']; 
  322.                       //<SALT>|status||||||udf5|udf4|udf3|udf2|udf1|email|firstname|productinfo|amount|txnid|key 
  323.                       $checkhash = hash('sha512', "$this->key_secret|$_REQUEST[status]||||||||||$_REQUEST[udf1]|$_REQUEST[email]|$_REQUEST[firstname]|$_REQUEST[productinfo]|$_REQUEST[amount]|$_REQUEST[txnid]|$this->key_id"); 
  324.                       $trans_authorised = false; 
  325.                        
  326.                       if( $order->status !=='completed' ) { 
  327.                           if($hash == $checkhash) { 
  328.                               $status = strtolower($status); 
  329.                               if($status=="success") { 
  330.                                   $trans_authorised = true; 
  331.                                   $this->msg['message'] = "Thank you for shopping with us. Your account has been charged and your transaction is successful."; 
  332.                                   $this->msg['class'] = 'woocommerce-message'; 
  333.                                   if($order->status == 'processing') { 
  334.                                       $order->add_order_note('PayU Money ID: '.$_REQUEST['mihpayid'].' ('.$_REQUEST['txnid'].')<br/>PG: '.$_REQUEST['PG_TYPE'].'('.$_REQUEST['unmappedstatus'].')<br/>Bank Ref: '.$_REQUEST['bank_ref_num'].'('.$_REQUEST['mode'].')'); 
  335.                                   }else{ 
  336.                                       $order->payment_complete(); 
  337.                                       $order->add_order_note('PayU Money payment successful.<br/>PayU Money ID: '.$_REQUEST['mihpayid'].' ('.$_REQUEST['txnid'].')<br/>PG: '.$_REQUEST['PG_TYPE'].'('.$_REQUEST['unmappedstatus'].')<br/>Bank Ref: '.$_REQUEST['bank_ref_num'].'('.$_REQUEST['mode'].')'); 
  338.                                       $woocommerce->cart->empty_cart(); 
  339.                                   } 
  340.                               }else if($status=="pending") { 
  341.                                   $trans_authorised = true; 
  342.                                   $this->msg['message'] = "Thank you for shopping with us. Right now your payment status is pending. We will keep you posted regarding the status of your order through eMail"; 
  343.                                   $this->msg['class'] = 'woocommerce-info'; 
  344.                                   $order->add_order_note('PayU Money payment status is pending<br/>PayU Money ID: '.$_REQUEST['mihpayid'].' ('.$_REQUEST['txnid'].')<br/>PG: '.$_REQUEST['PG_TYPE'].'('.$_REQUEST['unmappedstatus'].')<br/>Bank Ref: '.$_REQUEST['bank_ref_num'].'('.$_REQUEST['mode'].')'); 
  345.                                   $order->update_status('on-hold'); 
  346.                                   $woocommerce -> cart -> empty_cart(); 
  347.                               }else{ 
  348.                                   $this->msg['class'] = 'woocommerce-error'; 
  349.                                   $this->msg['message'] = "Thank you for shopping with us. However, the transaction has been declined."; 
  350.                                   $order->add_order_note('Transaction ERROR: '.$_REQUEST['error'].'<br/>PayU Money ID: '.$_REQUEST['mihpayid'].' ('.$_REQUEST['txnid'].')<br/>PG: '.$_REQUEST['PG_TYPE'].'('.$_REQUEST['unmappedstatus'].')<br/>Bank Ref: '.$_REQUEST['bank_ref_num'].'('.$_REQUEST['mode'].')'); 
  351.                               } 
  352.                           }else{ 
  353.                               $this->msg['class'] = 'error'; 
  354.                               $this->msg['message'] = "Security Error. Illegal access detected."; 
  355.                               $order->add_order_note('Checksum ERROR: '.json_encode($_REQUEST)); 
  356.                           } 
  357.                           if($trans_authorised==false) { 
  358.                               $order->update_status('failed'); 
  359.                           } 
  360.                           //removed for WooCommerce 2.0 
  361.                           //add_action('the_content', array(&$this, 'payupaisa_showMessage')); 
  362.                       } 
  363.                   }catch(Exception $e) { 
  364.                       // $errorOccurred = true; 
  365.                       $msg = "Error"; 
  366.                   } 
  367.               } 
  368.  
  369.               if ( $this->redirect_page == '' || $this->redirect_page == 0 ) { 
  370.                   //$redirect_url = $order->get_checkout_payment_url( true ); 
  371.                   $redirect_url = get_permalink( get_option('woocommerce_myaccount_page_id') ); 
  372.               } else { 
  373.                   $redirect_url = get_permalink( $this->redirect_page ); 
  374.               } 
  375.                
  376.               wp_redirect( $redirect_url ); 
  377.               exit; 
  378.    
  379.           } 
  380.  
  381.       } //END-check_payuindia_response

  386.       function payuindia_get_pages($title = false, $indent = true) { 
  387.           $wp_pages = get_pages('sort_column=menu_order'); 
  388.           $page_list = array(); 
  389.           if ($title) $page_list[] = $title; 
  390.           foreach ($wp_pages as $page) { 
  391.               $prefix = ''; 
  392.               // show indented child pages? 
  393.               if ($indent) { 
  394.                   $has_parent = $page->post_parent; 
  395.                   while($has_parent) { 
  396.                       $prefix .=  ' - '; 
  397.                       $next_page = get_post($has_parent); 
  398.                       $has_parent = $next_page->post_parent; 
  399.                   } 
  400.               } 
  401.               // add to page list array array 
  402.               $page_list[$page->ID] = $prefix . $page->post_title; 
  403.           } 
  404.           return $page_list; 
  405.       } //END-payuindia_get_pages
