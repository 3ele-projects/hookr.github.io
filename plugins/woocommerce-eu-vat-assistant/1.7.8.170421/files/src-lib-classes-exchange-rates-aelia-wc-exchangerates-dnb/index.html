  7. /** 
  8.  * Retrieves the exchange rates from the Danish National Bank. 
  9.  * 
  10.  * @link http://www.nationalbanken.dk/en/statistics/exchange_rates/Pages/default.aspx 
  11.  * @link http://www.nationalbanken.dk/_vti_bin/DN/DataService.svc/CurrencyRatesXML?lang=en 
  12.  */ 
  13. class Exchange_Rates_DNB_Model extends \Aelia\WC\ExchangeRatesModel { 
  14.   // @var string The URL template to use to query DNB 
  15.   private $dnb_api_rates_url = 'http://www.nationalbanken.dk/_vti_bin/DN/DataService.svc/CurrencyRatesXML?lang=en'; 
  16.  
  17.   /** 
  18.    * Tranforms the exchange rates received from DNB into an array of 
  19.    * currency code => exchange rate pairs. 
  20.    * 
  21.    * @param string dnb_rates The XML received from DNB. 
  22.    * @retur array 
  23.    */ 
  24.   protected function decode_rates($dnb_rates) { 
  25.       $exchange_rates = array(); 
  26.  
  27.       foreach($dnb_rates->dailyrates->currency as $rate) { 
  28.           /** DNB provides exchange rates work in its own peculiar way. The "rate" 
  29.            * does not indicate how many units of the target currency correspond to 
  30.            * one unit of the base currency, but how many units of the base currency 
  31.            * are needed to get 100 units of the target currency. Because of that,  
  32.            * we have to make a simple calculation to get the actual exchange rate. 
  33.            */ 
  34.           $exchange_rates[(string)$rate['code']] = 100 / (float)$rate['rate']; 
  35.       } 
  36.       // DNB feed is based against DKK, but it doesn't contain such currency. We 
  37.       // can safely add it manually, with an exchange rate of 1 
  38.       $exchange_rates['DKK'] = 1; 
  39.       return $exchange_rates; 
  40.   } 
  41.  
  42.   /** 
  43.    * Fetches all exchange rates from DNB API. 
  44.    * 
  45.    * @return object|bool An object containing the response from Open Exchange, or 
  46.    * False in case of failure. 
  47.    */ 
  48.   private function fetch_all_rates() { 
  49.       try { 
  50.           $response = \Httpful\Request::get($this->dnb_api_rates_url) 
  51.               ->expectsXml() 
  52.               ->send(); 
  53.  
  54.           // Debug 
  55.           //var_dump("DNB RATES RESPONSE:", $response); die(); 
  56.           if($response->hasErrors()) { 
  57.               // OpenExchangeRates sends error details in response body 
  58.               if($response->hasBody()) { 
  59.                   $response_data = $response->body; 
  60.  
  61.                   $this->add_error(self::ERR_ERROR_RETURNED,  
  62.                                                    sprintf(__('Error returned by DNB. ' . 
  63.                                                                           'Error code: %s. Error message: %s - %s.',  
  64.                                                                           Definitions::TEXT_DOMAIN),  
  65.                                                                    $response_data->status,  
  66.                                                                    $response_data->message,  
  67.                                                                    $response_data->description)); 
  68.               } 
  69.               return false; 
  70.           } 
  71.           return $response->body; 
  72.       } 
  73.       catch(Exception $e) { 
  74.           $this->add_error(self::ERR_EXCEPTION_OCCURRED,  
  75.                                            sprintf(__('Exception occurred while retrieving the exchange rates from DNB. ' . 
  76.                                                                   'Error message: %s.',  
  77.                                                                   Definitions::TEXT_DOMAIN),  
  78.                                                            $e->getMessage())); 
  79.           return null; 
  80.       } 
  81.   } 
  82.  
  83.   /** 
  84.    * Returns current exchange rates for the specified currency. 
  85.    * 
  86.    * @param string base_currency The base currency. 
  87.    * @return array An array of Currency => Exchange Rate pairs. 
  88.    */ 
  89.   private function current_rates($base_currency) { 
  90.       if(empty($this->_current_rates) || 
  91.            $this->_base_currency != $base_currency) { 
  92.  
  93.           $cache_key = md5(get_class($this)) . $base_currency; 
  94.           // Try to get the cached rates for the specified base currency, if any 
  95.           $this->_current_rates = $this->get_cached_exchange_rates($cache_key); 
  96.           if(!empty($this->_current_rates)) { 
  97.               return $this->_current_rates; 
  98.           } 
  99.  
  100.           // Fetch exchange rates 
  101.           $dnb_exchange_rates = $this->fetch_all_rates(); 
  102.           if($dnb_exchange_rates === false) { 
  103.               return null; 
  104.           } 
  105.  
  106.           // Debug 
  107.           //var_dump($dnb_exchange_rates);die(); 
  108.  
  109.           // DNB rates are returned as JSON representation of an array of objects. 
  110.           // We need to transform it into an array of currency => rate pairs 
  111.           $exchange_rates = $this->decode_rates($dnb_exchange_rates); 
  112.  
  113.           // Debug 
  114.           //var_dump($exchange_rates);die(); 
  115.  
  116.           if(!is_array($exchange_rates)) { 
  117.               $this->add_error(self::ERR_UNEXPECTED_ERROR_FETCHING_EXCHANGE_RATES,  
  118.                                                __('An unexpected error occurred while fetching exchange rates ' . 
  119.                                                       'from DNB. The most common cause of this issue is the ' . 
  120.                                                       'absence of PHP CURL extension. Please make sure that ' . 
  121.                                                       'PHP CURL is installed and configured in your system.',  
  122.                                                       Definitions::TEXT_DOMAIN)); 
  123.               return array(); 
  124.           } 
  125.  
  126.           // Since we didn't get the exchange rates related to the base currency,  
  127.           // but in the default base currency used by OpenExchange, we need to 
  128.           // recalculate them against the base currency we would like to use 
  129.           $this->_current_rates = $this->rebase_rates($exchange_rates, $base_currency); 
  130.           $this->_base_currency = $base_currency; 
  131.  
  132.           // Cache the exchange rates 
  133.           $this->cache_exchange_rates($cache_key, $this->_current_rates); 
  134.       } 
  135.       return $this->_current_rates; 
  136.   } 
  137.  
  138.   /** 
  139.    * Recaculates the exchange rates using another base currency. This method 
  140.    * is invoked because the rates fetched from DNB are relative to BitCoin,  
  141.    * but another currency is most likely is used by WooCommerce. 
  142.    * 
  143.    * @param array exchange_rates The exchange rates retrieved from DNB. 
  144.    * @param string base_currency The base currency against which the rates should 
  145.    * be recalculated. 
  146.    * @return array An array of currency => exchange rate pairs. 
  147.    */ 
  148.   private function rebase_rates(array $exchange_rates, $base_currency) { 
  149.       $recalc_rate = get_value($base_currency, $exchange_rates); 
  150.       //var_dump($base_currency, $exchange_rates); 
  151.  
  152.       if(empty($recalc_rate)) { 
  153.           $this->add_error(self::ERR_BASE_CURRENCY_NOT_FOUND,  
  154.                                            sprintf(__('Could not rebase rates against base currency "%s". ' . 
  155.                                                                   'Currency not found in data returned by DNB.',  
  156.                                                                   Definitions::TEXT_DOMAIN),  
  157.                                                            $base_currency)); 
  158.           return null; 
  159.       } 
  160.  
  161.       $result = array(); 
  162.       foreach($exchange_rates as $currency => $rate) { 
  163.           $result[$currency] = $rate / $recalc_rate; 
  164.       } 
  165.  
  166.       // Debug 
  167.       //var_dump($result); die(); 
  168.       return $result; 
  169.   } 
  170.  
  171.   /** 
  172.    * Returns the exchange rate of a currency in respect to a base currency. 
  173.    * 
  174.    * @param string base_currency The code of the base currency. 
  175.    * @param string currency The code of the currency for which to find the 
  176.    * Exchange Rate. 
  177.    * @return float 
  178.    */ 
  179.   protected function get_rate($base_currency, $currency) { 
  180.       $current_rates = $this->current_rates($base_currency); 
  181.       return get_value($currency, $current_rates); 
  182.   }

  191.   public function __construct($settings = null) { 
  192.       parent::__construct($settings); 
  193.   } 
  194. }
