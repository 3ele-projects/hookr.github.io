  539.    * @param $fields 
  540.    * @param $server 
  541.    * @return array 
  542.    */ 
  543.   public function order_response( $order_data, $order, $fields, $server ) { 
  544.  
  545.     // add cashier data 
  546.     $cashier_details = isset( $order_data['cashier'] ) ? $order_data['cashier'] : array(); 
  547.     $order_data['cashier'] = $this->add_cashier_details( $order, $cashier_details ); 
  548.  
  549.     // add pos payment info 
  550.     $payment_details = isset( $order_data['payment_details'] ) ? $order_data['payment_details'] : array(); 
  551.     $order_data['payment_details'] = $this->add_payment_details( $order, $payment_details ); 
  552.  
  553.     // addresses 
  554. //    $order_data['billing_address'] = $this->filter_address($order_data['billing_address'], $order); 
  555. //    $order_data['shipping_address'] = $this->filter_address($order_data['shipping_address'], $order, 'shipping'); 
  556.  
  557.     // allow decimal quantity 
  558.     // fixed in WC 2.4+ 
  559.     if( version_compare( WC()->version, '2.4', '<' ) && wc_pos_get_option( 'general', 'decimal_qty' ) ) { 
  560.       $order_data['line_items'] = $this->filter_qty($order_data['line_items']); 
  561.     } 
  562.  
  563.     // add subtotal_tax 
  564.     $subtotal_tax = 0; 
  565.     foreach( $order_data['line_items'] as $item ) { 
  566.       if(isset( $item['subtotal_tax'] )) { 
  567.         $subtotal_tax += wc_format_decimal( $item['subtotal_tax'] ); 
  568.       } 
  569.     } 
  570.     $order_data['subtotal_tax'] = wc_format_decimal( $subtotal_tax ); 
  571.  
  572.     // add shipping tax 
  573.     foreach( $order_data['shipping_lines'] as &$item ) { 
  574.       if(isset( $item['id'] )) { 
  575.         $taxes = wc_get_order_item_meta( $item['id'], 'taxes' ); 
  576.         $item['total_tax'] = wc_format_decimal( array_sum($taxes) ); 
  577.       } 
  578.     } 
  579.  
  580.     // compatibility for templates 
  581.     foreach( $order_data['fee_lines'] as &$item ) { 
  582.       if( !isset($item['name']) && isset($item['title'] ) ) { 
  583.         $item['name'] = $item['title']; 
  584.       } 
  585.     } 
  586.  
  587.     // add cart discount tax 
  588.     $order_data['cart_discount_tax'] = get_post_meta($order->id, '_cart_discount_tax', true); 
  589.  
  590.     // Backwards compatibility for WC legacy API 
  591.     $order_data['discount_total'] = $order_data['total_discount']; 
  592.  
  593.     return $order_data; 
  594.   }
