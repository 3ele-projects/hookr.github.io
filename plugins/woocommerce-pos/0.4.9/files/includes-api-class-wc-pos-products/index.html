  96. /** 
  97.  * Filter each product response from WC REST API for easier handling by the POS 
  98.  * - use the thumbnails rather than fullsize 
  99.  * - add barcode field 
  100.  * - unset unnecessary data

  104.  * 
  105.  * @return array modified data array $product_data 
  106.  */ 
  107. public function product_response( $data, $product, $fields, $server ) { 
  108.   $type = isset( $data['type'] ) ? $data['type'] : ''; 
  109.  
  110.   // variable products 
  111.   if( $type == 'variable' ) : 
  112.     // nested variations 
  113.     foreach( $data['variations'] as &$variation ) : 
  114.       $_product = wc_get_product( $variation['id'] ); 
  115.       $variation = $this->filter_response_data( $variation, $_product ); 
  116.       $variation['attributes'] = $this->patch_variation_attributes( $_product ); 
  117.     endforeach; 
  118.   endif; 
  119.  
  120.   // variation 
  121.   if( $type == 'variation' ) : 
  122.     $data['attributes'] = $this->patch_variation_attributes( $product ); 
  123.   endif; 
  124.  
  125.   return $this->filter_response_data( $data, $product ); 
  126. }

  128. /** 
  129.  * https://github.com/woothemes/woocommerce/issues/8457 
  130.  * patches WC_Product_Variable->get_variation_attributes() 
  131.  * @param $product 
  132.  * @return array 
  133.  */ 
  134. private function patch_variation_attributes( $product ) { 
  135.   $patched_attributes = array(); 
  136.   $attributes = $product->get_attributes(); 
  137.   $variation_attributes = $product->get_variation_attributes(); 
  138.  
  139.   // patch for corrupted data, depreciate asap 
  140.   if( empty( $attributes ) ) { 
  141.     $attributes = $product->parent->product_attributes; 
  142.     delete_post_meta( $product->variation_id, '_product_attributes' ); 
  143.   } 
  144.  
  145.   foreach( $variation_attributes as $slug => $option ) { 
  146.     $slug = str_replace( 'attribute_', '', $slug ); 
  147.  
  148.     if( isset( $attributes[$slug] ) ) { 
  149.       $patched_attributes[] = array( 
  150.         'name' => $this->get_variation_name( $attributes[$slug] ),  
  151.         'option' => $this->get_variation_option( $product, $attributes[$slug], $option ) 
  152.  ); 
  153.     } 
  154.  
  155.   } 
  156.  
  157.   return $patched_attributes; 
  158. }

  164. private function get_variation_name( $attribute ) { 
  165.   if( $attribute['is_taxonomy'] ) { 
  166.     global $wpdb; 
  167.     $name = str_replace( 'pa_', '', $attribute['name'] ); 
  168.  
  169.     $label = $wpdb->get_var( 
  170.       $wpdb->prepare(" 
  171.         SELECT attribute_label 
  172.         FROM {$wpdb->prefix}woocommerce_attribute_taxonomies 
  173.         WHERE attribute_name = %s; 
  174.       ", $name ) ); 
  175.  
  176.     return $label ? $label : $name; 
  177.   } 
  178.  
  179.   return $attribute['name']; 
  180. }

  182. /**

  184.  * @param $option 
  185.  * @param $attribute 
  186.  * @return mixed 
  187.  */ 
  188. private function get_variation_option( $product, $attribute, $option ) { 
  189.   $name = $option; 
  190.  
  191.   // taxonomy attributes 
  192.   if ( $attribute['is_taxonomy'] ) { 
  193.     $terms = wp_get_post_terms( $product->parent->id, $attribute['name'] ); 
  194.     if( !is_wp_error($terms) ) : foreach( $terms as $term ) : 
  195.       if( $option === $term->slug ) $name = $term->name; 
  196.     endforeach; endif; 
  197.  
  198.   // piped attributes 
  199.   } else { 
  200.     $values = array_map( 'trim', explode( WC_DELIMITER, $attribute['value'] ) ); 
  201.     $options = array_combine( array_map( 'sanitize_title', $values) , $values ); 
  202.     if( $options && isset( $options[$option] ) ) { 
  203.       $name = $options[$option]; 
  204.     } 
  205.   } 
  206.  
  207.   return $name; 
  208. }

  210. /** 
  211.  * Filter product response data 
  212.  * - add featured_src 
  213.  * - add special key for barcode, defaults to sku 
  214.  * @param array $data 
  215.  * @param $product 
  216.  * @return array 
  217.  */ 
  218. private function filter_response_data( array $data, $product ) { 
  219.   $id = isset( $data['id'] ) ? $data['id'] : ''; 
  220.   $barcode = isset( $data['sku'] ) ? $data['sku'] : ''; 
  221.  
  222.   // allow custom barcode field 
  223.   $barcode_meta_key = apply_filters( 'woocommerce_pos_barcode_meta_key', '_sku' ); 
  224.   if( $barcode_meta_key !== '_sku' ) { 
  225.     $barcode = get_post_meta( $id, $barcode_meta_key, true ); 
  226.   } 
  227.  
  228.   $data['featured_src'] = $this->get_thumbnail( $id ); 
  229.   $data['barcode'] = apply_filters( 'woocommerce_pos_product_barcode', $barcode, $id ); 
  230.  
  231.   // allow decimal stock quantities, fixed in WC 2.4 
  232.   if( version_compare( WC()->version, '2.4', '<' ) ) { 
  233.     $data['stock_quantity'] = $product->get_stock_quantity(); 
  234.   } 
  235.  
  236.   // Transition to name property 
  237.   if( isset($data['title']) ) { 
  238.     $data['name'] = $data['title']; 
  239.     unset($data['title']); 
  240.   } 
  241.  
  242.   // filter by whitelist 
  243.   // - note, this uses the same method as WC REST API fields parameter 
  244.   // - this doesn't speed up queries as it should 
  245.   // - when WC REST API properly filters requests POS should use fields param 
  246.   return array_intersect_key( $data, array_flip( $this->whitelist ) ); 
  247. }

  320. /**

  325. public function posts_where( $where, $query ) { 
  326.   global $wpdb; 
  327.  
  328.   if( isset( $_GET['filter'] ) ) { 
  329.  
  330.     $filter = $_GET['filter']; 
  331.  
  332.     if( isset($filter['barcode']) ) { 
  333.  
  334.       $barcode_meta_key = apply_filters( 'woocommerce_pos_barcode_meta_key', '_sku' ); 
  335.  
  336.       // gets post ids and parent ids 
  337.       $result = $wpdb->get_results( 
  338.         $wpdb->prepare(" 
  339.           SELECT p.ID, p.post_parent 
  340.           FROM $wpdb->posts AS p 
  341.           JOIN $wpdb->postmeta AS pm 
  342.           ON p.ID = pm.post_id 
  343.           WHERE pm.meta_key = %s 
  344.           AND pm.meta_value LIKE %s 
  345.         ", $barcode_meta_key, '%'.$filter['barcode'].'%' ),  
  346.         ARRAY_N 
  347.  ); 
  348.  
  349.       if($result) { 
  350.         $ids = call_user_func_array('array_merge', $result); 
  351.         $where .= " AND ID IN (" . implode( ', ', array_unique($ids) ) . ")"; 
  352.       } else { 
  353.         // no matches 
  354.         $where .= " AND 1=0"; 
  355.       } 
  356.  
  357.     } 
  358.  
  359.   } 
  360.  
  361.   return $where; 
  362. }
