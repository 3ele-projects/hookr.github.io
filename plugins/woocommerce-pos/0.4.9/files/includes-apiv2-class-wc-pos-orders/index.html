  135.  
  136.   // payment method 
  137.   $payment_method = $order->get_payment_method(); 
  138.  
  139.   // some gateways check if a user is signed in, so let's switch to customer 
  140.   $logged_in_user = get_current_user_id(); 
  141.   $customer_id = $order->get_customer_id(); 
  142.   wp_set_current_user( $customer_id ); 
  143.  
  144.   // load the gateways & process payment

  147.   $gateways = $settings->load_enabled_gateways(); 
  148.  
  149.   // process payment

  151.   $response = $gateways[ $payment_method ]->process_payment( $order->get_id() ); 
  152.  
  153.   if(isset($response['result']) && $response['result'] == 'success') { 
  154.  
  155.     $this->payment_success($payment_method, $order, $response); 
  156.  
  157.     if( ! get_post_meta( $order->get_id(), '_pos_payment_redirect' ) ) { 
  158.       $order->set_date_paid( current_time( 'timestamp' ) ); 
  159.       $order->set_date_completed( current_time( 'timestamp' ) ); 
  160.       $message = __('POS Transaction completed.', 'woocommerce-pos'); 
  161.       $order->update_status( wc_pos_get_option( 'checkout', 'order_status' ), $message ); 
  162.     } 
  163.  
  164.   } else { 
  165.  
  166.     $this->payment_failure($payment_method, $order, $response); 
  167.     $order->update_status( 'wc-failed', __( 'There was an error processing the payment', 'woocommerce-pos') ); 
  168.  
  169.   } 
  170.  
  171.   // switch back to logged in user 
  172.   wp_set_current_user( $logged_in_user ); 
  173.  
  174.   // clear any payment gateway messages 
  175.   wc_clear_notices(); 
  176.  
  177. }

  362.   $success = wp_parse_args( parse_url( $success_url ), array( 'host' => '', 'path' => '' )); 
  363.   $redirect = wp_parse_args( parse_url( $response['redirect'] ), array( 'host' => '', 'path' => '' ));

  366.  
  367.   if($offsite) { 
  368.     update_post_meta( $order->get_id(), '_pos_payment_redirect', $response['redirect'] ); 
  369.     $message = __('You are now being redirected offsite to complete the payment. ', 'woocommerce-pos'); 
  370.     $message .= sprintf( __('<a href="%s">Click here</a> if you are not redirected automatically. ', 'woocommerce-pos'), $response['redirect'] ); 
  371.   } 
  372.  
  373.   return $message; 
  374. }
