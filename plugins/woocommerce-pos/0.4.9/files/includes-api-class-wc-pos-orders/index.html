  514.  
  515.   // compare url fragments 
  516.   $success_url = wc_get_endpoint_url( 'order-received', $order_id, get_permalink( wc_get_page_id( 'checkout' ) ) ); 
  517.   $success = wp_parse_args( parse_url( $success_url ), array( 'host' => '', 'path' => '' ));

  525.     $message = __('You are now being redirected offsite to complete the payment. ', 'woocommerce-pos');

  549.  * @param array $order_data 
  550.  * @param $order 
  551.  * @param $fields 
  552.  * @param $server 
  553.  * @return array 
  554.  */ 
  555. public function order_response( $order_data, $order, $fields, $server ) { 
  556.  
  557.   // add cashier data 
  558.   $cashier_details = isset( $order_data['cashier'] ) ? $order_data['cashier'] : array(); 
  559.   $order_data['cashier'] = $this->add_cashier_details( $order, $cashier_details ); 
  560.  
  561.   // add pos payment info 
  562.   $payment_details = isset( $order_data['payment_details'] ) ? $order_data['payment_details'] : array(); 
  563.   $order_data['payment_details'] = $this->add_payment_details( $order, $payment_details ); 
  564.  
  565.   // addresses 
  566. //    $order_data['billing_address'] = $this->filter_address($order_data['billing_address'], $order); 
  567. //    $order_data['shipping_address'] = $this->filter_address($order_data['shipping_address'], $order, 'shipping'); 
  568.  
  569.   // allow decimal quantity 
  570.   // fixed in WC 2.4+ 
  571.   if( version_compare( WC()->version, '2.4', '<' ) && wc_pos_get_option( 'general', 'decimal_qty' ) ) { 
  572.     $order_data['line_items'] = $this->filter_qty($order_data['line_items']); 
  573.   } 
  574.  
  575.   // add subtotal_tax 
  576.   $subtotal_tax = 0; 
  577.   foreach( $order_data['line_items'] as $item ) { 
  578.     if(isset( $item['subtotal_tax'] )) { 
  579.       $subtotal_tax += wc_format_decimal( $item['subtotal_tax'] ); 
  580.     } 
  581.   } 
  582.   $order_data['subtotal_tax'] = wc_format_decimal( $subtotal_tax ); 
  583.  
  584.   // add shipping tax 
  585.   foreach( $order_data['shipping_lines'] as &$item ) { 
  586.     if(isset( $item['id'] )) { 
  587.       $taxes = wc_get_order_item_meta( $item['id'], 'taxes' ); 
  588.       $item['total_tax'] = wc_format_decimal( array_sum($taxes) ); 
  589.     } 
  590.   } 
  591.  
  592.   // compatibility for templates 
  593.   foreach( $order_data['fee_lines'] as &$item ) { 
  594.     if( !isset($item['name']) && isset($item['title'] ) ) { 
  595.       $item['name'] = $item['title']; 
  596.     } 
  597.   } 
  598.  
  599.   // add cart discount tax 
  600.   $order_data['cart_discount_tax'] = get_post_meta($order->id, '_cart_discount_tax', true); 
  601.  
  602.   // Backwards compatibility for WC legacy API 
  603.   $order_data['discount_total'] = $order_data['total_discount']; 
  604.  
  605.   return $order_data; 
  606. }
