  530.                     $nextview = monsterinsights_google_auth_enterkey_view( $reauth, $auth_url, esc_html__( 'Please paste in your Google code.', 'google-analytics-for-wordpress' ) ); 
  531.                 } 
  532.                 break; 
  533.  
  534.             case 'selectprofile': 
  535.                 $profile = ! empty( $_POST['stepdata'] ) ? absint( sanitize_text_field( $_POST['stepdata'] ) ) : ''; 
  536.                 if ( ! empty( $profile ) ) { 
  537.                     $this->set_test_client(); 
  538.                     $profile = $this->find_selected_profile( $profile ); 
  539.                     if ( ! empty( $profile ) ) { 
  540.  
  541.                         $this->clear_manual_data(); // Just in case we were manual, clear out UA 
  542.  
  543.                         $this->client->move_test_to_live(); 
  544.                         $this->save_selected_profile( $profile ); 
  545.                         $this->set_client(); 
  546.                         $this->reinitialize(); 
  547.  
  548.                         // Refresh reporting data 
  549.                         $this->base->reports->refresh_aggregate_data(); 
  550.  
  551.                         $nextview = monsterinsights_google_auth_done_view( $reauth ); 
  552.                     } else { 
  553.                         // Invalid profile selected 
  554.                         $profiles = get_option( 'monsterinsights_get_profiles', array() ); 
  555.                         $select = $this->ga_select( $profiles ); 
  556.                         $nextview = monsterinsights_google_auth_selectprofile_view( $reauth, $select, esc_html__( 'Invalid profile selected.', 'google-analytics-for-wordpress' ) ); 
  557.                     } 
  558.                 } else {  
  559.                     // No profile selected 
  560.                     $profiles = get_option( 'monsterinsights_get_profiles', array() ); 
  561.                     $select = $this->ga_select( $profiles ); 
  562.                     $nextview = monsterinsights_google_auth_selectprofile_view( $reauth, $select, esc_html__( 'Please select a profile.', 'google-analytics-for-wordpress' ) ); 
  563.                 } 
  564.                 break; 
  565.  
  566.             case 'done': 
  567.                 $nextview = monsterinsights_google_auth_done_view( $reauth ); 
  568.                 break; 
  569.                  
  570.             case 'prestart': 
  571.             default: 
  572.                  $nextview = monsterinsights_google_auth_start_view( $reauth ); 
  573.                 break; 
  574.         } 
  575.         echo $nextview; 
  576.         wp_die(); 
  577.     }

  579.     public function google_cancel() { 
  580.         $view = isset( $_POST['view'] ) && in_array( $_POST['view'], array( 'prestart', 'start', 'enterkey', 'selectprofile', 'done' ) ) ? $_POST['view'] : ''; 
  581.         $reauth = isset( $_POST['reauth'] ) && $_POST['reauth'] && $_POST['reauth'] !== 'false'  ? true : false; 
  582.  
  583.         if ( ! current_user_can( 'monsterinsights_save_settings' ) ) {

  585.             wp_die(); 
  586.         } 
  587.  
  588.         // If we cancelled on enterkey or selectprofile, delete the temp access tokens and set client back to normal client

  590.             $this->client->clear_data(); 
  591.             delete_option( 'monsterinsights_get_profiles' ); 
  592.             $this->set_client(); 
  593.         }

  607.     private function ga_select( $profiles = array() ) { 
  608.         if ( empty( $profiles ) || ! is_array( $profiles ) ) {  
  609.             $profiles = $this->get_profiles(); 
  610.         } 
  611.  
  612.         $optgroups = array(); 
  613.         foreach ( $profiles as $key => $value ) { 
  614.             foreach ( $value['items'] as $subitem ) { 
  615.                 $optgroups[ $subitem['name'] ]['items'] = $subitem['items']; 
  616.             } 
  617.         } 
  618.  
  619.         $values = $optgroups; 
  620.         $select = ''; 
  621.         $select .= '<div class="monsterinsights_ga_form">'; 
  622.         $select .= '<label for="monsterinsights_step_data" id="monsterinsights_select_ga_profile_label">' . esc_html__( 'Analytics profile', 'google-analytics-for-wordpress' ) . ':</label>';

'" name="monsterinsights_step_data" class="monsterinsights-select300 monsterinsights_select_ga_profile" id="monsterinsights_step_data" style="width:80%;margin-left:10%;margin-right:10%;background-color: #FFF;">'; 
  624.         $select .= '<option></option>'; 
  625.  
  626.         if ( count( $values ) >= 1 ) { 
  627.             foreach ( $values as $optgroup => $value ) { 
  628.                 if ( ! empty( $value['items'] ) ) { 
  629.                     $select .= $this->create_optgroup( $optgroup, $value ); 
  630.                 } 
  631.                 else { 
  632.                     $select .= '<option value="' . esc_attr( $value['id'] ) . '">' . esc_attr( stripslashes( $value['name'] ) ) . '</option>'; 
  633.                 } 
  634.             } 
  635.         } 
  636.         $select .= '</select>'; 
  637.         $select .= '</div>'; 
  638.         return $select;
