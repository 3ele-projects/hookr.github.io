  76.         return '/' 
  77.                . '<a'               // matches the characters <a literally 
  78.                . '\s'                // Match any sort of whitespace 
  79.                . '([^>]*)'          // 1. match a single character not present in the list (Between 0 and * times) 
  80.                . '\s'               // match any white space character 
  81.                . 'href='            // matches the characters href= literally 
  82.                . '([\'\"])'         // 2. match a single or a double quote 
  83.                . '('                // 3. matches the link protocol (between 0 and 1 times) 
  84.                .   '([a-zA-Z]+)'    // 4. matches the link protocol name 
  85.                .   ':'              // matches the character : literally 
  86.                .   '(?:\/\/)??'     // matches the characters 
  87.                .  ')??'             // literally (between 0 and 1 times) 
  88.                .  '(.*)'            // 5. matches any character (except newline) (Between 0 and * times) 
  89.                .  '\2'              // matches the same quote (2nd capturing group) as was used to open the href value 
  90.                .  '([^>]*)'         // 6. match a single character not present in the list below 
  91.                . '>'                // matches the characters > literally 
  92.                . '(.*)'             // 7. matches any character (except newline) (Between 0 and * times) 
  93.                . '<\/a>'            // matches the characters </a> literally 
  94.                . '/isU';            // case insensitive, single line, ungreedy
