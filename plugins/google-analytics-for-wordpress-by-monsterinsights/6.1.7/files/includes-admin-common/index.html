  290. /** 
  291.  * Called whenever an upgrade button / link is displayed in Lite, this function will 
  292.  * check if there's a shareasale ID specified. 
  293.  * 
  294.  * There are three ways to specify an ID, ordered by highest to lowest priority 
  295.  * - add_filter( 'monsterinsights_shareasale_id', function() { return 1234; } ); 
  296.  * - define( 'MONSTERINSIGHTS_SHAREASALE_ID', 1234 ); 
  297.  * - get_option( 'monsterinsights_shareasale_id' ); (with the option being in the wp_options table) 
  298.  * 
  299.  * If an ID is present, returns the ShareASale link with the affiliate ID, and tells 
  300.  * ShareASale to then redirect to monsterinsights.com/lite 
  301.  * 
  302.  * If no ID is present, just returns the monsterinsights.com/lite URL with UTM tracking. 
  303.  * 
  304.  * @since 6.0.0 
  305.  * @access public 
  306.  * 
  307.  * @return string Upgrade link. 
  308.  */ 
  309. function monsterinsights_get_upgrade_link() { 
  310.  
  311.   if ( class_exists( 'MonsterInsights' ) ) { 
  312.       // User is using MonsterInsights, so just take them to the Pricing page. 
  313.       // Note: On the Addons screen, if the user has a license, we won't hit this function,  
  314.       // as the API will tell us the direct URL to send the user to based on their license key,  
  315.       // so they see pro-rata pricing. 
  316.       return 'https://www.monsterinsights.com/pricing/?utm_source=proplugin&utm_medium=link&utm_campaign=WordPress'; 
  317.   } 
  318.  
  319.   $shareasale_id = monsterinsights_get_shareasale_id(); 
  320.    
  321.   // If at this point we still don't have an ID, we really don't have one! 
  322.   // Just return the standard upgrade URL. 
  323.   if ( empty( $shareasale_id ) ) { 
  324.       return 'https://www.monsterinsights.com/pricing/?utm_source=liteplugin&utm_medium=link&utm_campaign=WordPress'; 
  325.   } 
  326.  
  327.   // If here, we have a ShareASale ID 
  328.   // Return ShareASale URL with redirect. 
  329.   return 'https://www.shareasale.com/r.cfm?u=' . $shareasale_id . '&b=971799&m=69975&afftrack=&urllink=monsterinsights%2Ecom%2Flite%2F'; 
  330. }

  332. function monsterinsights_get_shareasale_id() { 
  333.  // Check if there's a constant. 
  334.   $shareasale_id = ''; 
  335.   if ( defined( 'MONSTERINSIGHTS_SHAREASALE_ID' ) ) { 
  336.       $shareasale_id = MONSTERINSIGHTS_SHAREASALE_ID; 
  337.   } 
  338.  
  339.   // If there's no constant, check if there's an option. 
  340.   if ( empty( $shareasale_id ) ) { 
  341.       $shareasale_id = get_option( 'monsterinsights_shareasale_id', '' ); 
  342.   } 
  343.  
  344.   // Whether we have an ID or not, filter the ID. 
  345.   $shareasale_id = apply_filters( 'monsterinsights_shareasale_id', $shareasale_id ); 
  346.   return $shareasale_id; 
  347. }
