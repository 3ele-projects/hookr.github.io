  58.   /** 
  59.    * Setup the data that is going to be tracked 
  60.    * 
  61.    * @access private 
  62.    * @return void 
  63.    */ 
  64.   private function setup_data() { 
  65.       $data = array(); 
  66.  
  67.       // Retrieve current theme info 
  68.       $theme_data = wp_get_theme(); 
  69.       $theme = $theme_data->Name . ' ' . $theme_data->Version; 
  70.       $tracking_mode = monsterinsights_get_option( 'tracking_mode', 'analytics' ); 
  71.       $events_mode = monsterinsights_get_option( 'events_mode', 'none' ); 
  72.  
  73.       if ( $tracking_mode === false ) { 
  74.           $tracking_mode = 'analytics'; 
  75.       } 
  76.  
  77.       if ( $events_mode === false ) { 
  78.           $events_mode = 'none'; 
  79.       } 
  80.  
  81.       $data['php_version'] = phpversion(); 
  82.       $data['mi_version'] = MONSTERINSIGHTS_VERSION; 
  83.       $data['wp_version'] = get_bloginfo( 'version' ); 
  84.       $data['server'] = isset( $_SERVER['SERVER_SOFTWARE'] ) ? $_SERVER['SERVER_SOFTWARE'] : ''; 
  85.       $data['over_time'] = get_option( 'monsterinsights_over_time', array() ); 
  86.       $data['multisite'] = is_multisite(); 
  87.       $data['url'] = home_url(); 
  88.       $data['theme'] = $theme; 
  89.       $data['email'] = get_bloginfo( 'admin_email' ); 
  90.       $data['key'] = monsterinsights_get_license(); 
  91.       $data['sas'] = monsterinsights_get_shareasale_id(); 
  92.       $data['setttings'] = monsterinsights_get_options(); 
  93.       $data['tracking_mode'] = $tracking_mode; 
  94.       $data['events_mode'] = $events_mode; 
  95.  
  96.       // Retrieve current plugin information 
  97.       if( ! function_exists( 'get_plugins' ) ) { 
  98.           include ABSPATH . '/wp-admin/includes/plugin.php'; 
  99.       } 
  100.  
  101.       $plugins = array_keys( get_plugins() ); 
  102.       $active_plugins = get_option( 'active_plugins', array() ); 
  103.  
  104.       foreach ( $plugins as $key => $plugin ) { 
  105.           if ( in_array( $plugin, $active_plugins ) ) { 
  106.               // Remove active plugins from list so we can show active and inactive separately 
  107.               unset( $plugins[ $key ] ); 
  108.           } 
  109.       } 
  110.  
  111.       $data['active_plugins'] = $active_plugins; 
  112.       $data['inactive_plugins'] = $plugins; 
  113.       $data['locale'] = get_locale(); 
  114.  
  115.       $this->data = $data; 
  116.   }

  118.   /** 
  119.    * Send the data to the EDD server 
  120.    * 
  121.    * @access private 
  122.    * @return void 
  123.    */ 
  124.   public function send_checkin( $override = false, $ignore_last_checkin = false ) { 
  125.  
  126.       if( ! $this->tracking_allowed() && ! $override ) { 
  127.           return false; 
  128.       } 
  129.  
  130.       // Send a maximum of once per day 
  131.       $last_send = $this->get_last_send(); 
  132.       if( is_numeric( $last_send ) && $last_send > strtotime( '-1 day' ) && ! $ignore_last_checkin ) { 
  133.           return false; 
  134.       } 
  135.  
  136.       $this->setup_data(); 
  137.  
  138.       $request = wp_remote_post( 'https://www.monsterinsights.com/?edd_action=checkin', array( 
  139.           'method' => 'POST',  
  140.           'timeout' => 20,  
  141.           'redirection' => 5,  
  142.           'httpversion' => '1.1',  
  143.           'blocking' => true,  
  144.           'body' => $this->data,  
  145.           'user-agent' => 'MI/' . MONSTERINSIGHTS_VERSION . '; ' . get_bloginfo( 'url' ) 
  146.  ) ); 
  147.  
  148.       if( is_wp_error( $request ) ) { 
  149.           return $request; 
  150.       } 
  151.  
  152.       update_option( 'mi_tracking_last_send', time() ); 
  153.  
  154.       return true; 
  155.  
  156.   }

  158.   /** 
  159.    * Check for a new opt-in on settings save 
  160.    * 
  161.    * This runs during the sanitation of General settings, thus the return 
  162.    * 
  163.    * @access public 
  164.    * @return array 
  165.    */ 
  166.   public function check_for_settings_optin() { 
  167.       if ( ! current_user_can( 'monsterinsights_save_settings' ) ) { 
  168.           return; 
  169.       } 
  170.  
  171.       // Send an intial check in on settings save 
  172.       $anonymous_data = isset( $_POST['anonymous_data'] ) ? 1 : 0; 
  173.       if ( $anonymous_data ) { 
  174.           $this->send_checkin( true ); 
  175.       } 
  176.  
  177.   }

  179.   /** 
  180.    * Check for a new opt-in via the admin notice 
  181.    *

  183.    * @return void 
  184.    */ 
  185.   public function check_for_optin() { 
  186.       if ( ! ( ! empty( $_REQUEST['mi_action'] ) && 'opt_into_tracking' === $_REQUEST['mi_action'] ) ) { 
  187.           return; 
  188.       } 
  189.  
  190.       if ( monsterinsights_get_option( 'anonymous_data', false ) ) { 
  191.           return; 
  192.       } 
  193.  
  194.       if ( ! current_user_can( 'monsterinsights_save_settings' ) ) { 
  195.           return; 
  196.       } 
  197.        
  198.       monsterinsights_update_option( 'anonymous_data', 1 ); 
  199.       $this->send_checkin( true ); 
  200.       update_option( 'monsterinsights_tracking_notice', 1 ); 
  201.   }

  209.   public function check_for_optout() {

  211.           return; 
  212.       } 
  213.  
  214.       if ( monsterinsights_get_option( 'anonymous_data', false ) ) {

  236.   /** 
  237.    * Schedule a weekly checkin 
  238.    * 
  239.    * @access public 
  240.    * @return void 
  241.    */ 
  242.   public function schedule_send() { 
  243.       // We send once a day (while tracking is allowed) to check in, which can be used to determine active sites 
  244.       if ( ! wp_next_scheduled( 'monsterinsights_daily_cron' ) ) { 
  245.           // Set the next event of fetching data 
  246.           wp_schedule_event( strtotime( date( 'Y-m-d', strtotime( 'tomorrow' ) ) . ' 00:01:00 ' ), 'daily', 'monsterinsights_daily_cron' ); 
  247.       } 
  248.   }

  250.   /** 
  251.    * Display the admin notice to users that have not opted-in or out 
  252.    * 
  253.    * @access public 
  254.    * @return void 
  255.    */ 
  256.   public function monsterinsights_admin_notice() { 
  257.  
  258.       $hide_notice = get_option( 'monsterinsights_tracking_notice' ); 
  259.  
  260.       if ( $hide_notice ) { 
  261.           return; 
  262.       } 
  263.  
  264.       if ( monsterinsights_get_option( 'anonymous_data', false ) ) { 
  265.           return; 
  266.       } 
  267.  
  268.       if ( ! current_user_can( 'monsterinsights_save_settings' ) ) { 
  269.           return; 
  270.       } 
  271.  
  272.       if ( 
  273.           stristr( network_site_url( '/' ), 'dev' ) !== false || 
  274.           stristr( network_site_url( '/' ), 'localhost' ) !== false || 
  275.           stristr( network_site_url( '/' ), ':8888' ) !== false // This is common with MAMP on OS X 
  276.  ) { 
  277.           update_option( 'monsterinsights_tracking_notice', '1' ); 
  278.       } else { 
  279.           $optin_url = add_query_arg( 'mi_action', 'opt_into_tracking' ); 
  280.           $optout_url = add_query_arg( 'mi_action', 'opt_out_of_tracking' ); 
  281.           echo '<div class="updated"><p>'; 
  282.           echo esc_html__( 'Allow MonsterInsights to track plugin usage? Opt-in to tracking and our newsletter to stay informed of the latest changes to MonsterInsights and help us ensure compatibility.', 'google-analytics-for-wordpress' ); 
  283.           echo ' <a href="' . esc_url( $optin_url ) . '" class="button-secondary">' . __( 'Allow', 'google-analytics-for-wordpress' ) . '</a>'; 
  284.           echo ' <a href="' . esc_url( $optout_url ) . '" class="button-secondary">' . __( 'Do not allow', 'google-analytics-for-wordpress' ) . '</a>'; 
  285.           echo '</p></div>'; 
  286.       } 
  287.   }
