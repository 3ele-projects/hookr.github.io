  245.   function _sanitize_text_fields( $str, $keep_newlines = false ) { 
  246.       $filtered = wp_check_invalid_utf8( $str ); 
  247.     
  248.       if ( strpos($filtered, '<') !== false ) { 
  249.           $filtered = wp_pre_kses_less_than( $filtered ); 
  250.           // This will strip extra whitespace for us. 
  251.           $filtered = wp_strip_all_tags( $filtered, false ); 
  252.     
  253.           // Use html entities in a special case to make sure no later 
  254.           // newline stripping stage could lead to a functional tag 
  255.           $filtered = str_replace("<\n", "<\n", $filtered); 
  256.       } 
  257.     
  258.       if ( ! $keep_newlines ) { 
  259.           $filtered = preg_replace( '/[\r\n\t ]+/', ' ', $filtered ); 
  260.       } 
  261.       $filtered = trim( $filtered ); 
  262.     
  263.       $found = false; 
  264.       while ( preg_match('/%[a-f0-9]{2}/i', $filtered, $match) ) { 
  265.           $filtered = str_replace($match[0], '', $filtered); 
  266.           $found = true; 
  267.       } 
  268.     
  269.       if ( $found ) { 
  270.           // Strip out the whitespace that may now exist after removing the octets. 
  271.           $filtered = trim( preg_replace('/ +/', ' ', $filtered) ); 
  272.       } 
  273.     
  274.       return $filtered; 
  275.   }
