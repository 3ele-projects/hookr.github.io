  127.       /**  
  128.        * Explanation of MonsterInsights core options

  130.        * By now your head is probably spinning trying to figure

  132.        * "monsterinsights" to "mi" in the options names to make this table easier 
  133.        * to read. 
  134.        * 
  135.        * Here's a basic rundown: 
  136.        * 
  137.        * mi_settings_version: Used to store the version  
  138.        *                         of the MI settings. We use this 
  139.        *                         so we can do upgrade routines where 
  140.        *                         we'd have to do different actions based 
  141.        *                         on the version the settings were installed 
  142.        *                         in. For example: if we made a mistake with  
  143.        *                         the value we saved as the default for  
  144.        *                         a select setting, we can detect the version 
  145.        *                         containing this mistake and correct it. 
  146.        * 
  147.        * mi_current_version:  This starts with the actual version MI was 
  148.        *                         installed on. We use this version to  
  149.        *                         determine whether or not a site needs 
  150.        *                         to run one of the behind the scenes 
  151.        *                         MI upgrade routines. This version is updated 
  152.        *                         every time a minor or major background upgrade 
  153.        *                         routine is run. Generally lags behind the  
  154.        *                         MONSTERINSIGHTS_VERSION constant by at most a couple minor 
  155.        *                         versions. Never lags behind by 1 major version 
  156.        *                         or more. 
  157.        * 
  158.        * mi_db_version:         This is different from mi_current_version. 
  159.        *                         Unlike the former, this is used to determine 
  160.        *                         if a site needs to run a *user* initiated 
  161.        *                         upgrade routine (see MI_Upgrade class). This 
  162.        *                         value is only update when a user initiated 
  163.        *                         upgrade routine is done. Because we do very 
  164.        *                         few user initiated upgrades compared to  
  165.        *                         automatic ones, this version can lag behind by 
  166.        *                         2 or even 3 major versions. Generally contains 
  167.        *                         the current major version. 
  168.        * 
  169.        * mi_settings:            Returned by monsterinsights_get_option_name(), this 
  170.        *                         is actually "monsterinsights_settings" for both pro 
  171.        *                         and lite version. However we use a helper function to  
  172.        *                         retrieve the option name in case we ever decide down the 
  173.        *                         road to maintain seperate options for the Lite and Pro versions. 
  174.        *                          If you need to access MI's settings directly, (as opposed to our 
  175.        *                          monsterinsights_get_option helper which uses the option name helper 
  176.        *                          automatically), you should use this function to get the  
  177.        *                          name of the option to retrieve. 
  178.        * 
  179.        * yst_ga:                Yoast's old settings option. We no longer use this, though 
  180.        *                         for backwards compatibility reasons we store the updated settings 
  181.        *                         in this just for a little while longer. These settings are migrated 
  182.        *                         to the new settings option when you upgrade to MonsterInsights 
  183.        *                         6.0 or higher automatically. 
  184.        * 
  185.        * yst_* & yoast_*:        These are options from when the plugin was developed by 
  186.        *                         Yoast, and also of the few point releases we did after 
  187.        *                         the acquisition. Note, while we currently do backcompat 
  188.        *                         on some of these options so other plugins will continue working 
  189.        *                         please realize there will be a point in the near future that we  
  190.        *                         will no longer support them. Please do not use them anymore. 
  191.        */

  503.                       if ( isset( $options['analytics_profile_code'] ) ) {

  505.                       } 
  506.                       if ( isset( $options['analytics_profile'] ) ) { 
  507.                           unset( $options['analytics_profile'] ); 
  508.                       } 
  509.                       delete_option( 'yoast-ga-access_token' ); 
  510.                       delete_option( 'yoast-ga-refresh_token' ); 
  511.                       delete_option( 'yst_ga_api' ); 
  512.                   }  else { 
  513.                   // if UA in profile profiles, remove others and use that 
  514.                       if ( ! empty( $options['analytics_profile' ] ) && ! empty( $profiles['ga_api_response_accounts'] ) && is_array( $profiles['ga_api_response_accounts'] ) ) { 
  515.                           foreach ( $profiles as $account ) { 
  516.                               foreach ( $account['items'] as $profile ) { 
  517.                                   foreach ( $profile['items'] as $subprofile ) { 
  518.                                       if ( isset( $subprofile['id'] ) && $subprofile['id'] == $options['analytics_profile' ] ) { 
  519.                                           $options['manual_ua_code'] = $subprofile['ua_code']; 
  520.                                           break 3; 
  521.                                       } 
  522.                                   } 
  523.                               } 
  524.                           } 
  525.                       } 
  526.                       delete_option( 'yoast-ga-access_token' ); 
  527.                       delete_option( 'yoast-ga-refresh_token' ); 
  528.                       delete_option( 'yst_ga_api' ); 
  529.                   } 
  530.               } 
  531.            
  532.           if ( isset( $options['manual_ua_code_field'] ) ) { 
  533.               unset( $options['manual_ua_code_field'] ); 
  534.           } 
  535.  
  536.       // oAuth Stir Data Tank 
  537.           // Will happen automatically as cron_last_run set to 25 hours ago 
  538.  
  539.       // Add oAuth version  
  540.               $options['oauth_version'] = '1.0.0'; 
  541.  
  542.       $data = array( 
  543.           'installed_version' => MONSTERINSIGHTS_VERSION,  
  544.           'installed_date' => time(),  
  545.           'installed_pro' => monsterinsights_is_pro_version(),  
  546.  ); 
  547.  
  548.       update_option( 'monsterinsights_over_time', $data ); 
  549.        
  550.       // Add the cron job 
  551.       if ( ! wp_next_scheduled( 'monsterinsights_daily_cron' ) ) { 
  552.           // Set the next event of fetching data 
  553.           wp_schedule_event( strtotime( date( 'Y-m-d', strtotime( 'tomorrow' ) ) . ' 00:05:00 ' ), 'daily', 'monsterinsights_daily_cron' ); 
  554.       } 
  555.        
  556.       // Finish up 
  557.           // Save the new settings 
  558.           $this->new_settings = $options; 
  559.   }
