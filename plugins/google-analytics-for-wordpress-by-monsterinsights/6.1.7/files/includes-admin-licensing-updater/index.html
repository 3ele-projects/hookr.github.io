  150.    * Infuse plugin update details when WordPress runs its update checker. 
  151.    *

  154.    * @param object $value  The WordPress update object. 
  155.    * @return object $value Amended WordPress update object on success, default if object is empty.

  157.   public function update_plugins_filter( $value ) { 
  158.  
  159.       // If no update object exists, return early. 
  160.       if ( empty( $value ) ) { 
  161.           return $value; 
  162.       } 
  163.  
  164.       // Run update check by pinging the external API. If it fails, return the default update object. 
  165.       if ( ! $this->update ) { 
  166.           $this->update = $this->perform_remote_request( 'get-plugin-update', array( 'tgm-updater-plugin' => $this->plugin_slug ) ); 
  167.           if ( ! $this->update || ! empty( $this->update->error ) ) { 
  168.               $this->update = false; 
  169.               return $value; 
  170.           } 
  171.       } 
  172.  
  173.       // Infuse the update object with our data if the version from the remote API is newer. 
  174.       if ( isset( $this->update->new_version ) && version_compare( $this->version, $this->update->new_version, '<' ) ) { 
  175.           // The $plugin_update object contains new_version, package, slug and last_update keys. 
  176.           $value->response[$this->plugin_path] = $this->update; 
  177.       } 
  178.  
  179.       // Return the update object. 
  180.       return $value; 
  181.  
  182.   }

  282.   /** 
  283.    * Queries the remote URL via wp_remote_post and returns a json decoded response. 
  284.    * 
  285.    * @since 6.0.0 
  286.    * 
  287.    * @param string $action        The name of the $_POST action var. 
  288.    * @param array $body           The content to retrieve from the remote URL. 
  289.    * @param array $headers        The headers to send to the remote URL. 
  290.    * @param string $return_format The format for returning content from the remote URL. 
  291.    * @return string|bool          Json decoded response on success, false on failure. 
  292.    */ 
  293.   public function perform_remote_request( $action, $body = array(), $headers = array(), $return_format = 'json' ) { 
  294.  
  295.       // Build the body of the request. 
  296.       $body = wp_parse_args( 
  297.           $body,  
  298.           array( 
  299.               'tgm-updater-action' => $action,  
  300.               'tgm-updater-key' => $this->key,  
  301.               'tgm-updater-wp-version' => get_bloginfo( 'version' ),  
  302.               'tgm-updater-referer' => site_url(),  
  303.               'tgm-updater-mi-version' => MONSTERINSIGHTS_VERSION,  
  304.               'tgm-updater-is-pro' => monsterinsights_is_pro_version(),  
  305.  ) 
  306.  ); 
  307.       $body = http_build_query( $body, '', '&' ); 
  308.  
  309.       // Build the headers of the request. 
  310.       $headers = wp_parse_args( 
  311.           $headers,  
  312.           array( 
  313.               'Content-Type' => 'application/x-www-form-urlencoded',  
  314.               'Content-Length' => strlen( $body ) 
  315.  ) 
  316.  ); 
  317.  
  318.       // Setup variable for wp_remote_post. 
  319.       $post = array( 
  320.           'headers' => $headers,  
  321.           'body' => $body 
  322.  ); 
  323.  
  324.       // Perform the query and retrieve the response. 
  325.       $response = wp_remote_post( esc_url_raw( $this->remote_url ), $post ); 
  326.       $response_code = wp_remote_retrieve_response_code( $response ); 
  327.       $response_body = wp_remote_retrieve_body( $response ); 
  328.  
  329.       // Bail out early if there are any errors. 
  330.       if ( 200 != $response_code || is_wp_error( $response_body ) ) { 
  331.           return false; 
  332.       } 
  333.  
  334.       // Return the json decoded content. 
  335.       return json_decode( $response_body ); 
  336.  
  337.   }
