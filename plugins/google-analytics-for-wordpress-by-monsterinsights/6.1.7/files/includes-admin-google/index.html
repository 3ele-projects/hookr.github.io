  145.   private function get_status() { 
  146.       $status = 'valid'; 
  147.       if ( ! empty( $this->profile ) ) { 
  148.           // We are using oAuth 
  149.            
  150.           $last_run = monsterinsights_get_option( 'cron_last_run', false ); 
  151.           $failed = monsterinsights_get_option( 'cron_failed', false ); 
  152.           $dash_dis = monsterinsights_get_option( 'dashboards_disabled', false ); 
  153.  
  154.           // See if issue connecting or expired 
  155.           if ( ! $dash_dis && $failed && ( $last_run === false || monsterinsights_hours_between( $last_run ) >= 48 ) ) {  
  156.               $status = 'blocked'; 
  157.           } 
  158.  
  159.           $access_token = $this->client->get_access_token(); 
  160.  
  161.           // Check to make sure access token is there and not expired 
  162.           if ( empty( $access_token ) || empty( $access_token['expires'] ) || current_time( 'timestamp' ) >= $access_token['expires'] ) { 
  163.               $status = 'expired'; 
  164.               return $status; 
  165.           } 
  166.  
  167.           // See if needs permissions 
  168.           if ( version_compare( $this->oauth_version, '1.0', '<' ) ) {  
  169.               $status = 'needs-permissions'; 
  170.           } 
  171.            
  172.       } else if ( ! empty( $this->ua ) ) { 
  173.           // We are using manual 
  174.            
  175.       } else { 
  176.           // We do not have oAuth or manual active 
  177.           $status = 'empty'; 
  178.       } 
  179.  
  180.       return $status; 
  181.   }

  287.    * Get accounts request 
  288.    * 
  289.    * @param array $response 
  290.    * 
  291.    * @return mixed 
  292.    */ 
  293.   private function get_profiles_request() { 
  294.       global $wp_version; 
  295.       $version = str_replace( '-src', '', $wp_version ); 
  296.       $accounts = array(); 
  297.       $start_index = 1; 
  298.       $paginate = false; 
  299.       $continue = true; 
  300.       while ( $continue ) { 
  301.           $body = array( 
  302.               'max-results' => 1000,  
  303.               'start-index' => $paginate ? $start_index + 1000 : $start_index,  
  304.  ); 
  305.           if ( $paginate ) { 
  306.               $start_index = $start_index + 1000; 
  307.           } 
  308.           $response = $this->client->do_request( 'https://www.googleapis.com/analytics/v3/management/accounts/~all/webproperties/~all/profiles', false, 'GET', $body ); 
  309.           if ( ! empty( $response ) ) { 
  310.               $response = array( 
  311.                   'response' => array( 'code' => $this->client->get_http_response_code() ),  
  312.                   'body' => json_decode( $response->getResponseBody(), true ),  
  313.  ); 
  314.           } else { 
  315.               if ( version_compare( $version, '4.6', '<' ) ) { 
  316.                   return esc_html__( 'MonsterInsights requires WordPress version 4.6 or newer to use oAuth. Please update your WordPress version.', 'google-analytics-for-wordpress' ); 
  317.               } else { 
  318.                   if ( ! empty( $accounts ) ) { 
  319.                       return $accounts; 
  320.                   } else { 
  321.                       return esc_html__( 'Google Analytics had a connection error or your Google account is not signed up for Google Analytics.', 'google-analytics-for-wordpress' ); 
  322.                   } 
  323.               } 
  324.           } 
  325.            
  326.           if ( isset( $response['response']['code'] ) && $response['response']['code'] == 200 ) { 
  327.               if ( ! empty( $response['body']['items'] ) && is_array( $response['body']['items'] ) ) { 
  328.                   foreach ( $response['body']['items'] as $item ) { 
  329.  
  330.                       // Only deal with web properties, not apps. 
  331.                       if ( isset( $item['type'] ) && 'WEB' != $item['type'] ) { 
  332.                           continue; 
  333.                       } 
  334.                        
  335.                       if ( empty( $accounts[ $item['accountId'] ] ) ) { 
  336.                           $accounts[ $item['accountId'] ] = array(  
  337.                               'id' => $item['accountId'],  
  338.                               'ua_code' => $item['webPropertyId'],  
  339.                               'parent_name' => $item['websiteUrl'],  
  340.                               'items' => array(),  
  341.  ); 
  342.                       } 
  343.  
  344.                       if ( empty( $accounts[ $item['accountId'] ]['items'][ $item['internalWebPropertyId'] ] ) ) { 
  345.                           $accounts[ $item['accountId'] ]['items'][ $item['internalWebPropertyId'] ]= array(  
  346.                               'id' => $item['webPropertyId'],  
  347.                               'name' => $item['websiteUrl'],  
  348.                               'items' => array(),  
  349.  ); 
  350.                       } 
  351.  
  352.                       if ( empty( $accounts[ $item['accountId'] ]['items'][ $item['internalWebPropertyId'] ]['items'][ $item['id'] ] ) ) { 
  353.                           $accounts[ $item['accountId'] ]['items'][ $item['internalWebPropertyId'] ]['items'][ $item['id'] ] = array(  
  354.                               'name' => $item['name'] . ' (' . $item['webPropertyId'] . ')',  
  355.                               'ua_code' => $item['webPropertyId'],  
  356.                               'id' => $item['id'],  
  357.  ); 
  358.                       } 
  359.                   } 
  360.               } 
  361.           } else if ( isset( $response['response']['code'] ) && $response['response']['code'] !== 200 && ! $paginate ) { 
  362.               if ( version_compare( $version, '4.6', '<' ) ) { 
  363.                   return esc_html__( 'MonsterInsights requires WordPress version 4.6 or newer to use oAuth. Please update your WordPress version.', 'google-analytics-for-wordpress' ); 
  364.               } else { 
  365.                   if ( ! empty( $accounts ) ) { 
  366.                       return $accounts; 
  367.                   } else { 
  368.                       $code = isset( $response['response']['code'] ) ? $response['response']['code'] : 'Unknown'; 
  369.                       $type = isset( $response['body']['error']['errors'][0]['reason'] ) ? $response['body']['error']['errors'][0]['reason'] : false; 
  370.                       if ( $type === 'insufficientPermissions' ) { 
  371.                           return esc_html__( 'Please ensure your Google Account is signed up for Google Analytics.', 'google-analytics-for-wordpress' ); 
  372.                       } else if ( $type === 'badRequest' ) { 
  373.                           return esc_html__( 'Bad Request. Please contact support.', 'google-analytics-for-wordpress' ); 
  374.                       } else if ( $type === 'dailyLimitExceeded' ) { 
  375.                           return esc_html__( 'Daily Limit Exceeded. Please contact support.', 'google-analytics-for-wordpress' ); 
  376.                       } else if ( $type === 'userRateLimitExceeded' ) { 
  377.                           return esc_html__( 'User Rate Limit Exceeded. Wait 2 minutes and then try again.', 'google-analytics-for-wordpress' ); 
  378.                       } else if ( $type === 'rateLimitExceeded' ) { 
  379.                           return esc_html__( 'Project Rate Limit Exceeded. Wait 2 minutes and then try again.', 'google-analytics-for-wordpress' ); 
  380.                       } else if ( $type === 'quotaExceeded' ) { 
  381.                           return esc_html__( 'Project Rate Limit Quota Exceeded. Wait 2 minutes and then try again.', 'google-analytics-for-wordpress' ); 
  382.                       } else if ( $type === 'internalServerError' || $type === 'backendError' ) { 
  383.                           return esc_html__( 'Google Analytics is having API issues on their side. Wait 2 minutes and then try again.', 'google-analytics-for-wordpress' ); 
  384.                       } else { 
  385.                           return sprintf( esc_html__( 'Google Analytics had a connection error. Error code: %1$s. Reason: %2$s', 'google-analytics-for-wordpress' ), $code, $type ); 
  386.                       } 
  387.                   } 
  388.               } 
  389.           } 
  390.  
  391.           if ( isset( $response['body']['totalResults'] ) && $start_index < $response['body']['totalResults'] && ! empty( $response['body']['nextLink'] ) ) { 
  392.               $paginate = true; 
  393.           } else { 
  394.               $continue = false; 
  395.           } 
  396.       } 
  397.       return $accounts; 
  398.   }

  491.   public function google_auth_view() { 
  492.       $view = isset( $_POST['view'] ) && in_array( $_POST['view'], array( 'prestart', 'start', 'enterkey', 'selectprofile', 'done' ) ) ? $_POST['view'] : ''; 
  493.       $reauth = isset( $_POST['reauth'] ) && $_POST['reauth'] && $_POST['reauth'] !== 'false'  ? true : false; 
  494.  
  495.       if ( ! current_user_can( 'monsterinsights_save_settings' ) ) { 
  496.           echo esc_html__( 'Permission Denied', 'google-analytics-for-wordpress'); 
  497.           wp_die(); 
  498.       } 
  499.  
  500.       // We run the save routines, if required, for a view, and then after that send back the results + next view 
  501.       $nextview = array(); 
  502.  
  503.       switch ( $view ) { 
  504.           case 'start': 
  505.               if ( ! $this->is_wp_blocking_google() && ! $this->is_google_on_blacklist() ) { 
  506.                   $auth_url = $this->create_auth_url(); 
  507.                   $nextview = monsterinsights_google_auth_enterkey_view( $reauth, $auth_url ); 
  508.               } else { 
  509.                   $error = esc_html__( 'Cannot connect to Google', 'google-analytics-for-wordpress' ); 
  510.                   $nextview = monsterinsights_google_auth_error_view( $reauth, $error ); 
  511.               } 
  512.               break; 
  513.  
  514.           case 'enterkey': 
  515.               $auth_key = ! empty( $_POST['stepdata'] ) ? sanitize_text_field( $_POST['stepdata'] ) : ''; 
  516.               if ( $auth_key ) { 
  517.                   if ( $this->test_authkey( $auth_key ) ) { 
  518.                       $profiles = $this->get_profiles_request(); 
  519.                       delete_option( 'monsterinsights_get_profiles' ); 
  520.                       update_option( 'monsterinsights_get_profiles', $profiles ); 
  521.                       if ( ! empty( $profiles ) && is_array( $profiles ) ) { 
  522.                           $select = $this->ga_select( $profiles ); 
  523.                           $nextview = monsterinsights_google_auth_selectprofile_view( $reauth, $select ); 
  524.                       } else if ( ! empty( $profiles ) && is_string( $profiles ) ) { 
  525.                           // Error from Google 
  526.                           $auth_url = $this->create_auth_url(); 
  527.                           $this->client->clear_data(); 
  528.                           $this->set_test_client(); 
  529.                           $nextview = monsterinsights_google_auth_enterkey_view( $reauth, $auth_url, esc_html( $profiles ) ); 
  530.                       } else { 
  531.                           // No profiles or not enough permissions 
  532.                           $auth_url = $this->create_auth_url(); 
  533.                           $this->client->clear_data(); 
  534.                           $this->set_test_client(); 
  535.                           $nextview = monsterinsights_google_auth_enterkey_view( $reauth, $auth_url, esc_html__( 'No profiles viewable for that account. Please use another account.', 'google-analytics-for-wordpress' ) ); 
  536.                       } 
  537.                   } else { 
  538.                       // if bad authentication error message 
  539.                       $auth_url = $this->create_auth_url(); 
  540.                       $this->client->clear_data(); 
  541.                       $this->set_test_client(); 
  542.                       $nextview = monsterinsights_google_auth_enterkey_view( $reauth, $auth_url, esc_html__( 'Bad Google Code. Please try again.', 'google-analytics-for-wordpress' ) ); 
  543.                   } 
  544.               } else { 
  545.                   $auth_url = $this->create_auth_url(); 
  546.                   $this->client->clear_data(); 
  547.                   $this->set_test_client(); 
  548.                   // if no auth key error message 
  549.                   $nextview = monsterinsights_google_auth_enterkey_view( $reauth, $auth_url, esc_html__( 'Please paste in your Google code.', 'google-analytics-for-wordpress' ) ); 
  550.               } 
  551.               break; 
  552.  
  553.           case 'selectprofile': 
  554.               $profile = ! empty( $_POST['stepdata'] ) ? absint( sanitize_text_field( $_POST['stepdata'] ) ) : ''; 
  555.               if ( ! empty( $profile ) ) { 
  556.                   $this->set_test_client(); 
  557.                   $profile = $this->find_selected_profile( $profile ); 
  558.                   if ( ! empty( $profile ) ) { 
  559.  
  560.                       $this->clear_manual_data(); // Just in case we were manual, clear out UA 
  561.  
  562.                       $this->client->move_test_to_live(); 
  563.                       $this->save_selected_profile( $profile ); 
  564.                       $this->set_client(); 
  565.                       $this->reinitialize(); 
  566.  
  567.                       // Refresh reporting data 
  568.                       $this->base->reports->refresh_aggregate_data(); 
  569.  
  570.                       $nextview = monsterinsights_google_auth_done_view( $reauth ); 
  571.                   } else { 
  572.                       // Invalid profile selected 
  573.                       $profiles = get_option( 'monsterinsights_get_profiles', array() ); 
  574.                       $select = $this->ga_select( $profiles ); 
  575.                       $nextview = monsterinsights_google_auth_selectprofile_view( $reauth, $select, esc_html__( 'Invalid profile selected.', 'google-analytics-for-wordpress' ) ); 
  576.                   } 
  577.               } else {  
  578.                   // No profile selected 
  579.                   $profiles = get_option( 'monsterinsights_get_profiles', array() ); 
  580.                   $select = $this->ga_select( $profiles ); 
  581.                   $nextview = monsterinsights_google_auth_selectprofile_view( $reauth, $select, esc_html__( 'Please select a profile.', 'google-analytics-for-wordpress' ) ); 
  582.               } 
  583.               break; 
  584.  
  585.           case 'done': 
  586.               $nextview = monsterinsights_google_auth_done_view( $reauth ); 
  587.               break; 
  588.                
  589.           case 'prestart': 
  590.           default: 
  591.                $nextview = monsterinsights_google_auth_start_view( $reauth ); 
  592.               break; 
  593.       } 
  594.       echo $nextview; 
  595.       wp_die(); 
  596.   }

  600.       $reauth = isset( $_POST['reauth'] ) && $_POST['reauth'] && $_POST['reauth'] !== 'false'  ? true : false;

  626.   private function ga_select( $profiles = array() ) { 
  627.       if ( empty( $profiles ) || ! is_array( $profiles ) ) {

  629.       }

  635.           } 
  636.       }

'" name="monsterinsights_step_data" class="monsterinsights-select300 monsterinsights_select_ga_profile" id="monsterinsights_step_data" style="width:80%;margin-left:10%;margin-right:10%;background-color: #FFF;">'; 
  643.       $select .= '<option></option>'; 
  644.  
  645.       if ( count( $values ) >= 1 ) {
