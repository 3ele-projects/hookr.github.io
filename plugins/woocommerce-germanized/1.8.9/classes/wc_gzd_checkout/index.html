  332.      * Recalculate fee taxes to split tax based on different tax rates contained within cart 
  333.      *   
  334.      * @param  WC_Cart $cart 
  335.      */ 
  336.     public function do_fee_tax_calculation( WC_Cart $cart ) { 
  337.         if ( get_option( 'woocommerce_gzd_fee_tax' ) != 'yes' ) 
  338.             return; 
  339.         if ( ! empty( $cart->fees ) ) { 
  340.             $tax_shares = wc_gzd_get_cart_tax_share( 'fee' ); 
  341.             foreach ( $cart->fees as $key => $fee ) { 
  342.                 if ( ! $fee->taxable && get_option( 'woocommerce_gzd_fee_tax_force' ) != 'yes' ) 
  343.                     continue;     
  344.                 // Calculate gross price if necessary 
  345.                 if ( $fee->taxable ) { 
  346.                     $fee_tax_rates = WC_Tax::get_rates( $fee->tax_class ); 
  347.                     $fee_tax = WC_Tax::calc_tax( $fee->amount, $fee_tax_rates, false ); 
  348.                     $fee->amount += array_sum( $fee_tax ); 
  349.                 } 
  350.                 // Set fee to nontaxable to avoid WooCommerce default tax calculation 
  351.                 $fee->taxable = false; 
  352.                 // Calculate tax class share 
  353.                 if ( ! empty( $tax_shares ) ) { 
  354.                     $fee_taxes = array(); 
  355.                     foreach ( $tax_shares as $rate => $class ) { 
  356.                         $tax_rates = WC_Tax::get_rates( $rate ); 
  357.                         $tax_shares[ $rate ][ 'fee_tax_share' ] = $fee->amount * $class[ 'share' ]; 
  358.                         $tax_shares[ $rate ][ 'fee_tax' ] = WC_Tax::calc_tax( ( $fee->amount * $class[ 'share' ] ), $tax_rates, true ); 
  359.                         $fee_taxes += $tax_shares[ $rate ][ 'fee_tax' ]; 
  360.                     } 
  361.                     foreach ( $tax_shares as $rate => $class ) { 
  362.                         $cart->fees[ $key ]->tax_data = $cart->fees[ $key ]->tax_data + $class[ 'fee_tax' ]; 
  363.                     } 
  364.                     // Add fee taxes to cart taxes 
  365.                     foreach ( array_keys( $cart->taxes + $fee_taxes ) as $sub ) { 
  366.                         $cart->taxes[ $sub ] = ( isset( $fee_taxes[ $sub ] ) ? $fee_taxes[ $sub ] : 0 ) + ( isset( $cart->taxes[ $sub ] ) ? $cart->taxes[ $sub ] : 0 ); 
  367.                     } 
  368.                     // Update fee 
  369.                     $cart->fees[ $key ]->tax = array_sum( $cart->fees[ $key ]->tax_data ); 
  370.                     $cart->fees[ $key ]->amount = $cart->fees[ $key ]->amount - $cart->fees[ $key ]->tax; 
  371.                 } 
  372.             } 
  373.         } 
  374.     }
