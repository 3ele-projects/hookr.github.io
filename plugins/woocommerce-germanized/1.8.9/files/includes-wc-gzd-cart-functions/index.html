  175.  * Calculates tax share for shipping/fees

  180. function wc_gzd_get_cart_tax_share( $type = 'shipping' ) { 
  181.    
  182.   $cart = WC()->cart->get_cart(); 
  183.   $tax_shares = array(); 
  184.   $item_totals = 0; 
  185.    
  186.   // Get tax classes and tax amounts 
  187.   if ( ! empty( $cart ) ) { 
  188.        
  189.       foreach ( $cart as $key => $item ) { 
  190.            
  191.           $_product = apply_filters( 'woocommerce_cart_item_product', $item[ 'data' ], $item, $key ); 
  192.            
  193.           // Dont calculate share if is shipping and product is virtual or vat exception 
  194.           if ( $type == 'shipping' && $_product->is_virtual() || ( wc_gzd_get_gzd_product( $_product )->is_virtual_vat_exception() && $type == 'shipping' ) ) 
  195.               continue; 
  196.            
  197.           $class = $_product->get_tax_class(); 
  198.            
  199.           if ( ! isset( $tax_shares[ $class ] ) ) { 
  200.               $tax_shares[ $class ] = array(); 
  201.               $tax_shares[ $class ][ 'total' ] = 0; 
  202.               $tax_shares[ $class ][ 'key' ] = ''; 
  203.           } 
  204.            
  205.           $tax_shares[ $class ][ 'total' ] += ( $item[ 'line_total' ] + $item[ 'line_tax' ] );  
  206.           $tax_shares[ $class ][ 'key' ] = key( $item[ 'line_tax_data' ][ 'total' ] ); 
  207.           $item_totals += ( $item[ 'line_total' ] + $item[ 'line_tax' ] );  
  208.       } 
  209.   } 
  210.    
  211.   if ( ! empty( $tax_shares ) ) { 
  212.  
  213.       $default = ( $item_totals == 0 ? 1 / sizeof( $tax_shares ) : 0 ); 
  214.  
  215.       foreach ( $tax_shares as $key => $class ) 
  216.           $tax_shares[ $key ][ 'share' ] = ( $item_totals > 0 ? $class[ 'total' ] / $item_totals : $default ); 
  217.  
  218.   } 
  219.  
  220.   return $tax_shares; 
  221. }
