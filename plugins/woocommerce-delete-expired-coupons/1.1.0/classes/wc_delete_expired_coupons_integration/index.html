  148.      * Delete expired coupons everytime somebody visit the web site

  152.     public function delete_expired_coupons() { 
  153.         global $pagenow; 
  154.  
  155.         if  ( //Do nothing if on new/edit coupon pages 
  156.             ( 'post-new.php' == $pagenow || 'post.php' == $pagenow || ( isset( $_GET['post_type'] ) && 'shop_coupon' == $_GET['post_type'] ) ) || 
  157.             //Do nothing if in trash page 
  158.             ( 'edit.php' == $pagenow && isset( $_GET['post_status'] ) && 'trash' == $_GET['post_status'] ) || 
  159.             //Do nothing if doing autosave 
  160.             ( defined( 'DOING_AUTOSAVE' ) && DOING_AUTOSAVE ) || 
  161.             //Doing AJAX 
  162.             ( is_admin() && defined('DOING_AJAX') && DOING_AJAX ) ) 
  163.         { 
  164.             return; 
  165.         } 
  166.  
  167.  
  168.         $args = array( 
  169.             'posts_per_page' => -1,  
  170.             'post_type' => 'shop_coupon',  
  171.             'post_status' => 'publish',  
  172.             'meta_query' => array( 
  173.                 'relation' => 'AND',  
  174.                 array( 
  175.                     'key' => 'expiry_date',  
  176.                     'value' => current_time( 'Y-m-d' ),  
  177.                     'compare' => '<=' 
  178.  ),  
  179.                 array( 
  180.                     'key' => 'expiry_date',  
  181.                     'value' => '',  
  182.                     'compare' => '!=' 
  183.  ) 
  184.  ) 
  185.  ); 
  186.  
  187.         $coupons = get_posts( $args ); 
  188.  
  189.         if ( ! empty( $coupons ) ) { 
  190.  
  191.             if ( ! $this->_should_run() ) { 
  192.                 add_action( 'admin_notices', array( $this, 'scheduled_run_notice' ) ); 
  193.                 return; 
  194.             } 
  195.  
  196.             //Delete the undeleted coupons transient 
  197.             delete_transient( 'woocommerce_unable_to_delete_expired_coupons' ); 
  198.             $unable_to_delete = array(); 
  199.             foreach( $coupons as $coupon ) { 
  200.                 //Check if the expired coupons should be kept for X time and if this time expired, then trash/delete them 
  201.                 $current_time = current_time( 'timestamp' ); 
  202.                 if ( $current_time - $this->_keeping_time >= strtotime( get_post_meta( $coupon->ID, 'expiry_date', true ) ) ) { 
  203.                     if ( 'no' === $this->force_deletion ) { 
  204.                         //Trash coupon 
  205.                         if ( false === wp_trash_post( $coupon->ID ) ) { 
  206.                             $unable_to_delete[ $coupon->ID ] = $coupon->post_title; 
  207.                         } 
  208.                     } else { 
  209.                         //Delete coupon 
  210.                         if ( false === wp_delete_post( $coupon->ID ) ) { 
  211.                             $unable_to_delete[ $coupon->ID ] = $coupon->post_title; 
  212.                         } 
  213.                     } 
  214.                 } 
  215.             } 
  216.  
  217.             if ( ! empty( $unable_to_delete ) ) { 
  218.                 set_transient( 'woocommerce_unable_to_delete_expired_coupons', $unable_to_delete ); 
  219.             } 
  220.         } 
  221.     }

  276.     private function _calculate_keeping_time() {

  279.  
  280.         switch( $keep_expired_coupons_type ) {

  282.             case 'hours':   $this->_keeping_time = HOUR_IN_SECONDS * $keep_expired_coupons; break; 
  283.             case 'days':    $this->_keeping_time = DAY_IN_SECONDS * $keep_expired_coupons; break;
