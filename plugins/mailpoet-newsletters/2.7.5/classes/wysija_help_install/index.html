  457.     function createTables($sql_file, $main_table_ms=false) { 
  458.         // prepare some parameters 
  459.         $model_user=WYSIJA::get('user', 'model'); 
  460.         $prefix = $model_user->getPrefix(); 
  461.         $array_tables_to_test = array('user_list', 'user', 'list', 'campaign', 'campaign_list', 'email', 'user_field', 'queue', 'user_history', 'email_user_stat', 'url', 'email_user_url', 'url_mail'); 
  462.  
  463.         // if we are running a ms file we use a different prefix and we verify different tables

  467.         } 
  468.  
  469.  
  470.         // load the sql file, read it and separate the queries 
  471.         $handle = fopen($sql_file, 'r'); 
  472.         $query = fread($handle, filesize($sql_file)); 
  473.         fclose($handle); 
  474.  
  475.         $query=str_replace('CREATE TABLE IF NOT EXISTS `', 'CREATE TABLE IF NOT EXISTS `'.$prefix, $query); 
  476.         $queries=explode('-- QUERY ---', $query); 
  477.  
  478.         // execute the queries one by one 
  479.         global $wpdb; 
  480.         $has_errors = false; 
  481.  
  482.         foreach($queries as $qry) { 
  483.             $last_error = $wpdb->last_error; 
  484.             $wpdb->query($qry); 
  485.  
  486.             if( !empty($wpdb->last_error) && $last_error != $wpdb->last_error ) { 
  487.                 $this->notice($wpdb->last_error); 
  488.                 $has_errors = true; 
  489.             } 
  490.  
  491.         } 
  492.  
  493.         // list the tables that haven't been created 
  494.         $missingtables=array(); 
  495.         foreach($array_tables_to_test as $table_name) { 
  496.             if(!$model_user->query("SHOW TABLES like '".$prefix.$table_name."';")) { 
  497.                 $missingtables[]=$prefix.$table_name; 
  498.             } 
  499.         } 
  500.  
  501.         // return the result 
  502.         if($missingtables) { 
  503.             $this->error(sprintf(__('These tables could not be created on installation: %1$s', WYSIJA), implode(', ', $missingtables)), 1);

  539.         if($targetDir === FALSE) {

  542.         } else {

  561.     function rrmdir($dir) {

  565.         if ($file != '.' && $file != '..') $this->rrmdir("$dir".DS."$file"); 
  566.         rmdir($dir); 
  567.       } 
  568.       else if (file_exists($dir)) { 
  569.           $dir=str_replace('/', DS, $dir); 
  570.           unlink($dir); 
  571.       }
