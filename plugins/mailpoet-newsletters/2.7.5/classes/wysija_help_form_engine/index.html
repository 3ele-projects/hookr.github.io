  665.     protected function get_validation_class($block) { 
  666.         $rules = array(); 
  667.  
  668.         // if it's the email field, it's mandatory and needs to be valid 
  669.         if($block['field'] === 'email') { 
  670.             $rules[] = 'required'; 
  671.             $rules[] = 'custom[email]'; 
  672.         } 
  673.  
  674.         // if it's the list field, at least one option needs to be selected 
  675.         if($block['field'] === 'list') { 
  676.             $rules[] = 'required'; 
  677.         } 
  678.  
  679.         // check if the field is required 
  680.         if(isset($block['params']['required']) && (bool)$block['params']['required'] === true) { 
  681.             $rules[] = 'required'; 
  682.         } 
  683.  
  684.         // check for validation rules 
  685.         if(isset($block['params']['validate'])) { 
  686.             if(is_array($block['params']['validate'])) { 
  687.                 // handle multiple validation rules 
  688.                 foreach($block['params']['validate'] as $rule) { 
  689.                     $rules[] = 'custom['.$rule.']'; 
  690.                 } 
  691.             } else if(strlen(trim($block['params']['validate'])) > 0) { 
  692.                 // handle single validation rule 
  693.                 $rules[] = 'custom['.$block['params']['validate'].']'; 
  694.             } 
  695.         } 
  696.  
  697.         // generate string if there is at least one rule to validate against 
  698.         if(empty($rules)) { 
  699.             return ''; 
  700.         } else { 
  701.             // make sure rules are not duplicated 
  702.             $rules = array_unique($rules); 
  703.             return 'validate['.join(', ', $rules).']'; 
  704.         } 
  705.     }

  707.     protected function render_web_body() { 
  708.         $helper_render_engine = WYSIJA::get('render_engine', 'helper'); 
  709.         $helper_render_engine->setTemplatePath(WYSIJA_EDITOR_TOOLS); 
  710.  
  711.         $blocks = $this->get_data('body'); 
  712.  
  713.         if(empty($blocks)) return ''; 
  714.  
  715.         $body = ''; 
  716.         foreach($blocks as $key => $block) { 
  717.             // get extra data depending on field type 
  718.             $block = $this->_get_extra_data($block); 
  719.  
  720.             // special case for email widget 
  721.             if($block['field'] === 'email') { 
  722.                 $user_email = WYSIJA::wp_get_userdata('user_email'); 
  723.  
  724.                 if($user_email && is_string($user_email) && is_user_logged_in() && !current_user_can('switch_themes') && !is_admin()) { 
  725.                     $block['value'] = $user_email; 
  726.                 } 
  727.             } 
  728.  
  729.             // set field name 'prefix' depending whether it's a custom field or not 
  730.             if($this->is_custom_field($block)) { 
  731.                 $field_prefix = 'wysija[field]'; 
  732.             } else { 
  733.                 $field_prefix = 'wysija[user]'; 
  734.             } 
  735.  
  736.             // generate block template 
  737.             $data = array_merge($block, array( 
  738.                 'field_prefix' => $field_prefix,  
  739.                 'preview' => ($this->get_mode() === 'preview'),  
  740.                 'i18n' => $this->get_translations(),  
  741.                 'validation' => $this->get_validation_class($block) 
  742.  )); 
  743.             $body .= $helper_render_engine->render($data, 'templates/form/web/widgets/template.html'); 
  744.         } 
  745.
