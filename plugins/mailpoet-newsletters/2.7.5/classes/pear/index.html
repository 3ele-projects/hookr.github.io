  603.     /** 
  604.     * Pop the last error handler used 
  605.     * 
  606.     * @return bool Always true 
  607.     * 
  608.     * @see PEAR::pushErrorHandling 
  609.     */ 
  610.     function popErrorHandling() 
  611.     { 
  612.         $stack = &$GLOBALS['_PEAR_error_handler_stack']; 
  613.         array_pop($stack); 
  614.         list($mode, $options) = $stack[sizeof($stack) - 1]; 
  615.         array_pop($stack); 
  616.         if (isset($this) && is_a($this, 'PEAR')) { 
  617.             $this->setErrorHandling($mode, $options); 
  618.         } else { 
  619.             PEAR::setErrorHandling($mode, $options); 
  620.         } 
  621.         return true; 
  622.     } 
  623.  
  624.     // }}} 
  625.     // {{{ loadExtension() 
  626.  
  627.     /** 
  628.     * OS independant PHP extension load. Remember to take care 
  629.     * on the correct extension name for case sensitive OSes. 
  630.     * 
  631.     * @param string $ext The extension name 
  632.     * @return bool Success or not on the dl() call 
  633.     */ 
  634.     function loadExtension($ext) 
  635.     { 
  636.         if (!extension_loaded($ext)) { 
  637.             // if either returns true dl() will produce a FATAL error, stop that 
  638.             if ((ini_get('enable_dl') != 1) || (ini_get('safe_mode') == 1)) { 
  639.                 return false; 
  640.             } 
  641.             if (OS_WINDOWS) { 
  642.                 $suffix = '.dll'; 
  643.             } elseif (PHP_OS == 'HP-UX') { 
  644.                 $suffix = '.sl'; 
  645.             } elseif (PHP_OS == 'AIX') { 
  646.                 $suffix = '.a'; 
  647.             } elseif (PHP_OS == 'OSX') { 
  648.                 $suffix = '.bundle'; 
  649.             } else { 
  650.                 $suffix = '.so'; 
  651.             } 
  652.             return @dl('php_'.$ext.$suffix) || @dl($ext.$suffix); 
  653.         } 
  654.         return true; 
  655.     }
