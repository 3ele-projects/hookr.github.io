  325.     private function range_finder($value, $ranges_increment) { 
  326.  
  327.         $limit_max = 0; 
  328.         foreach( $ranges_increment as $limit => $range_increment ) { 
  329.             $small_limit = $limit_max + $range_increment; 
  330.             $limit_max = $limit; 
  331.  
  332.             if( $value > $limit_max) { 
  333.                 continue; 
  334.             } 
  335.  
  336.             while( $value >= $small_limit && $small_limit <= $limit_max ) { 
  337.  
  338.                 if( $value > $small_limit ) { 
  339.                     $min_value = $small_limit - $range_increment + 1; 
  340.                 } 
  341.                 if( $value == $small_limit ) { 
  342.                     break; 
  343.                 } 
  344.                 $small_limit += $range_increment; 
  345.             } 
  346.  
  347.             if( $value <= $small_limit) { 
  348.                 break; 
  349.             } 
  350.         } 
  351.  
  352.         if( $value > $limit_max) { 
  353.             return array( 'lower' => $limit_max , 'upper' => 'above' ); 
  354.         }else{ 
  355.             if( $value < 1 ) { 
  356.                 return array( 'lower' => 0 , 'upper' => 'or undefined' ); 
  357.             }else{ 
  358.                 $min_value = $small_limit - $range_increment + 1; 
  359.                 return array( 'lower' => $min_value , 'upper' => $small_limit ); 
  360.             } 
  361.         } 
  362.  
  363.     }

  860.         $main_version = (int)$php_version_factors[0];
