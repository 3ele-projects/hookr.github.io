  165. function reposter_cron() 
  166. { 
  167.   global $fb_reposter_settings; 
  168.  
  169.   $fb_reposter_settings = updateTokens('', $fb_reposter_settings); 
  170.  
  171.   $reposter_posts = get_option('reposter_posts'); 
  172.  
  173.   if (is_array($reposter_posts) === true && count($reposter_posts) > 0) { 
  174.  
  175.       $access_token = $fb_reposter_settings['long_lived_user_access_token']; 
  176.  
  177.       foreach ($fb_reposter_settings['user_pages']['data'] as $fb_page) { 
  178.  
  179.           if ($fb_page['id'] == $fb_reposter_settings['userid']) { 
  180.  
  181.               $access_token = $fb_page['access_token']; 
  182.           } 
  183.       } 
  184.  
  185.       if (!empty($access_token)) { 
  186.  
  187.           foreach ($reposter_posts as $post_id => $num_posted) { 
  188.  
  189.               if ($num_posted < $fb_reposter_settings['xtimes']) { 
  190.  
  191.                   $post = generatePost($post_id); 
  192.                   postToWall($post['title'], $post['link'], $access_token, $fb_reposter_settings['userid']); 
  193.  
  194.                   if ($reposter_posts[$post_id] + 1 == $fb_reposter_settings['xtimes']) { 
  195.                       unset($reposter_posts[$post_id]); 
  196.                   } else { 
  197.                       $reposter_posts[$post_id]++; 
  198.                   } 
  199.  
  200.               } 
  201.           } 
  202.       } 
  203.       update_option('reposter_posts', $reposter_posts); 
  204.   }
