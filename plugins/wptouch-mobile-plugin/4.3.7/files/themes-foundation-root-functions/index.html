  80. function foundation_process_image_file( $file_name, $setting_name ) { 
  81.   if ( $setting_name->domain == FOUNDATION_SETTING_DOMAIN && $setting_name->name == 'logo_image' ) { 
  82.       // Need to make sure this isn't too big 
  83.       if ( function_exists( 'getimagesize' ) && function_exists( 'imagecreatefrompng' ) && function_exists( 'imagecopyresampled' ) && function_exists( 'imagepng' ) ) { 
  84.  
  85.           $size = getimagesize( $file_name ); 
  86.           if ( $size ) { 
  87.               $width = $size[0]; 
  88.               $height = $size[1]; 
  89.  
  90.               if ( $size['mime'] == 'image/png' ) { 
  91.                   if ( $width > FOUNDATION_MAX_LOGO_SIZE ) { 
  92.                       $new_width = FOUNDATION_MAX_LOGO_SIZE; 
  93.                       $new_height = $height*$new_width/$width; 
  94.  
  95.                       $src_image = imagecreatefrompng( $file_name ); 
  96.  
  97.                       $saved_image = imagecreatetruecolor( $new_width, $new_height ); 
  98.  
  99.                       // Preserve Transparency 
  100.                       imagecolortransparent( $saved_image, imagecolorallocate( $saved_image, 0, 0, 0 ) ); 
  101.  
  102.                       imagecopyresampled( $saved_image, $src_image, 0, 0, 0, 0, $new_width, $new_height, $width, $height); 
  103.  
  104.                       // Get rid of the old file 
  105.                       unlink( $file_name ); 
  106.  
  107.                       // New image, compression level 5 (make it a bit smaller) 
  108.                       imagepng( $saved_image, $file_name, 5 ); 
  109.                   } 
  110.               } 
  111.           } 
  112.       } 
  113.   } 
  114. }

  971. function wptouch_fdn_hierarchical_cat_list( $num, $include_count = true, $taxonomy = 'category', $opening_tag = '<ul>', $closing_tag = '</ul>' ) { 
  972.   $walker = new WPtouchProMainNavMenuWalker; 
  973.   $defaults = array( 
  974.       'number' => $num,  
  975.       'show_option_all' => false,  
  976.       'show_option_none' => false,  
  977.       'orderby' => 'name',  
  978.       'order' => 'ASC',  
  979.       'style' => 'list',  
  980.       'show_count' => $include_count,  
  981.       'hide_empty' => 1,  
  982.       'use_desc_for_title' => 1,  
  983.       'child_of' => 0,  
  984.       'feed' => '',  
  985.       'feed_type' => '',  
  986.       'feed_image' => '',  
  987.       'exclude' => '',  
  988.       'exclude_tree' => '',  
  989.       'current_category' => 0,  
  990.       'hierarchical' => true,  
  991.       'title_li' => false,  
  992.       'echo' => 1,  
  993.       'depth' => 0,  
  994.       'taxonomy' => $taxonomy,  
  995.       'walker' => new WPtouchProCategoryWalker 
  996.  ); 
  997.  
  998.   if ( isset( $args ) ) { 
  999.       $r = wp_parse_args( $args, $defaults ); 
  1000.   } else { 
  1001.       $r = $defaults; 
  1002.   } 
  1003.  
  1004.   if ( !isset( $r['pad_counts'] ) && $r['show_count'] && $r['hierarchical'] ) 
  1005.       $r['pad_counts'] = true; 
  1006.  
  1007.   if ( true == $r['hierarchical'] ) { 
  1008.       $r['exclude_tree'] = $r['exclude']; 
  1009.       $r['exclude'] = ''; 
  1010.   } 
  1011.  
  1012.   if ( ! isset( $r['class'] ) ) 
  1013.       $r['class'] = ( 'category' == $r['taxonomy'] ) ? 'categories' : $r['taxonomy']; 
  1014.  
  1015.   if ( ! taxonomy_exists( $r['taxonomy'] ) ) { 
  1016.       return false; 
  1017.   } 
  1018.  
  1019.   $show_option_all = $r['show_option_all']; 
  1020.   $show_option_none = $r['show_option_none']; 
  1021.  
  1022.   $categories = get_categories( $r ); 
  1023.  
  1024.   $output = ''; 
  1025.  
  1026.   if ( empty( $categories ) ) { 
  1027.       if ( ! empty( $show_option_none ) ) { 
  1028.           if ( 'list' == $r['style'] ) { 
  1029.               $output .= '<li class="cat-item-none">' . $show_option_none . '</li>'; 
  1030.           } else { 
  1031.               $output .= $show_option_none; 
  1032.           } 
  1033.       } 
  1034.   } else { 
  1035.       $output = $opening_tag; 
  1036.  
  1037.       if ( ! empty( $show_option_all ) ) { 
  1038.           $posts_page = ( 'page' == get_option( 'show_on_front' ) && get_option( 'page_for_posts' ) ) ? get_permalink( get_option( 'page_for_posts' ) ) : home_url( '/' ); 
  1039.           $posts_page = esc_url( $posts_page ); 
  1040.           if ( 'list' == $r['style'] ) { 
  1041.               $output .= "<li class='cat-item-all'><a href='$posts_page'>$show_option_all</a></li>"; 
  1042.           } else { 
  1043.               $output .= "<a href='$posts_page'>$show_option_all</a>"; 
  1044.           } 
  1045.       } 
  1046.  
  1047.       if ( empty( $r['current_category'] ) && ( is_category() || is_tax() || is_tag() ) ) { 
  1048.           $current_term_object = get_queried_object(); 
  1049.           if ( $current_term_object && $r['taxonomy'] === $current_term_object->taxonomy ) { 
  1050.               $r['current_category'] = get_queried_object_id(); 
  1051.           } 
  1052.       } 
  1053.  
  1054.       if ( $r['hierarchical'] ) { 
  1055.           $depth = $r['depth'];
