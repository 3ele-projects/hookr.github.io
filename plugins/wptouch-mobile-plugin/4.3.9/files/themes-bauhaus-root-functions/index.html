  220. function bauhaus_body_classes( $classes ) { 
  221.   $settings = bauhaus_get_settings(); 
  222.  
  223.   $classes[] = 'circles'; 
  224.  
  225.   if ( !$settings->bauhaus_show_comment_bubbles ) { 
  226.       $classes[] = 'no-com-bubbles'; 
  227.   } 
  228.  
  229.   if ( $settings->bauhaus_menu_style == 'drop-down' ) { 
  230.       $classes[] = 'drop-down'; 
  231.   } else { 
  232.       $classes[] = 'off-canvas'; 
  233.   } 
  234.  
  235.   if ( $settings->bauhaus_post_listing_view == 'default' ) { 
  236.       $classes[] = 'list-view'; 
  237.   } 
  238.  
  239.   if ( $settings->bauhaus_post_listing_view == 'carousel' && $settings->bauhaus_featured_carousel_enabled == false && $settings->bauhaus_popular_enabled == false ) { 
  240.       $classes[] = 'slider-latest-only'; 
  241.   } 
  242.  
  243.   return $classes; 
  244. }
