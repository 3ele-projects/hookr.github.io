  134. function foundation_featured_get_slides() { 
  135.   global $post; 
  136.  
  137.   $settings = foundation_get_settings(); 
  138.  
  139.   $foundation_featured_posts = array(); 
  140.   $foundation_featured_data = array(); 
  141.  
  142.   $args = foundation_featured_get_args(); 
  143.  
  144.   $new_posts = false; 
  145.   switch( $settings->featured_type ) { 
  146.       case 'tag': 
  147.           $new_posts = new WP_Query( array( 'ignore_sticky_posts' => 1, 'posts_per_page' => $args[ 'max_search' ], 'tag' => $settings->featured_tag, 'meta_query' => array( array('key' => '_thumbnail_id') ) ) ); 
  148.           break; 
  149.       case 'category': 
  150.           $new_posts = new WP_Query( array( 'ignore_sticky_posts' => 1, 'posts_per_page' => $args[ 'max_search' ], 'category_name' => $settings->featured_category, 'meta_query' => array( array('key' => '_thumbnail_id') ) ) ); 
  151.           break; 
  152.       case 'posts': 
  153.           if ( function_exists( 'wptouch_custom_posts_add_to_search' ) ) { 
  154.               $post_types = wptouch_custom_posts_add_to_search( array( 'post', 'page' ) ); 
  155.           } else { 
  156.               $post_types = array( 'post', 'page' ); 
  157.           } 
  158.           $post_ids = explode( ', ', str_replace( ' ', '', $settings->featured_post_ids ) ); 
  159.           if ( is_array( $post_ids ) && count( $post_ids ) ) { 
  160.               $new_posts = new WP_Query( array( 'post__in' => $post_ids, 'posts_per_page' => $args[ 'max_search' ], 'post_type' => $post_types, 'orderby' => 'post__in' , 'meta_query' => array( array('key' => '_thumbnail_id') ) ) ); 
  161.           } 
  162.           break; 
  163.       case 'post_type': 
  164.           $new_posts = new WP_Query( array( 'post_type' => $settings->featured_post_type, 'posts_per_page' => $args[ 'max_search' ], 'meta_query' => array( array('key' => '_thumbnail_id') ) ) ); 
  165.           break; 
  166.       case 'latest': 
  167.       default: 
  168.           break; 
  169.   } 
  170.  
  171.   if ( !$new_posts ) { 
  172.       $new_posts = new WP_Query( array( 'posts_per_page' => $args[ 'max_search' ], 'meta_query' => array( array( 'key' => '_thumbnail_id' ) ) ) ); 
  173.   } 
  174.  
  175.   return $new_posts; 
  176. }
