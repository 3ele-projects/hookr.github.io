  160. function bauhaus_featured_modify_query( $query ) {

  175.           $query 
  176.  );

  179.           return;

  224. function bauhaus_featured_get_posts() { 
  225.   global $post; 
  226.  
  227.   $settings = bauhaus_get_settings(); 
  228.  
  229.   $bauhaus_featured_posts = array(); 
  230.  
  231.   $args = bauhaus_featured_get_args(); 
  232.  
  233.   $new_posts = false; 
  234.   switch( $settings->bauhaus_featured_type ) { 
  235.       case 'tag': 
  236.           $new_posts = new WP_Query( array( 'ignore_sticky_posts' => 1, 'posts_per_page' => $args[ 'max_search' ], 'tag' => $settings->bauhaus_featured_tag, 'meta_query' => array( array('key' => '_thumbnail_id') ) ) ); 
  237.           break; 
  238.       case 'category': 
  239.           $new_posts = new WP_Query( array( 'ignore_sticky_posts' => 1, 'posts_per_page' => $args[ 'max_search' ], 'category_name' => $settings->bauhaus_featured_category, 'meta_query' => array( array('key' => '_thumbnail_id') ) ) ); 
  240.           break; 
  241.       case 'posts': 
  242.           if ( function_exists( 'wptouch_custom_posts_add_to_search' ) ) { 
  243.               $post_types = wptouch_custom_posts_add_to_search( array( 'post', 'page' ) ); 
  244.           } else { 
  245.               $post_types = array( 'post', 'page' ); 
  246.           } 
  247.           $post_ids = explode( ', ', str_replace( ' ', '', $settings->bauhaus_featured_post_ids ) ); 
  248.           if ( is_array( $post_ids ) && count( $post_ids ) ) { 
  249.               $new_posts = new WP_Query( array( 'ignore_sticky_posts' => 1, 'post__in' => $post_ids, 'posts_per_page' => $args[ 'max_search' ], 'post_type' => $post_types, 'orderby' => 'post__in' , 'meta_query' => array( array('key' => '_thumbnail_id') ) ) ); 
  250.           } 
  251.           break; 
  252.       case 'post_type': 
  253.           $new_posts = new WP_Query( array( 'ignore_sticky_posts' => 1, 'post_type' => $settings->bauhaus_featured_post_type, 'posts_per_page' => $args[ 'max_search' ], 'meta_query' => array( array('key' => '_thumbnail_id') ) ) ); 
  254.           break; 
  255.       case 'latest': 
  256.       default: 
  257.           $new_posts = new WP_Query( array( 'ignore_sticky_posts' => 1, 'posts_per_page' => $args[ 'max_search' ], 'meta_query' => array( array( 'key' => '_thumbnail_id' ) ) ) ); 
  258.           break; 
  259.   } 
  260.  
  261.   return $new_posts; 
  262. }

  294.

  306.  
  307.   if ( $settings->bauhaus_post_listing_dots ) { 
  308.       $classes[] = 'dots';

  310.  
  311.   if ( function_exists( 'wptouch_custom_posts_add_to_search' ) ) { 
  312.       $post_types = wptouch_custom_posts_add_to_search( array( 'post', 'page' ) );

  314.       $post_types = array( 'post', 'page' ); 
  315.   } 
  316.  
  317.   $slides = new WP_Query( array( 'ignore_sticky_posts' => 1, 'post__in' => $bauhaus_featured_posts, 'post_type' => $post_types, 'posts_per_page' => $settings->bauhaus_featured_max_number_of_posts ) ); 
  318.  
  319.   if ( $slides->post_count > 0 ) { 
  320.  
  321.       echo '<div class="carousel list-carousel ' . implode( ' ', $classes ) .'">';

  332.   }
