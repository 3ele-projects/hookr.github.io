", Version: $jfb_version, WP: " . $GLOBALS['wp_version'] . ", Browser: " . $browser['shortname'] . " " . $browser['version'] . " for " . $browser['platform'] . ")\n"; 
  18.  
  19.   //Run one hook before ANYTHING happens. 
  20.   $jfb_log .= "WP: Running action wpfb_prelogin\n"; 
  21.   do_action('wpfb_prelogin'); 
  22.  
  23.   //Check the nonce to make sure this was a valid login attempt (unless the user has disabled nonce checking) 
  24.   global $opt_jfb_disablenonce, $jfb_nonce_name; 
  25.   if( !get_option($opt_jfb_disablenonce) ) 
  26.   { 
  27.       if( wp_verify_nonce ($_REQUEST[$jfb_nonce_name], $jfb_nonce_name) != 1 ) 
  28.       { 
  29.           //If there's already a user logged in, tell the user and give them a link back to where they were. 
  30.           $currUser = wp_get_current_user();  
  31.           if( $currUser->ID )

  37.           } 
  38.              
  39.           j_die("Nonce check failed, login aborted.\n<br/><br/>\n" . __("This error is usually due to your browser's privacy settings or a server-side caching plugin.  If you get this error on multiple browsers, please contact the site administrator.", 'wp-fb-autoconnect')); 
  40.       } 
  41.       $jfb_log .= "WP: nonce check passed\n"; 
  42.   } 
  43.   else 
  44.       $jfb_log .= "WP: nonce check DISABLED\n"; 
  45.

  51.   $jfb_log .= "WP: Found redirect URL ($redirectTo)\n"; 
  52.

  57.       j_die("Error: Missing Facebook access token.\n<br/><br/>\n" . sprintf(__("If you're receiving this notice via e-mail as a site administrator, it's nearly always safe to ignore (these errors are due to spambots automatically hitting your site).  If you're seeing this as a real person attempting to login, please report it to the plugin author at %s.", 'wp-fb-autoconnect'), $jfb_homepage), !get_option($opt_jfb_email_logs_missingpost));

  63.   $jfb_log .= "FB: Initiating Facebook connection...\n";

  72.   //NOTE: Up to 08/11/2014, I was fetching both avatars in one fql call: 
  73.   //jfb_api_get("https://graph.facebook.com/fql?q=".urlencode("SELECT pic_square, pic_big FROM user WHERE uid=$fb_uid")."&access_token=$access_token");

  76.   $pic = jfb_api_get("https://graph.facebook.com/me/picture?".apply_filters("wpfb_avatar_size", "type=square", "thumb")."&redirect=false&access_token=$access_token"); 
  77.   $fbuser['pic_square'] = $pic['data']['url'];  
  78.   $pic = jfb_api_get("https://graph.facebook.com/me/picture?".apply_filters("wpfb_avatar_size", "type=large", "full")."&redirect=false&access_token=$access_token"); 
  79.   $fbuser['pic_big'] = $pic['data']['url']; 
  80.   $jfb_log .= "FB: Got user info (".$fbuser['name'].")\n"; 
  81.    
  82.   //See if we were given permission to access the user's email 
  83.   //This isn't required, and will only matter if it's a new user without an existing WP account 
  84.   //(since we'll auto-register an account for them, using the contact_email we get from Facebook - if we can...)

  87.   { 
  88.       $jfb_log .= "FB: Email privilege granted (" .$fbuser['email'] . ")\n"; 
  89.       $userRevealedEmail = true; 
  90.   } 
  91.   else if( isset($fbuser['email']) && strlen($fbuser['email']) != 0 ) 
  92.   { 
  93.       $jfb_log .= "FB: Email privilege granted, but only for an anonymous proxy address (" . $fbuser['email'] . ")\n"; 
  94.   } 
  95.   else 
  96.   { 
  97.       global $jfb_default_email; 
  98.       $jfb_log .= "FB: Email priviledge denied.\n"; 
  99.       $fbuser['email'] = "FB_" . $fb_uid . $jfb_default_email; 
  100.   }  
  101.  
  102.  
  103.   //Run a hook so users can`examine this Facebook user *before* letting them login.  You might use this 
  104.   //to limit logins based on friendship status - if someone isn't your friend, you could redirect them 
  105.   //to an error page (and terminate this script). 
  106.   $jfb_log .= "WP: Running action wpfb_connect\n";

  108.  
  109.  
  110.   //Examine all existing WP users to see if any of them match this Facebook user.  
  111.   //The base query for getting the users comes from get_users_from_blog(), to which I add a subquery

  125.   } 
  126.  
  127.   //Although $wp_users should only contain the one matching user (or be empty), this "loop" method of searching

  139.           $user_login_name = $user_data->user_login; 
  140.           $jfb_log .= "WP: Found existing user by meta (" . $user_login_name . ")\n"; 
  141.           break; 
  142.       } 
  143.   } 
  144.  
  145.  
  146.   //Next, try to lookup their email directly (via WordPress).  Obviously this will only work if they've revealed 
  147.   //their "real" address (vs denying access, or changing it to a "proxy" in the popup) 
  148.   if ( !$user_login_id && $userRevealedEmail ) 
  149.   { 
  150.       $jfb_log .= "WP: Searching for user by email address...\n"; 
  151.       if ( $wp_user = get_user_by('email', $fbuser['email']) ) 
  152.       { 
  153.           $user_login_id = $wp_user->ID; 
  154.           $user_data = get_userdata($wp_user->ID); 
  155.           $user_login_name = $user_data->user_login; 
  156.           $jfb_log .= "WP: Found existing user (" . $user_login_name . ") by email (" . $fbuser['email'] . ")\n"; 
  157.       } 
  158.   } 
  159.  
  160.  
  161.   //If we found an existing user, check if they'd previously denied access to their email but have now allowed it. 
  162.   //If so, we'll want to update their WP account with their *real* email. 
  163.   global $jfb_default_email; 
  164.   if( $user_login_id ) 
  165.   { 
  166.       //Check 1: It was previously denied, but is now allowed 
  167.       $updateEmail = false; 
  168.       if( strpos($user_data->user_email, $jfb_default_email) !== FALSE && strpos($fbuser['email'], $jfb_default_email) === FALSE ) 
  169.       { 
  170.           $jfb_log .= "WP: Previously DENIED email has now been allowed; updating to (".$fbuser['email'].")\n"; 
  171.           $updateEmail = true; 
  172.       } 
  173.       //Check 2: It was previously allowed, but only as an anonymous proxy.  They've now revealed their "true" email. 
  174.       if( strpos($user_data->user_email, "@proxymail.facebook.com") !== FALSE && strpos($fbuser['email'], "@proxymail.facebook.com") === FALSE ) 
  175.       { 
  176.           $jfb_log .= "WP: Previously PROXIED email has now been allowed; updating to (".$fbuser['email'].")\n"; 
  177.           $updateEmail = true; 
  178.       } 
  179.       if( $updateEmail ) 
  180.       { 
  181.           $user_upd = array(); 
  182.           $user_upd['ID'] = $user_login_id; 
  183.           $user_upd['user_email'] = $fbuser['email']; 
  184.           wp_update_user($user_upd); 
  185.       } 
  186.        
  187.       //Run a hook when an existing user logs in 
  188.       $jfb_log .= "WP: Running action wpfb_existing_user\n"; 
  189.       do_action('wpfb_existing_user', array('WP_ID' => $user_login_id, 'FB_ID' => $fb_uid, 'WP_UserData' => $user_data, 'access_token'=>$access_token) );

  191.  
  192.  
  193.   //If we still don't have a user_login_id, the FB user who's logging in has never been to this blog. 
  194.   //We'll auto-register them a new account.  Note that if they haven't allowed email permissions, the 
  195.   //account we register will have a bogus email address (but that's OK, since we still know their Facebook ID) 
  196.   if( !$user_login_id ) 
  197.   { 
  198.       $jfb_log .= "WP: No user found. Automatically registering (FB_". $fb_uid . ")\n";

  200.       $user_data['user_login'] = "FB_" . $fb_uid; 
  201.       $user_data['user_pass'] = wp_generate_password(); 
  202.       $user_data['first_name'] = $fbuser['first_name']; 
  203.       $user_data['last_name'] = $fbuser['last_name']; 
  204.       $user_data['user_url'] = $fbuser["profile_url"]; 
  205.       $user_data['user_email'] = $fbuser["email"]; 
  206.        
  207.       //Nicename is a sanitized copy of the username; it's not visible/editable from the admin panel.  It's what's used for author links. 
  208.       $user_data['user_nicename'] = sanitize_title($user_data['user_login']); 
  209.  
  210.       //Display_name is selectable as the "Display name publicly as" dropdown on the profile page. 
  211.       //BP also duplicates this to the primary XProfile "name" field when a new user is registered. 
  212.       $user_data['display_name'] = $fbuser['first_name'] . " " . $fbuser['last_name']; 
  213.        
  214.       //Run a filter so the user can be modified to something different before registration 
  215.       //NOTE: If the user has selected "pretty names", this'll change FB_xxx to i.e. "John.Smith" 
  216.       $jfb_log .= "WP: Applying filters wpfb_insert_user/wpfb_inserting_user\n"; 
  217.       $user_data = apply_filters('wpfb_insert_user', $user_data, $fbuser ); 
  218.       $user_data = apply_filters('wpfb_inserting_user', $user_data, array('WP_ID' => $user_login_id, 'FB_ID' => $fb_uid, 'FB_UserData' => $fbuser, 'access_token'=>$access_token) ); 
  219.        
  220.       //Insert a new user to our database and make sure it worked 
  221.       $user_login_id = wp_insert_user($user_data); 
  222.       if( is_wp_error($user_login_id) ) 
  223.       { 
  224.           $errMsg = "Error: wp_insert_user failed!<br /><br />"; 
  225.           $errMsg .= __("If you get this error while running WordPress MultiSite, it means you'll need to purchase the premium addon to enable full MultiSite support.  Please see here for more information:", 'wp-fb-autoconnect'); 
  226.           $errMsg .= " <a href=\"$jfb_homepage#premium\"><b>Premium Addon</b></a><br/>"; 
  227.           $errMsg .= __("If you're NOT using MultiSite, please report this bug to the plugin author on the support page:", 'wp-fb-autoconnect'); 
  228.           $errMsg .= " <a href=\"$jfb_homepage#feedback\">$jfb_homepage</a>.<br /><br />"; 
  229.           $errMsg .= "Message: " . (method_exists($user_login_id, 'get_error_message')?$user_login_id->get_error_message():"Undefined") . "<br />"; 
  230.           $errMsg .= "Allow Multisite: " . (defined('WP_ALLOW_MULTISITE')?constant('WP_ALLOW_MULTISITE'):"Undefined") . "<br />"; 
  231.           $errMsg .= "Is Multisite: " . (function_exists('is_multisite')?is_multisite():"Undefined") . "<br />"; 
  232.           $user_data['user_pass'] = '(Removed)'; 
  233.           $errMsg .= "Userdata: " . print_r($user_data, true); 
  234.           j_die($errMsg); 
  235.       } 
  236.        
  237.       //Success! Notify the site admin. 
  238.       //(Implementation comes from pre-4.3's version of wp_new_user_notification) 
  239.       $user_login_name = $user_data['user_login']; 
  240.       $user = get_userdata( $user_login_id ); 
  241.       $blogname = wp_specialchars_decode(get_option('blogname'), ENT_QUOTES); 
  242.       $message = sprintf(__('New user registration on your site %s:'), $blogname) . "\r\n\r\n"; 
  243.       $message .= sprintf(__('Username: %s'), $user->user_login) . "\r\n\r\n"; 
  244.       $message .= sprintf(__('E-mail: %s'), $user->user_email) . "\r\n"; 
  245.       @wp_mail(get_option('admin_email'), sprintf(__('[%s] New User Registration'), $blogname), $message); 
  246.        
  247.       //Run an action so i.e. usermeta can be added to a user after registration 
  248.       $jfb_log .= "WP: Running action wpfb_inserted_user\n"; 
  249.       do_action('wpfb_inserted_user', array('WP_ID' => $user_login_id, 'FB_ID' => $fb_uid, 'WP_UserData' => $user_data, 'access_token'=>$access_token) ); 
  250.   } 
  251.    
  252.   //Tag the user with our meta so we can recognize them next time, without resorting to email hashes 
  253.   global $jfb_uid_meta_name; 
  254.   update_user_meta($user_login_id, $jfb_uid_meta_name, $fb_uid); 
  255.   $jfb_log .= "WP: Updated usermeta ($jfb_uid_meta_name -> $fb_uid)\n"; 
  256.
