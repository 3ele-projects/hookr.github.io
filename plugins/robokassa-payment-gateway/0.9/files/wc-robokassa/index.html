  295.           else{ 
  296.               wp_die('IPN Request Failure'); 
  297.           }

  299.       else if (isset($_GET['robokassa']) AND $_GET['robokassa'] == 'success') { 
  300.           $inv_id = $_POST['InvId']; 
  301.           $order = new WC_Order($inv_id); 
  302.           $order->update_status('processing', __('****** ******* *******', 'woocommerce')); 
  303.           WC()->cart->empty_cart(); 
  304.

  306.       }

  308.           $inv_id = $_POST['InvId'];
