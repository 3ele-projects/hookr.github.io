  182. function bbp_pre_get_option( $value = '' ) { 
  183.  
  184.   // Remove the filter prefix 
  185.   $option = str_replace( 'pre_option_', '', current_filter() ); 
  186.  
  187.   // Check the options global for preset value 
  188.   if ( isset( bbpress()->options[$option] ) ) 
  189.       $value = bbpress()->options[$option]; 
  190.  
  191.   // Always return a value, even if false 
  192.   return $value;
