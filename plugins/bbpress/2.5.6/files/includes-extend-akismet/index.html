  234.           // Mysterious, and straight from the can

  239.  
  240.           // Honey-glazed, a straight off the bone 
  241.           case 'bbp_unspammed_topic' : 
  242.           case 'bbp_unspammed_reply' : 
  243.               $request_type = 'ham'; 
  244.               break; 
  245.  
  246.           // Possibly poison... 
  247.           default : 
  248.               return;

  253.

  256.

  258.       if ( empty( $_post ) ) 
  259.           return; 
  260.  
  261.       // Bail if we're spamming, but the post_status isn't spam 
  262.       if ( ( 'spam' === $request_type ) && ( bbp_get_spam_status_id() !== $_post->post_status ) ) 
  263.           return; 
  264.  
  265.       // Set some default post_data

  267.           'comment_approved' => $_post->post_status,

  275.           'comment_type' => $_post->post_type,  
  276.           'permalink' => get_permalink( $post_id ),  
  277.           'user_ID' => $_post->post_author,  
  278.           'user_ip' => get_post_meta( $post_id, '_bbp_author_ip', true ),

  332.   /** 
  333.    * Ping Akismet service and check for spam/ham response 
  334.    * 
  335.    * @since bbPress (r3277) 
  336.    * 
  337.    * @param array $post_data 
  338.    * @param string $check Accepts check|submit 
  339.    * @param string $spam Accepts spam|ham 
  340.    * 
  341.    * @global string $akismet_api_host 
  342.    * @global string $akismet_api_port 
  343.    * 
  344.    * @uses akismet_test_mode() To determine if Akismet is in test mode 
  345.    * @uses akismet_http_post() To send data to the mothership for processing 
  346.    * 
  347.    * @return array Array of post data 
  348.    */ 
  349.   private function maybe_spam( $post_data, $check = 'check', $spam = 'spam' ) { 
  350.       global $akismet_api_host, $akismet_api_port; 
  351.  
  352.       // Define variables 
  353.       $query_string = $path = $response = ''; 
  354.  
  355.       // Populate post data 
  356.       $post_data['blog'] = get_option( 'home' ); 
  357.       $post_data['blog_charset'] = get_option( 'blog_charset' ); 
  358.       $post_data['blog_lang'] = get_locale(); 
  359.       $post_data['referrer'] = $_SERVER['HTTP_REFERER']; 
  360.       $post_data['user_agent'] = $_SERVER['HTTP_USER_AGENT']; 
  361.  
  362.       // Akismet Test Mode 
  363.       if ( akismet_test_mode() ) 
  364.           $post_data['is_test'] = 'true'; 
  365.  
  366.       // Loop through _POST args and rekey strings 
  367.       foreach ( $_POST as $key => $value ) 
  368.           if ( is_string( $value ) ) 
  369.               $post_data['POST_' . $key] = $value; 
  370.  
  371.       // Keys to ignore 
  372.       $ignore = array( 'HTTP_COOKIE', 'HTTP_COOKIE2', 'PHP_AUTH_PW' ); 
  373.  
  374.       // Loop through _SERVER args and remove whitelisted keys 
  375.       foreach ( $_SERVER as $key => $value ) { 
  376.  
  377.           // Key should not be ignored 
  378.           if ( !in_array( $key, $ignore ) && is_string( $value ) ) { 
  379.               $post_data[$key] = $value; 
  380.  
  381.           // Key should be ignored 
  382.           } else { 
  383.               $post_data[$key] = ''; 
  384.           } 
  385.       } 
  386.  
  387.       // Ready... 
  388.       foreach ( $post_data as $key => $data ) 
  389.           $query_string .= $key . '=' . urlencode( stripslashes( $data ) ) . '&'; 
  390.  
  391.       // Aim... 
  392.       if ( 'check' === $check ) { 
  393.           $path = '/1.1/comment-check'; 
  394.       } elseif ( 'submit' === $check ) { 
  395.           $path = '/1.1/submit-' . $spam; 
  396.       } 
  397.  
  398.       // Fire! 
  399.       $response = $this->http_post( $query_string, $akismet_api_host, $path, $akismet_api_port ); 
  400.  
  401.       // Check the high-speed cam 
  402.       if ( !empty( $response[1] ) ) { 
  403.           $post_data['bbp_akismet_result'] = $response[1]; 
  404.       } else { 
  405.           $post_data['bbp_akismet_result'] = esc_html__( 'No response', 'bbpress' ); 
  406.       } 
  407.  
  408.       // This is ham 
  409.       return $post_data; 
  410.   }

  428.   public function update_post_meta( $post_id = 0, $_post = false ) { 
  429.  
  430.       // Define local variable(s) 
  431.       $as_submitted = false; 
  432.  
  433.       // Setup some variables 
  434.       $post_id = (int) $post_id; 
  435.  
  436.       // Ensure we have a post object 
  437.       if ( empty( $_post ) ) 
  438.           $_post = get_post( $post_id ); 
  439.  
  440.       // Set up Akismet last post data 
  441.       if ( !empty( $this->last_post ) ) 
  442.           $as_submitted = $this->last_post['bbp_post_as_submitted']; 
  443.  
  444.       // wp_insert_post() might be called in other contexts. Ensure this is 
  445.       // the same topic/reply as was checked by BBP_Akismet::check_post() 
  446.       if ( is_object( $_post ) && !empty( $this->last_post ) && is_array( $as_submitted ) ) { 
  447.  
  448.           // Get user data 
  449.           $userdata = get_userdata( $_post->post_author ); 
  450.           $anonymous_data = bbp_filter_anonymous_post_data(); 
  451.  
  452.           // More checks 
  453.           if (    intval( $as_submitted['comment_post_ID'] )    === intval( $_post->post_parent ) 
  454.                   &&      $as_submitted['comment_author'] === ( $anonymous_data ? $anonymous_data['bbp_anonymous_name']  : $userdata->display_name ) 
  455.                   &&      $as_submitted['comment_author_email'] === ( $anonymous_data ? $anonymous_data['bbp_anonymous_email'] : $userdata->user_email ) 
  456.  ) { 
  457.  
  458.               // Normal result: true 
  459.               if ( $this->last_post['bbp_akismet_result'] === 'true' ) { 
  460.  
  461.                   // Leave a trail so other's know what we did 
  462.                   update_post_meta( $post_id, '_bbp_akismet_result', 'true' ); 
  463.                   $this->update_post_history( $post_id, esc_html__( 'Akismet caught this post as spam', 'bbpress' ), 'check-spam' ); 
  464.  
  465.                   // If post_status isn't the spam status, as expected, leave a note 
  466.                   if ( bbp_get_spam_status_id() !== $_post->post_status ) { 
  467.                       $this->update_post_history( $post_id, sprintf( esc_html__( 'Post status was changed to %s', 'bbpress' ), $_post->post_status ), 'status-changed-' . $_post->post_status ); 
  468.                   } 
  469.  
  470.               // Normal result: false 
  471.               } elseif ( $this->last_post['bbp_akismet_result'] === 'false' ) { 
  472.  
  473.                   // Leave a trail so other's know what we did 
  474.                   update_post_meta( $post_id, '_bbp_akismet_result', 'false' ); 
  475.                   $this->update_post_history( $post_id, esc_html__( 'Akismet cleared this post as not spam', 'bbpress' ), 'check-ham' ); 
  476.  
  477.                   // If post_status is the spam status, which isn't expected, leave a note 
  478.                   if ( bbp_get_spam_status_id() === $_post->post_status ) { 
  479.  
  480.                       // @todo Use wp_blacklist_check() 
  481.  
  482.                       $this->update_post_history( $post_id, sprintf( esc_html__( 'Post status was changed to %s', 'bbpress' ), $_post->post_status ), 'status-changed-' . $_post->post_status ); 
  483.                   } 
  484.  
  485.               // Abnormal result: error 
  486.               } else { 
  487.                   // Leave a trail so other's know what we did 
  488.                   update_post_meta( $post_id, '_bbp_akismet_error', time() ); 
  489.                   $this->update_post_history( $post_id, sprintf( esc_html__( 'Akismet was unable to check this post (response: %s), will automatically retry again later.', 'bbpress' ), $this->last_post['bbp_akismet_result'] ), 'check-error' ); 
  490.               } 
  491.  
  492.               // Record the complete original data as submitted for checking 
  493.               if ( isset( $this->last_post['bbp_post_as_submitted'] ) ) { 
  494.                   update_post_meta( $post_id, '_bbp_akismet_as_submitted', $this->last_post['bbp_post_as_submitted'] ); 
  495.               }

  592.   /** 
  593.    * Submit data to Akismet service with unique bbPress User Agent 
  594.    * 
  595.    * This code is directly taken from the akismet_http_post() function and 
  596.    * documented to bbPress 2.0 standard. 
  597.    * 
  598.    * @since bbPress (r3466) 
  599.    * 
  600.    * @param string $request The request we are sending 
  601.    * @param string $host The host to send our request to 
  602.    * @param string $path The path from the host 
  603.    * @param string $port The port to use 
  604.    * @param string $ip Optional Override $host with an IP address 
  605.    * @uses bbp_get_version() To get the current bbPress version 
  606.    * @return mixed WP_Error on error, array on success, empty on failure 
  607.    */ 
  608.   private function http_post( $request, $host, $path, $port = 80, $ip = '' ) { 
  609.  
  610.       // Preload required variables 
  611.       $bbp_version = bbp_get_version(); 
  612.       $content_length = strlen( $request ); 
  613.       $http_host = $host; 
  614.       $blog_charset = get_option( 'blog_charset' ); 
  615.       $response = ''; 
  616.       $errno = null; 
  617.       $errstr = null; 
  618.  
  619.       // Untque User Agent 
  620.       $akismet_ua = "bbPress/{$bbp_version} | "; 
  621.       $akismet_ua    .= 'Akismet/' . constant( 'AKISMET_VERSION' ); 
  622.  
  623.       // Use specific IP (if provided) 
  624.       if ( !empty( $ip ) && long2ip( ip2long( $ip ) ) ) 
  625.           $http_host = $ip; 
  626.  
  627.       // WP HTTP class is available 
  628.       if ( function_exists( 'wp_remote_post' ) ) { 
  629.  
  630.           // Setup the arguments 
  631.           $http_args = array( 
  632.               'body' => $request,  
  633.               'headers' => array( 
  634.                   'Content-Type' => 'application/x-www-form-urlencoded; charset=' . $blog_charset,  
  635.                   'Host' => $host,  
  636.                   'User-Agent' => $akismet_ua 
  637.  ),  
  638.               'httpversion' => '1.0',  
  639.               'timeout' => 15 
  640.  ); 
  641.  
  642.           // Where we are sending our request 
  643.           $akismet_url = 'http://' . $http_host . $path; 
  644.  
  645.           // Send the request 
  646.           $response = wp_remote_post( $akismet_url, $http_args ); 
  647.  
  648.           // Bail if the response is an error 
  649.           if ( is_wp_error( $response ) ) 
  650.               return ''; 
  651.  
  652.           // No errors so return response 
  653.           return array( $response['headers'], $response['body'] ); 
  654.  
  655.       // WP HTTP class is not available (Why not?) 
  656.       } else { 
  657.  
  658.           // Header info to use with our socket 
  659.           $http_request = "POST {$path} HTTP/1.0\r\n"; 
  660.           $http_request .= "Host: {$host}\r\n"; 
  661.           $http_request .= "Content-Type: application/x-www-form-urlencoded; charset={$blog_charset}\r\n"; 
  662.           $http_request .= "Content-Length: {$content_length}\r\n"; 
  663.           $http_request .= "User-Agent: {$akismet_ua}\r\n"; 
  664.           $http_request .= "\r\n"; 
  665.           $http_request .= $request; 
  666.  
  667.           // Open a socket connection 
  668.           if ( false !== ( $fs = @fsockopen( $http_host, $port, $errno, $errstr, 10 ) ) ) { 
  669.  
  670.               // Write our request to the pointer 
  671.               fwrite( $fs, $http_request ); 
  672.  
  673.               // Loop through pointer and compile a response 
  674.               while ( !feof( $fs ) ) { 
  675.                   // One TCP-IP packet at a time 
  676.                   $response .= fgets( $fs, 1160 ); 
  677.               } 
  678.  
  679.               // Close our socket 
  680.               fclose( $fs ); 
  681.  
  682.               // Explode the response into usable data 
  683.               $response = explode( "\r\n\r\n", $response, 2 ); 
  684.           } 
  685.  
  686.           // Return the response ('' if error/empty) 
  687.           return $response; 
  688.       } 
  689.   }
