  316.    * Save the Group Forum data on edit 
  317.    * 
  318.    * @since bbPress (r3465) 
  319.    * @param int $group_id (to handle Group Admin UI hook bp_group_admin_edit_after ) 
  320.    * @uses bbp_new_forum_handler() To check for forum creation 
  321.    * @uses bbp_edit_forum_handler() To check for forum edit 
  322.    */ 
  323.   public function edit_screen_save( $group_id = 0 ) { 
  324.  
  325.       // Bail if not a POST action 
  326.       if ( ! bbp_is_post_request() ) 
  327.           return; 
  328.  
  329.       // Admin Nonce check 
  330.       if ( is_admin() ) { 
  331.           check_admin_referer( 'groups_edit_save_' . $this->slug, 'forum_group_admin_ui' ); 
  332.  
  333.       // Theme-side Nonce check 
  334.       } elseif ( ! bbp_verify_nonce_request( 'groups_edit_save_' . $this->slug ) ) { 
  335.           bbp_add_error( 'bbp_edit_group_forum_screen_save', __( '<strong>ERROR</strong>: Are you sure you wanted to do that?', 'bbpress' ) ); 
  336.           return; 
  337.        } 
  338.  
  339.       $edit_forum = !empty( $_POST['bbp-edit-group-forum'] ) ? true : false; 
  340.       $forum_id = 0; 
  341.       $group_id = !empty( $group_id ) ? $group_id : bp_get_current_group_id(); 
  342.  
  343.       // Keymasters have the ability to reconfigure forums 
  344.       if ( bbp_is_user_keymaster() ) { 
  345.           $forum_ids = ! empty( $_POST['bbp_group_forum_id'] ) ? (array) (int) $_POST['bbp_group_forum_id'] : array(); 
  346.  
  347.       // Use the existing forum IDs 
  348.       } else { 
  349.           $forum_ids = array_values( bbp_get_group_forum_ids( $group_id ) ); 
  350.       } 
  351.  
  352.       // Normalize group forum relationships now 
  353.       if ( !empty( $forum_ids ) ) { 
  354.  
  355.           // Loop through forums, and make sure they exist 
  356.           foreach ( $forum_ids as $forum_id ) { 
  357.  
  358.               // Look for forum 
  359.               $forum = bbp_get_forum( $forum_id ); 
  360.  
  361.               // No forum exists, so break the relationship 
  362.               if ( empty( $forum ) ) { 
  363.                   $this->remove_forum( array( 'forum_id' => $forum_id ) ); 
  364.                   unset( $forum_ids[$forum_id] ); 
  365.               } 
  366.           } 
  367.  
  368.           // No support for multiple forums yet 
  369.           $forum_id = (int) ( is_array( $forum_ids ) ? $forum_ids[0] : $forum_ids ); 
  370.       } 
  371.  
  372.       // Update the group ID and forum ID relationships 
  373.       bbp_update_group_forum_ids( $group_id, (array) $forum_ids ); 
  374.       bbp_update_forum_group_ids( $forum_id, (array) $group_id ); 
  375.  
  376.       // Update the group forum setting 
  377.       $group = $this->toggle_group_forum( $group_id, $edit_forum ); 
  378.  
  379.       // Create a new forum 
  380.       if ( empty( $forum_id ) && ( true === $edit_forum ) ) { 
  381.  
  382.           // Set the default forum status 
  383.           switch ( $group->status ) { 
  384.               case 'hidden'  : 
  385.                   $status = bbp_get_hidden_status_id(); 
  386.                   break; 
  387.               case 'private' : 
  388.                   $status = bbp_get_private_status_id(); 
  389.                   break; 
  390.               case 'public'  : 
  391.               default        : 
  392.                   $status = bbp_get_public_status_id(); 
  393.                   break; 
  394.           } 
  395.  
  396.           // Create the initial forum 
  397.           $forum_id = bbp_insert_forum( array( 
  398.               'post_parent' => bbp_get_group_forums_root_id(),  
  399.               'post_title' => $group->name,  
  400.               'post_content' => $group->description,  
  401.               'post_status' => $status 
  402.  ) ); 
  403.  
  404.           // Setup forum args with forum ID 
  405.           $new_forum_args = array( 'forum_id' => $forum_id ); 
  406.  
  407.           // If in admin, also include the group ID 
  408.           if ( is_admin() && !empty( $group_id ) ) { 
  409.               $new_forum_args['group_id'] = $group_id; 
  410.           } 
  411.  
  412.           // Run the BP-specific functions for new groups 
  413.           $this->new_forum( $new_forum_args ); 
  414.       } 
  415.  
  416.       // Redirect after save when not in admin 
  417.       if ( !is_admin() ) { 
  418.           bp_core_redirect( trailingslashit( bp_get_group_permalink( buddypress()->groups->current_group ) . '/admin/' . $this->slug ) ); 
  419.       } 
  420.   }
