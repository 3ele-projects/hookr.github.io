  434.    
  435.   /** 
  436.    * Bail if a root template was already found. This prevents unintended 
  437.    * recursive filtering of 'the_content'. 
  438.    * 
  439.    * @link http://bbpress.trac.wordpress.org/ticket/2429 
  440.    */ 
  441.   if ( bbp_is_template_included() ) { 
  442.       return $template; 
  443.   } 
  444.  
  445.   /** 
  446.    * If BuddyPress is activated at a network level, the action order is 
  447.    * reversed, which causes the template integration to fail. If we're looking 
  448.    * at a BuddyPress page here, bail to prevent the extra processing. 
  449.    * 
  450.    * This is a bit more brute-force than is probably necessary, but gets the 
  451.    * job done while we work towards something more elegant. 
  452.    */ 
  453.   if ( function_exists( 'is_buddypress' ) && is_buddypress() ) 
  454.       return $template; 
  455.  
  456.   // Define local variable(s) 
  457.   $bbp_shortcodes = bbpress()->shortcodes; 
  458.  
  459.   // Bail if shortcodes are unset somehow 
  460.   if ( !is_a( $bbp_shortcodes, 'BBP_Shortcodes' ) ) 
  461.       return $template; 
  462.  
  463.   /** Users *************************************************************/ 
  464.  
  465.   if ( bbp_is_single_user_edit() || bbp_is_single_user() ) { 
  466.  
  467.       // Reset post 
  468.       bbp_theme_compat_reset_post( array( 
  469.           'ID' => 0,  
  470.           'post_author' => 0,  
  471.           'post_date' => 0,  
  472.           'post_content' => bbp_buffer_template_part( 'content', 'single-user', false ),  
  473.           'post_type' => '',  
  474.           'post_title' => bbp_get_displayed_user_field( 'display_name' ),  
  475.           'post_status' => bbp_get_public_status_id(),  
  476.           'is_archive' => false,  
  477.           'comment_status' => 'closed' 
  478.  ) ); 
  479.  
  480.   /** Forums ************************************************************/ 
  481.  
  482.   // Forum archive 
  483.   } elseif ( bbp_is_forum_archive() ) { 
  484.  
  485.       // Page exists where this archive should be 
  486.       $page = bbp_get_page_by_path( bbp_get_root_slug() ); 
  487.  
  488.       // Should we replace the content... 
  489.       if ( empty( $page->post_content ) ) { 
  490.  
  491.           // Use the topics archive 
  492.           if ( 'topics' === bbp_show_on_root() ) { 
  493.               $new_content = $bbp_shortcodes->display_topic_index(); 
  494.  
  495.           // No page so show the archive 
  496.           } else { 
  497.               $new_content = $bbp_shortcodes->display_forum_index(); 
  498.           } 
  499.  
  500.       // ...or use the existing page content? 
  501.       } else { 
  502.           $new_content = apply_filters( 'the_content', $page->post_content ); 
  503.       } 
  504.  
  505.       // Should we replace the title... 
  506.       if ( empty( $page->post_title ) ) { 
  507.  
  508.           // Use the topics archive 
  509.           if ( 'topics' === bbp_show_on_root() ) { 
  510.               $new_title = bbp_get_topic_archive_title(); 
  511.  
  512.           // No page so show the archive 
  513.           } else { 
  514.               $new_title = bbp_get_forum_archive_title(); 
  515.           } 
  516.  
  517.       // ...or use the existing page title? 
  518.       } else { 
  519.           $new_title = apply_filters( 'the_title', $page->post_title ); 
  520.       } 
  521.  
  522.       // Reset post 
  523.       bbp_theme_compat_reset_post( array( 
  524.           'ID' => !empty( $page->ID ) ? $page->ID : 0,  
  525.           'post_title' => $new_title,  
  526.           'post_author' => 0,  
  527.           'post_date' => 0,  
  528.           'post_content' => $new_content,  
  529.           'post_type' => bbp_get_forum_post_type(),  
  530.           'post_status' => bbp_get_public_status_id(),  
  531.           'is_archive' => true,  
  532.           'comment_status' => 'closed' 
  533.  ) ); 
  534.  
  535.   // Single Forum 
  536.   } elseif ( bbp_is_forum_edit() ) { 
  537.  
  538.       // Reset post 
  539.       bbp_theme_compat_reset_post( array( 
  540.           'ID' => bbp_get_forum_id(),  
  541.           'post_title' => bbp_get_forum_title(),  
  542.           'post_author' => bbp_get_forum_author_id(),  
  543.           'post_date' => 0,  
  544.           'post_content' => $bbp_shortcodes->display_forum_form(),

  551.   } elseif ( bbp_is_single_forum() ) { 
  552.  
  553.       // Reset post

  749.    * template files for users, topics, forums, etc...

  751.    * We do this after the above checks to prevent incorrect 404 body classes 
  752.    * and header statuses, as well as to set the post global as needed.

  763.    * 
  764.    * To do this, we first remove all filters from 'the_content' and hook 
  765.    * our own function into it, which runs a series of checks to determine 
  766.    * the context, and then uses the built in shortcodes to output the

  771.    * navigation, comments, date/time, etc...) 
  772.    * 
  773.    * Hook into the 'bbp_get_bbpress_template' to override the array of 
  774.    * possible templates, or 'bbp_bbpress_template' to override the result. 
  775.    */ 
  776.   } elseif ( bbp_is_theme_compat_active() ) { 
  777.       bbp_remove_all_filters( 'the_content' ); 
  778.  
  779.       $template = bbp_get_theme_compat_templates(); 
  780.   } 
  781.  
  782.   return apply_filters( 'bbp_template_include_theme_compat', $template ); 
  783. }

  787. /** 
  788.  * Remove the canonical redirect to allow pretty pagination 
  789.  * 
  790.  * @since bbPress (r2628) 
  791.  * @param string $redirect_url Redirect url 
  792.  * @uses WP_Rewrite::using_permalinks() To check if the blog is using permalinks 
  793.  * @uses bbp_get_paged() To get the current page number 
  794.  * @uses bbp_is_single_topic() To check if it's a topic page 
  795.  * @uses bbp_is_single_forum() To check if it's a forum page 
  796.  * @return bool|string False if it's a topic/forum and their first page,  
  797.  *                      otherwise the redirect url 
  798.  */ 
  799. function bbp_redirect_canonical( $redirect_url ) { 
  800.   global $wp_rewrite; 
  801.  
  802.   // Canonical is for the beautiful 
  803.   if ( $wp_rewrite->using_permalinks() ) { 
  804.  
  805.       // If viewing beyond page 1 of several 
  806.       if ( 1 < bbp_get_paged() ) { 
  807.  
  808.           // Only on single topics... 
  809.           if ( bbp_is_single_topic() ) { 
  810.               $redirect_url = false; 
  811.  
  812.           // ...and single forums... 
  813.           } elseif ( bbp_is_single_forum() ) { 
  814.               $redirect_url = false; 
  815.  
  816.           // ...and single replies... 
  817.           } elseif ( bbp_is_single_reply() ) { 
  818.               $redirect_url = false; 
  819.  
  820.           // ...and any single anything else... 
  821.           // 
  822.           // @todo - Find a more accurate way to disable paged canonicals for 
  823.           //          paged shortcode usage within other posts. 
  824.           } elseif ( is_page() || is_singular() ) { 
  825.               $redirect_url = false; 
  826.           } 
  827.  
  828.       // If editing a topic 
  829.       } elseif ( bbp_is_topic_edit() ) { 
  830.           $redirect_url = false; 
  831.  
  832.       // If editing a reply 
  833.       } elseif ( bbp_is_reply_edit() ) { 
  834.           $redirect_url = false; 
  835.       } 
  836.   } 
  837.  
  838.   return $redirect_url; 
  839. }

  843. /** 
  844.  * Removes all filters from a WordPress filter, and stashes them in the $bbp 
  845.  * global in the event they need to be restored later. 
  846.  * 
  847.  * @since bbPress (r3251) 
  848.  * @global WP_filter $wp_filter 
  849.  * @global array $merged_filters 
  850.  * @param string $tag 
  851.  * @param int $priority 
  852.  * @return bool 
  853.  */ 
  854. function bbp_remove_all_filters( $tag, $priority = false ) { 
  855.   global $wp_filter, $merged_filters; 
  856.  
  857.   $bbp = bbpress(); 
  858.  
  859.   // Filters exist 
  860.   if ( isset( $wp_filter[$tag] ) ) { 
  861.  
  862.       // Filters exist in this priority 
  863.       if ( !empty( $priority ) && isset( $wp_filter[$tag][$priority] ) ) { 
  864.  
  865.           // Store filters in a backup 
  866.           $bbp->filters->wp_filter[$tag][$priority] = $wp_filter[$tag][$priority]; 
  867.  
  868.           // Unset the filters 
  869.           unset( $wp_filter[$tag][$priority] ); 
  870.  
  871.       // Priority is empty 
  872.       } else { 
  873.  
  874.           // Store filters in a backup 
  875.           $bbp->filters->wp_filter[$tag] = $wp_filter[$tag]; 
  876.  
  877.           // Unset the filters 
  878.           unset( $wp_filter[$tag] ); 
  879.       } 
  880.   } 
  881.  
  882.   // Check merged filters 
  883.   if ( isset( $merged_filters[$tag] ) ) { 
  884.  
  885.       // Store filters in a backup 
  886.       $bbp->filters->merged_filters[$tag] = $merged_filters[$tag]; 
  887.  
  888.       // Unset the filters 
  889.       unset( $merged_filters[$tag] ); 
  890.   } 
  891.  
  892.   return true; 
  893. }
