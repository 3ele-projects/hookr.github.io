  1111.   public function widget( $args, $instance ) {

  1115.  
  1116.       // Typical WordPress filter

  1118.  
  1119.       // bbPress filter 
  1120.       $settings['title'] = apply_filters( 'bbp_replies_widget_title', $settings['title'], $instance, $this->id_base ); 
  1121.  
  1122.       // Note: private and hidden forums will be excluded via the 
  1123.       // bbp_pre_get_posts_normalize_forum_visibility action and function. 
  1124.       $widget_query = new WP_Query( array( 
  1125.           'post_type' => bbp_get_reply_post_type(),  
  1126.           'post_status' => array( bbp_get_public_status_id(), bbp_get_closed_status_id() ),  
  1127.           'posts_per_page' => (int) $settings['max_shown'],  
  1128.           'ignore_sticky_posts' => true,  
  1129.           'no_found_rows' => true,  
  1130.  ) );

  1135.       }

  1138.

  1140.           echo $args['before_title'] . $settings['title'] . $args['after_title']; 
  1141.       } ?>

  1154.  
  1155.                   // Only query user if showing them

  1168.                   // Reply link and timestamp

  1173.  
  1174.                   // Reply author and title 
  1175.                   elseif ( !empty( $author_link ) ) : 
  1176.  
  1177.                       // translators: 1: reply author, 2: reply link 
  1178.                       printf( _x( '%1$s on %2$s', 'widgets', 'bbpress' ), $author_link, $reply_link ); 
  1179.  
  1180.                   // Only the reply title 
  1181.                   else : 
  1182.  
  1183.                       // translators: 1: reply link 
  1184.                       printf( _x( '%1$s', 'widgets', 'bbpress' ), $reply_link ); 
  1185.  
  1186.                   endif; 
  1187.  
  1188.                   ?> 
  1189.  
  1190.               </li> 
  1191.  
  1192.           <?php endwhile; ?> 
  1193.  
  1194.       </ul> 
  1195.  
  1196.       <?php echo $args['after_widget']; 
  1197.  
  1198.       // Reset the $post global 
  1199.       wp_reset_postdata();
