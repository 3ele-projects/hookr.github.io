  586.  * @uses bbp_is_user_spammer() To check if user is spammer 
  587.  * @uses bbp_is_user_deleted() To check if user is deleted 
  588.  * 
  589.  * @param int $user_id The user ID to check 
  590.  * @return bool True if public, false if not 
  591.  */ 
  592. function bbp_is_user_active( $user_id = 0 ) { 
  593.  
  594.   // Default to current user 
  595.   if ( empty( $user_id ) && is_user_logged_in() ) 
  596.       $user_id = bbp_get_current_user_id(); 
  597.  
  598.   // No user to check 
  599.   if ( empty( $user_id ) ) 
  600.       return false; 
  601.  
  602.   // Check spam 
  603.   if ( bbp_is_user_spammer( $user_id ) ) 
  604.       return false; 
  605.  
  606.   // Check deleted 
  607.   if ( bbp_is_user_deleted( $user_id ) ) 
  608.       return false; 
  609.  
  610.   // Assume true if not spam or deleted 
  611.   return true; 
  612. }

  615.  * Checks if user is not active. 
  616.  * 
  617.  * @since bbPress (r3502) 
  618.  * 
  619.  * @uses is_user_logged_in() To check if user is logged in 
  620.  * @uses bbp_get_displayed_user_id() To get current user ID 
  621.  * @uses bbp_is_user_active() To check if user is active 
  622.  * 
  623.  * @param int $user_id The user ID to check. Defaults to current user ID 
  624.  * @return bool True if inactive, false if active 
  625.  */ 
  626. function bbp_is_user_inactive( $user_id = 0 ) { 
  627.  
  628.   // Default to current user 
  629.   if ( empty( $user_id ) && is_user_logged_in() ) 
  630.       $user_id = bbp_get_current_user_id(); 
  631.  
  632.   // No user to check 
  633.   if ( empty( $user_id ) ) 
  634.       return false; 
  635.  
  636.   // Return the inverse of active 
  637.   return !bbp_is_user_active( $user_id ); 
  638. }

  640. /** 
  641.  * Checks if user is a keymaster 
  642.  * 
  643.  * @since bbPress (r4783) 
  644.  * 
  645.  * @param int $user_id  
  646.  * @return bool True if keymaster, false if not 
  647.  */ 
  648. function bbp_is_user_keymaster( $user_id = 0 ) { 
  649.  
  650.   // Default to current user ID if none is passed 
  651.   $_user_id = (int) ! empty( $user_id ) ? $user_id : bbp_get_current_user_id(); 
  652.  
  653.   // Filter and return 
  654.   return (bool) apply_filters( 'bbp_is_user_keymaster', user_can( $_user_id, 'keep_gate' ), $_user_id, $user_id ); 
  655. }

  657. /** 
  658.  * Does a user have a profile for the current site 
  659.  * 
  660.  * @since bbPress (r4362) 
  661.  * 
  662.  * @param int $user_id User ID to check 
  663.  * @param int $blog_id Blog ID to check 
  664.  * 
  665.  * @uses bbp_get_user_id() To verify the user ID 
  666.  * @uses get_userdata() To get the user's data 
  667.  * @uses bbp_is_user_keymaster() To determine if user can see inactive users 
  668.  * @uses bbp_is_user_inactive() To check if user is spammer or deleted 
  669.  * @uses apply_filters() To allow override of this functions result 
  670.  * 
  671.  * @return boolean Whether or not the user has a profile on this blog_id 
  672.  */ 
  673. function bbp_user_has_profile( $user_id = 0 ) { 
  674.  
  675.   // Assume every user has a profile 
  676.   $retval = true; 
  677.  
  678.   // Validate user ID, default to displayed or current user 
  679.   $user_id = bbp_get_user_id( $user_id, true, true ); 
  680.  
  681.   // Try to get this user's data 
  682.   $user = get_userdata( $user_id ); 
  683.  
  684.   // No user found, return false 
  685.   if ( empty( $user ) ) { 
  686.       $retval = false; 
  687.  
  688.   // User is inactive, and current user is not a keymaster 
  689.   } elseif ( ! bbp_is_user_keymaster() && bbp_is_user_inactive( $user->ID ) ) { 
  690.       $retval = false; 
  691.   } 
  692.  
  693.   // Filter and return 
  694.   return (bool) apply_filters( 'bbp_show_user_profile', $retval, $user_id ); 
  695. }
