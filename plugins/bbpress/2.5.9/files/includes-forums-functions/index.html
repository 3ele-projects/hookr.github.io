  49.       return false;

  110. function bbp_new_forum_handler( $action = '' ) { 
  111.  
  112.   // Bail if action is not bbp-new-forum 
  113.   if ( 'bbp-new-forum' !== $action ) 
  114.       return; 
  115.  
  116.   // Nonce check 
  117.   if ( ! bbp_verify_nonce_request( 'bbp-new-forum' ) ) { 
  118.       bbp_add_error( 'bbp_new_forum_nonce', __( '<strong>ERROR</strong>: Are you sure you wanted to do that?', 'bbpress' ) ); 
  119.       return; 
  120.   } 
  121.  
  122.   // Define local variable(s) 
  123.   $view_all = $anonymous_data = false; 
  124.   $forum_parent_id = $forum_author = 0; 
  125.   $forum_title = $forum_content = ''; 
  126.  
  127.   /** Forum Author **********************************************************/ 
  128.  
  129.   // User cannot create forums 
  130.   if ( !current_user_can( 'publish_forums' ) ) { 
  131.       bbp_add_error( 'bbp_forum_permissions', __( '<strong>ERROR</strong>: You do not have permission to create new forums.', 'bbpress' ) ); 
  132.       return; 
  133.   } 
  134.  
  135.   // Forum author is current user 
  136.   $forum_author = bbp_get_current_user_id(); 
  137.  
  138.   // Remove kses filters from title and content for capable users and if the nonce is verified 
  139.   if ( current_user_can( 'unfiltered_html' ) && !empty( $_POST['_bbp_unfiltered_html_forum'] ) && wp_create_nonce( 'bbp-unfiltered-html-forum_new' ) === $_POST['_bbp_unfiltered_html_forum'] ) { 
  140.       remove_filter( 'bbp_new_forum_pre_title', 'wp_filter_kses' ); 
  141.       remove_filter( 'bbp_new_forum_pre_content', 'bbp_encode_bad', 10 ); 
  142.       remove_filter( 'bbp_new_forum_pre_content', 'bbp_filter_kses', 30 ); 
  143.   } 
  144.  
  145.   /** Forum Title ***********************************************************/ 
  146.  
  147.   if ( !empty( $_POST['bbp_forum_title'] ) ) 
  148.       $forum_title = esc_attr( strip_tags( $_POST['bbp_forum_title'] ) ); 
  149.  
  150.   // Filter and sanitize 
  151.   $forum_title = apply_filters( 'bbp_new_forum_pre_title', $forum_title ); 
  152.  
  153.   // No forum title 
  154.   if ( empty( $forum_title ) ) 
  155.       bbp_add_error( 'bbp_forum_title', __( '<strong>ERROR</strong>: Your forum needs a title.', 'bbpress' ) ); 
  156.  
  157.   /** Forum Content *********************************************************/ 
  158.  
  159.   if ( !empty( $_POST['bbp_forum_content'] ) ) 
  160.       $forum_content = $_POST['bbp_forum_content']; 
  161.  
  162.   // Filter and sanitize 
  163.   $forum_content = apply_filters( 'bbp_new_forum_pre_content', $forum_content ); 
  164.  
  165.   // No forum content 
  166.   if ( empty( $forum_content ) ) 
  167.       bbp_add_error( 'bbp_forum_content', __( '<strong>ERROR</strong>: Your forum description cannot be empty.', 'bbpress' ) ); 
  168.  
  169.   /** Forum Parent **********************************************************/ 
  170.  
  171.   // Forum parent was passed (the norm) 
  172.   if ( !empty( $_POST['bbp_forum_parent_id'] ) ) { 
  173.       $forum_parent_id = bbp_get_forum_id( $_POST['bbp_forum_parent_id'] ); 
  174.   } 
  175.  
  176.   // Filter and sanitize 
  177.   $forum_parent_id = apply_filters( 'bbp_new_forum_pre_parent_id', $forum_parent_id ); 
  178.  
  179.   // No forum parent was passed (should never happen) 
  180.   if ( empty( $forum_parent_id ) ) { 
  181.       bbp_add_error( 'bbp_new_forum_missing_parent', __( '<strong>ERROR</strong>: Your forum must have a parent.', 'bbpress' ) ); 
  182.  
  183.   // Forum exists 
  184.   } elseif ( !empty( $forum_parent_id ) ) { 
  185.  
  186.       // Forum is a category 
  187.       if ( bbp_is_forum_category( $forum_parent_id ) ) { 
  188.           bbp_add_error( 'bbp_new_forum_forum_category', __( '<strong>ERROR</strong>: This forum is a category. No forums can be created in this forum.', 'bbpress' ) ); 
  189.       } 
  190.  
  191.       // Forum is closed and user cannot access 
  192.       if ( bbp_is_forum_closed( $forum_parent_id ) && !current_user_can( 'edit_forum', $forum_parent_id ) ) { 
  193.           bbp_add_error( 'bbp_new_forum_forum_closed', __( '<strong>ERROR</strong>: This forum has been closed to new forums.', 'bbpress' ) ); 
  194.       } 
  195.  
  196.       // Forum is private and user cannot access 
  197.       if ( bbp_is_forum_private( $forum_parent_id ) && !current_user_can( 'read_private_forums' ) ) { 
  198.           bbp_add_error( 'bbp_new_forum_forum_private', __( '<strong>ERROR</strong>: This forum is private and you do not have the capability to read or create new forums in it.', 'bbpress' ) ); 
  199.       } 
  200.  
  201.       // Forum is hidden and user cannot access 
  202.       if ( bbp_is_forum_hidden( $forum_parent_id ) && !current_user_can( 'read_hidden_forums' ) ) { 
  203.           bbp_add_error( 'bbp_new_forum_forum_hidden', __( '<strong>ERROR</strong>: This forum is hidden and you do not have the capability to read or create new forums in it.', 'bbpress' ) ); 
  204.       } 
  205.   } 
  206.  
  207.   /** Forum Flooding ********************************************************/ 
  208.  
  209.   if ( !bbp_check_for_flood( $anonymous_data, $forum_author ) ) 
  210.       bbp_add_error( 'bbp_forum_flood', __( '<strong>ERROR</strong>: Slow down; you move too fast.', 'bbpress' ) ); 
  211.  
  212.   /** Forum Duplicate *******************************************************/ 
  213.  
  214.   if ( !bbp_check_for_duplicate( array( 'post_type' => bbp_get_forum_post_type(), 'post_author' => $forum_author, 'post_content' => $forum_content, 'anonymous_data' => $anonymous_data ) ) ) 
  215.       bbp_add_error( 'bbp_forum_duplicate', __( '<strong>ERROR</strong>: This forum already exists.', 'bbpress' ) ); 
  216.  
  217.   /** Forum Blacklist *******************************************************/ 
  218.  
  219.   if ( !bbp_check_for_blacklist( $anonymous_data, $forum_author, $forum_title, $forum_content ) ) 
  220.       bbp_add_error( 'bbp_forum_blacklist', __( '<strong>ERROR</strong>: Your forum cannot be created at this time.', 'bbpress' ) ); 
  221.  
  222.   /** Forum Moderation ******************************************************/ 
  223.  
  224.   $post_status = bbp_get_public_status_id(); 
  225.   if ( !bbp_check_for_moderation( $anonymous_data, $forum_author, $forum_title, $forum_content ) ) 
  226.       $post_status = bbp_get_pending_status_id(); 
  227.  
  228.   /** Additional Actions (Before Save) **************************************/ 
  229.  
  230.   do_action( 'bbp_new_forum_pre_extras', $forum_parent_id ); 
  231.  
  232.   // Bail if errors 
  233.   if ( bbp_has_errors() ) 
  234.       return; 
  235.  
  236.   /** No Errors *************************************************************/ 
  237.  
  238.   // Add the content of the form to $forum_data as an array 
  239.   // Just in time manipulation of forum data before being created 
  240.   $forum_data = apply_filters( 'bbp_new_forum_pre_insert', array( 
  241.       'post_author' => $forum_author,  
  242.       'post_title' => $forum_title,  
  243.       'post_content' => $forum_content,  
  244.       'post_parent' => $forum_parent_id,  
  245.       'post_status' => $post_status,  
  246.       'post_type' => bbp_get_forum_post_type(),  
  247.       'comment_status' => 'closed' 
  248.  ) ); 
  249.  
  250.   // Insert forum 
  251.   $forum_id = wp_insert_post( $forum_data ); 
  252.  
  253.   /** No Errors *************************************************************/ 
  254.  
  255.   if ( !empty( $forum_id ) && !is_wp_error( $forum_id ) ) { 
  256.  
  257.       /** Trash Check *******************************************************/ 
  258.  
  259.       // If the forum is trash, or the forum_status is switched to 
  260.       // trash, trash it properly 
  261.       if ( ( get_post_field( 'post_status', $forum_id ) === bbp_get_trash_status_id() ) || ( $forum_data['post_status'] === bbp_get_trash_status_id() ) ) { 
  262.  
  263.           // Trash the reply 
  264.           wp_trash_post( $forum_id ); 
  265.  
  266.           // Force view=all 
  267.           $view_all = true; 
  268.       } 
  269.  
  270.       /** Spam Check ********************************************************/ 
  271.  
  272.       // If reply or forum are spam, officially spam this reply 
  273.       if ( $forum_data['post_status'] === bbp_get_spam_status_id() ) { 
  274.           add_post_meta( $forum_id, '_bbp_spam_meta_status', bbp_get_public_status_id() ); 
  275.  
  276.           // Force view=all 
  277.           $view_all = true; 
  278.       } 
  279.  
  280.       /** Update counts, etc... *********************************************/ 
  281.  
  282.       do_action( 'bbp_new_forum', array( 
  283.           'forum_id' => $forum_id,  
  284.           'post_parent' => $forum_parent_id,  
  285.           'forum_author' => $forum_author,  
  286.           'last_topic_id' => 0,  
  287.           'last_reply_id' => 0,  
  288.           'last_active_id' => 0,  
  289.           'last_active_time' => 0,  
  290.           'last_active_status' => bbp_get_public_status_id() 
  291.  ) ); 
  292.  
  293.       /** Additional Actions (After Save) ***********************************/ 
  294.  
  295.       do_action( 'bbp_new_forum_post_extras', $forum_id ); 
  296.  
  297.       /** Redirect **********************************************************/ 
  298.  
  299.       // Redirect to 
  300.       $redirect_to = bbp_get_redirect_to(); 
  301.  
  302.       // Get the forum URL 
  303.       $redirect_url = bbp_get_forum_permalink( $forum_id, $redirect_to ); 
  304.  
  305.       // Add view all? 
  306.       if ( bbp_get_view_all() || !empty( $view_all ) ) { 
  307.  
  308.           // User can moderate, so redirect to forum with view all set 
  309.           if ( current_user_can( 'moderate' ) ) { 
  310.               $redirect_url = bbp_add_view_all( $redirect_url ); 
  311.  
  312.           // User cannot moderate, so redirect to forum 
  313.           } else { 
  314.               $redirect_url = bbp_get_forum_permalink( $forum_id ); 
  315.           } 
  316.       } 
  317.  
  318.       // Allow to be filtered 
  319.       $redirect_url = apply_filters( 'bbp_new_forum_redirect_to', $redirect_url, $redirect_to ); 
  320.  
  321.       /** Successful Save ***************************************************/ 
  322.  
  323.       // Redirect back to new forum 
  324.       wp_safe_redirect( $redirect_url ); 
  325.  
  326.       // For good measure 
  327.       exit(); 
  328.  
  329.   // Errors 
  330.   } else { 
  331.       $append_error = ( is_wp_error( $forum_id ) && $forum_id->get_error_message() ) ? $forum_id->get_error_message() . ' ' : ''; 
  332.       bbp_add_error( 'bbp_forum_error', __( '<strong>ERROR</strong>: The following problem(s) have been found with your forum:' . $append_error, 'bbpress' ) ); 
  333.   } 
  334. }

  369. function bbp_edit_forum_handler( $action = '' ) { 
  370.  
  371.   // Bail if action is not bbp-edit-forum 
  372.   if ( 'bbp-edit-forum' !== $action ) 
  373.       return; 
  374.  
  375.   // Define local variable(s) 
  376.   $anonymous_data = array(); 
  377.   $forum = $forum_id = $forum_parent_id = 0; 
  378.   $forum_title = $forum_content = $forum_edit_reason = ''; 
  379.  
  380.   /** Forum *****************************************************************/ 
  381.  
  382.   // Forum id was not passed 
  383.   if ( empty( $_POST['bbp_forum_id'] ) ) { 
  384.       bbp_add_error( 'bbp_edit_forum_id', __( '<strong>ERROR</strong>: Forum ID not found.', 'bbpress' ) ); 
  385.       return; 
  386.  
  387.   // Forum id was passed 
  388.   } elseif ( is_numeric( $_POST['bbp_forum_id'] ) ) { 
  389.       $forum_id = (int) $_POST['bbp_forum_id']; 
  390.       $forum = bbp_get_forum( $forum_id ); 
  391.   } 
  392.  
  393.   // Nonce check 
  394.   if ( ! bbp_verify_nonce_request( 'bbp-edit-forum_' . $forum_id ) ) { 
  395.       bbp_add_error( 'bbp_edit_forum_nonce', __( '<strong>ERROR</strong>: Are you sure you wanted to do that?', 'bbpress' ) ); 
  396.       return; 
  397.  
  398.   // Forum does not exist 
  399.   } elseif ( empty( $forum ) ) { 
  400.       bbp_add_error( 'bbp_edit_forum_not_found', __( '<strong>ERROR</strong>: The forum you want to edit was not found.', 'bbpress' ) ); 
  401.       return; 
  402.  
  403.   // User cannot edit this forum 
  404.   } elseif ( !current_user_can( 'edit_forum', $forum_id ) ) { 
  405.       bbp_add_error( 'bbp_edit_forum_permissions', __( '<strong>ERROR</strong>: You do not have permission to edit that forum.', 'bbpress' ) ); 
  406.       return; 
  407.   } 
  408.  
  409.   // Remove kses filters from title and content for capable users and if the nonce is verified 
  410.   if ( current_user_can( 'unfiltered_html' ) && !empty( $_POST['_bbp_unfiltered_html_forum'] ) && ( wp_create_nonce( 'bbp-unfiltered-html-forum_' . $forum_id ) === $_POST['_bbp_unfiltered_html_forum'] ) ) { 
  411.       remove_filter( 'bbp_edit_forum_pre_title', 'wp_filter_kses' ); 
  412.       remove_filter( 'bbp_edit_forum_pre_content', 'bbp_encode_bad', 10 ); 
  413.       remove_filter( 'bbp_edit_forum_pre_content', 'bbp_filter_kses', 30 ); 
  414.   } 
  415.  
  416.   /** Forum Parent ***********************************************************/ 
  417.  
  418.   // Forum parent id was passed 
  419.   if ( !empty( $_POST['bbp_forum_parent_id'] ) ) { 
  420.       $forum_parent_id = bbp_get_forum_id( $_POST['bbp_forum_parent_id'] ); 
  421.   } 
  422.  
  423.   // Current forum this forum is in 
  424.   $current_parent_forum_id = bbp_get_forum_parent_id( $forum_id ); 
  425.  
  426.   // Forum exists 
  427.   if ( !empty( $forum_parent_id ) && ( $forum_parent_id !== $current_parent_forum_id ) ) { 
  428.  
  429.       // Forum is closed and user cannot access 
  430.       if ( bbp_is_forum_closed( $forum_parent_id ) && !current_user_can( 'edit_forum', $forum_parent_id ) ) { 
  431.           bbp_add_error( 'bbp_edit_forum_forum_closed', __( '<strong>ERROR</strong>: This forum has been closed to new forums.', 'bbpress' ) ); 
  432.       } 
  433.  
  434.       // Forum is private and user cannot access 
  435.       if ( bbp_is_forum_private( $forum_parent_id ) && !current_user_can( 'read_private_forums' ) ) { 
  436.           bbp_add_error( 'bbp_edit_forum_forum_private', __( '<strong>ERROR</strong>: This forum is private and you do not have the capability to read or create new forums in it.', 'bbpress' ) ); 
  437.       } 
  438.  
  439.       // Forum is hidden and user cannot access 
  440.       if ( bbp_is_forum_hidden( $forum_parent_id ) && !current_user_can( 'read_hidden_forums' ) ) { 
  441.           bbp_add_error( 'bbp_edit_forum_forum_hidden', __( '<strong>ERROR</strong>: This forum is hidden and you do not have the capability to read or create new forums in it.', 'bbpress' ) ); 
  442.       } 
  443.   } 
  444.  
  445.   /** Forum Title ***********************************************************/ 
  446.  
  447.   if ( !empty( $_POST['bbp_forum_title'] ) ) 
  448.       $forum_title = esc_attr( strip_tags( $_POST['bbp_forum_title'] ) ); 
  449.  
  450.   // Filter and sanitize 
  451.   $forum_title = apply_filters( 'bbp_edit_forum_pre_title', $forum_title, $forum_id ); 
  452.

  652.           // Publish (default) 
  653.           case bbp_get_public_status_id()  : 
  654.           default        :

  657.       } 
  658.   }
