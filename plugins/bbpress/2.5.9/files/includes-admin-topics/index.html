  265.   /** 
  266.    * Pass the topic attributes for processing 
  267.    * 
  268.    * @since bbPress (r2746) 
  269.    * 
  270.    * @param int $topic_id Topic id 
  271.    * @uses current_user_can() To check if the current user is capable of 
  272.    *                           editing the topic 
  273.    * @uses do_action() Calls 'bbp_topic_attributes_metabox_save' with the 
  274.    *                    topic id and parent id 
  275.    * @return int Parent id 
  276.    */ 
  277.   public function attributes_metabox_save( $topic_id ) { 
  278.  
  279.       if ( $this->bail() ) return $topic_id; 
  280.  
  281.       // Bail if doing an autosave 
  282.       if ( defined( 'DOING_AUTOSAVE' ) && DOING_AUTOSAVE ) 
  283.           return $topic_id; 
  284.  
  285.       // Bail if not a post request 
  286.       if ( ! bbp_is_post_request() ) 
  287.           return $topic_id; 
  288.  
  289.       // Nonce check 
  290.       if ( empty( $_POST['bbp_topic_metabox'] ) || !wp_verify_nonce( $_POST['bbp_topic_metabox'], 'bbp_topic_metabox_save' ) ) 
  291.           return $topic_id; 
  292.  
  293.       // Bail if current user cannot edit this topic 
  294.       if ( !current_user_can( 'edit_topic', $topic_id ) ) 
  295.           return $topic_id; 
  296.  
  297.       // Get the forum ID 
  298.       $forum_id = !empty( $_POST['parent_id'] ) ? (int) $_POST['parent_id'] : 0; 
  299.  
  300.       // Get topic author data 
  301.       $anonymous_data = bbp_filter_anonymous_post_data(); 
  302.       $author_id = bbp_get_topic_author_id( $topic_id ); 
  303.       $is_edit = (bool) isset( $_POST['save'] ); 
  304.  
  305.       // Formally update the topic 
  306.       bbp_update_topic( $topic_id, $forum_id, $anonymous_data, $author_id, $is_edit ); 
  307.  
  308.       // Stickies 
  309.       if ( !empty( $_POST['bbp_stick_topic'] ) && in_array( $_POST['bbp_stick_topic'], array( 'stick', 'super', 'unstick' ) ) ) { 
  310.  
  311.           // What's the haps? 
  312.           switch ( $_POST['bbp_stick_topic'] ) { 
  313.  
  314.               // Sticky in this forum 
  315.               case 'stick'   : 
  316.                   bbp_stick_topic( $topic_id ); 
  317.                   break; 
  318.  
  319.               // Super sticky in all forums 
  320.               case 'super'   : 
  321.                   bbp_stick_topic( $topic_id, true ); 
  322.                   break; 
  323.  
  324.               // Normal 
  325.               case 'unstick' : 
  326.               default        : 
  327.                   bbp_unstick_topic( $topic_id ); 
  328.                   break; 
  329.           } 
  330.       } 
  331.  
  332.       // Allow other fun things to happen 
  333.       do_action( 'bbp_topic_attributes_metabox_save', $topic_id, $forum_id ); 
  334.       do_action( 'bbp_author_metabox_save', $topic_id, $anonymous_data ); 
  335.  
  336.       return $topic_id; 
  337.   }

  474.  
  475.       // Only proceed if GET is a topic toggle action 
  476.       if ( bbp_is_get_request() && !empty( $_GET['action'] ) && in_array( $_GET['action'], array( 'bbp_toggle_topic_close', 'bbp_toggle_topic_stick', 'bbp_toggle_topic_spam' ) ) && !empty( $_GET['topic_id'] ) ) { 
  477.           $action = $_GET['action'];            // What action is taking place? 
  478.           $topic_id = (int) $_GET['topic_id'];    // What's the topic id? 
  479.           $success = false;                      // Flag 
  480.           $post_data = array( 'ID' => $topic_id ); // Prelim array 
  481.           $topic = bbp_get_topic( $topic_id ); 
  482.  
  483.           // Bail if topic is missing 
  484.           if ( empty( $topic ) ) 
  485.               wp_die( __( 'The topic was not found!', 'bbpress' ) ); 
  486.  
  487.           if ( !current_user_can( 'moderate', $topic->ID ) ) // What is the user doing here? 
  488.               wp_die( __( 'You do not have the permission to do that!', 'bbpress' ) ); 
  489.  
  490.           switch ( $action ) { 
  491.               case 'bbp_toggle_topic_close' : 
  492.                   check_admin_referer( 'close-topic_' . $topic_id ); 
  493.  
  494.                   $is_open = bbp_is_topic_open( $topic_id ); 
  495.                   $message = true === $is_open ? 'closed' : 'opened'; 
  496.                   $success = true === $is_open ? bbp_close_topic( $topic_id ) : bbp_open_topic( $topic_id ); 
  497.  
  498.                   break; 
  499.  
  500.               case 'bbp_toggle_topic_stick' : 
  501.                   check_admin_referer( 'stick-topic_' . $topic_id ); 
  502.  
  503.                   $is_sticky = bbp_is_topic_sticky( $topic_id ); 
  504.                   $is_super = false === $is_sticky && !empty( $_GET['super'] ) && ( "1" === $_GET['super'] ) ? true : false; 
  505.                   $message = true === $is_sticky ? 'unsticked'     : 'sticked'; 
  506.                   $message = true === $is_super  ? 'super_sticked' : $message; 
  507.                   $success = true === $is_sticky ? bbp_unstick_topic( $topic_id ) : bbp_stick_topic( $topic_id, $is_super ); 
  508.  
  509.                   break; 
  510.  
  511.               case 'bbp_toggle_topic_spam'  : 
  512.                   check_admin_referer( 'spam-topic_' . $topic_id ); 
  513.  
  514.                   $is_spam = bbp_is_topic_spam( $topic_id ); 
  515.                   $message = true === $is_spam ? 'unspammed' : 'spammed'; 
  516.                   $success = true === $is_spam ? bbp_unspam_topic( $topic_id ) : bbp_spam_topic( $topic_id ); 
  517.  
  518.                   break; 
  519.           } 
  520.  
  521.           $message = array( 'bbp_topic_toggle_notice' => $message, 'topic_id' => $topic->ID ); 
  522.  
  523.           if ( false === $success || is_wp_error( $success ) ) 
  524.               $message['failed'] = '1'; 
  525.  
  526.           // Do additional topic toggle actions (admin side) 
  527.           do_action( 'bbp_toggle_topic_admin', $success, $post_data, $action, $message ); 
  528.  
  529.           // Redirect back to the topic 
  530.           $redirect = add_query_arg( $message, remove_query_arg( array( 'action', 'topic_id' ) ) ); 
  531.           wp_safe_redirect( $redirect ); 
  532.  
  533.           // For good measure 
  534.           exit(); 
  535.       } 
  536.   }
