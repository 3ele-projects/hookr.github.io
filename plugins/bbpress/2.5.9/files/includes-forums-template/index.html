  676. function bbp_forum_get_subforums( $args = '' ) {

  725. /** 
  726.  * Output a list of forums (can be used to list subforums) 
  727.  * 
  728.  * @param mixed $args The function supports these args: 
  729.  *  - before: To put before the output. Defaults to '<ul class="bbp-forums">' 
  730.  *  - after: To put after the output. Defaults to '</ul>' 
  731.  *  - link_before: To put before every link. Defaults to '<li class="bbp-forum">' 
  732.  *  - link_after: To put after every link. Defaults to '</li>' 
  733.  *  - separator: Separator. Defaults to ', ' 
  734.  *  - forum_id: Forum id. Defaults to '' 
  735.  *  - show_topic_count - To show forum topic count or not. Defaults to true 
  736.  *  - show_reply_count - To show forum reply count or not. Defaults to true 
  737.  * @uses bbp_forum_get_subforums() To check if the forum has subforums or not 
  738.  * @uses bbp_get_forum_permalink() To get forum permalink 
  739.  * @uses bbp_get_forum_title() To get forum title 
  740.  * @uses bbp_is_forum_category() To check if a forum is a category 
  741.  * @uses bbp_get_forum_topic_count() To get forum topic count 
  742.  * @uses bbp_get_forum_reply_count() To get forum reply count 
  743.  */ 
  744. function bbp_list_forums( $args = '' ) { 
  745.  
  746.   // Define used variables 
  747.   $output = $sub_forums = $topic_count = $reply_count = $counts = ''; 
  748.   $i = 0; 
  749.   $count = array(); 
  750.  
  751.   // Parse arguments against default values 
  752.   $r = bbp_parse_args( $args, array( 
  753.       'before' => '<ul class="bbp-forums-list">',  
  754.       'after' => '</ul>',  
  755.       'link_before' => '<li class="bbp-forum">',  
  756.       'link_after' => '</li>',  
  757.       'count_before' => ' (',  
  758.       'count_after' => ')',  
  759.       'count_sep' => ', ',  
  760.       'separator' => ', ',  
  761.       'forum_id' => '',  
  762.       'show_topic_count' => true,  
  763.       'show_reply_count' => true,  
  764.  ), 'list_forums' ); 
  765.  
  766.   // Loop through forums and create a list 
  767.   $sub_forums = bbp_forum_get_subforums( $r['forum_id'] ); 
  768.   if ( !empty( $sub_forums ) ) { 
  769.  
  770.       // Total count (for separator) 
  771.       $total_subs = count( $sub_forums ); 
  772.       foreach ( $sub_forums as $sub_forum ) { 
  773.           $i++; // Separator count 
  774.  
  775.           // Get forum details 
  776.           $count = array(); 
  777.           $show_sep = $total_subs > $i ? $r['separator'] : ''; 
  778.           $permalink = bbp_get_forum_permalink( $sub_forum->ID ); 
  779.           $title = bbp_get_forum_title( $sub_forum->ID ); 
  780.  
  781.           // Show topic count 
  782.           if ( !empty( $r['show_topic_count'] ) && !bbp_is_forum_category( $sub_forum->ID ) ) { 
  783.               $count['topic'] = bbp_get_forum_topic_count( $sub_forum->ID ); 
  784.           } 
  785.  
  786.           // Show reply count 
  787.           if ( !empty( $r['show_reply_count'] ) && !bbp_is_forum_category( $sub_forum->ID ) ) { 
  788.               $count['reply'] = bbp_get_forum_reply_count( $sub_forum->ID ); 
  789.           } 
  790.  
  791.           // Counts to show 
  792.           if ( !empty( $count ) ) { 
  793.               $counts = $r['count_before'] . implode( $r['count_sep'], $count ) . $r['count_after']; 
  794.           } 
  795.  
  796.           // Build this sub forums link 
  797.           $output .= $r['link_before'] . '<a href="' . esc_url( $permalink ) . '" class="bbp-forum-link">' . $title . $counts . '</a>' . $show_sep . $r['link_after']; 
  798.       } 
  799.  
  800.       // Output the list 
  801.       echo apply_filters( 'bbp_list_forums', $r['before'] . $output . $r['after'], $r ); 
  802.   } 
  803. }
