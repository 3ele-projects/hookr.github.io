  330.  
  331.         // Define variables

  333.

  338.         $post_data['referrer'] = $_SERVER['HTTP_REFERER']; 
  339.         $post_data['user_agent'] = $_SERVER['HTTP_USER_AGENT']; 
  340.  
  341.         // Akismet Test Mode 
  342.         if ( akismet_test_mode() ) 
  343.             $post_data['is_test'] = 'true'; 
  344.  
  345.         // Loop through _POST args and rekey strings 
  346.         foreach ( $_POST as $key => $value ) 
  347.             if ( is_string( $value ) ) 
  348.                 $post_data['POST_' . $key] = $value; 
  349.  
  350.         // Keys to ignore 
  351.         $ignore = array( 'HTTP_COOKIE', 'HTTP_COOKIE2', 'PHP_AUTH_PW' );

  354.         foreach ( $_SERVER as $key => $value ) { 
  355.  
  356.             // Key should not be ignored 
  357.             if ( !in_array( $key, $ignore ) && is_string( $value ) ) { 
  358.                 $post_data[$key] = $value; 
  359.  
  360.             // Key should be ignored 
  361.             } else { 
  362.                 $post_data[$key] = ''; 
  363.             }

  365.  
  366.         // Ready...

  571.     /** 
  572.      * Submit data to Akismet service with unique bbPress User Agent 
  573.      * 
  574.      * This code is directly taken from the akismet_http_post() function and 
  575.      * documented to bbPress 2.0 standard. 
  576.      * 
  577.      * @since bbPress (r3466) 
  578.      * 
  579.      * @param string $request The request we are sending 
  580.      * @param string $host The host to send our request to 
  581.      * @param string $path The path from the host 
  582.      * @param string $port The port to use 
  583.      * @param string $ip Optional Override $host with an IP address 
  584.      * @uses bbp_get_version() To get the current bbPress version 
  585.      * @return mixed WP_Error on error, array on success, empty on failure 
  586.      */ 
  587.     private function http_post( $request, $host, $path, $port = 80, $ip = '' ) { 
  588.  
  589.         // Preload required variables 
  590.         $bbp_version = bbp_get_version(); 
  591.         $content_length = strlen( $request ); 
  592.         $http_host = $host; 
  593.         $blog_charset = get_option( 'blog_charset' ); 
  594.         $response = ''; 
  595.         $errno = null; 
  596.         $errstr = null; 
  597.  
  598.         // Untque User Agent 
  599.         $akismet_ua = "bbPress/{$bbp_version} | "; 
  600.         $akismet_ua    .= 'Akismet/' . constant( 'AKISMET_VERSION' ); 
  601.  
  602.         // Use specific IP (if provided) 
  603.         if ( !empty( $ip ) && long2ip( ip2long( $ip ) ) ) 
  604.             $http_host = $ip; 
  605.  
  606.         // WP HTTP class is available 
  607.         if ( function_exists( 'wp_remote_post' ) ) { 
  608.  
  609.             // Setup the arguments 
  610.             $http_args = array( 
  611.                 'body' => $request,  
  612.                 'headers' => array( 
  613.                     'Content-Type' => 'application/x-www-form-urlencoded; charset=' . $blog_charset,  
  614.                     'Host' => $host,  
  615.                     'User-Agent' => $akismet_ua 
  616.  ),  
  617.                 'httpversion' => '1.0',  
  618.                 'timeout' => 15 
  619.  ); 
  620.  
  621.             // Where we are sending our request 
  622.             $akismet_url = 'http://' . $http_host . $path; 
  623.  
  624.             // Send the request 
  625.             $response = wp_remote_post( $akismet_url, $http_args ); 
  626.  
  627.             // Bail if the response is an error 
  628.             if ( is_wp_error( $response ) ) 
  629.                 return ''; 
  630.  
  631.             // No errors so return response 
  632.             return array( $response['headers'], $response['body'] ); 
  633.  
  634.         // WP HTTP class is not available (Why not?) 
  635.         } else { 
  636.  
  637.             // Header info to use with our socket 
  638.             $http_request = "POST {$path} HTTP/1.0\r\n"; 
  639.             $http_request .= "Host: {$host}\r\n"; 
  640.             $http_request .= "Content-Type: application/x-www-form-urlencoded; charset={$blog_charset}\r\n"; 
  641.             $http_request .= "Content-Length: {$content_length}\r\n"; 
  642.             $http_request .= "User-Agent: {$akismet_ua}\r\n"; 
  643.             $http_request .= "\r\n"; 
  644.             $http_request .= $request; 
  645.  
  646.             // Open a socket connection 
  647.             if ( false !== ( $fs = @fsockopen( $http_host, $port, $errno, $errstr, 10 ) ) ) { 
  648.  
  649.                 // Write our request to the pointer 
  650.                 fwrite( $fs, $http_request ); 
  651.  
  652.                 // Loop through pointer and compile a response 
  653.                 while ( !feof( $fs ) ) { 
  654.                     // One TCP-IP packet at a time 
  655.                     $response .= fgets( $fs, 1160 ); 
  656.                 } 
  657.  
  658.                 // Close our socket 
  659.                 fclose( $fs ); 
  660.  
  661.                 // Explode the response into usable data 
  662.                 $response = explode( "\r\n\r\n", $response, 2 ); 
  663.             } 
  664.  
  665.             // Return the response ('' if error/empty) 
  666.             return $response; 
  667.         } 
  668.     }
