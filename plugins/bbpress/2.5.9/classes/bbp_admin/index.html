  665.     /** 
  666.      * Ajax action for facilitating the forum auto-suggest 
  667.      * 
  668.      * @since bbPress (r4261) 
  669.      * 
  670.      * @uses get_posts() 
  671.      * @uses bbp_get_topic_post_type() 
  672.      * @uses bbp_get_topic_id() 
  673.      * @uses bbp_get_topic_title() 
  674.      */ 
  675.     public function suggest_topic() { 
  676.         global $wpdb; 
  677.  
  678.         // Bail early if no request 
  679.         if ( empty( $_REQUEST['q'] ) ) { 
  680.             wp_die( '0' ); 
  681.         } 
  682.  
  683.         // Bail if user cannot moderate - only moderators can change hierarchy 
  684.         if ( ! current_user_can( 'moderate' ) ) { 
  685.             wp_die( '0' ); 
  686.         } 
  687.  
  688.         // Check the ajax nonce 
  689.         check_ajax_referer( 'bbp_suggest_topic_nonce' ); 
  690.  
  691.         // Try to get some topics 
  692.         $topics = get_posts( array( 
  693.             's' => $wpdb->esc_like( $_REQUEST['q'] ),  
  694.             'post_type' => bbp_get_topic_post_type() 
  695.  ) ); 
  696.  
  697.         // If we found some topics, loop through and display them 
  698.         if ( ! empty( $topics ) ) { 
  699.             foreach ( (array) $topics as $post ) { 
  700.                 printf( esc_html__( '%s - %s', 'bbpress' ), bbp_get_topic_id( $post->ID ), bbp_get_topic_title( $post->ID ) . "\n" ); 
  701.             } 
  702.         } 
  703.         die(); 
  704.     }

  707.      * Ajax action for facilitating the topic and reply author auto-suggest 
  708.      * 
  709.      * @since bbPress (r5014) 
  710.      */ 
  711.     public function suggest_user() { 
  712.         global $wpdb; 
  713.  
  714.         // Bail early if no request 
  715.         if ( empty( $_REQUEST['q'] ) ) { 
  716.             wp_die( '0' ); 
  717.         } 
  718.  
  719.         // Bail if user cannot moderate - only moderators can change authorship 
  720.         if ( ! current_user_can( 'moderate' ) ) { 
  721.             wp_die( '0' ); 
  722.         } 
  723.  
  724.         // Check the ajax nonce 
  725.         check_ajax_referer( 'bbp_suggest_user_nonce' ); 
  726.  
  727.         // Try to get some users 
  728.         $users_query = new WP_User_Query( array( 
  729.             'search' => '*' . $wpdb->esc_like( $_REQUEST['q'] ) . '*',  
  730.             'fields' => array( 'ID', 'user_nicename' ),  
  731.             'search_columns' => array( 'ID', 'user_nicename', 'user_email' ),  
  732.             'orderby' => 'ID' 
  733.  ) ); 
  734.  
  735.         // If we found some users, loop through and display them 
  736.         if ( ! empty( $users_query->results ) ) { 
  737.             foreach ( (array) $users_query->results as $user ) { 
  738.                 printf( esc_html__( '%s - %s', 'bbpress' ), bbp_get_user_id( $user->ID ), bbp_get_user_nicename( $user->ID, array( 'force' => $user->user_nicename ) ) . "\n" ); 
  739.             } 
  740.         } 
  741.         die(); 
  742.     }
