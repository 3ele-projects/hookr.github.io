  95. /**

  103. function bbp_code_trick( $content = '' ) { 
  104.   $content = str_replace( array( "\r\n", "\r" ), "\n", $content ); 
  105.   $content = preg_replace_callback( "|(`)(.*?)`|", 'bbp_encode_callback', $content ); 
  106.   $content = preg_replace_callback( "!(^|\n)`(.*?)`!s", 'bbp_encode_callback', $content ); 
  107.  
  108.   return $content;

  111. /** 
  112.  * When editing a topic or reply, reverse the code trick so the textarea 
  113.  * contains the correct editable content. 
  114.  * 
  115.  * @since bbPress (r4641) 
  116.  * 
  117.  * @param string $content Topic and reply content 
  118.  * @return string Partially encodedd content 
  119.  */ 
  120. function bbp_code_trick_reverse( $content = '' ) { 
  121.  
  122.   // Setup variables 
  123.   $openers = array( '<p>', '<br />' ); 
  124.   $content = preg_replace_callback( "!(<pre><code>|<code>)(.*?)(</code></pre>|</code>)!s", 'bbp_decode_callback', $content ); 
  125.  
  126.   // Do the do 
  127.   $content = str_replace( $openers, '', $content ); 
  128.   $content = str_replace( '</p>', "\n", $content ); 
  129.   $content = str_replace( '<coded_br />', '<br />', $content ); 
  130.   $content = str_replace( '<coded_p>', '<p>', $content ); 
  131.   $content = str_replace( '</coded_p>', '</p>', $content ); 
  132.  
  133.   return $content; 
  134. }

  136. /** 
  137.  * Filter the content and encode any bad HTML tags 
  138.  * 
  139.  * @since bbPress (r4641) 
  140.  * 
  141.  * @param string $content Topic and reply content 
  142.  * @return string Partially encodedd content 
  143.  */ 
  144. function bbp_encode_bad( $content = '' ) { 
  145.  
  146.   // Setup variables 
  147.   $content = _wp_specialchars( $content, ENT_NOQUOTES ); 
  148.   $content = preg_split( '@(`[^`]*`)@m', $content, -1, PREG_SPLIT_NO_EMPTY + PREG_SPLIT_DELIM_CAPTURE ); 
  149.   $allowed = bbp_kses_allowed_tags(); 
  150.   $empty = array( 
  151.       'br' => true,  
  152.       'hr' => true,  
  153.       'img' => true,  
  154.       'input' => true,  
  155.       'param' => true,  
  156.       'area' => true,  
  157.       'col' => true,  
  158.       'embed' => true 
  159.  ); 
  160.  
  161.   // Loop through allowed tags and compare for empty and normal tags 
  162.   foreach ( $allowed as $tag => $args ) { 
  163.       $preg = $args ? "{$tag}(?:\s.*?)?" : $tag; 
  164.  
  165.       // Which walker to use based on the tag and arguments 
  166.       if ( isset( $empty[$tag] ) ) { 
  167.           array_walk( $content, 'bbp_encode_empty_callback', $preg ); 
  168.       } else { 
  169.           array_walk( $content, 'bbp_encode_normal_callback', $preg ); 
  170.       } 
  171.   } 
  172.  
  173.   // Return the joined content array 
  174.   return implode( '', $content ); 
  175. }

  177. /** Code Callbacks ************************************************************/

  179. /** 
  180.  * Callback to encode the tags in topic or reply content 
  181.  * 
  182.  * @since bbPress (r4641) 
  183.  * 
  184.  * @param array $matches 
  185.  * @return string 
  186.  */ 
  187. function bbp_encode_callback( $matches = array() ) { 
  188.  
  189.   // Trim inline code, not pre blocks (to prevent removing indentation) 
  190.   if ( "`" === $matches[1] ) { 
  191.       $content = trim( $matches[2] ); 
  192.   } else { 
  193.       $content = $matches[2]; 
  194.   } 
  195.  
  196.   // Do some replacing 
  197.   $content = htmlspecialchars( $content, ENT_QUOTES ); 
  198.   $content = str_replace( array( "\r\n", "\r" ), "\n", $content ); 
  199.   $content = preg_replace( "|\n\n\n+|", "\n\n", $content ); 
  200.   $content = str_replace( '&amp;', '&', $content ); 
  201.   $content = str_replace( '&lt;', '<', $content ); 
  202.   $content = str_replace( '&gt;', '>', $content ); 
  203.  
  204.   // Wrap in code tags 
  205.   $content = '<code>' . $content . '</code>'; 
  206.  
  207.   // Wrap blocks in pre tags 
  208.   if ( "`" !== $matches[1] ) { 
  209.       $content = "\n<pre>" . $content . "</pre>\n"; 
  210.   } 
  211.  
  212.   return $content; 
  213. }

  215. /** 
  216.  * Callback to decode the tags in topic or reply content 
  217.  * 
  218.  * @since bbPress (r4641) 
  219.  * 
  220.  * @param array $matches 
  221.  * @todo Experiment with _wp_specialchars() 
  222.  * @return string 
  223.  */ 
  224. function bbp_decode_callback( $matches = array() ) { 
  225.  
  226.   // Setup variables 
  227.   $trans_table = array_flip( get_html_translation_table( HTML_ENTITIES ) ); 
  228.   $amps = array( '&', '&', '&' ); 
  229.   $single = array( ''', ''' ); 
  230.   $content = $matches[2]; 
  231.   $content = strtr( $content, $trans_table ); 
  232.  
  233.   // Do the do 
  234.   $content = str_replace( '<br />', '<coded_br />', $content ); 
  235.   $content = str_replace( '<p>', '<coded_p>', $content ); 
  236.   $content = str_replace( '</p>', '</coded_p>', $content ); 
  237.   $content = str_replace( $amps, '&', $content ); 
  238.   $content = str_replace( $single, "'", $content ); 
  239.  
  240.   // Return content wrapped in code tags 
  241.   return '`' . $content . '`'; 
  242. }

  304. /** 
  305.  * Convert plaintext URI to HTML links. 
  306.  * 
  307.  * Converts URI, www and ftp, and email addresses. Finishes by fixing links 
  308.  * within links. 
  309.  * 
  310.  * This custom version of WordPress's make_clickable() skips links inside of 
  311.  * pre and code tags. 
  312.  * 
  313.  * @since bbPress (r4941) 
  314.  * 
  315.  * @param string $text Content to convert URIs. 
  316.  * @return string Content with converted URIs. 
  317.  */ 
  318. function bbp_make_clickable( $text ) { 
  319.   $r = ''; 
  320.   $in_code = false; 
  321.   $textarr = preg_split( '/(<[^<>]+>)/', $text, -1, PREG_SPLIT_DELIM_CAPTURE ); // split out HTML tags 
  322.  
  323.   foreach ( $textarr as $piece ) { 
  324.  
  325.       switch ( $piece ) { 
  326.           case '<code>' : 
  327.           case '<pre>'  : 
  328.               $in_code = true; 
  329.               break; 
  330.           case '</code>' : 
  331.           case '</pre>'  : 
  332.               $in_code = false; 
  333.               break; 
  334.       } 
  335.  
  336.       if ( $in_code || empty( $piece ) || ( $piece[0] === '<' && ! preg_match('|^<\s*[\w]{1, 20}+://|', $piece) ) ) { 
  337.           $r .= $piece; 
  338.           continue; 
  339.       } 
  340.  
  341.       // Long strings might contain expensive edge cases ... 
  342.       if ( 10000 < strlen( $piece ) ) { 
  343.           // ... break it up 
  344.           foreach ( _split_str_by_whitespace( $piece, 2100 ) as $chunk ) { // 2100: Extra room for scheme and leading and trailing paretheses 
  345.               if ( 2101 < strlen( $chunk ) ) { 
  346.                   $r .= $chunk; // Too big, no whitespace: bail. 
  347.               } else { 
  348.                   $r .= make_clickable( $chunk ); 
  349.               } 
  350.           } 
  351.       } else { 
  352.           $ret = " $piece "; // Pad with whitespace to simplify the regexes 
  353.  
  354.           $url_clickable = '~ 
  355.               ([\\s(<., ;:!?])                                        # 1: Leading whitespace, or punctuation 
  356.               (                                                      # 2: URL 
  357.                   [\\w]{1, 20}+://                                # Scheme and hier-part prefix 
  358.                   (?=\S{1, 2000}\s)                               # Limit to URLs less than about 2000 characters long 
  359.                   [\\w\\x80-\\xff#%\\~/@\\[\\]*(+=&$-]*+         # Non-punctuation URL character 
  360.                   (?:                                            # Unroll the Loop: Only allow puctuation URL character if followed by a non-punctuation URL character 
  361.                       [\'., ;:!?)]                            # Punctuation URL character 
  362.                       [\\w\\x80-\\xff#%\\~/@\\[\\]*(+=&$-]++ # Non-punctuation URL character 
  363.  )* 
  364.  ) 
  365.               (\)?)                                                  # 3: Trailing closing parenthesis (for parethesis balancing post processing) 
  366.           ~xS'; // The regex is a non-anchored pattern and does not have a single fixed starting character. 
  367.                 // Tell PCRE to spend more time optimizing since, when used on a page load, it will probably be used several times. 
  368.  
  369.           $ret = preg_replace_callback( $url_clickable, '_make_url_clickable_cb', $ret ); 
  370.  
  371.           $ret = preg_replace_callback( '#([\s>])((www|ftp)\.[\w\\x80-\\xff\#$%&~/.\-;:=, ?@\[\]+]+)#is', '_make_web_ftp_clickable_cb', $ret ); 
  372.           $ret = preg_replace_callback( '#([\s>])([.0-9a-z_+-]+)@(([0-9a-z-]+\.)+[0-9a-z]{2, })#i', '_make_email_clickable_cb', $ret ); 
  373.  
  374.           $ret = substr( $ret, 1, -1 ); // Remove our whitespace padding. 
  375.           $r .= $ret; 
  376.       } 
  377.   } 
  378.  
  379.   // Cleanup of accidental links within links 
  380.   $r = preg_replace( '#(<a( [^>]+?>|>))<a [^>]+?>([^>]+?)</a></a>#i', "$1$3</a>", $r ); 
  381.   return $r; 
  382. }
