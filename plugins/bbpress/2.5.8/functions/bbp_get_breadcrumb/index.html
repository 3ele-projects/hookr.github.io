bbp_get_breadcrumb

Return a breadcrumb ( forum -> topic -> reply ).

Description

(string) bbp_get_breadcrumb( (array) $args = array() ); 

Returns (string)

Breadcrumbs

Parameters (1)

0. $args — Optional. (array) => array()
Separator. Defaults to ←

Usage

  1. if ( !function_exists( 'bbp_get_breadcrumb' ) ) { 
  2.     require_once ABSPATH . PLUGINDIR . 'bbpress/includes/common/template.php'; 
  3. }

  21.         if ( !empty( $front_id ) ) { 
  22.             $pre_front_text = get_the_title( $front_id );

  34.         $page = bbp_get_page_by_path( bbp_get_root_slug() ); 
  35.         if ( !empty( $page ) ) { 
  36.             $root_id = $page->ID; 
  37.         } 
  38.         $pre_root_text = bbp_get_forum_archive_title(); 
  39.     } 
  40.  
  41.     /** Includes **********************************************************/ 
  42.

  45.         $pre_include_root = false; 
  46.     } 
  47.

  66.         $pre_current_text = bbp_get_forum_archive_title(); 
  67.

  69.     } elseif ( bbp_is_topic_archive() ) { 
  70.         $pre_current_text = bbp_get_topic_archive_title(); 
  71.  
  72.     // View 
  73.     } elseif ( bbp_is_single_view() ) { 
  74.         $pre_current_text = bbp_get_view_title(); 
  75.  
  76.     // Single Forum 
  77.     } elseif ( bbp_is_single_forum() ) { 
  78.         $pre_current_text = bbp_get_forum_title(); 
  79.  
  80.     // Single Topic 
  81.     } elseif ( bbp_is_single_topic() ) { 
  82.         $pre_current_text = bbp_get_topic_title(); 
  83.  
  84.     // Single Topic 
  85.     } elseif ( bbp_is_single_reply() ) { 
  86.         $pre_current_text = bbp_get_reply_title(); 
  87.  
  88.     // Topic Tag (or theme compat topic tag)

  104.         $pre_current_text = __( 'Edit', bbpress ); 
  105.  
  106.     // Single 
  107.     } else { 
  108.         $pre_current_text = get_the_title(); 
  109.     }

  111.     /** Parse Args ********************************************************/ 
  112.  
  113.     // Parse args 
  114.     $r = bbp_parse_args( $args, array(

  117.         'before' => '<div class="bbp-breadcrumb"><p>',  
  118.         'after' => '</p></div>',  
  119.  
  120.         // Separator 
  121.         'sep' => is_rtl() ? __( '‹', bbpress ) : __( '›', bbpress ),
