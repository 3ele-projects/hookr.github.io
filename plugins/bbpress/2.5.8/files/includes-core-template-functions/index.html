  46. /** 
  47.  * Retrieve the name of the highest priority template file that exists. 
  48.  * 
  49.  * Searches in the child theme before parent theme so that themes which 
  50.  * inherit from a parent theme can just overload one file. If the template is 
  51.  * not found in either of those, it looks in the theme-compat folder last. 
  52.  * 
  53.  * @since bbPress (r3618) 
  54.  * 
  55.  * @param string|array $template_names Template file(s) to search for, in order. 
  56.  * @param bool $load If true the template file will be loaded if it is found. 
  57.  * @param bool $require_once Whether to require_once or require. Default true. 
  58.  *                            Has no effect if $load is false. 
  59.  * @return string The template filename if one is located. 
  60.  */ 
  61. function bbp_locate_template( $template_names, $load = false, $require_once = true ) { 
  62.  
  63.   // No file found yet 
  64.   $located = false; 
  65.   $template_locations = bbp_get_template_stack(); 
  66.  
  67.   // Try to find a template file 
  68.   foreach ( (array) $template_names as $template_name ) { 
  69.  
  70.       // Continue if template is empty 
  71.       if ( empty( $template_name ) ) { 
  72.           continue; 
  73.       } 
  74.  
  75.       // Trim off any slashes from the template name 
  76.       $template_name = ltrim( $template_name, '/' ); 
  77.  
  78.       // Loop through template stack 
  79.       foreach ( (array) $template_locations as $template_location ) { 
  80.  
  81.           // Continue if $template_location is empty 
  82.           if ( empty( $template_location ) ) { 
  83.               continue; 
  84.           } 
  85.  
  86.           // Check child theme first 
  87.           if ( file_exists( trailingslashit( $template_location ) . $template_name ) ) { 
  88.               $located = trailingslashit( $template_location ) . $template_name; 
  89.               break 2; 
  90.           } 
  91.       } 
  92.   } 
  93.  
  94.   /** 
  95.    * This action exists only to follow the standard bbPress coding convention,  
  96.    * and should not be used to short-circuit any part of the template locator. 
  97.    * 
  98.    * If you want to override a specific template part, please either filter 
  99.    * 'bbp_get_template_part' or add a new location to the template stack. 
  100.    */ 
  101.   do_action( 'bbp_locate_template', $located, $template_name, $template_names, $template_locations, $load, $require_once ); 
  102.  
  103.   // Maybe load the template if one was located 
  104.   if ( ( true === $load ) && !empty( $located ) ) { 
  105.       load_template( $located, $require_once ); 
  106.   } 
  107.  
  108.   return $located; 
  109. }

  130. function bbp_enqueue_style( $handle = '', $file = '', $dependencies = array(), $version = false, $media = 'all' ) { 
  131.  
  132.   // No file found yet 
  133.   $located = false; 
  134.  
  135.   // Trim off any slashes from the template name 
  136.   $file = ltrim( $file, '/' ); 
  137.  
  138.   // Make sure there is always a version 
  139.   if ( empty( $version ) ) { 
  140.       $version = bbp_get_version(); 
  141.   } 
  142.  
  143.   // Loop through template stack 
  144.   foreach ( (array) bbp_get_template_stack() as $template_location ) { 
  145.  
  146.       // Continue if $template_location is empty 
  147.       if ( empty( $template_location ) ) { 
  148.           continue; 
  149.       } 
  150.  
  151.       // Check child theme first 
  152.       if ( file_exists( trailingslashit( $template_location ) . $file ) ) { 
  153.           $located = trailingslashit( $template_location ) . $file; 
  154.           break; 
  155.       } 
  156.   } 
  157.  
  158.   // Enqueue if located 
  159.   if ( !empty( $located ) ) { 
  160.  
  161.       $content_dir = constant( 'WP_CONTENT_DIR' ); 
  162.  
  163.       // IIS (Windows) here

  167.           $content_dir = str_replace( '\\', '/', $content_dir );

  171.       $located = str_replace( $content_dir, content_url(), $located );

  174.       wp_enqueue_style( $handle, $located, $dependencies, $version, $media );

  290. /** 
  291.  * Call the functions added to the 'bbp_template_stack' filter hook, and return 
  292.  * an array of the template locations. 
  293.  * 
  294.  * @see bbp_register_template_stack() 
  295.  * 
  296.  * @since bbPress (r4323) 
  297.  * 
  298.  * @global array $wp_filter Stores all of the filters 
  299.  * @global array $merged_filters Merges the filter hooks using this function. 
  300.  * @global array $wp_current_filter stores the list of current filters with the current one last 
  301.  * 
  302.  * @return array The filtered value after all hooked functions are applied to it. 
  303.  */ 
  304. function bbp_get_template_stack() { 
  305.   global $wp_filter, $merged_filters, $wp_current_filter; 
  306.  
  307.   // Setup some default variables 
  308.   $tag = 'bbp_template_stack'; 
  309.   $args = $stack = array(); 
  310.  
  311.   // Add 'bbp_template_stack' to the current filter array 
  312.   $wp_current_filter[] = $tag; 
  313.  
  314.   // Sort 
  315.   if ( ! isset( $merged_filters[ $tag ] ) ) { 
  316.       ksort( $wp_filter[$tag] ); 
  317.       $merged_filters[ $tag ] = true; 
  318.   } 
  319.  
  320.   // Ensure we're always at the beginning of the filter array 
  321.   reset( $wp_filter[ $tag ] ); 
  322.  
  323.   // Loop through 'bbp_template_stack' filters, and call callback functions 
  324.   do { 
  325.       foreach ( (array) current( $wp_filter[$tag] ) as $the_ ) { 
  326.           if ( ! is_null( $the_['function'] ) ) { 
  327.               $args[1] = $stack; 
  328.               $stack[] = call_user_func_array( $the_['function'], array_slice( $args, 1, (int) $the_['accepted_args'] ) ); 
  329.           } 
  330.       } 
  331.   } while ( next( $wp_filter[$tag] ) !== false ); 
  332.  
  333.   // Remove 'bbp_template_stack' from the current filter array 
  334.   array_pop( $wp_current_filter ); 
  335.  
  336.   // Remove empties and duplicates 
  337.   $stack = array_unique( array_filter( $stack ) ); 
  338.  
  339.   return (array) apply_filters( 'bbp_get_template_stack', $stack ) ; 
  340. }

  472. function bbp_parse_query( $posts_query ) { 
  473.  
  474.   // Bail if $posts_query is not the main loop 
  475.   if ( ! $posts_query->is_main_query() ) 
  476.       return; 
  477.  
  478.   // Bail if filters are suppressed on this query 
  479.   if ( true === $posts_query->get( 'suppress_filters' ) ) 
  480.       return; 
  481.  
  482.   // Bail if in admin 
  483.   if ( is_admin() ) 
  484.       return; 
  485.  
  486.   // Get query variables 
  487.   $bbp_view = $posts_query->get( bbp_get_view_rewrite_id() ); 
  488.   $bbp_user = $posts_query->get( bbp_get_user_rewrite_id() ); 
  489.   $is_edit = $posts_query->get( bbp_get_edit_rewrite_id() ); 
  490.  
  491.   // It is a user page - We'll also check if it is user edit 
  492.   if ( !empty( $bbp_user ) ) { 
  493.  
  494.       /** Find User *********************************************************/ 
  495.  
  496.       // Setup the default user variable 
  497.       $the_user = false; 
  498.  
  499.       // If using pretty permalinks, always use slug 
  500.       if ( get_option( 'permalink_structure' ) ) { 
  501.           $the_user = get_user_by( 'slug', $bbp_user ); 
  502.  
  503.       // If not using pretty permalinks, always use numeric ID 
  504.       } elseif ( is_numeric( $bbp_user ) ) { 
  505.           $the_user = get_user_by( 'id', $bbp_user ); 
  506.       } 
  507.  
  508.       // 404 and bail if user does not have a profile 
  509.       if ( empty( $the_user->ID ) || ! bbp_user_has_profile( $the_user->ID ) ) { 
  510.           $posts_query->set_404(); 
  511.           return; 
  512.       } 
  513.  
  514.       /** User Exists *******************************************************/ 
  515.  
  516.       $is_favs = $posts_query->get( bbp_get_user_favorites_rewrite_id() ); 
  517.       $is_subs = $posts_query->get( bbp_get_user_subscriptions_rewrite_id() ); 
  518.       $is_topics = $posts_query->get( bbp_get_user_topics_rewrite_id() ); 
  519.       $is_replies = $posts_query->get( bbp_get_user_replies_rewrite_id() ); 
  520.  
  521.       // View or edit? 
  522.       if ( !empty( $is_edit ) ) { 
  523.  
  524.           // We are editing a profile 
  525.           $posts_query->bbp_is_single_user_edit = true; 
  526.  
  527.           // Load the core WordPress contact methods 
  528.           if ( !function_exists( '_wp_get_user_contactmethods' ) ) { 
  529.               include_once( ABSPATH . 'wp-includes/registration.php' ); 
  530.           } 
  531.  
  532.           // Load the edit_user functions 
  533.           if ( !function_exists( 'edit_user' ) ) { 
  534.               require_once( ABSPATH . 'wp-admin/includes/user.php' ); 
  535.           } 
  536.  
  537.           // Load the grant/revoke super admin functions 
  538.           if ( is_multisite() && !function_exists( 'revoke_super_admin' ) ) { 
  539.               require_once( ABSPATH . 'wp-admin/includes/ms.php' ); 
  540.           } 
  541.  
  542.           // Editing a user 
  543.           $posts_query->bbp_is_edit = true; 
  544.  
  545.       // User favorites 
  546.       } elseif ( ! empty( $is_favs ) ) { 
  547.           $posts_query->bbp_is_single_user_favs = true; 
  548.  
  549.       // User subscriptions 
  550.       } elseif ( ! empty( $is_subs ) ) { 
  551.           $posts_query->bbp_is_single_user_subs = true; 
  552.  
  553.       // User topics 
  554.       } elseif ( ! empty( $is_topics ) ) { 
  555.           $posts_query->bbp_is_single_user_topics = true; 
  556.  
  557.       // User topics 
  558.       } elseif ( ! empty( $is_replies ) ) { 
  559.           $posts_query->bbp_is_single_user_replies = true; 
  560.  
  561.       // User profile 
  562.       } else { 
  563.           $posts_query->bbp_is_single_user_profile = true; 
  564.       } 
  565.  
  566.       // Looking at a single user 
  567.       $posts_query->bbp_is_single_user = true; 
  568.  
  569.       // Make sure 404 is not set 
  570.       $posts_query->is_404 = false; 
  571.  
  572.       // Correct is_home variable 
  573.       $posts_query->is_home = false; 
  574.  
  575.       // User is looking at their own profile 
  576.       if ( get_current_user_id() === $the_user->ID ) { 
  577.           $posts_query->bbp_is_single_user_home = true; 
  578.       } 
  579.  
  580.       // Set bbp_user_id for future reference 
  581.       $posts_query->set( 'bbp_user_id', $the_user->ID ); 
  582.  
  583.       // Set author_name as current user's nicename to get correct posts 
  584.       $posts_query->set( 'author_name', $the_user->user_nicename ); 
  585.  
  586.       // Set the displayed user global to this user 
  587.       bbpress()->displayed_user = $the_user; 
  588.  
  589.   // View Page 
  590.   } elseif ( !empty( $bbp_view ) ) { 
  591.  
  592.       // Check if the view exists by checking if there are query args are set 
  593.       $view_args = bbp_get_view_query_args( $bbp_view ); 
  594.  
  595.       // Bail if view args is false (view isn't registered) 
  596.       if ( false === $view_args ) { 
  597.           $posts_query->set_404(); 
  598.           return; 
  599.       } 
  600.  
  601.       // Correct is_home variable 
  602.       $posts_query->is_home = false; 
  603.  
  604.       // We are in a custom topic view 
  605.       $posts_query->bbp_is_view = true; 
  606.  
  607.   // Search Page 
  608.   } elseif ( isset( $posts_query->query_vars[ bbp_get_search_rewrite_id() ] ) ) { 
  609.  
  610.       // Check if there are search query args set 
  611.       $search_terms = bbp_get_search_terms(); 
  612.       if ( !empty( $search_terms ) ) 
  613.           $posts_query->bbp_search_terms = $search_terms; 
  614.  
  615.       // Correct is_home variable 
  616.       $posts_query->is_home = false; 
  617.  
  618.       // We are in a search query 
  619.       $posts_query->bbp_is_search = true; 
  620.  
  621.   // Forum/Topic/Reply Edit Page 
  622.   } elseif ( !empty( $is_edit ) ) { 
  623.  
  624.       // Get the post type from the main query loop 
  625.       $post_type = $posts_query->get( 'post_type' ); 
  626.  
  627.       // Check which post_type we are editing, if any 
  628.       if ( !empty( $post_type ) ) { 
  629.           switch( $post_type ) { 
  630.  
  631.               // We are editing a forum 
  632.               case bbp_get_forum_post_type() : 
  633.                   $posts_query->bbp_is_forum_edit = true; 
  634.                   $posts_query->bbp_is_edit = true; 
  635.                   break; 
  636.  
  637.               // We are editing a topic 
  638.               case bbp_get_topic_post_type() : 
  639.                   $posts_query->bbp_is_topic_edit = true; 
  640.                   $posts_query->bbp_is_edit = true; 
  641.                   break; 
  642.  
  643.               // We are editing a reply 
  644.               case bbp_get_reply_post_type() : 
  645.                   $posts_query->bbp_is_reply_edit = true; 
  646.                   $posts_query->bbp_is_edit = true; 
  647.                   break; 
  648.           } 
  649.  
  650.       // We are editing a topic tag 
  651.       } elseif ( bbp_is_topic_tag() ) { 
  652.           $posts_query->bbp_is_topic_tag_edit = true; 
  653.           $posts_query->bbp_is_edit = true; 
  654.       } 
  655.  
  656.       // We save post revisions on our own 
  657.       remove_action( 'pre_post_update', 'wp_save_post_revision' ); 
  658.  
  659.   // Topic tag page 
  660.   } elseif ( bbp_is_topic_tag() ) { 
  661.       $posts_query->set( 'bbp_topic_tag', get_query_var( 'term' ) ); 
  662.       $posts_query->set( 'post_type', bbp_get_topic_post_type() ); 
  663.       $posts_query->set( 'posts_per_page', bbp_get_topics_per_page() ); 
  664.  
  665.   // Do topics on forums root 
  666.   } elseif ( is_post_type_archive( array( bbp_get_forum_post_type(), bbp_get_topic_post_type() ) ) && ( 'topics' === bbp_show_on_root() ) ) { 
  667.       $posts_query->bbp_show_topics_on_root = true; 
  668.   } 
  669. }
