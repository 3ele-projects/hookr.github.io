  287.  * @since bbPress (r3653) 
  288.  */ 
  289. function bbp_get_group_forum_ids( $group_id = 0 ) { 
  290.  
  291.   // Assume no forums 
  292.   $forum_ids = array(); 
  293.  
  294.   // Use current group if none is set 
  295.   if ( empty( $group_id ) ) 
  296.       $group_id = bp_get_current_group_id(); 
  297.  
  298.   // Get the forums 
  299.   if ( !empty( $group_id ) ) 
  300.       $forum_ids = groups_get_groupmeta( $group_id, 'forum_id' ); 
  301.  
  302.   // Make sure result is an array 
  303.   if ( !is_array( $forum_ids ) ) 
  304.       $forum_ids = (array) $forum_ids; 
  305.  
  306.   // Trim out any empty array items 
  307.   $forum_ids = array_filter( $forum_ids ); 
  308.  
  309.   return (array) apply_filters( 'bbp_get_group_forum_ids', $forum_ids, $group_id ); 
  310. }
