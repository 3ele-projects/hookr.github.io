  319.   private function get_notes( $reply_id = 0 ) {

  330.   /** 
  331.    * Store a new note 
  332.    * 
  333.    * @since 1.0 
  334.    * @access public 
  335.    * @return int The ID of the new note 
  336.    */ 
  337.   public function save_note() { 
  338.  
  339.       if ( ! current_user_can( 'moderate' ) ) 
  340.           return; 
  341.  
  342.       if ( empty( $_POST['bbp-add-note'] ) ) 
  343.           return; 
  344.  
  345.       if ( empty( $_POST['bbp-reply-id'] ) ) 
  346.           return; 
  347.  
  348.       $reply_id = absint( $_POST['bbp-reply-id'] ); 
  349.  
  350.       $reply_note = wp_kses( $_POST['bbp-reply-note'], array() ); 
  351.  
  352.       $user = get_userdata( get_current_user_id() ); 
  353.  
  354.       $note_id = wp_insert_comment( wp_filter_comment( array( 
  355.           'comment_post_ID' => $reply_id,  
  356.           'comment_author' => $user->user_login,  
  357.           'comment_author_url' => $user->user_url,  
  358.           'comment_author_IP' => bbp_current_author_ip(),  
  359.           'comment_author_email' => $user->user_email,  
  360.           'comment_content' => $reply_note,  
  361.           'user_id' => $user->ID,  
  362.           'comment_agent' => bbp_current_author_ua(),  
  363.           'comment_date' => current_time( 'mysql' ),  
  364.           'comment_date_gmt' => current_time( 'mysql', 1 ),  
  365.           'comment_approved' => 1,  
  366.           'comment_parent' => 0,  
  367.           'comment_type' => 'bbp_note',  
  368.  
  369.  ) ) ); 
  370.  
  371.       /** Email all moderators who are suscribed to this topic */ 
  372.       $user_ids = bbp_get_topic_subscribers( $_POST['bbp-topic-id'] ); 
  373.  
  374.       foreach ( (array) $user_ids as $user_id ) { 
  375.           if ( ( (int) $user_id == (int) $user->ID ) ) 
  376.               continue; 
  377.  
  378.           if ( ! user_can( $user_id, 'moderate' ) ) 
  379.               continue; 
  380.  
  381.           $link = bbp_get_reply_url( $_POST['bbp-reply-id'] ); 
  382.  
  383.           $message = sprintf( __( '%1$s added a new note on: 
  384.  
  385. %2$s 
  386. ----------- 
  387.  
  388. View note: %3$s 
  389.  
  390. You are receiving this email because you subscribed to a forum topic. 
  391.  
  392. Login and visit the topic to unsubscribe from these emails.', 'bbp-admin-notes' ),  
  393.  
  394.               get_the_author_meta( 'display_name', $user->ID ),  
  395.               $reply_note,  
  396.               $link 
  397.  ); 
  398.  
  399.           // For plugins to filter titles per reply/topic/user 
  400.           $subject = '[' . wp_specialchars_decode( get_option( 'blogname' ), ENT_QUOTES ) . '] ' . strip_tags( bbp_get_topic_title( $_POST['bbp-topic-id'] ) ) . ' (' . __( 'New Admin Note', 'bbp-admin-notes' ) . ')'; 
  401.  
  402.           $headers = apply_filters( 'bbp_subscription_mail_headers', array() ); 
  403.  
  404.           // Get user data of this user 
  405.           $user_data = get_userdata( $user_id ); 
  406.  
  407.           // Send notification email 
  408.           wp_mail( $user_data->user_email, $subject, $message, $headers ); 
  409.       } 
  410.  
  411.       return $note_id; 
  412.   }
