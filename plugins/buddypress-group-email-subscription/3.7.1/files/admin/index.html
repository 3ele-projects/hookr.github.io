  170. function ass_update_dashboard_settings() { 
  171.   if ( !check_admin_referer( 'ass_admin_settings' ) )

  173.  
  174.   if ( !is_super_admin() ) 
  175.       return; 
  176.  
  177.   /** The daily digest time has been changed */ 
  178.   if ( $_POST['ass_digest_time'] ) 
  179.       ass_set_daily_digest_time( $_POST['ass_digest_time']['hours'], $_POST['ass_digest_time']['minutes'] ); 
  180.  
  181.   /** The weekly digest day has been changed */ 
  182.   if ( $_POST['ass_weekly_digest'] ) 
  183.       ass_set_weekly_digest_time( $_POST['ass_weekly_digest'] ); 
  184.  
  185.   if ( $_POST['ass-global-unsubscribe-link'] != bp_get_option( 'ass-global-unsubscribe-link' ) ) 
  186.       bp_update_option( 'ass-global-unsubscribe-link', $_POST['ass-global-unsubscribe-link'] ); 
  187.  
  188.   if ( $_POST['ass-admin-can-edit-email'] != bp_get_option( 'ass-admin-can-edit-email' ) ) 
  189.       bp_update_option( 'ass-admin-can-edit-email', $_POST['ass-admin-can-edit-email'] ); 
  190.  
  191.   if ( $_POST['ass-admin-can-send-email'] != bp_get_option( 'ass-admin-can-send-email' ) ) 
  192.       bp_update_option( 'ass-admin-can-send-email', $_POST['ass-admin-can-send-email'] ); 
  193.  
  194.   if ( $_POST['ass_registered_req'] != bp_get_option( 'ass_registered_req' ) ) 
  195.       bp_update_option( 'ass_registered_req', $_POST['ass_registered_req'] );

  197.   return true;

  214. /** 
  215.  * Install GES emails during email installation routine for BuddyPress. 
  216.  * 
  217.  * @since 3.7.0 
  218.  * 
  219.  * @param bool $post_exists_check Should we check to see if our email post types exist before installing? 
  220.  *                                Default: false. 
  221.  */ 
  222. function ass_install_emails( $post_exists_check = false ) { 
  223.   // No need to check if our post types exist. 
  224.   if ( ! $post_exists_check ) { 
  225.       ass_set_email_type( 'bp-ges-single', false ); 
  226.       ass_set_email_type( 'bp-ges-digest', false ); 
  227.       ass_set_email_type( 'bp-ges-notice', false ); 
  228.       ass_set_email_type( 'bp-ges-welcome', false ); 
  229.  
  230.   // Only create email post types if they do not exist. 
  231.   } else { 
  232.       switch_to_blog( bp_get_root_blog_id() ); 
  233.  
  234.       $ges_types = array( 'bp-ges-single', 'bp-ges-digest', 'bp-ges-notice', 'bp-ges-welcome' ); 
  235.  
  236.       // Try to fetch email posts with our taxonomy. 
  237.       $emails = get_posts( array( 
  238.           'fields' => 'ids',  
  239.           'post_status' => 'publish',  
  240.           'post_type' => bp_get_email_post_type(),  
  241.           'posts_per_page' => 4,  
  242.           'suppress_filters' => false,  
  243.           'tax_query' => array( 
  244.               'relation' => 'OR',  
  245.               array( 
  246.                   'taxonomy' => bp_get_email_tax_type(),  
  247.                   'field' => 'slug',  
  248.                   'terms' => $ges_types,  
  249.  ),  
  250.  ),  
  251.  ) ); 
  252.  
  253.       // See if our taxonomies are attached to our email posts. 
  254.       $found = array(); 
  255.       foreach ( $emails as $post_id ) { 
  256.           $tax = wp_get_object_terms( $post_id, bp_get_email_tax_type(), array( 'fields' => 'slugs' ) ); 
  257.           $found = array_merge( $found, (array) $tax ); 
  258.       } 
  259.  
  260.       restore_current_blog(); 
  261.  
  262.       // Find out if we need to create any posts. 
  263.       $to_create = array_diff( $ges_types, $found ); 
  264.       if ( empty( $to_create ) ) { 
  265.           return; 
  266.       } 
  267.  
  268.       // Create posts with our email types. 
  269.       if ( ! function_exists( 'ass_set_email_type' ) ) { 
  270.           require_once( dirname( __FILE__ ) . '/bp-activity-subscription-functions.php' ); 
  271.       } 
  272.       foreach ( $to_create as $email_type ) { 
  273.           ass_set_email_type( $email_type, false ); 
  274.       } 
  275.   } 
  276. } 
  277. add_action( 'bp_core_install_emails', 'ass_install_emails' );
