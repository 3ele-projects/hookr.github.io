  215.       // do the following for forum replies 
  216.       } else { 
  217.           $topic_status = isset( $user_topic_status[$user_id] ) ? $user_topic_status[$user_id] : ''; 
  218.  
  219.           // the topic mute button will override the subscription options below

  224.           // maybe not neccesary, but good to be cautious 
  225.           if ( $group_status == 'sum' && $topic_status != 'sub' )

  367.   global $bp;

  377.  
  378.   if ( $topic_status == 'mute' || ( $group_status != 'supersub' && !$topic_status ) ) { 
  379.       $action = 'follow'; 
  380.       $link_text = __('Follow', 'bp-ass'); 
  381.       $title = __('You are not following this topic. Click to follow it and get email updates for new posts', 'bp-ass'); 
  382.   } else if ( $topic_status == 'sub' || ( $group_status == 'supersub' && !$topic_status ) ) { 
  383.       $action = 'mute'; 
  384.       $link_text = __('Mute', 'bp-ass'); 
  385.       $title = __('You are following this topic. Click to stop getting email updates', 'bp-ass');

  387.       echo 'nothing'; // do nothing 
  388.   } 
  389.  
  390.   if ( $topic_status == 'mute' ) 
  391.       $title = __('This conversation is muted. Click to follow it', 'bp-ass'); 
  392.  
  393.   if ( $action && bp_is_action_variable( 'topic', 0 ) ) { // we're viewing one topic 
  394.       echo '<div class="generic-button ass-topic-subscribe"><a title="'.$title.'" 
  395.           id="'.$action.'-'.$topic_id.'-'.bp_get_current_group_id().'">'.$link_text.' '.__('this topic', 'bp-ass').'</a></div>'; 
  396.   } else if ( $action )  { // we're viewing a list of topics 
  397.       echo '<td class="td-email-sub"><div class="generic-button ass-topic-subscribe"><a title="'.$title.'" 
  398.           id="'.$action.'-'.$topic_id.'-'.bp_get_current_group_id().'">'.$link_text.'</a></div></td>'; 
  399.   } 
  400. }

  437. /**

  440. function ass_topic_subscribe_or_mute( $action, $user_id, $topic_id, $group_id ) { 
  441.   global $bp; 
  442.  
  443.   if ( !$action || !$user_id || !$topic_id || !$group_id ) 
  444.       return false; 
  445.  
  446.   //$mute_list = get_usermeta( $user_id, 'ass_topic_mute' ); 
  447.   $user_topic_status = groups_get_groupmeta( $group_id, 'ass_user_topic_status_' . $topic_id ); 
  448.  
  449.   if ( $action == 'unsubscribe' ||  $action == 'mute' ) { 
  450.       //$mute_list[ $topic_id ] = 'mute'; 
  451.       $user_topic_status[ $user_id ] = 'mute'; 
  452.   } elseif ( $action == 'subscribe' ||  $action == 'follow' ) { 
  453.       //$mute_list[ $topic_id ] = 'subscribe'; 
  454.       $user_topic_status[ $user_id ] = 'sub'; 
  455.   } 
  456.  
  457.   //update_usermeta( $user_id, 'ass_topic_mute', $mute_list ); 
  458.   groups_update_groupmeta( $group_id , 'ass_user_topic_status_' . $topic_id, $user_topic_status ); 
  459.   //bb_update_topicmeta( $topic_id, 'ass_mute_users', $user_id );
