  36. function ass_manage_members_email_update() { 
  37.   global $bp; 
  38.  
  39.   if ( bp_is_groups_component() && bp_is_action_variable( 'manage-members', 0 ) ) { 
  40.  
  41.       if ( !$bp->is_item_admin ) 
  42.           return false; 
  43.  
  44.       if ( bp_is_action_variable( 'email', 1 ) && ( bp_is_action_variable( 'no', 2 ) || bp_is_action_variable( 'sum', 2 ) || bp_is_action_variable( 'dig', 2 ) || bp_is_action_variable( 'sub', 2 ) || bp_is_action_variable( 'supersub', 2 ) ) && isset( $bp->action_variables[3] ) && is_numeric( $bp->action_variables[3] ) ) { 
  45.  
  46.           $user_id = $bp->action_variables[3]; 
  47.           $action = $bp->action_variables[2]; 
  48.  
  49.           /** Check the nonce first. */ 
  50.           if ( !check_admin_referer( 'ass_member_email_status' ) ) 
  51.               return false; 
  52.  
  53.           ass_group_subscription( $action, $user_id, bp_get_current_group_id() ); 
  54.           bp_core_add_message( __( 'User email status changed successfully', 'bp-ass' ) ); 
  55.           bp_core_redirect( bp_get_group_permalink( $bp->groups->current_group ) . 'admin/manage-members/' ); 
  56.       } 
  57.   } 
  58. } 
  59. add_action( 'bp_actions', 'ass_manage_members_email_update' );

  170. function ass_admin_notice() { 
  171.   if ( bp_is_groups_component() && bp_is_current_action( 'admin' ) && bp_is_action_variable( 'notifications', 0 ) ) { 
  172.  
  173.           // Make sure the user is an admin 
  174.       if ( !groups_is_user_admin( bp_loggedin_user_id(), bp_get_current_group_id() ) && ! is_super_admin() ) 
  175.           return; 
  176.  
  177.       if ( get_option('ass-admin-can-send-email') == 'no' ) 
  178.           return; 
  179.  
  180.       // make sure the correct form variables are here 
  181.       if ( ! isset( $_POST[ 'ass_admin_notice_send' ] ) ) 
  182.           return; 
  183.  
  184.       if ( empty( $_POST[ 'ass_admin_notice' ] ) ) { 
  185.           bp_core_add_message( __( 'The email notice was sent not sent. Please enter email content.', 'bp-ass' ), 'error' ); 
  186.       } else { 
  187.           $group = groups_get_current_group(); 
  188.           $group_id = $group->id; 
  189.           $group_name = bp_get_current_group_name(); 
  190.           $group_link = bp_get_group_permalink( $group ); 
  191.  
  192.           if ( $group->status != 'public' ) { 
  193.               $group_link = ass_get_login_redirect_url( $group_link, 'admin_notice' ); 
  194.           } 
  195.  
  196.           $blogname = '[' . get_blog_option( BP_ROOT_BLOG, 'blogname' ) . ']'; 
  197.           $subject = $_POST[ 'ass_admin_notice_subject' ]; 
  198.           $subject   .= __(' - sent from the group ', 'bp-ass') . $group_name . ' ' . $blogname; 
  199.           $subject = apply_filters( 'ass_admin_notice_subject', $subject, $_POST[ 'ass_admin_notice_subject' ], $group_name, $blogname ); 
  200.           $subject = ass_clean_subject( $subject, false ); 
  201.           $notice = apply_filters( 'ass_admin_notice_message', $_POST['ass_admin_notice'] ); 
  202.           $notice = ass_clean_content( $notice ); 
  203.  
  204.           $message = sprintf( __( 
  205. 'This is a notice from the group \'%s\': 
  206.  
  207. "%s" 
  208.  
  209.  
  210. To view this group log in and follow the link below: 
  211. %s 
  212.  
  213. --------------------- 
  214. ', 'bp-ass' ), $group_name, $notice, $group_link ); 
  215.  
  216.           $message .= __( 'Please note: admin notices are sent to everyone in the group and cannot be disabled. 
  217. If you feel this service is being misused please speak to the website administrator.', 'bp-ass' ); 
  218.  
  219.           $user_ids = BP_Groups_Member::get_group_member_ids( $group_id ); 
  220.  
  221.           // allow others to perform an action when this type of email is sent, like adding to the activity feed 
  222.           do_action( 'ass_admin_notice', $group_id, $subject, $notice ); 
  223.  
  224.           // cycle through all group members 
  225.           foreach ( (array)$user_ids as $user_id ) { 
  226.               $user = bp_core_get_core_userdata( $user_id ); // Get the details for the user 
  227.  
  228.               if ( $user->user_email ) 
  229.                   wp_mail( $user->user_email, $subject, $message );  // Send the email 
  230.  
  231.               //echo '<br>Email: ' . $user->user_email; 
  232.           } 
  233.  
  234.           bp_core_add_message( __( 'The email notice was sent successfully.', 'bp-ass' ) ); 
  235.           //echo '<p>Subject: ' . $subject; 
  236.           //echo '<pre>'; print_r( $message ); echo '</pre>'; 
  237.       } 
  238.  
  239.       bp_core_redirect( bp_get_group_permalink( groups_get_current_group() ) . 'admin/notifications/' ); 
  240.   } 
  241. } 
  242. add_action( 'bp_actions', 'ass_admin_notice', 1 );
