  658.     /** 
  659.      * Fitler avatar to eventually replace it by a suggestion 
  660.      * 
  661.      * @package BP Avatar Suggestions 
  662.      * @subpackage Front 
  663.      * @since   1.1.0 
  664.      */ 
  665.     function suggestion_avatar( $image = '', $params = array() ) { 
  666.         if ( ! empty( $this->avatar_removed ) ) { 
  667.             return $image; 
  668.         } 
  669.  
  670.         if ( ! empty( $params['no_grav'] ) || empty( $params['item_id'] ) ) { 
  671.             return $image; 
  672.         } 
  673.  
  674.         if ( ! $this->enable_users && 'user' == $params['object'] ) { 
  675.             return $image; 
  676.         } 
  677.  
  678.         if ( ! $this->enable_groups && 'group' == $params['object'] ) { 
  679.             return $image; 
  680.         } 
  681.  
  682.         $component_items = array( 'user' => 1, 'group' => 1 ); 
  683.         if ( empty( $component_items[ $params['object'] ] ) ) { 
  684.             return $image; 
  685.         } 
  686.  
  687.         $item_id = absint( $params['item_id'] ); 
  688.  
  689.         switch( $params['object'] ) { 
  690.             case 'group' : 
  691.                 if ( ! bp_get_group_has_avatar( $item_id ) ) { 
  692.  
  693.                     $group_choice = groups_get_groupmeta( $item_id, 'group_avatar_choice', true ); 
  694.  
  695.                     if ( empty( $group_choice ) ) { 
  696.                         return $image; 
  697.                     } else { 
  698.                         $group_choice = str_replace( array( 'https:', 'http:' ), '', $group_choice ); 
  699.                      } 
  700.  
  701.                     if ( ! empty( $params['html'] ) ) { 
  702.                         $image = preg_replace('/src="([^"]*)"/i', 'src="' . $group_choice . '"', $image ); 
  703.                     } else { 
  704.                         $image = $group_choice; 
  705.                     } 
  706.                 } 
  707.                 break; 
  708.  
  709.             case 'user' : 
  710.                 if ( ! bp_get_user_has_avatar( $item_id ) ) { 
  711.  
  712.                     $user_choice = get_user_meta( $item_id, 'user_avatar_choice', true ); 
  713.  
  714.                     if ( empty( $user_choice ) ) { 
  715.                         return $image; 
  716.                     } else { 
  717.                         $user_choice = str_replace( array( 'https:', 'http:' ), '', $user_choice ); 
  718.                      } 
  719.  
  720.                     if ( ! empty( $params['html'] ) ) { 
  721.                         $image = preg_replace('/src="([^"]*)"/i', 'src="' . $user_choice . '"', $image ); 
  722.                     } else { 
  723.                         $image = $user_choice; 
  724.                     } 
  725.  
  726.                 } 
  727.                 break; 
  728.         } 
  729.  
  730.         /** in case you need to filter with your own function... */ 
  731.         return apply_filters( 'bp_as_fetch_suggested_avatar', $image, $params ); 
  732.     }
