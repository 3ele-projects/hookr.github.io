  276.      * Checks if your plugin's config is similar to BuddyPress 
  277.      * 
  278.      * @package BP Avatar Suggestions

  280.      */ 
  281.     public static function config_check() { 
  282.         /** 
  283.          * blog_status    : true if your plugin is activated on the same blog 
  284.          * network_active : true when your plugin is activated on the network 
  285.          * network_status : BuddyPress & your plugin share the same network status 
  286.          */ 
  287.         self::$bp_config = array( 
  288.             'blog_status' => false,  
  289.             'network_active' => false,  
  290.             'network_status' => true,  
  291.  ); 
  292.  
  293.         if ( get_current_blog_id() == bp_get_root_blog_id() ) { 
  294.             self::$bp_config['blog_status'] = true; 
  295.         } 
  296.  
  297.         $network_plugins = get_site_option( 'active_sitewide_plugins', array() ); 
  298.  
  299.         // No Network plugins 
  300.         if ( empty( $network_plugins ) ) { 
  301.             return self::$bp_config; 
  302.         } 
  303.  
  304.         $plugin_basename = plugin_basename( __FILE__ ); 
  305.  
  306.         // Looking for BuddyPress and this plugin 
  307.         $check = array( buddypress()->basename, $plugin_basename ); 
  308.  
  309.         // Are they active on the network ? 
  310.         $network_active = array_diff( $check, array_keys( $network_plugins ) ); 
  311.  
  312.         // If result is 1, your plugin is network activated 
  313.         // and not BuddyPress or vice & versa. Config is not ok 
  314.         if ( count( $network_active ) == 1 ) { 
  315.             self::$bp_config['network_status'] = false; 
  316.         } 
  317.  
  318.         self::$bp_config['network_active'] = isset( $network_plugins[ $plugin_basename ] ); 
  319.  
  320.         return self::$bp_config; 
  321.     }
