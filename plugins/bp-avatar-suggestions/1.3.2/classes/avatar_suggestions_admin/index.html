  538.             <form id="bp-avatar-suggestions-form" action="<?php echo esc_url( $form_url );?>" method="post"> 
  539.                 <?php $this->suggestion_list_table->display(); ?> 
  540.             </form> 
  541.  
  542.         </div> 
  543.         <?php

  546.     /** 
  547.      * Upload the suggestions as WordPress attachments 
  548.      * 
  549.      * @package BP Avatar Suggestions 
  550.      * @subpackage Admin 
  551.      * @since   1.3.0 
  552.      */ 
  553.     public function handle_upload() { 
  554.         // Bail if not a POST action 
  555.         if ( 'POST' !== strtoupper( $_SERVER['REQUEST_METHOD'] ) ) { 
  556.             wp_die(); 
  557.         } 
  558.  
  559.         /** 
  560.          * Sending the json response will be different if 
  561.          * the current Plupload runtime is html4 
  562.          */ 
  563.         $is_html4 = false; 
  564.         if ( ! empty( $_POST['html4' ] ) ) { 
  565.             $is_html4 = true; 
  566.         } 
  567.  
  568.         // Check the nonce 
  569.         check_admin_referer( 'bp-uploader' ); 
  570.  
  571.         // Init the BuddyPress parameters 
  572.         $bp_params = array(); 
  573.  
  574.         // We need it to carry on 
  575.         if ( ! empty( $_POST['bp_params' ] ) ) { 
  576.             $bp_params = $_POST['bp_params' ]; 
  577.         } else { 
  578.             bp_attachments_json_response( false, $is_html4 ); 
  579.         } 
  580.  
  581.         // Check params 
  582.         if ( empty( $bp_params['object'] ) || empty( $bp_params['item_id'] ) || 'post' !== $bp_params['object'] || (int) $this->avatar_post_id !== (int) $bp_params['item_id'] ) { 
  583.             bp_attachments_json_response( false, $is_html4 ); 
  584.         } 
  585.  
  586.         // Capability check 
  587.         if ( ! bp_current_user_can( 'upload_files' ) ) { 
  588.             bp_attachments_json_response( false, $is_html4 ); 
  589.         } 
  590.  
  591.         $suggestion_attachment = new Avatar_Suggestions_Attachment(); 
  592.         $suggestion = $suggestion_attachment->upload( $_FILES ); 
  593.  
  594.         // Error while trying to upload the file 
  595.         if ( ! empty( $suggestion['error'] ) ) { 
  596.             bp_attachments_json_response( false, $is_html4, array( 
  597.                 'type' => 'upload_error',  
  598.                 'message' => $suggestion['error'],  
  599.  ) ); 
  600.         } 
  601.  
  602.         /** 
  603.          * Simulate WordPress's media_handle_upload() function 
  604.          */ 
  605.         $time = current_time( 'mysql' ); 
  606.         $post = get_post( $bp_params['item_id'] ); 
  607.  
  608.         if ( is_a( $post, 'WP_Post' ) ) { 
  609.             if ( substr( $post->post_date, 0, 4 ) > 0 ) { 
  610.                 $time = $post->post_date; 
  611.             } 
  612.         } 
  613.  
  614.         $name_parts = pathinfo( $suggestion['file'] ); 
  615.         $url = $suggestion['url']; 
  616.         $type = $suggestion['type']; 
  617.         $file = $suggestion['file']; 
  618.         $title = $name_parts['filename']; 
  619.         $content = ''; 
  620.         $excerpt = ''; 
  621.         $image_meta = @wp_read_image_metadata( $file ); 
  622.  
  623.         if ( 0 === strpos( $type, 'image/' ) && ! empty( $image_meta ) ) { 
  624.             if ( trim( $image_meta['title'] ) && ! is_numeric( sanitize_title( $image_meta['title'] ) ) ) { 
  625.                 $title = $image_meta['title']; 
  626.             } 
  627.  
  628.             if ( trim( $image_meta['caption'] ) ) { 
  629.                 $excerpt = $image_meta['caption']; 
  630.             } 
  631.         } 
  632.  
  633.         // Construct the attachment array 
  634.         $attachment = array( 
  635.             'post_mime_type' => $type,  
  636.             'guid' => $url,  
  637.             'post_parent' => $post->ID,  
  638.             'post_title' => $title,  
  639.             'post_content' => $content,  
  640.             'post_excerpt' => $excerpt,  
  641.  ); 
  642.  
  643.         // Save the data 
  644.         $attachment_id = wp_insert_attachment( $attachment, $file, $post->ID ); 
  645.         if ( ! is_wp_error( $attachment_id ) ) { 
  646.  
  647.             // Add the avatar image sizes 
  648.             add_image_size( 'bp_avatar_suggestions', bp_core_avatar_full_width(), bp_core_avatar_full_height(), true ); 
  649.             add_filter( 'intermediate_image_sizes_advanced', array( $this, 'restrict_image_sizes' ), 10, 1 ); 
  650.  
  651.             wp_update_attachment_metadata( $attachment_id, wp_generate_attachment_metadata( $attachment_id, $file ) ); 
  652.  
  653.             // Remove it so no other attachments will be affected 
  654.             remove_image_size( 'bp_avatar_suggestions' ); 
  655.             remove_filter( 'intermediate_image_sizes_advanced', array( $this, 'restrict_image_sizes' ), 10, 1 ); 
  656.  
  657.             // Try to get the bp_avatar_suggestions size 
  658.             $avatar = wp_get_attachment_image_src( $attachment_id, 'bp_avatar_suggestions', true ); 
  659.             if ( ! empty( $avatar[0] ) ) { 
  660.                 $url = $avatar[0]; 
  661.             } 
  662.  
  663.             // Finally return the avatar to the editor 
  664.             bp_attachments_json_response( true, $is_html4, array( 
  665.                 'name' => esc_html( $title ),  
  666.                 'url' => esc_url_raw( $url ),  
  667.  ) ); 
  668.         } else { 
  669.             bp_attachments_json_response( false, $is_html4, array( 
  670.                 'type' => 'upload_error',  
  671.                 'message' => esc_html__( 'Something went wrong while saving the image', 'bp-avatar-suggestions' ),  
  672.  ) ); 
  673.         } 
  674.     } 
  675.  
  676.     /** 
  677.      * Only keep the avatar suggestions size and thumbnail one for backcompat 
  678.      * 
  679.      * @package BP Avatar Suggestions 
  680.      * @subpackage Admin 
  681.      * @since   1.3.0 
  682.      * 
  683.      * @param  array $sizes the available WordPress image sizes 
  684.      */ 
  685.     public function restrict_image_sizes( $sizes = array() ) { 
  686.         return array_intersect_key( $sizes, array( 'thumbnail' => true, 'bp_avatar_suggestions' => true ) ); 
  687.     } 
  688. }
