  787.        * 
  788.        * @return bool

  790.       protected function _selfTransient($qObj, $quote = null) 
  791.       {

  796.           { 
  797.               set_transient($hashKey, $quote, 1200);

  800.               // Add one 
  801.               if (empty($transients))

  805.               WC()->session->set('ssTransients', $transients); 
  806.               return true; 
  807.           } 
  808.  
  809.           if ( !( $return = get_transient($hashKey))) 
  810.               return false; 
  811.           else 
  812.               return $return; 
  813.       }
