  320.               'id' => 'shipping',  
  321.               'name' => __('Shipping', PAYNL_WOOCOMMERCE_TEXTDOMAIN),  
  322.               'price' => $shipping,  
  323.               'tax' => $order->get_shipping_tax(),  
  324.               'qty' => 1,  
  325.               'type' => \Paynl\Transaction::PRODUCT_TYPE_SHIPPING 
  326.  );
