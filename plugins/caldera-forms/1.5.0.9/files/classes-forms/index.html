  639.    * Get all fields of a form

  642.    * 
  643.    * @param array $form The form config

  645.    *

  647.    */ 
  648.   public static function get_fields( array $form, $in_order = true ) { 
  649.       if( empty( $form[ 'fields' ] ) ) {

  651.       }

  654.

  674.
