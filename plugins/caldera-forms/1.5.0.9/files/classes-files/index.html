  35.   public static function upload( $file, array  $args = array() ) { 
  36.       $args = wp_parse_args($args, array( 
  37.           'private' => false,  
  38.           'field_id' => null,  
  39.           'form_id' => null,  
  40.  )); 
  41.       if( true == $args[ 'private' ] && ! empty( $args[ 'field_id' ] ) && ! empty( $args[ 'form_id' ] )) { 
  42.           $private = true;

  57.  
  58.       if( $private ) { 
  59.           self::remove_upload_filter(); 
  60.       } 
  61.

  63.  
  64.   }

  66.   /**

  69.    * @since 1.4.4

  75.   public static function add_to_media_library( $upload ) { 
  76.       if( isset( $upload[ 'error' ] ) ) { 
  77.           return false; 
  78.       } 
  79.  
  80.       require_once( ABSPATH . 'wp-admin/includes/media.php' ); 
  81.       require_once( ABSPATH . 'wp-admin/includes/image.php' ); 
  82.  
  83.       $media_item = array( 
  84.           'guid' => $upload['file'],  
  85.           'post_mime_type' => $upload['type'],  
  86.           'post_title' => preg_replace( '/\.[^.]+$/', '', basename( $upload['file'] ) ),  
  87.           'post_content' => '',  
  88.           'post_status' => 'inherit' 
  89.  ); 
  90.  
  91.       $media_id = wp_insert_attachment( $media_item, $upload['file'] ); 
  92.  
  93.       $media_data = wp_generate_attachment_metadata( $media_id, $upload['file'] );

  99.   /** 
  100.    * Setup upload directory filter 
  101.    * 
  102.    * @since 1.4.4 
  103.    * 
  104.    * @param string $field_id The field ID for file field 
  105.    * @param string $form_id The form ID

  107.   public static function add_upload_filter( $field_id , $form_id, $private = true ) { 
  108.       if ( $private ) { 
  109.           self::$dir = self::secret_dir( $field_id, $form_id ); 
  110.       }else{ 
  111.           /** 
  112.            * Filter directory for uploaded files 
  113.            * 
  114.            * If null, the file path will be WordPress' default 
  115.            * 
  116.            * @since 1.4.5 
  117.            * 
  118.            * @param string|null Directory 
  119.            * @param string $field_id Field ID 
  120.            * @param string $form_id Form ID 
  121.            */ 
  122.           $dir = apply_filters( 'caldera_forms_upload_directory', null, $field_id, $form_id ); 
  123.           if( null != $dir ) { 
  124.               self::$dir = $dir; 
  125.           } 
  126.       } 
  127.       add_filter( 'upload_dir', array( __CLASS__, 'uploads_filter' ) ); 
  128.   }

  139.   /** 
  140.    * Filter upload directory 
  141.    * 
  142.    * @uses "upload_dir" filter 
  143.    * 
  144.    * @since 1.4.4 
  145.    * 
  146.    * @param array $args 
  147.    * @return array

  149.   public static function uploads_filter( $args ) { 
  150.  
  151.       if (  self::$dir ) { 
  152.           $newdir = '/' . self::$dir; 
  153.  
  154.           $args[ 'path' ]   = str_replace( $args[ 'subdir' ], '', $args[ 'path' ] ); 
  155.           $args[ 'url' ]    = str_replace( $args[ 'subdir' ], '', $args[ 'url' ] ); 
  156.           $args[ 'subdir' ] = $newdir; 
  157.           $args[ 'path' ] .= $newdir; 
  158.           $args[ 'url' ] .= $newdir; 
  159.       }

  164.   /** 
  165.    * Get a secret file fir by field ID and form ID 
  166.    * 
  167.    * @since 1.4.4 
  168.    *

  180.    * Delete all files from the secret dir for a field 
  181.    * 
  182.    * @since 1.4.4

  189.       $dir = $uploads[ 'basedir' ] . '/' . self::secret_dir($field_id, $form_id); 
  190.       if (is_dir($dir)) { 
  191.           array_map('unlink', glob($dir . '/*')); 
  192.           rmdir($dir); 
  193.       } 
  194.   
  195.   }

  205.   public static function cleanup( $form, $second_run = false ) { 
  206.       if( false === $second_run && Caldera_Forms::should_send_mail( $form ) ) { 
  207.           add_action( 'caldera_forms_mailer_complete', array( __CLASS__, 'delete_after_mail' ), 10, 3 ); 
  208.           add_action( 'caldera_forms_mailer_failed', array( __CLASS__, 'delete_after_mail' ), 10, 3 ); 
  209.           return; 
  210.       } 
  211.  
  212.       $form_id = $form[ 'ID' ]; 
  213.       $fields = Caldera_Forms_Forms::get_fields( $form, false ); 
  214.       foreach( $fields as $id => $field ) { 
  215.           if( Caldera_Forms_Field_Util::is_file_field( $field, $form ) ) { 
  216.               self::delete_uploaded_files( $field[ 'ID' ], $form_id ); 
  217.           }

  219.       } 
  220.  
  221.   }
