  120. function cf_ajax_redirect($type, $url, $form) { 
  121.  
  122.   if(empty($form['form_ajax'])) { 
  123.       return; 
  124.   } 
  125.  
  126.   if( empty( $_POST['cfajax'] ) || empty( $_POST['action'] ) || $_POST['action'] != 'cf_process_ajax_submit' ) { 
  127.       return; 
  128.   } 
  129.  
  130.   $data = Caldera_Forms::get_submission_data($form); 
  131.    
  132.   // setup notcies 
  133.   $urlparts = parse_url($url); 
  134.   $query = array(); 
  135.  
  136.   if(!empty($urlparts['query'])) { 
  137.       parse_str($urlparts['query'], $query); 
  138.   } 
  139.  
  140.   $notices = array(); 
  141.   $note_general_classes = Caldera_Forms_Render_Notices::get_note_general_classes( $form ); 
  142.  
  143.   // base id 
  144.   $form_id = 'caldera_form_1'; 
  145.  
  146.   if($type == 'complete') { 
  147.       if(isset($query['cf_su'])) { 
  148.           $notices['success']['note'] = $form['success']; 
  149.           $form_id = 'caldera_form_' . $query['cf_su']; 
  150.       }else{ 
  151.           $out['url'] = $url; 
  152.           $notices['success']['note'] = esc_html__( 'Redirecting', 'caldera-forms'); 
  153.       } 
  154.   }elseif($type == 'preprocess') { 
  155.       if(isset($query['cf_er'])) { 
  156.           $data = Caldera_Forms_Transient::get_transient( $query['cf_er'] ); 
  157.           if(!empty($data['note'])) { 
  158.               $notices[$data['type']]['note'] = $data['note']; 
  159.           } 
  160.  
  161.       }else{ 
  162.           $out['url'] = $url; 
  163.           $notices['success']['note'] = esc_html__( 'Redirecting', 'caldera-forms'); 
  164.       } 
  165.  
  166.   }elseif($type == 'error') { 
  167.       $data = Caldera_Forms_Transient::get_transient( $query['cf_er'] ); 
  168.       if(!empty($data['note'])) { 
  169.           $notices['error']['note'] = $data['note']; 
  170.       } 
  171.  
  172.   } 
  173.   // check for field erors 
  174.   if(!empty($data['fields'])) { 
  175.       foreach($form['fields'] as $fieldid=>$field) { 
  176.           if( isset( $data['fields'][$fieldid] ) ) { 
  177.  
  178.               if($urlparts['path'] == 'api') { 
  179.                   $out['fields'][$field['slug']] = $data['fields'][$fieldid]; 
  180.               }else{ 
  181.                   $out['fields'][$fieldid] = $data['fields'][$fieldid]; 
  182.               } 
  183.           } 
  184.       } 
  185.   } 
  186.  
  187.   $notices = Caldera_Forms_Render_Notices::prepare_notices( $notices, $form ); 
  188.   $note_classes = Caldera_Forms_Render_Notices::get_note_classes( $note_general_classes, $form ); 
  189.  
  190.   $html = Caldera_Forms_Render_Notices::html_from_notices( $notices, $note_classes ); 
  191.  
  192.   if(!empty($result)) { 
  193.       $out['result'] = $result; 
  194.   } 
  195.  
  196.   if(!empty($query)) { 
  197.       if(!empty($query['cf_su'])) { 
  198.           unset($query['cf_su']); 
  199.       } 
  200.       if(!empty($query['cf_ee'])) { 
  201.           $out['entry'] = $query['cf_ee']; 
  202.       } 
  203.       $out['data'] = $query; 
  204.   } 
  205.   $out['html'] = $html; 
  206.   $out['type'] = ( isset($data['type']) ? $data['type'] : $type ); 
  207.   $out['form_id'] = $form['ID']; 
  208.   $out['form_name'] = $form['name'];     
  209.   $out['status'] = $type; 
  210.  
  211.   if( ! empty( $form[ 'scroll_top' ] ) ) { 
  212.       $out[ 'scroll' ] = Caldera_Forms_Render_Util::notice_element_id( $form, absint( $_POST[ '_cf_frm_ct' ] ) ); 
  213.   } 
  214.  
  215.   $out = apply_filters( 'caldera_forms_ajax_return', $out, $form); 
  216.  
  217.   wp_send_json( $out ); 
  218.   exit; 
  219.  
  220. }
