  238.     public function gdm_shortcode_display_drivefile($atts, $content=null) { 
  239.         if (!isset($atts['url'])) { 
  240.             return '<b>google-drive-embed requires a url attribute</b>'; 
  241.         } 
  242.         $url = $atts['url']; 
  243.         $title = isset($atts['title']) ? $atts['title'] : $url; // Should be html-encoded already 
  244.          
  245.         $linkstyle = isset($atts['style']) && in_array($atts['style'], Array('normal', 'plain', 'download', 'embed'))  
  246.                         ? $atts['style'] : 'normal'; 
  247.         $returnhtml = ''; 
  248.         switch ($linkstyle) { 
  249.             case 'normal':

  255.                 } 
  256.                 $newwindow = isset($atts['newwindow']) && $atts['newwindow']=='yes' ? ' target="_blank"' : ''; 
  257.                 $ahref = '<a href="'.$url.'"'.$newwindow.'>'.$title.'</a>'; 
  258.                 if ((isset($atts['plain']) && $atts['plain']=='yes') || $linkstyle=='plain') { 
  259.                     $returnhtml = $ahref; 
  260.                 } 
  261.                 else { 
  262.                     $returnhtml = '<p><span class="gdm-drivefile-embed">'.$imghtml.' '.$ahref.'</span></p>'; 
  263.                 } 
  264.                 break; 
  265.             case 'embed': 
  266.                 $width = isset($atts['width']) ? $atts['width'] : '100%'; 
  267.                 $height = isset($atts['height']) ? $atts['height'] : '400'; 
  268.                 $scrolling = isset($atts['scrolling']) && strtolower($atts['scrolling']) == 'yes' ? 'yes' : 'no'; 
  269.                 $allowfullscreen = isset($atts['allowfullscreen']) && strtolower($atts['allowfullscreen']) == 'no' ? '' : 'allowfullscreen'; 
  270.                 $returnhtml = "<iframe width='${width}' height='${height}' frameborder='0' scrolling='${scrolling}' src='${url}' ${allowfullscreen}></iframe>"; 
  271.                 break; 
  272.         }

  274.             $returnhtml .= do_shortcode($content); 
  275.         }
