  364.                       $checkhash = sprintf( "%u", crc32 ($_REQUEST['TRANSACTIONID'].':'.$_REQUEST['APTRANSACTIONID'].':'.$_REQUEST['AMOUNT'].':'.$_REQUEST['TRANSACTIONSTATUS'].':'.$_REQUEST['MESSAGE'].':'.$this->key_id.':'.$this->username ) ); 
  365.                       $trans_authorised = false; 
  366.  
  367.                       if( $order->status !== 'completed' ) { 
  368.                           if($hash == $checkhash) {

  373.                                   $this->msg['class'] = 'woocommerce-message';

  375.                                       $order->add_order_note('AirPay ID: '.$_REQUEST['APTRANSACTIONID'].' ('.$_REQUEST['TRANSACTIONID'].')');

  377.                                           $order->add_order_note('<br/>Msg: '.$_REQUEST['MESSAGE']); 
  378.                                       } 
  379.                                   }else{ 
  380.                                       $order->payment_complete(); 
  381.                                       $order->add_order_note('AirPay payment successful.<br/>AirPay ID: '.$_REQUEST['APTRANSACTIONID'].' ('.$_REQUEST['TRANSACTIONID'].')'); 
  382.                                       if( isset($_POST['MESSAGE']) && $_POST['MESSAGE'] != "" ) { 
  383.                                           $order->add_order_note('<br/>Msg: '.$_REQUEST['MESSAGE']); 
  384.                                       } 
  385.                                       $woocommerce->cart->empty_cart(); 
  386.                                   } 
  387.                               }else if($status=="pending") { 
  388.                                   $this->msg['message'] = "Thank you for shopping with us. Right now your payment status is pending. We will keep you posted regarding the status of your order through eMail"; 
  389.                                   $this->msg['class'] = 'woocommerce-info';

  392.                                       $order->add_order_note('<br/>Msg: '.$_REQUEST['MESSAGE']); 
  393.                                   }

  398.                                   $this->msg['message'] = "Thank you for shopping with us. However, the transaction has been declined.";

  409.                       }
