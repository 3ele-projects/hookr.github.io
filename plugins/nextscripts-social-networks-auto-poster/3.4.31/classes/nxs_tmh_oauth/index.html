  553.     // configure curl 
  554.     $c = curl_init(); 
  555.     curl_setopt_array($c, array( 
  556.       CURLOPT_USERAGENT => $this->config['user_agent'],  
  557.       CURLOPT_CONNECTTIMEOUT => $this->config['curl_connecttimeout'],  
  558.       CURLOPT_TIMEOUT => $this->config['curl_timeout'],  
  559.       CURLOPT_RETURNTRANSFER => true,  
  560.       CURLOPT_SSL_VERIFYPEER => $this->config['curl_ssl_verifypeer'],  
  561.       CURLOPT_SSL_VERIFYHOST => $this->config['curl_ssl_verifyhost'],  
  562.  
  563.       CURLOPT_FOLLOWLOCATION => $this->config['curl_followlocation'],  
  564.       CURLOPT_PROXY => $this->config['curl_proxy'],  
  565.       CURLOPT_ENCODING => $this->config['curl_encoding'],  
  566.       CURLOPT_URL => $this->url,  
  567.       // process the headers 
  568.       CURLOPT_HEADERFUNCTION => array($this, 'curlHeader'),  
  569.       CURLOPT_HEADER => false,  
  570.       CURLINFO_HEADER_OUT => true,  
  571.  )); 
  572.  
  573.     if ($this->config['curl_cainfo'] !== false) 
  574.       curl_setopt($c, CURLOPT_CAINFO, $this->config['curl_cainfo']); 
  575.  
  576.     if ($this->config['curl_capath'] !== false) 
  577.       curl_setopt($c, CURLOPT_CAPATH, $this->config['curl_capath']); 
  578.  
  579.     if ($this->config['curl_proxyuserpwd'] !== false) 
  580.       curl_setopt($c, CURLOPT_PROXYUSERPWD, $this->config['curl_proxyuserpwd']); 
  581.  
  582.     if ($this->config['is_streaming']) { 
  583.       // process the body 
  584.       $this->response['content-length'] = 0; 
  585.       curl_setopt($c, CURLOPT_TIMEOUT, 0); 
  586.       curl_setopt($c, CURLOPT_WRITEFUNCTION, array($this, 'curlWrite')); 
  587.     } 
  588.  
  589.     switch ($this->method) { 
  590.       case 'GET': 
  591.         break; 
  592.       case 'POST': 
  593.         curl_setopt($c, CURLOPT_POST, true); 
  594.         break; 
  595.       default: 
  596.         curl_setopt($c, CURLOPT_CUSTOMREQUEST, $this->method); 
  597.     } 
  598.  
  599.     if ( ! empty($this->request_params) ) { 
  600.       // if not doing multipart we need to implode the parameters 
  601.       if ( ! $this->config['multipart'] ) { 
  602.         foreach ($this->request_params as $k => $v) { 
  603.           $ps[] = "{$k}={$v}"; 
  604.         } 
  605.         $this->request_params = implode('&', $ps); 
  606.       } 
  607.       curl_setopt($c, CURLOPT_POSTFIELDS, $this->request_params); 
  608.     } else { 
  609.       // CURL will set length to -1 when there is no data, which breaks Twitter 
  610.       $this->headers['Content-Type'] = ''; 
  611.       $this->headers['Content-Length'] = ''; 
  612.     } 
  613.  
  614.     // CURL defaults to setting this to Expect: 100-Continue which Twitter rejects 
  615.     $this->headers['Expect'] = ''; 
  616.  
  617.     if ( ! empty($this->headers)) { 
  618.       foreach ($this->headers as $k => $v) { 
  619.         $headers[] = trim($k . ': ' . $v); 
  620.       } 
  621.       curl_setopt($c, CURLOPT_HTTPHEADER, $headers); 
  622.     } 
  623.  
  624.     if (isset($this->config['prevent_request']) && true == $this->config['prevent_request']) 
  625.       return; 
  626.  
  627.     // do it! 
  628.     global $nxs_skipSSLCheck; if ($nxs_skipSSLCheck===true) curl_setopt($c, CURLOPT_SSL_VERIFYPEER, false); 
  629.     curl_setopt($c, CURLOPT_URL, $this->url);  
  630.     $response = curl_exec($c); 
  631.     $code = curl_getinfo($c, CURLINFO_HTTP_CODE); 
  632.     $info = curl_getinfo($c); 
  633.     $error = curl_error($c); 
  634.     $errno = curl_errno($c); 
  635.     curl_close($c); 
  636.  
  637.     // store the response 
  638.     $this->response['code'] = $code; 
  639.     $this->response['response'] = $response; 
  640.     $this->response['info'] = $info; 
  641.     $this->response['error'] = $error; 
  642.     $this->response['errno'] = $errno; 
  643.     return $code; 
  644.   } 
  645. }
