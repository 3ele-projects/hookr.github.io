/includes/class-wc-payment-discounts-apply-discount.php

  1. <?php 
  2. /** 
  3.  * Apply discounts on checkout 
  4.  * 
  5.  * @package WC_Payment_Discounts/Classes 
  6.  * @since   3.0.0 
  7.  * @version 3.0.0 
  8.  */ 
  9.  
  10. if ( ! defined( 'ABSPATH' ) ) { 
  11.   exit; // Exit if accessed directly. 
  12. } 
  13.  
  14. /** 
  15.  * WC_Payment_Discounts_Apply_Discount class. 
  16.  */ 
  17. class WC_Payment_Discounts_Apply_Discount { 
  18.  
  19.   /** 
  20.    * Initialize the actions. 
  21.    */ 
  22.   public function __construct() { 
  23.       // Load public-facing JavaScript. 
  24.       add_action( 'wp_enqueue_scripts', array( $this, 'enqueue_scripts' ) ); 
  25.  
  26.       // Apply the discounts. 
  27.       add_action( 'woocommerce_calculate_totals', array( $this, 'add_discount' ), 10 ); 
  28.  
  29.       // Display the discount in payment gateways titles. 
  30.       add_filter( 'woocommerce_gateway_title', array( $this, 'payment_method_title' ), 10, 2 ); 
  31.  
  32.       // Fix salved payment method title and update the cart discount total. 
  33.       add_action( 'woocommerce_checkout_order_processed', array( $this, 'update_order_data' ), 10 ); 
  34.   } 
  35.  
  36.   /** 
  37.    * Register and enqueues public-facing JavaScript files. 
  38.    */ 
  39.   public function enqueue_scripts() { 
  40.       if ( is_checkout() ) { 
  41.           $suffix = defined( 'SCRIPT_DEBUG' ) && SCRIPT_DEBUG ? '' : '.min'; 
  42.           wp_enqueue_script( 'woocommerce-payment-discounts', plugins_url( 'assets/js/frontend/update-checkout' . $suffix . '.js', plugin_dir_path( __FILE__ ) ), array( 'wc-checkout' ), WC_Payment_Discounts::VERSION ); 
  43.       } 
  44.   } 
  45.  
  46.   /** 
  47.    * Display the discount in payment method title. 
  48.    * 
  49.    * @param  string $title Gateway title. 
  50.    * @param  string $id    Gateway ID. 
  51.    * 
  52.    * @return string 
  53.    */ 
  54.   public function payment_method_title( $title, $id ) { 
  55.       if ( ! is_checkout() && ! ( defined( 'DOING_AJAX' ) && DOING_AJAX ) ) { 
  56.           return $title; 
  57.       } 
  58.  
  59.       $settings = get_option( 'woocommerce_payment_discounts' ); 
  60.  
  61.       if ( isset( $settings[ $id ]['amount'] ) && 0 < $settings[ $id ]['amount'] ) { 
  62.           $amount = $settings[ $id ]['amount']; 
  63.  
  64.           if ( strstr( $amount, '%' ) ) { 
  65.               $value = $amount; 
  66.           } else { 
  67.               $value = wc_price( $amount ); 
  68.           } 
  69.  
  70.           $title .= ' <small>(' . sprintf( __( '%s off', 'woocommerce-payment-discounts' ), $value ) . ')</small>'; 
  71.       } 
  72.  
  73.       return $title; 
  74.   } 
  75.  
  76.   /** 
  77.    * Remove payment coupons. 
  78.    * 
  79.    * @param  WC_Cart $cart Cart object. 
  80.    * @param  string  $skip Payment method ID to skip during remotion. 
  81.    * @return bool 
  82.    */ 
  83.   protected function remove_payment_coupons( $cart, $skip = '' ) { 
  84.       $removed = false; 
  85.  
  86.       foreach ( $cart->get_applied_coupons() as $code ) { 
  87.           if ( 'wcpd_' === substr( $code, 0, 5 ) && $code !== $skip ) { 
  88.               $cart->remove_coupon( $code ); 
  89.               $removed = true; 
  90.           } 
  91.       } 
  92.  
  93.       return $removed; 
  94.   } 
  95.  
  96.   /** 
  97.    * Add discount. 
  98.    * 
  99.    * @param WC_Cart $cart Cart object. 
  100.    */ 
  101.   public function add_discount( $cart ) { 
  102.       if ( is_admin() && ! defined( 'DOING_AJAX' ) || is_cart() ) { 
  103.           return; 
  104.       } 
  105.  
  106.       // Gets the settings. 
  107.       $settings = get_option( 'woocommerce_payment_discounts' ); 
  108.  
  109.       // Always set coupons calculation as enabled. 
  110.       add_filter( 'woocommerce_coupons_enabled', '__return_true' ); 
  111.  
  112.       if ( isset( $settings[ WC()->session->chosen_payment_method ]['amount'] ) ) { 
  113.           // Gets the gateway discount. 
  114.           $amount = $settings[ WC()->session->chosen_payment_method ]['amount']; 
  115.  
  116.           if ( apply_filters( 'wc_payment_discounts_apply_discount', 0 < $amount, $cart ) ) { 
  117.               // Gets the gateway data. 
  118.               $methods = WC()->payment_gateways->payment_gateways(); 
  119.               $gateway = $methods[ WC()->session->chosen_payment_method ]; 
  120.               $coupon = WC_Payment_Discounts_Coupons::get_coupon( $gateway->id, array( 'amount' => $amount ) ); 
  121.  
  122.               // Remove other coupons and apply method coupon. 
  123.               $this->remove_payment_coupons( $cart, $coupon->get_code() ); 
  124.               if ( ! $cart->has_discount( $coupon->get_code() ) ) { 
  125.                   $cart->add_discount( $coupon->get_code() ); 
  126.               } 
  127.           } else { 
  128.               $this->remove_payment_coupons( $cart ); 
  129.           } 
  130.       } 
  131.   } 
  132.  
  133.   /** 
  134.    * Remove the discount in the payment method title. 
  135.    * 
  136.    * @param int $order_id Order ID. 
  137.    */ 
  138.   public function update_order_data( $order_id ) { 
  139.       $payment_method_title = get_post_meta( $order_id, '_payment_method_title', true ); 
  140.       $new_payment_method_title = preg_replace( '/<small>.*<\/small>/', '', $payment_method_title ); 
  141.  
  142.       // Save the new payment method title. 
  143.       update_post_meta( $order_id, '_payment_method_title', $new_payment_method_title ); 
  144.   } 
  145. } 
  146.  
  147. new WC_Payment_Discounts_Apply_Discount();
