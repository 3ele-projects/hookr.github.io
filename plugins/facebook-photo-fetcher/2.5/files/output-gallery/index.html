  1. <?php 
  2. /** 
  3.  * This file handles the creation of a Facebook Gallery. 
  4.  * When a page containing the "magic tag" is saved, this function will: 
  5.  *   -Fetch the album from facebook 
  6.  *   -Fill in the album content between the tags, formatting it based on the parameters in the tag 
  7.  *   -Add postmeta "_fpf_album_size" with the number of items in the album (which the user can optionally reference) 
  8.  *   -Add postmeta "_fpf_album_cover" with the Facebook URL of the cover photo, if found 
  9.  *    
  10.  * Re-saving the same page will re-fetch the album from facebook and regenerate its content again.   
  11.  */ 
  12. add_action('wp_insert_post_data', 'fpf_run_main'); 
  13. function fpf_run_main($data) 
  14. { 
  15.   //Don't process REVISIONS (would result in 2 fetches per save) 
  16.   if( $data['post_type'] == 'revision') 
  17.       return $data; 
  18.        
  19.   //Check the content for our magic tag (and parse out everything we need if found) 
  20.   $parsed_content = fpf_find_tags($data['post_content']); 
  21.   if( !$parsed_content ) return $data; 
  22.        
  23.   //Connect to Facebook and generate the album content 
  24.   $album_content = fpf_fetch_album_content($parsed_content['aid'], $parsed_content); 
  25.    
  26.   //Update the post we're about to save 
  27.   $data['post_content'] = $parsed_content['before'] .  
  28.                           $parsed_content['startTag'] .  
  29.                           $album_content['content'] .  
  30.                           $parsed_content['endTag'] .  
  31.                           $parsed_content['after']; 
  32.    
  33.   //Set postmeta with the album's size and cover photo (can be optionally referenced by the user) 
  34.   //(Note: for some stupid reason, $data doesn't have the ID - we need to parse it out of the guid.) 
  35.   $post_ID = substr(strrchr($data['guid'], '='), 1); 
  36.   update_post_meta( $post_ID, '_fpf_album_size', $album_content['count'] ); 
  37.   if(isset($album_content['cover'])) update_post_meta( $post_ID, '_fpf_album_cover', $album_content['cover'] ); 
  38.   else                               delete_post_meta($post_ID, '_fpf_album_cover');  
  39.  
  40.   //Done! 
  41.   return $data; 
  42. }

  45. /** 
  46. * Check a post's content for valid "magic tags".  If not found, return 0.  Otherwise, return: 
  47. * $retVal['before']   //Content before the start tag 
  48. * $retVal['after']    //Content after the end tag 
  49. * $retVal['aid']      //The albumID parsed from the start tag 
  50. * $retVal['startTag'] //The complete starttag 
  51. * $retVal['endTag']   //The complete endTag 
  52. * $retVal[....]       //Additional supported parameters found in the startTag. 
  53. *                     //For a full list of what's available see fpf_fetch_album_content(). 
  54. */ 
  55. function fpf_find_tags($post_content) 
  56. {  
  57.   //Start by splitting the content at startTag, and check for "none" or "too many" occurrences 
  58.   global $fpf_identifier; 
  59.   $result = preg_split("/(\<!--[ ]*".$fpf_identifier."[ ]*?([\d_-]+).*?--\>)/", $post_content, -1, PREG_SPLIT_DELIM_CAPTURE ); 
  60.   if( count($result) < 4 )            //No tags found 
  61.       return 0; 
  62.   if( count($result) > 4 )            //Too many tags found 
  63.   { 
  64.       echo __("Sorry, this plugin currently supports only one Facebook gallery per page.", 'facebook-photo-fetcher') . "<br />"; 
  65.       return 0; 
  66.   } 
  67.   $retVal = Array(); 
  68.   $retVal['before'] = $result[0]; 
  69.   $retVal['startTag'] = $result[1]; 
  70.   $retVal['aid'] = $result[2]; 
  71.   $retVal['after'] = $result[3]; 
  72.    
  73.   //Now search the remaining content and split it at the endTag, again checking for "none" or "too many" 
  74.   $result = preg_split("/(\<!--[ ]*\/".$fpf_identifier."[ ]*--\>)/", $retVal['after'], -1, PREG_SPLIT_DELIM_CAPTURE); 
  75.   if( count($result) < 3 ) 
  76.   { 
  77.       echo __("Missing gallery end-tag.", 'facebook-photo-fetcher') . "<br />"; 
  78.       return 0; 
  79.   } 
  80.   if( count($result) > 3 ) 
  81.   { 
  82.       echo __("Duplicate gallery end-tag found.", 'facebook-photo-fetcher') . "<br />"; 
  83.       return 0; 
  84.   } 
  85.   $retVal['endTag'] = $result[1]; 
  86.   $retVal['after'] = $result[2]; 
  87.    
  88.   //Check for optional params in the startTag: 
  89.   if( preg_match('/cols=(\d+)/', $retVal['startTag'], $matches) )     $retVal['cols'] = $matches[1]; 
  90.   if( preg_match('/start=(\d+)/', $retVal['startTag'], $matches) )    $retVal['start'] = $matches[1]; 
  91.   if( preg_match('/max=(\d+)/', $retVal['startTag'], $matches) )      $retVal['max'] = $matches[1]; 
  92.   if( preg_match('/swapHead=(\d+)/', $retVal['startTag'], $matches) ) $retVal['swapHead'] = $matches[1]?true:false; 
  93.   if( preg_match('/hideHead=(\d+)/', $retVal['startTag'], $matches) ) $retVal['hideHead'] = $matches[1]?true:false; 
  94.   if( preg_match('/hideDesc=(\d+)/', $retVal['startTag'], $matches) ) $retVal['hideDesc'] = $matches[1]?true:false; 
  95.   if( preg_match('/hideCaps=(\d+)/', $retVal['startTag'], $matches) ) $retVal['hideCaps'] = $matches[1]?true:false; 
  96.   if( preg_match('/noLB=(\d+)/', $retVal['startTag'], $matches) )     $retVal['noLB'] = $matches[1]?true:false; 
  97.   if( preg_match('/hideCred=(\d+)/', $retVal['startTag'], $matches) ) $retVal['hideCred'] = $matches[1]?true:false; 
  98.   if( preg_match('/rand=(\d+)/', $retVal['startTag'], $matches) )     $retVal['rand'] = $matches[1]; 
  99.   if( preg_match('/orderby=(\w+)/', $retVal['startTag'], $matches) )  $retVal['orderby'] = $matches[1]; 
  100.   return apply_filters('fpf_parse_params', $retVal); 
  101. }

  105. /** 
  106. * Given a Facebook AlbumID, fetch its content and return: 
  107. * $retVal['content'] - The generated HTML content we'll use to display the album 
  108. * $retVal['cover']   - The Facebook album's cover photo (if set) 
  109. * $retVal['count']   - The number of SHOWN photos in the album 
  110. *  
  111. * $params is a array of extra options, parsed from the startTag by fpf_find_tags(). 
  112. * For a list of supported options and their meanings see the $defaults array below.    
  113. */ 
  114. function fpf_fetch_album_content($aid, $params) 
  115. { 
  116.   //Combine optional parameters with default values 
  117.   global $fpf_homepage; 
  118.   $defaults = array('cols' => 4, //Number of columns of images (aka Number of images per row) 
  119.                     'start' => 0, //The first photo index to show (aka skip some initially) 
  120.                     'max' => 99999999999, //The max number of items to show 
  121.                     'swapHead'=> false, //Swap the order of the 2 lines in the album header? 
  122.                     'hideHead'=> false, //Hide the album header entirely? 
  123.                     'hideDesc'=> false, //Hide just the description portion of the header 
  124.                     'hideCaps'=> false, //Hide the per-photo captions on the main listing? 
  125.                     'noLB' => false, //Suppress outputting the lightbox javascript? 
  126.                     'hideCred'=> false, //Omit the "Generated by Facebook Photo Fetcher" footer (please don't :)) 
  127.                     'rand' => false, //Randomly select n photos from the album (or from photos between "start" and "max") 
  128.                     'orderby' => 'normal');       //Can be "normal" or "reverse" (for now) 
  129.   $params = array_merge( apply_filters('fpf_default_albumparams', $defaults), $params ); 
  130.   $itemwidth = $params['cols'] > 0 ? floor(100/$params['cols']) : 100; 
  131.   $itemwidth -= (0.5/$params['cols']); //For stupid IE7, which rounds fractional percentages UP (shave off 0.5%, or the last item will wrap to the next row) 
  132.   $retVal = Array(); 
  133.   $retVal['content'] = ""; 
  134.   $retVal['count'] = 0; 
  135.    
  136.   //Get our saved access token (and make sure it exists)  
  137.   global $fpf_opt_access_token, $fpf_apiver; 
  138.   $access_token = get_option($fpf_opt_access_token); 
  139.   if(!$access_token) 
  140.   { 
  141.       $retVal['content'] = "Error 0: " . __('This plugin does not have a valid Facebook access token.  Please use your admin panel to login with Facebook.', 'facebook-photo-fetcher'); 
  142.       return $retVal; 
  143.   } 
  144.    
  145.   //Try to fetch the album object from Facebook, and check for common errors. 
  146.   $album_fetch_url = "https://graph.facebook.com/$fpf_apiver/$aid?access_token=$access_token&fields=id, cover_photo, count, link, name, from, created_time, description"; 
  147.   $album = fpf_get($album_fetch_url); 
  148.   if(!$album || isset($album->error)) 
  149.   { 
  150.       if(!$album)                       $retVal['content'] = "Error 1: " . __("An unknown error occurred while trying to fetch the album (empty reply).", 'facebook-photo-fetcher'); 
  151.       else if($album->error->code==190) $retVal['content'] = "Error 190: " . __("Invalid OAuth Access Token.  Try using the admin panel to re-validate your plugin.", 'facebook-photo-fetcher'); 
  152.       else if($album->error->code==803) $retVal['content'] = "Error 803: " . __("Your album id doesn't appear to exist.", 'facebook-photo-fetcher'); 
  153.       else if($album->error->code==100) $retVal['content'] = "Error 100: " . __("Your album id doesn't appear to be accessible.", 'facebook-photo-fetcher'); 
  154.       return $retVal; 
  155.   } 
  156.   if(!isset($album->id) || $album->id != $aid) 
  157.   { 
  158.       $retVal['content'] = "Error 2: " . __("An unknown error occurred while trying to fetch the album (id mismatch).", 'facebook-photo-fetcher'); 
  159.       return $retVal; 
  160.   } 
  161.   if(!isset($album->cover_photo) || $album->id != $aid) 
  162.   { 
  163.       $retVal['content'] = "Error 3: " . __("An error occurred while trying to fetch the album: the ID specified does not appear to be an album.", 'facebook-photo-fetcher'); 
  164.       return $retVal; 
  165.   } 
  166.   if($album->count == 0) 
  167.   { 
  168.       $retVal['content'] = "Error 4: " . __("An error occurred while trying to fetch the album: it appears to be empty.", 'facebook-photo-fetcher'); 
  169.       return $retVal; 
  170.   } 
  171.    
  172.   //Now that we know the album is OK, try to fetch its photos.  Note that as of Feb 2014, it seems like Facebook 
  173.   //won't return more than 100 photos, so I'll have to fetch them in paged groups... 
  174.   $photos = Array(); 
  175.   $photoGroupNum = 0; 
  176.   $debugString = "Starting to fetch $album->count photos.\nAlbum: <a href='$album_fetch_url'>$album_fetch_url</a>\n"; 
  177.   $debugPhotoCount = 0; 
  178.   $fetch_url = "https://graph.facebook.com/$fpf_apiver/$aid/photos?access_token=$access_token&limit=9999&fields=name, source, picture"; 
  179.   while(true) 
  180.   { 
  181.       //Fetch this group (as many as FB will give us at once...might not be all of them, even though I specify a limit of 9999) 
  182.       $photosThisGroup = fpf_get($fetch_url); 
  183.        
  184.       //Make sure no error 
  185.       if(!$photosThisGroup || !isset($photosThisGroup->data)) 
  186.       { 
  187.           $retVal['content'] = "Error 5: " . __("An unknown error occurred while trying to fetch the photos (empty data).", 'facebook-photo-fetcher'); 
  188.           return $retVal; 
  189.       } 
  190.        
  191.       //Just for testing... 
  192.       $debugString .= "**********************************************\n"; 
  193.       $debugString .= "Group: $photoGroupNum\nFetch URL: <a href='$fetch_url'>$fetch_url</a>\nItems: " . count($photosThisGroup->data) . "\n"; 
  194.       $debugString .= "**********************************************\n"; 
  195.       foreach($photosThisGroup->data as $photo) $debugString .= ($debugPhotoCount++) . ") <a href='$photo->source'>$photo->source</a>\n";      
  196.       $debugString .= "\n\n"; 
  197.        
  198.       //If we didn't get any back, we must've already fetched all available photos - break out of this loop. 
  199.       //I don't think this should ever happen, but check just in case (to avoid infinite loop) 
  200.       if(count($photosThisGroup->data) == 0) 
  201.       { 
  202.           $debugString .= "--->Done: No results returned."; 
  203.           break; 
  204.       } 
  205.        
  206.       //Likewise - just be sure there's no infinite loop.  I'm pretty sure no album will ever have >2000 photos. 
  207.       if($photoGroupNum >= 20) 
  208.       { 
  209.           $debugString .= "--->Done: Stopped to prevent infinite loop (Limit: 2000 photos)."; 
  210.           break; 
  211.       } 
  212.        
  213.       //Tack these results onto our 'main overall' set of photos 
  214.       $photos = array_merge($photos, $photosThisGroup->data); 
  215.        
  216.       //If we've got the total expected number of photos, we're done - break out of this loop 
  217.       if( count($photos) == $album->count ) 
  218.       { 
  219.           $debugString .= "--->Done: Successfully fetched all " . count($photos) . " photos."; 
  220.           break; 
  221.       } 
  222.        
  223.       //If the next 'paging' url isn't specified, it's telling us there are no more photos available - break out of this loop 
  224.       if(!isset($photosThisGroup->paging->next)) 
  225.       { 
  226.           $debugString .= "--->Done: Paging->next wasn't set."; 
  227.           break;             
  228.       } 
  229.        
  230.       //Otherwise, get the URL to fetch the next group of photos & keep going. 
  231.       $fetch_url = $photosThisGroup->paging->next; 
  232.       $photoGroupNum++; 
  233.   } 
  234.   //echo "<pre>$debugString</pre>"; 
  235.    
  236.   //Sanity check 
  237.   //if(count($photos) != $album->count) $retVal['content'] = "<i>Warning: A size mismatch error occurred while trying to fetch the photos (the album reported $album->count entries, but only " . count($photos) . " were returned).</i><br />"; 
  238.    
  239.   //Run filters so we can modify the album and photo data 
  240.   $album = apply_filters('fpf_album_data', $album ); 
  241.   $photos = apply_filters('fpf_photos_presort', $photos ); 
  242.    
  243.   //Store the filename of the album cover 
  244.   //We must do this here, prior to slicing down the array of photos. 
  245.   if( isset($album->cover_photo) ) 
  246.   { 
  247.       foreach($photos as $photo) 
  248.       { 
  249.           if( strcmp($photo->id, $album->cover_photo->id) == 0 ) 
  250.               $retVal['cover'] = $photo->source; 
  251.       } 
  252.   } 
  253.    
  254.   //Reorder the photos if necessary 
  255.   if( $params['orderby'] == 'reverse' ) 
  256.   { 
  257.       $photos = array_reverse($photos); 
  258.   } 
  259.    
  260.   //Slice the photo array as necessary 
  261.   if( count($photos) > 0 ) 
  262.   { 
  263.       //Slice the photos between "start" and "max" 
  264.       if( $params['start'] > $album->count ) 
  265.       { 
  266.           $retVal['content'] .= "Error 6: " . sprintf(__("Start index %s is greater than the total number of photos in this album; Defaulting to 0.", 'facebook-photo-fetcher'), $params['start']) . "<br /><br />"; 
  267.           $params['start'] = 0; 
  268.       } 
  269.       if( $params['max'] > $album->count - $params['start'] ) 
  270.           $params['max'] = $album->count - $params['start']; 
  271.       $photos = array_slice($photos, $params['start'], $params['max']);  
  272.        
  273.       //If "rand" is specified, randomize the order and slice again 
  274.       if( $params['rand'] ) 
  275.       { 
  276.           shuffle($photos); 
  277.           $photos = array_slice($photos, 0, $params['rand']); 
  278.       } 
  279.   } 
  280.    
  281.   //Run a filter so addons can modify/process the photos 
  282.   $photos = apply_filters('fpf_photos_postsort', $photos ); 
  283.    
  284.   //Create a header with some info about the album 
  285.   $retVal['count'] = count($photos); 
  286.   if(!$params['hideHead']) 
  287.   { 
  288.       /** translators: This is what displays "From (album name), " above imported albums.*/ 
  289.       $headerTitle = sprintf(__('From %s.', 'facebook-photo-fetcher'), '<a href="' . htmlspecialchars($album->link) . '">' . $album->name . '</a>'); 
  290.       if( isset($album->from->id) && isset($album->created_time) ) 
  291.       { 
  292.           /** translators: This is what displays the Facebook user & album date, above imported albums (i.e. "posted by John Smith on 1/1/2001")*/ 
  293.           $headerTitle .= ' ' . sprintf(__('Posted by %s on %s', 'facebook-photo-fetcher'), '<a href="http://www.facebook.com/profile.php?id=' . $album->from->id . '">' . $album->from->name . '</a>', date('n/d/Y', strtotime($album->created_time))); 
  294.       } 
  295.       /** translators: Shows the photo count in the header above imported albums*/ 
  296.       if( $retVal['count'] < $album->count) $headerTitle .= ' (' . sprintf(__('Showing %s of %s items', 'facebook-photo-fetcher'), $retVal['count'], $album->count).")\n"; 
  297.       /** translators: Shows the photo count in the header above imported albums*/ 
  298.       else                                  $headerTitle .= ' (' . sprintf(__('%s items', 'facebook-photo-fetcher'), $retVal['count']) . ")\n"; 
  299.       $headerTitle .= '<br /><br />'; 
  300.        
  301.       if(!$params['hideDesc']) 
  302.       {             
  303.           if( isset($album->description) ) $headerDesc = '"'.$album->description.'"<br /><br />'."\n"; 
  304.           else                             $headerDesc = ""; 
  305.       } 
  306.   }  
  307.  
  308.   //Output the album!  Starting with a (hidden) timestamp, then the header, then each photo. 
  309.   global $fpf_version; 
  310.   $retVal['content'] .= "<!-- ID ". $aid ." Last fetched on " . date('m/d/Y H:i:s') . " v$fpf_version-->\n"; 
  311.   if( $params['swapHead'] )   $retVal['content'] .= $headerTitle . $headerDesc; 
  312.   else                        $retVal['content'] .= $headerDesc . $headerTitle;  
  313.   $retVal['content'] .= "<div class='gallery fpf-gallery'>\n"; 
  314.   $i = 0; 
  315.   foreach($photos as $photo) 
  316.   { 
  317.       //Strip [], or WP will try to run it as shortcode 
  318.       if(!isset($photo->name)) $photo->name = ""; 
  319.       $caption = preg_replace("/\[/", "(", $photo->name); 
  320.       $caption = preg_replace("/\]/", ")", $caption); 
  321.        
  322.       //Strip emoji. 
  323.       //Emoji come from FB as surrogate pairs (like "\udbb8\udf2c"), which get converted to UTF8 when we json_decode() the string (see http://stackoverflow.com/questions/17445901/replace-iphone-emoji-in-html-page) 
  324.       //First, strip these (http://apps.timwhitlock.info/emoji/tables/unicode) (Code from: http://stackoverflow.com/questions/12807176/php-writing-a-simple-removeemoji-function) 
  325.       $caption = preg_replace('/[\x{1F600}-\x{1F64F}]/u', '', $caption); 
  326.       $caption = preg_replace('/[\x{1F300}-\x{1F5FF}]/u', '', $caption); 
  327.       $caption = preg_replace('/[\x{1F680}-\x{1F6FF}]/u', '', $caption); 
  328.       //And here are some more (This was the range that was messing up Mark Laurich's albums: https://github.com/adamrocker/Japanese-Mobile-Emoji/blob/master/EmojiData.csv) 
  329.       $caption = preg_replace('/[\x{FE000}-\x{FE4E4}]/u', '', $caption);  
  330.  
  331.       //Output this photo 
  332.       $caption = preg_replace("/\r/", "", $caption); 
  333.       $caption_with_br = htmlspecialchars(preg_replace("/\n/", "<br />", $caption)); 
  334.       $caption_no_br = htmlspecialchars(preg_replace("/\n/", " ", $caption)); 
  335.       if ($caption_with_br != '') 
  336.           $link = '<a rel="' . htmlspecialchars($album->link) . '" class="fbPhoto" href="'. htmlspecialchars($photo->source) . '" title="'.$caption_with_br.' " ><img src="' . htmlspecialchars($photo->picture) . '" alt="" /></a>'; 
  337.       else 
  338.           $link = '<a rel="' . htmlspecialchars($album->link) . '" class="fbPhoto" href="'. htmlspecialchars($photo->source) . '"><img src="' . htmlspecialchars($photo->picture) . '" alt="" /></a>'; 
  339.       $retVal['content'] .= "<dl class='gallery-item' style=\"width:$itemwidth%\">"; 
  340.       $retVal['content'] .= "<dt class='gallery-icon'>$link</dt>"; 
  341.       if(!$params['hideCaps']) 
  342.       { 
  343.           $retVal['content'] .= "<dd class='gallery-caption'>"; 
  344.           $retVal['content'] .= mb_substr($caption_no_br, 0, 85) . (strlen($caption_no_br)>85?"...":""); 
  345.           $retVal['content'] .= "</dd>"; 
  346.       } 
  347.       $retVal['content'] .= "</dl>\n"; 
  348.        
  349.       //Move on to the next row? 
  350.       if( $params['cols'] > 0 && ++$i % $params['cols'] == 0 ) $retVal['content'] .= "<br style=\"clear: both\" />\n\n"; 
  351.   } 
  352.   if( $i%$params['cols'] != 0 ) $retVal['content'] .= "<br style=\"clear: both\" />\n\n"; 
  353.   $retVal['content'] .= "</div>\n"; 
  354.   if( !$params['hideCred'] )    $retVal['content'] .= "<span class=\"fpfcredit\">" . sprintf(__("Generated by %s", 'facebook-photo-fetcher'), "<i>Facebook Photo Fetcher 2</i>") . "</span>\n"; 
  355.    
  356.   //Activate the lightbox when the user clicks a photo (only if the Lightbox plugin isn't already there) 
  357.   if( !$params['noLB'] && !function_exists('lightbox_2_options_page') ) 
  358.   { 
  359.       $retVal['content'] .= '<script type="text/javascript"> 
  360.       jQuery(document).ready(function() { 
  361.           jQuery("a[rel*=\''.$aid.'\']").fancybox({ 
  362.               "transitionIn"    : "elastic",  
  363.               "transitionOut"    : "elastic",  
  364.               "titlePosition" : "inside",  
  365.               "titleFormat"    : function(title, currentArray, currentIndex, currentOpts) 
  366.               { 
  367.                   return "<span id=\'fancybox-title-over\' style=\'background-image:none; text-align:left;\'>" + (title.length ? title : "") + "</span>"; 
  368.               } 
  369.           }); 
  370.       });'. 
  371.       "\n</script>\n"; 
  372.   } 
  373.   $retVal['content'] .= "<!-- End Album ". $aid ." -->\n"; 
  374.   return $retVal; 
  375. }
