  119. /** 
  120.  * Conditionally flushes rewrite rules if we have reason to. 
  121.  * 
  122.  * @since 1.3.0 
  123.  */ 
  124. function cptui_flush_rewrite_rules() { 
  125.  
  126.   if ( defined( 'DOING_AJAX' ) && DOING_AJAX ) { 
  127.       return; 
  128.   } 
  129.  
  130.   /** 
  131.    * Wise men say that you should not do flush_rewrite_rules on init or admin_init. Due to the nature of our plugin 
  132.    * and how new post types or taxonomies can suddenly be introduced, we need to...potentially. For this,  
  133.    * we rely on a short lived transient. Only 5 minutes life span. If it exists, we do a soft flush before 
  134.    * deleting the transient to prevent subsequent flushes. The only times the transient gets created, is if 
  135.    * post types or taxonomies are created, updated, deleted, or imported. Any other time and this condition 
  136.    * should not be met. 
  137.    */ 
  138.   if ( 'true' === ( $flush_it = get_transient( 'cptui_flush_rewrite_rules' ) ) ) { 
  139.       flush_rewrite_rules( false ); 
  140.       // So we only run this once. 
  141.       delete_transient( 'cptui_flush_rewrite_rules' ); 
  142.   } 
  143. } 
  144. add_action( 'admin_init', 'cptui_flush_rewrite_rules' );

  666. /** 
  667.  * Mark site as not a new CPTUI install upon update to 1.5.0 
  668.  * 
  669.  * @since 1.5.0 
  670.  * 
  671.  * @param object $wp_upgrader WP_Upgrader instance. 
  672.  * @param array  $extras      Extra information about performed upgrade. 
  673.  */ 
  674. function cptui_not_new_install( $wp_upgrader, $extras ) { 
  675.  
  676.   if ( ! is_a( $wp_upgrader, 'Plugin_Upgrader' ) ) { 
  677.       return; 
  678.   } 
  679.  
  680.   if ( ! array_key_exists( 'plugins', $extras ) || ! is_array( $extras['plugins'] ) ) { 
  681.       return; 
  682.   } 
  683.  
  684.   // Was CPTUI updated? 
  685.   if ( ! in_array( 'custom-post-type-ui/custom-post-type-ui.php', $extras['plugins'] ) ) { 
  686.       return; 
  687.   } 
  688.  
  689.   // If we are already known as not new, return. 
  690.   if ( cptui_is_new_install() ) { 
  691.       return; 
  692.   } 
  693.  
  694.   // We need to mark ourselves as not new. 
  695.   cptui_set_not_new_install(); 
  696. } 
  697. add_action( 'upgrader_process_complete', 'cptui_not_new_install', 10, 2 ); 
  698.  
  699. /** 
  700.  * Check whether or not we're on a new install. 
  701.  * 
  702.  * @since 1.5.0 
  703.  * 
  704.  * @return bool 
  705.  */ 
  706. function cptui_is_new_install() { 
  707.   $new_or_not = true; 
  708.   $saved = get_option( 'cptui_new_install', '' ); 
  709.  
  710.   if ( 'false' === $saved ) { 
  711.       $new_or_not = false; 
  712.   } 
  713.  
  714.   /** 
  715.    * Filters the new install status. 
  716.    * 
  717.    * Offers third parties the ability to override if they choose to. 
  718.    * 
  719.    * @since 1.5.0 
  720.    * 
  721.    * @param bool $new_or_not Whether or not site is a new install. 
  722.    */ 
  723.   return (bool) apply_filters( 'cptui_is_new_install', $new_or_not ); 
  724. }

  769. /** 
  770.  * Checks if a requested post type has a custom CPTUI feature supported. 
  771.  * 
  772.  * @since 1.5.0
