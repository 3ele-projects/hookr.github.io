  62.   function add_global_discount_grouped( $price, $qty, $_product ) { 
  63.       if ( $_product->is_type( 'grouped' ) ) { 
  64.           foreach ( $_product->get_children() as $child_id ) { 
  65.               $the_price = get_post_meta( $child_id, '_price', true ); 
  66.               $the_product = wc_get_product( $child_id ); 
  67.               $the_price = wcj_get_product_display_price( $the_product, $the_price, 1 ); 
  68.               if ( $the_price == $price ) { 
  69.                   return $this->add_global_discount_price( $price, $the_product ); 
  70.               } 
  71.           } 
  72.       } 
  73.       return $price; 
  74.   }

  79.    * @version 2.5.7

  82.   function calculate_price( $price, $coefficient, $group ) { 
  83.       $return_price = ( 'percent' === get_option( 'wcj_global_discount_sale_coefficient_type_' . $group, 'percent' ) ) ? 
  84.           ( $price + $price * ( $coefficient / 100 ) ) : 
  85.           ( $price + $coefficient ); 
  86.       return ( $return_price >= 0 ) ? $return_price : 0; 
  87.   }

  96.   function check_if_applicable( $_product, $group ) { 
  97.       return ( 'yes' === get_option( 'wcj_global_discount_sale_enabled_' . $group, 'yes' ) && $this->check_product_categories( $_product, $group ) ); 
  98.   }

  104.    * @since   2.5.7

  106.    */ 
  107.   function check_product_categories( $_product, $group ) { 
  108.       // Check product category - include 
  109.       $categories_in = get_option( 'wcj_global_discount_sale_categories_incl_' . $group, '' ); 
  110.       if ( ! empty( $categories_in ) ) { 
  111.           $product_categories = get_the_terms( wcj_get_product_id_or_variation_parent_id( $_product ), 'product_cat' ); 
  112.           if ( empty( $product_categories ) ) { 
  113.               return false; // option set to some categories, but product has no categories 
  114.           } 
  115.           foreach( $product_categories as $product_category ) { 
  116.               if ( in_array( $product_category->term_id, $categories_in ) ) { 
  117.                   return true; // category found 
  118.               } 
  119.           } 
  120.           return false; // no categories found 
  121.       } 
  122.       return true; // option not set (i.e. left blank) 
  123.   }

  131.   function add_global_discount_any_price( $price_type, $price, $_product ) { 
  132.       $total_number = apply_filters( 'booster_get_option', 1, get_option( 'wcj_global_discount_groups_total_number', 1 ) ); 
  133.       for ( $i = 1; $i <= $total_number; $i++ ) { 
  134.           if ( ! $this->check_if_applicable( $_product, $i ) ) { 
  135.               continue; // no changes by current discount group 
  136.           } 
  137.           $coefficient = get_option( 'wcj_global_discount_sale_coefficient_' . $i, 0 ); 
  138.           if ( 0 != $coefficient ) { 
  139.               if ( 'sale_price' === $price_type ) { 
  140.                   if ( 0 == $price ) { 
  141.                       // The product is currently not on sale 
  142.                       if ( 'only_on_sale' === get_option( 'wcj_global_discount_sale_product_scope_' . $i, 'all' ) ) { 
  143.                           continue; // no changes by current discount group 
  144.                       } else { 
  145.                           $price = $_product->get_regular_price(); 
  146.                       } 
  147.                   } else { 
  148.                       // The product is currently on sale 
  149.                       if ( 'only_not_on_sale' === get_option( 'wcj_global_discount_sale_product_scope_' . $i, 'all' ) ) { 
  150.                           continue; // no changes by current discount group 
  151.                       } 
  152.                   } 
  153.               } else { // if ( 'price' === $price_type ) 
  154.                   remove_filter( WCJ_PRODUCT_GET_SALE_PRICE_FILTER, array( $this, 'add_global_discount_sale_price' ), PHP_INT_MAX, 2 ); 
  155.                   remove_filter( 'woocommerce_variation_prices_sale_price', array( $this, 'add_global_discount_sale_price' ), PHP_INT_MAX, 2 ); 
  156.                   if ( ! WCJ_IS_WC_VERSION_BELOW_3 ) { 
  157.                       remove_filter( 'woocommerce_product_variation_get_sale_price', array( $this, 'add_global_discount_sale_price' ), PHP_INT_MAX, 2 ); 
  158.                   } 
  159.                   if ( 'only_on_sale' === get_option( 'wcj_global_discount_sale_product_scope_' . $i, 'all' ) && 0 == $_product->get_sale_price() ) { 
  160.                       continue; // no changes by current discount group 
  161.                   } elseif ( 'only_not_on_sale' === get_option( 'wcj_global_discount_sale_product_scope_' . $i, 'all' ) && 0 != $_product->get_sale_price() ) { 
  162.                       continue; // no changes by current discount group 
  163.                   } 
  164.                   add_filter( WCJ_PRODUCT_GET_SALE_PRICE_FILTER, array( $this, 'add_global_discount_sale_price' ), PHP_INT_MAX, 2 ); 
  165.                   add_filter( 'woocommerce_variation_prices_sale_price', array( $this, 'add_global_discount_sale_price' ), PHP_INT_MAX, 2 ); 
  166.                   if ( ! WCJ_IS_WC_VERSION_BELOW_3 ) { 
  167.                       add_filter( 'woocommerce_product_variation_get_sale_price', array( $this, 'add_global_discount_sale_price' ), PHP_INT_MAX, 2 ); 
  168.                   } 
  169.               } 
  170.               return $this->calculate_price( $price, $coefficient, $i ); // discount applied 
  171.           } 
  172.       } 
  173.       return $price; // no changes 
  174.   }
