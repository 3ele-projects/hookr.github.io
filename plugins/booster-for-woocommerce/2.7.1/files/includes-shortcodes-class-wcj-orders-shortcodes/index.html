  379.       $total = 0; 
  380.       $the_items = $this->the_order->get_items(); 
  381.       foreach ( $the_items as $item_id => $item ) { 
  382.           $product_id = ( 0 != $item['variation_id'] ) ? $item['variation_id'] : $item['product_id']; 
  383.           $_product = wc_get_product( $product_id ); 
  384.           if ( $_product ) { 
  385.               switch ( $param ) { 
  386.                   case 'width': 
  387.                       $total += ( $item['qty'] * $_product->get_width() ); 
  388.                       break; 
  389.                   case 'height': 
  390.                       $total += ( $item['qty'] * $_product->get_height() ); 
  391.                       break; 
  392.                   case 'length': 
  393.                       $total += ( $item['qty'] * $_product->get_length() ); 
  394.                       break; 
  395.                   case 'weight': 
  396.                       $total += ( $item['qty'] * $_product->get_weight() ); 
  397.                       break; 
  398.               } 
  399.           } 
  400.       } 
  401.       return ( 0 == $total && 'yes' === $atts['hide_if_zero'] ) ? '' : $total; 
  402.   }

  467.   function wcj_order_items( $atts ) { 
  468.       $items = array(); 
  469.       $the_items = $this->the_order->get_items(); 
  470.       foreach ( $the_items as $item_id => $item ) { 
  471.           switch ( $atts['field'] ) { 
  472.               case 'debug': 
  473.                   $items[] = '<pre>' . print_r( $item, true ) . '</pre>'; 
  474.                   break; 
  475.               default: // case 'name' etc. 
  476.                   $items[] = ( isset( $item[ $atts['field'] ] ) ) ? $item[ $atts['field'] ] : ''; 
  477.                   break; 
  478.           } 
  479.       } 
  480.       if ( empty( $items ) ) { 
  481.           return ''; 
  482.       } 
  483.       if ( 'all' === $atts['item_number'] ) { 
  484.           return implode( $atts['sep'], $items ); 
  485.       } else { 
  486.           switch ( $atts['item_number'] ) { 
  487.               case 'first': 
  488.                   return current( $items ); 
  489.               case 'last': 
  490.                   return end( $items ); 
  491.               default: 
  492.                   $item_number = intval( $atts['item_number'] ) - 1; 
  493.                   if ( $item_number < 0 ) { 
  494.                       $item_number = 0; 
  495.                   } elseif ( $item_number >= count( $items ) ) { 
  496.                       $item_number = count( $items ) - 1; 
  497.                   } 
  498.                   return $items[ $item_number ]; 
  499.           } 
  500.       } 
  501.   }
