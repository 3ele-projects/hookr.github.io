  486.   /** 
  487.    * is_visible. 
  488.    * 
  489.    * @version 2.6.0 
  490.    * @since   2.6.0 
  491.    */ 
  492.   function is_visible( $i ) { 
  493.  
  494.       if ( apply_filters( 'wcj_checkout_custom_field_always_visible_on_empty_cart', false ) && WC()->cart->is_empty() ) { 
  495.           // Added for "One Page Checkout" plugin compatibility. 
  496.           return true; 
  497.       } 
  498.  
  499.       // Checking categories 
  500.       $categories_in = get_option( 'wcj_checkout_custom_field_categories_in_' . $i ); 
  501.       if ( ! empty( $categories_in ) ) { 
  502.           foreach ( WC()->cart->get_cart() as $cart_item_key => $values ) { 
  503.               $product_categories = get_the_terms( $values['product_id'], 'product_cat' ); 
  504.               if ( empty( $product_categories ) ) { 
  505.                   continue; 
  506.               } 
  507.               foreach( $product_categories as $product_category ) { 
  508.                   if ( in_array( $product_category->term_id, $categories_in ) ) { 
  509.                       return true; 
  510.                   } 
  511.               } 
  512.           } 
  513.           return false; 
  514.       } 
  515.  
  516.       // Checking products 
  517.       $products_in = get_option( 'wcj_checkout_custom_field_products_in_' . $i ); 
  518.       if ( ! empty( $products_in ) ) { 
  519.           foreach ( WC()->cart->get_cart() as $cart_item_key => $values ) { 
  520.               if ( in_array( $values['product_id'], $products_in ) ) { 
  521.                   return true; 
  522.               } 
  523.           } 
  524.           return false; 
  525.       } 
  526.  
  527.       // There were nothing to check 
  528.       return true; 
  529.   }

  534.    * @version 2.6.0

  536.   public function add_custom_checkout_fields( $fields ) { 
  537.  
  538.       for ( $i = 1; $i <= apply_filters( 'booster_get_option', 1, get_option( 'wcj_checkout_custom_fields_total_number', 1 ) ); $i++ ) {

  544.               } 
  545.  
  546.               $the_type = get_option( 'wcj_checkout_custom_field_type_' . $i ); 
  547.               $custom_attributes = array(); 
  548.               if ( 'datepicker' === $the_type || 'weekpicker' === $the_type || 'timepicker' === $the_type || 'number' === $the_type ) { 
  549.                   if ( 'datepicker' === $the_type || 'weekpicker' === $the_type ) {

  555.                       $custom_attributes['maxdate'] = get_option( 'wcj_checkout_custom_field_datepicker_maxdate_' . $i, 365 ); 
  556.                       $custom_attributes['firstday'] = get_option( 'wcj_checkout_custom_field_datepicker_firstday_' . $i, 0 ); 
  557.                       if ( 'yes' === get_option( 'wcj_checkout_custom_field_datepicker_changeyear_' . $i, 'yes' ) ) {

  559.                           $custom_attributes['yearrange'] = get_option( 'wcj_checkout_custom_field_datepicker_yearrange_' . $i, 'c-10:c+10' ); 
  560.                       } 
  561.                       $custom_attributes['display'] = ( 'datepicker' === $the_type ) ? 'date' : 'week'; 
  562.                   } elseif ( 'timepicker' === $the_type ) { 
  563.                       $custom_attributes['timeformat'] = get_option( 'wcj_checkout_custom_field_timepicker_format_' . $i, 'hh:mm p' ); 
  564.                       $custom_attributes['interval'] = get_option( 'wcj_checkout_custom_field_timepicker_interval_' . $i, 15 ); 
  565.                       $custom_attributes['display'] = 'time'; 
  566.                   } else/** if ( 'number' === $the_type ) */ { 
  567.                       $custom_attributes['display'] = $the_type; 
  568.                   } 
  569.                   $the_type = 'text'; 
  570.               } 
  571.               $the_section = get_option( 'wcj_checkout_custom_field_section_' . $i ); 
  572.               $the_key = 'wcj_checkout_field_' . $i; 
  573.  
  574.               $the_field = array( 
  575.                   'type' => $the_type,  
  576.                   'label' => get_option( 'wcj_checkout_custom_field_label_' . $i ),  
  577.                   'placeholder' => get_option( 'wcj_checkout_custom_field_placeholder_' . $i ),  
  578.                   'required' => ( 'yes' === get_option( 'wcj_checkout_custom_field_required_' . $i ) ) ? true : false,  
  579.                   'custom_attributes' => $custom_attributes,  
  580.                   'clear' => ( 'yes' === get_option( 'wcj_checkout_custom_field_clear_' . $i ) ) ? true : false,  
  581.                   'class' => array( get_option( 'wcj_checkout_custom_field_class_' . $i ), ),

  585.                   $select_options_raw = get_option( 'wcj_checkout_custom_field_select_options_' . $i ); 
  586.                   $select_options = wcj_get_select_options( $select_options_raw ); 
  587.                   if ( 'select' === $the_type ) { 
  588.                       $placeholder = get_option( 'wcj_checkout_custom_field_placeholder_' . $i ); 
  589.                       if ( '' != $placeholder ) { 
  590.                           $select_options = array_merge( array( '' => $placeholder ), $select_options ); 
  591.                       } 
  592.                   } 
  593.                   $the_field['options'] = $select_options; 
  594.                   if ( ! empty( $select_options ) ) { 
  595.                       reset( $select_options ); 
  596.                       $the_field['default'] = key( $select_options ); 
  597.                   } 
  598.               } 
  599.  
  600.               if ( 'checkbox' === $the_type ) { 
  601.                   $the_field['default'] = ( 'yes' === get_option( 'wcj_checkout_custom_field_checkbox_default_' . $i ) ) ? 1 : 0; 
  602.               } 
  603.  
  604.               $fields[ $the_section ][ $the_section . '_' . $the_key ] = $the_field; 
  605.           } 
  606.       } 
  607.       return $fields; 
  608.   }
