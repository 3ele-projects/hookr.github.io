  263.   function validate_open_price_on_add_to_cart( $passed, $product_id ) { 
  264.       $the_product = wc_get_product( $product_id ); 
  265.       if ( $this->is_open_price_product( $the_product ) ) { 
  266.           $min_price = get_post_meta( $product_id, '_' . 'wcj_product_open_price_min_price', true ); 
  267.           $max_price = get_post_meta( $product_id, '_' . 'wcj_product_open_price_max_price', true ); 
  268.           if ( $min_price > 0 ) { 
  269.               if ( ! isset( $_POST['wcj_open_price'] ) || '' === $_POST['wcj_open_price'] ) { 
  270.                   wc_add_notice( get_option( 'wcj_product_open_price_messages_required', __( 'Price is required!', 'woocommerce-jetpack' ) ), 'error' ); 
  271.                   return false; 
  272.               } 
  273.               if ( $_POST['wcj_open_price'] < $min_price ) { 
  274.                   wc_add_notice( get_option( 'wcj_product_open_price_messages_to_small', __( 'Entered price is too small!', 'woocommerce-jetpack' ) ), 'error' ); 
  275.                   return false; 
  276.               } 
  277.           } 
  278.           if ( $max_price > 0 ) { 
  279.               if ( isset( $_POST['wcj_open_price'] ) && $_POST['wcj_open_price'] > $max_price ) { 
  280.                   wc_add_notice( get_option( 'wcj_product_open_price_messages_to_big', __( 'Entered price is too big!', 'woocommerce-jetpack' ) ), 'error' ); 
  281.                   return false; 
  282.               } 
  283.           } 
  284.       } 
  285.       return $passed; 
  286.   }
