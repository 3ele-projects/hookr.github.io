  73.               } 
  74.               elseif ( 'fixed' === $discount_type ) {

  76.               } else { 
  77.                   $discount = $discount . '%';

  98.   private function get_discount_by_quantity( $quantity, $product_id ) { 
  99.  
  100.       // Check for user role options 
  101.       $role_option_name_addon = ''; 
  102.       $user_roles = get_option( 'wcj_wholesale_price_by_user_role_roles', '' ); 
  103.       if ( ! empty( $user_roles ) ) { 
  104.           $current_user_role = wcj_get_current_user_first_role(); 
  105.           foreach ( $user_roles as $user_role_key ) { 
  106.               if ( $current_user_role === $user_role_key ) { 
  107.                   $role_option_name_addon = '_' . $user_role_key; 
  108.                   break; 
  109.               } 
  110.           } 
  111.       } 
  112.  
  113.       // Get discount 
  114.       $max_qty_level = 1; 
  115.       $discount = 0; 
  116.       if ( wcj_is_product_wholesale_enabled_per_product( $product_id ) ) { 
  117.           for ( $i = 1; $i <= apply_filters( 'booster_get_option', 1, get_post_meta( $product_id, '_' . 'wcj_wholesale_price_levels_number' . $role_option_name_addon, true ) ); $i++ ) { 
  118.               $level_qty = get_post_meta( $product_id, '_' . 'wcj_wholesale_price_level_min_qty' . $role_option_name_addon . '_' . $i, true ); 
  119.               if ( $quantity >= $level_qty && $level_qty >= $max_qty_level ) { 
  120.                   $max_qty_level = $level_qty; 
  121.                   $discount = get_post_meta( $product_id, '_' . 'wcj_wholesale_price_level_discount' . $role_option_name_addon . '_' . $i, true ); 
  122.               } 
  123.           } 
  124.       } else { 
  125.           for ( $i = 1; $i <= apply_filters( 'booster_get_option', 1, get_option( 'wcj_wholesale_price_levels_number' . $role_option_name_addon, 1 ) ); $i++ ) { 
  126.               $level_qty = get_option( 'wcj_wholesale_price_level_min_qty' . $role_option_name_addon . '_' . $i, PHP_INT_MAX ); 
  127.               if ( $quantity >= $level_qty && $level_qty >= $max_qty_level ) { 
  128.                   $max_qty_level = $level_qty; 
  129.                   $discount = get_option( 'wcj_wholesale_price_level_discount_percent' . $role_option_name_addon . '_' . $i, 0 ); 
  130.               } 
  131.           } 
  132.       } 
  133.  
  134.       return $discount; 
  135.   }

  142.   private function get_wholesale_price( $price, $quantity, $product_id ) { 
  143.       $discount = $this->get_discount_by_quantity( $quantity, $product_id ); 
  144.       $discount_type = ( wcj_is_product_wholesale_enabled_per_product( $product_id ) ) 
  145.           ? get_post_meta( $product_id, '_' . 'wcj_wholesale_price_discount_type', true ) 
  146.           : get_option( 'wcj_wholesale_price_discount_type', 'percent' ); 
  147.       if ( 'price_directly' === $discount_type ) { 
  148.           return ( 0 != $discount ) ? apply_filters( 'wcj_get_wholesale_price', $discount, $product_id ) : $price; 
  149.       } elseif ( 'percent' === $discount_type ) { 
  150.           return $price * ( 1.0 - ( $discount / 100.0 ) ); 
  151.       } else { 
  152.           $discounted_price = $price - $discount; 
  153.           return ( $discounted_price >= 0 ) ? $discounted_price : 0; 
  154.       } 
  155.   }

  177.   function calculate_totals( $cart ) { 
  178.  
  179.       foreach ( $cart->cart_contents as $item_key => $item ) { 
  180.  
  181.           if ( isset( WC()->cart->cart_contents[ $item_key ]['data']->wcj_wholesale_price ) ) { 
  182.               unset( WC()->cart->cart_contents[ $item_key ]['data']->wcj_wholesale_price ); 
  183.           } 
  184.           if ( isset( WC()->cart->cart_contents[ $item_key ]['wcj_wholesale_price'] ) ) { 
  185.               unset( WC()->cart->cart_contents[ $item_key ]['wcj_wholesale_price'] ); 
  186.           } 
  187.           if ( isset( WC()->cart->cart_contents[ $item_key ]['wcj_wholesale_price_old'] ) ) { 
  188.               unset( WC()->cart->cart_contents[ $item_key ]['wcj_wholesale_price_old'] ); 
  189.           } 
  190.  
  191.           $_product = wc_get_product( $item['product_id'] ); 
  192.           if ( ! wcj_is_product_wholesale_enabled( wcj_get_product_id_or_variation_parent_id( $_product ) ) ) { 
  193.               continue; 
  194.           } 
  195.  
  196.           // Prices 
  197.           if ( 0 != ( $variation_id = WC()->cart->cart_contents[ $item_key ]['variation_id'] ) ) { 
  198.               $variation = wc_get_product( $variation_id ); 
  199.               $price = $variation->get_price(); 
  200.               $price_old = wcj_get_product_display_price( $variation ); // used for display only 
  201.           } else { 
  202.               $price = $_product->get_price(); 
  203.               $price_old = wcj_get_product_display_price( $_product ); // used for display only 
  204.           } 
  205.  
  206.           // If other discount was applied in cart... 
  207.           if ( 'yes' === get_option( 'wcj_wholesale_price_apply_only_if_no_other_discounts', 'no' ) ) { 
  208.               if ( WC()->cart->get_total_discount() > 0 || sizeof( WC()->cart->applied_coupons ) > 0 ) { 
  209.                   continue; 
  210.               } 
  211.           } 
  212.  
  213.           // Maybe set wholesale price 
  214.           $the_quantity = ( 'yes' === get_option( 'wcj_wholesale_price_use_total_cart_quantity', 'no' ) ) 
  215.               ? $cart->cart_contents_count 
  216.               : $item['quantity']; 
  217.           if ( $the_quantity > 1 ) { 
  218.               $wholesale_price = $this->get_wholesale_price( $price, $the_quantity, wcj_get_product_id_or_variation_parent_id( $_product ) ); 
  219.               if ( $wholesale_price != $price ) { 
  220.                   // Setting wholesale price 
  221.                   $precision = get_option( 'woocommerce_price_num_decimals', 2 ); 
  222.                   $wcj_wholesale_price = round( $wholesale_price, $precision ); 
  223.                   WC()->cart->cart_contents[ $item_key ]['data']->wcj_wholesale_price = $wcj_wholesale_price; 
  224.                   WC()->cart->cart_contents[ $item_key ]['wcj_wholesale_price'] = $wcj_wholesale_price; 
  225.                   WC()->cart->cart_contents[ $item_key ]['wcj_wholesale_price_old'] = $price_old; 
  226.               } 
  227.           } 
  228.  
  229.       } 
  230.   }
