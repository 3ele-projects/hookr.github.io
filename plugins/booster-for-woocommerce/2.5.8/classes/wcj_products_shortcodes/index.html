  563.                 $max = $max * $atts['multiply_by'];

  579.     /** 
  580.      * wcj_product_wholesale_price_table. 
  581.      * 
  582.      * @version 2.5.7 
  583.      */ 
  584.     function wcj_product_wholesale_price_table( $atts ) { 
  585.  
  586.         if ( ! wcj_is_product_wholesale_enabled( $this->the_product->id ) ) return ''; 
  587.  
  588.         // Check for user role options 
  589.         $role_option_name_addon = ''; 
  590.         $user_roles = get_option( 'wcj_wholesale_price_by_user_role_roles', '' ); 
  591.         if ( ! empty( $user_roles ) ) { 
  592.             $current_user_role = wcj_get_current_user_first_role(); 
  593.             foreach ( $user_roles as $user_role_key ) { 
  594.                 if ( $current_user_role === $user_role_key ) { 
  595.                     $role_option_name_addon = '_' . $user_role_key; 
  596.                     break; 
  597.                 } 
  598.             } 
  599.         } 
  600.  
  601.         $wholesale_price_levels = array(); 
  602.         if ( wcj_is_product_wholesale_enabled_per_product( $this->the_product->id ) ) { 
  603.             for ( $i = 1; $i <= apply_filters( 'wcj_get_option_filter', 1, get_post_meta( $this->the_product->id, '_' . 'wcj_wholesale_price_levels_number' . $role_option_name_addon, true ) ); $i++ ) { 
  604.                 $level_qty = get_post_meta( $this->the_product->id, '_' . 'wcj_wholesale_price_level_min_qty' . $role_option_name_addon . '_' . $i, true ); 
  605.                 $discount = get_post_meta( $this->the_product->id, '_' . 'wcj_wholesale_price_level_discount' . $role_option_name_addon . '_' . $i, true ); 
  606.                 $wholesale_price_levels[] = array( 'quantity' => $level_qty, 'discount' => $discount, ); 
  607.             } 
  608.         } else { 
  609.             for ( $i = 1; $i <= apply_filters( 'wcj_get_option_filter', 1, get_option( 'wcj_wholesale_price_levels_number' . $role_option_name_addon, 1 ) ); $i++ ) { 
  610.                 $level_qty = get_option( 'wcj_wholesale_price_level_min_qty' . $role_option_name_addon . '_' . $i, PHP_INT_MAX ); 
  611.                 $discount = get_option( 'wcj_wholesale_price_level_discount_percent' . $role_option_name_addon . '_' . $i, 0 ); 
  612.                 $wholesale_price_levels[] = array( 'quantity' => $level_qty, 'discount' => $discount, ); 
  613.             } 
  614.         } 
  615.  
  616.         $discount_type = ( wcj_is_product_wholesale_enabled_per_product( $this->the_product->id ) ) 
  617.             ? get_post_meta( $this->the_product->id, '_' . 'wcj_wholesale_price_discount_type', true ) 
  618.             : get_option( 'wcj_wholesale_price_discount_type', 'percent' ); 
  619.  
  620.         $data_qty = array(); 
  621.         $data_price = array(); 
  622.         $data_discount = array(); 
  623.         $columns_styles = array(); 
  624.         $i = -1; 
  625.         foreach ( $wholesale_price_levels as $wholesale_price_level ) { 
  626.             $i++; 
  627.             if ( 0 == $wholesale_price_level['quantity'] && 'yes' === $atts['hide_if_zero_quantity'] ) { 
  628.                 continue; 
  629.             } 
  630.  
  631.             $the_price = ''; 
  632.  
  633.             if ( $this->the_product->is_type( 'variable' ) ) { 
  634.                 // Variable 
  635.                 $prices = $this->the_product->get_variation_prices( false ); 
  636.                 $min_key = key( $prices['price'] ); 
  637.                 end( $prices['price'] ); 
  638.                 $max_key = key( $prices['price'] ); 
  639.                 $min_product = wc_get_product( $min_key ); 
  640.                 $max_product = wc_get_product( $max_key ); 
  641.                 $get_price_method = 'get_price_' . get_option( 'woocommerce_tax_display_shop' ) . 'uding_tax'; 
  642.                 $min = $min_product->$get_price_method(); 
  643.                 $max = $max_product->$get_price_method(); 
  644.                 $min_original = $min; 
  645.                 $max_original = $max; 
  646.                 if ( 'fixed' === $discount_type ) { 
  647.                     $min = $min - $wholesale_price_level['discount']; 
  648.                     $max = $max - $wholesale_price_level['discount']; 
  649.                 } else { 
  650.                     $coefficient = 1.0 - ( $wholesale_price_level['discount'] / 100.0 ); 
  651.                     $min = $min * $coefficient; 
  652.                     $max = $max * $coefficient; 
  653.                 } 
  654.                 if ( 'yes' !== $atts['hide_currency'] ) { 
  655.                     $min = wc_price( $min ); 
  656.                     $max = wc_price( $max ); 
  657.                     $min_original = wc_price( $min_original ); 
  658.                     $max_original = wc_price( $max_original ); 
  659.                 } 
  660.                 $the_price = ( $min != $max ) ? sprintf( '%s-%s', $min, $max ) : $min; 
  661.                 $the_price_original = ( $min_original != $max_original ) ? sprintf( '%s-%s', $min_original, $max_original ) : $min_original; 
  662.             } else { 
  663.                 // Simple etc. 
  664.                 $get_price_method = 'get_price_' . get_option( 'woocommerce_tax_display_shop' ) . 'uding_tax'; 
  665.                 $the_price = $this->the_product->$get_price_method(); 
  666.                 $the_price = apply_filters( 'wcj_product_wholesale_price_table_price_before', $the_price, $this->the_product ); 
  667.                 $the_price_original = $the_price; 
  668.                 if ( 'price_directly' === $discount_type ) { 
  669.                     $the_price = $wholesale_price_level['discount']; 
  670.                 } elseif ( 'fixed' === $discount_type ) { 
  671.                     $the_price = $the_price - $wholesale_price_level['discount']; 
  672.                 } else { // 'percent' 
  673.                     $coefficient = 1.0 - ( $wholesale_price_level['discount'] / 100.0 ); 
  674.                     $the_price = $the_price * $coefficient; 
  675.                 } 
  676.                 $the_price_original = apply_filters( 'wcj_product_wholesale_price_table_price_after', $the_price_original, $this->the_product ); 
  677.                 $the_price = apply_filters( 'wcj_product_wholesale_price_table_price_after', $the_price, $this->the_product ); 
  678.                 if ( 'yes' !== $atts['hide_currency'] ) { 
  679.                     $the_price = wc_price( $the_price ); 
  680.                     $the_price_original = wc_price( $the_price_original ); 
  681.                 } 
  682.             } 
  683.  
  684.             $level_max_qty = ( isset( $wholesale_price_levels[ $i + 1 ]['quantity'] ) ) ? $atts['before_level_max_qty'] . ( $wholesale_price_levels[ $i + 1 ]['quantity'] - 1 ) : $atts['last_level_max_qty']; 
  685.             $data_qty[] = str_replace( 
  686.                 array( '%level_qty%', '%level_min_qty%', '%level_max_qty%' ), // %level_qty% is deprecated 
  687.                 array( $wholesale_price_level['quantity'], $wholesale_price_level['quantity'], $level_max_qty ),  
  688.                 $atts['heading_format'] 
  689.  ); 
  690.             if ( 'yes' === $atts['add_price_row'] ) { 
  691.                 $data_price[] = str_replace( array( '%old_price%', '%price%' ), array( $the_price_original, $the_price ), $atts['price_row_format'] ); 
  692.             } 
  693.             if ( 'yes' === $atts['add_percent_row'] ) { 
  694.                 if ( 'percent' === $discount_type ) { 
  695.                     $data_discount[] = '-' . $wholesale_price_level['discount'] . '%'; 
  696.                 } else { // 'fixed' or 'price_directly' 
  697.                     // todo (maybe) 
  698.                 } 
  699.             } 
  700.             if ( 'yes' === $atts['add_discount_row'] ) { 
  701.                 if ( 'fixed' === $discount_type ) { 
  702.                     $data_discount[] = '-' . wc_price( $wholesale_price_level['discount'] ); 
  703.                 } else { // 'percent' or 'price_directly' 
  704.                     // todo (maybe) 
  705.                 } 
  706.             } 
  707.  
  708.             $columns_styles[] = 'text-align: center;'; 
  709.         } 
  710.  
  711.         $table_rows = array( $data_qty, ); 
  712.         if ( 'yes' === $atts['add_price_row'] ) { 
  713.             $table_rows[] = $data_price; 
  714.         } 
  715.         if ( 'yes' === $atts['add_percent_row'] ) { 
  716.             $table_rows[] = $data_discount; 
  717.         } 
  718.  
  719.         if ( 'vertical' === $atts['table_format'] ) { 
  720.             $table_rows_modified = array(); 
  721.             foreach ( $table_rows as $row_number => $table_row ) { 
  722.                 foreach ( $table_row as $column_number => $cell ) { 
  723.                     $table_rows_modified[ $column_number ][ $row_number ] = $cell; 
  724.                 } 
  725.             } 
  726.             $table_rows = $table_rows_modified; 
  727.         } 
  728.  
  729.         return wcj_get_table_html( $table_rows, array( 'table_class' => 'wcj_product_wholesale_price_table', 'columns_styles' => $columns_styles, 'table_heading_type' => $atts['table_format'] ) ); 
  730.     }
