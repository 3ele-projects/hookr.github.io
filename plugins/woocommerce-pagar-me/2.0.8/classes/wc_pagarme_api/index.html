  615.     /** 
  616.      * Process regular payment. 
  617.      * 
  618.      * @param int $order_id Order ID. 
  619.      * 
  620.      * @return array Redirect data. 
  621.      */ 
  622.     public function process_regular_payment( $order_id ) { 
  623.         $order = wc_get_order( $order_id ); 
  624.  
  625.         if ( isset( $this->gateway->checkout ) && 'yes' === $this->gateway->checkout ) { 
  626.             if ( ! empty( $_POST['pagarme_checkout_token'] ) ) { 
  627.                 $token = sanitize_text_field( wp_unslash( $_POST['pagarme_checkout_token'] ) ); 
  628.                 $data = $this->generate_checkout_data( $order, $token ); 
  629.  
  630.                 // Cancel the payment is irregular. 
  631.                 if ( isset( $data['error'] ) ) { 
  632.                     $this->cancel_transaction( $order, $token ); 
  633.                     $order->update_status( 'failed', $data['error'] ); 
  634.  
  635.                     return array( 
  636.                         'result' => 'success',  
  637.                         'redirect' => $this->gateway->get_return_url( $order ),  
  638.  ); 
  639.                 } 
  640.  
  641.                 $transaction = $this->do_transaction( $order, $data, $token ); 
  642.             } else { 
  643.                 $transaction = array( 'errors' => array( array( 'message' => __( 'Missing credit card data, please review your data and try again or contact us for assistance.', 'woocommerce-pagarme' ) ) ) ); 
  644.             } 
  645.         } else { 
  646.             $data = $this->generate_transaction_data( $order, $_POST ); 
  647.             $transaction = $this->do_transaction( $order, $data ); 
  648.         } 
  649.  
  650.         if ( isset( $transaction['errors'] ) ) { 
  651.             foreach ( $transaction['errors'] as $error ) { 
  652.                 wc_add_notice( $error['message'], 'error' ); 
  653.             } 
  654.  
  655.             return array( 
  656.                 'result' => 'fail',  
  657.  ); 
  658.         } else { 
  659.             // Save transaction data. 
  660.             update_post_meta( $order->id, '_wc_pagarme_transaction_id', intval( $transaction['id'] ) ); 
  661.             $payment_data = array_map( 
  662.                 'sanitize_text_field',  
  663.                 array( 
  664.                     'payment_method' => $transaction['payment_method'],  
  665.                     'installments' => $transaction['installments'],  
  666.                     'card_brand' => $this->get_card_brand_name( $transaction['card_brand'] ),  
  667.                     'antifraud_score' => $transaction['antifraud_score'],  
  668.                     'boleto_url' => $transaction['boleto_url'],  
  669.  ) 
  670.  ); 
  671.             update_post_meta( $order->id, '_wc_pagarme_transaction_data', $payment_data ); 
  672.             update_post_meta( $order->id, '_transaction_id', intval( $transaction['id'] ) ); 
  673.             $this->save_order_meta_fields( $order->id, $transaction ); 
  674.  
  675.             // Change the order status. 
  676.             $this->process_order_status( $order, $transaction['status'] ); 
  677.  
  678.             // Empty the cart. 
  679.             WC()->cart->empty_cart(); 
  680.  
  681.             // Redirect to thanks page. 
  682.             return array( 
  683.                 'result' => 'success',  
  684.                 'redirect' => $this->gateway->get_return_url( $order ),  
  685.  ); 
  686.         } 
  687.     }

  689.     /** 
  690.      * Check if Pagar.me response is validity. 
  691.      * 
  692.      * @param  array $ipn_response IPN response data. 
  693.      * 
  694.      * @return bool 
  695.      */ 
  696.     public function check_fingerprint( $ipn_response ) { 
  697.         if ( isset( $ipn_response['id'] ) && isset( $ipn_response['current_status'] ) && isset( $ipn_response['fingerprint'] ) ) { 
  698.             $fingerprint = sha1( $ipn_response['id'] . '#' . $this->gateway->api_key ); 
  699.  
  700.             if ( $fingerprint === $ipn_response['fingerprint'] ) { 
  701.                 return true; 
  702.             } 
  703.         } 
  704.  
  705.         return false; 
  706.     }

  720.     /** 
  721.      * IPN handler. 
  722.      */ 
  723.     public function ipn_handler() { 
  724.         @ob_clean(); 
  725.  
  726.         $ipn_response = ! empty( $_POST ) ? $_POST : false; 
  727.  
  728.         if ( $ipn_response && $this->check_fingerprint( $ipn_response ) ) { 
  729.             header( 'HTTP/1.1 200 OK' ); 
  730.  
  731.             $this->process_successful_ipn( $ipn_response ); 
  732.  
  733.             // Deprecated action since 2.0.0. 
  734.             do_action( 'wc_pagarme_valid_ipn_request', $ipn_response ); 
  735.  
  736.             exit; 
  737.         } else { 
  738.             wp_die( esc_html__( 'Pagar.me Request Failure', 'woocommerce-pagarme' ), '', array( 'response' => 401 ) ); 
  739.         } 
  740.     }

  742.     /** 
  743.      * Process successeful IPN requests. 
  744.      * 
  745.      * @param array $posted Posted data. 
  746.      */ 
  747.     public function process_successful_ipn( $posted ) { 
  748.         global $wpdb; 
  749.  
  750.         $posted = wp_unslash( $posted ); 
  751.         $order_id = absint( $wpdb->get_var( $wpdb->prepare( "SELECT post_id FROM $wpdb->postmeta WHERE meta_key = '_wc_pagarme_transaction_id' AND meta_value = %d", $posted['id'] ) ) ); 
  752.         $order = wc_get_order( $order_id ); 
  753.         $status = sanitize_text_field( $posted['current_status'] ); 
  754.  
  755.         if ( $order && $order->id === $order_id ) { 
  756.             $this->process_order_status( $order, $status ); 
  757.         } 
  758.     }

  760.     /** 
  761.      * Process the order status. 
  762.      * 
  763.      * @param WC_Order $order  Order data. 
  764.      * @param string   $status Transaction status. 
  765.      */ 
  766.     public function process_order_status( $order, $status ) { 
  767.         if ( 'yes' === $this->gateway->debug ) { 
  768.             $this->gateway->log->add( $this->gateway->id, 'Payment status for order ' . $order->get_order_number() . ' is now: ' . $status ); 
  769.         } 
  770.  
  771.         switch ( $status ) { 
  772.             case 'authorized' : 
  773.                 if ( ! in_array( $order->get_status(), array( 'processing', 'completed' ), true ) ) { 
  774.                     $order->update_status( 'on-hold', __( 'Pagar.me: The transaction was authorized.', 'woocommerce-pagarme' ) ); 
  775.                 } 
  776.  
  777.                 break; 
  778.             case 'processing' : 
  779.                 $order->update_status( 'on-hold', __( 'Pagar.me: The transaction is being processed.', 'woocommerce-pagarme' ) ); 
  780.  
  781.                 break; 
  782.             case 'paid' : 
  783.                 if ( ! in_array( $order->get_status(), array( 'processing', 'completed' ), true ) ) { 
  784.                     $order->add_order_note( __( 'Pagar.me: Transaction paid.', 'woocommerce-pagarme' ) ); 
  785.                 } 
  786.  
  787.                 // Changing the order for processing and reduces the stock. 
  788.                 $order->payment_complete(); 
  789.  
  790.                 break; 
  791.             case 'waiting_payment' : 
  792.                 $order->update_status( 'on-hold', __( 'Pagar.me: The banking ticket was issued but not paid yet.', 'woocommerce-pagarme' ) ); 
  793.  
  794.                 break; 
  795.             case 'refused' : 
  796.                 $order->update_status( 'failed', __( 'Pagar.me: The transaction was rejected by the card company or by fraud.', 'woocommerce-pagarme' ) ); 
  797.  
  798.                 $transaction_id = get_post_meta( $order->id, '_wc_pagarme_transaction_id', true ); 
  799.                 $transaction_url = '<a href="https://dashboard.pagar.me/#/transactions/' . intval( $transaction_id ) . '">https://dashboard.pagar.me/#/transactions/' . intval( $transaction_id ) . '</a>'; 
  800.  
  801.                 $this->send_email( 
  802.                     sprintf( esc_html__( 'The transaction for order %s was rejected by the card company or by fraud', 'woocommerce-pagarme' ), $order->get_order_number() ),  
  803.                     esc_html__( 'Transaction failed', 'woocommerce-pagarme' ),  
  804.                     sprintf( esc_html__( 'Order %1$s has been marked as failed, because the transaction was rejected by the card company or by fraud, for more details, see %2$s.', 'woocommerce-pagarme' ), $order->get_order_number(), $transaction_url ) 
  805.  ); 
  806.  
  807.                 break; 
  808.             case 'refunded' : 
  809.                 $order->update_status( 'refunded', __( 'Pagar.me: The transaction was refunded/canceled.', 'woocommerce-pagarme' ) ); 
  810.  
  811.                 $transaction_id = get_post_meta( $order->id, '_wc_pagarme_transaction_id', true ); 
  812.                 $transaction_url = '<a href="https://dashboard.pagar.me/#/transactions/' . intval( $transaction_id ) . '">https://dashboard.pagar.me/#/transactions/' . intval( $transaction_id ) . '</a>'; 
  813.  
  814.                 $this->send_email( 
  815.                     sprintf( esc_html__( 'The transaction for order %s refunded', 'woocommerce-pagarme' ), $order->get_order_number() ),  
  816.                     esc_html__( 'Transaction refunded', 'woocommerce-pagarme' ),  
  817.                     sprintf( esc_html__( 'Order %1$s has been marked as refunded by Pagar.me, for more details, see %2$s.', 'woocommerce-pagarme' ), $order->get_order_number(), $transaction_url ) 
  818.  ); 
  819.  
  820.                 break; 
  821.  
  822.             default : 
  823.                 break; 
  824.         } 
  825.     }
