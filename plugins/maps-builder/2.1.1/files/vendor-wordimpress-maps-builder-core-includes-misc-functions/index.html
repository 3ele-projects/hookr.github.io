  54. function gmb_test_ajax_works() { 
  55.  
  56.   // Check if the Airplane Mode plugin is installed 
  57.   if ( class_exists( 'Airplane_Mode_Core' ) ) { 
  58.  
  59.       $airplane = Airplane_Mode_Core::getInstance(); 
  60.  
  61.       if ( method_exists( $airplane, 'enabled' ) ) { 
  62.  
  63.           if ( $airplane->enabled() ) { 
  64.               return true; 
  65.           } 
  66.  
  67.       } else { 
  68.  
  69.           if ( $airplane->check_status() == 'on' ) { 
  70.               return true; 
  71.           } 
  72.       } 
  73.   } 
  74.  
  75.   add_filter( 'block_local_requests', '__return_false' ); 
  76.  
  77.   if ( get_transient( '_gmb_ajax_works' ) ) { 
  78.       return true; 
  79.   } 
  80.  
  81.   $params = array( 
  82.       'sslverify' => false,  
  83.       'timeout' => 30,  
  84.       'body' => array( 
  85.           'action' => 'gmb_test_ajax' 
  86.  ) 
  87.  ); 
  88.  
  89.   $ajax = wp_remote_post( gmb_get_ajax_url(), $params ); 
  90.   $works = true; 
  91.  
  92.   if ( is_wp_error( $ajax ) ) { 
  93.  
  94.       $works = false; 
  95.  
  96.   } else { 
  97.  
  98.       if ( empty( $ajax['response'] ) ) { 
  99.           $works = false; 
  100.       } 
  101.  
  102.       if ( empty( $ajax['response']['code'] ) || 200 !== (int) $ajax['response']['code'] ) { 
  103.           $works = false; 
  104.       } 
  105.  
  106.       if ( empty( $ajax['response']['message'] ) || 'OK' !== $ajax['response']['message'] ) { 
  107.           $works = false; 
  108.       } 
  109.  
  110.       if ( ! isset( $ajax['body'] ) || 0 !== (int) $ajax['body'] ) { 
  111.           $works = false; 
  112.       } 
  113.  
  114.   } 
  115.  
  116.   if ( $works ) { 
  117.       set_transient( '_gmb_ajax_works', '1', DAY_IN_SECONDS ); 
  118.   } 
  119.  
  120.   return $works; 
  121. }

  137.       $ajax_url = preg_replace( '/^http/', 'https', $ajax_url );

  149. function gmb_get_current_page_url() { 
  150.  
  151.   if ( is_front_page() ) : 
  152.       $page_url = home_url(); 
  153.   else : 
  154.       $page_url = 'http'; 
  155.  
  156.       if ( isset( $_SERVER["HTTPS"] ) && $_SERVER["HTTPS"] == "on" ) { 
  157.           $page_url .= "s"; 
  158.       } 
  159.  
  160.       $page_url .= "://"; 
  161.  
  162.       if ( isset( $_SERVER["SERVER_PORT"] ) && $_SERVER["SERVER_PORT"] != "80" ) { 
  163.           $page_url .= $_SERVER["SERVER_NAME"] . ":" . $_SERVER["SERVER_PORT"] . $_SERVER["REQUEST_URI"]; 
  164.       } else { 
  165.           $page_url .= $_SERVER["SERVER_NAME"] . $_SERVER["REQUEST_URI"]; 
  166.       } 
  167.   endif; 
  168.  
  169.   return apply_filters( 'gmb_get_current_page_url', esc_url( $page_url ) ); 
  170. }

  181. function gmb_get_host() { 
  182.   $host = false; 
  183.  
  184.   if ( defined( 'WPE_APIKEY' ) ) {

  193.       $host = 'iPage';

  195.       $host = 'IPower'; 
  196.   } elseif ( strpos( DB_HOST, '.gridserver.com' ) !== false ) { 
  197.       $host = 'MediaTemple Grid'; 
  198.   } elseif ( strpos( DB_HOST, '.pair.com' ) !== false ) { 
  199.       $host = 'pair Networks'; 
  200.   } elseif ( strpos( DB_HOST, '.stabletransit.com' ) !== false ) { 
  201.       $host = 'Rackspace Cloud'; 
  202.   } elseif ( strpos( DB_HOST, '.sysfix.eu' ) !== false ) { 
  203.       $host = 'SysFix.eu Power Hosting'; 
  204.   } elseif ( strpos( $_SERVER['SERVER_NAME'], 'Flywheel' ) !== false ) { 
  205.       $host = 'Flywheel'; 
  206.   } else {

  208.       $host = 'DBH: ' . DB_HOST . ', SRV: ' . $_SERVER['SERVER_NAME'];

  326. /** 
  327.  * Include a view 
  328.  * 
  329.  * NOTE: First this attempts to load from GMB_PLUGIN_PATH . '/includes/' then it trys GMP_CORE_PATH .'/includes', unless $full 
  330.  * NOTE: Uses include() not include_once() 
  331.  * 
  332.  * @since 2.1 
  333.  * 
  334.  * @return string 
  335.  * 
  336.  * @param string $file File path relative to either core includes path or plugin includes path. Use full absolute path if $full param is true 
  337.  * @param bool $full Optional. If true, $file param should be a full absolute path. Default is false. 
  338.  * @param array $data Optional. An array of values to be used in the view via output buffering. Default is an empty array which skips output buffering. 
  339.  * 
  340.  * @return bool True if file was included, false if not. 
  341.  */ 
  342. function gmb_include_view( $file, $full = false, $data = array() ) { 
  343.   $file = gmb_find_view( $file, $full ); 
  344.  
  345.   /** 
  346.    * Filter file path for gmb_include_view 
  347.    * 
  348.    * @since 2.1 
  349.    * 
  350.    * @param string $file File path -- should be a full absolute path 
  351.    * @param bool $full If this function is using full file path mode or not 
  352.    */ 
  353.   $file = apply_filters( 'gmb_gmb_include_view_file', $file, $full ); 
  354.  
  355.   if ( file_exists( $file ) ) { 
  356.       if ( ! empty( $data ) ) { 
  357.           extract( $data, EXTR_SKIP ); 
  358.           ob_start(); 
  359.       } 
  360.       include( $file ); 
  361.       if ( ! empty( $data ) ) { 
  362.           echo ob_get_clean(); 
  363.       } 
  364.  
  365.       return true; 
  366.   } else { 
  367.       return false; 
  368.   } 
  369. }

  371. /** 
  372.  * Find view file 
  373.  * 
  374.  * NOTE: First this attempts to load from GMB_PLUGIN_PATH . '/includes/' then it trys GMP_CORE_PATH .'/includes', unless $full 
  375.  * 
  376.  * @since 2.1 
  377.  * 
  378.  * @param string $file File path relative to either core includes path or plugin includes path. Use full absolute path if $full param is true 
  379.  * @param bool $full Optional. If true, $file param should be a full absolute path. Default is false. 
  380.  * 
  381.  * @return string

  383. function gmb_find_view( $file, $full = false ) { 
  384.   if ( ! $full ) { 
  385.       $_file = GMB_PLUGIN_PATH . 'includes/' . $file; 
  386.       if ( ! file_exists( $_file ) ) {

  389.

  391.   }
