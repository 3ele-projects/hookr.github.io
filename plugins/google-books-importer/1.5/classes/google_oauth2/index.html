  345.   function verifySignedJwtWithCerts($jwt, $certs, $required_audience) { 
  346.     $segments = explode(".", $jwt); 
  347.     if (count($segments) != 3) { 
  348.       throw new Google_AuthException("Wrong number of segments in token: $jwt"); 
  349.     } 
  350.     $signed = $segments[0] . "." . $segments[1]; 
  351.     $signature = Google_Utils::urlSafeB64Decode($segments[2]); 
  352.  
  353.     // Parse envelope. 
  354.     $envelope = json_decode(Google_Utils::urlSafeB64Decode($segments[0]), true); 
  355.     if (!$envelope) { 
  356.       throw new Google_AuthException("Can't parse token envelope: " . $segments[0]); 
  357.     } 
  358.  
  359.     // Parse token 
  360.     $json_body = Google_Utils::urlSafeB64Decode($segments[1]); 
  361.     $payload = json_decode($json_body, true); 
  362.     if (!$payload) { 
  363.       throw new Google_AuthException("Can't parse token payload: " . $segments[1]); 
  364.     } 
  365.  
  366.     // Check signature 
  367.     $verified = false; 
  368.     foreach ($certs as $keyName => $pem) { 
  369.       $public_key = new Google_PemVerifier($pem); 
  370.       if ($public_key->verify($signed, $signature)) { 
  371.         $verified = true; 
  372.         break; 
  373.       } 
  374.     } 
  375.  
  376.     if (!$verified) { 
  377.       throw new Google_AuthException("Invalid token signature: $jwt"); 
  378.     } 
  379.  
  380.     // Check issued-at timestamp 
  381.     $iat = 0; 
  382.     if (array_key_exists("iat", $payload)) { 
  383.       $iat = $payload["iat"]; 
  384.     } 
  385.     if (!$iat) { 
  386.       throw new Google_AuthException("No issue time in token: $json_body"); 
  387.     } 
  388.     $earliest = $iat - self::CLOCK_SKEW_SECS; 
  389.  
  390.     // Check expiration timestamp 
  391.     $now = time(); 
  392.     $exp = 0; 
  393.     if (array_key_exists("exp", $payload)) { 
  394.       $exp = $payload["exp"]; 
  395.     } 
  396.     if (!$exp) { 
  397.       throw new Google_AuthException("No expiration time in token: $json_body"); 
  398.     } 
  399.     if ($exp >= $now + self::MAX_TOKEN_LIFETIME_SECS) { 
  400.       throw new Google_AuthException( 
  401.           "Expiration time too far in future: $json_body"); 
  402.     } 
  403.  
  404.     $latest = $exp + self::CLOCK_SKEW_SECS; 
  405.     if ($now < $earliest) { 
  406.       throw new Google_AuthException( 
  407.           "Token used too early, $now < $earliest: $json_body"); 
  408.     } 
  409.     if ($now > $latest) { 
  410.       throw new Google_AuthException( 
  411.           "Token used too late, $now > $latest: $json_body"); 
  412.     }

  415.
