  316.   /** 
  317.    * Create a WP user at signup. 
  318.    * 
  319.    * Since BP 2.0, non-multisite configurations have stored signups in 
  320.    * the same way as Multisite configs traditionally have: in the 
  321.    * wp_signups table. However, because some plugins may be looking 
  322.    * directly in the wp_users table for non-activated signups, we 
  323.    * mirror signups there by creating "phantom" users, mimicking WP's 
  324.    * default behavior. 
  325.    * 
  326.    * @since 2.0.0 
  327.    * 
  328.    * @param string $user_login    User login string. 
  329.    * @param string $user_password User password.

  332.    * @return int User id. 
  333.    */ 
  334.   public static function add_backcompat( $user_login = '', $user_password = '', $user_email = '', $usermeta = array() ) { 
  335.       global $wpdb; 
  336.  
  337.       $user_id = wp_insert_user( array( 
  338.           'user_login' => $user_login,  
  339.           'user_pass' => $user_password,  
  340.           'display_name' => sanitize_title( $user_login ),  
  341.           'user_email' => $user_email 
  342.  ) ); 
  343.  
  344.       if ( is_wp_error( $user_id ) || empty( $user_id ) ) { 
  345.           return $user_id; 
  346.       } 
  347.  
  348.       // Update the user status to '2', ie "not activated" 
  349.       // (0 = active, 1 = spam, 2 = not active). 
  350.       $wpdb->query( $wpdb->prepare( "UPDATE {$wpdb->users} SET user_status = 2 WHERE ID = %d", $user_id ) ); 
  351.  
  352.       // WordPress creates these options automatically on 
  353.       // wp_insert_user(), but we delete them so that inactive 
  354.       // signups don't appear in various user counts. 
  355.       delete_user_option( $user_id, 'capabilities' ); 
  356.       delete_user_option( $user_id, 'user_level' ); 
  357.  
  358.       // Set any profile data. 
  359.       if ( bp_is_active( 'xprofile' ) ) { 
  360.           if ( ! empty( $usermeta['profile_field_ids'] ) ) { 
  361.               $profile_field_ids = explode( ', ', $usermeta['profile_field_ids'] ); 
  362.  
  363.               foreach ( (array) $profile_field_ids as $field_id ) { 
  364.                   if ( empty( $usermeta["field_{$field_id}"] ) ) { 
  365.                       continue; 
  366.                   } 
  367.  
  368.                   $current_field = $usermeta["field_{$field_id}"]; 
  369.                   xprofile_set_field_data( $field_id, $user_id, $current_field ); 
  370.  
  371.                   // Save the visibility level. 
  372.                   $visibility_level = ! empty( $usermeta['field_' . $field_id . '_visibility'] ) ? $usermeta['field_' . $field_id . '_visibility'] : 'public'; 
  373.                   xprofile_set_field_visibility_level( $field_id, $user_id, $visibility_level ); 
  374.               } 
  375.           } 
  376.       }

  541.   /** 
  542.    * Resend an activation email.

  547.    * @return array 
  548.    */ 
  549.   public static function resend( $signup_ids = array() ) { 
  550.       if ( empty( $signup_ids ) || ! is_array( $signup_ids ) ) { 
  551.           return false; 
  552.       } 
  553.  
  554.       $to_resend = self::get( array( 
  555.           'include' => $signup_ids,  
  556.  ) ); 
  557.  
  558.       if ( ! $signups = $to_resend['signups'] ) { 
  559.           return false; 
  560.       } 
  561.  
  562.       $result = array(); 
  563.  
  564.       /** 
  565.        * Fires before activation emails are resent. 
  566.        * 
  567.        * @since 2.0.0 
  568.        * 
  569.        * @param array $signup_ids Array of IDs to resend activation emails to. 
  570.        */ 
  571.       do_action( 'bp_core_signup_before_resend', $signup_ids ); 
  572.  
  573.       foreach ( $signups as $signup ) { 
  574.  
  575.           $meta = $signup->meta; 
  576.           $meta['sent_date'] = current_time( 'mysql', true ); 
  577.           $meta['count_sent'] = $signup->count_sent + 1; 
  578.  
  579.           // Send activation email. 
  580.           if ( is_multisite() ) { 
  581.               wpmu_signup_user_notification( $signup->user_login, $signup->user_email, $signup->activation_key, serialize( $meta ) ); 
  582.           } else { 
  583.  
  584.               // Check user status before sending email. 
  585.               $user_id = email_exists( $signup->user_email ); 
  586.  
  587.               if ( ! empty( $user_id ) && 2 != self::check_user_status( $user_id ) ) { 
  588.  
  589.                   // Status is not 2, so user's account has been activated. 
  590.                   $result['errors'][ $signup->signup_id ] = array( $signup->user_login, esc_html__( 'the sign-up has already been activated.', 'buddypress' ) ); 
  591.  
  592.                   // Repair signups table. 
  593.                   self::validate( $signup->activation_key );

  640.           return false; 
  641.       }

  643.       $to_activate = self::get( array( 
  644.           'include' => $signup_ids,  
  645.  ) );

  649.       } 
  650.  
  651.       $result = array(); 
  652.  
  653.       /** 
  654.        * Fires before activation of user accounts.

  667.  
  668.               $user_id = username_exists( $signup->user_login ); 
  669.

  671.                   $user_id = false; 
  672.               } 
  673.  
  674.               if ( empty( $user_id ) ) { 
  675.  
  676.                   // Status is not 2, so user's account has been activated. 
  677.                   $result['errors'][ $signup->signup_id ] = array( $signup->user_login, esc_html__( 'the sign-up has already been activated.', 'buddypress' ) ); 
  678.  
  679.                   // Repair signups table. 
  680.                   self::validate( $signup->activation_key ); 
  681.  
  682.               // We have a user id, account is not active, let's delete it. 
  683.               } else { 
  684.                   $result['errors'][ $signup->signup_id ] = array( $signup->user_login, $user->get_error_message() ); 
  685.               } 
  686.  
  687.           } else { 
  688.               $result['activated'][] = $user; 
  689.           } 
  690.       }
