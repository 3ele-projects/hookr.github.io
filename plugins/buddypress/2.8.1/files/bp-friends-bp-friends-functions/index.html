  18. /** 
  19.  * Create a new friendship. 
  20.  * 
  21.  * @since 1.0.0 
  22.  * 
  23.  * @param int  $initiator_userid ID of the "initiator" user (the user who is 
  24.  *                               sending the friendship request). 
  25.  * @param int  $friend_userid    ID of the "friend" user (the user whose friendship 
  26.  *                               is being requested). 
  27.  * @param bool $force_accept     Optional. Whether to force acceptance. When false,  
  28.  *                               running friends_add_friend() will result in a friendship request. 
  29.  *                               When true, running friends_add_friend() will result in an accepted 
  30.  *                               friendship, with no notifications being sent. Default: false. 
  31.  * @return bool True on success, false on failure. 
  32.  */ 
  33. function friends_add_friend( $initiator_userid, $friend_userid, $force_accept = false ) { 
  34.  
  35.   // You cannot be friends with yourself! 
  36.   if ( $initiator_userid == $friend_userid ) { 
  37.       return false; 
  38.   } 
  39.  
  40.   // Check if already friends, and bail if so. 
  41.   if ( friends_check_friendship( $initiator_userid, $friend_userid ) ) { 
  42.       return true; 
  43.   } 
  44.  
  45.   // Setup the friendship data. 
  46.   $friendship = new BP_Friends_Friendship; 
  47.   $friendship->initiator_user_id = $initiator_userid; 
  48.   $friendship->friend_user_id = $friend_userid; 
  49.   $friendship->is_confirmed = 0; 
  50.   $friendship->is_limited = 0; 
  51.   $friendship->date_created = bp_core_current_time(); 
  52.  
  53.   if ( ! empty( $force_accept ) ) { 
  54.       $friendship->is_confirmed = 1; 
  55.   } 
  56.  
  57.   // Bail if friendship could not be saved (how sad!). 
  58.   if ( ! $friendship->save() ) { 
  59.       return false; 
  60.   } 
  61.  
  62.   // Send notifications. 
  63.   if ( empty( $force_accept ) ) { 
  64.       $action = 'requested'; 
  65.  
  66.   // Update friend totals. 
  67.   } else { 
  68.       $action = 'accepted'; 
  69.       friends_update_friend_totals( $friendship->initiator_user_id, $friendship->friend_user_id, 'add' ); 
  70.   } 
  71.  
  72.   /** 
  73.    * Fires at the end of initiating a new friendship connection. 
  74.    * 
  75.    * This is a variable hook, depending on context. 
  76.    * The two potential hooks are: friends_friendship_requested, friends_friendship_accepted. 
  77.    * 
  78.    * @since 1.0.0 
  79.    * 
  80.    * @param int    $id                ID of the pending friendship connection. 
  81.    * @param int    $initiator_user_id ID of the friendship initiator. 
  82.    * @param int    $friend_user_id    ID of the friend user. 
  83.    * @param object $friendship        BuddyPress Friendship Object. 
  84.    */ 
  85.   do_action( 'friends_friendship_' . $action, $friendship->id, $friendship->initiator_user_id, $friendship->friend_user_id, $friendship ); 
  86.  
  87.   return true; 
  88. }

  90. /** 
  91.  * Remove a friendship. 
  92.  * 
  93.  * Will also delete the related "friendship_accepted" activity item. 
  94.  * 
  95.  * @since 1.0.0 
  96.  * 
  97.  * @param int $initiator_userid ID of the friendship initiator. 
  98.  * @param int $friend_userid    ID of the friend user. 
  99.  * @return bool True on success, false on failure. 
  100.  */ 
  101. function friends_remove_friend( $initiator_userid, $friend_userid ) { 
  102.  
  103.   $friendship_id = BP_Friends_Friendship::get_friendship_id( $initiator_userid, $friend_userid ); 
  104.   $friendship = new BP_Friends_Friendship( $friendship_id ); 
  105.  
  106.   /** 
  107.    * Fires before the deletion of a friendship activity item 
  108.    * for the user who canceled the friendship. 
  109.    * 
  110.    * @since 1.5.0 
  111.    * 
  112.    * @param int $friendship_id    ID of the friendship object, if any, between a pair of users. 
  113.    * @param int $initiator_userid ID of the friendship initiator. 
  114.    * @param int $friend_userid    ID of the friend user. 
  115.    */ 
  116.   do_action( 'friends_before_friendship_delete', $friendship_id, $initiator_userid, $friend_userid ); 
  117.  
  118.   // Remove the activity stream items about the friendship id. 
  119.   friends_delete_activity( array( 'item_id' => $friendship_id, 'type' => 'friendship_created', 'user_id' => 0 ) ); 
  120.  
  121.   /** 
  122.    * Fires before the friendship connection is removed. 
  123.    * 
  124.    * This hook is misleadingly named - the friendship is not yet deleted. 
  125.    * This is your last chance to do something while the friendship exists. 
  126.    * 
  127.    * @since 1.0.0 
  128.    * 
  129.    * @param int $friendship_id    ID of the friendship object, if any, between a pair of users. 
  130.    * @param int $initiator_userid ID of the friendship initiator. 
  131.    * @param int $friend_userid    ID of the friend user. 
  132.    */ 
  133.   do_action( 'friends_friendship_deleted', $friendship_id, $initiator_userid, $friend_userid ); 
  134.  
  135.   if ( $friendship->delete() ) { 
  136.       friends_update_friend_totals( $initiator_userid, $friend_userid, 'remove' ); 
  137.  
  138.       /** 
  139.        * Fires after the friendship connection is removed. 
  140.        * 
  141.        * @since 1.8.0 
  142.        * 
  143.        * @param int $initiator_userid ID of the friendship initiator. 
  144.        * @param int $friend_userid    ID of the friend user. 
  145.        */ 
  146.       do_action( 'friends_friendship_post_delete', $initiator_userid, $friend_userid ); 
  147.  
  148.       return true; 
  149.   } 
  150.  
  151.   return false; 
  152. }

  154. /**

  164. function friends_accept_friendship( $friendship_id ) { 
  165.  
  166.   // Get the friendship data. 
  167.   $friendship = new BP_Friends_Friendship( $friendship_id, true, false ); 
  168.  
  169.   // Accepting friendship. 
  170.   if ( empty( $friendship->is_confirmed ) && BP_Friends_Friendship::accept( $friendship_id ) ) { 
  171.  
  172.       // Bump the friendship counts. 
  173.       friends_update_friend_totals( $friendship->initiator_user_id, $friendship->friend_user_id ); 
  174.  
  175.       /** 
  176.        * Fires after a friendship is accepted. 
  177.        * 
  178.        * @since 1.0.0 
  179.        * 
  180.        * @param int    $id                ID of the pending friendship object. 
  181.        * @param int    $initiator_user_id ID of the friendship initiator. 
  182.        * @param int    $friend_user_id    ID of the user requested friendship with. 
  183.        * @param object $friendship        BuddyPress Friendship Object. 
  184.        */ 
  185.       do_action( 'friends_friendship_accepted', $friendship->id, $friendship->initiator_user_id, $friendship->friend_user_id, $friendship ); 
  186.  
  187.       return true; 
  188.   }

  190.   return false;

  537.  * Get a list of friends that a user can invite into this group. 
  538.  * 
  539.  * Excludes friends that are already in the group, and banned friends if the 
  540.  * user is not a group admin. 
  541.  * 
  542.  * @since 1.0.0 
  543.  * 
  544.  * @param int $user_id  User ID whose friends to see can be invited. Default: 
  545.  *                      ID of the logged-in user. 
  546.  * @param int $group_id Group to check possible invitations against. 
  547.  * @return mixed False if no friends, array of users if friends. 
  548.  */ 
  549. function friends_get_friends_invite_list( $user_id = 0, $group_id = 0 ) { 
  550.  
  551.   // Default to logged in user id. 
  552.   if ( empty( $user_id ) ) 
  553.       $user_id = bp_loggedin_user_id(); 
  554.  
  555.   // Only group admins can invited previously banned users. 
  556.   $user_is_admin = (bool) groups_is_user_admin( $user_id, $group_id ); 
  557.  
  558.   // Assume no friends. 
  559.   $friends = array(); 
  560.  
  561.   /** 
  562.    * Filters default arguments for list of friends a user can invite into this group. 
  563.    * 
  564.    * @since 1.5.4 
  565.    * 
  566.    * @param array $value Array of default parameters for invite list. 
  567.    */ 
  568.   $args = apply_filters( 'bp_friends_pre_get_invite_list', array( 
  569.       'user_id' => $user_id,  
  570.       'type' => 'alphabetical',  
  571.       'per_page' => 0 
  572.  ) ); 
  573.  
  574.   // User has friends. 
  575.   if ( bp_has_members( $args ) ) { 
  576.  
  577.       /** 
  578.        * Loop through all friends and try to add them to the invitation list. 
  579.        * 
  580.        * Exclude friends that: 
  581.        *     1. are already members of the group 
  582.        *     2. are banned from this group if the current user is also not a 
  583.        *        group admin. 
  584.        */ 
  585.       while ( bp_members() ) : 
  586.  
  587.           // Load the member. 
  588.           bp_the_member(); 
  589.  
  590.           // Get the user ID of the friend. 
  591.           $friend_user_id = bp_get_member_user_id(); 
  592.  
  593.           // Skip friend if already in the group. 
  594.           if ( groups_is_user_member( $friend_user_id, $group_id ) ) 
  595.               continue; 
  596.  
  597.           // Skip friend if not group admin and user banned from group. 
  598.           if ( ( false === $user_is_admin ) && groups_is_user_banned( $friend_user_id, $group_id ) ) 
  599.               continue; 
  600.  
  601.           // Friend is safe, so add it to the array of possible friends. 
  602.           $friends[] = array( 
  603.               'id' => $friend_user_id,  
  604.               'full_name' => bp_get_member_name() 
  605.  ); 
  606.  
  607.       endwhile; 
  608.   } 
  609.  
  610.   // If no friends, explicitly set to false. 
  611.   if ( empty( $friends ) ) 
  612.       $friends = false; 
  613.  
  614.   /** 
  615.    * Filters the list of potential friends that can be invited to this group. 
  616.    * 
  617.    * @since 1.5.4 
  618.    * 
  619.    * @param array|bool $friends  Array friends available to invite or false for no friends. 
  620.    * @param int        $user_id  ID of the user checked for who they can invite. 
  621.    * @param int        $group_id ID of the group being checked on. 
  622.    */ 
  623.   return apply_filters( 'bp_friends_get_invite_list', $friends, $user_id, $group_id ); 
  624. }

  671. function friends_search_users( $search_terms, $user_id, $pag_num = 0, $pag_page = 0 ) { 
  672.  
  673.   $user_ids = BP_Friends_Friendship::search_users( $search_terms, $user_id, $pag_num, $pag_page );

  698. /** 
  699.  * Update user friend counts.

  701.  * Friend counts are cached in usermeta for performance reasons. After a

  703.  * the cached values. 
  704.  * 
  705.  * @since 1.0.0 
  706.  * 
  707.  * @param int    $initiator_user_id ID of the first user. 
  708.  * @param int    $friend_user_id    ID of the second user. 
  709.  * @param string $status            Optional. The friendship event that's been triggered. 
  710.  *                                  'add' will ++ each user's friend counts, while any other string 
  711.  *                                  will --. 
  712.  */ 
  713. function friends_update_friend_totals( $initiator_user_id, $friend_user_id, $status = 'add' ) { 
  714.  
  715.   if ( 'add' == $status ) { 
  716.       bp_update_user_meta( $initiator_user_id, 'total_friend_count', (int)bp_get_user_meta( $initiator_user_id, 'total_friend_count', true ) + 1 ); 
  717.       bp_update_user_meta( $friend_user_id, 'total_friend_count', (int)bp_get_user_meta( $friend_user_id, 'total_friend_count', true ) + 1 ); 
  718.   } else { 
  719.       bp_update_user_meta( $initiator_user_id, 'total_friend_count', (int)bp_get_user_meta( $initiator_user_id, 'total_friend_count', true ) - 1 ); 
  720.       bp_update_user_meta( $friend_user_id, 'total_friend_count', (int)bp_get_user_meta( $friend_user_id, 'total_friend_count', true ) - 1 ); 
  721.   } 
  722. }

  766. /** 
  767.  * Used by the Activity component's @mentions to print a JSON list of the current user's friends. 
  768.  * 
  769.  * This is intended to speed up @mentions lookups for a majority of use cases. 
  770.  * 
  771.  * @since 2.1.0 
  772.  * 
  773.  * @see bp_activity_mentions_script() 
  774.  */ 
  775. function bp_friends_prime_mentions_results() { 
  776.   if ( ! bp_activity_maybe_load_mentions_scripts() ) { 
  777.       return; 
  778.   } 
  779.  
  780.   // Bail out if the site has a ton of users. 
  781.   if ( is_multisite() && wp_is_large_network( 'users' ) ) { 
  782.       return; 
  783.   } 
  784.  
  785.   if ( friends_get_total_friend_count( get_current_user_id() ) > 150 ) { 
  786.       return; 
  787.   } 
  788.  
  789.   $friends_query = array( 
  790.       'count_total' => '', // Prevents total count. 
  791.       'populate_extras' => false,  
  792.  
  793.       'type' => 'alphabetical',  
  794.       'user_id' => get_current_user_id(),  
  795.  ); 
  796.  
  797.   $friends_query = new BP_User_Query( $friends_query ); 
  798.   $results = array(); 
  799.  
  800.   foreach ( $friends_query->results as $user ) { 
  801.       $result = new stdClass(); 
  802.       $result->ID = $user->user_nicename; 
  803.       $result->image = bp_core_fetch_avatar( array( 'html' => false, 'item_id' => $user->ID ) ); 
  804.  
  805.       if ( ! empty( $user->display_name ) && ! bp_disable_profile_sync() ) { 
  806.           $result->name = $user->display_name; 
  807.       } else { 
  808.           $result->name = bp_core_get_user_displayname( $user->ID ); 
  809.       } 
  810.  
  811.       $results[] = $result; 
  812.   } 
  813.  
  814.   wp_localize_script( 'bp-mentions', 'BP_Suggestions', array( 
  815.       'friends' => $results,  
  816.  ) ); 
  817. } 
  818. add_action( 'bp_activity_mentions_prime_results', 'bp_friends_prime_mentions_results' );
