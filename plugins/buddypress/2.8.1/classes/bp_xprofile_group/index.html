  605.     /** 
  606.      * Fetch the field visibility level for the fields returned by the query. 
  607.      * 
  608.      * @since 1.6.0 
  609.      * 
  610.      * @param int   $user_id The profile owner's user_id. 
  611.      * @param array $fields  The database results returned by the get() query. 
  612.      * @return array $fields The database results, with field_visibility added 
  613.      */ 
  614.     public static function fetch_visibility_level( $user_id = 0, $fields = array() ) { 
  615.  
  616.         // Get the user's visibility level preferences. 
  617.         $visibility_levels = bp_get_user_meta( $user_id, 'bp_xprofile_visibility_levels', true ); 
  618.  
  619.         foreach( (array) $fields as $key => $field ) { 
  620.  
  621.             // Does the admin allow this field to be customized? 
  622.             $visibility = bp_xprofile_get_meta( $field->id, 'field', 'allow_custom_visibility' ); 
  623.             $allow_custom = (bool) ( 'disabled' !== $visibility ); 
  624.  
  625.             // Look to see if the user has set the visibility for this field. 
  626.             if ( ( true === $allow_custom ) && isset( $visibility_levels[ $field->id ] ) ) { 
  627.                 $field_visibility = $visibility_levels[ $field->id ]; 
  628.  
  629.             // If no admin-set default is saved, fall back on a global default. 
  630.             } else { 
  631.                 $fallback_visibility = bp_xprofile_get_meta( $field->id, 'field', 'default_visibility' ); 
  632.  
  633.                 /** 
  634.                  * Filters the XProfile default visibility level for a field. 
  635.                  * 
  636.                  * @since 1.6.0 
  637.                  * 
  638.                  * @param string $value Default visibility value. 
  639.                  */ 
  640.                 $field_visibility = ! empty( $fallback_visibility ) 
  641.                     ? $fallback_visibility 
  642.                     : apply_filters( 'bp_xprofile_default_visibility_level', 'public' ); 
  643.             } 
  644.  
  645.             $fields[ $key ]->visibility_level = $field_visibility; 
  646.         } 
  647.  
  648.         return $fields; 
  649.     }

  651.     /** 
  652.      * Fetch the admin-set preferences for all fields. 
  653.      * 
  654.      * @since 1.6.0 
  655.      * 
  656.      * @return array $default_visibility_levels An array, keyed by field_id, of default 
  657.      *                                          visibility level + allow_custom 
  658.      *                                          (whether the admin allows this 
  659.      *                                          field to be set by user) 
  660.      */ 
  661.     public static function fetch_default_visibility_levels() { 
  662.         global $wpdb; 
  663.  
  664.         $default_visibility_levels = wp_cache_get( 'default_visibility_levels', 'bp_xprofile' ); 
  665.  
  666.         if ( false === $default_visibility_levels ) { 
  667.             $bp = buddypress(); 
  668.  
  669.             $levels = $wpdb->get_results( "SELECT object_id, meta_key, meta_value FROM {$bp->profile->table_name_meta} WHERE object_type = 'field' AND ( meta_key = 'default_visibility' OR meta_key = 'allow_custom_visibility' )" ); 
  670.  
  671.             // Arrange so that the field id is the key and the visibility level the value. 
  672.             $default_visibility_levels = array(); 
  673.             foreach ( $levels as $level ) { 
  674.                 switch ( $level->meta_key ) { 
  675.                     case 'default_visibility' : 
  676.                         $default_visibility_levels[ $level->object_id ]['default'] = $level->meta_value; 
  677.                         break; 
  678.                     case 'allow_custom_visibility' : 
  679.                         $default_visibility_levels[ $level->object_id ]['allow_custom'] = $level->meta_value; 
  680.                         break; 
  681.                 } 
  682.             } 
  683.  
  684.             wp_cache_set( 'default_visibility_levels', $default_visibility_levels, 'bp_xprofile' ); 
  685.         } 
  686.  
  687.         return $default_visibility_levels; 
  688.     }
