  99.   /** 
  100.    * Set up component navigation.

  111.   public function setup_nav( $main_nav = array(), $sub_nav = array() ) { 
  112.  
  113.       // Determine user to use. 
  114.       if ( bp_displayed_user_domain() ) { 
  115.           $user_domain = bp_displayed_user_domain(); 
  116.       } elseif ( bp_loggedin_user_domain() ) { 
  117.           $user_domain = bp_loggedin_user_domain(); 
  118.       } else { 
  119.           return; 
  120.       } 
  121.  
  122.       $access = bp_core_can_edit_settings(); 
  123.       $slug = bp_get_notifications_slug(); 
  124.       $notifications_link = trailingslashit( $user_domain . $slug ); 
  125.  
  126.       // Only grab count if we're on a user page and current user has access. 
  127.       if ( bp_is_user() && bp_user_has_access() ) { 
  128.           $count = bp_notifications_get_unread_notification_count( bp_displayed_user_id() ); 
  129.           $class = ( 0 === $count ) ? 'no-count' : 'count'; 
  130.           $nav_name = sprintf( 
  131.               /** translators: %s: Unread notification count for the current user */ 
  132.               _x( 'Notifications %s', 'Profile screen nav', 'buddypress' ),  
  133.               sprintf( 
  134.                   '<span class="%s">%s</span>',  
  135.                   esc_attr( $class ),  
  136.                   bp_core_number_format( $count ) 
  137.  ) 
  138.  ); 
  139.       } else { 
  140.           $nav_name = _x( 'Notifications', 'Profile screen nav', 'buddypress' ); 
  141.       } 
  142.  
  143.       // Add 'Notifications' to the main navigation. 
  144.       $main_nav = array( 
  145.           'name' => $nav_name,  
  146.           'slug' => $slug,  
  147.           'position' => 30,  
  148.           'show_for_displayed_user' => $access,  
  149.           'screen_function' => 'bp_notifications_screen_unread',  
  150.           'default_subnav_slug' => 'unread',  
  151.           'item_css_id' => $this->id,  
  152.  ); 
  153.  
  154.       // Add the subnav items to the notifications nav item. 
  155.       $sub_nav[] = array( 
  156.           'name' => _x( 'Unread', 'Notification screen nav', 'buddypress' ),  
  157.           'slug' => 'unread',  
  158.           'parent_url' => $notifications_link,  
  159.           'parent_slug' => $slug,  
  160.           'screen_function' => 'bp_notifications_screen_unread',  
  161.           'position' => 10,  
  162.           'item_css_id' => 'notifications-my-notifications',  
  163.           'user_has_access' => $access,  
  164.  ); 
  165.  
  166.       $sub_nav[] = array( 
  167.           'name' => _x( 'Read', 'Notification screen nav', 'buddypress' ),  
  168.           'slug' => 'read',  
  169.           'parent_url' => $notifications_link,  
  170.           'parent_slug' => $slug,  
  171.           'screen_function' => 'bp_notifications_screen_read',  
  172.           'position' => 20,

  190.   public function setup_admin_bar( $wp_admin_nav = array() ) { 
  191.  
  192.       // Menus for logged in user. 
  193.       if ( is_user_logged_in() ) { 
  194.  
  195.           // Setup the logged in user variables. 
  196.           $notifications_link = trailingslashit( bp_loggedin_user_domain() . bp_get_notifications_slug() ); 
  197.  
  198.           // Pending notification requests. 
  199.           $count = bp_notifications_get_unread_notification_count( bp_loggedin_user_id() ); 
  200.           if ( ! empty( $count ) ) { 
  201.               $title = sprintf( 
  202.                   /** translators: %s: Unread notification count for the current user */ 
  203.                   _x( 'Notifications %s', 'My Account Notification pending', 'buddypress' ),  
  204.                   '<span class="count">' . bp_core_number_format( $count ) . '</span>' 
  205.  ); 
  206.               $unread = sprintf( 
  207.                   /** translators: %s: Unread notification count for the current user */ 
  208.                   _x( 'Unread %s', 'My Account Notification pending', 'buddypress' ),  
  209.                   '<span class="count">' . bp_core_number_format( $count ) . '</span>' 
  210.  ); 
  211.           } else { 
  212.               $title = _x( 'Notifications', 'My Account Notification', 'buddypress' ); 
  213.               $unread = _x( 'Unread', 'My Account Notification sub nav', 'buddypress' ); 
  214.           } 
  215.  
  216.           // Add the "My Account" sub menus. 
  217.           $wp_admin_nav[] = array( 
  218.               'parent' => buddypress()->my_account_menu_id,  
  219.               'id' => 'my-account-' . $this->id,  
  220.               'title' => $title,  
  221.               'href' => $notifications_link 
  222.  ); 
  223.  
  224.           // Unread. 
  225.           $wp_admin_nav[] = array( 
  226.               'parent' => 'my-account-' . $this->id,  
  227.               'id' => 'my-account-' . $this->id . '-unread',  
  228.               'title' => $unread,  
  229.               'href' => $notifications_link,  
  230.               'position' => 10 
  231.  ); 
  232.  
  233.           // Read. 
  234.           $wp_admin_nav[] = array( 
  235.               'parent' => 'my-account-' . $this->id,  
  236.               'id' => 'my-account-' . $this->id . '-read',  
  237.               'title' => _x( 'Read', 'My Account Notification sub nav', 'buddypress' ),  
  238.               'href' => trailingslashit( $notifications_link . 'read' ),  
  239.               'position' => 20 
  240.  ); 
  241.       }
