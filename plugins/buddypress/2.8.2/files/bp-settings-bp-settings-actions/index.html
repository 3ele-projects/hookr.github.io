  14. /** 
  15.  * Handles the changing and saving of user email addresses and passwords. 
  16.  * 
  17.  * We do quite a bit of logic and error handling here to make sure that users 
  18.  * do not accidentally lock themselves out of their accounts. We also try to 
  19.  * provide as accurate of feedback as possible without exposing anyone else's 
  20.  * information to them. 
  21.  * 
  22.  * Special considerations are made for super admins that are able to edit any 
  23.  * users accounts already, without knowing their existing password. 
  24.  * 
  25.  * @since 1.6.0 
  26.  * 
  27.  * @global BuddyPress $bp 
  28.  */ 
  29. function bp_settings_action_general() { 
  30.  
  31.   // Bail if not a POST action. 
  32.   if ( 'POST' !== strtoupper( $_SERVER['REQUEST_METHOD'] ) ) 
  33.       return; 
  34.  
  35.   // Bail if no submit action. 
  36.   if ( ! isset( $_POST['submit'] ) ) 
  37.       return; 
  38.  
  39.   // Bail if not in settings. 
  40.   if ( ! bp_is_settings_component() || ! bp_is_current_action( 'general' ) ) 
  41.       return; 
  42.  
  43.   // 404 if there are any additional action variables attached 
  44.   if ( bp_action_variables() ) { 
  45.       bp_do_404(); 
  46.       return; 
  47.   } 
  48.  
  49.   // Define local defaults 
  50.   $bp = buddypress(); // The instance 
  51.   $email_error = false;        // invalid|blocked|taken|empty|nochange 
  52.   $pass_error = false;        // invalid|mismatch|empty|nochange 
  53.   $pass_changed = false;        // true if the user changes their password 
  54.   $email_changed = false;        // true if the user changes their email 
  55.   $feedback_type = 'error';      // success|error 
  56.   $feedback = array();      // array of strings for feedback. 
  57.  
  58.   // Nonce check. 
  59.   check_admin_referer('bp_settings_general'); 
  60.  
  61.   // Validate the user again for the current password when making a big change. 
  62.   if ( ( is_super_admin() ) || ( !empty( $_POST['pwd'] ) && wp_check_password( $_POST['pwd'], $bp->displayed_user->userdata->user_pass, bp_displayed_user_id() ) ) ) { 
  63.  
  64.       $update_user = get_userdata( bp_displayed_user_id() ); 
  65.  
  66.       /** Email Change Attempt ******************************************/ 
  67.  
  68.       if ( !empty( $_POST['email'] ) ) { 
  69.  
  70.           // What is missing from the profile page vs signup - 
  71.           // let's double check the goodies. 
  72.           $user_email = sanitize_email( esc_html( trim( $_POST['email'] ) ) ); 
  73.           $old_user_email = $bp->displayed_user->userdata->user_email; 
  74.  
  75.           // User is changing email address. 
  76.           if ( $old_user_email != $user_email ) { 
  77.  
  78.               // Run some tests on the email address. 
  79.               $email_checks = bp_core_validate_email_address( $user_email ); 
  80.  
  81.               if ( true !== $email_checks ) { 
  82.                   if ( isset( $email_checks['invalid'] ) ) { 
  83.                       $email_error = 'invalid'; 
  84.                   } 
  85.  
  86.                   if ( isset( $email_checks['domain_banned'] ) || isset( $email_checks['domain_not_allowed'] ) ) { 
  87.                       $email_error = 'blocked'; 
  88.                   } 
  89.  
  90.                   if ( isset( $email_checks['in_use'] ) ) { 
  91.                       $email_error = 'taken'; 
  92.                   } 
  93.               } 
  94.  
  95.               // Store a hash to enable email validation. 
  96.               if ( false === $email_error ) { 
  97.                   $hash = wp_generate_password( 32, false ); 
  98.  
  99.                   $pending_email = array( 
  100.                       'hash' => $hash,  
  101.                       'newemail' => $user_email,  
  102.  ); 
  103.  
  104.                   bp_update_user_meta( bp_displayed_user_id(), 'pending_email_change', $pending_email ); 
  105.                   $verify_link = bp_displayed_user_domain() . bp_get_settings_slug() . '/?verify_email_change=' . $hash; 
  106.  
  107.                   // Send the verification email. 
  108.                   $args = array( 
  109.                       'tokens' => array( 
  110.                           'displayname' => bp_core_get_user_displayname( bp_displayed_user_id() ),  
  111.                           'old-user.email' => $old_user_email,  
  112.                           'user.email' => $user_email,  
  113.                           'verify.url' => esc_url( $verify_link ),  
  114.  ),  
  115.  ); 
  116.                   bp_send_email( 'settings-verify-email-change', bp_displayed_user_id(), $args ); 
  117.  
  118.                   // We mark that the change has taken place so as to ensure a 
  119.                   // success message, even though verification is still required. 
  120.                   $_POST['email'] = $update_user->user_email; 
  121.                   $email_changed = true; 
  122.               } 
  123.  
  124.           // No change. 
  125.           } else { 
  126.               $email_error = false; 
  127.           } 
  128.  
  129.       // Email address cannot be empty. 
  130.       } else { 
  131.           $email_error = 'empty'; 
  132.       } 
  133.  
  134.       /** Password Change Attempt ***************************************/ 
  135.  
  136.       if ( !empty( $_POST['pass1'] ) && !empty( $_POST['pass2'] ) ) { 
  137.  
  138.           if ( ( $_POST['pass1'] == $_POST['pass2'] ) && !strpos( " " . $_POST['pass1'], "\\" ) ) { 
  139.  
  140.               // Password change attempt is successful. 
  141.               if ( ( ! empty( $_POST['pwd'] ) && $_POST['pwd'] != $_POST['pass1'] ) || is_super_admin() )  { 
  142.                   $update_user->user_pass = $_POST['pass1']; 
  143.                   $pass_changed = true; 
  144.  
  145.               // The new password is the same as the current password. 
  146.               } else { 
  147.                   $pass_error = 'same'; 
  148.               } 
  149.  
  150.           // Password change attempt was unsuccessful. 
  151.           } else { 
  152.               $pass_error = 'mismatch'; 
  153.           } 
  154.  
  155.       // Both password fields were empty. 
  156.       } elseif ( empty( $_POST['pass1'] ) && empty( $_POST['pass2'] ) ) { 
  157.           $pass_error = false; 
  158.  
  159.       // One of the password boxes was left empty. 
  160.       } elseif ( ( empty( $_POST['pass1'] ) && !empty( $_POST['pass2'] ) ) || ( !empty( $_POST['pass1'] ) && empty( $_POST['pass2'] ) ) ) { 
  161.           $pass_error = 'empty'; 
  162.       } 
  163.  
  164.       // The structure of the $update_user object changed in WP 3.3, but 
  165.       // wp_update_user() still expects the old format. 
  166.       if ( isset( $update_user->data ) && is_object( $update_user->data ) ) { 
  167.           $update_user = $update_user->data; 
  168.           $update_user = get_object_vars( $update_user ); 
  169.  
  170.           // Unset the password field to prevent it from emptying out the 
  171.           // user's user_pass field in the database. 
  172.           // @see wp_update_user(). 
  173.           if ( false === $pass_changed ) { 
  174.               unset( $update_user['user_pass'] ); 
  175.           } 
  176.       } 
  177.  
  178.       // Clear cached data, so that the changed settings take effect 
  179.       // on the current page load. 
  180.       if ( ( false === $email_error ) && ( false === $pass_error ) && ( wp_update_user( $update_user ) ) ) { 
  181.           wp_cache_delete( 'bp_core_userdata_' . bp_displayed_user_id(), 'bp' ); 
  182.           $bp->displayed_user->userdata = bp_core_get_core_userdata( bp_displayed_user_id() ); 
  183.       } 
  184.  
  185.   // Password Error. 
  186.   } else { 
  187.       $pass_error = 'invalid'; 
  188.   } 
  189.  
  190.   // Email feedback. 
  191.   switch ( $email_error ) { 
  192.       case 'invalid' : 
  193.           $feedback['email_invalid'] = __( 'That email address is invalid. Check the formatting and try again.', 'buddypress' ); 
  194.           break; 
  195.       case 'blocked' : 
  196.           $feedback['email_blocked'] = __( 'That email address is currently unavailable for use.', 'buddypress' ); 
  197.           break; 
  198.       case 'taken' : 
  199.           $feedback['email_taken'] = __( 'That email address is already taken.', 'buddypress' ); 
  200.           break; 
  201.       case 'empty' : 
  202.           $feedback['email_empty'] = __( 'Email address cannot be empty.', 'buddypress' ); 
  203.           break; 
  204.       case false : 
  205.           // No change. 
  206.           break; 
  207.   } 
  208.  
  209.   // Password feedback. 
  210.   switch ( $pass_error ) { 
  211.       case 'invalid' : 
  212.           $feedback['pass_error'] = __( 'Your current password is invalid.', 'buddypress' ); 
  213.           break; 
  214.       case 'mismatch' : 
  215.           $feedback['pass_mismatch'] = __( 'The new password fields did not match.', 'buddypress' ); 
  216.           break; 
  217.       case 'empty' : 
  218.           $feedback['pass_empty'] = __( 'One of the password fields was empty.', 'buddypress' ); 
  219.           break; 
  220.       case 'same' : 
  221.           $feedback['pass_same'] = __( 'The new password must be different from the current password.', 'buddypress' ); 
  222.           break; 
  223.       case false : 
  224.           // No change. 
  225.           break; 
  226.   } 
  227.  
  228.   // No errors so show a simple success message. 
  229.   if ( ( ( false === $email_error ) || ( false == $pass_error ) ) && ( ( true === $pass_changed ) || ( true === $email_changed ) ) ) { 
  230.       $feedback[] = __( 'Your settings have been saved.', 'buddypress' ); 
  231.       $feedback_type = 'success'; 
  232.  
  233.   // Some kind of errors occurred. 
  234.   } elseif ( ( ( false === $email_error ) || ( false === $pass_error ) ) && ( ( false === $pass_changed ) || ( false === $email_changed ) ) ) { 
  235.       if ( bp_is_my_profile() ) { 
  236.           $feedback['nochange'] = __( 'No changes were made to your account.', 'buddypress' ); 
  237.       } else { 
  238.           $feedback['nochange'] = __( 'No changes were made to this account.', 'buddypress' ); 
  239.       } 
  240.   } 
  241.  
  242.   // Set the feedback. 
  243.   bp_core_add_message( implode( "\n", $feedback ), $feedback_type ); 
  244.  
  245.   /** 
  246.    * Fires after the general settings have been saved, and before redirect. 
  247.    * 
  248.    * @since 1.5.0 
  249.    */ 
  250.   do_action( 'bp_core_general_settings_after_save' ); 
  251.  
  252.   // Redirect to prevent issues with browser back button. 
  253.   bp_core_redirect( trailingslashit( bp_displayed_user_domain() . bp_get_settings_slug() . '/general' ) ); 
  254. } 
  255. add_action( 'bp_actions', 'bp_settings_action_general' );

  348.  
  349.   if ( bp_is_user_spammer( bp_displayed_user_id() ) != $is_spammer ) { 
  350.       $status = ( true == $is_spammer ) ? 'spam' : 'ham';

  359.        * @param string $status Determined status of "spam" or "ham" for the displayed user.

  383. function bp_settings_action_delete_account() { 
  384.  
  385.   // Bail if not a POST action. 
  386.   if ( 'POST' !== strtoupper( $_SERVER['REQUEST_METHOD'] ) ) 
  387.       return; 
  388.  
  389.   // Bail if no submit action. 
  390.   if ( ! isset( $_POST['delete-account-understand'] ) ) 
  391.       return; 
  392.  
  393.   // Bail if not in settings. 
  394.   if ( ! bp_is_settings_component() || ! bp_is_current_action( 'delete-account' ) ) 
  395.       return false; 
  396.  
  397.   // 404 if there are any additional action variables attached 
  398.   if ( bp_action_variables() ) { 
  399.       bp_do_404(); 
  400.       return; 
  401.   } 
  402.  
  403.   // Bail if account deletion is disabled. 
  404.   if ( bp_disable_account_deletion() && ! bp_current_user_can( 'delete_users' ) ) { 
  405.       return false; 
  406.   } 
  407.  
  408.   // Nonce check. 
  409.   check_admin_referer( 'delete-account' ); 
  410.  
  411.   // Get username now because it might be gone soon! 
  412.   $username = bp_get_displayed_user_fullname(); 
  413.  
  414.   // Delete the users account. 
  415.   if ( bp_core_delete_account( bp_displayed_user_id() ) ) { 
  416.  
  417.       // Add feedback after deleting a user. 
  418.       bp_core_add_message( sprintf( __( '%s was successfully deleted.', 'buddypress' ), $username ), 'success' ); 
  419.  
  420.       // Redirect to the root domain. 
  421.       bp_core_redirect( bp_get_root_domain() ); 
  422.   } 
  423. }

  426. /** 
  427.  * Process email change verification or cancel requests. 
  428.  * 
  429.  * @since 2.1.0 
  430.  */ 
  431. function bp_settings_verify_email_change() { 
  432.   if ( ! bp_is_settings_component() ) { 
  433.       return; 
  434.   } 
  435.  
  436.   if ( ! bp_is_my_profile() ) { 
  437.       return; 
  438.   } 
  439.  
  440.   $redirect_to = trailingslashit( bp_displayed_user_domain() . bp_get_settings_slug() ); 
  441.  
  442.   // Email change is being verified. 
  443.   if ( isset( $_GET['verify_email_change'] ) ) { 
  444.       $pending_email = bp_get_user_meta( bp_displayed_user_id(), 'pending_email_change', true ); 
  445.  
  446.       // Bail if the hash provided doesn't match the one saved in the database. 
  447.       if ( ! hash_equals( urldecode( $_GET['verify_email_change'] ), $pending_email['hash'] ) ) { 
  448.           return; 
  449.       } 
  450.  
  451.       $email_changed = wp_update_user( array( 
  452.           'ID' => bp_displayed_user_id(),  
  453.           'user_email' => trim( $pending_email['newemail'] ),  
  454.  ) ); 
  455.  
  456.       if ( $email_changed ) { 
  457.           // Delete object cache for displayed user. 
  458.           wp_cache_delete( 'bp_core_userdata_' . bp_displayed_user_id(), 'bp' ); 
  459.  
  460.           // Delete the pending email change key. 
  461.           bp_delete_user_meta( bp_displayed_user_id(), 'pending_email_change' ); 
  462.  
  463.           // Post a success message and redirect. 
  464.           bp_core_add_message( __( 'You have successfully verified your new email address.', 'buddypress' ) ); 
  465.       } else { 
  466.           // Unknown error. 
  467.           bp_core_add_message( __( 'There was a problem verifying your new email address. Please try again.', 'buddypress' ), 'error' ); 
  468.       } 
  469.  
  470.       bp_core_redirect( $redirect_to ); 
  471.       die(); 
  472.  
  473.   // Email change is being dismissed. 
  474.   } elseif ( ! empty( $_GET['dismiss_email_change'] ) ) { 
  475.       $nonce_check = isset( $_GET['_wpnonce'] ) && wp_verify_nonce( wp_unslash( $_GET['_wpnonce'] ), 'bp_dismiss_email_change' ); 
  476.  
  477.       if ( $nonce_check ) { 
  478.           bp_delete_user_meta( bp_displayed_user_id(), 'pending_email_change' ); 
  479.           bp_core_add_message( __( 'You have successfully dismissed your pending email change.', 'buddypress' ) ); 
  480.       } 
  481.  
  482.       bp_core_redirect( $redirect_to ); 
  483.       die(); 
  484.   } 
  485. } 
  486. add_action( 'bp_actions', 'bp_settings_verify_email_change' );
