  288. /** 
  289.  * Save our settings. 
  290.  * 
  291.  * @since 1.6.0 
  292.  */ 
  293. function bp_core_admin_settings_save() { 
  294.   global $wp_settings_fields; 
  295.  
  296.   if ( isset( $_GET['page'] ) && 'bp-settings' == $_GET['page'] && !empty( $_POST['submit'] ) ) { 
  297.       check_admin_referer( 'buddypress-options' ); 
  298.  
  299.       // Because many settings are saved with checkboxes, and thus will have no values 
  300.       // in the $_POST array when unchecked, we loop through the registered settings. 
  301.       if ( isset( $wp_settings_fields['buddypress'] ) ) { 
  302.           foreach( (array) $wp_settings_fields['buddypress'] as $section => $settings ) { 
  303.               foreach( $settings as $setting_name => $setting ) { 
  304.                   $value = isset( $_POST[$setting_name] ) ? $_POST[$setting_name] : ''; 
  305.  
  306.                   bp_update_option( $setting_name, $value ); 
  307.               } 
  308.           } 
  309.       } 
  310.  
  311.       // Some legacy options are not registered with the Settings API, or are reversed in the UI. 
  312.       $legacy_options = array( 
  313.           'bp-disable-account-deletion',  
  314.           'bp-disable-avatar-uploads',  
  315.           'bp-disable-cover-image-uploads',  
  316.           'bp-disable-group-avatar-uploads',  
  317.           'bp-disable-group-cover-image-uploads',  
  318.           'bp_disable_blogforum_comments',  
  319.           'bp-disable-profile-sync',  
  320.           'bp_restrict_group_creation',  
  321.           'hide-loggedout-adminbar',  
  322.  ); 
  323.  
  324.       foreach( $legacy_options as $legacy_option ) { 
  325.           // Note: Each of these options is represented by its opposite in the UI 
  326.           // Ie, the Profile Syncing option reads "Enable Sync", so when it's checked,  
  327.           // the corresponding option should be unset. 
  328.           $value = isset( $_POST[$legacy_option] ) ? '' : 1; 
  329.           bp_update_option( $legacy_option, $value ); 
  330.       } 
  331.  
  332.       bp_core_redirect( add_query_arg( array( 'page' => 'bp-settings', 'updated' => 'true' ), bp_get_admin_url( 'admin.php' ) ) ); 
  333.   } 
  334. } 
  335. add_action( 'bp_admin_init', 'bp_core_admin_settings_save', 100 );

  337. /** 
  338.  * Output settings API option. 
  339.  * 
  340.  * @since 1.6.0 
  341.  * 
  342.  * @param string $option  Form option to echo. 
  343.  * @param string $default Form option default. 
  344.  * @param bool   $slug    Form option slug. 
  345.  */ 
  346. function bp_form_option( $option, $default = '' , $slug = false ) { 
  347.   echo bp_get_form_option( $option, $default, $slug ); 
  348. } 
  349.   /** 
  350.    * Return settings API option 
  351.    * 
  352.    * @since 1.6.0 
  353.    * 
  354.    * 
  355.    * @param string $option  Form option to return. 
  356.    * @param string $default Form option default. 
  357.    * @param bool   $slug    Form option slug. 
  358.    * @return string 
  359.    */ 
  360.   function bp_get_form_option( $option, $default = '', $slug = false ) { 
  361.  
  362.       // Get the option and sanitize it. 
  363.       $value = bp_get_option( $option, $default ); 
  364.  
  365.       // Slug? 
  366.       if ( true === $slug ) { 
  367.  
  368.           /** 
  369.            * Filters the slug value in the form field. 
  370.            * 
  371.            * @since 1.6.0 
  372.            * 
  373.            * @param string $value Value being returned for the requested option. 
  374.            */ 
  375.           $value = esc_attr( apply_filters( 'editable_slug', $value ) ); 
  376.       } else { // Not a slug. 
  377.           $value = esc_attr( $value ); 
  378.       } 
  379.  
  380.       // Fallback to default. 
  381.       if ( empty( $value ) ) 
  382.           $value = $default; 
  383.  
  384.       /** 
  385.        * Filters the settings API option. 
  386.        * 
  387.        * @since 1.6.0 
  388.        * 
  389.        * @param string $value  Value being returned for the requested option. 
  390.        * @param string $option Option whose value is being requested. 
  391.        */ 
  392.       return apply_filters( 'bp_get_form_option', $value, $option ); 
  393.   }
