  219. /** 
  220.  * Output the inline JS needed for the cropper to work on a per-page basis. 
  221.  * 
  222.  * @since 1.1.0 
  223.  */ 
  224. function bp_core_add_cropper_inline_js() { 
  225.  
  226.   /** 
  227.    * Filters the return value of getimagesize to determine if an image was uploaded. 
  228.    * 
  229.    * @since 1.1.0 
  230.    * 
  231.    * @param array $value Array of data found by getimagesize. 
  232.    */ 
  233.   $image = apply_filters( 'bp_inline_cropper_image', getimagesize( bp_core_avatar_upload_path() . buddypress()->avatar_admin->image->dir ) ); 
  234.   if ( empty( $image ) ) { 
  235.       return; 
  236.   } 
  237.  
  238.   // Get avatar full width and height. 
  239.   $full_height = bp_core_avatar_full_height(); 
  240.   $full_width = bp_core_avatar_full_width(); 
  241.  
  242.   // Calculate Aspect Ratio. 
  243.   if ( !empty( $full_height ) && ( $full_width != $full_height ) ) { 
  244.       $aspect_ratio = $full_width / $full_height; 
  245.   } else { 
  246.       $aspect_ratio = 1; 
  247.   } 
  248.  
  249.   // Default cropper coordinates. 
  250.   // Smaller than full-width: cropper defaults to entire image. 
  251.   if ( $image[0] < $full_width ) { 
  252.       $crop_left = 0; 
  253.       $crop_right = $image[0]; 
  254.  
  255.   // Less than 2x full-width: cropper defaults to full-width. 
  256.   } elseif ( $image[0] < ( $full_width * 2 ) ) { 
  257.       $padding_w = round( ( $image[0] - $full_width ) / 2 ); 
  258.       $crop_left = $padding_w; 
  259.       $crop_right = $image[0] - $padding_w; 
  260.  
  261.   // Larger than 2x full-width: cropper defaults to 1/2 image width. 
  262.   } else { 
  263.       $crop_left = round( $image[0] / 4 ); 
  264.       $crop_right = $image[0] - $crop_left; 
  265.   } 
  266.  
  267.   // Smaller than full-height: cropper defaults to entire image. 
  268.   if ( $image[1] < $full_height ) { 
  269.       $crop_top = 0; 
  270.       $crop_bottom = $image[1]; 
  271.  
  272.   // Less than double full-height: cropper defaults to full-height. 
  273.   } elseif ( $image[1] < ( $full_height * 2 ) ) { 
  274.       $padding_h = round( ( $image[1] - $full_height ) / 2 ); 
  275.       $crop_top = $padding_h; 
  276.       $crop_bottom = $image[1] - $padding_h; 
  277.  
  278.   // Larger than 2x full-height: cropper defaults to 1/2 image height. 
  279.   } else { 
  280.       $crop_top = round( $image[1] / 4 ); 
  281.       $crop_bottom = $image[1] - $crop_top; 
  282.   } 
  283.  
  284.   ?> 
  285.  
  286.   <script type="text/javascript"> 
  287.       jQuery(window).load( function() { 
  288.           jQuery('#avatar-to-crop').Jcrop({ 
  289.               onChange: showPreview,  
  290.               onSelect: updateCoords,  
  291.               aspectRatio: <?php echo (int) $aspect_ratio; ?>,  
  292.               setSelect: [ <?php echo (int) $crop_left; ?>, <?php echo (int) $crop_top; ?>, <?php echo (int) $crop_right; ?>, <?php echo (int) $crop_bottom; ?> ] 
  293.           }); 
  294.       }); 
  295.  
  296.       function updateCoords(c) { 
  297.           jQuery('#x').val(c.x); 
  298.           jQuery('#y').val(c.y); 
  299.           jQuery('#w').val(c.w); 
  300.           jQuery('#h').val(c.h); 
  301.       } 
  302.  
  303.       function showPreview(coords) { 
  304.           if ( parseInt(coords.w) > 0 ) { 
  305.               var fw = <?php echo (int) $full_width; ?>; 
  306.               var fh = <?php echo (int) $full_height; ?>; 
  307.               var rx = fw / coords.w; 
  308.               var ry = fh / coords.h; 
  309.  
  310.               jQuery( '#avatar-crop-preview' ).css({ 
  311.                   width: Math.round(rx * <?php echo (int) $image[0]; ?>) + 'px',  
  312.                   height: Math.round(ry * <?php echo (int) $image[1]; ?>) + 'px',  
  313.                   marginLeft: '-' + Math.round(rx * coords.x) + 'px',  
  314.                   marginTop: '-' + Math.round(ry * coords.y) + 'px' 
  315.               }); 
  316.           } 
  317.       } 
  318.   </script> 
  319.  
  320. <?php 
  321. }

  407. /** 
  408.  * Add inline css to display the component's single item cover image.

  411.  *

  413.  * @return string|array the inline css or an associative array containing 
  414.  *                      the css rules and the style handle 
  415.  */ 
  416. function bp_add_cover_image_inline_css( $return = false ) { 
  417.   $bp = buddypress(); 
  418.  
  419.   // Find the component of the current item. 
  420.   if ( bp_is_user() ) { 
  421.  
  422.       // User is not allowed to upload cover images 
  423.       // no need to carry on. 
  424.       if ( bp_disable_cover_image_uploads() ) { 
  425.           return; 
  426.       } 
  427.  
  428.       $cover_image_object = array( 
  429.           'component' => 'xprofile',  
  430.           'object' => $bp->displayed_user 
  431.  ); 
  432.   } elseif ( bp_is_group() ) { 
  433.  
  434.       // Users are not allowed to upload cover images for their groups 
  435.       // no need to carry on. 
  436.       if ( bp_disable_group_cover_image_uploads() ) { 
  437.           return; 
  438.       } 
  439.  
  440.       $cover_image_object = array( 
  441.           'component' =>'groups',  
  442.           'object' => $bp->groups->current_group 
  443.  ); 
  444.   } else { 
  445.       $cover_image_object = apply_filters( 'bp_current_cover_image_object_inline_css', array() ); 
  446.   } 
  447.  
  448.   // Bail if no component were found. 
  449.   if ( empty( $cover_image_object['component'] ) || empty( $cover_image_object['object'] ) || ! bp_is_active( $cover_image_object['component'], 'cover_image' ) ) { 
  450.       return; 
  451.   } 
  452.  
  453.   // Get the settings of the cover image feature for the current component. 
  454.   $params = bp_attachments_get_cover_image_settings( $cover_image_object['component'] ); 
  455.  
  456.   // Bail if no params. 
  457.   if ( empty( $params ) ) { 
  458.       return; 
  459.   } 
  460.  
  461.   // Try to call the callback. 
  462.   if ( is_callable( $params['callback'] ) ) { 
  463.  
  464.       $object_dir = $cover_image_object['component']; 
  465.  
  466.       if ( 'xprofile' === $object_dir ) { 
  467.           $object_dir = 'members'; 
  468.       } 
  469.  
  470.       $cover_image = bp_attachments_get_attachment( 'url', array( 
  471.           'object_dir' => $object_dir,  
  472.           'item_id' => $cover_image_object['object']->id,  
  473.  ) ); 
  474.  
  475.       if ( empty( $cover_image ) ) { 
  476.           if ( ! empty( $params['default_cover'] ) ) { 
  477.               $cover_image = $params['default_cover']; 
  478.           } 
  479.       } 
  480.  
  481.       $inline_css = call_user_func_array( $params['callback'], array( array( 
  482.           'cover_image' => esc_url_raw( $cover_image ),  
  483.           'component' => sanitize_key( $cover_image_object['component'] ),  
  484.           'object_id' => (int) $cover_image_object['object']->id,  
  485.           'width' => (int) $params['width'],  
  486.           'height' => (int) $params['height'],  
  487.  ) ) ); 
  488.  
  489.       // Finally add the inline css to the handle. 
  490.       if ( ! empty( $inline_css ) ) { 
  491.  
  492.           // Used to get the css when Ajax setting the cover image. 
  493.           if ( true === $return ) { 
  494.               return array( 
  495.                   'css_rules' => '<style type="text/css">' . "\n" . $inline_css . "\n" . '</style>',  
  496.                   'handle' => $params['theme_handle'],  
  497.  ); 
  498.           } 
  499.  
  500.           wp_add_inline_style( $params['theme_handle'], $inline_css ); 
  501.       } else { 
  502.           return false; 
  503.       } 
  504.   } 
  505. } 
  506. add_action( 'bp_enqueue_scripts', 'bp_add_cover_image_inline_css', 11 );
