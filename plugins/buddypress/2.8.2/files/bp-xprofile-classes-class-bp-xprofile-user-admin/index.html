  172.    * Save the profile fields in Members community profile page. 
  173.    * 
  174.    * Loaded before the page is rendered, this function is processing form 
  175.    * requests. 
  176.    * 
  177.    * @since 2.0.0

  179.    * @param string $doaction    Action being run. 
  180.    * @param int    $user_id     ID for the user whose profile is being saved. 
  181.    * @param array  $request     Request being made. 
  182.    * @param string $redirect_to Where to redirect user to. 
  183.    */ 
  184.   public function user_admin_load( $doaction = '', $user_id = 0, $request = array(), $redirect_to = '' ) { 
  185.  
  186.       // Eventually delete avatar. 
  187.       if ( 'delete_avatar' === $doaction ) { 
  188.  
  189.           check_admin_referer( 'delete_avatar' ); 
  190.  
  191.           $redirect_to = remove_query_arg( '_wpnonce', $redirect_to ); 
  192.  
  193.           if ( bp_core_delete_existing_avatar( array( 'item_id' => $user_id ) ) ) { 
  194.               $redirect_to = add_query_arg( 'updated', 'avatar', $redirect_to ); 
  195.           } else { 
  196.               $redirect_to = add_query_arg( 'error', 'avatar', $redirect_to ); 
  197.           } 
  198.  
  199.           bp_core_redirect( $redirect_to ); 
  200.  
  201.       } elseif ( isset( $_POST['field_ids'] ) ) { 
  202.           // Update profile fields. 
  203.           // Check the nonce. 
  204.           check_admin_referer( 'edit-bp-profile_' . $user_id ); 
  205.  
  206.           // Check we have field ID's. 
  207.           if ( empty( $_POST['field_ids'] ) ) { 
  208.               $redirect_to = add_query_arg( 'error', '1', $redirect_to ); 
  209.               bp_core_redirect( $redirect_to ); 
  210.           } 
  211.  
  212.           /** 
  213.            * Unlike front-end edit-fields screens, the wp-admin/profile 
  214.            * displays all groups of fields on a single page, so the list of 
  215.            * field ids is an array gathering for each group of fields a 
  216.            * distinct comma separated list of ids. 
  217.            * 
  218.            * As a result, before using the wp_parse_id_list() function, we 
  219.            * must ensure that these ids are "merged" into a single comma 
  220.            * separated list. 
  221.            */ 
  222.           $merge_ids = join( ', ', $_POST['field_ids'] ); 
  223.  
  224.           // Explode the posted field IDs into an array so we know which fields have been submitted. 
  225.           $posted_field_ids = wp_parse_id_list( $merge_ids ); 
  226.           $is_required = array(); 
  227.  
  228.           // Loop through the posted fields formatting any datebox values then validate the field. 
  229.           foreach ( (array) $posted_field_ids as $field_id ) { 
  230.               bp_xprofile_maybe_format_datebox_post_data( $field_id ); 
  231.  
  232.               $is_required[ $field_id ] = xprofile_check_is_required_field( $field_id ) && ! bp_current_user_can( 'bp_moderate' ); 
  233.               if ( $is_required[ $field_id ] && empty( $_POST['field_' . $field_id ] ) ) { 
  234.                   $redirect_to = add_query_arg( 'error', '2', $redirect_to ); 
  235.                   bp_core_redirect( $redirect_to ); 
  236.               } 
  237.           } 
  238.  
  239.           // Set the errors var. 
  240.           $errors = false; 
  241.  
  242.           // Now we've checked for required fields, let's save the values. 
  243.           $old_values = $new_values = array(); 
  244.           foreach ( (array) $posted_field_ids as $field_id ) { 
  245.  
  246.               /** 
  247.                * Certain types of fields (checkboxes, multiselects) may come 
  248.                * through empty. Save them as an empty array so that they don't 
  249.                * get overwritten by the default on the next edit. 
  250.                */ 
  251.               $value = isset( $_POST['field_' . $field_id] ) ? $_POST['field_' . $field_id] : ''; 
  252.  
  253.               $visibility_level = ! empty( $_POST['field_' . $field_id . '_visibility'] ) ? $_POST['field_' . $field_id . '_visibility'] : 'public'; 
  254.               /** 
  255.                * Save the old and new values. They will be 
  256.                * passed to the filter and used to determine 
  257.                * whether an activity item should be posted. 
  258.                */ 
  259.               $old_values[ $field_id ] = array( 
  260.                   'value' => xprofile_get_field_data( $field_id, $user_id ),  
  261.                   'visibility' => xprofile_get_field_visibility_level( $field_id, $user_id ),  
  262.  ); 
  263.  
  264.               // Update the field data and visibility level. 
  265.               xprofile_set_field_visibility_level( $field_id, $user_id, $visibility_level ); 
  266.               $field_updated = xprofile_set_field_data( $field_id, $user_id, $value, $is_required[ $field_id ] ); 
  267.               $value = xprofile_get_field_data( $field_id, $user_id ); 
  268.  
  269.               $new_values[ $field_id ] = array( 
  270.                   'value' => $value,  
  271.                   'visibility' => xprofile_get_field_visibility_level( $field_id, $user_id ),  
  272.  ); 
  273.  
  274.               if ( ! $field_updated ) { 
  275.                   $errors = true; 
  276.               } else { 
  277.  
  278.                   /** 
  279.                    * Fires after the saving of each profile field, if successful. 
  280.                    * 
  281.                    * @since 1.1.0 
  282.                    * 
  283.                    * @param int    $field_id ID of the field being updated. 
  284.                    * @param string $value    Value that was saved to the field. 
  285.                    */ 
  286.                   do_action( 'xprofile_profile_field_data_updated', $field_id, $value ); 
  287.               } 
  288.           } 
  289.  
  290.           /** 
  291.            * Fires after all XProfile fields have been saved for the current profile. 
  292.            * 
  293.            * @since 1.0.0 
  294.            * @since 2.6.0 Added $old_values and $new_values parameters. 
  295.            * 
  296.            * @param int   $user_id          ID for the user whose profile is being saved. 
  297.            * @param array $posted_field_ids Array of field IDs that were edited. 
  298.            * @param bool  $errors           Whether or not any errors occurred. 
  299.            * @param array $old_values       Array of original values before update. 
  300.            * @param array $new_values       Array of newly saved values after update. 
  301.            */ 
  302.           do_action( 'xprofile_updated_profile', $user_id, $posted_field_ids, $errors, $old_values, $new_values ); 
  303.  
  304.           // Set the feedback messages. 
  305.           if ( ! empty( $errors ) ) { 
  306.               $redirect_to = add_query_arg( 'error', '3', $redirect_to ); 
  307.           } else { 
  308.               $redirect_to = add_query_arg( 'updated', '1', $redirect_to ); 
  309.           } 
  310.  
  311.           bp_core_redirect( $redirect_to ); 
  312.       } 
  313.   }
