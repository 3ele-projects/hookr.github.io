  890.    * Set token names and replacement values for this email. 
  891.    * 
  892.    * In templates, tokens are inserted with a Handlebars-like syntax, e.g. `{{token_name}}`. 
  893.    * { and } are reserved characters. There's no need to specify these brackets in your token names.

  898.    *                         Values are a string or a callable function. 
  899.    * @return BP_Email 
  900.    */ 
  901.   public function set_tokens( array $tokens ) { 
  902.       $formatted_tokens = array(); 
  903.  
  904.       foreach ( $tokens as $name => $value ) { 
  905.           $name = str_replace( array( '{', '}' ), '', sanitize_text_field( $name ) ); 
  906.           $formatted_tokens[ $name ] = $value; 
  907.       }
