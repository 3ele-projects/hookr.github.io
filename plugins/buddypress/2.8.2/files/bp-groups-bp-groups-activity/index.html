  539.  * If the user joined this group less than five minutes ago, remove the 
  540.  * joined_group activity so users cannot flood the activity stream by 
  541.  * joining/leaving the group in quick succession. 
  542.  * 
  543.  * @since 1.9.0 
  544.  * 
  545.  * @param int $group_id ID of the group. 
  546.  * @param int $user_id  ID of the user leaving the group. 
  547.  */ 
  548. function bp_groups_leave_group_delete_recent_activity( $group_id, $user_id ) { 
  549.  
  550.   // Bail if Activity component is not active. 
  551.   if ( ! bp_is_active( 'activity' ) ) { 
  552.       return; 
  553.   } 
  554.  
  555.   // Get the member's group membership information. 
  556.   $membership = new BP_Groups_Member( $user_id, $group_id ); 
  557.  
  558.   // Check the time period, and maybe delete their recent group activity. 
  559.   if ( time() <= strtotime( '+5 minutes', (int) strtotime( $membership->date_modified ) ) ) { 
  560.       bp_activity_delete( array( 
  561.           'component' => buddypress()->groups->id,  
  562.           'type' => 'joined_group',  
  563.           'user_id' => $user_id,  
  564.           'item_id' => $group_id 
  565.  ) ); 
  566.   } 
  567. } 
  568. add_action( 'groups_leave_group', 'bp_groups_leave_group_delete_recent_activity', 10, 2 ); 
  569. add_action( 'groups_remove_member', 'bp_groups_leave_group_delete_recent_activity', 10, 2 ); 
  570. add_action( 'groups_ban_member', 'bp_groups_leave_group_delete_recent_activity', 10, 2 );
