  2. /** 
  3.  * BuddyPress XProfile Actions. 
  4.  * 
  5.  * Action functions are exactly the same as screen functions, however they do not 
  6.  * have a template screen associated with them. Usually they will send the user 
  7.  * back to the default screen after execution. 
  8.  * 
  9.  * @package BuddyPress 
  10.  * @subpackage XProfileActions 
  11.  * @since 1.5.0 
  12.  */ 
  13.  
  14. // Exit if accessed directly. 
  15. defined( 'ABSPATH' ) || exit; 
  16.  
  17. /** 
  18.  * This function runs when an action is set for a screen: 
  19.  * example.com/members/andy/profile/change-avatar/ [delete-avatar] 
  20.  * 
  21.  * The function will delete the active avatar for a user. 
  22.  * 
  23.  * @since 1.0.0 
  24.  * 
  25.  */ 
  26. function xprofile_action_delete_avatar() { 
  27.  
  28.   if ( ! bp_is_user_change_avatar() || ! bp_is_action_variable( 'delete-avatar', 0 ) ) { 
  29.       return false; 
  30.   } 
  31.  
  32.   // Check the nonce. 
  33.   check_admin_referer( 'bp_delete_avatar_link' ); 
  34.  
  35.   if ( ! bp_is_my_profile() && ! bp_current_user_can( 'bp_moderate' ) ) { 
  36.       return false; 
  37.   } 
  38.  
  39.   if ( bp_core_delete_existing_avatar( array( 'item_id' => bp_displayed_user_id() ) ) ) { 
  40.       bp_core_add_message( __( 'Your profile photo was deleted successfully!', 'buddypress' ) ); 
  41.   } else { 
  42.       bp_core_add_message( __( 'There was a problem deleting your profile photo. Please try again.', 'buddypress' ), 'error' ); 
  43.   } 
  44.  
  45.   bp_core_redirect( wp_get_referer() ); 
  46. } 
  47. add_action( 'bp_actions', 'xprofile_action_delete_avatar' ); 
  48.  
  49. /** 
  50.  * Handles the saving of xprofile field visibilities. 
  51.  * 
  52.  * @since 1.9.0 
  53.  */ 
  54. function bp_xprofile_action_settings() { 
  55.  
  56.   // Bail if not a POST action. 
  57.   if ( 'POST' !== strtoupper( $_SERVER['REQUEST_METHOD'] ) ) { 
  58.       return; 
  59.   } 
  60.  
  61.   // Bail if no submit action. 
  62.   if ( ! isset( $_POST['xprofile-settings-submit'] ) ) { 
  63.       return; 
  64.   } 
  65.  
  66.   // Bail if not in settings. 
  67.   if ( ! bp_is_user_settings_profile() ) { 
  68.       return; 
  69.   } 
  70.  
  71.   // 404 if there are any additional action variables attached 
  72.   if ( bp_action_variables() ) { 
  73.       bp_do_404(); 
  74.       return; 
  75.   } 
  76.  
  77.   // Nonce check. 
  78.   check_admin_referer( 'bp_xprofile_settings' ); 
  79.  
  80.   /** 
  81.    * Fires before saving xprofile field visibilities. 
  82.    * 
  83.    * @since 2.0.0 
  84.    */ 
  85.   do_action( 'bp_xprofile_settings_before_save' ); 
  86.  
  87.   /** Save ******************************************************************/ 
  88.  
  89.   // Only save if there are field ID's being posted. 
  90.   if ( ! empty( $_POST['field_ids'] ) ) { 
  91.  
  92.       // Get the POST'ed field ID's. 
  93.       $posted_field_ids = explode( ', ', $_POST['field_ids'] ); 
  94.  
  95.       // Backward compatibility: a bug in BP 2.0 caused only a single 
  96.       // group's field IDs to be submitted. Look for values submitted 
  97.       // in the POST request that may not appear in 'field_ids', and 
  98.       // add them to the list of IDs to save. 
  99.       foreach ( $_POST as $posted_key => $posted_value ) { 
  100.           preg_match( '/^field_([0-9]+)_visibility$/', $posted_key, $matches ); 
  101.           if ( ! empty( $matches[1] ) && ! in_array( $matches[1], $posted_field_ids ) ) { 
  102.               $posted_field_ids[] = $matches[1]; 
  103.           } 
  104.       } 
  105.  
  106.       // Save the visibility settings. 
  107.       foreach ( $posted_field_ids as $field_id ) { 
  108.  
  109.           $visibility_level = 'public'; 
  110.  
  111.           if ( !empty( $_POST['field_' . $field_id . '_visibility'] ) ) { 
  112.               $visibility_level = $_POST['field_' . $field_id . '_visibility']; 
  113.           } 
  114.  
  115.           xprofile_set_field_visibility_level( $field_id, bp_displayed_user_id(), $visibility_level ); 
  116.       } 
  117.   } 
  118.  
  119.   /** Other *****************************************************************/ 
  120.  
  121.   /** 
  122.    * Fires after saving xprofile field visibilities. 
  123.    * 
  124.    * @since 2.0.0 
  125.    */ 
  126.   do_action( 'bp_xprofile_settings_after_save' ); 
  127.  
  128.   // Redirect to the root domain. 
  129.   bp_core_redirect( bp_displayed_user_domain() . bp_get_settings_slug() . '/profile' ); 
  130. } 
  131. add_action( 'bp_actions', 'bp_xprofile_action_settings' );
