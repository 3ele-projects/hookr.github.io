  167.   if ( empty( $field_value ) ) {

  171.   // Value might be serialized.

  185.        * @param string                  $field_value          The original value before filtering. 
  186.        * @param BP_XProfile_ProfileData $data_obj             The BP_XProfile_ProfileData object. 
  187.        */ 
  188.       $filtered_field_value = apply_filters( 'xprofile_filtered_data_value_before_save', $filtered_field_value, $field_value, $data_obj ); 
  189.  
  190.   // Filter each array item independently. 
  191.   } else { 
  192.       $filtered_values = array(); 
  193.       foreach ( (array) $field_value as $value ) { 
  194.           $kses_field_value = xprofile_filter_kses( $value, $data_obj );

  199.  
  200.       } 
  201.  
  202.       if ( !empty( $reserialize ) ) { 
  203.           $filtered_field_value = serialize( $filtered_values ); 
  204.       } else { 
  205.           $filtered_field_value = $filtered_values; 
  206.       } 
  207.   } 
  208.  
  209.   return $filtered_field_value;

  322. /** 
  323.  * Filter an Extended Profile field value, and attempt to make clickable links 
  324.  * to members search results out of them. 
  325.  * 
  326.  * - Not run on datebox field types. 
  327.  * - Not run on values without commas with less than 5 words. 
  328.  * - URL's are made clickable. 
  329.  * 
  330.  * To disable globally: 
  331.  *     remove_filter( 'bp_get_the_profile_field_value', 'xprofile_filter_link_profile_data', 9, 3 ); 
  332.  * 
  333.  * To disable for a single field, use the 'Autolink' settings in Dashboard > Users > Profile Fields. 
  334.  * 
  335.  * @since 1.1.0 
  336.  * 
  337.  * @param string $field_value Profile field data value. 
  338.  * @param string $field_type  Profile field type. 
  339.  * @return string 
  340.  */ 
  341. function xprofile_filter_link_profile_data( $field_value, $field_type = 'textbox' ) { 
  342.   global $field; 
  343.  
  344.   if ( ! $field->get_do_autolink() ) { 
  345.       return $field_value; 
  346.   } 
  347.  
  348.   if ( 'datebox' === $field_type ) { 
  349.       return $field_value; 
  350.   } 
  351.  
  352.   if ( strpos( $field_value, ', ' ) === false && strpos( $field_value, ';' ) === false && ( count( explode( ' ', $field_value ) ) > 5 ) ) { 
  353.       return $field_value; 
  354.   } 
  355.  
  356.   if ( strpos( $field_value, ', ' ) !== false ) { 
  357.       $list_type = 'comma'; 
  358.       $values = explode( ', ', $field_value ); // Comma-separated lists. 
  359.   } else { 
  360.       $list_type = 'semicolon'; 
  361.       $values = explode( ';', $field_value ); // Semicolon-separated lists. 
  362.   } 
  363.  
  364.   if ( ! empty( $values ) ) { 
  365.       foreach ( (array) $values as $value ) { 
  366.           $value = trim( $value ); 
  367.  
  368.           // If the value is a URL, skip it and just make it clickable. 
  369.           if ( preg_match( '@(https?://([-\w\.]+)+(:\d+)?(/([\w/_\.]*(\?\S+)?)?)?)@', $value ) ) { 
  370.               $new_values[] = make_clickable( $value ); 
  371.  
  372.           // Is not clickable. 
  373.           } else { 
  374.  
  375.               // More than 5 spaces. 
  376.               if ( count( explode( ' ', $value ) ) > 5 ) { 
  377.                   $new_values[] = $value; 
  378.  
  379.               // Less than 5 spaces. 
  380.               } else { 
  381.                   $query_arg = bp_core_get_component_search_query_arg( 'members' ); 
  382.                   $search_url = add_query_arg( array( $query_arg => urlencode( $value ) ), bp_get_members_directory_permalink() ); 
  383.                   $new_values[] = '<a href="' . esc_url( $search_url ) . '" rel="nofollow">' . $value . '</a>'; 
  384.               } 
  385.           } 
  386.       } 
  387.  
  388.       if ( 'comma' === $list_type ) { 
  389.           $values = implode( ', ', $new_values ); 
  390.       } else { 
  391.           $values = implode( '; ', $new_values ); 
  392.       } 
  393.   } 
  394.  
  395.   return $values; 
  396. }

  398. /** 
  399.  * Ensures that BP data appears in comments array. 
  400.  * 
  401.  * This filter loops through the comments return by a normal WordPress request 
  402.  * and swaps out user data with BP xprofile data, where available. 
  403.  * 
  404.  * @since 1.2.0 
  405.  * 
  406.  * @param array $comments Comments to filter in. 
  407.  * @param int   $post_id  Post ID the comments are for. 
  408.  * @return array $comments 
  409.  */ 
  410. function xprofile_filter_comments( $comments, $post_id = 0 ) { 
  411.  
  412.   // Locate comment authors with WP accounts. 
  413.   foreach( (array) $comments as $comment ) { 
  414.       if ( $comment->user_id ) { 
  415.           $user_ids[] = $comment->user_id; 
  416.       } 
  417.   } 
  418.  
  419.   // If none are found, just return the comments array. 
  420.   if ( empty( $user_ids ) ) { 
  421.       return $comments; 
  422.   } 
  423.  
  424.   // Pull up the xprofile fullname of each commenter. 
  425.   if ( $fullnames = bp_core_get_user_displaynames( $user_ids ) ) { 
  426.       foreach( (array) $fullnames as $user_id => $user_fullname ) { 
  427.           $users[ $user_id ] = trim( stripslashes( $user_fullname ) ); 
  428.       } 
  429.   } 
  430.  
  431.   // Loop through and match xprofile fullname with commenters. 
  432.   foreach( (array) $comments as $i => $comment ) { 
  433.       if ( ! empty( $comment->user_id ) ) { 
  434.           if ( ! empty( $users[ $comment->user_id ] ) ) { 
  435.               $comments[ $i ]->comment_author = $users[ $comment->user_id ]; 
  436.           } 
  437.       } 
  438.   } 
  439.  
  440.   return $comments; 
  441. }

  484.   $sql = $xprofile_query->get_sql( 'u', $q->uid_name ); 
  485.  
  486.   if ( ! empty( $sql['join'] ) ) { 
  487.       $q->uid_clauses['select'] .= $sql['join']; 
  488.       $q->uid_clauses['where'] .= $sql['where']; 
  489.   } 
  490. } 
  491. add_action( 'bp_pre_user_query', 'bp_xprofile_add_xprofile_query_to_user_query' ); 
  492.  
  493. /** 
  494.  * Filter meta queries to modify for the xprofile data schema. 
  495.  * 
  496.  * @since 2.0.0 
  497.  * 
  498.  * @access private Do not use. 
  499.  * 
  500.  * @param string $q SQL query. 
  501.  * @return string 
  502.  */ 
  503. function bp_xprofile_filter_meta_query( $q ) { 
  504.   global $wpdb; 
  505.  
  506.   $raw_q = $q; 
  507.  
  508.   /** 
  509.    * Replace quoted content with __QUOTE__ to avoid false positives. 
  510.    * This regular expression will match nested quotes. 
  511.    */ 
  512.   $quoted_regex = "/'[^'\\\\]*(?:\\\\.[^'\\\\]*)*'/s"; 
  513.   preg_match_all( $quoted_regex, $q, $quoted_matches ); 
  514.   $q = preg_replace( $quoted_regex, '__QUOTE__', $q ); 
  515.  
  516.   // Get the first word of the command. 
  517.   preg_match( '/^(\S+)/', $q, $first_word_matches ); 
  518.  
  519.   if ( empty( $first_word_matches[0] ) ) { 
  520.       return $raw_q; 
  521.   } 
  522.  
  523.   // Get the field type. 
  524.   preg_match( '/xprofile_(group|field|data)_id/', $q, $matches ); 
  525.  
  526.   if ( empty( $matches[0] ) || empty( $matches[1] ) ) { 
  527.       return $raw_q; 
  528.   } 
  529.  
  530.   switch ( $first_word_matches[0] ) { 
  531.  
  532.       /** 
  533.        * SELECT: 
  534.        * - replace 'xprofile_{fieldtype}_id' with 'object_id' 
  535.        * - ensure that 'object_id' is aliased to 'xprofile_{fieldtype}_id',  
  536.        *   because update_meta_cache() needs the column name to parse 
  537.        *   the query results 
  538.        * - append the 'object type' WHERE clause 
  539.        */ 
  540.       case 'SELECT' : 
  541.           $q = str_replace( 
  542.               array( 
  543.                   $matches[0],  
  544.                   'SELECT object_id',  
  545.                   'WHERE ',  
  546.  ),  
  547.               array( 
  548.                   'object_id',  
  549.                   'SELECT object_id AS ' . $matches[0],  
  550.                   $wpdb->prepare( 'WHERE object_type = %s AND ', $matches[1] ),  
  551.  ),  
  552.               $q 
  553.  ); 
  554.           break; 
  555.  
  556.       /** 
  557.        * UPDATE and DELETE: 
  558.        * - replace 'xprofile_{fieldtype}_id' with 'object_id' 
  559.        * - append the 'object type' WHERE clause 
  560.        */ 
  561.       case 'UPDATE' : 
  562.       case 'DELETE' : 
  563.           $q = str_replace( 
  564.               array( 
  565.                   $matches[0],  
  566.                   'WHERE ',  
  567.  ),  
  568.               array( 
  569.                   'object_id',  
  570.                   $wpdb->prepare( 'WHERE object_type = %s AND ', $matches[1] ),  
  571.  ),  
  572.               $q 
  573.  ); 
  574.           break; 
  575.  
  576.       /** 
  577.        * UPDATE and DELETE: 
  578.        * - replace 'xprofile_{fieldtype}_id' with 'object_id' 
  579.        * - ensure that the object_type field gets filled in 
  580.        */ 
  581.       case 'INSERT' : 
  582.           $q = str_replace( 
  583.               array( 
  584.                   '`' . $matches[0] . '`',  
  585.                   'VALUES (',  
  586.  ),  
  587.               array( 
  588.                   '`object_type`, `object_id`',  
  589.                   $wpdb->prepare( "VALUES (%s, ", $matches[1] ),  
  590.  ),  
  591.               $q 
  592.  ); 
  593.           break; 
  594.   } 
  595.  
  596.   // Put quoted content back into the string. 
  597.   if ( ! empty( $quoted_matches[0] ) ) { 
  598.       for ( $i = 0; $i < count( $quoted_matches[0] ); $i++ ) { 
  599.           $quote_pos = strpos( $q, '__QUOTE__' ); 
  600.           $q = substr_replace( $q, $quoted_matches[0][ $i ], $quote_pos, 9 ); 
  601.       } 
  602.   } 
  603.  
  604.   return $q; 
  605. }
