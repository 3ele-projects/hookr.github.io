  471.   // Nonce check. 
  472.   if ( ! check_admin_referer( 'groups_leave_group' ) ) { 
  473.       return false; 
  474.   } 
  475.  
  476.   // User wants to leave any group. 
  477.   if ( groups_is_user_member( bp_loggedin_user_id(), bp_get_current_group_id() ) ) { 
  478.       $bp = buddypress(); 
  479.  
  480.       // Stop sole admins from abandoning their group. 
  481.       $group_admins = groups_get_group_admins( bp_get_current_group_id() ); 
  482.  
  483.       if ( 1 == count( $group_admins ) && $group_admins[0]->user_id == bp_loggedin_user_id() ) { 
  484.           bp_core_add_message( __( 'This group must have at least one admin', 'buddypress' ), 'error' ); 
  485.       } elseif ( ! groups_leave_group( $bp->groups->current_group->id ) ) { 
  486.           bp_core_add_message( __( 'There was an error leaving the group.', 'buddypress' ), 'error' ); 
  487.       } else { 
  488.           bp_core_add_message( __( 'You successfully left the group.', 'buddypress' ) ); 
  489.       } 
  490.  
  491.       $redirect = bp_get_group_permalink( groups_get_current_group() ); 
  492.  
  493.       if( 'hidden' == $bp->groups->current_group->status ) { 
  494.           $redirect = trailingslashit( bp_loggedin_user_domain() . bp_get_groups_slug() ); 
  495.       } 
  496.  
  497.       bp_core_redirect( $redirect ); 
  498.   } 
  499.  
  500.   /** This filter is documented in bp-groups/bp-groups-actions.php */ 
  501.   bp_core_load_template( apply_filters( 'groups_template_group_home', 'groups/single/home' ) ); 
  502. } 
  503. add_action( 'bp_actions', 'groups_action_leave_group' );

  505. /** 
  506.  * Sort the group creation steps. 
  507.  * 
  508.  * @since 1.1.0 
  509.  * 
  510.  * @return false|null False on failure. 
  511.  */ 
  512. function groups_action_sort_creation_steps() { 
  513.  
  514.   if ( !bp_is_groups_component() || !bp_is_current_action( 'create' ) ) 
  515.       return false; 
  516.  
  517.   $bp = buddypress(); 
  518.  
  519.   if ( !is_array( $bp->groups->group_creation_steps ) ) 
  520.       return false; 
  521.  
  522.   foreach ( (array) $bp->groups->group_creation_steps as $slug => $step ) { 
  523.       while ( !empty( $temp[$step['position']] ) ) 
  524.           $step['position']++; 
  525.  
  526.       $temp[$step['position']] = array( 'name' => $step['name'], 'slug' => $slug ); 
  527.   } 
  528.  
  529.   // Sort the steps by their position key. 
  530.   ksort($temp); 
  531.   unset($bp->groups->group_creation_steps); 
  532.  
  533.   foreach( (array) $temp as $position => $step ) 
  534.       $bp->groups->group_creation_steps[$step['slug']] = array( 'name' => $step['name'], 'position' => $position ); 
  535.  
  536.   /** 
  537.    * Fires after group creation sets have been sorted. 
  538.    * 
  539.    * @since 2.3.0 
  540.    */ 
  541.   do_action( 'groups_action_sort_creation_steps' ); 
  542. }
