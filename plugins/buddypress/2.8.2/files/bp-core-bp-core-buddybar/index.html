  114.   $bp = buddypress();

  116.   $defaults = array( 
  117.       'name' => false, // Display name for the nav item. 
  118.       'slug' => false, // URL slug for the nav item. 
  119.       'item_css_id' => false, // The CSS ID to apply to the HTML of the nav item. 
  120.       'show_for_displayed_user' => true, // When viewing another user does this nav item show up? 
  121.       'site_admin_only' => false, // Can only site admins see this nav item? 
  122.       'position' => 99, // Index of where this nav item should be positioned.

  195. function bp_core_register_nav_screen_function( $args = '' ) { 
  196.   $bp = buddypress(); 
  197.  
  198.   $defaults = array( 
  199.       'name' => false, // Display name for the nav item. 
  200.       'slug' => false, // URL slug for the nav item. 
  201.       'item_css_id' => false, // The CSS ID to apply to the HTML of the nav item. 
  202.       'show_for_displayed_user' => true, // When viewing another user does this nav item show up? 
  203.       'site_admin_only' => false, // Can only site admins see this nav item? 
  204.       'position' => 99, // Index of where this nav item should be positioned. 
  205.       'screen_function' => false, // The name of the function to run when clicked. 
  206.       'default_subnav_slug' => false  // The slug of the default subnav item to select when clicked. 
  207.  ); 
  208.  
  209.   $r = wp_parse_args( $args, $defaults ); 
  210.  
  211.   // If we don't have the required info we need, don't register this screen function. 
  212.   if ( empty( $r['slug'] ) ) { 
  213.       return false; 
  214.   } 
  215.  
  216.   /** 
  217.    * If this is for site admins only and the user is not one,  
  218.    * don't register this screen function. 
  219.    */ 
  220.   if ( ! empty( $r['site_admin_only'] ) && ! bp_current_user_can( 'bp_moderate' ) ) { 
  221.       return false; 
  222.   } 
  223.  
  224.   /** 
  225.    * If this nav item is hidden for the displayed user, and 
  226.    * the logged in user is not the displayed user 
  227.    * looking at their own profile, don't don't register this screen function. 
  228.    */ 
  229.   if ( empty( $r['show_for_displayed_user'] ) && ! bp_user_has_access() ) { 
  230.       return false; 
  231.   } 
  232.  
  233.   /** 
  234.    * If the nav item is visible, we are not viewing a user, and this is a root 
  235.    * component, don't attach the default subnav function so we can display a 
  236.    * directory or something else. 
  237.    */ 
  238.   if ( ( -1 != $r['position'] ) && bp_is_root_component( $r['slug'] ) && ! bp_displayed_user_id() ) { 
  239.       return; 
  240.   } 
  241.  
  242.   // Look for current component. 
  243.   if ( bp_is_current_component( $r['slug'] ) || bp_is_current_item( $r['slug'] ) ) { 
  244.  
  245.       // The requested URL has explicitly included the default subnav 
  246.       // (eg: http://example.com/members/membername/activity/just-me/) 
  247.       // The canonical version will not contain this subnav slug. 
  248.       if ( ! empty( $r['default_subnav_slug'] ) && bp_is_current_action( $r['default_subnav_slug'] ) && ! bp_action_variable( 0 ) ) { 
  249.           unset( $bp->canonical_stack['action'] ); 
  250.       } elseif ( ! bp_current_action() ) { 
  251.  
  252.           // Add our screen hook if screen function is callable. 
  253.           if ( is_callable( $r['screen_function'] ) ) { 
  254.               add_action( 'bp_screens', $r['screen_function'], 3 ); 
  255.           } 
  256.  
  257.           if ( ! empty( $r['default_subnav_slug'] ) ) { 
  258.  
  259.               /** 
  260.                * Filters the default component subnav item. 
  261.                * 
  262.                * @since 1.5.0 
  263.                * 
  264.                * @param string $value The slug of the default subnav item 
  265.                *                      to select when clicked. 
  266.                * @param array  $r     Parsed arguments for the nav item. 
  267.                */ 
  268.               $bp->current_action = apply_filters( 'bp_default_component_subnav', $r['default_subnav_slug'], $r ); 
  269.           } 
  270.       } 
  271.   } 
  272.  
  273.   /** 
  274.    * Fires after the screen function for an item in the BuddyPress main 
  275.    * navigation is registered. 
  276.    * 
  277.    * @since 2.4.0 
  278.    * 
  279.    * @param array $r        Parsed arguments for the nav item. 
  280.    * @param array $args     Originally passed in arguments for the nav item. 
  281.    * @param array $defaults Default arguments for a nav item. 
  282.    */ 
  283.   do_action( 'bp_core_register_nav_screen_function', $r, $args, $defaults ); 
  284. }

  286. /** 
  287.  * Modify the default subnav item that loads when a top level nav item is clicked. 
  288.  * 
  289.  * @since 1.1.0 
  290.  * 
  291.  * @param array|string $args { 
  292.  *     @type string   $parent_slug     The slug of the nav item whose default is being changed. 
  293.  *     @type callable $screen_function The new default callback function that will run when the nav item is clicked. 
  294.  *     @type string   $subnav_slug     The slug of the new default subnav item. 
  295.  * } 
  296.  */ 
  297. function bp_core_new_nav_default( $args = '' ) { 
  298.   $bp = buddypress(); 
  299.  
  300.   $defaults = array( 
  301.       'parent_slug' => false, // Slug of the parent. 
  302.       'screen_function' => false, // The name of the function to run when clicked. 
  303.       'subnav_slug' => false  // The slug of the subnav item to select when clicked. 
  304.  ); 
  305.  
  306.   $r = wp_parse_args( $args, $defaults ); 
  307.  
  308.   // This is specific to Members - it's not available in Groups. 
  309.   $parent_nav = $bp->members->nav->get_primary( array( 'slug' => $r['parent_slug'] ), false ); 
  310.  
  311.   if ( ! $parent_nav ) { 
  312.       return ; 
  313.   } 
  314.  
  315.   $parent_nav = reset( $parent_nav ); 
  316.  
  317.   if ( ! empty( $parent_nav->screen_function ) ) { 
  318.       // Remove our screen hook if screen function is callable. 
  319.       if ( is_callable( $parent_nav->screen_function ) ) { 
  320.           remove_action( 'bp_screens', $parent_nav->screen_function, 3 ); 
  321.       } 
  322.   } 
  323.  
  324.   // Edit the screen function for the parent nav. 
  325.   $bp->members->nav->edit_nav( array( 
  326.       'screen_function' => &$r['screen_function'],  
  327.       'default_subnav_slug' => $r['subnav_slug'],  
  328.  ), $parent_nav->slug ); 
  329.  
  330.   if ( bp_is_current_component( $parent_nav->slug ) ) { 
  331.  
  332.       // The only way to tell whether to set the subnav is to peek at the unfiltered_uri 
  333.       // Find the component. 
  334.       $component_uri_key = array_search( $parent_nav->slug, $bp->unfiltered_uri ); 
  335.  
  336.       if ( false !== $component_uri_key ) { 
  337.           if ( ! empty( $bp->unfiltered_uri[$component_uri_key + 1] ) ) { 
  338.               $unfiltered_action = $bp->unfiltered_uri[$component_uri_key + 1]; 
  339.           } 
  340.       } 
  341.  
  342.       // No subnav item has been requested in the URL, so set a new nav default. 
  343.       if ( empty( $unfiltered_action ) ) { 
  344.           if ( ! bp_is_current_action( $r['subnav_slug'] ) ) { 
  345.               if ( is_callable( $r['screen_function'] ) ) { 
  346.                   add_action( 'bp_screens', $r['screen_function'], 3 ); 
  347.               } 
  348.  
  349.               $bp->current_action = $r['subnav_slug']; 
  350.               unset( $bp->canonical_stack['action'] ); 
  351.           } 
  352.  
  353.       // The URL is explicitly requesting the new subnav item, but should be 
  354.       // directed to the canonical URL. 
  355.       } elseif ( $unfiltered_action == $r['subnav_slug'] ) { 
  356.           unset( $bp->canonical_stack['action'] ); 
  357.  
  358.       // In all other cases (including the case where the original subnav item 
  359.       // is explicitly called in the URL), the canonical URL will contain the 
  360.       // subnav slug. 
  361.       } else { 
  362.           $bp->canonical_stack['action'] = bp_current_action(); 
  363.       } 
  364.   } 
  365.  
  366.   return; 
  367. }

  400. function bp_core_new_subnav_item( $args, $component = null ) { 
  401.   // Backward compatibility for plugins using `bp_core_new_subnav_item()` without `$component` 
  402.   // to add group subnav items. 
  403.   if ( null === $component && bp_is_active( 'groups' ) && bp_is_group() && isset( $args['parent_slug'] ) ) { 
  404.       /** 
  405.        * Assume that this item is intended to belong to the current group if: 
  406.        * a) the 'parent_slug' is the same as the slug of the current group, or 
  407.        * b) the 'parent_slug' starts with the slug of the current group, and the members nav doesn't have 
  408.        *    a primary item with that slug. 
  409.        */ 
  410.       $group_slug = bp_get_current_group_slug(); 
  411.       if ( 
  412.           $group_slug === $args['parent_slug'] || 
  413.           ( 0 === strpos( $args['parent_slug'], $group_slug ) && ! buddypress()->members->nav->get_primary( array( 'slug' => $args['parent_slug'] ), false ) ) 
  414.  ) { 
  415.           $component = 'groups'; 
  416.       } 
  417.   } 
  418.  
  419.   if ( ! $component ) { 
  420.       $component = 'members'; 
  421.   } 
  422.  
  423.   if ( ! bp_is_active( $component ) ) { 
  424.       return; 
  425.   } 
  426.  
  427.   // First, register the subnav item in the nav. 
  428.   $subnav_item = bp_core_create_subnav_link( $args, $component ); 
  429.  
  430.   /** 
  431.    * To mimic legacy behavior, if bp_core_create_subnav_link() returns false, we make an 
  432.    * early exit and don't attempt to register the screen function. 
  433.    */ 
  434.   if ( false === $subnav_item ) { 
  435.       return false; 
  436.   } 
  437.  
  438.   // Then, hook the screen function for the added subnav item. 
  439.   $hooked = bp_core_register_subnav_screen_function( $subnav_item, $component ); 
  440.   if ( false === $hooked ) { 
  441.       return false; 
  442.   } 
  443. }

  445. /** 
  446.  * Add a subnav link to the BuddyPress navigation.

  450.  * 
  451.  * @param array|string $args { 
  452.  *     Array describing the new subnav item. 
  453.  *     @type string      $name              Display name for the subnav item. 
  454.  *     @type string      $slug              Unique URL slug for the subnav item. 
  455.  *     @type string      $parent_slug       Slug of the top-level nav item under which the 
  456.  *                                          new subnav item should be added. 
  457.  *     @type string      $parent_url        URL of the parent nav item. 
  458.  *     @type bool|string $item_css_id       Optional. 'id' attribute for the nav 
  459.  *                                          item. Default: the value of $slug. 
  460.  *     @type bool        $user_has_access   Optional. True if the logged-in user has access to the 
  461.  *                                          subnav item, otherwise false. Can be set dynamically 
  462.  *                                          when registering the subnav; eg, use bp_is_my_profile() 
  463.  *                                          to restrict access to profile owners only. Default: true. 
  464.  *     @type bool        $site_admin_only   Optional. Whether the nav item should be visible only 
  465.  *                                          to site admins (those with the 'bp_moderate' cap). 
  466.  *                                          Default: false. 
  467.  *     @type int         $position          Optional. Numerical index specifying where the item 
  468.  *                                          should appear in the subnav array. Default: 90. 
  469.  *     @type callable    $screen_function   The callback function that will run 
  470.  *                                          when the nav item is clicked. 
  471.  *     @type string      $link              Optional. The URL that the subnav item should point 
  472.  *                                          to. Defaults to a value generated from the $parent_url + $slug. 
  473.  *     @type bool        $show_in_admin_bar Optional. Whether the nav item should be added into 
  474.  *                                          the group's "Edit" Admin Bar menu for group admins. 
  475.  *                                          Default: false. 
  476.  * } 
  477.  * @param string       $component The component the navigation is attached to. Defaults to 'members'. 
  478.  * @return bool|object Returns false on failure, new BP_Core_Nav_Item instance on success. 
  479.  */ 
  480. function bp_core_create_subnav_link( $args = '', $component = 'members' ) { 
  481.   $bp = buddypress(); 
  482.  
  483.   $r = wp_parse_args( $args, array( 
  484.       'name' => false, // Display name for the nav item. 
  485.       'slug' => false, // URL slug for the nav item. 
  486.       'parent_slug' => false, // URL slug of the parent nav item. 
  487.       'parent_url' => false, // URL of the parent item. 
  488.       'item_css_id' => false, // The CSS ID to apply to the HTML of the nav item. 
  489.       'user_has_access' => true, // Can the logged in user see this nav item? 
  490.       'no_access_url' => '',  
  491.       'site_admin_only' => false, // Can only site admins see this nav item? 
  492.       'position' => 90, // Index of where this nav item should be positioned. 
  493.       'screen_function' => false, // The name of the function to run when clicked. 
  494.       'link' => '', // The link for the subnav item; optional, not usually required. 
  495.       'show_in_admin_bar' => false, // Show the Manage link in the current group's "Edit" Admin Bar menu. 
  496.  ) ); 
  497.  
  498.   // If we don't have the required info we need, don't create this subnav item. 
  499.   if ( empty( $r['name'] ) || empty( $r['slug'] ) || empty( $r['parent_slug'] ) || empty( $r['parent_url'] ) || empty( $r['screen_function'] ) ) 
  500.       return false; 
  501.  
  502.   // Link was not forced, so create one. 
  503.   if ( empty( $r['link'] ) ) { 
  504.       $r['link'] = trailingslashit( $r['parent_url'] . $r['slug'] ); 
  505.  
  506.       $parent_nav = $bp->{$component}->nav->get_primary( array( 'slug' => $r['parent_slug'] ), false ); 
  507.  
  508.       // If this sub item is the default for its parent, skip the slug. 
  509.       if ( $parent_nav ) { 
  510.           $parent_nav_item = reset( $parent_nav ); 
  511.           if ( ! empty( $parent_nav_item->default_subnav_slug ) && $r['slug'] === $parent_nav_item->default_subnav_slug ) { 
  512.               $r['link'] = trailingslashit( $r['parent_url'] ); 
  513.           } 
  514.       } 
  515.   } 
  516.  
  517.   // If this is for site admins only and the user is not one, don't create the subnav item. 
  518.   if ( ! empty( $r['site_admin_only'] ) && ! bp_current_user_can( 'bp_moderate' ) ) { 
  519.       return false; 
  520.   } 
  521.  
  522.   if ( empty( $r['item_css_id'] ) ) { 
  523.       $r['item_css_id'] = $r['slug']; 
  524.   } 
  525.  
  526.   $subnav_item = array( 
  527.       'name' => $r['name'],  
  528.       'link' => $r['link'],  
  529.       'slug' => $r['slug'],  
  530.       'parent_slug' => $r['parent_slug'],  
  531.       'css_id' => $r['item_css_id'],  
  532.       'position' => $r['position'],  
  533.       'user_has_access' => $r['user_has_access'],  
  534.       'no_access_url' => $r['no_access_url'],  
  535.       'screen_function' => &$r['screen_function'],  
  536.       'show_in_admin_bar' => (bool) $r['show_in_admin_bar'],  
  537.  ); 
  538.  
  539.   buddypress()->{$component}->nav->add_nav( $subnav_item ); 
  540.  
  541.   return $subnav_item; 
  542. } 
  543.  
  544. /** 
  545.  * Register a screen function, whether or not a related subnav link exists. 
  546.  * 
  547.  * @since 2.4.0 
  548.  * @since 2.6.0 Introduced the `$component` parameter. 
  549.  * 
  550.  * @param array|string $args { 
  551.  *     Array describing the new subnav item. 
  552.  *     @type string   $slug              Unique URL slug for the subnav item. 
  553.  *     @type string   $parent_slug       Slug of the top-level nav item under which the 
  554.  *                                       new subnav item should be added. 
  555.  *     @type string   $parent_url        URL of the parent nav item. 
  556.  *     @type bool     $user_has_access   Optional. True if the logged-in user has access to the 
  557.  *                                       subnav item, otherwise false. Can be set dynamically 
  558.  *                                       when registering the subnav; eg, use bp_is_my_profile() 
  559.  *                                       to restrict access to profile owners only. Default: true. 
  560.  *     @type bool     $site_admin_only   Optional. Whether the nav item should be visible 
  561.  *                                       only to site admins (those with the 'bp_moderate' cap). 
  562.  *                                       Default: false. 
  563.  *     @type int      $position          Optional. Numerical index specifying where the item 
  564.  *                                       should appear in the subnav array. Default: 90. 
  565.  *     @type callable $screen_function   The callback function that will run 
  566.  *                                       when the nav item is clicked. 
  567.  *     @type string   $link              Optional. The URL that the subnav item should point to. 
  568.  *                                       Defaults to a value generated from the $parent_url + $slug. 
  569.  *     @type bool     $show_in_admin_bar Optional. Whether the nav item should be added into 
  570.  *                                       the group's "Edit" Admin Bar menu for group admins. 
  571.  *                                       Default: false. 
  572.  * } 
  573.  * @param string       $component The component the navigation is attached to. Defaults to 'members'. 
  574.  * @return bool|null Returns false on failure. 
  575.  */ 
  576. function bp_core_register_subnav_screen_function( $args = '', $component = 'members' ) { 
  577.   $bp = buddypress(); 
  578.  
  579.   $r = wp_parse_args( $args, array( 
  580.       'slug' => false, // URL slug for the screen. 
  581.       'parent_slug' => false, // URL slug of the parent screen. 
  582.       'user_has_access' => true, // Can the user visit this screen? 
  583.       'no_access_url' => '',  
  584.       'site_admin_only' => false, // Can only site admins visit this screen? 
  585.       'screen_function' => false, // The name of the function to run when clicked. 
  586.  ) ); 
  587.  
  588.   /** 
  589.    * Hook the screen function for the added subnav item. But this only needs to 
  590.    * be done if this subnav item is the current view, and the user has access to the 
  591.    * subnav item. We figure out whether we're currently viewing this subnav by 
  592.    * checking the following two conditions: 
  593.    *   (1) Either: 
  594.    *       (a) the parent slug matches the current_component, or 
  595.    *       (b) the parent slug matches the current_item 
  596.    *   (2) And either: 
  597.    *       (a) the current_action matches $slug, or 
  598.    *       (b) there is no current_action (ie, this is the default subnav for the parent nav) 
  599.    *       and this subnav item is the default for the parent item (which we check by 
  600.    *       comparing this subnav item's screen function with the screen function of the 
  601.    *       parent nav item in the component's primary nav). This condition only arises 
  602.    *       when viewing a user, since groups should always have an action set. 
  603.    */ 
  604.  
  605.   // If we *don't* meet condition (1), return. 
  606.   if ( ! bp_is_current_component( $r['parent_slug'] ) && ! bp_is_current_item( $r['parent_slug'] ) ) { 
  607.       return; 
  608.   } 
  609.  
  610.   $parent_nav = $bp->{$component}->nav->get_primary( array( 'slug' => $r['parent_slug'] ), false ); 
  611.  
  612.   // If we *do* meet condition (2), then the added subnav item is currently being requested. 
  613.   if ( ( bp_current_action() && bp_is_current_action( $r['slug'] ) ) || ( bp_is_user() && ! bp_current_action() && ! empty( $parent_nav->screen_function ) && $r['screen_function'] == $parent_nav->screen_function ) ) { 
  614.  
  615.       // If this is for site admins only and the user is not one, don't create the subnav item. 
  616.       if ( ! empty( $r['site_admin_only'] ) && ! bp_current_user_can( 'bp_moderate' ) ) { 
  617.           return false; 
  618.       } 
  619.  
  620.       $hooked = bp_core_maybe_hook_new_subnav_screen_function( $r, $component ); 
  621.  
  622.       // If redirect args have been returned, perform the redirect now. 
  623.       if ( ! empty( $hooked['status'] ) && 'failure' === $hooked['status'] && isset( $hooked['redirect_args'] ) ) { 
  624.           bp_core_no_access( $hooked['redirect_args'] ); 
  625.       } 
  626.   } 
  627. } 
  628.  
  629. /** 
  630.  * For a given subnav item, either hook the screen function or generate redirect arguments, as necessary. 
  631.  * 
  632.  * @since 2.1.0 
  633.  * @since 2.6.0 Introduced the `$component` parameter. 
  634.  * 
  635.  * @param array  $subnav_item The subnav array added to the secondary navigation of 
  636.  *                            the component in bp_core_new_subnav_item(). 
  637.  * @param string $component   The component the navigation is attached to. Defaults to 'members'. 
  638.  * @return array 
  639.  */ 
  640. function bp_core_maybe_hook_new_subnav_screen_function( $subnav_item, $component = 'members' ) { 
  641.   $retval = array( 
  642.       'status' => '',  
  643.  ); 
  644.  
  645.   // Is this accessible by site admins only? 
  646.   $site_admin_restricted = false; 
  647.   if ( ! empty( $subnav_item['site_admin_only'] ) && ! bp_current_user_can( 'bp_moderate' ) ) { 
  648.       $site_admin_restricted = true; 
  649.   } 
  650.  
  651.   // User has access, so let's try to hook the display callback. 
  652.   if ( ! empty( $subnav_item['user_has_access'] ) && ! $site_admin_restricted ) { 
  653.  
  654.       // Screen function is invalid. 
  655.       if ( ! is_callable( $subnav_item['screen_function'] ) ) { 
  656.           $retval['status'] = 'failure'; 
  657.  
  658.       // Success - hook to bp_screens. 
  659.       } else { 
  660.           add_action( 'bp_screens', $subnav_item['screen_function'], 3 ); 
  661.           $retval['status'] = 'success'; 
  662.       } 
  663.  
  664.   // User doesn't have access. Determine redirect arguments based on 
  665.   // user status. 
  666.   } else { 
  667.       $retval['status'] = 'failure'; 
  668.  
  669.       if ( is_user_logged_in() ) { 
  670.  
  671.           $bp = buddypress(); 
  672.  
  673.           // If a redirect URL has been passed to the subnav 
  674.           // item, respect it. 
  675.           if ( ! empty( $subnav_item['no_access_url'] ) ) { 
  676.               $message = __( 'You do not have access to this page.', 'buddypress' ); 
  677.               $redirect_to = trailingslashit( $subnav_item['no_access_url'] ); 
  678.  
  679.           // In the case of a user page, we try to assume a 
  680.           // redirect URL. 
  681.           } elseif ( bp_is_user() ) { 
  682.  
  683.               $parent_nav_default = $bp->{$component}->nav->get_primary( array( 'slug' => $bp->default_component ), false ); 
  684.               if ( $parent_nav_default ) { 
  685.                   $parent_nav_default_item = reset( $parent_nav_default ); 
  686.               } 
  687.  
  688.               // Redirect to the displayed user's default 
  689.               // component, as long as that component is 
  690.               // publicly accessible. 
  691.               if ( bp_is_my_profile() || ( isset( $parent_nav_default_item ) && $parent_nav_default_item->show_for_displayed_user ) ) { 
  692.                   $message = __( 'You do not have access to this page.', 'buddypress' ); 
  693.                   $redirect_to = bp_displayed_user_domain(); 
  694.  
  695.               // In some cases, the default tab is not accessible to 
  696.               // the logged-in user. So we fall back on a tab that we 
  697.               // know will be accessible. 
  698.               } else { 
  699.                   // Try 'activity' first. 
  700.                   if ( bp_is_active( 'activity' ) && isset( $bp->pages->activity ) ) { 
  701.                       $redirect_to = trailingslashit( bp_displayed_user_domain() . bp_get_activity_slug() ); 
  702.                   // Then try 'profile'. 
  703.                   } else { 
  704.                       $redirect_to = trailingslashit( bp_displayed_user_domain() . ( 'xprofile' == $bp->profile->id ? 'profile' : $bp->profile->id ) ); 
  705.                   } 
  706.  
  707.                   $message = ''; 
  708.               } 
  709.  
  710.           // Fall back to the home page. 
  711.           } else { 
  712.               $message = __( 'You do not have access to this page.', 'buddypress' ); 
  713.               $redirect_to = bp_get_root_domain(); 
  714.           } 
  715.  
  716.           $retval['redirect_args'] = array( 
  717.               'message' => $message,  
  718.               'root' => $redirect_to,  
  719.               'redirect' => false,  
  720.  ); 
  721.  
  722.       } else { 
  723.           // When the user is logged out, pass an empty array 
  724.           // This indicates that the default arguments should be 
  725.           // used in bp_core_no_access(). 
  726.           $retval['redirect_args'] = array(); 
  727.       } 
  728.   } 
  729.  
  730.   return $retval; 
  731. }

  733. /** 
  734.  * Check whether a given nav item has subnav items. 
  735.  * 
  736.  * @since 1.5.0 
  737.  * @since 2.6.0 Introduced the `$component` parameter. 
  738.  * 
  739.  * @param string $nav_item  The slug of the top-level nav item whose subnav items you're checking. 
  740.  *                          Default: the current component slug. 
  741.  * @param string $component The component the navigation is attached to. Defaults to 'members'. 
  742.  * @return bool $has_subnav True if the nav item is found and has subnav items; false otherwise. 
  743.  */ 
  744. function bp_nav_item_has_subnav( $nav_item = '', $component = 'members' ) { 
  745.   $bp = buddypress(); 
  746.  
  747.   if ( ! isset( $bp->{$component}->nav ) ) { 
  748.       return false; 
  749.   } 
  750.  
  751.   if ( ! $nav_item ) { 
  752.       $nav_item = bp_current_component(); 
  753.  
  754.       if ( bp_is_group() ) { 
  755.           $nav_item = bp_current_item(); 
  756.       } 
  757.   } 
  758.  
  759.   $has_subnav = (bool) $bp->{$component}->nav->get_secondary( array( 'parent_slug' => $nav_item ), false ); 
  760.  
  761.   /** 
  762.    * Filters whether or not a given nav item has subnav items. 
  763.    * 
  764.    * @since 1.5.0 
  765.    * 
  766.    * @param bool   $has_subnav Whether or not there is any subnav items. 
  767.    * @param string $nav_item   The slug of the top-level nav item whose subnav items you're checking. 
  768.    */ 
  769.   return apply_filters( 'bp_nav_item_has_subnav', $has_subnav, $nav_item ); 
  770. }

  772. /** 
  773.  * Deletes an item from the primary navigation of the specified component. 
  774.  * 
  775.  * @since 1.0.0 
  776.  * @since 2.6.0 Introduced the `$component` parameter. 
  777.  * 
  778.  * @param string $slug      The slug of the primary navigation item. 
  779.  * @param string $component The component the navigation is attached to. Defaults to 'members'. 
  780.  * @return bool Returns false on failure, True on success. 
  781.  */ 
  782. function bp_core_remove_nav_item( $slug, $component = null ) { 
  783.   $bp = buddypress(); 
  784.  
  785.   // Backward compatibility for removing group nav items using the group slug as `$parent_slug`. 
  786.   if ( ! $component && bp_is_active( 'groups' ) && isset( $bp->groups->nav ) ) { 
  787.       if ( $bp->groups->nav->get_primary( array( 'slug' => $slug ) ) ) { 
  788.           $component = 'groups'; 
  789.       } 
  790.   } 
  791.  
  792.   if ( ! $component ) { 
  793.       $component = 'members'; 
  794.   } 
  795.  
  796.   if ( ! isset( $bp->{$component}->nav ) ) { 
  797.       return false; 
  798.   } 
  799.  
  800.   $screen_functions = $bp->{$component}->nav->delete_nav( $slug ); 
  801.  
  802.   // Reset backcompat nav items so that subsequent references will be correct. 
  803.   $bp->bp_nav->reset(); 
  804.   $bp->bp_options_nav->reset(); 
  805.  
  806.   if ( ! is_array( $screen_functions ) ) { 
  807.       return false; 
  808.   } 
  809.  
  810.   foreach ( $screen_functions as $screen_function ) { 
  811.       // Remove our screen hook if screen function is callable. 
  812.       if ( is_callable( $screen_function ) ) { 
  813.           remove_action( 'bp_screens', $screen_function, 3 ); 
  814.       } 
  815.   } 
  816.  
  817.   return true; 
  818. }
