  359. /** 
  360.  * Fetch global BP options. 
  361.  * 
  362.  * BuddyPress uses common options to store configuration settings. Many of these 
  363.  * settings are needed at run time. Instead of fetching them all and adding many 
  364.  * initial queries to each page load, let's fetch them all in one go. 
  365.  * 
  366.  * @since 1.5.0 
  367.  * 
  368.  * @todo Use settings API and audit these methods. 
  369.  * 
  370.  * @return array $root_blog_options_meta List of options. 
  371.  */ 
  372. function bp_core_get_root_options() { 
  373.   global $wpdb; 
  374.  
  375.   // Get all the BuddyPress settings, and a few useful WP ones too. 
  376.   $root_blog_options = bp_get_default_options(); 
  377.   $root_blog_options['registration'] = '0'; 
  378.   $root_blog_options['avatar_default'] = 'mysteryman'; 
  379.   $root_blog_option_keys = array_keys( $root_blog_options ); 
  380.  
  381.   // Do some magic to get all the root blog options in 1 swoop 
  382.   // Check cache first - We cache here instead of using the standard WP 
  383.   // settings cache because the current blog may not be the root blog,  
  384.   // and it's not practical to access the cache across blogs. 
  385.   $root_blog_options_meta = wp_cache_get( 'root_blog_options', 'bp' ); 
  386.  
  387.   if ( false === $root_blog_options_meta ) { 
  388.       $blog_options_keys = "'" . join( "', '", (array) $root_blog_option_keys ) . "'"; 
  389.       $blog_options_table = bp_is_multiblog_mode() ? $wpdb->options : $wpdb->get_blog_prefix( bp_get_root_blog_id() ) . 'options'; 
  390.       $blog_options_query = "SELECT option_name AS name, option_value AS value FROM {$blog_options_table} WHERE option_name IN ( {$blog_options_keys} )"; 
  391.       $root_blog_options_meta = $wpdb->get_results( $blog_options_query ); 
  392.  
  393.       // On Multisite installations, some options must always be fetched from sitemeta. 
  394.       if ( is_multisite() ) { 
  395.  
  396.           /** 
  397.            * Filters multisite options retrieved from sitemeta. 
  398.            * 
  399.            * @since 1.5.0 
  400.            * 
  401.            * @param array $value Array of multisite options from sitemeta table. 
  402.            */ 
  403.           $network_options = apply_filters( 'bp_core_network_options', array( 
  404.               'tags_blog_id' => '0',  
  405.               'sitewide_tags_blog' => '',  
  406.               'registration' => '0',  
  407.               'fileupload_maxk' => '1500' 
  408.  ) ); 
  409.  
  410.           $current_site = get_current_site(); 
  411.           $network_option_keys = array_keys( $network_options ); 
  412.           $sitemeta_options_keys = "'" . join( "', '", (array) $network_option_keys ) . "'"; 
  413.           $sitemeta_options_query = $wpdb->prepare( "SELECT meta_key AS name, meta_value AS value FROM {$wpdb->sitemeta} WHERE meta_key IN ( {$sitemeta_options_keys} ) AND site_id = %d", $current_site->id ); 
  414.           $network_options_meta = $wpdb->get_results( $sitemeta_options_query ); 
  415.  
  416.           // Sitemeta comes second in the merge, so that network 'registration' value wins. 
  417.           $root_blog_options_meta = array_merge( $root_blog_options_meta, $network_options_meta ); 
  418.       } 
  419.  
  420.       // Loop through our results and make them usable. 
  421.       foreach ( $root_blog_options_meta as $root_blog_option ) { 
  422.           $root_blog_options[$root_blog_option->name] = $root_blog_option->value; 
  423.       } 
  424.  
  425.       // Copy the options no the return val. 
  426.       $root_blog_options_meta = $root_blog_options; 
  427.  
  428.       // Clean up our temporary copy. 
  429.       unset( $root_blog_options ); 
  430.  
  431.       wp_cache_set( 'root_blog_options', $root_blog_options_meta, 'bp' ); 
  432.   } 
  433.

  435.    * Filters the global BP options. 
  436.    * 
  437.    * @since 1.5.0 
  438.    * 
  439.    * @param array $root_blog_options_meta Array of global BP options. 
  440.    */ 
  441.   return apply_filters( 'bp_core_get_root_options', $root_blog_options_meta ); 
  442. }
