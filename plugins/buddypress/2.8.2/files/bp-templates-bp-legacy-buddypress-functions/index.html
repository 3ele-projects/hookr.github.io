  346.   /** 
  347.    * Get the URL and handle of a web-accessible CSS or JS asset 
  348.    * 
  349.    * We provide two levels of customizability with respect to where CSS 
  350.    * and JS files can be stored: (1) the child theme/parent theme/theme 
  351.    * compat hierarchy, and (2) the "template stack" of /buddypress/css/,  
  352.    * /community/css/, and /css/. In this way, CSS and JS assets can be 
  353.    * overloaded, and default versions provided, in exactly the same way 
  354.    * as corresponding PHP templates. 
  355.    * 
  356.    * We are duplicating some of the logic that is currently found in 
  357.    * bp_locate_template() and the _template_stack() functions. Those 
  358.    * functions were built with PHP templates in mind, and will require 
  359.    * refactoring in order to provide "stack" functionality for assets 
  360.    * that must be accessible both using file_exists() (the file path) 
  361.    * and at a public URI. 
  362.    * 
  363.    * This method is marked private, with the understanding that the 
  364.    * implementation is subject to change or removal in an upcoming 
  365.    * release, in favor of a unified _template_stack() system. Plugin 
  366.    * and theme authors should not attempt to use what follows. 
  367.    * 
  368.    * @since 1.8.0 
  369.    * @param string $file A filename like buddypress.css. 
  370.    * @param string $type Optional. Either "js" or "css" (the default). 
  371.    * @param string $script_handle Optional. If set, used as the script name in `wp_enqueue_script`. 
  372.    * @return array An array of data for the wp_enqueue_* function: 
  373.    *   'handle' (eg 'bp-child-css') and a 'location' (the URI of the 
  374.    *   asset) 
  375.    */ 
  376.   private function locate_asset_in_stack( $file, $type = 'css', $script_handle = '' ) { 
  377.       $locations = array(); 
  378.  
  379.       // Ensure the assets can be located when running from /src/. 
  380.       if ( defined( 'BP_SOURCE_SUBDIRECTORY' ) && BP_SOURCE_SUBDIRECTORY === 'src' ) { 
  381.           $file = str_replace( '.min', '', $file ); 
  382.       } 
  383.  
  384.       // No need to check child if template == stylesheet. 
  385.       if ( is_child_theme() ) { 
  386.           $locations['bp-child'] = array( 
  387.               'dir' => get_stylesheet_directory(),  
  388.               'uri' => get_stylesheet_directory_uri(),  
  389.               'file' => str_replace( '.min', '', $file ),  
  390.  ); 
  391.       } 
  392.  
  393.       $locations['bp-parent'] = array( 
  394.           'dir' => get_template_directory(),  
  395.           'uri' => get_template_directory_uri(),  
  396.           'file' => str_replace( '.min', '', $file ),  
  397.  ); 
  398.  
  399.       $locations['bp-legacy'] = array( 
  400.           'dir' => bp_get_theme_compat_dir(),  
  401.           'uri' => bp_get_theme_compat_url(),  
  402.           'file' => $file,  
  403.  ); 
  404.  
  405.       // Subdirectories within the top-level $locations directories. 
  406.       $subdirs = array( 
  407.           'buddypress/' . $type,  
  408.           'community/' . $type,  
  409.           $type,  
  410.  ); 
  411.  
  412.       $retval = array(); 
  413.  
  414.       foreach ( $locations as $location_type => $location ) { 
  415.           foreach ( $subdirs as $subdir ) { 
  416.               if ( file_exists( trailingslashit( $location['dir'] ) . trailingslashit( $subdir ) . $location['file'] ) ) { 
  417.                   $retval['location'] = trailingslashit( $location['uri'] ) . trailingslashit( $subdir ) . $location['file']; 
  418.                   $retval['handle'] = ( $script_handle ) ? $script_handle : "{$location_type}-{$type}"; 
  419.  
  420.                   break 2; 
  421.               } 
  422.           } 
  423.       } 
  424.  
  425.       return $retval; 
  426.   }

  653.  * This function looks scarier than it actually is. :)

  655.  * parameters to show specific information based on the page we are currently 
  656.  * looking at. 
  657.  * 
  658.  * The following function will take into account any cookies set in the JS and 
  659.  * allow us to override the parameters sent. That way we can change the results 
  660.  * returned without reloading the page. 
  661.  * 
  662.  * By using cookies we can also make sure that user settings are retained 
  663.  * across page loads. 
  664.  * 
  665.  * @param string $query_string Query string for the current request. 
  666.  * @param string $object       Object for cookie. 
  667.  * @return string Query string for the component loops 
  668.  * @since 1.2.0 
  669.  */ 
  670. function bp_legacy_theme_ajax_querystring( $query_string, $object ) { 
  671.   if ( empty( $object ) ) 
  672.       return ''; 
  673.  
  674.   // Set up the cookies passed on this AJAX request. Store a local var to avoid conflicts. 
  675.   if ( ! empty( $_POST['cookie'] ) ) { 
  676.       $_BP_COOKIE = wp_parse_args( str_replace( '; ', '&', urldecode( $_POST['cookie'] ) ) ); 
  677.   } else { 
  678.       $_BP_COOKIE = &$_COOKIE; 
  679.   } 
  680.  
  681.   $qs = array(); 
  682.  
  683.   /** 
  684.    * Check if any cookie values are set. If there are then override the 
  685.    * default params passed to the template loop. 
  686.    */ 
  687.  
  688.   // Activity stream filtering on action. 
  689.   if ( ! empty( $_BP_COOKIE['bp-' . $object . '-filter'] ) && '-1' != $_BP_COOKIE['bp-' . $object . '-filter'] ) { 
  690.       $qs[] = 'type=' . $_BP_COOKIE['bp-' . $object . '-filter']; 
  691.  
  692.       if ( bp_is_active( 'activity' ) ) { 
  693.           $actions = bp_activity_get_actions_for_context(); 
  694.           foreach ( $actions as $action ) { 
  695.               if ( $action['key'] === $_BP_COOKIE['bp-' . $object . '-filter'] ) { 
  696.                   $qs[] = 'action=' . $_BP_COOKIE['bp-' . $object . '-filter']; 
  697.               } 
  698.           } 
  699.       } 
  700.   } 
  701.  
  702.   if ( ! empty( $_BP_COOKIE['bp-' . $object . '-scope'] ) ) { 
  703.       if ( 'personal' == $_BP_COOKIE['bp-' . $object . '-scope'] ) { 
  704.           $user_id = ( bp_displayed_user_id() ) ? bp_displayed_user_id() : bp_loggedin_user_id(); 
  705.           $qs[] = 'user_id=' . $user_id; 
  706.       } 
  707.  
  708.       // Activity stream scope only on activity directory. 
  709.       if ( 'all' != $_BP_COOKIE['bp-' . $object . '-scope'] && ! bp_displayed_user_id() && ! bp_is_single_item() ) 
  710.           $qs[] = 'scope=' . $_BP_COOKIE['bp-' . $object . '-scope']; 
  711.   } 
  712.  
  713.   // If page and search_terms have been passed via the AJAX post request, use those. 
  714.   if ( ! empty( $_POST['page'] ) && '-1' != $_POST['page'] ) 
  715.       $qs[] = 'page=' . absint( $_POST['page'] );

  718.   if ( ! empty( $_POST['exclude_just_posted'] ) ) {

  731.

  733.   $query_string = empty( $qs ) ? '' : join( '&', (array) $qs ); 
  734.  
  735.   $object_filter = ''; 
  736.   if ( isset( $_BP_COOKIE['bp-' . $object . '-filter'] ) ) 
  737.       $object_filter = $_BP_COOKIE['bp-' . $object . '-filter']; 
  738.  
  739.   $object_scope = ''; 
  740.   if ( isset( $_BP_COOKIE['bp-' . $object . '-scope'] ) ) 
  741.       $object_scope = $_BP_COOKIE['bp-' . $object . '-scope']; 
  742.  
  743.   $object_page = ''; 
  744.   if ( isset( $_BP_COOKIE['bp-' . $object . '-page'] ) ) 
  745.       $object_page = $_BP_COOKIE['bp-' . $object . '-page']; 
  746.  
  747.   $object_search_terms = ''; 
  748.   if ( isset( $_BP_COOKIE['bp-' . $object . '-search-terms'] ) ) 
  749.       $object_search_terms = $_BP_COOKIE['bp-' . $object . '-search-terms']; 
  750.  
  751.   $object_extras = ''; 
  752.   if ( isset( $_BP_COOKIE['bp-' . $object . '-extras'] ) ) 
  753.       $object_extras = $_BP_COOKIE['bp-' . $object . '-extras']; 
  754.  
  755.   /** 
  756.    * Filters the AJAX query string for the component loops. 
  757.    * 
  758.    * @since 1.7.0 
  759.    * 
  760.    * @param string $query_string        The query string we are working with.
