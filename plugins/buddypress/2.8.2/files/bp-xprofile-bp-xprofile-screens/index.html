  45. /** 
  46.  * Handles the display of the profile edit page by loading the correct template file. 
  47.  * Also checks to make sure this can only be accessed for the logged in users profile. 
  48.  * 
  49.  * @since 1.0.0 
  50.  * 
  51.  */ 
  52. function xprofile_screen_edit_profile() { 
  53.  
  54.   if ( ! bp_is_my_profile() && ! bp_current_user_can( 'bp_moderate' ) ) { 
  55.       return false; 
  56.   } 
  57.  
  58.   // Make sure a group is set. 
  59.   if ( ! bp_action_variable( 1 ) ) { 
  60.       bp_core_redirect( trailingslashit( bp_displayed_user_domain() . bp_get_profile_slug() . '/edit/group/1' ) ); 
  61.   } 
  62.  
  63.   // Check the field group exists. 
  64.   if ( ! bp_is_action_variable( 'group' ) || ! xprofile_get_field_group( bp_action_variable( 1 ) ) ) { 
  65.       bp_do_404(); 
  66.       return; 
  67.   } 
  68.  
  69.   // No errors. 
  70.   $errors = false; 
  71.  
  72.   // Check to see if any new information has been submitted. 
  73.   if ( isset( $_POST['field_ids'] ) ) { 
  74.  
  75.       // Check the nonce. 
  76.       check_admin_referer( 'bp_xprofile_edit' ); 
  77.  
  78.       // Check we have field ID's. 
  79.       if ( empty( $_POST['field_ids'] ) ) { 
  80.           bp_core_redirect( trailingslashit( bp_displayed_user_domain() . bp_get_profile_slug() . '/edit/group/' . bp_action_variable( 1 ) ) ); 
  81.       } 
  82.  
  83.       // Explode the posted field IDs into an array so we know which 
  84.       // fields have been submitted. 
  85.       $posted_field_ids = wp_parse_id_list( $_POST['field_ids'] ); 
  86.       $is_required = array(); 
  87.  
  88.       // Loop through the posted fields formatting any datebox values then validate the field. 
  89.       foreach ( (array) $posted_field_ids as $field_id ) { 
  90.           bp_xprofile_maybe_format_datebox_post_data( $field_id ); 
  91.  
  92.           $is_required[ $field_id ] = xprofile_check_is_required_field( $field_id ) && ! bp_current_user_can( 'bp_moderate' ); 
  93.           if ( $is_required[$field_id] && empty( $_POST['field_' . $field_id] ) ) { 
  94.               $errors = true; 
  95.           } 
  96.       } 
  97.  
  98.       // There are errors. 
  99.       if ( !empty( $errors ) ) { 
  100.           bp_core_add_message( __( 'Please make sure you fill in all required fields in this profile field group before saving.', 'buddypress' ), 'error' ); 
  101.  
  102.       // No errors. 
  103.       } else { 
  104.  
  105.           // Reset the errors var. 
  106.           $errors = false; 
  107.  
  108.           // Now we've checked for required fields, lets save the values. 
  109.           $old_values = $new_values = array(); 
  110.           foreach ( (array) $posted_field_ids as $field_id ) { 
  111.  
  112.               // Certain types of fields (checkboxes, multiselects) may come through empty. Save them as an empty array so that they don't get overwritten by the default on the next edit. 
  113.               $value = isset( $_POST['field_' . $field_id] ) ? $_POST['field_' . $field_id] : ''; 
  114.  
  115.               $visibility_level = !empty( $_POST['field_' . $field_id . '_visibility'] ) ? $_POST['field_' . $field_id . '_visibility'] : 'public'; 
  116.  
  117.               // Save the old and new values. They will be 
  118.               // passed to the filter and used to determine 
  119.               // whether an activity item should be posted. 
  120.               $old_values[ $field_id ] = array( 
  121.                   'value' => xprofile_get_field_data( $field_id, bp_displayed_user_id() ),  
  122.                   'visibility' => xprofile_get_field_visibility_level( $field_id, bp_displayed_user_id() ),  
  123.  ); 
  124.  
  125.               // Update the field data and visibility level. 
  126.               xprofile_set_field_visibility_level( $field_id, bp_displayed_user_id(), $visibility_level ); 
  127.               $field_updated = xprofile_set_field_data( $field_id, bp_displayed_user_id(), $value, $is_required[ $field_id ] ); 
  128.               $value = xprofile_get_field_data( $field_id, bp_displayed_user_id() ); 
  129.  
  130.               $new_values[ $field_id ] = array( 
  131.                   'value' => $value,  
  132.                   'visibility' => xprofile_get_field_visibility_level( $field_id, bp_displayed_user_id() ),  
  133.  ); 
  134.  
  135.               if ( ! $field_updated ) { 
  136.                   $errors = true; 
  137.               } else { 
  138.  
  139.                   /** 
  140.                    * Fires on each iteration of an XProfile field being saved with no error. 
  141.                    * 
  142.                    * @since 1.1.0 
  143.                    * 
  144.                    * @param int    $field_id ID of the field that was saved. 
  145.                    * @param string $value    Value that was saved to the field. 
  146.                    */ 
  147.                   do_action( 'xprofile_profile_field_data_updated', $field_id, $value ); 
  148.               } 
  149.           } 
  150.  
  151.           /** 
  152.            * Fires after all XProfile fields have been saved for the current profile. 
  153.            * 
  154.            * @since 1.0.0 
  155.            * 
  156.            * @param int   $value            Displayed user ID. 
  157.            * @param array $posted_field_ids Array of field IDs that were edited. 
  158.            * @param bool  $errors           Whether or not any errors occurred. 
  159.            * @param array $old_values       Array of original values before updated. 
  160.            * @param array $new_values       Array of newly saved values after update. 
  161.            */ 
  162.           do_action( 'xprofile_updated_profile', bp_displayed_user_id(), $posted_field_ids, $errors, $old_values, $new_values ); 
  163.  
  164.           // Set the feedback messages. 
  165.           if ( !empty( $errors ) ) { 
  166.               bp_core_add_message( __( 'There was a problem updating some of your profile information. Please try again.', 'buddypress' ), 'error' ); 
  167.           } else { 
  168.               bp_core_add_message( __( 'Changes saved.', 'buddypress' ) ); 
  169.           } 
  170.  
  171.           // Redirect back to the edit screen to display the updates and message. 
  172.           bp_core_redirect( trailingslashit( bp_displayed_user_domain() . bp_get_profile_slug() . '/edit/group/' . bp_action_variable( 1 ) ) ); 
  173.       } 
  174.   } 
  175.  
  176.   /** 
  177.    * Fires right before the loading of the XProfile edit screen template file. 
  178.    * 
  179.    * @since 1.0.0 
  180.    */ 
  181.   do_action( 'xprofile_screen_edit_profile' ); 
  182.  
  183.   /** 
  184.    * Filters the template to load for the XProfile edit screen. 
  185.    * 
  186.    * @since 1.0.0 
  187.    * 
  188.    * @param string $template Path to the XProfile edit template to load. 
  189.    */ 
  190.   bp_core_load_template( apply_filters( 'xprofile_template_edit_profile', 'members/single/home' ) ); 
  191. }

  193. /** 
  194.  * Handles the uploading and cropping of a user avatar. Displays the change avatar page. 
  195.  * 
  196.  * @since 1.0.0 
  197.  * 
  198.  */ 
  199. function xprofile_screen_change_avatar() { 
  200.  
  201.   // Bail if not the correct screen. 
  202.   if ( ! bp_is_my_profile() && ! bp_current_user_can( 'bp_moderate' ) ) { 
  203.       return false; 
  204.   } 
  205.  
  206.   // Bail if there are action variables. 
  207.   if ( bp_action_variables() ) { 
  208.       bp_do_404(); 
  209.       return; 
  210.   } 
  211.  
  212.   $bp = buddypress(); 
  213.  
  214.   if ( ! isset( $bp->avatar_admin ) ) { 
  215.       $bp->avatar_admin = new stdClass(); 
  216.   } 
  217.  
  218.   $bp->avatar_admin->step = 'upload-image'; 
  219.  
  220.   if ( !empty( $_FILES ) ) { 
  221.  
  222.       // Check the nonce. 
  223.       check_admin_referer( 'bp_avatar_upload' ); 
  224.  
  225.       // Pass the file to the avatar upload handler. 
  226.       if ( bp_core_avatar_handle_upload( $_FILES, 'xprofile_avatar_upload_dir' ) ) { 
  227.           $bp->avatar_admin->step = 'crop-image'; 
  228.  
  229.           // Make sure we include the jQuery jCrop file for image cropping. 
  230.           add_action( 'wp_print_scripts', 'bp_core_add_jquery_cropper' ); 
  231.       } 
  232.   } 
  233.  
  234.   // If the image cropping is done, crop the image and save a full/thumb version. 
  235.   if ( isset( $_POST['avatar-crop-submit'] ) ) { 
  236.  
  237.       // Check the nonce. 
  238.       check_admin_referer( 'bp_avatar_cropstore' ); 
  239.  
  240.       $args = array( 
  241.           'item_id' => bp_displayed_user_id(),  
  242.           'original_file' => $_POST['image_src'],  
  243.           'crop_x' => $_POST['x'],  
  244.           'crop_y' => $_POST['y'],  
  245.           'crop_w' => $_POST['w'],  
  246.           'crop_h' => $_POST['h'] 
  247.  ); 
  248.  
  249.       if ( ! bp_core_avatar_handle_crop( $args ) ) { 
  250.           bp_core_add_message( __( 'There was a problem cropping your profile photo.', 'buddypress' ), 'error' ); 
  251.       } else { 
  252.  
  253.           /** 
  254.            * Fires right before the redirect, after processing a new avatar. 
  255.            * 
  256.            * @since 1.1.0 
  257.            * @since 2.3.4 Add two new parameters to inform about the user id and 
  258.            *              about the way the avatar was set (eg: 'crop' or 'camera'). 
  259.            * 
  260.            * @param string $item_id Inform about the user id the avatar was set for. 
  261.            * @param string $value   Inform about the way the avatar was set ('crop'). 
  262.            */ 
  263.           do_action( 'xprofile_avatar_uploaded', (int) $args['item_id'], 'crop' ); 
  264.           bp_core_add_message( __( 'Your new profile photo was uploaded successfully.', 'buddypress' ) ); 
  265.           bp_core_redirect( bp_displayed_user_domain() ); 
  266.       } 
  267.   } 
  268.  
  269.   /** 
  270.    * Fires right before the loading of the XProfile change avatar screen template file. 
  271.    * 
  272.    * @since 1.0.0 
  273.    */ 
  274.   do_action( 'xprofile_screen_change_avatar' ); 
  275.  
  276.   /** 
  277.    * Filters the template to load for the XProfile change avatar screen. 
  278.    * 
  279.    * @since 1.0.0 
  280.    * 
  281.    * @param string $template Path to the XProfile change avatar template to load. 
  282.    */ 
  283.   bp_core_load_template( apply_filters( 'xprofile_template_change_avatar', 'members/single/home' ) ); 
  284. }
