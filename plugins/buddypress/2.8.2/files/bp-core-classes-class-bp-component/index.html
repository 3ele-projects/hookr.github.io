  326.   /** 
  327.    * Include required files. 
  328.    * 
  329.    * Please note that, by default, this method is fired on the bp_include 
  330.    * hook, with priority 8. This is necessary so that core components are 
  331.    * loaded in time to be available to third-party plugins. However, this 
  332.    * load order means that third-party plugins whose main files are 
  333.    * loaded at bp_include with priority 10 (as recommended), will not be 
  334.    * loaded in time for their includes() method to fire automatically. 
  335.    * 
  336.    * For this reason, it is recommended that your plugin has its own 
  337.    * method or function for requiring necessary files. If you must use 
  338.    * this method, you will have to call it manually in your constructor 
  339.    * class, ie 
  340.    *   $this->includes(); 
  341.    * 
  342.    * Note that when you pass an array value like 'actions' to includes,  
  343.    * it looks for the following three files (assuming your component is 
  344.    * called 'my_component'): 
  345.    *   - ./actions 
  346.    *   - ./bp-my_component/actions 
  347.    *   - ./bp-my_component/bp-my_component-actions.php 
  348.    * 
  349.    * @since 1.5.0 
  350.    * 
  351.    * 
  352.    * @param array $includes An array of file names, or file name chunks,  
  353.    *                        to be parsed and then included. 
  354.    */ 
  355.   public function includes( $includes = array() ) { 
  356.  
  357.       // Bail if no files to include. 
  358.       if ( ! empty( $includes ) ) { 
  359.           $slashed_path = trailingslashit( $this->path ); 
  360.  
  361.           // Loop through files to be included. 
  362.           foreach ( (array) $includes as $file ) { 
  363.  
  364.               $paths = array( 
  365.  
  366.                   // Passed with no extension. 
  367.                   'bp-' . $this->id . '/bp-' . $this->id . '-' . $file  . '.php',  
  368.                   'bp-' . $this->id . '-' . $file . '.php',  
  369.                   'bp-' . $this->id . '/' . $file . '.php',  
  370.  
  371.                   // Passed with extension. 
  372.                   $file,  
  373.                   'bp-' . $this->id . '-' . $file,  
  374.                   'bp-' . $this->id . '/' . $file,  
  375.  ); 
  376.  
  377.               foreach ( $paths as $path ) { 
  378.                   if ( @is_file( $slashed_path . $path ) ) { 
  379.                       require( $slashed_path . $path ); 
  380.                       break; 
  381.                   } 
  382.               } 
  383.           } 
  384.       }

  389.        * This is a dynamic hook that is based on the component string ID. 
  390.        * 
  391.        * @since 1.5.0 
  392.        */ 
  393.       do_action( 'bp_' . $this->id . '_includes' ); 
  394.   }

  517.    *                            array is passed to {@link WP_Admin_Bar::add_menu()}. 
  518.    *                            See that method for a description of the required syntax for 
  519.    *                            each item. 
  520.    */ 
  521.   public function setup_admin_bar( $wp_admin_nav = array() ) { 
  522.  
  523.       // Bail if this is an ajax request. 
  524.       if ( defined( 'DOING_AJAX' ) ) { 
  525.           return; 
  526.       } 
  527.  
  528.       // Do not proceed if BP_USE_WP_ADMIN_BAR constant is not set or is false. 
  529.       if ( ! bp_use_wp_admin_bar() ) { 
  530.           return; 
  531.       } 
  532.  
  533.       /** 
  534.        * Filters the admin navigation passed into setup_admin_bar. 
  535.        * 
  536.        * This is a dynamic hook that is based on the component string ID. 
  537.        * 
  538.        * @since 1.9.0 
  539.        * 
  540.        * @param array $wp_admin_nav Array of navigation items to add. 
  541.        */ 
  542.       $wp_admin_nav = apply_filters( 'bp_' . $this->id . '_admin_nav', $wp_admin_nav ); 
  543.  
  544.       // Do we have Toolbar menus to add? 
  545.       if ( !empty( $wp_admin_nav ) ) { 
  546.           // Fill in position if one wasn't passed for backpat. 
  547.           $pos = 0; 
  548.           $not_set_pos = 1; 
  549.           foreach( $wp_admin_nav as $key => $nav ) { 
  550.               if ( ! isset( $nav['position'] ) ) { 
  551.                   $wp_admin_nav[$key]['position'] = $pos + $not_set_pos; 
  552.  
  553.                   if ( 9 !== $not_set_pos ) { 
  554.                       ++$not_set_pos; 
  555.                   } 
  556.               } else { 
  557.                   $pos = $nav['position']; 
  558.  
  559.                   // Reset not set pos to 1 
  560.                   if ( $pos % 10 === 0 ) { 
  561.                       $not_set_pos = 1; 
  562.                   } 
  563.               } 
  564.           } 
  565.  
  566.           // Sort admin nav by position. 
  567.           $wp_admin_nav = bp_sort_by_key( $wp_admin_nav, 'position', 'num' ); 
  568.  
  569.           // Set this objects menus. 
  570.           $this->admin_menu = $wp_admin_nav; 
  571.  
  572.           // Define the WordPress global. 
  573.           global $wp_admin_bar; 
  574.  
  575.           // Add each admin menu. 
  576.           foreach( $this->admin_menu as $admin_menu ) { 
  577.               $wp_admin_bar->add_menu( $admin_menu ); 
  578.           } 
  579.       } 
  580.  
  581.       /** 
  582.        * Fires at the end of the setup_admin_bar method inside BP_Component.

  634.   public function register_global_tables( $tables = array() ) { 
  635.  
  636.       /** 
  637.        * Filters the global tables for the component, so that it may use WordPress' database API. 
  638.        *

  677.   public function register_meta_tables( $tables = array() ) {

  681.        * Filters the global meta_tables for the component.
