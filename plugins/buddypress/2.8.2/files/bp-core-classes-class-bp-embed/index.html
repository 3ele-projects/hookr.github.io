  13. /** 
  14.  * Enable oEmbeds in BuddyPress contexts. 
  15.  * 
  16.  * Extends WP_Embed class for use with BuddyPress. 
  17.  * 
  18.  * @since 1.5.0 
  19.  * 
  20.  * @see WP_Embed 
  21.  */ 
  22. class BP_Embed extends WP_Embed { 
  23.  
  24.   /** 
  25.    * Constructor 
  26.    * 
  27.    * @global WP_Embed $wp_embed 
  28.    */ 
  29.   public function __construct() { 
  30.       global $wp_embed; 
  31.  
  32.       // Make sure we populate the WP_Embed handlers array. 
  33.       // These are providers that use a regex callback on the URL in question. 
  34.       // Do not confuse with oEmbed providers, which require an external ping. 
  35.       // Used in WP_Embed::shortcode(). 
  36.       $this->handlers = $wp_embed->handlers; 
  37.  
  38.       if ( bp_use_embed_in_activity() ) { 
  39.           add_filter( 'bp_get_activity_content_body', array( &$this, 'autoembed' ), 8 ); 
  40.           add_filter( 'bp_get_activity_content_body', array( &$this, 'run_shortcode' ), 7 ); 
  41.       } 
  42.  
  43.       if ( bp_use_embed_in_activity_replies() ) { 
  44.           add_filter( 'bp_get_activity_content', array( &$this, 'autoembed' ), 8 ); 
  45.           add_filter( 'bp_get_activity_content', array( &$this, 'run_shortcode' ), 7 ); 
  46.       } 
  47.  
  48.       if ( bp_use_embed_in_forum_posts() ) { 
  49.           add_filter( 'bp_get_the_topic_post_content', array( &$this, 'autoembed' ), 8 ); 
  50.           add_filter( 'bp_get_the_topic_post_content', array( &$this, 'run_shortcode' ), 7 ); 
  51.       } 
  52.  
  53.       if ( bp_use_embed_in_private_messages() ) { 
  54.           add_filter( 'bp_get_the_thread_message_content', array( &$this, 'autoembed' ), 8 ); 
  55.           add_filter( 'bp_get_the_thread_message_content', array( &$this, 'run_shortcode' ), 7 ); 
  56.       } 
  57.  
  58.       /** 
  59.        * Filters the BuddyPress Core oEmbed setup. 
  60.        * 
  61.        * @since 1.5.0 
  62.        * 
  63.        * @param BP_Embed $this Current instance of the BP_Embed. Passed by reference. 
  64.        */ 
  65.       do_action_ref_array( 'bp_core_setup_oembed', array( &$this ) ); 
  66.   } 
  67.  
  68.   /** 
  69.    * The {@link do_shortcode()} callback function. 
  70.    * 
  71.    * Attempts to convert a URL into embed HTML. Starts by checking the 
  72.    * URL against the regex of the registered embed handlers. Next, checks 
  73.    * the URL against the regex of registered {@link WP_oEmbed} providers 
  74.    * if oEmbed discovery is false. If none of the regex matches and it's 
  75.    * enabled, then the URL will be passed to {@link BP_Embed::parse_oembed()} 
  76.    * for oEmbed parsing. 
  77.    * 
  78.    * 
  79.    * @param array  $attr Shortcode attributes. 
  80.    * @param string $url  The URL attempting to be embeded. 
  81.    * @return string The embed HTML on success, otherwise the original URL. 
  82.    */ 
  83.   public function shortcode( $attr, $url = '' ) { 
  84.       if ( empty( $url ) ) 
  85.           return ''; 
  86.  
  87.       $rawattr = $attr; 
  88.       $attr = wp_parse_args( $attr, wp_embed_defaults() ); 
  89.  
  90.       // Use kses to convert & into & and we need to undo this 
  91.       // See https://core.trac.wordpress.org/ticket/11311. 
  92.       $url = str_replace( '&', '&', $url ); 
  93.  
  94.       // Look for known internal handlers. 
  95.       ksort( $this->handlers ); 
  96.       foreach ( $this->handlers as $priority => $handlers ) { 
  97.           foreach ( $handlers as $hid => $handler ) { 
  98.               if ( preg_match( $handler['regex'], $url, $matches ) && is_callable( $handler['callback'] ) ) { 
  99.                   if ( false !== $return = call_user_func( $handler['callback'], $matches, $attr, $url, $rawattr ) ) { 
  100.  
  101.                       /** 
  102.                        * Filters the oEmbed handler result for the provided URL. 
  103.                        * 
  104.                        * @since 1.5.0 
  105.                        * 
  106.                        * @param string $return Handler callback for the oEmbed. 
  107.                        * @param string $url    URL attempting to be embedded. 
  108.                        * @param array  $attr   Shortcode attributes. 
  109.                        */ 
  110.                       return apply_filters( 'embed_handler_html', $return, $url, $attr ); 
  111.                   } 
  112.               } 
  113.           } 
  114.       } 
  115.  
  116.       /** 
  117.        * Filters the embed object ID. 
  118.        * 
  119.        * @since 1.5.0 
  120.        * 
  121.        * @param int $value Value of zero. 
  122.        */ 
  123.       $id = apply_filters( 'embed_post_id', 0 ); 
  124.  
  125.       $unfiltered_html = current_user_can( 'unfiltered_html' ); 
  126.       $default_discovery = false; 
  127.  
  128.       // Since 4.4, WordPress is now an oEmbed provider. 
  129.       if ( function_exists( 'wp_oembed_register_route' ) ) { 
  130.           $unfiltered_html = true; 
  131.           $default_discovery = true; 
  132.       } 
  133.  
  134.       /** 
  135.        * Filters whether or not oEmbed discovery is on. 
  136.        * 
  137.        * @since 1.5.0 
  138.        * @since 2.5.0 Default status of oEmbed discovery has been switched 
  139.        *              to true to apply changes introduced in WordPress 4.4 
  140.        * 
  141.        * @param bool $default_discovery Current status of oEmbed discovery. 
  142.        */ 
  143.       $attr['discover'] = ( apply_filters( 'bp_embed_oembed_discover', $default_discovery ) && $unfiltered_html ); 
  144.  
  145.       // Set up a new WP oEmbed object to check URL with registered oEmbed providers. 
  146.       require_once( ABSPATH . WPINC . '/class-oembed.php' ); 
  147.       $oembed_obj = _wp_oembed_get_object(); 
  148.  
  149.       // If oEmbed discovery is true, skip oEmbed provider check. 
  150.       $is_oembed_link = false; 
  151.       if ( !$attr['discover'] ) { 
  152.           foreach ( (array) $oembed_obj->providers as $provider_matchmask => $provider ) { 
  153.               $regex = ( $is_regex = $provider[1] ) ? $provider_matchmask : '#' . str_replace( '___wildcard___', '(.+)', preg_quote( str_replace( '*', '___wildcard___', $provider_matchmask ), '#' ) ) . '#i'; 
  154.  
  155.               if ( preg_match( $regex, $url ) ) 
  156.                   $is_oembed_link = true; 
  157.           } 
  158.  
  159.           // If url doesn't match a WP oEmbed provider, stop parsing. 
  160.           if ( !$is_oembed_link ) 
  161.               return $this->maybe_make_link( $url ); 
  162.       } 
  163.  
  164.       return $this->parse_oembed( $id, $url, $attr, $rawattr ); 
  165.   } 
  166.  
  167.   /** 
  168.    * Base function so BP components/plugins can parse links to be embedded. 
  169.    * 
  170.    * View an example to add support in {@link bp_activity_embed()}. 
  171.    * 
  172.    *       on success. 
  173.    *       oEmbed failure. 
  174.    * 
  175.    * @param int    $id      ID to do the caching for. 
  176.    * @param string $url     The URL attempting to be embedded. 
  177.    * @param array  $attr    Shortcode attributes from {@link WP_Embed::shortcode()}. 
  178.    * @param array  $rawattr Untouched shortcode attributes from 
  179.    *                        {@link WP_Embed::shortcode()}. 
  180.    * @return string The embed HTML on success, otherwise the original URL. 
  181.    */ 
  182.   public function parse_oembed( $id, $url, $attr, $rawattr ) { 
  183.       $id = intval( $id ); 
  184.  
  185.       if ( $id ) { 
  186.           // Setup the cachekey. 
  187.           $cachekey = '_oembed_' . md5( $url . serialize( $attr ) ); 
  188.  
  189.           // Let components / plugins grab their cache. 
  190.           $cache = ''; 
  191.  
  192.           /** 
  193.            * Filters the cache value to be used in the oEmbed, if exists. 
  194.            * 
  195.            * @since 1.5.0 
  196.            * 
  197.            * @param string $cache    Empty initial cache value. 
  198.            * @param int    $id       ID that the caching is for. 
  199.            * @param string $cachekey Key to use for the caching in the database. 
  200.            * @param string $url      The URL attempting to be embedded. 
  201.            * @param array  $attr     Parsed shortcode attributes. 
  202.            * @param array  $rawattr  Unparsed shortcode attributes. 
  203.            */ 
  204.           $cache = apply_filters( 'bp_embed_get_cache', $cache, $id, $cachekey, $url, $attr, $rawattr ); 
  205.  
  206.           // Grab cache and return it if available. 
  207.           if ( !empty( $cache ) ) { 
  208.  
  209.               /** 
  210.                * Filters the found cache for the provided URL. 
  211.                * 
  212.                * @since 1.5.0 
  213.                * 
  214.                * @param string $cache   Cached HTML markup for embed. 
  215.                * @param string $url     The URL being embedded. 
  216.                * @param array  $attr    Parsed shortcode attributes. 
  217.                * @param array  $rawattr Unparased shortcode attributes. 
  218.                */ 
  219.               return apply_filters( 'bp_embed_oembed_html', $cache, $url, $attr, $rawattr ); 
  220.  
  221.           // If no cache, ping the oEmbed provider and cache the result. 
  222.           } else { 
  223.               $html = wp_oembed_get( $url, $attr ); 
  224.               $cache = ( $html ) ? $html : $url; 
  225.  
  226.               /** 
  227.                * Fires if there is no existing cache and triggers cache setting. 
  228.                * 
  229.                * Lets components / plugins save their cache. 
  230.                * 
  231.                * @since 1.5.0 
  232.                * 
  233.                * @param string $cache    Newly cached HTML markup for embed. 
  234.                * @param string $cachekey Key to use for the caching in the database. 
  235.                * @param int    $id       ID to do the caching for. 
  236.                */ 
  237.               do_action( 'bp_embed_update_cache', $cache, $cachekey, $id ); 
  238.  
  239.               // If there was a result, return it. 
  240.               if ( $html ) { 
  241.  
  242.                   /** This filter is documented in bp-core/classes/class-bp-embed.php */ 
  243.                   return apply_filters( 'bp_embed_oembed_html', $html, $url, $attr, $rawattr ); 
  244.               } 
  245.           } 
  246.       } 
  247.  
  248.       // Still unknown. 
  249.       return $this->maybe_make_link( $url ); 
  250.   } 
  251. }
