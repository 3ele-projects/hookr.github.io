  147. /** 
  148.  * Determine which items from a list do not have cached values. 
  149.  * 
  150.  * @since 2.0.0 
  151.  * 
  152.  * @param array  $item_ids    ID list. 
  153.  * @param string $cache_group The cache group to check against. 
  154.  * @return array 
  155.  */ 
  156. function bp_get_non_cached_ids( $item_ids, $cache_group ) { 
  157.   $uncached = array(); 
  158.  
  159.   foreach ( $item_ids as $item_id ) { 
  160.       $item_id = (int) $item_id; 
  161.       if ( false === wp_cache_get( $item_id, $cache_group ) ) { 
  162.           $uncached[] = $item_id; 
  163.       } 
  164.   } 
  165.  
  166.   return $uncached; 
  167. }

  169. /** 
  170.  * Update the metadata cache for the specified objects. 
  171.  * 
  172.  * Based on WordPress's {@link update_meta_cache()}, this function primes the 
  173.  * cache with metadata related to a set of objects. This is typically done when 
  174.  * querying for a loop of objects; pre-fetching metadata for each queried 
  175.  * object can lead to dramatic performance improvements when using metadata 
  176.  * in the context of template loops. 
  177.  * 
  178.  * @since 1.6.0 
  179.  * 
  180.  * @global object $wpdb WordPress database object for queries.. 
  181.  * 
  182.  * @param array $args { 
  183.  *     Array of arguments. 
  184.  *     @type array|string $object_ids       List of object IDs to fetch metadata for. 
  185.  *                                          Accepts an array or a comma-separated list of numeric IDs. 
  186.  *     @type string       $object_type      The type of object, eg 'groups' or 'activity'. 
  187.  *     @type string       $meta_table       The name of the metadata table being queried. 
  188.  *     @type string       $object_column    Optional. The name of the database column where IDs 
  189.  *                                          (those provided by $object_ids) are found. Eg, 'group_id' 
  190.  *                                          for the groups metadata tables. Default: $object_type . '_id'. 
  191.  *     @type string       $cache_key_prefix Optional. The prefix to use when creating 
  192.  *                                          cache key names. Default: the value of $meta_table. 
  193.  * } 
  194.  * @return array|bool Metadata cache for the specified objects, or false on failure. 
  195.  */ 
  196. function bp_update_meta_cache( $args = array() ) { 
  197.   global $wpdb; 
  198.  
  199.   $defaults = array( 
  200.       'object_ids' => array(), // Comma-separated list or array of item ids. 
  201.       'object_type' => '', // Canonical component id: groups, members, etc. 
  202.       'cache_group' => '', // Cache group. 
  203.       'meta_table' => '', // Name of the table containing the metadata. 
  204.       'object_column' => '', // DB column for the object ids (group_id, etc). 
  205.       'cache_key_prefix' => ''       // Prefix to use when creating cache key names. Eg 'bp_groups_groupmeta'. 
  206.  ); 
  207.   $r = wp_parse_args( $args, $defaults ); 
  208.   extract( $r ); 
  209.  
  210.   if ( empty( $object_ids ) || empty( $object_type ) || empty( $meta_table ) || empty( $cache_group ) ) { 
  211.       return false; 
  212.   } 
  213.  
  214.   if ( empty( $cache_key_prefix ) ) { 
  215.       $cache_key_prefix = $meta_table; 
  216.   } 
  217.  
  218.   if ( empty( $object_column ) ) { 
  219.       $object_column = $object_type . '_id'; 
  220.   } 
  221.  
  222.   if ( ! $cache_group ) { 
  223.       return false; 
  224.   } 
  225.  
  226.   $object_ids = wp_parse_id_list( $object_ids ); 
  227.   $uncached_ids = bp_get_non_cached_ids( $object_ids, $cache_group ); 
  228.  
  229.   $cache = array(); 
  230.  
  231.   // Get meta info. 
  232.   if ( ! empty( $uncached_ids ) ) { 
  233.       $id_list = join( ', ', wp_parse_id_list( $uncached_ids ) ); 
  234.       $meta_list = $wpdb->get_results( esc_sql( "SELECT {$object_column}, meta_key, meta_value FROM {$meta_table} WHERE {$object_column} IN ({$id_list})" ), ARRAY_A ); 
  235.  
  236.       if ( ! empty( $meta_list ) ) { 
  237.           foreach ( $meta_list as $metarow ) { 
  238.               $mpid = intval( $metarow[$object_column] ); 
  239.               $mkey = $metarow['meta_key']; 
  240.               $mval = $metarow['meta_value']; 
  241.  
  242.               // Force subkeys to be array type. 
  243.               if ( !isset( $cache[$mpid] ) || !is_array( $cache[$mpid] ) ) 
  244.                   $cache[$mpid] = array(); 
  245.               if ( !isset( $cache[$mpid][$mkey] ) || !is_array( $cache[$mpid][$mkey] ) ) 
  246.                   $cache[$mpid][$mkey] = array(); 
  247.  
  248.               // Add a value to the current pid/key. 
  249.               $cache[$mpid][$mkey][] = $mval; 
  250.           } 
  251.       } 
  252.  
  253.       foreach ( $uncached_ids as $uncached_id ) { 
  254.           // Cache empty values as well. 
  255.           if ( ! isset( $cache[ $uncached_id ] ) ) { 
  256.               $cache[ $uncached_id ] = array(); 
  257.           } 
  258.  
  259.           wp_cache_set( $uncached_id, $cache[ $uncached_id ], $cache_group ); 
  260.       } 
  261.   } 
  262.  
  263.   return $cache; 
  264. }

  278. function bp_core_get_incremented_cache( $key, $group ) { 
  279.   $cache_key = bp_core_get_incremented_cache_key( $key, $group ); 
  280.   return wp_cache_get( $cache_key, $group ); 
  281. }

  283. /** 
  284.  * Caches a value using an incremented key. 
  285.  * 
  286.  * An "incremented key" is a cache key that is hashed with a unique incrementor,  
  287.  * allowing for bulk invalidation. 
  288.  * 
  289.  * Use this method when caching data that should be invalidated whenever any 
  290.  * object of a given type is created, updated, or deleted. This usually means 
  291.  * data related to object queries, which can only reliably cached until the 
  292.  * underlying set of objects has been modified. See, eg, BP_Activity_Activity::get(). 
  293.  * 
  294.  * @since 2.7.0 
  295.  * 
  296.  * @param string $key   Unique key for the query. Usually a SQL string. 
  297.  * @param string $group Cache group. Eg 'bp_activity'. 
  298.  * @param array  $ids   Array of IDs. 
  299.  * @return bool 
  300.  */ 
  301. function bp_core_set_incremented_cache( $key, $group, $ids ) { 
  302.   $cache_key = bp_core_get_incremented_cache_key( $key, $group ); 
  303.   return wp_cache_set( $cache_key, $ids, $group ); 
  304. }

  306. /** 
  307.  * Gets the key to be used when caching a value using an incremented cache key. 
  308.  * 
  309.  * The $key is hashed with a component-specific incrementor, which is used to 
  310.  * invalidate multiple caches at once. 
  311.   
  312.  * @since 2.7.0 
  313.  * 
  314.  * @param string $key   Unique key for the query. Usually a SQL string. 
  315.  * @param string $group Cache group. Eg 'bp_activity'. 
  316.  * @return string 
  317.  */ 
  318. function bp_core_get_incremented_cache_key( $key, $group ) { 
  319.   $incrementor = bp_core_get_incrementor( $group ); 
  320.   $cache_key = md5( $key . $incrementor ); 
  321.   return $cache_key; 
  322. }

  324. /** 
  325.  * Gets a group-specific cache incrementor. 
  326.  * 
  327.  * The incrementor is paired with query identifiers (like SQL strings) to 
  328.  * create cache keys that can be invalidated en masse. 
  329.  * 
  330.  * If an incrementor does not yet exist for the given `$group`, one will 
  331.  * be created. 
  332.  * 
  333.  * @since 2.7.0 
  334.  * 
  335.  * @param string $group Cache group. Eg 'bp_activity'. 
  336.  * @return string 
  337.  */ 
  338. function bp_core_get_incrementor( $group ) { 
  339.   $incrementor = wp_cache_get( 'incrementor', $group ); 
  340.   if ( ! $incrementor ) { 
  341.       $incrementor = microtime(); 
  342.       wp_cache_set( 'incrementor', $incrementor, $group ); 
  343.   } 
  344.  
  345.   return $incrementor; 
  346. }

  348. /** 
  349.  * Reset a group-specific cache incrementor. 
  350.  * 
  351.  * Call this function when all incrementor-based caches associated with a given 
  352.  * cache group should be invalidated. 
  353.  * 
  354.  * @since 2.7.0 
  355.  * 
  356.  * @param string $group Cache group. Eg 'bp_activity'. 
  357.  * @return bool True on success, false on failure. 
  358.  */ 
  359. function bp_core_reset_incrementor( $group ) { 
  360.   return wp_cache_delete( 'incrementor', $group );
