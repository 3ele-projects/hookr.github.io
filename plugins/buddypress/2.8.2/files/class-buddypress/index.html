  541.

  549.  
  550.       $component = null; 
  551.  
  552.       // First check to see if the class is one without a properly namespaced name. 
  553.       if ( isset( $irregular_map[ $class ] ) ) { 
  554.           $component = $irregular_map[ $class ]; 
  555.  
  556.       // Next chunk is usually the component name. 
  557.       } elseif ( in_array( $class_parts[1], $components, true ) ) { 
  558.           $component = $class_parts[1]; 
  559.       } 
  560.  
  561.       if ( ! $component ) { 
  562.           return;

  567.  
  568.       $path = dirname( __FILE__ ) . "/bp-{$component}/classes/class-{$class}.php"; 
  569.  
  570.       // Sanity check. 
  571.       if ( ! file_exists( $path ) ) { 
  572.           return; 
  573.       } 
  574.  
  575.       /** 
  576.        * Sanity check 2 - Check if component is active before loading class. 
  577.        * Skip if PHPUnit is running, or BuddyPress is installing for the first time. 
  578.        */ 
  579.       if ( 
  580.           ! in_array( $component, array( 'core', 'members' ), true ) && 
  581.           ! bp_is_active( $component ) && 
  582.           ! function_exists( 'tests_add_filter' ) 
  583.  ) { 
  584.           return; 
  585.       } 
  586.  
  587.       require $path; 
  588.   }

  590.   /** 
  591.    * Set up the default hooks and actions. 
  592.    * 
  593.    * @since 1.6.0 
  594.    * 
  595.    */ 
  596.   private function setup_actions() { 
  597.  
  598.       // Add actions to plugin activation and deactivation hooks 
  599.       add_action( 'activate_'   . $this->basename, 'bp_activation' ); 
  600.       add_action( 'deactivate_' . $this->basename, 'bp_deactivation' ); 
  601.  
  602.       // If BuddyPress is being deactivated, do not add any actions 
  603.       if ( bp_is_deactivation( $this->basename ) ) { 
  604.           return; 
  605.       } 
  606.  
  607.       // Array of BuddyPress core actions 
  608.       $actions = array( 
  609.           'setup_theme', // Setup the default theme compat 
  610.           'setup_current_user', // Setup currently logged in user 
  611.           'register_post_types', // Register post types 
  612.           'register_post_statuses', // Register post statuses 
  613.           'register_taxonomies', // Register taxonomies 
  614.           'register_views', // Register the views 
  615.           'register_theme_directory', // Register the theme directory 
  616.           'register_theme_packages', // Register bundled theme packages (bp-themes) 
  617.           'load_textdomain', // Load textdomain 
  618.           'add_rewrite_tags', // Add rewrite tags 
  619.           'generate_rewrite_rules'    // Generate rewrite rules 
  620.  ); 
  621.  
  622.       // Add the actions 
  623.       foreach( $actions as $class_action ) { 
  624.           if ( method_exists( $this, $class_action ) ) { 
  625.               add_action( 'bp_' . $class_action, array( $this, $class_action ), 5 ); 
  626.           } 
  627.       } 
  628.  
  629.       /** 
  630.        * Fires after the setup of all BuddyPress actions. 
  631.        * 
  632.        * Includes bbp-core-hooks.php. 
  633.        * 
  634.        * @since 1.7.0 
  635.        * 
  636.        * @param BuddyPress $this. Current BuddyPress instance. Passed by reference. 
  637.        */ 
  638.       do_action_ref_array( 'bp_after_setup_actions', array( &$this ) ); 
  639.   }

  641.   /** 
  642.    * Private method to align the active and database versions. 
  643.    * 
  644.    * @since 1.7.0 
  645.    */ 
  646.   private function versions() { 
  647.  
  648.       // Get the possible DB versions (boy is this gross) 
  649.       $versions = array(); 
  650.       $versions['1.6-single'] = get_blog_option( $this->root_blog_id, '_bp_db_version' ); 
  651.  
  652.       // 1.6-single exists, so trust it 
  653.       if ( !empty( $versions['1.6-single'] ) ) { 
  654.           $this->db_version_raw = (int) $versions['1.6-single']; 
  655.  
  656.       // If no 1.6-single exists, use the max of the others 
  657.       } else { 
  658.           $versions['1.2'] = get_site_option(                      'bp-core-db-version' ); 
  659.           $versions['1.5-multi'] = get_site_option(                           'bp-db-version' ); 
  660.           $versions['1.6-multi'] = get_site_option(                          '_bp_db_version' ); 
  661.           $versions['1.5-single'] = get_blog_option( $this->root_blog_id, 'bp-db-version' ); 
  662.  
  663.           // Remove empty array items 
  664.           $versions = array_filter( $versions ); 
  665.           $this->db_version_raw = (int) ( !empty( $versions ) ) ? (int) max( $versions ) : 0; 
  666.       } 
  667.   }
