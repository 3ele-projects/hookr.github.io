  728.     /** 
  729.      * Get the user id associated with a given activity item. 
  730.      * 
  731.      * Wraps bp_activity_get_specific(), with some additional logic for 
  732.      * avoiding duplicate queries. 
  733.      * 
  734.      * @since 1.6.0 
  735.      * 
  736.      * @param int $activity_id Activity ID to retrieve User ID for. 
  737.      * @return int User ID of the activity item in question. 
  738.      */ 
  739.     protected function get_activity_user_id( $activity_id ) { 
  740.         // If there is an existing activity/user ID mapping, just return the user ID. 
  741.         if ( ! empty( $this->activity_user_id[$activity_id] ) ) { 
  742.             return $this->activity_user_id[$activity_id]; 
  743.  
  744.         /** 
  745.          * We don't have a mapping. This means the $activity_id is not on the current 
  746.          * page of results, so fetch its details from the database. 
  747.          */ 
  748.         } else { 
  749.             $activity = bp_activity_get_specific( array( 'activity_ids' => $activity_id, 'show_hidden' => true, 'spam' => 'all', ) ); 
  750.  
  751.             /** 
  752.              * If, somehow, the referenced activity has been deleted, leaving its associated 
  753.              * activities as orphans, use the logged in user's ID to avoid errors. 
  754.              */ 
  755.             if ( empty( $activity['activities'] ) ) 
  756.                 return bp_loggedin_user_id(); 
  757.  
  758.             // Store the new activity/user ID mapping for any later re-use. 
  759.             $this->activity_user_id[ $activity['activities'][0]->id ] = $activity['activities'][0]->user_id; 
  760.  
  761.             // Return the user ID. 
  762.             return $activity['activities'][0]->user_id; 
  763.         }

  766.     /** 
  767.      * Checks if an activity item can be replied to. 
  768.      * 
  769.      * This method merges functionality from {@link bp_activity_can_comment()} and 
  770.      * {@link bp_blogs_disable_activity_commenting()}. This is done because the activity 
  771.      * list table doesn't use a BuddyPress activity loop, which prevents those 
  772.      * functions from working as intended. 
  773.      * 
  774.      * @since 2.0.0 
  775.      * @since 2.5.0 Include Post type activities types 
  776.      * 
  777.      * @param array $item An array version of the BP_Activity_Activity object. 
  778.      * @return bool $can_comment 
  779.      */ 
  780.     protected function can_comment( $item ) { 
  781.         $can_comment = bp_activity_type_supports( $item['type'], 'comment-reply' ); 
  782.  
  783.         if ( ! $this->disable_blogforum_comments && bp_is_active( 'blogs' ) ) { 
  784.             $parent_activity = false; 
  785.  
  786.             if ( bp_activity_type_supports( $item['type'], 'post-type-comment-tracking' ) ) { 
  787.                 $parent_activity = (object) $item; 
  788.             } elseif ( 'activity_comment' === $item['type'] ) { 
  789.                 $parent_activity = new BP_Activity_Activity( $item['item_id'] ); 
  790.                 $can_comment = bp_activity_can_comment_reply( (object) $item ); 
  791.             } 
  792.  
  793.             if ( isset( $parent_activity->type ) && bp_activity_post_type_get_tracking_arg( $parent_activity->type, 'post_type' ) ) { 
  794.                 // Fetch blog post comment depth and if the blog post's comments are open. 
  795.                 bp_blogs_setup_activity_loop_globals( $parent_activity ); 
  796.  
  797.                 $can_comment = bp_blogs_can_comment_reply( true, $item ); 
  798.             } 
  799.         }

  808.          * @param array $item        An array version of the BP_Activity_Activity object.

  810.         return apply_filters( 'bp_activity_list_table_can_comment', $can_comment, $item );

  829.                     $tree[] = $child; 
  830.                 } 
  831.  
  832.                 unset( $node->children ); 
  833.             } 
  834.         } 
  835.
