  599.     public static function search_users( $search_terms, $limit = null, $page = 1, $populate_extras = true ) { 
  600.         global $wpdb; 
  601.  
  602.         $bp = buddypress(); 
  603.  
  604.         $user_ids = array(); 
  605.         $pag_sql = $limit && $page ? $wpdb->prepare( " LIMIT %d, %d", intval( ( $page - 1 ) * intval( $limit ) ), intval( $limit ) ) : ''; 
  606.  
  607.         $search_terms_like = '%' . bp_esc_like( $search_terms ) . '%'; 
  608.         $status_sql = bp_core_get_status_sql( 'u.' ); 
  609.  
  610.         /** 
  611.          * Filters the SQL used to query for searched users count. 
  612.          * 
  613.          * @since 1.0.0 
  614.          * 
  615.          * @param string $value SQL statement for the searched users count query. 
  616.          */ 
  617.         $total_users_sql = apply_filters( 'bp_core_search_users_count_sql', $wpdb->prepare( "SELECT COUNT(DISTINCT u.ID) as id FROM {$wpdb->users} u LEFT JOIN {$bp->profile->table_name_data} pd ON u.ID = pd.user_id WHERE {$status_sql} AND pd.value LIKE %s ORDER BY pd.value ASC", $search_terms_like ), $search_terms ); 
  618.  
  619.         /** 
  620.          * Filters the SQL used to query for searched users. 
  621.          * 
  622.          * @since 1.0.0 
  623.          * 
  624.          * @param string $value SQL statement for the searched users query. 
  625.          */ 
  626.         $paged_users_sql = apply_filters( 'bp_core_search_users_sql', $wpdb->prepare( "SELECT DISTINCT u.ID as id, u.user_registered, u.user_nicename, u.user_login, u.user_email FROM {$wpdb->users} u LEFT JOIN {$bp->profile->table_name_data} pd ON u.ID = pd.user_id WHERE {$status_sql} AND pd.value LIKE %s ORDER BY pd.value ASC{$pag_sql}", $search_terms_like ), $search_terms, $pag_sql ); 
  627.  
  628.         $total_users = $wpdb->get_var( $total_users_sql ); 
  629.         $paged_users = $wpdb->get_results( $paged_users_sql ); 
  630.  
  631.         /** 
  632.          * Lets fetch some other useful data in a separate queries, this will be faster than querying the data for every user in a list. 
  633.          * We can't add these to the main query above since only users who have this information will be returned (since the much of the data is in usermeta and won't support any type of directional join) 
  634.          */ 
  635.         foreach ( (array) $paged_users as $user ) 
  636.             $user_ids[] = $user->id; 
  637.  
  638.         // Add additional data to the returned results. 
  639.         if ( $populate_extras ) 
  640.             $paged_users = BP_Core_User::get_user_extras( $paged_users, $user_ids ); 
  641.  
  642.         return array( 'users' => $paged_users, 'total' => $total_users ); 
  643.     }

  645.     /** 
  646.      * Fetch extra user information, such as friend count and last profile update message. 
  647.      * 
  648.      * Accepts multiple user IDs to fetch data for. 
  649.      * 
  650.      * @global wpdb $wpdb WordPress database object. 
  651.      * 
  652.      * @param array       $paged_users An array of stdClass containing the users. 
  653.      * @param string      $user_ids    The user ids to select information about. 
  654.      * @param string|bool $type        The type of fields we wish to get. 
  655.      * @return mixed False on error, otherwise associative array of results. 
  656.      */ 
  657.     public static function get_user_extras( &$paged_users, &$user_ids, $type = false ) { 
  658.         global $wpdb; 
  659.  
  660.         $bp = buddypress(); 
  661.  
  662.         if ( empty( $user_ids ) ) 
  663.             return $paged_users; 
  664.  
  665.         // Sanitize user IDs. 
  666.         $user_ids = implode( ', ', wp_parse_id_list( $user_ids ) ); 
  667.  
  668.         // Fetch the user's full name. 
  669.         if ( bp_is_active( 'xprofile' ) && 'alphabetical' != $type ) { 
  670.             $names = $wpdb->get_results( $wpdb->prepare( "SELECT pd.user_id as id, pd.value as fullname FROM {$bp->profile->table_name_fields} pf, {$bp->profile->table_name_data} pd WHERE pf.id = pd.field_id AND pf.name = %s AND pd.user_id IN ( {$user_ids} )", bp_xprofile_fullname_field_name() ) ); 
  671.             for ( $i = 0, $count = count( $paged_users ); $i < $count; ++$i ) { 
  672.                 foreach ( (array) $names as $name ) { 
  673.                     if ( $name->id == $paged_users[$i]->id ) 
  674.                         $paged_users[$i]->fullname = $name->fullname; 
  675.                 } 
  676.             } 
  677.         } 
  678.  
  679.         // Fetch the user's total friend count. 
  680.         if ( 'popular' != $type ) { 
  681.             $friend_count = $wpdb->get_results( $wpdb->prepare( "SELECT user_id as id, meta_value as total_friend_count FROM {$wpdb->usermeta} WHERE meta_key = %s AND user_id IN ( {$user_ids} )", bp_get_user_meta_key( 'total_friend_count' ) ) ); 
  682.             for ( $i = 0, $count = count( $paged_users ); $i < $count; ++$i ) { 
  683.                 foreach ( (array) $friend_count as $fcount ) { 
  684.                     if ( $fcount->id == $paged_users[$i]->id ) 
  685.                         $paged_users[$i]->total_friend_count = (int) $fcount->total_friend_count; 
  686.                 } 
  687.             } 
  688.         } 
  689.  
  690.         // Fetch whether or not the user is a friend. 
  691.         if ( bp_is_active( 'friends' ) ) { 
  692.             $friend_status = $wpdb->get_results( $wpdb->prepare( "SELECT initiator_user_id, friend_user_id, is_confirmed FROM {$bp->friends->table_name} WHERE (initiator_user_id = %d AND friend_user_id IN ( {$user_ids} ) ) OR (initiator_user_id IN ( {$user_ids} ) AND friend_user_id = %d )", bp_loggedin_user_id(), bp_loggedin_user_id() ) ); 
  693.             for ( $i = 0, $count = count( $paged_users ); $i < $count; ++$i ) { 
  694.                 foreach ( (array) $friend_status as $status ) { 
  695.                     if ( $status->initiator_user_id == $paged_users[$i]->id || $status->friend_user_id == $paged_users[$i]->id ) 
  696.                         $paged_users[$i]->is_friend = $status->is_confirmed; 
  697.                 } 
  698.             } 
  699.         } 
  700.  
  701.         // Fetch the user's last_activity. 
  702.         if ( 'active' != $type ) { 
  703.             $user_activity = $wpdb->get_results( $wpdb->prepare( "SELECT user_id as id, meta_value as last_activity FROM {$wpdb->usermeta} WHERE meta_key = %s AND user_id IN ( {$user_ids} )", bp_get_user_meta_key( 'last_activity' ) ) ); 
  704.             for ( $i = 0, $count = count( $paged_users ); $i < $count; ++$i ) { 
  705.                 foreach ( (array) $user_activity as $activity ) { 
  706.                     if ( $activity->id == $paged_users[$i]->id ) 
  707.                         $paged_users[$i]->last_activity = $activity->last_activity; 
  708.                 } 
  709.             } 
  710.         } 
  711.  
  712.         // Fetch the user's latest update. 
  713.         $user_update = $wpdb->get_results( $wpdb->prepare( "SELECT user_id as id, meta_value as latest_update FROM {$wpdb->usermeta} WHERE meta_key = %s AND user_id IN ( {$user_ids} )", bp_get_user_meta_key( 'bp_latest_update' ) ) ); 
  714.         for ( $i = 0, $count = count( $paged_users ); $i < $count; ++$i ) { 
  715.             foreach ( (array) $user_update as $update ) { 
  716.                 if ( $update->id == $paged_users[$i]->id ) 
  717.                     $paged_users[$i]->latest_update = $update->latest_update; 
  718.             } 
  719.         } 
  720.  
  721.         return $paged_users; 
  722.     }

  741.     /** 
  742.      * Get last activity data for a user or set of users. 
  743.      * 
  744.      * @param int|array $user_id User IDs or multiple user IDs. 
  745.      * @return array 
  746.      */ 
  747.     public static function get_last_activity( $user_id ) { 
  748.         global $wpdb; 
  749.  
  750.         // Sanitize and remove empty values. 
  751.         $user_ids = array_filter( wp_parse_id_list( $user_id ) ); 
  752.  
  753.         if ( empty( $user_ids ) ) { 
  754.             return false; 
  755.         } 
  756.  
  757.         $uncached_user_ids = bp_get_non_cached_ids( $user_ids, 'bp_last_activity' ); 
  758.         if ( ! empty( $uncached_user_ids ) ) { 
  759.             $bp = buddypress(); 
  760.  
  761.             $user_ids_sql = implode( ', ', $uncached_user_ids ); 
  762.             $user_count = count( $uncached_user_ids ); 
  763.  
  764.             $last_activities = $wpdb->get_results( $wpdb->prepare( "SELECT id, user_id, date_recorded FROM {$bp->members->table_name_last_activity} WHERE component = %s AND type = 'last_activity' AND user_id IN ({$user_ids_sql}) LIMIT {$user_count}", $bp->members->id ) ); 
  765.  
  766.             foreach ( $last_activities as $last_activity ) { 
  767.                 wp_cache_set( $last_activity->user_id, array( 
  768.                     'user_id' => $last_activity->user_id,  
  769.                     'date_recorded' => $last_activity->date_recorded,  
  770.                     'activity_id' => $last_activity->id,  
  771.  ), 'bp_last_activity' ); 
  772.             } 
  773.         } 
  774.  
  775.         // Fetch all user data from the cache. 
  776.         $retval = array(); 
  777.         foreach ( $user_ids as $user_id ) { 
  778.             $retval[ $user_id ] = wp_cache_get( $user_id, 'bp_last_activity' ); 
  779.  
  780.             if ( isset( $retval['user_id'] ) ) { 
  781.                 $retval[ $user_id ]['user_id'] = (int) $retval[ $user_id ]['user_id']; 
  782.             } 
  783.             if ( isset( $retval['activity_id'] ) ) { 
  784.                 $retval[ $user_id ]['activity_id'] = (int) $retval[ $user_id ]['activity_id']; 
  785.             } 
  786.         } 
  787.  
  788.         return $retval; 
  789.     }
