  978.     /** 
  979.      * Writes the given alterations to file 
  980.      * 
  981.      * @param $file_source string The full path to the file to be read from 
  982.      * @param $file_target string The full path to the file to be written to 
  983.      * @param $alterations array An array of arrays containing alterations to be made 
  984.      * @return void 
  985.      */ 
  986.     function write_lines_to_file( $file_source, $file_target, $alterations ) 
  987.     { 
  988.         if ( !$file_source || !file_exists( $file_source ) || !is_file( $file_source ) ) { 
  989.             return -1; 
  990.         } 
  991.  
  992.         if ( !$file_target ) { 
  993.             $file_target = $file_source; 
  994.         } 
  995.  
  996.         if ( !$alterations || !is_array( $alterations ) ) { 
  997.             return -2; 
  998.         } 
  999.  
  1000.         /** 
  1001.         Alterations array takes the form 
  1002.         array( 
  1003.             '1st 20 chars of line' => array( 'Search string', 'Replacement string' ),  
  1004.             '1st 20 chars of line' => array( 'Search string', 'Replacement string' ) 
  1005.  ); 
  1006.         */ 
  1007.  
  1008.         // Get the existing lines in the file 
  1009.         $lines = file( $file_source ); 
  1010.  
  1011.         // Initialise an array to store the modified lines 
  1012.         $modified_lines = array(); 
  1013.  
  1014.         // Loop through the lines and modify them 
  1015.         foreach ( $lines as $line ) { 
  1016.             if ( isset( $alterations[substr( $line, 0, 20 )] ) ) { 
  1017.                 $alteration = $alterations[substr( $line, 0, 20 )]; 
  1018.                 $modified_lines[] = str_replace( $alteration[0], $alteration[1], $line ); 
  1019.             } else { 
  1020.                 $modified_lines[] = $line; 
  1021.             } 
  1022.         } 
  1023.  
  1024.         $writable = true; 
  1025.         if ( file_exists( $file_target ) ) { 
  1026.             if ( !is_writable( $file_target ) ) { 
  1027.                 $writable = false; 
  1028.             } 
  1029.         } else { 
  1030.             $dir_target = dirname( $file_target ); 
  1031.             if ( file_exists( $dir_target ) ) { 
  1032.                 if ( !is_writable( $dir_target ) || !is_dir( $dir_target ) ) { 
  1033.                     $writable = false; 
  1034.                 } 
  1035.             } else { 
  1036.                 $writable = false;
