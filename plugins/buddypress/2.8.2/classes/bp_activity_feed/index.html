  299.         // Cache-related variables.

  301.         $modified_timestamp = strtotime( $last_modified );

  313.         if ( isset( $_SERVER['HTTP_IF_NONE_MATCH'] ) ) { 
  314.             $client_etag = wp_unslash( $_SERVER['HTTP_IF_NONE_MATCH'] ); 
  315.

  319.             // Strip suffixes from ETag if they exist (eg. "-gzip").

  329.  
  330.         // Get client last modified timestamp from supported user agents.

  335.         if ( ( $client_last_modified && $client_etag ) ?
