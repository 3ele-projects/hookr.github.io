  568.             ! in_array( $component, array( 'core', 'members' ), true ) && 
  569.             ! bp_is_active( $component ) && 
  570.             ! function_exists( 'tests_add_filter' ) 
  571.  ) { 
  572.             return; 
  573.         } 
  574.  
  575.         require $path; 
  576.     }

  578.     /** 
  579.      * Set up the default hooks and actions. 
  580.      * 
  581.      * @since 1.6.0 
  582.      * 
  583.      */ 
  584.     private function setup_actions() { 
  585.  
  586.         // Add actions to plugin activation and deactivation hooks 
  587.         add_action( 'activate_'   . $this->basename, 'bp_activation' ); 
  588.         add_action( 'deactivate_' . $this->basename, 'bp_deactivation' ); 
  589.  
  590.         // If BuddyPress is being deactivated, do not add any actions 
  591.         if ( bp_is_deactivation( $this->basename ) ) { 
  592.             return; 
  593.         } 
  594.  
  595.         // Array of BuddyPress core actions 
  596.         $actions = array( 
  597.             'setup_theme', // Setup the default theme compat 
  598.             'setup_current_user', // Setup currently logged in user 
  599.             'register_post_types', // Register post types 
  600.             'register_post_statuses', // Register post statuses 
  601.             'register_taxonomies', // Register taxonomies 
  602.             'register_views', // Register the views 
  603.             'register_theme_directory', // Register the theme directory 
  604.             'register_theme_packages', // Register bundled theme packages (bp-themes) 
  605.             'load_textdomain', // Load textdomain 
  606.             'add_rewrite_tags', // Add rewrite tags 
  607.             'generate_rewrite_rules'    // Generate rewrite rules 
  608.  ); 
  609.  
  610.         // Add the actions 
  611.         foreach( $actions as $class_action ) { 
  612.             if ( method_exists( $this, $class_action ) ) { 
  613.                 add_action( 'bp_' . $class_action, array( $this, $class_action ), 5 ); 
  614.             } 
  615.         } 
  616.  
  617.         /** 
  618.          * Fires after the setup of all BuddyPress actions. 
  619.          * 
  620.          * Includes bbp-core-hooks.php. 
  621.          * 
  622.          * @since 1.7.0 
  623.          * 
  624.          * @param BuddyPress $this. Current BuddyPress instance. Passed by reference. 
  625.          */ 
  626.         do_action_ref_array( 'bp_after_setup_actions', array( &$this ) ); 
  627.     }

  629.     /** 
  630.      * Private method to align the active and database versions. 
  631.      * 
  632.      * @since 1.7.0 
  633.      */ 
  634.     private function versions() { 
  635.  
  636.         // Get the possible DB versions (boy is this gross) 
  637.         $versions = array(); 
  638.         $versions['1.6-single'] = get_blog_option( $this->root_blog_id, '_bp_db_version' ); 
  639.  
  640.         // 1.6-single exists, so trust it 
  641.         if ( !empty( $versions['1.6-single'] ) ) { 
  642.             $this->db_version_raw = (int) $versions['1.6-single']; 
  643.  
  644.         // If no 1.6-single exists, use the max of the others 
  645.         } else { 
  646.             $versions['1.2'] = get_site_option(                      'bp-core-db-version' ); 
  647.             $versions['1.5-multi'] = get_site_option(                           'bp-db-version' ); 
  648.             $versions['1.6-multi'] = get_site_option(                          '_bp_db_version' ); 
  649.             $versions['1.5-single'] = get_blog_option( $this->root_blog_id, 'bp-db-version' ); 
  650.  
  651.             // Remove empty array items 
  652.             $versions = array_filter( $versions ); 
  653.             $this->db_version_raw = (int) ( !empty( $versions ) ) ? (int) max( $versions ) : 0; 
  654.         } 
  655.     }
