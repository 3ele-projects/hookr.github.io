  379.         $this->last_activity->akismet_submission = $activity_data;

  381.         // Spam. 
  382.         if ( 'true' == $activity_data['bp_as_result'] ) {

  463.         } else { 
  464.             bp_activity_update_meta( $activity->id, '_bp_akismet_error', bp_core_current_time() ); 
  465.             $this->update_activity_history( $activity->id, sprintf( __( 'Akismet was unable to check this item (response: %s), will automatically retry again later.', 'buddypress' ), $this->last_activity->akismet_submission['bp_as_result'] ), 'check-error' ); 
  466.         } 
  467.  
  468.         // Record the original data which was submitted to Akismet for checking. 
  469.         bp_activity_update_meta( $activity->id, '_bp_akismet_submission', $this->last_activity->akismet_submission ); 
  470.     }

  472.     /** 
  473.      * Contact Akismet to check if this is spam or ham. 
  474.      * 
  475.      * Props to WordPress core Akismet plugin for a lot of this. 
  476.      * 
  477.      * @since 1.6.0 
  478.      * 
  479.      * @param array  $activity_data Packet of information to submit to Akismet. 
  480.      * @param string $check         "check" or "submit". 
  481.      * @param string $spam          "spam" or "ham". 
  482.      * @return array $activity_data Activity data, with Akismet data added. 
  483.      */ 
  484.     public function send_akismet_request( $activity_data, $check = 'check', $spam = 'spam' ) { 
  485.         $query_string = $path = ''; 
  486.  
  487.         $activity_data['blog'] = bp_get_option( 'home' ); 
  488.         $activity_data['blog_charset'] = bp_get_option( 'blog_charset' ); 
  489.         $activity_data['blog_lang'] = get_locale(); 
  490.         $activity_data['referrer'] = $_SERVER['HTTP_REFERER']; 
  491.         $activity_data['user_agent'] = bp_core_current_user_ua(); 
  492.         $activity_data['user_ip'] = bp_core_current_user_ip(); 
  493.  
  494.         if ( Akismet::is_test_mode() ) 
  495.             $activity_data['is_test'] = 'true'; 
  496.  
  497.         // Loop through _POST args and rekey strings. 
  498.         foreach ( $_POST as $key => $value ) 
  499.             if ( is_string( $value ) && 'cookie' != $key ) 
  500.                 $activity_data['POST_' . $key] = $value; 
  501.  
  502.         // Keys to ignore. 
  503.         $ignore = array( 'HTTP_COOKIE', 'HTTP_COOKIE2', 'PHP_AUTH_PW' ); 
  504.  
  505.         // Loop through _SERVER args and remove whitelisted keys. 
  506.         foreach ( $_SERVER as $key => $value ) { 
  507.  
  508.             // Key should not be ignored. 
  509.             if ( !in_array( $key, $ignore ) && is_string( $value ) ) { 
  510.                 $activity_data[$key] = $value; 
  511.  
  512.             // Key should be ignored. 
  513.             } else { 
  514.                 $activity_data[$key] = ''; 
  515.             } 
  516.         } 
  517.  
  518.         foreach ( $activity_data as $key => $data ) 
  519.             $query_string .= $key . '=' . urlencode( stripslashes( $data ) ) . '&'; 
  520.  
  521.         if ( 'check' == $check ) 
  522.             $path = 'comment-check'; 
  523.         elseif ( 'submit' == $check ) 
  524.             $path = 'submit-' . $spam; 
  525.  
  526.         // Send to Akismet. 
  527.         add_filter( 'akismet_ua', array( $this, 'buddypress_ua' ) ); 
  528.         $response = Akismet::http_post( $query_string, $path ); 
  529.         remove_filter( 'akismet_ua', array( $this, 'buddypress_ua' ) ); 
  530.  
  531.         // Get the response. 
  532.         if ( ! empty( $response[1] ) && ! is_wp_error( $response[1] ) ) 
  533.             $activity_data['bp_as_result'] = $response[1]; 
  534.         else 
  535.             $activity_data['bp_as_result'] = false; 
  536.  
  537.         // Perform a daily tidy up. 
  538.         if ( ! wp_next_scheduled( 'bp_activity_akismet_delete_old_metadata' ) ) 
  539.             wp_schedule_event( time(), 'daily', 'bp_activity_akismet_delete_old_metadata' ); 
  540.  
  541.         return $activity_data; 
  542.     }

  545.      * Filters user agent when sending to Akismet to add BuddyPress info. 
  546.      * 
  547.      * @since 1.6.0 
  548.      * 
  549.      * @param string $user_agent User agent string, as generated by Akismet. 
  550.      * @return string $user_agent Modified user agent string.

  552.     public function buddypress_ua( $user_agent ) {
