  535.     /** 
  536.      * Fetch blog data not caught in the main query and append it to results array. 
  537.      * 
  538.      * Gets the following information, which is either unavailable at the 
  539.      * time of the original query, or is more efficient to look up in one 
  540.      * fell swoop: 
  541.      *   - The latest post for each blog, include Featured Image data 
  542.      *   - The blog description 
  543.      * 
  544.      * @param array       $paged_blogs Array of results from the original query. 
  545.      * @param array       $blog_ids    Array of IDs returned from the original query. 
  546.      * @param string|bool $type        Not currently used. Default: false. 
  547.      * @return array $paged_blogs The located blogs array, with the extras added. 
  548.      */ 
  549.     public static function get_blog_extras( &$paged_blogs, &$blog_ids, $type = false ) { 
  550.         global $wpdb; 
  551.  
  552.         $bp = buddypress(); 
  553.  
  554.         if ( empty( $blog_ids ) ) 
  555.             return $paged_blogs; 
  556.  
  557.         $blog_ids = implode( ', ', wp_parse_id_list( $blog_ids ) ); 
  558.  
  559.         for ( $i = 0, $count = count( $paged_blogs ); $i < $count; ++$i ) { 
  560.             $blog_prefix = $wpdb->get_blog_prefix( $paged_blogs[$i]->blog_id ); 
  561.             $paged_blogs[$i]->latest_post = $wpdb->get_row( "SELECT ID, post_content, post_title, post_excerpt, guid FROM {$blog_prefix}posts WHERE post_status = 'publish' AND post_type = 'post' AND id != 1 ORDER BY id DESC LIMIT 1" ); 
  562.             $images = array(); 
  563.  
  564.             // Add URLs to any Featured Image this post might have. 
  565.             if ( ! empty( $paged_blogs[$i]->latest_post ) && has_post_thumbnail( $paged_blogs[$i]->latest_post->ID ) ) { 
  566.  
  567.                 // Grab 4 sizes of the image. Thumbnail. 
  568.                 $image = wp_get_attachment_image_src( get_post_thumbnail_id( $paged_blogs[$i]->latest_post->ID ), 'thumbnail', false ); 
  569.                 if ( ! empty( $image ) ) 
  570.                     $images['thumbnail'] = $image[0]; 
  571.  
  572.                 // Medium. 
  573.                 $image = wp_get_attachment_image_src( get_post_thumbnail_id( $paged_blogs[$i]->latest_post->ID ), 'medium', false ); 
  574.                 if ( ! empty( $image ) ) 
  575.                     $images['medium'] = $image[0]; 
  576.  
  577.                 // Large. 
  578.                 $image = wp_get_attachment_image_src( get_post_thumbnail_id( $paged_blogs[$i]->latest_post->ID ), 'large', false ); 
  579.                 if ( ! empty( $image ) ) 
  580.                     $images['large'] = $image[0]; 
  581.  
  582.                 // Post thumbnail. 
  583.                 $image = wp_get_attachment_image_src( get_post_thumbnail_id( $paged_blogs[$i]->latest_post->ID ), 'post-thumbnail', false ); 
  584.                 if ( ! empty( $image ) ) 
  585.                     $images['post-thumbnail'] = $image[0]; 
  586.  
  587.                 // Add the images to the latest_post object. 
  588.                 $paged_blogs[$i]->latest_post->images = $images; 
  589.             } 
  590.         } 
  591.  
  592.         /** Fetch the blog description for each blog (as it may be empty we can't fetch it in the main query). */ 
  593.         $blog_descs = $wpdb->get_results( "SELECT blog_id, meta_value as description FROM {$bp->blogs->table_name_blogmeta} WHERE meta_key = 'description' AND blog_id IN ( {$blog_ids} )" ); 
  594.  
  595.         for ( $i = 0, $count = count( $paged_blogs ); $i < $count; ++$i ) { 
  596.             foreach ( (array) $blog_descs as $desc ) { 
  597.                 if ( $desc->blog_id == $paged_blogs[$i]->blog_id ) 
  598.                     $paged_blogs[$i]->description = $desc->description; 
  599.             } 
  600.         } 
  601.  
  602.         return $paged_blogs; 
  603.     }
