  603.     /** 
  604.      * Register global tables for the component, so that it may use WordPress's database API. 
  605.      * 
  606.      * @since 2.0.0 
  607.      * 
  608.      * @param array $tables Table names to register. 
  609.      */ 
  610.     public function register_global_tables( $tables = array() ) { 
  611.  
  612.         /** 
  613.          * Filters the global tables for the component, so that it may use WordPress' database API. 
  614.          * 
  615.          * This is a dynamic hook that is based on the component string ID. 
  616.          * It allows for component-specific filtering of table names. To filter 
  617.          * *all* tables, use the 'bp_core_get_table_prefix' filter instead. 
  618.          * 
  619.          * @since 1.6.0 
  620.          */ 
  621.         $tables = apply_filters( 'bp_' . $this->id . '_global_tables', $tables ); 
  622.  
  623.         // Add to the BuddyPress global object. 
  624.         if ( !empty( $tables ) && is_array( $tables ) ) { 
  625.             foreach ( $tables as $global_name => $table_name ) { 
  626.                 $this->$global_name = $table_name; 
  627.             } 
  628.  
  629.             // Keep a record of the metadata tables in the component. 
  630.             $this->global_tables = $tables; 
  631.         } 
  632.  
  633.         /** 
  634.          * Fires at the end of the register_global_tables method inside BP_Component. 
  635.          * 
  636.          * This is a dynamic hook that is based on the component string ID. 
  637.          * 
  638.          * @since 2.0.0 
  639.          */ 
  640.         do_action( 'bp_' . $this->id . '_register_global_tables' ); 
  641.     }

  643.     /** 
  644.      * Register component metadata tables. 
  645.      * 
  646.      * Metadata tables are registered in the $wpdb global, for 
  647.      * compatibility with the WordPress metadata API. 
  648.      * 
  649.      * @since 2.0.0 
  650.      * 
  651.      * @param array $tables Table names to register. 
  652.      */ 
  653.     public function register_meta_tables( $tables = array() ) { 
  654.         global $wpdb; 
  655.  
  656.         /** 
  657.          * Filters the global meta_tables for the component. 
  658.          * 
  659.          * This is a dynamic hook that is based on the component string ID. 
  660.          * It allows for component-specific filtering of table names. To filter 
  661.          * *all* tables, use the 'bp_core_get_table_prefix' filter instead. 
  662.          * 
  663.          * @since 2.0.0 
  664.          */ 
  665.         $tables = apply_filters( 'bp_' . $this->id . '_meta_tables', $tables ); 
  666.  
  667.         /** 
  668.          * Add the name of each metadata table to WPDB to allow BuddyPress 
  669.          * components to play nicely with the WordPress metadata API. 
  670.          */ 
  671.         if ( !empty( $tables ) && is_array( $tables ) ) { 
  672.             foreach( $tables as $meta_prefix => $table_name ) { 
  673.                 $wpdb->{$meta_prefix . 'meta'} = $table_name; 
  674.             } 
  675.  
  676.             // Keep a record of the metadata tables in the component. 
  677.             $this->meta_tables = $tables; 
  678.         } 
  679.  
  680.         /** 
  681.          * Fires at the end of the register_meta_tables method inside BP_Component. 
  682.          * 
  683.          * This is a dynamic hook that is based on the component string ID. 
  684.          * 
  685.          * @since 2.0.0 
  686.          */ 
  687.         do_action( 'bp_' . $this->id . '_register_meta_tables' ); 
  688.     }
