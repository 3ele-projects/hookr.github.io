  577.     public static function check_is_friend( $initiator_userid, $possible_friend_userid ) { 
  578.         global $wpdb; 
  579.  
  580.         if ( empty( $initiator_userid ) || empty( $possible_friend_userid ) ) {

  582.         } 
  583.

  585.         if ( $initiator_userid == $possible_friend_userid ) { 
  586.             return 'not_friends'; 
  587.         } 
  588.  
  589.         /** 
  590.          * Find friendships where the possible_friend_userid is the 
  591.          * initiator or friend. 
  592.          */ 
  593.         $args = array( 
  594.             'initiator_user_id' => $possible_friend_userid,  
  595.             'friend_user_id' => $possible_friend_userid 
  596.  );

  601.             if ( ! $friendship->is_confirmed ) {

  605.             }

  849.     public static function get_invitable_friend_count( $user_id, $group_id ) {

  852.         $is_group_admin = groups_is_user_admin( $user_id, $group_id );

  856.         for ( $i = 0, $count = count( $friend_ids ); $i < $count; ++$i ) {

  865.                 continue; 
  866.             } 
  867.  
  868.             // If user is not group admin and friend is banned, they cannot be invited. 
  869.             if ( ( false === $is_group_admin ) && groups_is_user_banned( (int) $friend_ids[$i], $group_id ) ) { 
  870.                 continue; 
  871.             } 
  872.  
  873.             $invitable_count++; 
  874.         }
