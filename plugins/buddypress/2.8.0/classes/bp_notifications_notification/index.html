  574.     /** 
  575.      * Get notifications, based on provided filter parameters. 
  576.      * 
  577.      * @since 1.9.0 
  578.      * 
  579.      * @param array $args { 
  580.      *     Associative array of arguments. All arguments but $page and 
  581.      *     $per_page can be treated as filter values for get_where_sql() 
  582.      *     and get_query_clauses(). All items are optional. 
  583.      *     @type int|array    $id                ID of notification being updated. Can be an 
  584.      *                                           array of IDs. 
  585.      *     @type int|array    $user_id           ID of user being queried. Can be an 
  586.      *                                           array of user IDs. 
  587.      *     @type int|array    $item_id           ID of associated item. Can be an array 
  588.      *                                           of multiple item IDs. 
  589.      *     @type int|array    $secondary_item_id ID of secondary associated 
  590.      *                                           item. Can be an array of multiple IDs. 
  591.      *     @type string|array $component_name    Name of the component to 
  592.      *                                           filter by. Can be an array of component names. 
  593.      *     @type string|array $component_action  Name of the action to 
  594.      *                                           filter by. Can be an array of actions. 
  595.      *     @type bool         $is_new            Whether to limit to new notifications. True 
  596.      *                                           returns only new notifications, false returns only non-new 
  597.      *                                           notifications. 'both' returns all. Default: true. 
  598.      *     @type string       $search_terms      Term to match against component_name 
  599.      *                                           or component_action fields. 
  600.      *     @type string       $order_by          Database column to order notifications by. 
  601.      *     @type string       $sort_order        Either 'ASC' or 'DESC'. 
  602.      *     @type string       $order_by          Field to order results by. 
  603.      *     @type string       $sort_order        ASC or DESC. 
  604.      *     @type int          $page              Number of the current page of results. Default: 
  605.      *                                           false (no pagination - all items). 
  606.      *     @type int          $per_page          Number of items to show per page. Default: 
  607.      *                                           false (no pagination - all items). 
  608.      *     @type array        $meta_query        Array of meta_query conditions. See WP_Meta_Query::queries. 
  609.      *     @type array        $date_query        Array of date_query conditions. See first parameter of 
  610.      *                                           WP_Date_Query::__construct(). 
  611.      *     @type bool         $update_meta_cache Whether to update meta cache. Default: true. 
  612.      * } 
  613.      * @return array Located notifications. 
  614.      */ 
  615.     public static function get( $args = array() ) { 
  616.         global $wpdb; 
  617.  
  618.         // Parse the arguments. 
  619.         $r = self::parse_args( $args ); 
  620.  
  621.         // Get BuddyPress. 
  622.         $bp = buddypress(); 
  623.  
  624.         // METADATA. 
  625.         $meta_query_sql = self::get_meta_query_sql( $r['meta_query'] ); 
  626.  
  627.         // SELECT. 
  628.         $select_sql = "SELECT *"; 
  629.  
  630.         // FROM. 
  631.         $from_sql = "FROM {$bp->notifications->table_name} n "; 
  632.  
  633.         // JOIN. 
  634.         $join_sql = $meta_query_sql['join']; 
  635.  
  636.         // WHERE. 
  637.         $where_sql = self::get_where_sql( array( 
  638.             'id' => $r['id'],  
  639.             'user_id' => $r['user_id'],  
  640.             'item_id' => $r['item_id'],  
  641.             'secondary_item_id' => $r['secondary_item_id'],  
  642.             'component_name' => $r['component_name'],  
  643.             'component_action' => $r['component_action'],  
  644.             'is_new' => $r['is_new'],  
  645.             'search_terms' => $r['search_terms'],  
  646.             'date_query' => $r['date_query'] 
  647.  ), $select_sql, $from_sql, $join_sql, $meta_query_sql ); 
  648.  
  649.         // ORDER BY. 
  650.         $order_sql = self::get_order_by_sql( array( 
  651.             'order_by' => $r['order_by'],  
  652.             'sort_order' => $r['sort_order'] 
  653.  ) ); 
  654.  
  655.         // LIMIT %d, %d. 
  656.         $pag_sql = self::get_paged_sql( array( 
  657.             'page' => $r['page'],  
  658.             'per_page' => $r['per_page'] 
  659.  ) ); 
  660.  
  661.         // Concatenate query parts. 
  662.         $sql = "{$select_sql} {$from_sql} {$join_sql} {$where_sql} {$order_sql} {$pag_sql}"; 
  663.  
  664.         $results = $wpdb->get_results( $sql ); 
  665.  
  666.         // Integer casting. 
  667.         foreach ( $results as $key => $result ) { 
  668.             $results[$key]->id = (int) $results[$key]->id; 
  669.             $results[$key]->user_id = (int) $results[$key]->user_id; 
  670.             $results[$key]->item_id = (int) $results[$key]->item_id; 
  671.             $results[$key]->secondary_item_id = (int) $results[$key]->secondary_item_id; 
  672.             $results[$key]->is_new = (int) $results[$key]->is_new; 
  673.         } 
  674.  
  675.         // Update meta cache. 
  676.         if ( true === $r['update_meta_cache'] ) { 
  677.             bp_notifications_update_meta_cache( wp_list_pluck( $results, 'id' ) ); 
  678.         } 
  679.  
  680.         return $results; 
  681.     }

  734.     /** 
  735.      * Get the SQL for the 'meta_query' param in BP_Notifications_Notification::get(). 
  736.      * 
  737.      * We use WP_Meta_Query to do the heavy lifting of parsing the 
  738.      * meta_query array and creating the necessary SQL clauses. However,  
  739.      * since BP_Notifications_Notification::get() builds its SQL differently than 
  740.      * WP_Query, we have to alter the return value (stripping the leading 
  741.      * AND keyword from the 'where' clause). 
  742.      * 
  743.      * @since 2.3.0 
  744.      * 
  745.      * @param  array $meta_query An array of meta_query filters. See the 
  746.      *                           documentation for WP_Meta_Query for details. 
  747.      * @return array $sql_array 'join' and 'where' clauses. 
  748.      */ 
  749.     public static function get_meta_query_sql( $meta_query = array() ) { 
  750.  
  751.         // Default array keys & empty values. 
  752.         $sql_array = array( 
  753.             'join' => '',  
  754.             'where' => '',  
  755.  ); 
  756.  
  757.         // Bail if no meta query. 
  758.         if ( empty( $meta_query ) ) { 
  759.             return $sql_array; 
  760.         } 
  761.  
  762.         // WP_Meta_Query expects the table name at $wpdb->notificationmeta. 
  763.         $GLOBALS['wpdb']->notificationmeta = buddypress()->notifications->table_name_meta; 
  764.  
  765.         $n_meta_query = new WP_Meta_Query( $meta_query ); 
  766.         $meta_sql = $n_meta_query->get_sql( 'notification', 'n', 'id' ); 
  767.  
  768.         // Strip the leading AND - it's handled in get(). 
  769.         $sql_array['where'] = preg_replace( '/^\sAND/', '', $meta_sql['where'] ); 
  770.         $sql_array['join'] = $meta_sql['join']; 
  771.  
  772.         return $sql_array; 
  773.     }
