BP_XProfile_Meta_Query

Class for generating SQL clauses that filter a primary query according to XProfile metadata keys and values.

  57.                 $existing_meta_query,  
  58.  );

  60.             $meta_query = array( 
  61.                 $primary_meta_query,  
  62.  ); 
  63.         } elseif ( ! empty( $existing_meta_query ) ) {

  128.      *

  132.      *

  138.      *     Array containing JOIN and WHERE SQL clauses to append to a first-order query.

  159.             '=', '!=', '>', '>=', '<', '<=',  
  160.             'LIKE', 'NOT LIKE',  
  161.             'IN', 'NOT IN',  
  162.             'BETWEEN', 'NOT BETWEEN',  
  163.             'EXISTS', 'NOT EXISTS',  
  164.             'REGEXP', 'NOT REGEXP', 'RLIKE' 
  165.  ) ) ) {

  167.         } 
  168.  
  169.         $meta_compare = $clause['compare']; 
  170.  
  171.         // First build the JOIN clause, if one is required. 
  172.         $join = ''; 
  173.  
  174.         // We prefer to avoid joins if possible. Look for an existing join compatible with this clause. 
  175.         $alias = $this->find_compatible_table_alias( $clause, $parent_query ); 
  176.         if ( false === $alias ) { 
  177.             $i = count( $this->table_aliases ); 
  178.             $alias = $i ? 'mt' . $i : $this->meta_table; 
  179.  
  180.             // JOIN clauses for NOT EXISTS have their own syntax. 
  181.             if ( 'NOT EXISTS' === $meta_compare ) {

  183.                 $join .= $i ? " AS $alias" : '';

  189.                 $join .= $i ? " AS $alias" : '';
