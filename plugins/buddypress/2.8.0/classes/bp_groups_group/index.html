  895.     public static function get( $args = array() ) {

  897.

  899.         if ( ! is_array( $args ) || func_num_args() > 1 ) {

  903.                 0 => 'type',  
  904.                 1 => 'per_page',  
  905.                 2 => 'page',  
  906.                 3 => 'user_id',  
  907.                 4 => 'search_terms',  
  908.                 5 => 'include',  
  909.                 6 => 'populate_extras',  
  910.                 7 => 'exclude',  
  911.                 8 => 'show_hidden',  
  912.  ); 
  913.  
  914.             $args = bp_core_parse_args_array( $old_args_keys, func_get_args() ); 
  915.         } 
  916.  
  917.         $defaults = array( 
  918.             'type' => null,  
  919.             'orderby' => 'date_created',  
  920.             'order' => 'DESC',  
  921.             'per_page' => null,  
  922.             'page' => null,  
  923.             'user_id' => 0,  
  924.             'search_terms' => false,  
  925.             'search_columns' => array(),  
  926.             'group_type' => '',  
  927.             'group_type__in' => '',  
  928.             'group_type__not_in' => '',  
  929.             'meta_query' => false,  
  930.             'include' => false,  
  931.             'parent_id' => null,  
  932.             'update_meta_cache' => true,  
  933.             'update_admin_cache' => false,  
  934.             'exclude' => false,  
  935.             'show_hidden' => false,  
  936.  ); 
  937.  
  938.         $r = wp_parse_args( $args, $defaults ); 
  939.  
  940.         $bp = buddypress(); 
  941.  
  942.         $sql = array( 
  943.             'select' => "SELECT DISTINCT g.id",  
  944.             'from' => "{$bp->groups->table_name} g",  
  945.             'where' => '',  
  946.             'orderby' => '',  
  947.             'pagination' => '',  
  948.  ); 
  949.  
  950.         if ( ! empty( $r['user_id'] ) ) { 
  951.             $sql['from'] .= " JOIN {$bp->groups->table_name_members} m ON ( g.id = m.group_id )"; 
  952.         } 
  953.  
  954.         $where_conditions = array(); 
  955.  
  956.         if ( empty( $r['show_hidden'] ) ) { 
  957.             $where_conditions['hidden'] = "g.status != 'hidden'"; 
  958.         } 
  959.  
  960.         $search = ''; 
  961.         if ( isset( $r['search_terms'] ) ) { 
  962.             $search = trim( $r['search_terms'] ); 
  963.         } 
  964.

  966.             $leading_wild = ( ltrim( $search, '*' ) != $search ); 
  967.             $trailing_wild = ( rtrim( $search, '*' ) != $search ); 
  968.             if ( $leading_wild && $trailing_wild ) { 
  969.                 $wild = 'both'; 
  970.             } elseif ( $leading_wild ) { 
  971.                 $wild = 'leading'; 
  972.             } elseif ( $trailing_wild ) { 
  973.                 $wild = 'trailing'; 
  974.             } else { 
  975.                 // Default is to wrap in wildcard characters. 
  976.                 $wild = 'both'; 
  977.             } 
  978.             $search = trim( $search, '*' ); 
  979.  
  980.             $searches = array(); 
  981.             $leading_wild = ( 'leading' == $wild || 'both' == $wild ) ? '%' : ''; 
  982.             $trailing_wild = ( 'trailing' == $wild || 'both' == $wild ) ? '%' : '';

  984.

  986.             if ( $r['search_columns'] ) {

  991.                 $searches[] = $wpdb->prepare( "$search_column LIKE %s", $wildcarded ); 
  992.             } 
  993.

  1023.         }

  1038.         }

  1042.             $where_conditions['exclude'] = "g.id NOT IN ({$exclude})"; 
  1043.         }

  1046.  
  1047.         $order = $r['order']; 
  1048.         $orderby = $r['orderby']; 
  1049.  
  1050.         // If a 'type' parameter was passed, parse it and overwrite 
  1051.         // 'order' and 'orderby' params passed to the function. 
  1052.         if (  ! empty( $r['type'] ) ) { 
  1053.  
  1054.             /** 
  1055.              * Filters the 'type' parameter used to overwrite 'order' and 'orderby' values. 
  1056.              * 
  1057.              * @since 2.1.0 
  1058.              * 
  1059.              * @param array  $value Converted 'type' value for order and orderby. 
  1060.              * @param string $value Parsed 'type' value for the get method.

  1063.

  1065.             // an array with empty values. In this case, we stick 
  1066.             // with the default values of $order and $orderby. 
  1067.             if ( ! empty( $order_orderby['order'] ) ) { 
  1068.                 $order = $order_orderby['order']; 
  1069.             } 
  1070.

  1072.                 $orderby = $order_orderby['orderby'];

  1087.         $order = bp_esc_sql_order( $order ); 
  1088.

  1091.          * 
  1092.          * @since 2.1.0

  1099.  
  1100.         // Random order is a special case. 
  1101.         if ( 'rand()' === $orderby ) { 
  1102.             $sql['orderby'] = "ORDER BY rand()"; 
  1103.         } else { 
  1104.             $sql['orderby'] = "ORDER BY {$orderby} {$order}"; 
  1105.         } 
  1106.  
  1107.         if ( ! empty( $r['per_page'] ) && ! empty( $r['page'] ) && $r['per_page'] != -1 ) { 
  1108.             $sql['pagination'] = $wpdb->prepare( "LIMIT %d, %d", intval( ( $r['page'] - 1 ) * $r['per_page']), intval( $r['per_page'] ) ); 
  1109.         } 
  1110.  
  1111.         $where = '';
