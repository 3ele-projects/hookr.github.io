  80. /** 
  81.  * Filter output of Group Description through WordPress's KSES API. 
  82.  * 
  83.  * @since 1.1.0 
  84.  * 
  85.  * @param string $content Content to filter. 
  86.  * @return string 
  87.  */ 
  88. function bp_groups_filter_kses( $content = '' ) { 
  89.  
  90.   /** 
  91.    * Note that we don't immediately bail if $content is empty. This is because 
  92.    * WordPress's KSES API calls several other filters that might be relevant 
  93.    * to someone's workflow (like `pre_kses`) 
  94.    */ 
  95.  
  96.   // Get allowed tags using core WordPress API allowing third party plugins 
  97.   // to target the specific `buddypress-groups` context. 
  98.   $allowed_tags = wp_kses_allowed_html( 'buddypress-groups' ); 
  99.  
  100.   // Add our own tags allowed in group descriptions. 
  101.   $allowed_tags['a']['class'] = array(); 
  102.   $allowed_tags['img'] = array(); 
  103.   $allowed_tags['img']['src'] = array(); 
  104.   $allowed_tags['img']['alt'] = array(); 
  105.   $allowed_tags['img']['width'] = array(); 
  106.   $allowed_tags['img']['height'] = array(); 
  107.   $allowed_tags['img']['class'] = array(); 
  108.   $allowed_tags['img']['id'] = array(); 
  109.   $allowed_tags['code'] = array(); 
  110.  
  111.   /** 
  112.    * Filters the HTML elements allowed for a given context. 
  113.    * 
  114.    * @since 1.2.0 
  115.    * 
  116.    * @param string $allowed_tags Allowed tags, attributes, and/or entities. 
  117.    */ 
  118.   $tags = apply_filters( 'bp_groups_filter_kses', $allowed_tags ); 
  119.  
  120.   // Return KSES'ed content, allowing the above tags. 
  121.   return wp_kses( $content, $tags ); 
  122. }

  167. /** 
  168.  * Add WHERE clauses to bbPress query for group-specific data and access protection. 
  169.  * 
  170.  * @since 1.1.0 
  171.  * 
  172.  * @param string $sql SQL Statement to amend. 
  173.  * @return string 
  174.  */ 
  175. function groups_add_forum_where_sql( $sql = '' ) { 
  176.  
  177.   // Define locale variable. 
  178.   $parts = array(); 
  179.  
  180.   // Set this for groups. 
  181.   $parts['groups'] = "(gm.meta_key = 'forum_id' AND gm.meta_value = t.forum_id)"; 
  182.  
  183.   // Restrict to public... 
  184.   $parts['private'] = "g.status = 'public'"; 
  185.  
  186.   /** 
  187.    * ...but do some checks to possibly remove public restriction. 
  188.    * 
  189.    * Decide if private are visible 
  190.    */ 
  191.  
  192.   // Are we in our own profile? 
  193.   if ( bp_is_my_profile() ) 
  194.       unset( $parts['private'] ); 
  195.  
  196.   // Are we a super admin? 
  197.   elseif ( bp_current_user_can( 'bp_moderate' ) ) 
  198.       unset( $parts['private'] ); 
  199.  
  200.   // No need to filter on a single item. 
  201.   elseif ( bp_is_single_item() ) 
  202.       unset( $parts['private'] ); 
  203.  
  204.   // Check the SQL filter that was passed. 
  205.   if ( !empty( $sql ) ) 
  206.       $parts['passed'] = $sql; 
  207.  
  208.   // Assemble Voltron. 
  209.   $parts_string = implode( ' AND ', $parts ); 
  210.  
  211.   $bp = buddypress(); 
  212.  
  213.   // Set it to the global filter. 
  214.   $bp->groups->filter_sql = $parts_string; 
  215.  
  216.   // Return the global filter. 
  217.   return $bp->groups->filter_sql; 
  218. }

  307.  * @param int                  $user_id   ID of the user to be notified. 
  308.  * @param BP_Activity_Activity $activity  Activity object. 
  309.  * @return bool 
  310.  */ 
  311. function bp_groups_disable_at_mention_notification_for_non_public_groups( $send, $usernames, $user_id, BP_Activity_Activity $activity ) { 
  312.   // Skip the check for administrators, who can get notifications from non-public groups. 
  313.   if ( user_can( $user_id, 'bp_moderate' ) ) { 
  314.       return $send; 
  315.   } 
  316.  
  317.   if ( 'groups' === $activity->component ) { 
  318.       $group = groups_get_group( $activity->item_id ); 
  319.       if ( 'public' !== $group->status && ! groups_is_user_member( $user_id, $group->id ) ) { 
  320.           $send = false; 
  321.       } 
  322.   } 
  323.  
  324.   return $send; 
  325. }
