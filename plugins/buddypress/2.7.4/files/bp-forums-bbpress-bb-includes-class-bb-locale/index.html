  360. /** 
  361.  * Retrieve the date in localized format, based on timestamp. 
  362.  * 
  363.  * If the locale specifies the locale month and weekday, then the locale will 
  364.  * take over the format for the date. If it isn't, then the date format string 
  365.  * will be used instead. 
  366.  * 
  367.  * @param string $dateformatstring Format to display the date. 
  368.  * @param int $unixtimestamp Optional. Unix timestamp. 
  369.  * @param bool $gmt Optional, default is true. Whether to convert to GMT for time. 
  370.  * @return string The date, translated if locale specifies it. 
  371.  */ 
  372. function bb_gmdate_i18n( $dateformatstring, $unixtimestamp = false, $gmt = true ) { 
  373.   global $bb_locale; 
  374.   $i = $unixtimestamp; 
  375.   // Sanity check for PHP 5.1.0- 
  376.   if ( false === $i || intval($i) < 0 ) { 
  377.       if ( ! $gmt ) 
  378.           $i = current_time( 'timestamp' ); 
  379.       else 
  380.           $i = time(); 
  381.       // we should not let date() interfere with our 
  382.       // specially computed timestamp 
  383.       $gmt = true; 
  384.   } 
  385.  
  386.   // store original value for language with untypical grammars 
  387.   // see http://core.trac.wordpress.org/ticket/9396 
  388.   $req_format = $dateformatstring; 
  389.  
  390.   $datefunc = $gmt? 'gmdate' : 'date'; 
  391.  
  392.   if ( ( !empty( $bb_locale->month ) ) && ( !empty( $bb_locale->weekday ) ) ) { 
  393.       $datemonth = $bb_locale->get_month( $datefunc( 'm', $i ) ); 
  394.       $datemonth_abbrev = $bb_locale->get_month_abbrev( $datemonth ); 
  395.       $dateweekday = $bb_locale->get_weekday( $datefunc( 'w', $i ) ); 
  396.       $dateweekday_abbrev = $bb_locale->get_weekday_abbrev( $dateweekday ); 
  397.       $datemeridiem = $bb_locale->get_meridiem( $datefunc( 'a', $i ) ); 
  398.       $datemeridiem_capital = $bb_locale->get_meridiem( $datefunc( 'A', $i ) ); 
  399.       $dateformatstring = ' '.$dateformatstring; 
  400.       $dateformatstring = preg_replace( "/([^\\\])D/", "\\1" . backslashit( $dateweekday_abbrev ), $dateformatstring ); 
  401.       $dateformatstring = preg_replace( "/([^\\\])F/", "\\1" . backslashit( $datemonth ), $dateformatstring ); 
  402.       $dateformatstring = preg_replace( "/([^\\\])l/", "\\1" . backslashit( $dateweekday ), $dateformatstring ); 
  403.       $dateformatstring = preg_replace( "/([^\\\])M/", "\\1" . backslashit( $datemonth_abbrev ), $dateformatstring ); 
  404.       $dateformatstring = preg_replace( "/([^\\\])a/", "\\1" . backslashit( $datemeridiem ), $dateformatstring ); 
  405.       $dateformatstring = preg_replace( "/([^\\\])A/", "\\1" . backslashit( $datemeridiem_capital ), $dateformatstring ); 
  406.  
  407.       $dateformatstring = substr( $dateformatstring, 1, strlen( $dateformatstring ) -1 ); 
  408.   } 
  409.   $j = @$datefunc( $dateformatstring, $i ); 
  410.   // allow plugins to redo this entirely for languages with untypical grammars 
  411.   $j = apply_filters('bb_gmdate_i18n', $j, $req_format, $i, $gmt); 
  412.   return $j; 
  413. }

  415. function bb_get_datetime_formatstring_i18n( $type = 'datetime' ) { 
  416.   $formatstring = bb_get_option( $type . '_format' ); 
  417.   if ( empty($formatstring) ) { 
  418.       global $bb_locale; 
  419.       $formatstring = $bb_locale->get_datetime_formatstring( $type ); 
  420.   } 
  421.   return $formatstring; 
  422. }

  424. function bb_datetime_format_i18n( $unixtimestamp, $type = 'datetime', $formatstring = '', $gmt = true ) { 
  425.   if ( empty($formatstring) ) { 
  426.       $formatstring = bb_get_datetime_formatstring_i18n( $type ); 
  427.   } 
  428.   return bb_gmdate_i18n( $formatstring, bb_offset_time( $unixtimestamp ), $gmt );

  431. /** 
  432.  * Convert number to format based on the locale. 
  433.  * 
  434.  * @since 2.3.0 
  435.  * 
  436.  * @param mixed $number The number to convert based on locale. 
  437.  * @param int $decimals Precision of the number of decimal places. 
  438.  * @return string Converted number in string format. 
  439.  */ 
  440. function bb_number_format_i18n( $number, $decimals = null ) { 
  441.   global $bb_locale; 
  442.   // let the user override the precision only 
  443.   $decimals = ( is_null( $decimals ) ) ? $bb_locale->number_format['decimals'] : intval( $decimals ); 
  444.  
  445.   $num = number_format( $number, $decimals, $bb_locale->number_format['decimal_point'], $bb_locale->number_format['thousands_sep'] ); 
  446.  
  447.   // let the user translate digits from latin to localized language 
  448.   return apply_filters( 'bb_number_format_i18n', $num ); 
  449. }
