  207. /** 
  208.  * Set up the currently logged-in user. 
  209.  * 
  210.  * We white-list the WordPress Customizer which purposely loads the user early. 
  211.  * If the current user is being setup before the "init" action has fired,  
  212.  * strange (and difficult to debug) role/capability issues will occur. 
  213.  * 
  214.  * @since 1.7.0 
  215.  * 
  216.  * @link https://buddypress.trac.wordpress.org/ticket/6046 
  217.  * @link https://core.trac.wordpress.org/ticket/24169 
  218.  * 
  219.  */ 
  220. function bp_setup_current_user() { 
  221.   $skip_warning = ( 
  222.       ( isset( $_REQUEST['wp_customize'] ) && 'on' === $_REQUEST['wp_customize'] ) || 
  223.       ( is_admin() && 'customize.php' === basename( $_SERVER['PHP_SELF'] ) ) 
  224.  ); 
  225.  
  226.   if ( ! $skip_warning && ! did_action( 'after_setup_theme' ) ) { 
  227.       $e = new Exception; 
  228.       $trace = $e->getTraceAsString(); 
  229.  
  230.       _doing_it_wrong( __FUNCTION__, __( 'The current user is being initialized without using $wp->init().', 'buddypress' ) . "\n===\nTrace:\n" . substr( $trace, strpos( $trace, '#6' ) ) . "\n===\n", '1.7' ); 
  231.   } 
  232.  
  233.   /** 
  234.    * Fires to set up the current user setup process. 
  235.    * 
  236.    * @since 1.7.0 
  237.    */ 
  238.   do_action( 'bp_setup_current_user' ); 
  239. }
