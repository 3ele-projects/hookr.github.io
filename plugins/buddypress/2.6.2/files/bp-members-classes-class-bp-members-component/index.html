  86.   /** 
  87.    * Set up bp-members global settings.

  89.    * The BP_MEMBERS_SLUG constant is deprecated, and only used here for

  98.   public function setup_globals( $args = array() ) { 
  99.       global $wpdb; 
  100.  
  101.       $bp = buddypress(); 
  102.  
  103.       /** Component Globals ************************************************ 
  104.        */ 
  105.  
  106.       // Define a slug, as a fallback for backpat. 
  107.       if ( !defined( 'BP_MEMBERS_SLUG' ) ) { 
  108.           define( 'BP_MEMBERS_SLUG', $this->id ); 
  109.       } 
  110.  
  111.       // Override any passed args. 
  112.       $args = array( 
  113.           'slug' => BP_MEMBERS_SLUG,  
  114.           'root_slug' => isset( $bp->pages->members->slug ) ? $bp->pages->members->slug : BP_MEMBERS_SLUG,  
  115.           'has_directory' => true,  
  116.           'directory_title' => _x( 'Members', 'component directory title', 'buddypress' ),  
  117.           'search_string' => __( 'Search Members...', 'buddypress' ),  
  118.           'global_tables' => array( 
  119.               'table_name_last_activity' => bp_core_get_table_prefix() . 'bp_activity',  
  120.               'table_name_signups' => $wpdb->base_prefix . 'signups', // Signups is a global WordPress table. 
  121.  ) 
  122.  ); 
  123.  
  124.       parent::setup_globals( $args ); 
  125.  
  126.       /** Logged in user *************************************************** 
  127.        */ 
  128.  
  129.       // The core userdata of the user who is currently logged in. 
  130.       $bp->loggedin_user->userdata = bp_core_get_core_userdata( bp_loggedin_user_id() ); 
  131.  
  132.       // Fetch the full name for the logged in user. 
  133.       $bp->loggedin_user->fullname = isset( $bp->loggedin_user->userdata->display_name ) ? $bp->loggedin_user->userdata->display_name : ''; 
  134.  
  135.       // Hits the DB on single WP installs so get this separately. 
  136.       $bp->loggedin_user->is_super_admin = $bp->loggedin_user->is_site_admin = is_super_admin( bp_loggedin_user_id() ); 
  137.  
  138.       // The domain for the user currently logged in. eg: http://example.com/members/andy. 
  139.       $bp->loggedin_user->domain = bp_core_get_user_domain( bp_loggedin_user_id() ); 
  140.  
  141.       /** Displayed user *************************************************** 
  142.        */ 
  143.  
  144.       // The core userdata of the user who is currently being displayed. 
  145.       $bp->displayed_user->userdata = bp_core_get_core_userdata( bp_displayed_user_id() ); 
  146.  
  147.       // Fetch the full name displayed user. 
  148.       $bp->displayed_user->fullname = isset( $bp->displayed_user->userdata->display_name ) ? $bp->displayed_user->userdata->display_name : ''; 
  149.  
  150.       // The domain for the user currently being displayed. 
  151.       $bp->displayed_user->domain = bp_core_get_user_domain( bp_displayed_user_id() ); 
  152.  
  153.       // Initialize the nav for the members component. 
  154.       $this->nav = new BP_Core_Nav(); 
  155.  
  156.       // If A user is displayed, check if there is a front template 
  157.       if ( bp_get_displayed_user() ) { 
  158.           $bp->displayed_user->front_template = bp_displayed_user_get_front_template(); 
  159.       } 
  160.  
  161.       /** Signup *********************************************************** 
  162.        */ 
  163.  
  164.       $bp->signup = new stdClass; 
  165.  
  166.       /** Profiles Fallback ************************************************ 
  167.        */ 
  168.  
  169.       if ( ! bp_is_active( 'xprofile' ) ) { 
  170.           $bp->profile = new stdClass; 
  171.           $bp->profile->slug = 'profile'; 
  172.           $bp->profile->id = 'profile'; 
  173.       }

  176.   /** 
  177.    * Set up canonical stack for this component. 
  178.    * 
  179.    * @since 2.1.0 
  180.    */ 
  181.   public function setup_canonical_stack() { 
  182.       $bp = buddypress(); 
  183.  
  184.       /** Default Profile Component **************************************** 
  185.        */ 
  186.       if ( bp_displayed_user_has_front_template() ) { 
  187.           $bp->default_component = 'front'; 
  188.       } elseif ( defined( 'BP_DEFAULT_COMPONENT' ) && BP_DEFAULT_COMPONENT ) { 
  189.           $bp->default_component = BP_DEFAULT_COMPONENT; 
  190.       } elseif ( bp_is_active( 'activity' ) && isset( $bp->pages->activity ) ) { 
  191.           $bp->default_component = bp_get_activity_slug(); 
  192.       } else { 
  193.           $bp->default_component = ( 'xprofile' === $bp->profile->id ) ? 'profile' : $bp->profile->id; 
  194.       } 
  195.  
  196.       /** Canonical Component Stack **************************************** 
  197.        */ 
  198.  
  199.       if ( bp_displayed_user_id() ) { 
  200.           $bp->canonical_stack['base_url'] = bp_displayed_user_domain(); 
  201.  
  202.           if ( bp_current_component() ) { 
  203.               $bp->canonical_stack['component'] = bp_current_component(); 
  204.           } 
  205.  
  206.           if ( bp_current_action() ) { 
  207.               $bp->canonical_stack['action'] = bp_current_action(); 
  208.           } 
  209.  
  210.           if ( !empty( $bp->action_variables ) ) { 
  211.               $bp->canonical_stack['action_variables'] = bp_action_variables(); 
  212.           } 
  213.  
  214.           // Looking at the single member root/home, so assume the default. 
  215.           if ( ! bp_current_component() ) { 
  216.               $bp->current_component = $bp->default_component; 
  217.  
  218.           // The canonical URL will not contain the default component. 
  219.           } elseif ( bp_is_current_component( $bp->default_component ) && ! bp_current_action() ) { 
  220.               unset( $bp->canonical_stack['component'] ); 
  221.           } 
  222.  
  223.           // If we're on a spammer's profile page, only users with the 'bp_moderate' cap 
  224.           // can view subpages on the spammer's profile. 
  225.           // 
  226.           // users without the cap trying to access a spammer's subnav page will get 
  227.           // redirected to the root of the spammer's profile page.  this occurs by 
  228.           // by removing the component in the canonical stack. 
  229.           if ( bp_is_user_spammer( bp_displayed_user_id() ) && ! bp_current_user_can( 'bp_moderate' ) ) { 
  230.               unset( $bp->canonical_stack['component'] ); 
  231.           } 
  232.       } 
  233.   }

  235.   /** 
  236.    * Set up fall-back component navigation if XProfile is inactive. 
  237.    * 
  238.    * @since 1.5.0 
  239.    * 
  240.    * @see BP_Component::setup_nav() for a description of arguments. 
  241.    * 
  242.    * @param array $main_nav Optional. See BP_Component::setup_nav() for 
  243.    *                        description. 
  244.    * @param array $sub_nav  Optional. See BP_Component::setup_nav() for 
  245.    *                        description. 
  246.    */ 
  247.   public function setup_nav( $main_nav = array(), $sub_nav = array() ) { 
  248.  
  249.       // Don't set up navigation if there's no member. 
  250.       if ( ! is_user_logged_in() && ! bp_is_user() ) { 
  251.           return; 
  252.       } 
  253.  
  254.       $is_xprofile_active = bp_is_active( 'xprofile' ); 
  255.  
  256.       // Bail if XProfile component is active and there's no custom front page for the user. 
  257.       if ( ! bp_displayed_user_has_front_template() && $is_xprofile_active ) { 
  258.           return; 
  259.       } 
  260.  
  261.       // Determine user to use. 
  262.       if ( bp_displayed_user_domain() ) { 
  263.           $user_domain = bp_displayed_user_domain(); 
  264.       } elseif ( bp_loggedin_user_domain() ) { 
  265.           $user_domain = bp_loggedin_user_domain(); 
  266.       } else { 
  267.           return; 
  268.       } 
  269.  
  270.       // Set slug to profile in case the xProfile component is not active 
  271.       $slug = bp_get_profile_slug(); 
  272.  
  273.       // Defaults to empty navs 
  274.       $this->main_nav = array(); 
  275.       $this->sub_nav = array(); 
  276.  
  277.       if ( ! $is_xprofile_active ) { 
  278.           $this->main_nav = array( 
  279.               'name' => _x( 'Profile', 'Member profile main navigation', 'buddypress' ),  
  280.               'slug' => $slug,  
  281.               'position' => 20,  
  282.               'screen_function' => 'bp_members_screen_display_profile',  
  283.               'default_subnav_slug' => 'public',  
  284.               'item_css_id' => buddypress()->profile->id 
  285.  ); 
  286.       } 
  287.  
  288.       /** 
  289.        * Setup the subnav items for the member profile. 
  290.        * 
  291.        * This is required in case there's a custom front or in case the xprofile component 
  292.        * is not active. 
  293.        */ 
  294.       $this->sub_nav = array( 
  295.           'name' => _x( 'View', 'Member profile view', 'buddypress' ),  
  296.           'slug' => 'public',  
  297.           'parent_url' => trailingslashit( $user_domain . $slug ),  
  298.           'parent_slug' => $slug,  
  299.           'screen_function' => 'bp_members_screen_display_profile',  
  300.           'position' => 10 
  301.  ); 
  302.  
  303.       /** 
  304.        * If there's a front template the members component nav 
  305.        * will be there to display the user's front page. 
  306.        */ 
  307.       if ( bp_displayed_user_has_front_template() ) { 
  308.           $main_nav = array( 
  309.               'name' => _x( 'Home', 'Member Home page', 'buddypress' ),  
  310.               'slug' => 'front',  
  311.               'position' => 5,  
  312.               'screen_function' => 'bp_members_screen_display_profile',  
  313.               'default_subnav_slug' => 'public',  
  314.  ); 
  315.  
  316.           // We need a dummy subnav for the front page to load. 
  317.           $front_subnav = $this->sub_nav; 
  318.           $front_subnav['parent_slug'] = 'front'; 
  319.  
  320.           // In case the subnav is displayed in the front template 
  321.           $front_subnav['parent_url'] = trailingslashit( $user_domain . 'front' ); 
  322.  
  323.           // Set the subnav 
  324.           $sub_nav[] = $front_subnav; 
  325.  
  326.           /** 
  327.            * If the profile component is not active, we need to create a new 
  328.            * nav to display the WordPress profile. 
  329.            */ 
  330.           if ( ! $is_xprofile_active ) { 
  331.               add_action( 'bp_members_setup_nav', array( $this, 'setup_profile_nav' ) ); 
  332.           } 
  333.  
  334.       /** 
  335.        * If there's no front template and xProfile is not active, the members 
  336.        * component nav will be there to display the WordPress profile 
  337.        */ 
  338.       } else { 
  339.           $main_nav = $this->main_nav; 
  340.           $sub_nav[] = $this->sub_nav; 
  341.       } 
  342.  
  343.  
  344.       parent::setup_nav( $main_nav, $sub_nav ); 
  345.   } 
  346.  
  347.   /** 
  348.    * Set up a profile nav in case the xProfile 
  349.    * component is not active and a front template is 
  350.    * used. 
  351.    * 
  352.    * @since 2.6.0 
  353.    */ 
  354.   public function setup_profile_nav() { 
  355.       if ( empty( $this->main_nav ) || empty( $this->sub_nav ) ) { 
  356.           return; 
  357.       } 
  358.  
  359.       // Add the main nav 
  360.       bp_core_new_nav_item( $this->main_nav, 'members' ); 
  361.  
  362.       // Add the sub nav item. 
  363.       bp_core_new_subnav_item( $this->sub_nav, 'members' ); 
  364.   }
