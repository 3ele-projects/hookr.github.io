  558.                         $inside = false;

  560.                     } 
  561.                 } 
  562.             } 
  563.         } 
  564.  
  565.         if ($trans) { 
  566.             $this->mpdf->gradients[$n]['trans'] = true; 
  567.             $s .= ' /TGS' . $n . ' gs '; 
  568.         } 
  569.         if (!is_array($extend) || count($extend) < 1) { 
  570.             $extend = array('true', 'true'); // These are supposed to be quoted - appear in PDF file as text 
  571.         } 
  572.         $this->mpdf->gradients[$n]['coords'] = $coords; 
  573.         $this->mpdf->gradients[$n]['extend'] = $extend; 
  574.         //paint the gradient 
  575.         $s .= '/Sh' . $n . ' sh ' . "\n"; 
  576.         //restore previous Graphic State 
  577.         $s .= ' Q ' . "\n"; 
  578.         if ($return) { 
  579.             return $s; 
  580.         } else { 
  581.             $this->mpdf->_out($s); 
  582.         } 
  583.     }
