  290.     public static function reordering_syllable(&$info, $GSUBdata, $start, $end) 
  291.     { 
  292.         /** vowel_syllable: We made the vowels look like consonants. So uses the consonant logic! */ 
  293.         /** broken_cluster: We already inserted dotted-circles, so just call the standalone_cluster. */ 
  294.  
  295.         $syllable_type = ($info[$start]['syllable'] & 0x0F); 
  296.         if ($syllable_type == self::NON_MYANMAR_CLUSTER) { 
  297.             return; 
  298.         } 
  299.         if ($syllable_type == self::BROKEN_CLUSTER) { 
  300.             //if ($uniscribe_bug_compatible) { 
  301.             /** For dotted-circle, this is what Uniscribe does: 
  302.              * If dotted-circle is the last glyph, it just does nothing. 
  303.              * i.e. It doesn't form Reph. */ 
  304.             if ($info[$end - 1]['myanmar_category'] == self::OT_DOTTEDCIRCLE) { 
  305.                 return; 
  306.             } 
  307.         } 
  308.  
  309.         $base = $end; 
  310.         $has_reph = false; 
  311.         $limit = $start; 
  312.  
  313.         if (($start + 3 <= $end) && 
  314.             $info[$start]['myanmar_category'] == self::OT_Ra && 
  315.             $info[$start + 1]['myanmar_category'] == self::OT_As && 
  316.             $info[$start + 2]['myanmar_category'] == self::OT_H) { 
  317.             $limit += 3; 
  318.             $base = $start; 
  319.             $has_reph = true; 
  320.         } 
  321.  
  322.         if (!$has_reph) 
  323.             $base = $limit; 
  324.  
  325.         for ($i = $limit; $i < $end; $i++) { 
  326.             if (self::is_consonant($info[$i])) { 
  327.                 $base = $i; 
  328.                 break; 
  329.             } 
  330.         } 
  331.  
  332.  
  333.         /** Reorder! */ 
  334.         $i = $start; 
  335.         for (; $i < $start + ($has_reph ? 3 : 0); $i++) 
  336.             $info[$i]['myanmar_position'] = self::POS_AFTER_MAIN; 
  337.         for (; $i < $base; $i++) 
  338.             $info[$i]['myanmar_position'] = self::POS_PRE_C; 
  339.         if ($i < $end) { 
  340.             $info[$i]['myanmar_position'] = self::POS_BASE_C; 
  341.             $i++; 
  342.         } 
  343.         $pos = self::POS_AFTER_MAIN; 
  344.         /** The following loop may be ugly, but it implements all of 
  345.          * Myanmar reordering! */ 
  346.         for (; $i < $end; $i++) { 
  347.             if ($info[$i]['myanmar_category'] == self::OT_MR) /** Pre-base reordering */ { 
  348.                 $info[$i]['myanmar_position'] = self::POS_PRE_C; 
  349.                 continue; 
  350.             } 
  351.             if ($info[$i]['myanmar_position'] < self::POS_BASE_C) /** Left matra */ { 
  352.                 continue; 
  353.             } 
  354.  
  355.             if ($pos == self::POS_AFTER_MAIN && $info[$i]['myanmar_category'] == self::OT_VBlw) { 
  356.                 $pos = self::POS_BELOW_C; 
  357.                 $info[$i]['myanmar_position'] = $pos; 
  358.                 continue; 
  359.             } 
  360.  
  361.             if ($pos == self::POS_BELOW_C && $info[$i]['myanmar_category'] == self::OT_A) { 
  362.                 $info[$i]['myanmar_position'] = self::POS_BEFORE_SUB; 
  363.                 continue; 
  364.             } 
  365.             if ($pos == self::POS_BELOW_C && $info[$i]['myanmar_category'] == self::OT_VBlw) { 
  366.                 $info[$i]['myanmar_position'] = $pos; 
  367.                 continue; 
  368.             } 
  369.             if ($pos == self::POS_BELOW_C && $info[$i]['myanmar_category'] != self::OT_A) { 
  370.                 $pos = self::POS_AFTER_SUB; 
  371.                 $info[$i]['myanmar_position'] = $pos; 
  372.                 continue; 
  373.             } 
  374.             $info[$i]['myanmar_position'] = $pos; 
  375.         } 
  376.  
  377.  
  378.         /** Sit tight, rock 'n roll! */ 
  379.         self::bubble_sort($info, $start, $end - $start); 
  380.     }

  464.     public static function bubble_sort(&$arr, $start, $len) 
  465.     { 
  466.         if ($len < 2) { 
  467.             return; 
  468.         } 
  469.         $k = $start + $len - 2;

  471.             for ($j = $start; $j <= $k; $j++) { 
  472.                 if ($arr[$j]['myanmar_position'] > $arr[$j + 1]['myanmar_position']) { 
  473.                     $t = $arr[$j];
