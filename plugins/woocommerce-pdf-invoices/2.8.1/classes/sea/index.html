  236.

  348.     public static function sea_get_categories($u) 
  349.     { 
  350.         if (0x1980 <= $u && $u <= 0x19DF) 
  351.             return self::$sea_table[$u - 0x1980]; // offset 0 for New Tai Lue 
  352.         if (0x1A20 <= $u && $u <= 0x1AAF) 
  353.             return self::$sea_table[$u - 0x1A20 + 96]; // offset for Tai Tham 
  354.         if (0xAA00 <= $u && $u <= 0xAA5F) 
  355.             return self::$sea_table[$u - 0xAA00 + 96 + 144];  // Cham 
  356.         if ($u == 0x00A0) 
  357.             return 3851; // (ISC_CP | (IMC_x << 8)) 
  358.         if ($u == 0x25CC) 
  359.             return 3851; // (ISC_CP | (IMC_x << 8)) 
  360.         return 3840; // (ISC_x | (IMC_x << 8)) 
  361.     }

  363.     public static function bubble_sort(&$arr, $start, $len) 
  364.     { 
  365.         if ($len < 2) { 
  366.             return; 
  367.         } 
  368.         $k = $start + $len - 2; 
  369.         while ($k >= $start) { 
  370.             for ($j = $start; $j <= $k; $j++) { 
  371.                 if ($arr[$j]['sea_position'] > $arr[$j + 1]['sea_position']) { 
  372.                     $t = $arr[$j]; 
  373.                     $arr[$j] = $arr[$j + 1]; 
  374.                     $arr[$j + 1] = $t; 
  375.                 } 
  376.             } 
  377.             $k--; 
  378.         } 
  379.     }
