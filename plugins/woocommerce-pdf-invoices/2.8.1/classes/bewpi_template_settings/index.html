  557.     /** 
  558.      * Validate input of settings. 
  559.      * 
  560.      * @param array $input form settings. 
  561.      * 
  562.      * @return mixed|void 
  563.      */ 
  564.     public function validate_input( $input ) { 
  565.         $output = array(); 
  566.  
  567.         // strip strings. 
  568.         foreach ( $input as $key => $value ) { 
  569.             if ( ! isset( $input[ $key ] ) ) { 
  570.                 continue; 
  571.             } 
  572.  
  573.             if ( 'bewpi_company_logo' === $key ) { 
  574.                 $output[ $key ] = ''; 
  575.                 continue; 
  576.             } 
  577.  
  578.             // strip all html and php tags and properly handle quoted strings. 
  579.             $output[ $key ] = $this->strip_str( stripslashes( $input[ $key ] ) ); 
  580.         } 
  581.  
  582.         if ( isset( $input['bewpi_company_logo'] ) && ! empty( $input['bewpi_company_logo'] ) ) { 
  583.             $image_url = $this->validate_image( $input['bewpi_company_logo'] ); 
  584.             if ( $image_url ) { 
  585.                 $output['bewpi_company_logo'] = $image_url; 
  586.             } else { 
  587.                 add_settings_error( 
  588.                     esc_attr( self::SETTINGS_KEY ),  
  589.                     'file-not-found',  
  590.                     __( 'Company logo not found. Upload the image to the Media Library and try again.', 'woocommerce-pdf-invoices' ) 
  591.  ); 
  592.             } 
  593.         } 
  594.  
  595.         if ( isset( $input['bewpi_reset_counter'] ) && $input['bewpi_reset_counter'] ) { 
  596.             set_transient( 'bewpi_next_invoice_number', intval( $input['bewpi_next_invoice_number'] ) ); 
  597.         } 
  598.  
  599.         // return the array processing any additional functions filtered by this action. 
  600.         return apply_filters( 'validate_input', $output, $input ); 
  601.     }
