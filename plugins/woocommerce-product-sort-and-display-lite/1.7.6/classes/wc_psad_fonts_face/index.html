  390.     public function is_valid_google_api_key( $cache=true ) { 
  391.         $is_valid = false; 
  392.  
  393.         $this->google_api_key = get_option( $this->google_api_key_option, '' ); 
  394.         $google_api_key_enable = get_option( $this->google_api_key_option . '_enable', 0 ); 
  395.  
  396.         if ( '' != trim( $this->google_api_key ) && 1 == $google_api_key_enable ) { 
  397.  
  398.             $google_api_key_status = get_transient( $this->google_api_key_option . '_status' ); 
  399.  
  400.             if ( ! $cache ) { 
  401.                 $google_api_key_status = null; 
  402.             } 
  403.  
  404.             if ( ! $google_api_key_status ) { 
  405.                 $respone_api = wp_remote_get( "https://www.googleapis.com/webfonts/v1/webfonts?sort=alpha&key=" . trim( $this->google_api_key ),  
  406.                     array( 
  407.                         'sslverify' => false,  
  408.                         'timeout' => 45 
  409.  ) 
  410.  ); 
  411.  
  412.                 $font_list = array(); 
  413.                 $response_fonts = array(); 
  414.

  416.                 if ( ! is_wp_error( $respone_api ) ) { 
  417.  
  418.                     $json_string = get_magic_quotes_gpc() ? stripslashes( $respone_api['body'] ) : $respone_api['body']; 
  419.                     $response_fonts = json_decode( $json_string, true ); 
  420.

  422.                     if ( ! isset( $response_fonts['error'] ) ) {
