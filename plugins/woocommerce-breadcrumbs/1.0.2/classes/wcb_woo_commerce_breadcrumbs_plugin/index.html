  254.     public function wcb_plugin_sanitize_options( $input ) { 
  255.         $valid = array(); 
  256.  
  257.         // If the Restore Defaults button is clicked, reset the values to the default values 
  258.         if ( isset( $_POST['restore_defaults'] ) ) { 
  259.             $message = esc_html('Default options restored', 'woocommerce-breadcrumbs' ); 
  260.             add_settings_error( 'woocommerce-breadcrumbs', esc_attr( 'wcb_restore_defaults' ), $message, 'updated' ); 
  261.             return $this->breadcrumb_defaults; 
  262.         } 
  263.  
  264.         // Validate the inputs 
  265.         $valid['wcb_enable_breadcrumbs'] = ( isset( $input['wcb_enable_breadcrumbs'] ) ? '1' : '0' ); 
  266.  
  267.         $valid['wcb_breadcrumb_delimiter'] = wp_kses_data( $input['wcb_breadcrumb_delimiter'] ); 
  268.  
  269.         $valid['wcb_wrap_before'] = wp_kses_post( $input['wcb_wrap_before'] ); 
  270.

  272.  
  273.         $valid['wcb_before'] = wp_kses_post( $input['wcb_before'] ); 
  274.  
  275.         $valid['wcb_after'] = wp_kses_post( $input['wcb_after'] ); 
  276.  
  277.         $valid['wcb_home_text'] = sanitize_text_field( $input['wcb_home_text'] );
