  702.     /** 
  703.      * Converts hexidecimal color value to rgb values and returns as array/string 
  704.      * 
  705.      * @param string $hex 
  706.      * @param bool $asString 
  707.      * @return array|string 
  708.      */ 
  709.     function hex2rgb($hex, $asString = false) 
  710.     { 
  711.         // strip off any leading # 
  712.         if (0 === strpos($hex, '#')) { 
  713.             $hex = substr($hex, 1); 
  714.         } else { 
  715.             if (0 === strpos($hex, '&H')) { 
  716.                 $hex = substr($hex, 2); 
  717.             } 
  718.         } 
  719.         // break into hex 3-tuple 
  720.         $cutpoint = ceil(strlen($hex) / 2) - 1; 
  721.         $rgb = explode(':', wordwrap($hex, $cutpoint, ':', $cutpoint), 3); 
  722.         // convert each tuple to decimal 
  723.         $rgb[0] = isset($rgb[0]) ? hexdec($rgb[0]) : 0; 
  724.         $rgb[1] = isset($rgb[1]) ? hexdec($rgb[1]) : 0; 
  725.         $rgb[2] = isset($rgb[2]) ? hexdec($rgb[2]) : 0; 
  726.         return $asString ? "{$rgb[0]} {$rgb[1]} {$rgb[2]}" : $rgb; 
  727.     }
