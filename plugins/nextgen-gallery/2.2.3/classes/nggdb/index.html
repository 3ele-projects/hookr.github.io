  715.     /** 
  716.      * search for images and return the result 
  717.      * 
  718.      * @since 1.3.0 
  719.      * @param string $request 
  720.      * @param int $limit number of results, 0 shows all results 
  721.      * @return Array Result of the request 
  722.      */ 
  723.     function search_for_images( $request, $limit = 0 ) { 
  724.         global $wpdb; 
  725.  
  726.         // If a search pattern is specified, load the posts that match 
  727.         if ( !empty($request) ) { 
  728.             // added slashes screw with quote grouping when done early, so done later 
  729.             $request = stripslashes($request); 
  730.  
  731.             // split the words it a array if seperated by a space or comma 
  732.             preg_match_all('/".*?("|$)|((?<=[\\s", +])|^)[^\\s", +]+/', $request, $matches); 
  733.             $search_terms = array_map(create_function('$a', 'return trim($a, "\\"\'\\n\\r ");'), $matches[0]); 
  734.  
  735.             $n = '%'; 
  736.             $searchand = ''; 
  737.             $search = ''; 
  738.  
  739.             foreach( (array) $search_terms as $term) { 
  740.                 $term = addslashes_gpc($term); 
  741.                 $search .= "{$searchand}((tt.description LIKE '{$n}{$term}{$n}') OR (tt.alttext LIKE '{$n}{$term}{$n}') OR (tt.filename LIKE '{$n}{$term}{$n}'))"; 
  742.                 $searchand = ' AND '; 
  743.             } 
  744.  
  745.             $term = esc_sql($request); 
  746.             if (count($search_terms) > 1 && $search_terms[0] != $request ) 
  747.                 $search .= " OR (tt.description LIKE '{$n}{$term}{$n}') OR (tt.alttext LIKE '{$n}{$term}{$n}') OR (tt.filename LIKE '{$n}{$term}{$n}')"; 
  748.  
  749.             if ( !empty($search) ) 
  750.                 $search = " AND ({$search}) "; 
  751.  
  752.             $limit_by = ( $limit > 0 ) ? 'LIMIT ' . intval($limit) : ''; 
  753.         } else 
  754.             return false; 
  755.  
  756.         // build the final query 
  757.         $query = "SELECT `tt`.`pid` FROM `{$wpdb->nggallery}` AS `t` INNER JOIN `{$wpdb->nggpictures}` AS `tt` ON `t`.`gid` = `tt`.`galleryid` WHERE 1=1 {$search} ORDER BY `tt`.`pid` ASC {$limit_by}"; 
  758.  
  759.         $result = $wpdb->get_col($query); 
  760.  
  761.         // TODO: Currently we didn't support a proper pagination 
  762.         $this->paged['total_objects'] = $this->paged['objects_per_page'] = intval ( $wpdb->get_var( "SELECT FOUND_ROWS()" ) ); 
  763.         $this->paged['max_objects_per_page'] = 1; 
  764.  
  765.         // Return the object from the query result 
  766.         if ($result) { 
  767.             $images = array(); 
  768.             $mapper = C_Image_Mapper::get_instance(); 
  769.             foreach ($result as $image_id) { 
  770.                 $images[] = $mapper->find($image_id); 
  771.             } 
  772.             return $images; 
  773.         } 
  774.  
  775.         return null; 
  776.     }

  906.     /** 
  907.      * Computes a unique slug for the gallery, album or image, when given the desired slug. 
  908.      * 
  909.      * @since 1.7.0 
  910.      * @param string $slug the desired slug (post_name) 
  911.      * @param string $type ('image', 'album' or 'gallery') 
  912.      * @param int (optional) $id of the object, so that it's not checked against itself 
  913.      * @return string unique slug for the object, based on $slug (with a -1, -2, etc. suffix) 
  914.      */ 
  915.     static function get_unique_slug( $slug, $type, $id = 0 ) 
  916.     { 
  917.         global $wpdb; 
  918.  
  919.         $slug = stripslashes($slug); 
  920.         $retval = $slug; 
  921.  
  922.         // We have to create a somewhat complex query to find the next available slug. The query could easily 
  923.         // be simplified if we could use MySQL REGEX, but there are still hosts using MySQL 5.0, and REGEX is 
  924.         // only supported in MySQL 5.1 and higher 
  925.         $field = ''; 
  926.         $table = ''; 
  927.         switch ($type) { 
  928.             case 'image': 
  929.                 $field = 'image_slug'; 
  930.                 $table = $wpdb->nggpictures; 
  931.                 break; 
  932.             case 'album': 
  933.                 $field = 'slug'; 
  934.                 $table = $wpdb->nggalbum; 
  935.                 break; 
  936.             case 'gallery': 
  937.                 $field = 'slug'; 
  938.                 $table = $wpdb->nggallery; 
  939.                 break; 
  940.         } 
  941.  
  942.         // Generate SQL query 
  943.         $query = array(); 
  944.         $query[] = "SELECT {$field}, SUBSTR({$field}, %d) AS 'i' FROM {$table}"; 
  945.         $query[] = "WHERE ({$field} LIKE '{$slug}-%%' AND CONVERT(SUBSTR({$field}, %d), SIGNED) BETWEEN 1 AND %d) OR {$field} = %s"; 
  946.         $query[] = "ORDER BY CAST(i AS SIGNED INTEGER) DESC LIMIT 1"; 
  947.         $query = $wpdb->prepare(implode(" ", $query), strlen("{$slug}-")+1, strlen("{$slug}-")+1, PHP_INT_MAX, $slug); 
  948.  
  949.         // If the above query returns a result, it means that the slug is already taken 
  950.         if (($last_slug = $wpdb->get_var($query))) { 
  951.  
  952.             // If the last known slug has an integer attached, then it means that we need to increment that integer 
  953.             $quoted_slug = preg_quote($slug, '/'); 
  954.             if (preg_match("/{$quoted_slug}-(\\d+)/", $last_slug, $matches)) { 
  955.                 $i = intval($matches[1]) + 1; 
  956.                 $retval = "{$slug}-{$i}"; 
  957.             } 
  958.             else $retval = "{$slug}-1"; 
  959.         } 
  960.  
  961.         return $retval; 
  962.     }
