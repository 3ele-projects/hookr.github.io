  94.                 tags :     album, gallery     -> /album-([0-9]+)/gallery-([0-9]+)/

  116.

  119.

  122.                 $url .= '/page-' . $args['nggpage']; 
  123.             elseif (isset ($args['nggpage']) && ($args['nggpage'] === false) && ( count($args) == 1 ) ) 
  124.                 $url = trailingslashit ( get_permalink ($post->ID) ); // special case instead of showing page-1, we show the clean url 
  125.

  262.      * Mainly a copy of the same function in wp-includes\rewrite.php 
  263.      * Adding the NGG tags to each post & page. Never found easier and proper way to handle this with other functions. 
  264.      * 
  265.      * @return array the permalink structure 
  266.      */ 
  267.     function generate_rewrite_rules() { 
  268.         global $wp_rewrite; 
  269.  
  270.         $rewrite_rules = array(); 
  271.         $permalink_structure =  $wp_rewrite->permalink_structure; 
  272.  
  273.         //get everything up to the first rewrite tag 
  274.         $front = substr($permalink_structure, 0, strpos($permalink_structure, '%')); 
  275.         //build an array of the tags (note that said array ends up being in $tokens[0]) 
  276.         preg_match_all('/%.+?%/', $permalink_structure, $tokens); 
  277.  
  278.         $num_tokens = count($tokens[0]); 
  279.  
  280.         $this->index = $wp_rewrite->index; //probably 'index.php' 
  281.  
  282.         //build a list from the rewritecode and queryreplace arrays, that will look something like 
  283.         //tagname=$matches[i] where i is the current $i 
  284.         for ( $i = 0; $i < $num_tokens; ++$i ) { 
  285.             if ( 0 < $i ) 
  286.                 $queries[$i] = $queries[$i - 1] . '&'; 
  287.             else 
  288.                 $queries[$i] = ''; 
  289.  
  290.             $query_token = str_replace($wp_rewrite->rewritecode, $wp_rewrite->queryreplace, $tokens[0][$i]) . $wp_rewrite->preg_index($i+1); 
  291.             $queries[$i] .= $query_token; 
  292.         } 
  293.  
  294.         //get the structure, minus any cruft (stuff that isn't tags) at the front 
  295.         $structure = $permalink_structure; 
  296.         if ( $front != '/' ) 
  297.             $structure = str_replace($front, '', $structure); 
  298.  
  299.         //create a list of dirs to walk over, making rewrite rules for each level 
  300.         //so for example, a $structure of /%year%/%month%/%postname% would create 
  301.         //rewrite rules for /%year%/, /%year%/%month%/ and /%year%/%month%/%postname% 
  302.         $structure = trim($structure, '/'); 
  303.  
  304.         //strip slashes from the front of $front 
  305.         $struct = preg_replace('|^/+|', '', $front); 
  306.  
  307.         //get the struct for this dir, and trim slashes off the front 
  308.         $struct .= $structure . '/'; //accumulate. see comment near explode('/', $structure) above 
  309.         $struct = ltrim($struct, '/'); 
  310.  
  311.         //replace tags with regexes 
  312.         $match = str_replace($wp_rewrite->rewritecode, $wp_rewrite->rewritereplace, $struct); 
  313.  
  314.         //make a list of tags, and store how many there are in $num_toks 
  315.         $num_toks = preg_match_all('/%.+?%/', $struct, $toks); 
  316.  
  317.         //get the 'tagname=$matches[i]' 
  318.         $query = ( isset($queries) && is_array($queries) ) ? $queries[$num_toks - 1] : ''; 
  319.  
  320.         if ( $num_toks ) { 
  321.             // In the case we build for each and every page ( based on a simple %pagename% rule ) the rewrite rules,  
  322.             // we need to add them first, then the post rules 
  323.             if ( $wp_rewrite->use_verbose_page_rules ) 
  324.                 $rewrite_rules = array_merge ( $this->page_rewrite_rules(), $this->add_rewrite_rules( $match, $query, $num_toks ) ); 
  325.             else 
  326.                 $rewrite_rules = array_merge ( $this->add_rewrite_rules( $match, $query, $num_toks ), $this->page_rewrite_rules() ); 
  327.         } 
  328.  
  329.         return $rewrite_rules; 
  330.     }

  333.      * Retrieve all of the rewrite rules for pages. 
  334.      * 
  335.      * If the 'use_verbose_page_rules' property is false, then there will only 
  336.      * be a single rewrite rule for pages for those matching '%pagename%'. With 
  337.      * the property set to true, the attachments and the pages will be added for 
  338.      * each individual attachment URI and page URI, respectively. 
  339.      * 
  340.      * @since 1.8.3 
  341.      * @access public 
  342.      * @return array

  344.     function page_rewrite_rules() { 
  345.         global $wp_rewrite; 
  346.  
  347.         $rewrite_rules = array(); 
  348.  
  349.         if ( ! $wp_rewrite->use_verbose_page_rules ) { 
  350.  
  351.             $rewrite_rules = $this->add_rewrite_rules( "(.+?)/", 'pagename=$matches[1]', 1 ); 
  352.             return $rewrite_rules; 
  353.         } 
  354.  
  355.         $page_uris = $wp_rewrite->page_uri_index(); 
  356.         $uris = $page_uris[0]; 
  357.  
  358.         if ( is_array( $uris ) ) { 
  359.  
  360.             foreach ( $uris as $uri => $pagename ) { 
  361.                 $rewrite_rules = array_merge($rewrite_rules, $this->add_rewrite_rules( "($uri)/", 'pagename=$matches[1]', 1 ) );

  370.      * Build the final structure of the rewrite rules based on match/query 
  371.      * 
  372.      * @since 1.8.3 
  373.      * @param string $match 
  374.      * @param string $query 
  375.      * @param int $num_toks 
  376.      * @return array 
  377.      */ 
  378.     function add_rewrite_rules( $match, $query, $num_toks ) { 
  379.         global $wp_rewrite; 
  380.  
  381.         $rewrite_rules = array(); 
  382.  
  383.         foreach ( $this->ngg_rules as $regex => $new_query) { 
  384.  
  385.             // first add your nextgen slug 
  386.             $final_match = $match . $this->slug; 
  387.  
  388.             //add regex parameter 
  389.             $final_match .= $regex; 
  390.             // check how often we found matches fields 
  391.             $count = substr_count($new_query, '[matches]'); 
  392.             // we need to know how many tags before 
  393.             $offset = $num_toks; 
  394.             // build the query and count up the matches : tagname=$matches[x] 
  395.             for ( $i = 0; $i < $count; $i++ ) { 
  396.                 $new_query = preg_replace('/\[matches\]/', '$matches[' . ++$offset . ']', $new_query, 1); 
  397.             } 
  398.             $final_query = $query . $new_query; 
  399.  
  400.             //close the match and finalise the query 
  401.             $final_match .= '?$'; 
  402.             $final_query = $this->index . '?' . $final_query; 
  403.  
  404.             $rewrite_rules = array_merge($rewrite_rules, array($final_match => $final_query)); 
  405.  
  406.         } 
  407.  
  408.         return $rewrite_rules; 
  409.     }
