  255.   static function rotate_image($image, $dir = false, $flip = false) 
  256.   {

  266.  
  267.       if ($dir === 'CW') { 
  268.           $rotation = 90; 
  269.       } 
  270.       else if ($dir === 'CCW') {

  281.                   case 5 : // vertical flip + 90 rotate right

  284.                       $rotation = 90; 
  285.                       break; 
  286.                   case 7 : // horizontal flip + 90 rotate right 
  287.                       $flip = 'H';

  289.                       $rotation = -90; 
  290.                       break; 
  291.                   case 4 : // vertical flip 
  292.                       $flip = 'V'; 
  293.                       break; 
  294.                   case 3 : // 180 rotate left 
  295.                       $rotation = -180; 
  296.                       break; 
  297.                   case 2 : // horizontal flip 
  298.                       $flip = 'H'; 
  299.                       break; 
  300.                   case 1 : // no action in the case it doesn't need a rotation 
  301.                   default: 
  302.                       return '0'; 
  303.                       break; 
  304.               } 
  305.           } else 
  306.               return '0'; 
  307.       }

  309.       if ($rotation != null) {

  311.       } 
  312.  
  313.       if ($flip != null) {

  317.       $result = $storage->generate_image_size($image, 'full', $params); 
  318.  
  319.       if (!$result)

  604.
