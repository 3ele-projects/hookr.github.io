  194.   function wpml_adjust_gallery_id($master_post_id, $lang, $post_array, $id) 
  195.   { 
  196.       if (!isset($post_array['post_type']) || $post_array['post_type'] == "displayed_gallery") 
  197.           return; 
  198.  
  199.       $re = "|preview/id--(\d+)|mi"; 
  200.       if (preg_match_all($re, $post_array['post_content'], $gallery_ids)) 
  201.       { 
  202.           foreach($gallery_ids[1] as $index => $gallery_id) { 
  203.               $translated_gallery_id = apply_filters('wpml_object_id', (int)$gallery_id, "displayed_gallery", true, $lang); 
  204.               $search[$index] = "preview/id--" . $gallery_id; 
  205.               $replace[$index] = "preview/id--" . $translated_gallery_id; 
  206.           } 
  207.  
  208.           $post_array['post_content'] = str_replace($search, $replace, $post_array['post_content']); 
  209.  
  210.           $to_save = array( 
  211.               'ID' => $id,  
  212.               'post_content' => $post_array['post_content'] 
  213.  ); 
  214.  
  215.           add_filter('ngg_cleanup_displayed_galleries', '__return_false', 10, 1);

  221.   function wpml_set_gallery_language_on_save_post($post_id) 
  222.   { 
  223.       if (wp_is_post_revision($post_id)) 
  224.           return; 
  225.  
  226.       if (isset($_POST['icl_ajx_action']) && $_POST['icl_ajx_action'] == 'make_duplicates') 
  227.           return; 
  228.  
  229.       $post = get_post($post_id); 
  230.  
  231.       if ($post->post_type == 'displayed_gallery') 
  232.           return; 
  233.  
  234.       if (preg_match_all("#<img.*http(s)?://(.*)?" . NGG_ATTACH_TO_POST_SLUG . "(=|/)preview(/|&|&)id(=|--)(\\d+).*?>#mi", $post->post_content, $matches, PREG_SET_ORDER)) 
  235.       { 
  236.           $mapper = C_Displayed_Gallery_Mapper::get_instance(); 
  237.           foreach ($matches as $match) { 
  238.               // Find the displayed gallery 
  239.               $displayed_gallery_id = $match[6]; 
  240.               add_filter('wpml_suppress_filters', '__return_true', 10, 1); 
  241.               $displayed_gallery = $mapper->find($displayed_gallery_id, TRUE); 
  242.               add_filter('wpml_suppress_filters', '__return_false', 11, 1); 
  243.               if ($displayed_gallery) 
  244.               { 
  245.                   $displayed_gallery_type = apply_filters('wpml_element_type', 'displayed_gallery'); 
  246.  
  247.                   // set language of this gallery 
  248.                   $displayed_gallery_lang = apply_filters('wpml_post_language_details', null, $displayed_gallery->ID); 
  249.                   $post_language = apply_filters('wpml_post_language_details', null, $post_id); 
  250.  
  251.                   if (!$displayed_gallery_lang || $displayed_gallery_lang['language_code'] != $post_language['language_code']) 
  252.                   { 
  253.                       if ($post_language) 
  254.                       { 
  255.                           $args = array( 
  256.                               'element_id' => $displayed_gallery->ID,  
  257.                               'element_type' => $displayed_gallery_type,  
  258.                               'language_code' => $post_language['language_code'] 
  259.  ); 
  260.                           do_action('wpml_set_element_language_details', $args); 
  261.                       } 
  262.                   } 
  263.  
  264.                   // duplicate gallery to other languages 
  265.                   $is_translated = apply_filters('wpml_element_has_translations', '', $displayed_gallery->ID, $displayed_gallery_type); 
  266.                   if (!$is_translated) 
  267.                       do_action('wpml_admin_make_post_duplicates', $displayed_gallery->ID); 
  268.               } 
  269.           } 
  270.       } 
  271.   }

  292.   static function mb_pathinfo($path, $options=null) 
  293.   { 
  294.       $ret = array( 
  295.           'dirname' => '',  
  296.           'basename' => '',  
  297.           'extension' => '',  
  298.           'filename' => '' 
  299.  ); 
  300.       $pathinfo = array(); 
  301.       if (preg_match('%^(.*?)[\\\\/]*(([^/\\\\]*?)(\.([^\.\\\\/]+?)|))[\\\\/\.]*$%im', $path, $pathinfo)) 
  302.       { 
  303.           if (array_key_exists(1, $pathinfo)) 
  304.               $ret['dirname'] = $pathinfo[1]; 
  305.           if (array_key_exists(2, $pathinfo)) 
  306.               $ret['basename'] = $pathinfo[2]; 
  307.           if (array_key_exists(5, $pathinfo)) 
  308.               $ret['extension'] = $pathinfo[5]; 
  309.           if (array_key_exists(3, $pathinfo)) 
  310.               $ret['filename'] = $pathinfo[3]; 
  311.       } 
  312.       switch ($options) { 
  313.           case PATHINFO_DIRNAME: 
  314.           case 'dirname': 
  315.               return $ret['dirname']; 
  316.           case PATHINFO_BASENAME: 
  317.           case 'basename': 
  318.               return $ret['basename']; 
  319.           case PATHINFO_EXTENSION: 
  320.           case 'extension': 
  321.               return $ret['extension']; 
  322.           case PATHINFO_FILENAME: 
  323.           case 'filename': 
  324.               return $ret['filename']; 
  325.           default: 
  326.               return $ret; 
  327.       } 
  328.   }
