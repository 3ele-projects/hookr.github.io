  157.   function locate_stale_displayed_galleries($post_id) 
  158.   { 
  159.       global $displayed_galleries_to_cleanup; 
  160.       $displayed_galleries_to_cleanup = array(); 
  161.       $post = get_post($post_id); 
  162.       $gallery_preview_url = C_NextGen_Settings::get_instance()->get('gallery_preview_url'); 
  163.       $preview_url = preg_quote($gallery_preview_url, '#'); 
  164.       if (preg_match_all("#{$preview_url}/id--(\d+)#", html_entity_decode($post->post_content), $matches, PREG_SET_ORDER)) { 
  165.           foreach ($matches as $match) { 
  166.               $preview_url = preg_quote($match[0], '/'); 
  167.               // The post was edited, and the displayed gallery placeholder was removed 
  168.               if (isset($_REQUEST['post_content']) && (!preg_match("/{$preview_url}/", $_POST['post_content']))) { 
  169.                   $displayed_galleries_to_cleanup[] = intval($match[1]); 
  170.               } 
  171.               // The post was deleted 
  172.               elseif (!isset($_REQUEST['action'])) { 
  173.                   $displayed_galleries_to_cleanup[] = intval($match[1]); 
  174.               } 
  175.           } 
  176.       } 
  177.   }

  291. static function _render_related_string($sluglist=array(), $maxImages=NULL, $type=NULL) 
  292. { 
  293.     $settings = C_NextGen_Settings::get_instance(); 
  294.     if (is_null($type)) $type = $settings->appendType; 
  295.     if (is_null($maxImages)) $maxImages = $settings->maxImages; 
  296.  
  297.     if (!$sluglist) { 
  298.         switch ($type) { 
  299.             case 'tags': 
  300.                 if (function_exists('get_the_tags')) 
  301.                 { 
  302.                     $taglist = get_the_tags(); 
  303.                     if (is_array($taglist)) { 
  304.                         foreach ($taglist as $tag) { 
  305.                             $sluglist[] = $tag->slug; 
  306.                         } 
  307.                     } 
  308.                 } 
  309.                 break; 
  310.             case 'category': 
  311.                 $catlist = get_the_category(); 
  312.                 if (is_array($catlist)) 
  313.                 { 
  314.                     foreach ($catlist as $cat) { 
  315.                         $sluglist[] = $cat->category_nicename; 
  316.                     } 
  317.                 } 
  318.                 break; 
  319.         } 
  320.     } 
  321.  
  322.     $taglist = implode(', ', $sluglist); 
  323.  
  324.     if ($taglist === 'uncategorized' || empty($taglist)) 
  325.         return; 
  326.  
  327.     $renderer = C_Displayed_Gallery_Renderer::get_instance(); 
  328.     $view = C_Component_Factory::get_instance()->create('mvc_view', ''); 
  329.     $retval = $renderer->display_images(array( 
  330.         'source' => 'tags',  
  331.         'container_ids' => $taglist,  
  332.         'display_type' => NGG_BASIC_THUMBNAILS,  
  333.         'images_per_page' => $maxImages,  
  334.         'maximum_entity_count' => $maxImages,  
  335.         'template' => $view->get_template_abspath('photocrati-nextgen_gallery_display#related'),  
  336.         'show_all_in_lightbox' => FALSE,  
  337.         'show_slideshow_link' => FALSE,  
  338.         'disable_pagination' => TRUE,  
  339.         'display_no_images_error' => FALSE 
  340.  )); 
  341.  
  342.     if ($retval) wp_enqueue_style('nextgen_gallery_related_images'); 
  343.  
  344.     return apply_filters('ngg_show_related_gallery_content', $retval, $taglist); 
  345. }
