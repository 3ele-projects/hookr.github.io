  664.    * 
  665.    * @param string $route 
  666.    * @return string 
  667.    */ 
  668.   function _route_to_regex($route) 
  669.   { 
  670.       // Get the settings manager 
  671.       $settings = $this->object->_settings; 
  672.       $param_slug = $settings->router_param_slug; 
  673.       // convert route to RegEx pattern 
  674.       $route_regex = preg_quote(str_replace(array('{', '}'), array('~', '~'), $route), '#'); 
  675.       // Wrap the route 
  676.       $route_regex = '(' . $route_regex . ')'; 
  677.       // If the route starts with a slash, then it must appear at the beginning 
  678.       // of a request uri 
  679.       if (strpos($route, '/') === 0) { 
  680.           $route_regex = '^' . $route_regex; 
  681.       } 
  682.       // If the route is not /, and perhaps /foo, then we need to optionally 
  683.       // look for a trailing slash as well 
  684.       if ($route != '/') { 
  685.           $route_regex .= '/?'; 
  686.       } 
  687.       // If parameters come after a slug, it might appear as well 
  688.       if ($param_slug) { 
  689.           $route_regex .= "(" . preg_quote($param_slug, '#') . '/)?'; 
  690.       } 
  691.       // Parameter might follow the request uri 
  692.       $route_regex .= "(/?([^/]+\\-\\-)?[^/]+\\-\\-[^/]+/?) {0, }"; 
  693.       // Create the regex 
  694.       $route_regex = '#' . $route_regex . '/?$#i'; 
  695.       // convert placeholders to regex as well 
  696.       return preg_replace('/~([^~]+)~/i', ($param_slug ? '(' . preg_quote($param_slug, '#') . '\\K)?' : '') . '(?P<\\1>[^/]+)/?', $route_regex); 
  697.   } 
  698.   /** 
  699.    * Gets a request parameter from either the request uri or querystring 
  700.    * This method takes into consideration the values of the router_param_prefix 
  701.    * and router_param_separator settings when searching for the parameter 
  702.    * 
  703.    * Parameter can take on the following forms: 
  704.    * /key--value 
  705.    * /[MVC_PARAM_PREFIX]key--value 
  706.    * /[MVC_PARAM_PREFIX]-key--value 
  707.    * /[MVC_PARAM_PREFIX]_key--value 
  708.    * /id--key--value 
  709.    * /id--[MVC_PARAM_PREFIX]key--value 
  710.    * /id--[MVC_PARAM_PREFIX]-key--value 
  711.    * /id--[MVC_PARAM_PREFIX]_key--value 
  712.    * 
  713.    * @param string $key 
  714.    * @param mixed $id 
  715.    * @param mixed $default 
  716.    * @return mixed 
  717.    */ 
  718.   function get_parameter($key, $id = NULL, $default = NULL, $segment = FALSE, $url = FALSE) 
  719.   { 
  720.       $retval = $default; 
  721.       $settings = $this->object->_settings; 
  722.       $quoted_key = preg_quote($key, '#'); 
  723.       $id = $id ? preg_quote($id, '#') : "[^/]+"; 
  724.       $param_prefix = preg_quote($settings->router_param_prefix, '#'); 
  725.       $param_sep = preg_quote($settings->router_param_separator, '#'); 
  726.       $param_regex = "#/((?P<id>{$id}) {$param_sep})?({$param_prefix}[-_]?)?{$quoted_key}{$param_sep}(?P<value>[^/\\?]+)/?#i"; 
  727.       $found = FALSE; 
  728.       $sources = $url ? array('custom' => $url) : $this->object->get_parameter_sources(); 
  729.       foreach ($sources as $source_name => $source) { 
  730.           if (preg_match($param_regex, $source, $matches)) { 
  731.               if ($segment) { 
  732.                   $retval = array('segment' => $matches[0], 'source' => $source_name); 
  733.               } else { 
  734.                   $retval = $this->object->recursive_stripslashes($matches['value']); 
  735.               } 
  736.               $found = TRUE; 
  737.               break; 
  738.           } 
  739.       } 
  740.       // Lastly, check the $_REQUEST 
  741.       if (!$found && !$url && isset($_REQUEST[$key])) { 
  742.           $found = TRUE; 
  743.           $retval = $this->object->recursive_stripslashes($_REQUEST[$key]); 
  744.       } 
  745.       if (!$found && isset($_SERVER['REQUEST_URI'])) { 
  746.           $params = array(); 
  747.           parse_str(@parse_url($_SERVER['REQUEST_URI'], PHP_URL_QUERY), $params); 
  748.           if (isset($params[$key])) { 
  749.               $found = TRUE; 
  750.               $retval = $this->object->recursive_stripslashes($params[$key]); 
  751.           } 
  752.       } 
  753.       return $retval; 
  754.   } 
  755.   /** 
  756.    * Checks and cleans a URL. This function is from WordPress. 
  757.    * 
  758.    * A number of characters are removed from the URL. If the URL is for displaying 
  759.    * (the default behaviour) ampersands are also replaced. The 'clean_url' filter 
  760.    * is applied to the returned cleaned URL. 
  761.    * 
  762.    * @since 2.8.0 
  763.    * @uses wp_kses_bad_protocol() To only permit protocols in the URL set 
  764.    *        via $protocols or the common ones set in the function. 
  765.    * 
  766.    * @param string $url The URL to be cleaned. 
  767.    * @param array $protocols Optional. An array of acceptable protocols. 
  768.    *        Defaults to 'http', 'https', 'ftp', 'ftps', 'mailto', 'news', 'irc', 'gopher', 'nntp', 'feed', 'telnet', 'mms', 'rtsp', 'svn' if not set. 
  769.    * @param string $_context Private. Use esc_url_raw() for database usage. 
  770.    * @return string The cleaned $url after the 'clean_url' filter is applied. 
  771.    */ 
  772.   function esc_url($url, $protocols = null, $_context = 'display') 
  773.   { 
  774.       $original_url = $url; 
  775.       $url = preg_replace('|[^a-z0-9-~+_.?#=!&;, /:%@$\\|*\'()\\x80-\\xff]|i', '', $url); 
  776.       $strip = array('%0d', '%0a', '%0D', '%0A'); 
  777.       $url = _deep_replace($strip, $url); 
  778.       $url = str_replace(';//', '://', $url); 
  779.       // Replace ampersands and single quotes only when displaying. 
  780.       if ('display' == $_context) { 
  781.           $url = wp_kses_normalize_entities($url); 
  782.           $url = str_replace('&', '&', $url); 
  783.           $url = str_replace("'", ''', $url); 
  784.       } 
  785.       if (!empty($url[0]) && '/' === $url[0]) { 
  786.           $good_protocol_url = $url; 
  787.       } else { 
  788.           if (!is_array($protocols)) { 
  789.               $protocols = wp_allowed_protocols(); 
  790.           } 
  791.           $good_protocol_url = wp_kses_bad_protocol($url, $protocols); 
  792.           if (strtolower($good_protocol_url) != strtolower($url)) { 
  793.               return ''; 
  794.           } 
  795.       } 
  796.       /** 
  797.        * Filter a string cleaned and escaped for output as a URL. 
  798.        * 
  799.        * @since 2.3.0 
  800.        * 
  801.        * @param string $good_protocol_url The cleaned URL to be returned. 
  802.        * @param string $original_url      The URL prior to cleaning. 
  803.        * @param string $_context          If 'display', replace ampersands and single quotes only. 
  804.        */ 
  805.       return apply_filters('clean_url', $good_protocol_url, $original_url, $_context); 
  806.   }
