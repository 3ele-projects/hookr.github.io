  588.   function prepare_legacy_album_params($displayed_gallery, $params) 
  589.   { 
  590.       $image_mapper = C_Image_Mapper::get_instance(); 
  591.       $storage = C_Gallery_Storage::get_instance(); 
  592.       $image_gen = C_Dynamic_Thumbnails_Manager::get_instance(); 
  593.       if (empty($displayed_gallery->display_settings['override_thumbnail_settings'])) { 
  594.           // legacy templates expect these dimensions 
  595.           $image_gen_params = array('width' => 91, 'height' => 68, 'crop' => TRUE); 
  596.       } else { 
  597.           // use settings requested by user 
  598.           $image_gen_params = array('width' => $displayed_gallery->display_settings['thumbnail_width'], 'height' => $displayed_gallery->display_settings['thumbnail_height'], 'quality' => isset($displayed_gallery->display_settings['thumbnail_quality']) ? $displayed_gallery->display_settings['thumbnail_quality'] : 100, 'crop' => isset($displayed_gallery->display_settings['thumbnail_crop']) ? $displayed_gallery->display_settings['thumbnail_crop'] : NULL, 'watermark' => isset($displayed_gallery->display_settings['thumbnail_watermark']) ? $displayed_gallery->display_settings['thumbnail_watermark'] : NULL); 
  599.       } 
  600.       // so user templates can know how big the images are expected to be 
  601.       $params['image_gen_params'] = $image_gen_params; 
  602.       // Transform entities 
  603.       $params['galleries'] = $params['entities']; 
  604.       unset($params['entities']); 
  605.       foreach ($params['galleries'] as &$gallery) { 
  606.           // Get the preview image url 
  607.           $gallery->previewurl = ''; 
  608.           if ($gallery->previewpic && $gallery->previewpic > 0) { 
  609.               if ($image = $image_mapper->find(intval($gallery->previewpic))) { 
  610.                   $gallery->previewpic_image = $image; 
  611.                   $gallery->previewpic_fullsized_url = $storage->get_image_url($image, 'full'); 
  612.                   $gallery->previewurl = $storage->get_image_url($image, $image_gen->get_size_name($image_gen_params), TRUE); 
  613.                   $gallery->previewname = $gallery->name; 
  614.               } 
  615.           } 
  616.           // Get the page link. If the entity is an album, then the url will 
  617.           // look like /nggallery/album--slug. 
  618.           $id_field = $gallery->id_field; 
  619.           if ($gallery->is_album) { 
  620.               if ($gallery->pageid > 0) { 
  621.                   $gallery->pagelink = @get_page_link($gallery->pageid); 
  622.               } else { 
  623.                   $pagelink = $this->object->get_routed_url(TRUE); 
  624.                   $pagelink = $this->object->remove_param_for($pagelink, 'album'); 
  625.                   $pagelink = $this->object->remove_param_for($pagelink, 'gallery'); 
  626.                   $pagelink = $this->object->remove_param_for($pagelink, 'nggpage'); 
  627.                   $pagelink = $this->object->set_param_for($pagelink, 'album', $gallery->slug); 
  628.                   $gallery->pagelink = $pagelink; 
  629.               } 
  630.           } else { 
  631.               if ($gallery->pageid > 0) { 
  632.                   $gallery->pagelink = @get_page_link($gallery->pageid); 
  633.               } 
  634.               if (empty($gallery->pagelink)) { 
  635.                   $pagelink = $this->object->get_routed_url(TRUE); 
  636.                   $parent_album = $this->object->get_parent_album_for($gallery->{$id_field}); 
  637.                   if ($parent_album) { 
  638.                       $pagelink = $this->object->remove_param_for($pagelink, 'album'); 
  639.                       $pagelink = $this->object->remove_param_for($pagelink, 'gallery'); 
  640.                       $pagelink = $this->object->remove_param_for($pagelink, 'nggpage'); 
  641.                       $pagelink = $this->object->set_param_for($pagelink, 'album', $parent_album->slug); 
  642.                   } else { 
  643.                       if ($displayed_gallery->container_ids === array('0') || $displayed_gallery->container_ids === array('')) { 
  644.                           $pagelink = $this->object->set_param_for($pagelink, 'album', 'all'); 
  645.                       } else { 
  646.                           $pagelink = $this->object->remove_param_for($pagelink, 'nggpage'); 
  647.                           $pagelink = $this->object->set_param_for($pagelink, 'album', 'album'); 
  648.                       } 
  649.                   } 
  650.                   $gallery->pagelink = $this->object->set_param_for($pagelink, 'gallery', $gallery->slug); 
  651.               } 
  652.           } 
  653.           // Mark the child type 
  654.           $gallery->entity_type = isset($gallery->is_gallery) && intval($gallery->is_gallery) ? 'gallery' : 'album'; 
  655.           // If this setting is on we need to inject an effect code 
  656.           if (!empty($displayed_gallery->display_settings['open_gallery_in_lightbox']) && $gallery->entity_type == 'gallery') { 
  657.               $gallery = $this->object->make_child_displayed_gallery($gallery, $displayed_gallery->display_settings); 
  658.               if ($this->does_lightbox_support_displayed_gallery($displayed_gallery)) { 
  659.                   $gallery->displayed_gallery->effect_code = $this->object->get_effect_code($gallery->displayed_gallery); 
  660.               } 
  661.           } 
  662.           // Let plugins modify the gallery 
  663.           $gallery = apply_filters('ngg_album_galleryobject', $gallery); 
  664.       } 
  665.       $params['galleries'] = apply_filters('ngg_album_prepared_child_entity', $params['galleries'], $params['displayed_gallery']); 
  666.       $params['album'] = reset($this->albums); 
  667.       $params['albums'] = $this->albums; 
  668.       // Clean up 
  669.       unset($storage); 
  670.       unset($image_mapper); 
  671.       unset($image_gen); 
  672.       unset($image_gen_params); 
  673.       return $params; 
  674.   }
