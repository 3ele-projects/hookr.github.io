  677.       } 
  678.

  685.   private function prepare_in_and_not_in( $sql, $post_ids ) { 
  686.       global $wpdb; 
  687.       $not_in_count = substr_count( $sql, '[IN_NOT_IN]' ); 
  688.       if ( $not_in_count > 0 ) { 
  689.           $args = array( str_replace( '[IN_NOT_IN]', implode( ', ', array_fill( 0, count( $post_ids ), '%d' ) ), str_replace( '%', '%%', $sql ) ) );
