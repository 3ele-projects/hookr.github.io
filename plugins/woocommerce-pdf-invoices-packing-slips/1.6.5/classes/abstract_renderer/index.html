  659.   protected function _border_line($x, $y, $length, $color, $widths, $side, $corner_style = "bevel", $pattern_name, $r1 = 0, $r2 = 0) { 
  660.     list($top, $right, $bottom, $left) = $widths; 
  661.      
  662.     $width = $$side; 
  663.     $pattern = $this->_get_dash_pattern($pattern_name, $width); 
  664.  
  665.     $half_width = $width/2; 
  666.     $r1 -= $half_width; 
  667.     $r2 -= $half_width; 
  668.     $adjust = $r1/80; 
  669.     $length -= $width;  
  670.      
  671.     switch ($side) { 
  672.       case "top": 
  673.         $x += $half_width;  
  674.         $y += $half_width; 
  675.    
  676.         if ( $r1 > 0 ) { 
  677.           $this->_canvas->arc($x + $r1, $y + $r1, $r1, $r1, 90-$adjust, 135+$adjust, $color, $width, $pattern); 
  678.         } 
  679.    
  680.         $this->_canvas->line($x + $r1, $y, $x + $length - $r2, $y, $color, $width, $pattern); 
  681.    
  682.         if ( $r2 > 0 ) { 
  683.           $this->_canvas->arc($x + $length - $r2, $y + $r2, $r2, $r2, 45-$adjust, 90+$adjust, $color, $width, $pattern); 
  684.         } 
  685.         break; 
  686.          
  687.       case "bottom": 
  688.         $x += $half_width;  
  689.         $y -= $half_width; 
  690.          
  691.         if ( $r1 > 0 ) { 
  692.           $this->_canvas->arc($x + $r1, $y - $r1, $r1, $r1, 225-$adjust, 270+$adjust, $color, $width, $pattern); 
  693.         } 
  694.          
  695.         $this->_canvas->line($x + $r1, $y, $x + $length - $r2, $y, $color, $width, $pattern); 
  696.          
  697.         if ( $r2 > 0 ) { 
  698.           $this->_canvas->arc($x + $length - $r2, $y - $r2, $r2, $r2, 270-$adjust, 315+$adjust, $color, $width, $pattern); 
  699.         } 
  700.         break; 
  701.          
  702.       case "left": 
  703.         $y += $half_width;  
  704.         $x += $half_width; 
  705.          
  706.         if ( $r1 > 0 ) { 
  707.           $this->_canvas->arc($x + $r1, $y + $r1, $r1, $r1, 135-$adjust, 180+$adjust, $color, $width, $pattern); 
  708.         } 
  709.          
  710.         $this->_canvas->line($x, $y + $r1, $x, $y + $length - $r2, $color, $width, $pattern); 
  711.          
  712.         if ( $r2 > 0 ) { 
  713.           $this->_canvas->arc($x + $r2, $y + $length - $r2, $r2, $r2, 180-$adjust, 225+$adjust, $color, $width, $pattern); 
  714.         } 
  715.         break; 
  716.          
  717.       case "right": 
  718.         $y += $half_width;  
  719.         $x -= $half_width; 
  720.          
  721.         if ( $r1 > 0 ) { 
  722.           $this->_canvas->arc($x - $r1, $y + $r1, $r1, $r1, 0-$adjust, 45+$adjust, $color, $width, $pattern); 
  723.         } 
  724.          
  725.         $this->_canvas->line($x, $y + $r1, $x, $y + $length - $r2, $color, $width, $pattern);

  729.         }

  731.     }
