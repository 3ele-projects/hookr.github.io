  711.   function prepend_child(Frame $child, $update_node = true) { 
  712.     if ( $update_node ) { 
  713.       $this->_node->insertBefore($child->_node, $this->_first_child ? $this->_first_child->_node : null); 
  714.     }

  716.     // Remove the child from its parent 
  717.     if ( $child->_parent ) { 
  718.       $child->_parent->remove_child($child, false); 
  719.     } 
  720.  
  721.     $child->_parent = $this; 
  722.     $child->_prev_sibling = null; 
  723.  
  724.     // Handle the first child 
  725.     if ( !$this->_first_child ) { 
  726.       $this->_first_child = $child; 
  727.       $this->_last_child = $child; 
  728.       $child->_next_sibling = null; 
  729.     } 
  730.     else { 
  731.       $this->_first_child->_prev_sibling = $child; 
  732.       $child->_next_sibling = $this->_first_child; 
  733.       $this->_first_child = $child; 
  734.     }

  771.    * 
  772.    * @param $new_child   Frame The new Frame to insert

  774.    * @param $update_node boolean Whether or not to update the DOM 
  775.    * 
  776.    * @throws DOMPDF_Exception 
  777.    */ 
  778.   function insert_child_before(Frame $new_child, Frame $ref, $update_node = true) { 
  779.     if ( $ref === $this->_first_child ) { 
  780.       $this->prepend_child($new_child, $update_node); 
  781.       return; 
  782.     } 
  783.  
  784.     if ( is_null($ref) ) { 
  785.       $this->append_child($new_child, $update_node); 
  786.       return; 
  787.     } 
  788.  
  789.     if ( $ref->_parent !== $this ) { 
  790.       throw new DOMPDF_Exception("Reference child is not a child of this node."); 
  791.     } 
  792.  
  793.     // Update the node 
  794.     if ( $update_node ) { 
  795.       $this->_node->insertBefore($new_child->_node, $ref->_node); 
  796.     } 
  797.  
  798.     // Remove the child from its parent 
  799.     if ( $new_child->_parent ) { 
  800.       $new_child->_parent->remove_child($new_child, false); 
  801.     } 
  802.  
  803.     $new_child->_parent = $this; 
  804.     $new_child->_next_sibling = $ref; 
  805.     $new_child->_prev_sibling = $ref->_prev_sibling; 
  806.  
  807.     if ( $ref->_prev_sibling ) { 
  808.       $ref->_prev_sibling->_next_sibling = $new_child; 
  809.     } 
  810.  
  811.     $ref->_prev_sibling = $new_child; 
  812.   }

  815.    * Inserts a new child immediately after the specified frame 
  816.    * 
  817.    * @param $new_child   Frame The new Frame to insert 
  818.    * @param $ref         Frame The Frame before the new Frame 
  819.    * @param $update_node boolean Whether or not to update the DOM 
  820.    * 
  821.    * @throws DOMPDF_Exception 
  822.    */ 
  823.   function insert_child_after(Frame $new_child, Frame $ref, $update_node = true) { 
  824.     if ( $ref === $this->_last_child ) { 
  825.       $this->append_child($new_child, $update_node); 
  826.       return; 
  827.     } 
  828.      
  829.     if ( is_null($ref) ) { 
  830.       $this->prepend_child($new_child, $update_node); 
  831.       return; 
  832.     } 
  833.  
  834.     if ( $ref->_parent !== $this ) { 
  835.       throw new DOMPDF_Exception("Reference child is not a child of this node."); 
  836.     } 
  837.  
  838.     // Update the node 
  839.     if ( $update_node ) { 
  840.       if ( $ref->_next_sibling ) { 
  841.         $next_node = $ref->_next_sibling->_node; 
  842.         $this->_node->insertBefore($new_child->_node, $next_node); 
  843.       } 
  844.       else { 
  845.         $new_child->_node = $this->_node->appendChild($new_child->_node); 
  846.       } 
  847.     } 
  848.  
  849.     // Remove the child from its parent 
  850.     if ( $new_child->_parent ) { 
  851.       $new_child->_parent->remove_child($new_child, false); 
  852.     } 
  853.  
  854.     $new_child->_parent = $this; 
  855.     $new_child->_prev_sibling = $ref; 
  856.     $new_child->_next_sibling = $ref->_next_sibling; 
  857.  
  858.     if ( $ref->_next_sibling ) { 
  859.       $ref->_next_sibling->_prev_sibling = $new_child; 
  860.     } 
  861.  
  862.     $ref->_next_sibling = $new_child; 
  863.   }

  868.    * 
  869.    * @param Frame   $child 
  870.    * @param boolean $update_node  Whether or not to remove the DOM node 
  871.    * 
  872.    * @throws DOMPDF_Exception 
  873.    * @return Frame The removed child frame 
  874.    */ 
  875.   function remove_child(Frame $child, $update_node = true) { 
  876.     if ( $child->_parent !== $this ) { 
  877.       throw new DOMPDF_Exception("Child not found in this frame"); 
  878.     } 
  879.  
  880.     if ( $update_node ) { 
  881.       $this->_node->removeChild($child->_node); 
  882.     } 
  883.  
  884.     if ( $child === $this->_first_child ) { 
  885.       $this->_first_child = $child->_next_sibling; 
  886.     } 
  887.  
  888.     if ( $child === $this->_last_child ) { 
  889.       $this->_last_child = $child->_prev_sibling; 
  890.     } 
  891.  
  892.     if ( $child->_prev_sibling ) { 
  893.       $child->_prev_sibling->_next_sibling = $child->_next_sibling; 
  894.     } 
  895.  
  896.     if ( $child->_next_sibling ) { 
  897.       $child->_next_sibling->_prev_sibling = $child->_prev_sibling; 
  898.     } 
  899.  
  900.     $child->_next_sibling = null; 
  901.     $child->_prev_sibling = null; 
  902.     $child->_parent = null; 
  903.     return $child;
