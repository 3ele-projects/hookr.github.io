  622.     //$this->_pdf->fonts[$this->_pdf->currentFont] sizes: 
  623.     //['FontBBox'][0] left, ['FontBBox'][1] bottom, ['FontBBox'][2] right, ['FontBBox'][3] top 
  624.     //Maximum extent of all glyphs of the font from the baseline point 
  625.     //['Ascender'] maximum height above baseline except accents 
  626.     //['Descender'] maximum depth below baseline, negative number means below baseline 
  627.     //['FontHeightOffset'] manual enhancement of .afm files to trim windows fonts. currently not used.

  636.     //FontBBox spans also the background box for the text. 
  637.     //If the lower corner would be used as reference point, the Descents of the glyphs would 
  638.     //hang over the background box border. 
  639.     //Therefore compensate only the extent above the Baseline.

  642.     //

  644.     $pdf->addText($x, $this->y($y) - $pdf->getFontHeight($size), $size, $text, $angle, $word_space, $char_space);

  773.   /** 
  774.    * Add text to each page after rendering is complete 
  775.    */ 
  776.   protected function _add_page_text() { 
  777.      
  778.     if ( !count($this->_page_text) ) { 
  779.       return; 
  780.     } 
  781.  
  782.     $page_number = 1; 
  783.     $eval = null; 
  784.  
  785.     foreach ($this->_pages as $pid) { 
  786.       $this->reopen_object($pid); 
  787.  
  788.       foreach ($this->_page_text as $pt) { 
  789.         extract($pt); 
  790.  
  791.         switch ($_t) { 
  792.           case "text": 
  793.             $text = str_replace(array("{PAGE_NUM}", "{PAGE_COUNT}"),  
  794.                                 array($page_number, $this->_page_count), $text); 
  795.             $this->text($x, $y, $text, $font, $size, $color, $word_space, $char_space, $angle); 
  796.             break; 
  797.              
  798.           case "script": 
  799.             if ( !$eval ) { 
  800.               $eval = new PHP_Evaluator($this); 
  801.             } 
  802.             $eval->evaluate($code, array('PAGE_NUM' => $page_number, 'PAGE_COUNT' => $this->_page_count)); 
  803.             break; 
  804.         } 
  805.       } 
  806.  
  807.       $this->close_object(); 
  808.       $page_number++; 
  809.     } 
  810.   }
