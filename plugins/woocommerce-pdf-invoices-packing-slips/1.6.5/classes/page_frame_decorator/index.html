  193.    * Determine if a page break is allowed before $frame 
  194.    * http://www.w3.org/TR/CSS21/page.html#allowed-page-breaks 
  195.    *  
  196.    * In the normal flow, page breaks can occur at the following places: 
  197.    *  
  198.    *    1. In the vertical margin between block boxes. When a page 
  199.    *    break occurs here, the used values of the relevant 
  200.    *    'margin-top' and 'margin-bottom' properties are set to '0'. 
  201.    *    2. Between line boxes inside a block box. 
  202.    * 
  203.    * These breaks are subject to the following rules: 
  204.    *  
  205.    *   * Rule A: Breaking at (1) is allowed only if the 
  206.    *     'page-break-after' and 'page-break-before' properties of 
  207.    *     all the elements generating boxes that meet at this margin 
  208.    *     allow it, which is when at least one of them has the value 
  209.    *     'always', 'left', or 'right', or when all of them are 
  210.    *     'auto'. 
  211.    * 
  212.    *   * Rule B: However, if all of them are 'auto' and the 
  213.    *     nearest common ancestor of all the elements has a 
  214.    *     'page-break-inside' value of 'avoid', then breaking here is 
  215.    *     not allowed. 
  216.    * 
  217.    *   * Rule C: Breaking at (2) is allowed only if the number of 
  218.    *     line boxes between the break and the start of the enclosing 
  219.    *     block box is the value of 'orphans' or more, and the number 
  220.    *     of line boxes between the break and the end of the box is 
  221.    *     the value of 'widows' or more. 
  222.    * 
  223.    *   * Rule D: In addition, breaking at (2) is allowed only if 
  224.    *     the 'page-break-inside' property is 'auto'. 
  225.    * 
  226.    * If the above doesn't provide enough break points to keep 
  227.    * content from overflowing the page boxes, then rules B and D are 
  228.    * dropped in order to find additional breakpoints. 
  229.    * 
  230.    * If that still does not lead to sufficient break points, rules A 
  231.    * and C are dropped as well, to find still more break points. 
  232.    * 
  233.    * We will also allow breaks between table rows.  However, when 
  234.    * splitting a table, the table headers should carry over to the 
  235.    * next page (but they don't yet). 
  236.    *  
  237.    * @param Frame $frame the frame to check 
  238.    * @return bool true if a break is allowed, false otherwise 
  239.    */

  394.    * Check if $frame will fit on the page.  If the frame does not fit,  
  395.    * the frame tree is modified so that a page break occurs in the

  397.    * 
  398.    * @param Frame $frame the frame to check 
  399.    * @return Frame the frame following the page break

  401.   function check_page_break(Frame $frame) { 
  402.     // Do not split if we have already or if the frame was already  
  403.     // pushed to the next page (prevents infinite loops) 
  404.     if ( $this->_page_full || $frame->_already_pushed ) { 
  405.       return false; 
  406.     } 
  407.      
  408.     // If the frame is absolute of fixed it shouldn't break 
  409.     $p = $frame; 
  410.     do { 
  411.       if ( $p->is_absolute() ) 
  412.         return false; 
  413.     } while ( $p = $p->get_parent() ); 
  414.      
  415.     $margin_height = $frame->get_margin_height(); 
  416.      
  417.     // FIXME If the row is taller than the page and  
  418.     // if it the first of the page, we don't break 
  419.     if ( $frame->get_style()->display === "table-row" && 
  420.          !$frame->get_prev_sibling() &&  
  421.          $margin_height > $this->get_margin_height() ) 
  422.       return false; 
  423.  
  424.     // Determine the frame's maximum y value

  433.                                            $style->padding_bottom,  
  434.                                            $style->border_bottom_width));

  492.         continue; 
  493.       } 
  494.

  496.          dompdf_debug("page-break", "following parent."); 
  497.  
  498.         if ( $iter->is_table() ) 
  499.           $this->_in_table--; 
  500.  
  501.         $iter = $next; 
  502.         $flg = true; 
  503.         continue; 
  504.       } 
  505.  
  506.       break; 
  507.     } 
  508.  
  509.     $this->_in_table = $in_table; 
  510.  
  511.     // No valid page break found.  Just break at $frame. 
  512.     dompdf_debug("page-break", "no valid break found, just splitting.");

  526.     $frame->_already_pushed = true; 
  527.     return true;
