  589.         $child->move(0, $lowest_y - $child->get_position("y")); 
  590.       } 
  591.     } 
  592.   } 
  593.  
  594.   /** 
  595.    * @param Frame $child 
  596.    * @param float $cb_x 
  597.    * @param float $cb_w 
  598.    */ 
  599.   function process_float(Frame $child, $cb_x, $cb_w) { 
  600.     $enable_css_float = $this->_frame->get_dompdf()->get_option("enable_css_float"); 
  601.     if ( !$enable_css_float ) { 
  602.       return; 
  603.     } 
  604.      
  605.     $child_style = $child->get_style(); 
  606.     $root = $this->_frame->get_root(); 
  607.      
  608.     // Handle "float" 
  609.     if ( $child_style->float !== "none" ) { 
  610.       $root->add_floating_frame($child); 
  611.        
  612.       // Remove next frame's beginning whitespace 
  613.       $next = $child->get_next_sibling(); 
  614.       if ( $next && $next instanceof Text_Frame_Decorator) { 
  615.         $next->set_text(ltrim($next->get_text())); 
  616.       } 
  617.        
  618.       $line_box = $this->_frame->get_current_line_box(); 
  619.       list($old_x, $old_y) = $child->get_position(); 
  620.        
  621.       $float_x = $cb_x; 
  622.       $float_y = $old_y; 
  623.       $float_w = $child->get_margin_width(); 
  624.        
  625.       if ( $child_style->clear === "none" ) { 
  626.         switch( $child_style->float ) { 
  627.           case "left":  
  628.             $float_x += $line_box->left; 
  629.             break; 
  630.           case "right":  
  631.             $float_x += ($cb_w - $line_box->right - $float_w); 
  632.             break; 
  633.         } 
  634.       } 
  635.       else { 
  636.         if ( $child_style->float === "right" ) { 
  637.           $float_x += ($cb_w - $float_w); 
  638.         } 
  639.       } 
  640.        
  641.       if ( $cb_w < $float_x + $float_w - $old_x ) { 
  642.         // TODO handle when floating elements don't fit 
  643.       } 
  644.        
  645.       $line_box->get_float_offsets(); 
  646.        
  647.       if ( $child->_float_next_line ) { 
  648.         $float_y += $line_box->h; 
  649.       } 
  650.        
  651.       $child->set_position($float_x, $float_y); 
  652.       $child->move($float_x - $old_x, $float_y - $old_y, true); 
  653.     } 
  654.   } 
  655.  
  656.   /** 
  657.    * @param Frame_Decorator $block 
  658.    */ 
  659.   function reflow(Block_Frame_Decorator $block = null) { 
  660.  
  661.     // Check if a page break is forced 
  662.     $page = $this->_frame->get_root(); 
  663.     $page->check_forced_page_break($this->_frame); 
  664.  
  665.     // Bail if the page is full 
  666.     if ( $page->is_full() ) { 
  667.       return; 
  668.     } 
  669.        
  670.     // Generated content 
  671.     $this->_set_content(); 
  672.  
  673.     // Collapse margins if required 
  674.     $this->_collapse_margins(); 
  675.  
  676.     $style = $this->_frame->get_style(); 
  677.     $cb = $this->_frame->get_containing_block(); 
  678.      
  679.     if ( $style->position === "fixed" ) { 
  680.       $cb = $this->_frame->get_root()->get_containing_block(); 
  681.     } 
  682.      
  683.     // Determine the constraints imposed by this frame: calculate the width 
  684.     // of the content area: 
  685.     list($w, $left_margin, $right_margin, $left, $right) = $this->_calculate_restricted_width(); 
  686.  
  687.     // Store the calculated properties 
  688.     $style->width = $w . "pt"; 
  689.     $style->margin_left = $left_margin."pt"; 
  690.     $style->margin_right = $right_margin."pt"; 
  691.     $style->left = $left ."pt"; 
  692.     $style->right = $right . "pt"; 
  693.      
  694.     // Update the position 
  695.     $this->_frame->position(); 
  696.     list($x, $y) = $this->_frame->get_position(); 
  697.  
  698.     // Adjust the first line based on the text-indent property 
  699.     $indent = $style->length_in_pt($style->text_indent, $cb["w"]); 
  700.     $this->_frame->increase_line_width($indent); 
  701.  
  702.     // Determine the content edge 
  703.     $top = $style->length_in_pt(array($style->margin_top,  
  704.                                       $style->padding_top,  
  705.                                       $style->border_top_width), $cb["h"]); 
  706.  
  707.     $bottom = $style->length_in_pt(array($style->border_bottom_width,  
  708.                                          $style->margin_bottom,  
  709.                                          $style->padding_bottom), $cb["h"]); 
  710.  
  711.     $cb_x = $x + $left_margin + $style->length_in_pt(array($style->border_left_width,  
  712.                                                            $style->padding_left), $cb["w"]); 
  713.  
  714.     $cb_y = $y + $top; 
  715.  
  716.     $cb_h = ($cb["h"] + $cb["y"]) - $bottom - $cb_y; 
  717.  
  718.     // Set the y position of the first line in this block 
  719.     $line_box = $this->_frame->get_current_line_box(); 
  720.     $line_box->y = $cb_y; 
  721.     $line_box->get_float_offsets(); 
  722.      
  723.     // Set the containing blocks and reflow each child 
  724.     foreach ( $this->_frame->get_children() as $child ) { 
  725.        
  726.       // Bail out if the page is full 
  727.       if ( $page->is_full() ) { 
  728.         break; 
  729.       } 
  730.        
  731.       $child->set_containing_block($cb_x, $cb_y, $w, $cb_h); 
  732.        
  733.       $this->process_clear($child); 
  734.        
  735.       $child->reflow($this->_frame); 
  736.        
  737.       // Don't add the child to the line if a page break has occurred 
  738.       if ( $page->check_page_break($child) ) { 
  739.         break; 
  740.       } 
  741.        
  742.       $this->process_float($child, $cb_x, $w); 
  743.     } 
  744.  
  745.     // Determine our height 
  746.     list($height, $margin_top, $margin_bottom, $top, $bottom) = $this->_calculate_restricted_height(); 
  747.     $style->height = $height; 
  748.     $style->margin_top = $margin_top; 
  749.     $style->margin_bottom = $margin_bottom; 
  750.     $style->top = $top; 
  751.     $style->bottom = $bottom; 
  752.      
  753.     $needs_reposition = ($style->position === "absolute" && ($style->right !== "auto" || $style->bottom !== "auto")); 
  754.      
  755.     // Absolute positioning measurement 
  756.     if ( $needs_reposition ) { 
  757.       $orig_style = $this->_frame->get_original_style(); 
  758.       if ( $orig_style->width === "auto" && ($orig_style->left === "auto" || $orig_style->right === "auto") ) { 
  759.         $width = 0; 
  760.         foreach ($this->_frame->get_line_boxes() as $line) { 
  761.           $width = max($line->w, $width); 
  762.         } 
  763.         $style->width = $width; 
  764.       } 
  765.        
  766.       $style->left = $orig_style->left; 
  767.       $style->right = $orig_style->right; 
  768.     } 
  769.  
  770.     $this->_text_align(); 
  771.     $this->vertical_align(); 
  772.      
  773.     // Absolute positioning 
  774.     if ( $needs_reposition ) { 
  775.       list($x, $y) = $this->_frame->get_position(); 
  776.       $this->_frame->position(); 
  777.       list($new_x, $new_y) = $this->_frame->get_position(); 
  778.       $this->_frame->move($new_x-$x, $new_y-$y, true); 
  779.     } 
  780.      
  781.     if ( $block && $this->_frame->is_in_flow() ) { 
  782.       $block->add_frame_to_line($this->_frame); 
  783.        
  784.       // May be inline-block 
  785.       if ( $style->display === "block" ) { 
  786.         $block->add_line(); 
  787.       } 
  788.     } 
  789.   } 
  790. }
