  53.   public function get_index_links( $max_entries ) { 
  54.  
  55.       $taxonomies = get_taxonomies( array( 'public' => true ), 'objects' ); 
  56.  
  57.       if ( empty( $taxonomies ) ) { 
  58.           return array(); 
  59.       } 
  60.  
  61.       $taxonomy_names = array_filter( array_keys( $taxonomies ), array( $this, 'is_valid_taxonomy' ) ); 
  62.       $taxonomies = array_intersect_key( $taxonomies, array_flip( $taxonomy_names ) ); 
  63.  
  64.       // Retrieve all the taxonomies and their terms so we can do a proper count on them. 
  65.       /** 
  66.        * Filter the setting of excluding empty terms from the XML sitemap. 
  67.        * 
  68.        * @param boolean $exclude        Defaults to true. 
  69.        * @param array   $taxonomy_names Array of names for the taxonomies being processed. 
  70.        */ 
  71.       $hide_empty = apply_filters( 'wpseo_sitemap_exclude_empty_terms', true, $taxonomy_names ); 
  72.  
  73.       $all_taxonomies = array(); 
  74.  
  75.       foreach ( $taxonomy_names as $taxonomy_name ) { 
  76.  
  77.           $taxonomy_terms = get_terms( $taxonomy_name, array( 
  78.               'hide_empty' => $hide_empty,  
  79.               'fields' => 'ids',  
  80.  ) ); 
  81.  
  82.           if ( count( $taxonomy_terms ) > 0 ) { 
  83.               $all_taxonomies[ $taxonomy_name ] = $taxonomy_terms; 
  84.           } 
  85.       } 
  86.  
  87.       $index = array(); 
  88.  
  89.       foreach ( $taxonomies as $tax_name => $tax ) { 
  90.  
  91.           if ( ! isset( $all_taxonomies[ $tax_name ] ) ) { // No eligible terms found. 
  92.               continue; 
  93.           } 
  94.  
  95.           $total_count = ( isset( $all_taxonomies[ $tax_name ] ) ) ? count( $all_taxonomies[ $tax_name ] ) : 1; 
  96.           $max_pages = 1; 
  97.  
  98.           if ( $total_count > $max_entries ) { 
  99.               $max_pages = (int) ceil( $total_count / $max_entries ); 
  100.           } 
  101.  
  102.           $last_modified_gmt = WPSEO_Sitemaps::get_last_modified_gmt( $tax->object_type ); 
  103.  
  104.           for ( $page_counter = 0; $page_counter < $max_pages; $page_counter++ ) { 
  105.  
  106.               $current_page = ( $max_pages > 1 ) ? ( $page_counter + 1 ) : ''; 
  107.  
  108.               if ( ! is_array( $tax->object_type ) || count( $tax->object_type ) == 0 ) { 
  109.                   continue; 
  110.               } 
  111.  
  112.               $terms = array_splice( $all_taxonomies[ $tax_name ], 0, $max_entries ); 
  113.  
  114.               if ( ! $terms ) { 
  115.                   continue; 
  116.               } 
  117.  
  118.               $args = array( 
  119.                   'post_type' => $tax->object_type,  
  120.                   'tax_query' => array( 
  121.                       array( 
  122.                           'taxonomy' => $tax_name,  
  123.                           'terms' => $terms,  
  124.  ),  
  125.  ),  
  126.                   'orderby' => 'modified',  
  127.                   'order' => 'DESC',  
  128.                   'posts_per_page' => 1,  
  129.  ); 
  130.               $query = new WP_Query( $args ); 
  131.  
  132.               if ( $query->have_posts() ) { 
  133.                   $date = $query->posts[0]->post_modified_gmt; 
  134.               } 
  135.               else { 
  136.                   $date = $last_modified_gmt; 
  137.               } 
  138.  
  139.               $index[] = array( 
  140.                   'loc' => WPSEO_Sitemaps_Router::get_base_url( $tax_name . '-sitemap' . $current_page . '.xml' ),  
  141.                   'lastmod' => $date,  
  142.  ); 
  143.           } 
  144.       } 
  145.  
  146.       return $index; 
  147.   }

  149.   /** 
  150.    * Get set of sitemap link data. 
  151.    * 
  152.    * @param string $type         Sitemap type. 
  153.    * @param int    $max_entries  Entries per sitemap. 
  154.    * @param int    $current_page Current page of the sitemap. 
  155.    * 
  156.    * @return array 
  157.    */ 
  158.   public function get_sitemap_links( $type, $max_entries, $current_page ) { 
  159.  
  160.       global $wpdb; 
  161.  
  162.       $links = array(); 
  163.       $taxonomy = get_taxonomy( $type ); 
  164.  
  165.       if ( $taxonomy === false || ! $this->is_valid_taxonomy( $taxonomy->name ) || ! $taxonomy->public ) { 
  166.           return $links; 
  167.       } 
  168.  
  169.       $options = $this->get_options(); 
  170.  
  171.       $steps = $max_entries; 
  172.       $offset = ( $current_page > 1 ) ? ( ( $current_page - 1 ) * $max_entries ) : 0; 
  173.  
  174.       /** This filter is documented in inc/sitemaps/class-taxonomy-sitemap-provider.php */ 
  175.       $hide_empty = apply_filters( 'wpseo_sitemap_exclude_empty_terms', true, $taxonomy ); 
  176.       $terms = get_terms( $taxonomy->name, array( 'hide_empty' => $hide_empty ) ); 
  177.       $terms = array_splice( $terms, $offset, $steps ); 
  178.  
  179.       if ( empty( $terms ) ) { 
  180.           $terms = array(); 
  181.       } 
  182.  
  183.       // Grab last modified date. 
  184.       $sql = " 
  185.           SELECT MAX(p.post_modified_gmt) AS lastmod 
  186.           FROM    $wpdb->posts AS p 
  187.           INNER JOIN $wpdb->term_relationships AS term_rel 
  188.               ON        term_rel.object_id = p.ID 
  189.           INNER JOIN $wpdb->term_taxonomy AS term_tax 
  190.               ON        term_tax.term_taxonomy_id = term_rel.term_taxonomy_id 
  191.               AND        term_tax.taxonomy = %s 
  192.               AND        term_tax.term_id = %d 
  193.           WHERE    p.post_status IN ('publish', 'inherit') 
  194.               AND        p.post_password = '' 
  195.       "; 
  196.  
  197.       foreach ( $terms as $term ) { 
  198.  
  199.           $url = array(); 
  200.  
  201.           $tax_noindex = WPSEO_Taxonomy_Meta::get_term_meta( $term, $term->taxonomy, 'noindex' ); 
  202.           $tax_sitemap_inc = WPSEO_Taxonomy_Meta::get_term_meta( $term, $term->taxonomy, 'sitemap_include' ); 
  203.  
  204.           if ( $tax_noindex === 'noindex' && $tax_sitemap_inc !== 'always' ) { 
  205.               continue; 
  206.           } 
  207.  
  208.           if ( $tax_sitemap_inc === 'never' ) { 
  209.               continue; 
  210.           } 
  211.  
  212.           $url['loc'] = WPSEO_Taxonomy_Meta::get_term_meta( $term, $term->taxonomy, 'canonical' ); 
  213.  
  214.           if ( ! is_string( $url['loc'] ) || $url['loc'] === '' ) { 
  215.  
  216.               $url['loc'] = get_term_link( $term, $term->taxonomy ); 
  217.  
  218.               if ( $options['trailingslash'] === true ) { 
  219.  
  220.                   $url['loc'] = trailingslashit( $url['loc'] ); 
  221.               } 
  222.           } 
  223.  
  224.           $url['mod'] = $wpdb->get_var( $wpdb->prepare( $sql, $term->taxonomy, $term->term_id ) ); 
  225.           $url['images'] = $this->get_image_parser()->get_term_images( $term ); 
  226.  
  227.           // Deprecated, kept for backwards data compat. R. 
  228.           $url['chf'] = 'daily'; 
  229.           $url['pri'] = 1; 
  230.  
  231.           /** This filter is documented at inc/sitemaps/class-post-type-sitemap-provider.php */ 
  232.           $url = apply_filters( 'wpseo_sitemap_entry', $url, 'term', $term ); 
  233.  
  234.           if ( ! empty( $url ) ) { 
  235.               $links[] = $url; 
  236.           } 
  237.       } 
  238.  
  239.       return $links; 
  240.   }

  260.       if ( 'post_format' === $taxonomy_name && ! empty( $options['disable-post_format'] ) ) {

  271.           return false; 
  272.       } 
  273.  
  274.       return true; 
  275.   }
