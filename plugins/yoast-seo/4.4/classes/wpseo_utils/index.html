  541.      *                            '+' or 'add' or 'addition',

  543.      *                            '*' or 'mul' or 'multiply',  
  544.      *                            '/' or 'div' or 'divide',

  559.     public static function calc( $number1, $action, $number2, $round = false, $decimals = 0, $precision = 10 ) { 
  560.         static $bc; 
  561.  
  562.         if ( ! is_scalar( $number1 ) || ! is_scalar( $number2 ) ) { 
  563.             return false; 
  564.         } 
  565.  
  566.         if ( ! isset( $bc ) ) { 
  567.             $bc = extension_loaded( 'bcmath' ); 
  568.         } 
  569.  
  570.         if ( $bc ) { 
  571.             $number1 = number_format( $number1, 10, '.', '' ); 
  572.             $number2 = number_format( $number2, 10, '.', '' ); 
  573.         } 
  574.  
  575.         $result = null; 
  576.         $compare = false; 
  577.  
  578.         switch ( $action ) { 
  579.             case '+': 
  580.             case 'add': 
  581.             case 'addition': 
  582.                 $result = ( $bc ) ? bcadd( $number1, $number2, $precision ) /** string */ : ( $number1 + $number2 ); 
  583.                 break; 
  584.  
  585.             case '-': 
  586.             case 'sub': 
  587.             case 'subtract': 
  588.                 $result = ( $bc ) ? bcsub( $number1, $number2, $precision ) /** string */ : ( $number1 - $number2 ); 
  589.                 break; 
  590.  
  591.             case '*': 
  592.             case 'mul': 
  593.             case 'multiply': 
  594.                 $result = ( $bc ) ? bcmul( $number1, $number2, $precision ) /** string */ : ( $number1 * $number2 ); 
  595.                 break; 
  596.  
  597.             case '/': 
  598.             case 'div': 
  599.             case 'divide': 
  600.                 if ( $bc ) { 
  601.                     $result = bcdiv( $number1, $number2, $precision ); // String, or NULL if right_operand is 0. 
  602.                 } 
  603.                 elseif ( $number2 != 0 ) { 
  604.                     $result = ( $number1 / $number2 ); 
  605.                 } 
  606.  
  607.                 if ( ! isset( $result ) ) { 
  608.                     $result = 0; 
  609.                 } 
  610.                 break; 
  611.  
  612.             case '%': 
  613.             case 'mod': 
  614.             case 'modulus': 
  615.                 if ( $bc ) { 
  616.                     $result = bcmod( $number1, $number2 ); // String, or NULL if modulus is 0. 
  617.                 } 
  618.                 elseif ( $number2 != 0 ) { 
  619.                     $result = ( $number1 % $number2 ); 
  620.                 } 
  621.  
  622.                 if ( ! isset( $result ) ) { 
  623.                     $result = 0; 
  624.                 } 
  625.                 break; 
  626.  
  627.             case '=': 
  628.             case 'comp': 
  629.             case 'compare': 
  630.                 $compare = true; 
  631.                 if ( $bc ) { 
  632.                     $result = bccomp( $number1, $number2, $precision ); // Returns int 0, 1 or -1. 
  633.                 } 
  634.                 else { 
  635.                     $result = ( $number1 == $number2 ) ? 0 : ( ( $number1 > $number2 ) ? 1 : - 1 ); 
  636.                 } 
  637.                 break; 
  638.         } 
  639.  
  640.         if ( isset( $result ) ) { 
  641.             if ( $compare === false ) { 
  642.                 if ( $round === true ) { 
  643.                     $result = round( floatval( $result ), $decimals ); 
  644.                     if ( $decimals === 0 ) { 
  645.                         $result = (int) $result; 
  646.                     } 
  647.                 } 
  648.                 else { 
  649.                     $result = ( intval( $result ) == $result ) ? intval( $result ) : floatval( $result ); 
  650.                 } 
  651.             } 
  652.  
  653.             return $result; 
  654.         } 
  655.  
  656.         return false; 
  657.     }

  871.     public static function home_url( $path = '', $scheme = null ) { 
  872.  
  873.         $home_url = home_url( $path, $scheme ); 
  874.  
  875.         if ( ! empty( $path ) ) { 
  876.             return $home_url; 
  877.         } 
  878.  
  879.         $home_path = parse_url( $home_url, PHP_URL_PATH ); 
  880.  
  881.         if ( '/' === $home_path ) { // Home at site root, already slashed. 
  882.             return $home_url;
