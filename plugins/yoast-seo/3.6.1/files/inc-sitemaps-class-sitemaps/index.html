  387.    * Get the GMT modification date for the last modified post in the post type. 
  388.    * 
  389.    * @param string|array $post_types Post type or array of types.

  394.   static public function get_last_modified_gmt( $post_types, $return_all = false ) { 
  395.  
  396.       global $wpdb; 
  397.  
  398.       static $post_type_dates = null; 
  399.  
  400.       if ( ! is_array( $post_types ) ) {

  402.       } 
  403.  
  404.       foreach ( $post_types as $post_type ) { 
  405.           if ( ! isset( $post_type_dates[ $post_type ] ) ) { // If we hadn't seen post type before. R. 
  406.               $post_type_dates = null; 
  407.               break; 
  408.           } 
  409.       } 
  410.  
  411.       if ( is_null( $post_type_dates ) ) { 
  412.  
  413.           $sql = "

  415.               FROM $wpdb->posts 
  416.               WHERE post_status IN ('publish', 'inherit') 
  417.                   AND post_type IN ('" . implode( "', '", get_post_types( array( 'public' => true ) ) ) . "') 
  418.               GROUP BY post_type 
  419.               ORDER BY post_modified_gmt DESC 
  420.           ";

  422.           $post_type_dates = array(); 
  423.
