  193.      * Find the deepest term in an array of term objects 
  194.      * 
  195.      * @param array $terms Terms set. 
  196.      * 
  197.      * @return object 
  198.      */ 
  199.     private function find_deepest_term( $terms ) { 
  200.         /** 
  201.         Let's find the deepest term in this array, by looping through and then 
  202.            unsetting every term that is used as a parent by another one in the array. 
  203.         */ 
  204.         $terms_by_id = array(); 
  205.         foreach ( $terms as $term ) { 
  206.             $terms_by_id[ $term->term_id ] = $term; 
  207.         } 
  208.         foreach ( $terms as $term ) { 
  209.             unset( $terms_by_id[ $term->parent ] ); 
  210.         } 
  211.         unset( $term ); 
  212.  
  213.         /** 
  214.         As we could still have two subcategories, from different parent categories,  
  215.            let's pick the one with the lowest ordered ancestor. 
  216.         */ 
  217.         $parents_count = 0; 
  218.         $term_order = 9999; // Because ASC. 
  219.         reset( $terms_by_id ); 
  220.         $deepest_term = current( $terms_by_id ); 
  221.         foreach ( $terms_by_id as $term ) { 
  222.             $parents = $this->get_term_parents( $term ); 
  223.  
  224.             if ( count( $parents ) >= $parents_count ) { 
  225.                 $parents_count = count( $parents ); 
  226.  
  227.                 // If higher count. 
  228.                 if ( count( $parents ) > $parents_count ) { 
  229.                     // Reset order. 
  230.                     $term_order = 9999; 
  231.                 } 
  232.  
  233.                 $parent_order = 9999; // Set default order. 
  234.                 foreach ( $parents as $parent ) { 
  235.                     if ( $parent->parent == 0 && isset( $parent->term_order ) ) { 
  236.                         $parent_order = $parent->term_order; 
  237.                     } 
  238.                 } 
  239.                 unset( $parent ); 
  240.  
  241.                 // Check if parent has lowest order. 
  242.                 if ( $parent_order < $term_order ) { 
  243.                     $term_order = $parent_order; 
  244.                     $deepest_term = $term; 
  245.                 } 
  246.             } 
  247.         } 
  248.  
  249.         return $deepest_term; 
  250.     }

  613.     /** 
  614.      * Take the crumbs array and convert each crumb to a single breadcrumb string. 
  615.      * 
  616.      * @link http://support.google.com/webmasters/bin/answer.py?hl=en&answer=185417 Google documentation on RDFA 
  617.      */ 
  618.     private function prepare_links() { 
  619.         if ( ! is_array( $this->crumbs ) || $this->crumbs === array() ) { 
  620.             return; 
  621.         } 
  622.  
  623.         foreach ( $this->crumbs as $i => $crumb ) { 
  624.             $link_info = $crumb; // Keep pre-set url/text combis. 
  625.  
  626.             if ( isset( $crumb['id'] ) ) { 
  627.                 $link_info = $this->get_link_info_for_id( $crumb['id'] ); 
  628.             } 
  629.             if ( isset( $crumb['term'] ) ) { 
  630.                 $link_info = $this->get_link_info_for_term( $crumb['term'] ); 
  631.             } 
  632.             if ( isset( $crumb['ptarchive'] ) ) { 
  633.                 $link_info = $this->get_link_info_for_ptarchive( $crumb['ptarchive'] ); 
  634.             } 
  635.  
  636.             $this->links[] = $this->crumb_to_link( $link_info, $i ); 
  637.         } 
  638.     }

  641.      * Retrieve link url and text based on post id 
  642.      * 
  643.      * @param int $id Post ID. 
  644.      * 
  645.      * @return array Array of link text and url 
  646.      */ 
  647.     private function get_link_info_for_id( $id ) { 
  648.         $link = array(); 
  649.  
  650.         $link['url'] = get_permalink( $id ); 
  651.         $link['text'] = WPSEO_Meta::get_value( 'bctitle', $id ); 
  652.         if ( $link['text'] === '' ) { 
  653.             $link['text'] = strip_tags( get_the_title( $id ) );

  660.          * 
  661.          * @param int $link_id The post ID 
  662.          */ 
  663.         $link['text'] = apply_filters( 'wp_seo_get_bc_title', $link['text'], $id ); 
  664.  
  665.         return $link;

  690.      * Retrieve link url and text based on post type 
  691.      * 
  692.      * @param string $pt Post type. 
  693.      * 
  694.      * @return array Array of link text and url 
  695.      */ 
  696.     private function get_link_info_for_ptarchive( $pt ) { 
  697.         $link = array(); 
  698.         $archive_title = ''; 
  699.  
  700.         if ( isset( $this->options[ 'bctitle-ptarchive-' . $pt ] ) && $this->options[ 'bctitle-ptarchive-' . $pt ] !== '' ) { 
  701.  
  702.             $archive_title = $this->options[ 'bctitle-ptarchive-' . $pt ]; 
  703.         } 
  704.         else { 
  705.             $post_type_obj = get_post_type_object( $pt ); 
  706.             if ( is_object( $post_type_obj ) ) { 
  707.                 if ( isset( $post_type_obj->label ) && $post_type_obj->label !== '' ) { 
  708.                     $archive_title = $post_type_obj->label; 
  709.                 } 
  710.                 elseif ( isset( $post_type_obj->labels->menu_name ) && $post_type_obj->labels->menu_name !== '' ) { 
  711.                     $archive_title = $post_type_obj->labels->menu_name; 
  712.                 } 
  713.                 else { 
  714.                     $archive_title = $post_type_obj->name; 
  715.                 } 
  716.             } 
  717.         } 
  718.  
  719.         $link['url'] = get_post_type_archive_link( $pt ); 
  720.         $link['text'] = $archive_title; 
  721.  
  722.         return $link; 
  723.     }

  726.     /** 
  727.      * Create a breadcrumb element string 
  728.      * 
  729.      * @todo The `$paged` variable only works for archives, not for paged articles, so this does not work 
  730.      * for paged article at this moment. 
  731.      * 
  732.      * @param  array $link Link info array containing the keys: 
  733.      *                     'text' => (string) link text 
  734.      *                     'url' => (string) link url 
  735.      *                     (optional) 'title' => (string) link title attribute text 
  736.      *                     (optional) 'allow_html' => (bool) whether to (not) escape html in the link text 
  737.      *                     This prevents html stripping from the text strings set in the 
  738.      *                     WPSEO -> Internal Links options page. 
  739.      * @param  int   $i    Index for the current breadcrumb. 
  740.      * 
  741.      * @return string 
  742.      */ 
  743.     private function crumb_to_link( $link, $i ) { 
  744.         $link_output = ''; 
  745.  
  746.         if ( isset( $link['text'] ) && ( is_string( $link['text'] ) && $link['text'] !== '' ) ) { 
  747.  
  748.             $link['text'] = trim( $link['text'] ); 
  749.             if ( ! isset( $link['allow_html'] ) || $link['allow_html'] !== true ) { 
  750.                 $link['text'] = esc_html( $link['text'] ); 
  751.             } 
  752.  
  753.             $inner_elm = 'span'; 
  754.             if ( $this->options['breadcrumbs-boldlast'] === true && $i === ( $this->crumb_count - 1 ) ) { 
  755.                 $inner_elm = 'strong'; 
  756.             } 
  757.  
  758.             if ( ( isset( $link['url'] ) && ( is_string( $link['url'] ) && $link['url'] !== '' ) ) && 
  759.                  ( $i < ( $this->crumb_count - 1 ) ) 
  760.  ) { 
  761.                 if ( $i === 0 ) { 
  762.                     $link_output .= '<' . $this->element . ' typeof="v:Breadcrumb">'; 
  763.                 } 
  764.                 else { 
  765.                     $link_output .= '<' . $this->element . ' rel="v:child" typeof="v:Breadcrumb">'; 
  766.                 } 
  767.                 $title_attr = isset( $link['title'] ) ? ' title="' . esc_attr( $link['title'] ) . '"' : ''; 
  768.                 $link_output .= '<a href="' . esc_url( $link['url'] ) . '" rel="v:url" property="v:title"' . $title_attr . '>' . $link['text'] . '</a>'; 
  769.             } 
  770.             else { 
  771.                 $link_output .= '<' . $inner_elm . ' class="breadcrumb_last">' . $link['text'] . '</' . $inner_elm . '>'; 
  772.                 // This is the last element, now close all previous elements. 
  773.                 while ( $i > 0 ) { 
  774.                     $link_output .= '</' . $this->element . '>'; 
  775.                     $i--; 
  776.                 } 
  777.             } 
  778.         } 
  779.  
  780.         /** 
  781.          * Filter: 'wpseo_breadcrumb_single_link' - Allow changing of each link being put out by the Yoast SEO breadcrumbs class 
  782.          * 
  783.          * @api string $link_output The output string 
  784.          * 
  785.          * @param array $link The link array. 
  786.          */ 
  787.  
  788.         return apply_filters( 'wpseo_breadcrumb_single_link', $link_output, $link ); 
  789.     }

  792.     /** 
  793.      * Create a complete breadcrumb string from an array of breadcrumb element strings 
  794.      */ 
  795.     private function links_to_string() { 
  796.         if ( is_array( $this->links ) && $this->links !== array() ) { 
  797.             // Remove any effectively empty links. 
  798.             $links = array_map( 'trim', $this->links ); 
  799.             $links = array_filter( $links ); 
  800.  
  801.             $this->output = implode( $this->separator, $links ); 
  802.         } 
  803.     }

  805.     /** 
  806.      * Wrap a complete breadcrumb string in a Breadcrumb RDFA wrapper 
  807.      */ 
  808.     private function wrap_breadcrumb() { 
  809.         if ( is_string( $this->output ) && $this->output !== '' ) { 
  810.             $output = '<' . $this->wrapper . $this->get_output_id() . $this->get_output_class() . ' xmlns:v="http://rdf.data-vocabulary.org/#">' . $this->output . '</' . $this->wrapper . '>'; 
  811.  
  812.             /** 
  813.              * Filter: 'wpseo_breadcrumb_output' - Allow changing the HTML output of the Yoast SEO breadcrumbs class 
  814.              * 
  815.              * @api string $unsigned HTML output 
  816.              */ 
  817.             $output = apply_filters( 'wpseo_breadcrumb_output', $output ); 
  818.  
  819.             if ( $this->options['breadcrumbs-prefix'] !== '' ) { 
  820.                 $output = "\t" . $this->options['breadcrumbs-prefix'] . "\n" . $output; 
  821.             } 
  822.  
  823.             $this->output = $output; 
  824.         } 
  825.     }
