  565.     /** 
  566.      * Parse the query to get items from database. 
  567.      * 
  568.      * Based on given parameters there will be parse a query which will get all the pages/posts and other post_types 
  569.      * from the database. 
  570.      * 
  571.      * @param string $subquery         SQL FROM part. 
  572.      * @param string $all_states       SQL IN part. 
  573.      * @param string $post_type_clause SQL post type part. 
  574.      * 
  575.      * @return string 
  576.      */ 
  577.     protected function parse_item_query( $subquery, $all_states, $post_type_clause ) { 
  578.         // Order By block. 
  579.         $orderby = filter_input( INPUT_GET, 'orderby' ); 
  580.  
  581.         $orderby = ! empty( $orderby ) ? esc_sql( sanitize_text_field( $orderby ) ) : 'post_title'; 
  582.         $orderby = $this->sanitize_orderby( $orderby ); 
  583.  
  584.         // Order clause. 
  585.         $order = filter_input( INPUT_GET, 'order' ); 
  586.         $order = ! empty( $order ) ? esc_sql( strtoupper( sanitize_text_field( $order ) ) ) : 'ASC'; 
  587.         $order = $this->sanitize_order( $order ); 
  588.  
  589.         // Get all needed results. 
  590.         $query = " 
  591.                 SELECT ID, post_title, post_type, post_status, post_modified, post_date 
  592.                 FROM {$subquery} 
  593.                 WHERE post_status IN ({$all_states}) $post_type_clause 
  594.                 ORDER BY {$orderby} {$order} 
  595.                 LIMIT %d, %d 
  596.             "; 
  597.  
  598.         return $query; 
  599.     }

  602.      * Heavily restricts the possible columns by which a user can order the table in the bulk editor, thereby preventing a possible CSRF vulnerability. 
  603.      * 
  604.      * @param string $orderby The column by which we want to order. 
  605.      * 
  606.      * @return string $orderby 
  607.      */ 
  608.     protected function sanitize_orderby( $orderby ) { 
  609.         $valid_column_names = array( 
  610.             'post_title',  
  611.             'post_type',  
  612.             'post_date',  
  613.  ); 
  614.  
  615.         if ( in_array( $orderby, $valid_column_names ) ) { 
  616.             return $orderby; 
  617.         } 
  618.  
  619.         return 'post_title'; 
  620.     }

  622.     /** 
  623.      * Makes sure the order clause is always ASC or DESC for the bulk editor table, thereby preventing a possible CSRF vulnerability. 
  624.      * 
  625.      * @param string $order Whether we want to sort ascending or descending. 
  626.      * 
  627.      * @return string $order SQL order string (ASC, DESC). 
  628.      */ 
  629.     protected function sanitize_order( $order ) { 
  630.         if ( in_array( strtoupper( $order ), array( 'ASC', 'DESC' ) ) ) { 
  631.             return $order; 
  632.         } 
  633.  
  634.         return 'ASC'; 
  635.     }

  752.     protected function parse_page_title_column( $rec ) { 
  753.  
  754.         $title = empty( $rec->post_title ) ? __( '(no title)', 'wordpress-seo' ) : $rec->post_title; 
  755.  
  756.         $return = sprintf( '<strong>%1$s</strong>', stripslashes( wp_strip_all_tags( $title ) ) );

  758.         $post_type_object = get_post_type_object( $rec->post_type );

  802.     /** 
  803.      * Parsing the column based on the $column_name. 
  804.      * 
  805.      * @param string  $column_name Column name. 
  806.      * @param WP_Post $rec         Post object. 
  807.      * 
  808.      * @return string 
  809.      */ 
  810.     protected function parse_column( $column_name, $rec ) { 
  811.  
  812.         static $date_format; 
  813.  
  814.         if ( $date_format == null ) { 
  815.             $date_format = get_option( 'date_format' ); 
  816.         } 
  817.  
  818.         switch ( $column_name ) { 
  819.             case 'col_page_title': 
  820.                 $column_value = $this->parse_page_title_column( $rec ); 
  821.                 break; 
  822.  
  823.             case 'col_page_slug': 
  824.                 $permalink = get_permalink( $rec->ID ); 
  825.                 $display_slug = str_replace( get_bloginfo( 'url' ), '', $permalink ); 
  826.                 $column_value = sprintf( '<a href="%2$s" target="_blank">%1$s</a>', stripslashes( $display_slug ), esc_url( $permalink ) ); 
  827.                 break; 
  828.  
  829.             case 'col_post_type': 
  830.                 $post_type = get_post_type_object( $rec->post_type ); 
  831.                 $column_value = $post_type->labels->singular_name; 
  832.                 break; 
  833.  
  834.             case 'col_post_status': 
  835.                 $post_status = get_post_status_object( $rec->post_status ); 
  836.                 $column_value = $post_status->label; 
  837.                 break; 
  838.  
  839.             case 'col_post_date': 
  840.                 $column_value = date_i18n( $date_format, strtotime( $rec->post_date ) ); 
  841.                 break; 
  842.  
  843.             case 'col_row_action': 
  844.                 $column_value = sprintf( 
  845.                     '<a href="#" role="button" class="wpseo-save" data-id="%1$s">%2$s</a> <span aria-hidden="true">|</span> <a href="#" role="button" class="wpseo-save-all">%3$s</a>',  
  846.                     $rec->ID,  
  847.                     esc_html__( 'Save', 'wordpress-seo' ),  
  848.                     esc_html__( 'Save all', 'wordpress-seo' ) 
  849.  ); 
  850.                 break; 
  851.         } 
  852.  
  853.         if ( ! empty( $column_value ) ) { 
  854.             return $column_value; 
  855.         } 
  856.     }
