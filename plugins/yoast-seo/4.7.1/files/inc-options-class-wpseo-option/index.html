  6. /** 
  7.  * This abstract class and it's concrete classes implement defaults and value validation for 
  8.  * all WPSEO options and subkeys within options. 
  9.  * 
  10.  * Some guidelines: 
  11.  * [Retrieving options] 
  12.  * - Use the normal get_option() to retrieve an option. You will receive a complete array for the option. 
  13.  *    Any subkeys which were not set, will have their default values in place. 
  14.  * - In other words, you will normally not have to check whether a subkey isset() as they will *always* be set. 
  15.  *    They will also *always* be of the correct variable type. 
  16.  *    The only exception to this are the options with variable option names based on post_type or taxonomy 
  17.  *    as those will not always be available before the taxonomy/post_type is registered. 
  18.  *    (they will be available if a value was set, they won't be if it wasn't as the class won't know 
  19.  *    that a default needs to be injected). 
  20.  * 
  21.  * [Updating/Adding options] 
  22.  * - For multisite site_options, please use the WPSEO_Options::update_site_option() method. 
  23.  * - For normal options, use the normal add/update_option() functions. As long a the classes here 
  24.  *   are instantiated, validation for all options and their subkeys will be automatic. 
  25.  * - On (succesfull) update of a couple of options, certain related actions will be run automatically. 
  26.  *    Some examples: 
  27.  *      - on change of wpseo[yoast_tracking], the cron schedule will be adjusted accordingly 
  28.  *      - on change of wpseo_permalinks and wpseo_xml, the rewrite rules will be flushed 
  29.  *      - on change of wpseo and wpseo_title, some caches will be cleared 
  30.  * 
  31.  * 
  32.  * [Important information about add/updating/changing these classes] 
  33.  * - Make sure that option array key names are unique across options. The WPSEO_Options::get_all() 
  34.  *    method merges most options together. If any of them have non-unique names, even if they 
  35.  *    are in a different option, they *will* overwrite each other. 
  36.  * - When you add a new array key in an option: make sure you add proper defaults and add the key 
  37.  *    to the validation routine in the proper place or add a new validation case. 
  38.  *    You don't need to do any upgrading as any option returned will always be merged with the 
  39.  *    defaults, so new options will automatically be available. 
  40.  *    If the default value is a string which need translating, add this to the concrete class 
  41.  *    translate_defaults() method. 
  42.  * - When you remove an array key from an option: if it's important that the option is really removed,  
  43.  *    add the WPSEO_Option::clean_up( $option_name ) method to the upgrade run. 
  44.  *    This will re-save the option and automatically remove the array key no longer in existance. 
  45.  * - When you rename a sub-option: add it to the clean_option() routine and run that in the upgrade run. 
  46.  * - When you change the default for an option sub-key, make sure you verify that the validation routine will 
  47.  *    still work the way it should. 
  48.  *    Example: changing a default from '' (empty string) to 'text' with a validation routine with tests 
  49.  *    for an empty string will prevent a user from saving an empty string as the real value. So the 
  50.  *    test for '' with the validation routine would have to be removed in that case. 
  51.  * - If an option needs specific actions different from defined in this abstract class, you can just overrule 
  52.  *    a method by defining it in the concrete class. 
  53.  * 
  54.  * @todo       - [JRF => testers] double check that validation will not cause errors when called 
  55.  *               from upgrade routine (some of the WP functions may not yet be available) 
  56.  */

  235.   /** 
  236.    * Validate webmaster tools & Pinterest verification strings 
  237.    * 
  238.    * @param string $key   Key to check, by type of service.

  243.   public function validate_verification_string( $key, $dirty, $old, &$clean ) { 
  244.       if ( isset( $dirty[ $key ] ) && $dirty[ $key ] !== '' ) { 
  245.           $meta = $dirty[ $key ]; 
  246.           if ( strpos( $meta, 'content=' ) ) { 
  247.               // Make sure we only have the real key, not a complete meta tag. 
  248.               preg_match( '`content=([\'"])?([^\'"> ]+)(?:\1|[ />])`', $meta, $match ); 
  249.               if ( isset( $match[2] ) ) { 
  250.                   $meta = $match[2]; 
  251.               } 
  252.               unset( $match ); 
  253.           } 
  254.  
  255.           $meta = sanitize_text_field( $meta ); 
  256.           if ( $meta !== '' ) { 
  257.               $regex = '`^[A-Fa-f0-9_-]+$`'; 
  258.               $service = ''; 
  259.  
  260.               switch ( $key ) { 
  261.                   case 'googleverify': 
  262.                       $regex = '`^[A-Za-z0-9_-]+$`'; 
  263.                       $service = 'Google Webmaster tools'; 
  264.                       break; 
  265.  
  266.                   case 'msverify': 
  267.                       $service = 'Bing Webmaster tools'; 
  268.                       break; 
  269.  
  270.                   case 'pinterestverify': 
  271.                       $service = 'Pinterest'; 
  272.                       break; 
  273.  
  274.                   case 'yandexverify': 
  275.                       $service = 'Yandex Webmaster tools'; 
  276.                       break; 
  277.               } 
  278.  
  279.               if ( preg_match( $regex, $meta ) ) { 
  280.                   $clean[ $key ] = $meta; 
  281.               } 
  282.               else { 
  283.                   if ( isset( $old[ $key ] ) && preg_match( $regex, $old[ $key ] ) ) { 
  284.                       $clean[ $key ] = $old[ $key ]; 
  285.                   } 
  286.                   if ( function_exists( 'add_settings_error' ) ) { 
  287.                       add_settings_error( 
  288.                           $this->group_name, // Slug title of the setting. 
  289.                           '_' . $key, // Suffix-id for the error message box. 
  290.                           /** translators: 1: Verification string from user input; 2: Service name. */ 
  291.                           sprintf( __( '%1$s does not seem to be a valid %2$s verification string. Please correct.', 'wordpress-seo' ), '<strong>' . esc_html( $meta ) . '</strong>', $service ), // The error message. 
  292.                           'error' // Error type, either 'error' or 'updated'. 
  293.  ); 
  294.                   } 
  295.               } 
  296.           } 
  297.       } 
  298.   }

  523.   /** 
  524.    * Update a site_option 
  525.    * 
  526.    * @internal This special method is only needed for multisite options, but very needed indeed there. 
  527.    * The order in which certain functions and hooks are run is different between get_option() and 
  528.    * get_site_option() which means in practice that the removing of the default filters would be 
  529.    * done too late and the re-adding of the default filters might not be done at all. 
  530.    * Aka: use the WPSEO_Options::update_site_option() method (which calls this method) for 
  531.    * safely adding/updating multisite options. 
  532.    * 
  533.    * @param mixed $value The new value for the option. 
  534.    * 
  535.    * @return bool whether the update was succesfull 
  536.    */ 
  537.   public function update_site_option( $value ) {

  587.   public function import( $option_value, $current_version = null, $all_old_option_values = null ) {

  653.    * Make sure that any set option values relating to post_types and/or taxonomies are retained,  
  654.    * even when that post_type or taxonomy may not yet have been registered. 
  655.    * 
  656.    * @internal The wpseo_titles concrete class overrules this method. Make sure that any changes 
  657.    * applied here, also get ported to that version. 
  658.    * 
  659.    * @param  array $dirty Original option as retrieved from the database. 
  660.    * @param  array $clean Filtered option where any options which shouldn't be in our option 
  661.    *                      have already been removed and any options which weren't set 
  662.    *                      have been set to their defaults. 
  663.    * 
  664.    * @return  array 
  665.    */ 
  666.   protected function retain_variable_keys( $dirty, $clean ) { 
  667.       if ( ( is_array( $this->variable_array_key_patterns ) && $this->variable_array_key_patterns !== array() ) && ( is_array( $dirty ) && $dirty !== array() ) ) { 
  668.           foreach ( $dirty as $key => $value ) { 
  669.  
  670.               // Do nothing if already in filtered options. 
  671.               if ( isset( $clean[ $key ] ) ) { 
  672.                   continue; 
  673.               } 
  674.  
  675.               foreach ( $this->variable_array_key_patterns as $pattern ) { 
  676.  
  677.                   if ( strpos( $key, $pattern ) === 0 ) { 
  678.                       $clean[ $key ] = $value; 
  679.                       break; 
  680.                   } 
  681.               } 
  682.           } 
  683.       } 
  684.  
  685.       return $clean; 
  686.   }

  694.    * @param  string $key Array key to check.

  699.   protected function get_switch_key( $key ) { 
  700.       if ( ! isset( $this->variable_array_key_patterns ) || ( ! is_array( $this->variable_array_key_patterns ) || $this->variable_array_key_patterns === array() ) ) { 
  701.           return $key; 
  702.       } 
  703.  
  704.       foreach ( $this->variable_array_key_patterns as $pattern ) {

  706.               return $pattern; 
  707.           } 
  708.       } 
  709.  
  710.       return $key;
