  265.   /** 
  266.    * Emulate the WP native sanitize_text_field function in a %%variable%% safe way 
  267.    * 
  268.    * @see   https://core.trac.wordpress.org/browser/trunk/src/wp-includes/formatting.php for the original 
  269.    * 
  270.    * Sanitize a string from user input or from the db 
  271.    * 
  272.    * check for invalid UTF-8,  
  273.    * Convert single < characters to entity,  
  274.    * strip all tags,  
  275.    * remove line breaks, tabs and extra white space,  
  276.    * strip octets - BUT DO NOT REMOVE (part of) VARIABLES WHICH WILL BE REPLACED. 
  277.    * 
  278.    * @static 
  279.    * 
  280.    * @since 1.8.0 
  281.    * 
  282.    * @param string $value String value to sanitize. 
  283.    * 
  284.    * @return string 
  285.    */ 
  286.   public static function sanitize_text_field( $value ) { 
  287.       $filtered = wp_check_invalid_utf8( $value ); 
  288.  
  289.       if ( strpos( $filtered, '<' ) !== false ) { 
  290.           $filtered = wp_pre_kses_less_than( $filtered ); 
  291.           // This will strip extra whitespace for us. 
  292.           $filtered = wp_strip_all_tags( $filtered, true ); 
  293.       } 
  294.       else { 
  295.           $filtered = trim( preg_replace( '`[\r\n\t ]+`', ' ', $filtered ) ); 
  296.       } 
  297.  
  298.       $found = false; 
  299.       while ( preg_match( '`[^%](%[a-f0-9]{2})`i', $filtered, $match ) ) { 
  300.           $filtered = str_replace( $match[1], '', $filtered ); 
  301.           $found = true; 
  302.       } 
  303.       unset( $match ); 
  304.  
  305.       if ( $found ) { 
  306.           // Strip out the whitespace that may now exist after removing the octets. 
  307.           $filtered = trim( preg_replace( '` +`', ' ', $filtered ) ); 
  308.       } 
  309.  
  310.       /** 
  311.        * Filter a sanitized text field string. 
  312.        * 
  313.        * @since WP 2.9.0 
  314.        * 
  315.        * @param string $filtered The sanitized string. 
  316.        * @param string $str      The string prior to being sanitized. 
  317.        */ 
  318.  
  319.       return apply_filters( 'sanitize_text_field', $filtered, $value ); 
  320.   }

  536.   /** 
  537.    * Do simple reliable math calculations without the risk of wrong results 
  538.    * 
  539.    * @see   http://floating-point-gui.de/ 
  540.    * @see   the big red warning on http://php.net/language.types.float.php 
  541.    * 
  542.    * In the rare case that the bcmath extension would not be loaded, it will return the normal calculation results 
  543.    * 
  544.    * @static 
  545.    * 
  546.    * @since 1.5.0 
  547.    * @since 1.8.0 Moved from stand-alone function to this class. 
  548.    * 
  549.    * @param mixed  $number1     Scalar (string/int/float/bool). 
  550.    * @param string $action      Calculation action to execute. Valid input: 
  551.    *                            '+' or 'add' or 'addition',  
  552.    *                            '-' or 'sub' or 'subtract',  
  553.    *                            '*' or 'mul' or 'multiply',  
  554.    *                            '/' or 'div' or 'divide',  
  555.    *                            '%' or 'mod' or 'modulus' 
  556.    *                            '=' or 'comp' or 'compare'. 
  557.    * @param mixed  $number2     Scalar (string/int/float/bool). 
  558.    * @param bool   $round       Whether or not to round the result. Defaults to false. 
  559.    *                            Will be disregarded for a compare operation. 
  560.    * @param int    $decimals    Decimals for rounding operation. Defaults to 0. 
  561.    * @param int    $precision   Calculation precision. Defaults to 10. 
  562.    * 
  563.    * @return mixed            Calculation Result or false if either or the numbers isn't scalar or 
  564.    *                          an invalid operation was passed 
  565.    *                          - for compare the result will always be an integer 
  566.    *                          - for all other operations, the result will either be an integer (preferred) 
  567.    *                            or a float 
  568.    */ 
  569.   public static function calc( $number1, $action, $number2, $round = false, $decimals = 0, $precision = 10 ) { 
  570.       static $bc; 
  571.  
  572.       if ( ! is_scalar( $number1 ) || ! is_scalar( $number2 ) ) { 
  573.           return false; 
  574.       } 
  575.  
  576.       if ( ! isset( $bc ) ) { 
  577.           $bc = extension_loaded( 'bcmath' ); 
  578.       } 
  579.  
  580.       if ( $bc ) { 
  581.           $number1 = number_format( $number1, 10, '.', '' ); 
  582.           $number2 = number_format( $number2, 10, '.', '' ); 
  583.       } 
  584.  
  585.       $result = null; 
  586.       $compare = false; 
  587.  
  588.       switch ( $action ) { 
  589.           case '+': 
  590.           case 'add': 
  591.           case 'addition': 
  592.               $result = ( $bc ) ? bcadd( $number1, $number2, $precision ) /** string */ : ( $number1 + $number2 ); 
  593.               break; 
  594.  
  595.           case '-': 
  596.           case 'sub': 
  597.           case 'subtract': 
  598.               $result = ( $bc ) ? bcsub( $number1, $number2, $precision ) /** string */ : ( $number1 - $number2 ); 
  599.               break; 
  600.  
  601.           case '*': 
  602.           case 'mul': 
  603.           case 'multiply': 
  604.               $result = ( $bc ) ? bcmul( $number1, $number2, $precision ) /** string */ : ( $number1 * $number2 ); 
  605.               break; 
  606.  
  607.           case '/': 
  608.           case 'div': 
  609.           case 'divide': 
  610.               if ( $bc ) { 
  611.                   $result = bcdiv( $number1, $number2, $precision ); // String, or NULL if right_operand is 0. 
  612.               } 
  613.               elseif ( $number2 != 0 ) { 
  614.                   $result = ( $number1 / $number2 ); 
  615.               } 
  616.  
  617.               if ( ! isset( $result ) ) { 
  618.                   $result = 0; 
  619.               } 
  620.               break; 
  621.  
  622.           case '%': 
  623.           case 'mod': 
  624.           case 'modulus': 
  625.               if ( $bc ) { 
  626.                   $result = bcmod( $number1, $number2 ); // String, or NULL if modulus is 0. 
  627.               } 
  628.               elseif ( $number2 != 0 ) { 
  629.                   $result = ( $number1 % $number2 ); 
  630.               } 
  631.  
  632.               if ( ! isset( $result ) ) { 
  633.                   $result = 0; 
  634.               } 
  635.               break; 
  636.  
  637.           case '=': 
  638.           case 'comp': 
  639.           case 'compare': 
  640.               $compare = true; 
  641.               if ( $bc ) { 
  642.                   $result = bccomp( $number1, $number2, $precision ); // Returns int 0, 1 or -1. 
  643.               } 
  644.               else { 
  645.                   $result = ( $number1 == $number2 ) ? 0 : ( ( $number1 > $number2 ) ? 1 : - 1 ); 
  646.               } 
  647.               break; 
  648.       } 
  649.  
  650.       if ( isset( $result ) ) { 
  651.           if ( $compare === false ) { 
  652.               if ( $round === true ) { 
  653.                   $result = round( floatval( $result ), $decimals ); 
  654.                   if ( $decimals === 0 ) { 
  655.                       $result = (int) $result; 
  656.                   } 
  657.               } 
  658.               else { 
  659.                   $result = ( intval( $result ) == $result ) ? intval( $result ) : floatval( $result ); 
  660.               } 
  661.           } 
  662.  
  663.           return $result; 
  664.       } 
  665.  
  666.       return false; 
  667.   }
