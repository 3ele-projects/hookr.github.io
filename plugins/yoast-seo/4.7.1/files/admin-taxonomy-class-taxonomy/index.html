  117.   /** 
  118.    * Update the taxonomy meta data on save. 
  119.    * 
  120.    * @param int    $term_id  ID of the term to save data for. 
  121.    * @param int    $tt_id    The taxonomy_term_id for the term. 
  122.    * @param string $taxonomy The taxonomy the term belongs to. 
  123.    */ 
  124.   public function update_term( $term_id, $tt_id, $taxonomy ) { 
  125.       /** Create post array with only our values */ 
  126.       $new_meta_data = array(); 
  127.       foreach ( WPSEO_Taxonomy_Meta::$defaults_per_term as $key => $default ) { 
  128.           if ( $posted_value = filter_input( INPUT_POST, $key ) ) { 
  129.               $new_meta_data[ $key ] = $posted_value; 
  130.           } 
  131.  
  132.           // If analysis is disabled remove that analysis score value from the DB. 
  133.           if ( $this->is_meta_value_disabled( $key ) ) { 
  134.               $new_meta_data[ $key ] = ''; 
  135.           } 
  136.       } 
  137.       unset( $key, $default ); 
  138.  
  139.       // Saving the values. 
  140.       WPSEO_Taxonomy_Meta::set_values( $term_id, $taxonomy, $new_meta_data ); 
  141.   }

  146.    * @param string $key The key of the meta value. 
  147.    * @return bool Whether the given meta value key is disabled. 
  148.    */ 
  149.   public function is_meta_value_disabled( $key ) { 
  150.       if ( 'wpseo_linkdex' === $key && ! $this->analysis_seo->is_enabled() ) { 
  151.           return true; 
  152.       } 
  153.  
  154.       if ( 'wpseo_content_score' === $key && ! $this->analysis_readability->is_enabled() ) { 
  155.           return true; 
  156.       } 
  157.  
  158.       return false; 
  159.   }
