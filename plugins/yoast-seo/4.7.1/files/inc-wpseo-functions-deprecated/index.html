  291. function wpseo_xml_sitemaps_init() { 
  292.   _deprecated_function( __FUNCTION__, 'WPSEO 2.4', 'WPSEO_Sitemaps_Router' ); 
  293.  
  294.   $options = get_option( 'wpseo_xml' ); 
  295.   if ( $options['enablexmlsitemap'] !== true ) { 
  296.       return; 
  297.   } 
  298.  
  299.   // Redirects sitemap.xml to sitemap_index.xml. 
  300.   add_action( 'template_redirect', 'wpseo_xml_redirect_sitemap', 0 ); 
  301.  
  302.   if ( ! is_object( $GLOBALS['wp'] ) ) { 
  303.       return; 
  304.   } 
  305.  
  306.   $GLOBALS['wp']->add_query_var( 'sitemap' ); 
  307.   $GLOBALS['wp']->add_query_var( 'sitemap_n' ); 
  308.   $GLOBALS['wp']->add_query_var( 'xsl' ); 
  309.   add_rewrite_rule( 'sitemap_index\.xml$', 'index.php?sitemap=1', 'top' ); 
  310.   add_rewrite_rule( '([^/]+?)-sitemap([0-9]+)?\.xml$', 'index.php?sitemap=$matches[1]&sitemap_n=$matches[2]', 'top' ); 
  311.   add_rewrite_rule( '([a-z]+)?-?sitemap\.xsl$', 'index.php?xsl=$matches[1]', 'top' );
