  39.   public function add_columns( array $columns ) {

  54.           if ( $column_name === 'description' && $this->analysis_readability->is_enabled() ) {

  62.   /**

  71.   public function parse_column( $content, $column_name, $term_id ) {

  73.       switch ( $column_name ) {

  75.               return $this->get_score_value( $term_id ); 
  76.  
  77.               break;

  80.               return $this->get_score_readability_value( $term_id );

  101.   /** 
  102.    * Parses the value for the score column.

  108.   private function get_score_value( $term_id ) {

  112.       if ( ! $this->is_indexable( $term ) ) { 
  113.           return $this->create_score_icon( 
  114.               new WPSEO_Rank( WPSEO_Rank::NO_INDEX ),  
  115.               __( 'Term is set to noindex.', 'wordpress-seo' ) 
  116.  ); 
  117.       } 
  118.  
  119.       // When there is a focus key word.

  123.  
  124.           return $this->create_score_icon( $rank, $rank->get_label() ); 
  125.       } 
  126.  
  127.       // Default icon. 
  128.       return $this->create_score_icon( 
  129.           new WPSEO_Rank( WPSEO_Rank::NO_FOCUS ),  
  130.           __( 'Focus keyword not set.', 'wordpress-seo' ) 
  131.  ); 
  132.   }

  135.    * Parses the value for the readability score column.

  144.

  160.

  167.    * @param mixed $term The current term.

  169.    * @return bool

  171.   private function is_indexable( $term ) { 
  172.       static $options; 
  173.  
  174.       // Saving the options once, because it's static. 
  175.       if ( $options === null ) { 
  176.           $options = WPSEO_Options::get_all(); 
  177.       } 
  178.  
  179.       // When the no_index value is not empty and not default, check if its value is index. 
  180.       $no_index = WPSEO_Taxonomy_Meta::get_term_meta( $term->term_id, $this->taxonomy, 'noindex' ); 
  181.       if ( ! empty( $no_index ) && $no_index !== 'default' ) { 
  182.           return ( $no_index === 'index' ); 
  183.       } 
  184.  
  185.       // Check if the default for taxonomy is empty (this will be index). 
  186.       $no_index_key = 'noindex-tax-' . $term->taxonomy; 
  187.       if ( is_object( $term ) && ( isset( $options[ $no_index_key ] ) ) ) { 
  188.           return ( empty( $options[ $no_index_key ] ) ); 
  189.       } 
  190.  
  191.       return true; 
  192.   }

  201.   private function get_focus_keyword( $term ) {

  223.   /**

  239.       if ( isset( $taxonomy ) ) {

  243.  
  244.           return ( ( isset( $options[ 'hideeditbox-tax-' . $taxonomy ] ) && $options[ 'hideeditbox-tax-' . $taxonomy ] === true ) || in_array( $taxonomy, $custom_taxonomies ) === false ); 
  245.       } 
  246.  
  247.       return false;
