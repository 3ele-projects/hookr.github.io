  104.   $options = get_option( 'wpseo' ); 
  105.   if ( is_array( $options['blocking_files'] ) && $options['blocking_files'] !== array() ) { 
  106.       foreach ( $options['blocking_files'] as $file ) { 
  107.           if ( is_file( $file ) ) { 
  108.               if ( ! @unlink( $file ) ) { 
  109.                   $errors[] = sprintf( 
  110.                       /** translators: %s expands to the file path and name. */ 
  111.                       __( 'The file %s could not be removed. Please remove it via FTP.', 'wordpress-seo' ),  
  112.                       '<code>' . $file . '</code>' 
  113.  ); 
  114.               } 
  115.           } 
  116.  
  117.           if ( is_dir( $file ) ) { 
  118.               if ( ! @ rmdir( $file ) ) { 
  119.                   $errors[] = sprintf( 
  120.                       /** translators: %s expands to the directory path and name. */ 
  121.                       __( 'The directory %s could not be removed. Please remove it via FTP.', 'wordpress-seo' ),  
  122.                       '<code>' . $file . '</code>' 
  123.  ); 
  124.               } 
  125.           } 
  126.       } 
  127.   } 
  128.  
  129.   if ( $errors ) { 
  130.       $message = implode( '<br />', $errors ); 
  131.       wp_send_json_error( array( 'message' => $message ) ); 
  132.   } 
  133.   else { 
  134.       $message = __( 'Files successfully removed.', 'wordpress-seo' ); 
  135.       wp_send_json_success( array( 'message' => $message ) ); 
  136.   } 
  137. }

  194. /** 
  195.  * Helper function to update a post's meta data, returning relevant information 
  196.  * about the information updated and the results or the meta update. 
  197.  * 
  198.  * @param int    $post_id         Post ID. 
  199.  * @param string $new_meta_value  New meta value to record. 
  200.  * @param string $orig_meta_value Original meta value. 
  201.  * @param string $meta_key        Meta key string. 
  202.  * @param string $return_key      Return key string to use in results. 
  203.  * 
  204.  * @return string 
  205.  */ 
  206. function wpseo_upsert_meta( $post_id, $new_meta_value, $orig_meta_value, $meta_key, $return_key ) { 
  207.  
  208.   $post_id = intval( $post_id ); 
  209.   $sanitized_new_meta_value = wp_strip_all_tags( $new_meta_value ); 
  210.   $orig_meta_value = wp_strip_all_tags( $orig_meta_value ); 
  211.  
  212.   $upsert_results = array( 
  213.       'status' => 'success',  
  214.       'post_id' => $post_id,  
  215.       "new_{$return_key}" => $sanitized_new_meta_value,  
  216.       "original_{$return_key}" => $orig_meta_value,  
  217.  ); 
  218.  
  219.   $the_post = get_post( $post_id ); 
  220.   if ( empty( $the_post ) ) { 
  221.  
  222.       $upsert_results['status'] = 'failure'; 
  223.       $upsert_results['results'] = __( 'Post doesn\'t exist.', 'wordpress-seo' ); 
  224.  
  225.       return $upsert_results; 
  226.   } 
  227.  
  228.   $post_type_object = get_post_type_object( $the_post->post_type ); 
  229.   if ( ! $post_type_object ) { 
  230.  
  231.       $upsert_results['status'] = 'failure'; 
  232.       $upsert_results['results'] = sprintf( __( 'Post has an invalid Post Type: %s.', 'wordpress-seo' ), $the_post->post_type ); 
  233.  
  234.       return $upsert_results; 
  235.   } 
  236.  
  237.   if ( ! current_user_can( $post_type_object->cap->edit_posts ) ) { 
  238.  
  239.       $upsert_results['status'] = 'failure'; 
  240.       $upsert_results['results'] = sprintf( __( 'You can\'t edit %s.', 'wordpress-seo' ), $post_type_object->label ); 
  241.  
  242.       return $upsert_results; 
  243.   } 
  244.  
  245.   if ( ! current_user_can( $post_type_object->cap->edit_others_posts ) && $the_post->post_author != get_current_user_id() ) { 
  246.  
  247.       $upsert_results['status'] = 'failure'; 
  248.       $upsert_results['results'] = sprintf( __( 'You can\'t edit %s that aren\'t yours.', 'wordpress-seo' ), $post_type_object->label ); 
  249.  
  250.       return $upsert_results; 
  251.  
  252.   } 
  253.  
  254.   if ( $sanitized_new_meta_value === $orig_meta_value && $sanitized_new_meta_value !== $new_meta_value ) { 
  255.       $upsert_results['status'] = 'failure'; 
  256.       $upsert_results['results'] = __( 'You have used HTML in your value which is not allowed.', 'wordpress-seo' ); 
  257.  
  258.       return $upsert_results; 
  259.   } 
  260.  
  261.   $res = update_post_meta( $post_id, $meta_key, $sanitized_new_meta_value ); 
  262.  
  263.   $upsert_results['status'] = ( $res !== false ) ? 'success' : 'failure'; 
  264.   $upsert_results['results'] = $res; 
  265.  
  266.   return $upsert_results; 
  267. }

  391. /** 
  392.  * Removes stopword from the sample permalink that is generated in an AJAX request 
  393.  * 
  394.  * @param array  $permalink The permalink generated for this post by WordPress. 
  395.  * @param int    $post_id   The ID of the post. 
  396.  * @param string $title     The title for the post that the user used. 
  397.  * @param string $name      The name for the post that the user used. 
  398.  * 
  399.  * @return array 
  400.  */ 
  401. function wpseo_remove_stopwords_sample_permalink( $permalink, $post_id, $title, $name ) { 
  402.   WPSEO_Options::get_instance(); 
  403.   $options = WPSEO_Options::get_options( array( 'wpseo_permalinks' ) ); 
  404.   if ( $options['cleanslugs'] !== true ) { 
  405.       return $permalink; 
  406.   } 
  407.  
  408.   /** 
  409.    * If the name is empty and the title is not, WordPress will generate a slug. In that case we want to remove stop

  412.   if ( empty( $name ) && ! empty( $title ) ) {

  415.       // The second element is the slug. 
  416.       $permalink[1] = $stop_words->remove_in( $permalink[1] ); 
  417.   } 
  418.  
  419.   return $permalink; 
  420. }
