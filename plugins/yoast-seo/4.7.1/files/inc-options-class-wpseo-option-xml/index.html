  152.

  154.  
  155.                               $error_message = sprintf( 
  156.                                   /** translators: 1: number; 2: number received from user. */ 
  157.                                   __( '"Max entries per sitemap page" should be below %1$s to meet Google\'s requirements, which %2$s is not.', 'wordpress-seo' ),  
  158.                                   number_format_i18n( 50000 ), '<strong>' . esc_html( sanitize_text_field( $dirty[ $key ] ) ) . '</strong>' 
  159.  ); 
  160.                               add_settings_error( $this->group_name, '_' . $key, $error_message, 'error' ); 
  161.  
  162.                               $int = 50000; 
  163.                           } 
  164.  
  165.                           $clean[ $key ] = $int; 
  166.                       } 
  167.                       else { 
  168.  
  169.                           if ( isset( $old[ $key ] ) && $old[ $key ] !== '' ) { 
  170.  
  171.                               $int = WPSEO_Utils::validate_int( $old[ $key ] ); 
  172.  
  173.                               if ( $int !== false && $int > 0 ) { 
  174.                                   $clean[ $key ] = $int; 
  175.                               } 
  176.                           }

  179.                               __( '"Max entries per sitemap page" should be a positive number, which %s is not. Please correct.', 'wordpress-seo' ),  
  180.                               '<strong>' . esc_html( sanitize_text_field( $dirty[ $key ] ) ) . '</strong>' 
  181.  ); 
  182.                           add_settings_error( $this->group_name, '_' . $key, $error_message, 'error' );

  197.                   break;

  203.               /**
