  51.    */ 
  52.   public function get_images( $post ) { 
  53.  
  54.       $images = array(); 
  55.  
  56.       if ( ! is_object( $post ) ) { 
  57.           return $images; 
  58.       } 
  59.  
  60.       $thumbnail_id = get_post_thumbnail_id( $post->ID ); 
  61.  
  62.       if ( $thumbnail_id ) { 
  63.  
  64.           $src = $this->get_absolute_url( $this->image_url( $thumbnail_id ) ); 
  65.           $alt = get_post_meta( $thumbnail_id, '_wp_attachment_image_alt', true ); 
  66.           $title = get_post_field( 'post_title', $thumbnail_id ); 
  67.           $images[] = $this->get_image_item( $post, $src, $title, $alt ); 
  68.       } 
  69.  
  70.       $unfiltered_images = $this->parse_html_images( $post->post_content ); 
  71.  
  72.       foreach ( $unfiltered_images as $image ) { 
  73.           $images[] = $this->get_image_item( $post, $image['src'], $image['title'], $image['alt'] ); 
  74.       } 
  75.  
  76.       foreach ( $this->parse_galleries( $post->post_content, $post->ID ) as $attachment ) { 
  77.  
  78.           $src = $this->get_absolute_url( $this->image_url( $attachment->ID ) ); 
  79.           $alt = get_post_meta( $attachment->ID, '_wp_attachment_image_alt', true ); 
  80.  
  81.           $images[] = $this->get_image_item( $post, $src, $attachment->post_title, $alt ); 
  82.       } 
  83.  
  84.       if ( 'attachment' === $post->post_type && wp_attachment_is_image( $post ) ) { 
  85.  
  86.           $src = $this->get_absolute_url( $this->image_url( $post->ID ) ); 
  87.           $alt = get_post_meta( $post->ID, '_wp_attachment_image_alt', true ); 
  88.  
  89.           $images[] = $this->get_image_item( $post, $src, $post->post_title, $alt ); 
  90.       } 
  91.  
  92.       foreach ( $images as $key => $image ) { 
  93.  
  94.           if ( empty( $image['src'] ) ) { 
  95.               unset( $images[ $key ] ); 
  96.           } 
  97.       } 
  98.  
  99.       /** 
  100.        * Filter images to be included for the post in XML sitemap. 
  101.        * 
  102.        * @param array $images  Array of image items. 
  103.        * @param int   $post_id ID of the post. 
  104.        */ 
  105.       $images = apply_filters( 'wpseo_sitemap_urlimages', $images, $post->ID ); 
  106.  
  107.       return $images; 
  108.   }

  131.   /** 
  132.    * Parse `<img />` tags in content. 
  133.    * 
  134.    * @param string $content Content string to parse. 
  135.    * 
  136.    * @return array 
  137.    */ 
  138.   private function parse_html_images( $content ) { 
  139.  
  140.       $images = array(); 
  141.  
  142.       if ( ! class_exists( 'DOMDocument' ) ) { 
  143.           return $images; 
  144.       } 
  145.  
  146.       if ( empty( $content ) ) { 
  147.           return $images; 
  148.       } 
  149.  
  150.       // Prevent DOMDocument from bubbling warnings about invalid HTML. 
  151.       libxml_use_internal_errors( true ); 
  152.  
  153.       $post_dom = new DOMDocument(); 
  154.       $post_dom->loadHTML( '<?xml encoding="' . $this->charset . '">' . $content ); 
  155.  
  156.       // Clear the errors, so they don't get kept in memory. 
  157.       libxml_clear_errors(); 
  158.  
  159.       /** @var DOMElement $img */ 
  160.       foreach ( $post_dom->getElementsByTagName( 'img' ) as $img ) { 
  161.  
  162.           $src = $img->getAttribute( 'src' ); 
  163.  
  164.           if ( empty( $src ) ) { 
  165.               continue; 
  166.           } 
  167.  
  168.           $class = $img->getAttribute( 'class' ); 
  169.  
  170.           if ( // This detects WP-inserted images, which we need to upsize. R. 
  171.               ! empty( $class ) 
  172.               && false === strpos( $class, 'size-full' ) 
  173.               && preg_match( '|wp-image-(?P<id>\d+)|', $class, $matches ) 
  174.               && get_post_status( $matches['id'] ) 
  175.  ) { 
  176.               $src = $this->image_url( $matches['id'] ); 
  177.           } 
  178.  
  179.           $src = $this->get_absolute_url( $src ); 
  180.  
  181.           if ( strpos( $src, $this->host ) === false ) { 
  182.               continue; 
  183.           } 
  184.  
  185.           if ( $src !== esc_url( $src ) ) { 
  186.               continue; 
  187.           } 
  188.  
  189.           $images[] = array( 
  190.               'src' => $src,  
  191.               'title' => $img->getAttribute( 'title' ),  
  192.               'alt' => $img->getAttribute( 'alt' ),  
  193.  ); 
  194.       } 
  195.  
  196.       return $images; 
  197.   }

  199.   /** 
  200.    * Parse gallery shortcodes in a given content. 
  201.    * 
  202.    * @param string $content Content string. 
  203.    * @param int    $post_id Optional ID of post being parsed. 
  204.    * 
  205.    * @return array Set of attachment objects. 
  206.    */ 
  207.   private function parse_galleries( $content, $post_id = 0 ) { 
  208.  
  209.       $attachments = array(); 
  210.       $galleries = $this->get_content_galleries( $content ); 
  211.  
  212.       foreach ( $galleries as $gallery ) { 
  213.  
  214.           $id = $post_id; 
  215.  
  216.           if ( ! empty( $gallery['id'] ) ) { 
  217.               $id = intval( $gallery['id'] ); 
  218.           } 
  219.  
  220.           // Forked from core gallery_shortcode() to have exact same logic. R. 
  221.           if ( ! empty( $gallery['ids'] ) ) { 
  222.               $gallery['include'] = $gallery['ids']; 
  223.           } 
  224.  
  225.           $gallery_attachments = array(); 
  226.  
  227.           if ( ! empty( $gallery['include'] ) ) { 
  228.  
  229.               $_attachments = get_posts( array( 
  230.                   'include' => $gallery['include'],  
  231.                   'post_status' => 'inherit',  
  232.                   'post_type' => 'attachment',  
  233.                   'post_mime_type' => 'image',  
  234.  ) ); 
  235.  
  236.               foreach ( $_attachments as $key => $val ) { 
  237.                   $gallery_attachments[ $val->ID ] = $_attachments[ $key ]; 
  238.               } 
  239.           } 
  240.           elseif ( ! empty( $gallery['exclude'] ) && ! empty( $id ) ) { 
  241.  
  242.               $gallery_attachments = get_children( array( 
  243.                   'post_parent' => $id,  
  244.                   'exclude' => $gallery['exclude'],  
  245.                   'post_status' => 'inherit',  
  246.                   'post_type' => 'attachment',  
  247.                   'post_mime_type' => 'image',  
  248.  ) ); 
  249.           } 
  250.           elseif ( ! empty( $id ) ) { 
  251.  
  252.               $gallery_attachments = get_children( array( 
  253.                   'post_parent' => $id,  
  254.                   'post_status' => 'inherit',  
  255.                   'post_type' => 'attachment',  
  256.                   'post_mime_type' => 'image',  
  257.  ) ); 
  258.           } 
  259.  
  260.           $attachments = array_merge( $attachments, $gallery_attachments ); 
  261.       } 
  262.  
  263.       return array_unique( $attachments, SORT_REGULAR ); 
  264.   }

  266.   /** 
  267.    * Retrieves galleries from the passed content. 
  268.    * 
  269.    * Forked from core to skip executing shortcodes for performance. 
  270.    * 
  271.    * @param string $content Content to parse for shortcodes. 
  272.    * 
  273.    * @return array A list of arrays, each containing gallery data. 
  274.    */ 
  275.   protected function get_content_galleries( $content ) { 
  276.  
  277.       if ( ! has_shortcode( $content, 'gallery' ) ) { 
  278.           return array(); 
  279.       } 
  280.  
  281.       $galleries = array(); 
  282.  
  283.       if ( ! preg_match_all( '/' . get_shortcode_regex() . '/s', $content, $matches, PREG_SET_ORDER ) ) { 
  284.           return $galleries; 
  285.       } 
  286.  
  287.       foreach ( $matches as $shortcode ) { 
  288.           if ( 'gallery' === $shortcode[2] ) { 
  289.  
  290.               $attributes = shortcode_parse_atts( $shortcode[3] ); 
  291.  
  292.               if ( '' === $attributes ) { // Valid shortcode without any attributes. R. 
  293.                   $attributes = array(); 
  294.               } 
  295.  
  296.               $galleries[] = $attributes; 
  297.           } 
  298.       } 
  299.  
  300.       return $galleries; 
  301.   }

  349.   /** 
  350.    * Get attached image URL. Adapted from core for speed. 
  351.    * 
  352.    * @param int $post_id ID of the post. 
  353.    * 
  354.    * @return string 
  355.    */ 
  356.   private function image_url( $post_id ) { 
  357.  
  358.       static $uploads; 
  359.  
  360.       if ( empty( $uploads ) ) { 
  361.           $uploads = wp_upload_dir(); 
  362.       } 
  363.  
  364.       if ( false !== $uploads['error'] ) { 
  365.           return ''; 
  366.       } 
  367.  
  368.       $file = get_post_meta( $post_id, '_wp_attached_file', true ); 
  369.  
  370.       if ( empty( $file ) ) { 
  371.           return ''; 
  372.       } 
  373.  
  374.       // Check that the upload base exists in the file location. 
  375.       if ( 0 === strpos( $file, $uploads['basedir'] ) ) { 
  376.           return str_replace( $uploads['basedir'], $uploads['baseurl'], $file ); 
  377.       } 
  378.  
  379.       // Replace file location with url location. 
  380.       if ( false !== strpos( $file, 'wp-content/uploads' ) ) { 
  381.           return $uploads['baseurl'] . substr( $file, ( strpos( $file, 'wp-content/uploads' ) + 18 ) ); 
  382.       } 
  383.  
  384.       // It's a newly uploaded file, therefore $file is relative to the baseurl. 
  385.       return $uploads['baseurl'] . "/$file"; 
  386.   }

  388.   /** 
  389.    * Make absolute URL for domain or protocol-relative one. 
  390.    * 
  391.    * @param string $src URL to process. 
  392.    * 
  393.    * @return string 
  394.    */ 
  395.   protected function get_absolute_url( $src ) { 
  396.  
  397.       if ( empty( $src ) || ! is_string( $src ) ) { 
  398.           return $src; 
  399.       } 
  400.  
  401.       if ( WPSEO_Utils::is_url_relative( $src ) === true ) { 
  402.  
  403.           if ( $src[0] !== '/' ) { 
  404.               return $src; 
  405.           } 
  406.  
  407.           // The URL is relative, we'll have to make it absolute. 
  408.           return $this->home_url . $src; 
  409.       } 
  410.  
  411.       if ( strpos( $src, 'http' ) !== 0 ) { 
  412.           // Protocol relative url, we add the scheme as the standard requires a protocol. 
  413.           return $this->scheme . ':' . $src; 
  414.       } 
  415.  
  416.       return $src; 
  417.   }
