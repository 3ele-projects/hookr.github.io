  16.    *

  18.    */

  20.  
  21.   /** 
  22.    * All the possible filters 
  23.    * 
  24.    * @var array

  31.    * @var string 
  32.    */ 
  33.   private $category; 
  34.  
  35.   /** 
  36.    * Constructing this object 
  37.    * 
  38.    * Setting the hook to create the issues categories as the links

  40.    * @param array $platform_counts Set of issue counts by platform.

  42.   public function __construct( array $platform_counts ) {

  44.           $this->set_counts( $platform_counts ); 
  45.       } 
  46.  
  47.       // Setting the filter values.

  49.

  53.   /** 
  54.    * Returns the value of the current category

  58.   public function get_category() {

  61.  
  62.   /**

  68.    */ 
  69.   public function as_array() { 
  70.       $new_views = array(); 
  71.  
  72.       foreach ( $this->category_counts as $category_name => $category ) { 
  73.           $new_views[] = $this->create_view_link( $category_name, $category['count'] ); 
  74.       } 
  75.  
  76.       return $new_views; 
  77.   }

  79.   /** 
  80.    * Getting the current view 
  81.    */ 
  82.   private function get_current_category() { 
  83.       if ( $current_category = filter_input( INPUT_GET, 'category' ) ) { 
  84.           return $current_category; 
  85.       } 
  86.  
  87.       // Just prevent redirect loops. 
  88.       if ( ! empty( $this->category_counts ) ) { 
  89.           $current_category = 'not_found'; 
  90.           if ( empty( $this->category_counts[ $current_category ] ) ) { 
  91.               $current_category = key( $this->category_counts ); 
  92.           } 
  93.  
  94.           // Just redirect to set the category. 
  95.           wp_redirect( add_query_arg( 'category', $current_category ) ); 
  96.           exit; 
  97.       } 
  98.   } 
  99.  
  100.   /** 
  101.    * Setting the view counts based on the saved data. The info will be used to display the category filters

  103.    * @param array $platform_counts Set of counts by platform.

  105.   private function set_counts( array $platform_counts ) {

  107.   }

  114.       $this->set_filter_value( 'faulty_redirects', __( 'Faulty redirects', 'wordpress-seo' ) ); 
  115.       $this->set_filter_value( 'not_followed', __( 'Not followed', 'wordpress-seo' ) ); 
  116.       $this->set_filter_value( 'not_found', __( 'Not found', 'wordpress-seo' ), __( 'URL points to a non-existent page.', 'wordpress-seo' ), __( 'Show information about errors in category \'Not Found\'', 'wordpress-seo' ) ); 
  117.       $this->set_filter_value( 'other', __( 'Other', 'wordpress-seo' ), __( 'Google was unable to crawl this URL due to an undetermined issue.', 'wordpress-seo' ), __( 'Show information about errors in category \'Other\'', 'wordpress-seo' ) ); 
  118.       /** Translators: %1$s: expands to '<code>robots.txt</code>'. */ 
  119.       $this->set_filter_value( 'roboted', __( 'Blocked', 'wordpress-seo' ), sprintf( __( 'Googlebot could access your site, but certain URLs are blocked for Googlebot in your %1$s file.   120.       $this->set_filter_value( 'server_error', __( 'Server Error', 'wordpress-seo' ), __( 'Request timed out or site is blocking Google.', 'wordpress-seo' ), __( 'Show information about errors in category \'Server\'', 'wordpress-seo' ) ); 
  121.       $this->set_filter_value( 'soft_404', __( 'Soft 404', 'wordpress-seo' ), __( "The target URL doesn't exist, but your server is not returning a 404 (file not found) error.", 'wordpress-seo' ), __( 'Show information about errors in category \'Soft 404\'', 'wordpress-seo' ) ); 
  122.   }

  124.   /** 
  125.    * Add new filter value to the filter_values 
  126.    * 
  127.    * @param string $key              Filter key. 
  128.    * @param string $value            Filter value. 
  129.    * @param string $description      Optional description string. 
  130.    * @param string $help_button_text Optional help button text.

  132.   private function set_filter_value( $key, $value, $description = '', $help_button_text = '' ) { 
  133.       $this->filter_values[ $key ] = array( 
  134.           'value' => $value,  
  135.           'description' => $description,  
  136.           'help-button' => $help_button_text,  
  137.  ); 
  138.   } 
  139.  
  140.   /** 
  141.    * Creates a filter link 
  142.    * 
  143.    * @param string  $category Issue type. 
  144.    * @param integer $count    Count for the type. 
  145.    * 
  146.    * @return string 
  147.    */ 
  148.   private function create_view_link( $category, $count ) { 
  149.       $href = add_query_arg( array( 'category' => $category, 'paged' => 1 ) ); 
  150.  
  151.       $class = 'gsc_category'; 
  152.  
  153.       if ( $this->category === $category ) { 
  154.           $class .= ' current'; 
  155.       } 
  156.  
  157.       $help_button = $help_panel = ''; 
  158.       if ( $this->filter_values[ $category ]['description'] !== '' ) { 
  159.           $help = new WPSEO_Admin_Help_Panel( $category, $this->filter_values[ $category ]['help-button'], $this->filter_values[ $category ]['description'], 'has-wrapper' ); 
  160.           $help_button = $help->get_button_html(); 
  161.           $help_panel = $help->get_panel_html(); 
  162.       } 
  163.  
  164.       return sprintf( 
  165.           '<a href="%1$s" class="%2$s">%3$s</a> (<span id="gsc_count_%4$s">%5$s</span>) %6$s %7$s',  
  166.           esc_attr( $href ),  
  167.           $class,  
  168.           $this->filter_values[ $category ]['value'],  
  169.           $category,  
  170.           $count,  
  171.           $help_button,  
  172.           $help_panel 
  173.  ); 
  174.   }

  176.   /** 
  177.    * Parsing the category counts. When there are 0 issues for a specific category, just remove that one from the array 
  178.    * 
  179.    * @param array $category_counts Set of counts for categories. 
  180.    * 
  181.    * @return mixed 
  182.    */ 
  183.   private function parse_counts( $category_counts ) { 
  184.       foreach ( $category_counts as $category_name => $category ) { 
  185.           if ( $category['count'] === '0' ) { 
  186.               unset( $category_counts[ $category_name ] ); 
  187.           } 
  188.       } 
  189.  
  190.       return $category_counts; 
  191.   }
