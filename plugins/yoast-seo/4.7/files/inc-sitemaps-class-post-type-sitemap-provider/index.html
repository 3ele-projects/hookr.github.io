  120.       global $wpdb;

  130.

  132.               continue; 
  133.           } 
  134.  
  135.           $max_pages = 1; 
  136.  
  137.           if ( $total_count > $max_entries ) { 
  138.               $max_pages = (int) ceil( $total_count / $max_entries ); 
  139.           } 
  140.  
  141.           $all_dates = array(); 
  142.  
  143.           if ( $max_pages > 1 ) {

  145.               $sql = " 
  146.               SELECT post_modified_gmt 
  147.                   FROM ( SELECT @rownum:=0 ) init

  149.                   WHERE post_status IN ( 'publish', 'inherit' ) 
  150.                     AND post_type = %s 
  151.                     AND ( @rownum:=@rownum+1 ) %% %d = 0 
  152.                   ORDER BY post_modified_gmt ASC 
  153.               "; 
  154.  
  155.               $all_dates = $wpdb->get_col( $wpdb->prepare( $sql, $post_type, $max_entries ) ); 
  156.           } 
  157.

  159.  
  160.               $current_page = ( $max_pages > 1 ) ? ( $page_counter + 1 ) : '';

  164.

  167.                   } 
  168.               }

  171.               }

  174.                   'loc' => WPSEO_Sitemaps_Router::get_base_url( $post_type . '-sitemap' . $current_page . '.xml' ),

  183.   /** 
  184.    * Get set of sitemap link data. 
  185.    * 
  186.    * @param string $type         Sitemap type. 
  187.    * @param int    $max_entries  Entries per sitemap. 
  188.    * @param int    $current_page Current page of the sitemap. 
  189.    * 
  190.    * @return array 
  191.    */ 
  192.   public function get_sitemap_links( $type, $max_entries, $current_page ) { 
  193.  
  194.       $links = array(); 
  195.       $post_type = $type; 
  196.  
  197.       if ( ! $this->is_valid_post_type( $post_type ) ) { 
  198.           return $links; 
  199.       } 
  200.  
  201.       $steps = min( 100, $max_entries ); 
  202.       $offset = ( $current_page > 1 ) ? ( ( $current_page - 1 ) * $max_entries ) : 0; 
  203.       $total = ( $offset + $max_entries ); 
  204.  
  205.       $typecount = $this->get_post_type_count( $post_type ); 
  206.  
  207.       if ( $total > $typecount ) { 
  208.           $total = $typecount; 
  209.       } 
  210.  
  211.       if ( $current_page === 1 ) { 
  212.           $links = array_merge( $links, $this->get_first_links( $post_type ) ); 
  213.       } 
  214.  
  215.       if ( $typecount === 0 ) { 
  216.  
  217.           return $links; 
  218.       } 
  219.  
  220.       $options = $this->get_options(); 
  221.  
  222.       $stacked_urls = array(); 
  223.  
  224.       while ( $total > $offset ) { 
  225.  
  226.           $posts = $this->get_posts( $post_type, $steps, $offset ); 
  227.  
  228.           $offset += $steps; 
  229.  
  230.           if ( empty( $posts ) ) { 
  231.               continue; 
  232.           } 
  233.  
  234.           $posts_to_exclude = explode( ', ', $options['excluded-posts'] ); 
  235.  
  236.           foreach ( $posts as $post ) { 
  237.  
  238.               if ( WPSEO_Meta::get_value( 'meta-robots-noindex', $post->ID ) === '1' ) { 
  239.                   continue; 
  240.               } 
  241.  
  242.               if ( in_array( $post->ID, $posts_to_exclude ) ) { 
  243.                   continue; 
  244.               } 
  245.  
  246.               $url = $this->get_url( $post ); 
  247.  
  248.               if ( ! isset( $url['loc'] ) ) { 
  249.                   continue; 
  250.               } 
  251.  
  252.               /** 
  253.                * Filter URL entry before it gets added to the sitemap. 
  254.                * 
  255.                * @param array  $url  Array of URL parts. 
  256.                * @param string $type URL type. 
  257.                * @param object $user Data object for the URL. 
  258.                */ 
  259.               $url = apply_filters( 'wpseo_sitemap_entry', $url, 'post', $post ); 
  260.  
  261.               if ( empty( $url ) ) { 
  262.                   continue; 
  263.               } 
  264.  
  265.               $stacked_urls[] = $url['loc']; 
  266.  
  267.               if ( (int) $post->ID === $this->get_page_for_posts_id() || (int) $post->ID === $this->get_page_on_front_id() ) { 
  268.  
  269.                   array_unshift( $links, $url ); 
  270.                   continue; 
  271.               } 
  272.               $links[] = $url; 
  273.           } 
  274.  
  275.           unset( $post, $url ); 
  276.       } 
  277.  
  278.       return $links; 
  279.   }

  445.    *

  448.   protected function get_posts( $post_type, $count, $offset ) { 
  449.  
  450.       global $wpdb; 
  451.  
  452.       static $filters = array(); 
  453.  
  454.       if ( ! isset( $filters[ $post_type ] ) ) { 
  455.           // Make sure you're wpdb->preparing everything you throw into this!! 
  456.           $filters[ $post_type ] = array( 
  457.               /** 
  458.                * Filter JOIN query part for the post type. 
  459.                * 
  460.                * @param string $join      SQL part, defaults to false. 
  461.                * @param string $post_type Post type name. 
  462.                */ 
  463.               'join' => apply_filters( 'wpseo_posts_join', false, $post_type ),  
  464.  
  465.               /** 
  466.                * Filter Where query part for the post type. 
  467.                * 
  468.                * @param string $where     SQL part, defaults to false. 
  469.                * @param string $post_type Post type name. 
  470.                */ 
  471.               'where' => apply_filters( 'wpseo_posts_where', false, $post_type ),  
  472.  ); 
  473.       } 
  474.  
  475.       $join_filter = $filters[ $post_type ]['join']; 
  476.       $where_filter = $filters[ $post_type ]['where']; 
  477.       $where = $this->get_sql_where_clause( $post_type ); 
  478.  
  479.       // Optimized query per this thread: http://wordpress.org/support/topic/plugin-wordpress-seo-by-yoast-performance-suggestion. 
  480.       // Also see http://explainextended.com/2009/10/23/mysql-order-by-limit-performance-late-row-lookups/. 
  481.       $sql = " 
  482.           SELECT l.ID, post_title, post_content, post_name, post_parent, post_author, post_modified_gmt, post_date, post_date_gmt 
  483.           FROM ( 
  484.               SELECT {$wpdb->posts}.ID 
  485.               FROM {$wpdb->posts} 
  486.               {$join_filter} 
  487.               {$where} 
  488.                   {$where_filter} 
  489.               ORDER BY {$wpdb->posts}.post_modified ASC LIMIT %d OFFSET %d 
  490.  ) 
  491.           o JOIN {$wpdb->posts} l ON l.ID = o.ID 
  492.       "; 
  493.

  495.  
  496.       $post_ids = array(); 
  497.  
  498.       foreach ( $posts as $post ) { 
  499.           $post->post_type = $post_type; 
  500.           $post->post_status = 'publish'; 
  501.           $post->filter = 'sample'; 
  502.           $post_ids[] = $post->ID; 
  503.       } 
  504.  
  505.       update_meta_cache( 'post', $post_ids ); 
  506.  
  507.       return $posts; 
  508.   }

  519.       $join = ''; 
  520.       $status = "{$wpdb->posts}.post_status = 'publish'"; 
  521.

  525.           $status = "p2.post_status = 'publish'"; 
  526.       }

  529.       {$join} 
  530.       WHERE {$status} 
  531.           AND {$wpdb->posts}.post_type = '%s'

  539.   /** 
  540.    * Produce array of URL parts for given post object. 
  541.    * 
  542.    * @param object $post Post object to get URL parts for. 
  543.    * 
  544.    * @return array|bool 
  545.    */ 
  546.   protected function get_url( $post ) { 
  547.  
  548.       $url = array(); 
  549.  
  550.       /** 
  551.        * Filter the URL Yoast SEO uses in the XML sitemap. 
  552.        * 
  553.        * Note that only absolute local URLs are allowed as the check after this removes external URLs. 
  554.        * 
  555.        * @param string $url  URL to use in the XML sitemap 
  556.        * @param object $post Post object for the URL. 
  557.        */ 
  558.       $url['loc'] = apply_filters( 'wpseo_xml_sitemap_post_url', get_permalink( $post ), $post ); 
  559.  
  560.       /** 
  561.        * Do not include external URLs. 
  562.        * 
  563.        * @see https://wordpress.org/plugins/page-links-to/ can rewrite permalinks to external URLs. 
  564.        */ 
  565.       if ( false === strpos( $url['loc'], $this->get_home_url() ) ) { 
  566.           return false; 
  567.       } 
  568.  
  569.       $modified = max( $post->post_modified_gmt, $post->post_date_gmt ); 
  570.  
  571.       if ( $modified !== '0000-00-00 00:00:00' ) { 
  572.           $url['mod'] = $modified; 
  573.       } 
  574.  
  575.       $url['chf'] = 'daily'; // Deprecated, kept for backwards data compat. R. 
  576.  
  577.       $canonical = WPSEO_Meta::get_value( 'canonical', $post->ID ); 
  578.  
  579.       if ( $canonical !== '' && $canonical !== $url['loc'] ) { 
  580.           /** 
  581.           Let's assume that if a canonical is set for this page and it's different from 
  582.              the URL of this post, that page is either already in the XML sitemap OR is on 
  583.              an external site, either way, we shouldn't include it here. 
  584.           */ 
  585.           return false; 
  586.       } 
  587.       unset( $canonical ); 
  588.  
  589.       $options = $this->get_options(); 
  590.       if ( $options['trailingslash'] === true && $post->post_type !== 'post' ) { 
  591.           $url['loc'] = trailingslashit( $url['loc'] ); 
  592.       } 
  593.  
  594.       $url['pri'] = 1; // Deprecated, kept for backwards data compat. R. 
  595.       $url['images'] = $this->get_image_parser()->get_images( $post ); 
  596.  
  597.       return $url; 
  598.   }

  600.   /** 
  601.    * Calculate the priority of the post. 
  602.    * 
  603.    * @deprecated 3.5 Priority data dropped from sitemaps. 
  604.    * 
  605.    * @param WP_Post $post Post object. 
  606.    *
