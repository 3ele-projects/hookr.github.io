  83.   /** 
  84.    * Register new replacement %%variables%% 
  85.    * For use by other plugins/themes to register extra variables 
  86.    * 
  87.    * @see wpseo_register_var_replacement() for a usage example 
  88.    * 
  89.    * @param  string $var              The name of the variable to replace, i.e. '%%var%%' 
  90.    *                                  - the surrounding %% are optional. 
  91.    * @param  mixed  $replace_function Function or method to call to retrieve the replacement value for the variable 
  92.    *                                  Uses the same format as add_filter/add_action function parameter and 
  93.    *                                  should *return* the replacement value. DON'T echo it. 
  94.    * @param  string $type             Type of variable: 'basic' or 'advanced', defaults to 'advanced'. 
  95.    * @param  string $help_text        Help text to be added to the help tab for this variable. 
  96.    * 
  97.    * @return bool     Whether the replacement function was succesfully registered 
  98.    */ 
  99.   public static function register_replacement( $var, $replace_function, $type = 'advanced', $help_text = '' ) { 
  100.       $success = false; 
  101.  
  102.       if ( is_string( $var ) && $var !== '' ) { 
  103.           $var = self::remove_var_delimiter( $var ); 
  104.  
  105.           if ( preg_match( '`^[A-Z0-9_-]+$`i', $var ) === false ) { 
  106.               trigger_error( __( 'A replacement variable can only contain alphanumeric characters, an underscore or a dash. Try renaming your variable.', 'wordpress-seo' ), E_USER_WARNING ); 
  107.           } 
  108.           elseif ( strpos( $var, 'cf_' ) === 0 || strpos( $var, 'ct_' ) === 0 ) { 
  109.               trigger_error( __( 'A replacement variable can not start with "%%cf_" or "%%ct_" as these are reserved for the WPSEO standard variable variables for custom fields and custom taxonomies. Try making your variable name unique.', 'wordpress-seo' ), E_USER_WARNING ); 
  110.           } 
  111.           elseif ( ! method_exists( __CLASS__, 'retrieve_' . $var ) ) { 
  112.               if ( ! isset( self::$external_replacements[ $var ] ) ) { 
  113.                   self::$external_replacements[ $var ] = $replace_function; 
  114.                   self::register_help_text( $type, $var, $help_text ); 
  115.                   $success = true; 
  116.               } 
  117.               else { 
  118.                   trigger_error( __( 'A replacement variable with the same name has already been registered. Try making your variable name unique.', 'wordpress-seo' ), E_USER_WARNING ); 
  119.               } 
  120.           } 
  121.           else { 
  122.               trigger_error( __( 'You cannot overrule a WPSEO standard variable replacement by registering a variable with the same name. Use the "wpseo_replacements" filter instead to adjust the replacement value.', 'wordpress-seo' ), E_USER_WARNING ); 
  123.           } 
  124.       } 
  125.  
  126.       return $success; 
  127.   }

  130.   /** 
  131.    * Replace `%%variable_placeholders%%` with their real value based on the current requested page/post/cpt/etc 
  132.    * 
  133.    * @param string $string the string to replace the variables in. 
  134.    * @param array  $args   the object some of the replacement values might come from,  
  135.    *                       could be a post, taxonomy or term. 
  136.    * @param array  $omit   variables that should not be replaced by this function. 
  137.    * 
  138.    * @return string 
  139.    */ 
  140.   public function replace( $string, $args, $omit = array() ) { 
  141.  
  142.       $string = strip_tags( $string ); 
  143.  
  144.       // Let's see if we can bail super early. 
  145.       if ( strpos( $string, '%%' ) === false ) { 
  146.           return WPSEO_Utils::standardize_whitespace( $string ); 
  147.       } 
  148.  
  149.       $args = (array) $args; 
  150.       if ( isset( $args['post_content'] ) && ! empty( $args['post_content'] ) ) { 
  151.           $args['post_content'] = WPSEO_Utils::strip_shortcode( $args['post_content'] ); 
  152.       } 
  153.       if ( isset( $args['post_excerpt'] ) && ! empty( $args['post_excerpt'] ) ) { 
  154.           $args['post_excerpt'] = WPSEO_Utils::strip_shortcode( $args['post_excerpt'] ); 
  155.       } 
  156.       $this->args = (object) wp_parse_args( $args, $this->defaults ); 
  157.  
  158.       // Clean $omit array. 
  159.       if ( is_array( $omit ) && $omit !== array() ) { 
  160.           $omit = array_map( array( __CLASS__, 'remove_var_delimiter' ), $omit ); 
  161.       } 
  162.  
  163.       $replacements = array(); 
  164.       if ( preg_match_all( '`%%([^%]+(%%single)?)%%?`iu', $string, $matches ) ) { 
  165.           $replacements = $this->set_up_replacements( $matches, $omit ); 
  166.       } 
  167.  
  168.       /** 
  169.        * Filter: 'wpseo_replacements' - Allow customization of the replacements before they are applied 
  170.        * 
  171.        * @api array $replacements The replacements 
  172.        */ 
  173.       $replacements = apply_filters( 'wpseo_replacements', $replacements ); 
  174.  
  175.       // Do the actual replacements. 
  176.       if ( is_array( $replacements ) && $replacements !== array() ) { 
  177.           $string = str_replace( array_keys( $replacements ), array_values( $replacements ), $string ); 
  178.       } 
  179.  
  180.       /** 
  181.        * Filter: 'wpseo_replacements_final' - Allow overruling of whether or not to remove placeholders 
  182.        * which didn't yield a replacement 
  183.        * 
  184.        * @example <code>add_filter( 'wpseo_replacements_final', '__return_false' );</code> 
  185.        * 
  186.        * @api     bool $final 
  187.        */ 
  188.       if ( apply_filters( 'wpseo_replacements_final', true ) === true && ( isset( $matches[1] ) && is_array( $matches[1] ) ) ) { 
  189.           // Remove non-replaced variables. 
  190.           $remove = array_diff( $matches[1], $omit ); // Make sure the $omit variables do not get removed. 
  191.           $remove = array_map( array( __CLASS__, 'add_var_delimiter' ), $remove ); 
  192.           $string = str_replace( $remove, '', $string ); 
  193.       } 
  194.  
  195.       // Undouble separators which have nothing between them, i.e. where a non-replaced variable was removed. 
  196.       if ( isset( $replacements['%%sep%%'] ) && ( is_string( $replacements['%%sep%%'] ) && $replacements['%%sep%%'] !== '' ) ) { 
  197.           $q_sep = preg_quote( $replacements['%%sep%%'], '`' ); 
  198.           $string = preg_replace( '`' . $q_sep . '(?:\s*' . $q_sep . ')*`u', $replacements['%%sep%%'], $string ); 
  199.       } 
  200.  
  201.       // Remove superfluous whitespace. 
  202.       $string = WPSEO_Utils::standardize_whitespace( $string ); 
  203.  
  204.       return trim( $string ); 
  205.   }

  208.   /** 
  209.    * Retrieve the replacements for the variables found. 
  210.    * 
  211.    * @param array $matches variables found in the original string - regex result. 
  212.    * @param array $omit    variables that should not be replaced by this function. 
  213.    * 
  214.    * @return array retrieved replacements - this might be a smaller array as some variables 
  215.    *               may not yield a replacement in certain contexts. 
  216.    */ 
  217.   private function set_up_replacements( $matches, $omit ) { 
  218.  
  219.       $replacements = array(); 
  220.  
  221.       // @todo -> figure out a way to deal with external functions starting with cf_/ct_. 
  222.       foreach ( $matches[1] as $k => $var ) { 
  223.  
  224.           // Don't set up replacements which should be omitted. 
  225.           if ( in_array( $var, $omit, true ) ) { 
  226.               continue; 
  227.           } 
  228.  
  229.           // Deal with variable variable names first. 
  230.           if ( strpos( $var, 'cf_' ) === 0 ) { 
  231.               $replacement = $this->retrieve_cf_custom_field_name( $var ); 
  232.           } 
  233.           elseif ( strpos( $var, 'ct_desc_' ) === 0 ) { 
  234.               $replacement = $this->retrieve_ct_desc_custom_tax_name( $var ); 
  235.           } 
  236.           elseif ( strpos( $var, 'ct_' ) === 0 ) { 
  237.               $single = ( isset( $matches[2][ $k ] ) && $matches[2][ $k ] !== '' ) ? true : false; 
  238.               $replacement = $this->retrieve_ct_custom_tax_name( $var, $single ); 
  239.           } // Deal with non-variable variable names. 
  240.           elseif ( method_exists( $this, 'retrieve_' . $var ) ) { 
  241.               $method_name = 'retrieve_' . $var; 
  242.               $replacement = $this->$method_name(); 
  243.           } // Deal with externally defined variable names. 
  244.           elseif ( isset( self::$external_replacements[ $var ] ) && ! is_null( self::$external_replacements[ $var ] ) ) { 
  245.               $replacement = call_user_func( self::$external_replacements[ $var ], $var, $this->args ); 
  246.           } 
  247.  
  248.           // Replacement retrievals can return null if no replacement can be determined, root those outs. 
  249.           if ( isset( $replacement ) ) { 
  250.               $var = self::add_var_delimiter( $var ); 
  251.               $replacements[ $var ] = $replacement; 
  252.           } 
  253.           unset( $replacement, $single, $method_name ); 
  254.       } 
  255.  
  256.       return $replacements; 
  257.   }

  542.   /** 
  543.    * Determine the page numbering of the current post/page/cpt 
  544.    * 
  545.    * @param string $request 'nr'|'max' - whether to return the page number or the max number of pages. 
  546.    * 
  547.    * @return int|null 
  548.    */ 
  549.   private function determine_pagenumbering( $request = 'nr' ) { 
  550.       global $wp_query, $post; 
  551.       $max_num_pages = null; 
  552.       $page_number = null; 
  553.  
  554.       $max_num_pages = 1; 
  555.  
  556.       if ( ! is_singular() ) { 
  557.           $page_number = get_query_var( 'paged' ); 
  558.           if ( $page_number === 0 || $page_number === '' ) { 
  559.               $page_number = 1; 
  560.           } 
  561.  
  562.           if ( isset( $wp_query->max_num_pages ) && ( $wp_query->max_num_pages != '' && $wp_query->max_num_pages != 0 ) ) { 
  563.               $max_num_pages = $wp_query->max_num_pages; 
  564.           } 
  565.       } 
  566.       else {
