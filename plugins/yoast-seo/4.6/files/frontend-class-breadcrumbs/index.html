  182.   /** 
  183.    * Get a term's parents. 
  184.    * 
  185.    * @param    object $term Term to get the parents for.

  189.   private function get_term_parents( $term ) { 
  190.       $tax = $term->taxonomy; 
  191.       $parents = array(); 
  192.       while ( $term->parent != 0 ) { 
  193.           $term = get_term( $term->parent, $tax ); 
  194.           $parents[] = $term; 
  195.       } 
  196.  
  197.       return array_reverse( $parents ); 
  198.   }

  200.   /** 
  201.    * Find the deepest term in an array of term objects 
  202.    * 
  203.    * @param array $terms Terms set. 
  204.    * 
  205.    * @return object 
  206.    */ 
  207.   private function find_deepest_term( $terms ) { 
  208.       /** 
  209.       Let's find the deepest term in this array, by looping through and then 
  210.          unsetting every term that is used as a parent by another one in the array. 
  211.       */ 
  212.       $terms_by_id = array(); 
  213.       foreach ( $terms as $term ) { 
  214.           $terms_by_id[ $term->term_id ] = $term; 
  215.       } 
  216.       foreach ( $terms as $term ) { 
  217.           unset( $terms_by_id[ $term->parent ] ); 
  218.       } 
  219.       unset( $term ); 
  220.  
  221.       /** 
  222.       As we could still have two subcategories, from different parent categories,  
  223.          let's pick the one with the lowest ordered ancestor. 
  224.       */ 
  225.       $parents_count = 0; 
  226.       $term_order = 9999; // Because ASC. 
  227.       reset( $terms_by_id ); 
  228.       $deepest_term = current( $terms_by_id ); 
  229.       foreach ( $terms_by_id as $term ) { 
  230.           $parents = $this->get_term_parents( $term ); 
  231.  
  232.           if ( count( $parents ) >= $parents_count ) { 
  233.               $parents_count = count( $parents ); 
  234.  
  235.               // If higher count. 
  236.               if ( count( $parents ) > $parents_count ) { 
  237.                   // Reset order. 
  238.                   $term_order = 9999; 
  239.               } 
  240.  
  241.               $parent_order = 9999; // Set default order. 
  242.               foreach ( $parents as $parent ) { 
  243.                   if ( $parent->parent == 0 && isset( $parent->term_order ) ) { 
  244.                       $parent_order = $parent->term_order; 
  245.                   } 
  246.               } 
  247.               unset( $parent ); 
  248.  
  249.               // Check if parent has lowest order. 
  250.               if ( $parent_order < $term_order ) { 
  251.                   $term_order = $parent_order; 
  252.                   $deepest_term = $term; 
  253.               } 
  254.           } 
  255.       } 
  256.  
  257.       return $deepest_term; 
  258.   }

  277.           $ancestors = array( $this->post->post_parent );

  281.        * Filter: Allow changing the ancestors for the Yoast SEO breadcrumbs output

  285.       $ancestors = apply_filters( 'wp_seo_get_bc_ancestors', $ancestors ); 
  286.

  288.           trigger_error( 'The return value for the "wp_seo_get_bc_ancestors" filter should be an array.', E_USER_WARNING );

  311.       elseif ( $this->show_on_front == 'page' && is_home() ) { 
  312.           $this->add_blog_crumb(); 
  313.       } 
  314.       elseif ( is_singular() ) { 
  315.           $this->maybe_add_pt_archive_crumb_for_post(); 
  316.  
  317.           if ( isset( $this->post->post_parent ) && 0 == $this->post->post_parent ) { 
  318.               $this->maybe_add_taxonomy_crumbs_for_post(); 
  319.           } 
  320.           else { 
  321.               $this->add_post_ancestor_crumbs(); 
  322.           } 
  323.  
  324.           if ( isset( $this->post->ID ) ) { 
  325.               $this->add_single_post_crumb( $this->post->ID ); 
  326.           } 
  327.       } 
  328.       else { 
  329.           if ( is_post_type_archive() ) { 
  330.               $post_type = $wp_query->get( 'post_type' ); 
  331.  
  332.               if ( $post_type && is_string( $post_type ) ) { 
  333.                   $this->add_ptarchive_crumb( $post_type ); 
  334.               } 
  335.           } 
  336.           elseif ( is_tax() || is_tag() || is_category() ) { 
  337.               $this->add_crumbs_for_taxonomy(); 
  338.           } 
  339.           elseif ( is_date() ) { 
  340.               if ( is_day() ) { 
  341.                   $this->add_linked_month_year_crumb(); 
  342.                   $this->add_date_crumb(); 
  343.               } 
  344.               elseif ( is_month() ) { 
  345.                   $this->add_month_crumb(); 
  346.               } 
  347.               elseif ( is_year() ) { 
  348.                   $this->add_year_crumb(); 
  349.               } 
  350.           } 
  351.           elseif ( is_author() ) { 
  352.               $user = $wp_query->get_queried_object(); 
  353.               $this->add_predefined_crumb( 
  354.                   $this->options['breadcrumbs-archiveprefix'] . ' ' . $user->display_name,  
  355.                   null,  
  356.                   true 
  357.  ); 
  358.           } 
  359.           elseif ( is_search() ) { 
  360.               $this->add_predefined_crumb( 
  361.                   $this->options['breadcrumbs-searchprefix'] . ' "' . esc_html( get_search_query() ) . '"',  
  362.                   null,  
  363.                   true 
  364.  ); 
  365.           } 
  366.           elseif ( is_404() ) { 
  367.               $this->add_predefined_crumb( 
  368.                   $this->options['breadcrumbs-404crumb'],  
  369.                   null,  
  370.                   true 
  371.  ); 
  372.           } 
  373.       }

  481.   private function maybe_add_taxonomy_crumbs_for_post() { 
  482.       if ( isset( $this->options[ 'post_types-' . $this->post->post_type . '-maintax' ] ) && $this->options[ 'post_types-' . $this->post->post_type . '-maintax' ] != '0' ) {

  484.           if ( isset( $this->post->ID ) ) { 
  485.               $terms = get_the_terms( $this->post, $main_tax ); 
  486.  
  487.               if ( is_array( $terms ) && $terms !== array() ) { 
  488.  
  489.                   $primary_term = new WPSEO_Primary_Term( $main_tax, $this->post->ID ); 
  490.                   if ( $primary_term->get_primary_term() ) { 
  491.                       $breadcrumb_term = get_term( $primary_term->get_primary_term(), $main_tax ); 
  492.                   } 
  493.                   else { 
  494.                       $breadcrumb_term = $this->find_deepest_term( $terms ); 
  495.                   } 
  496.

  498.                       $parent_terms = $this->get_term_parents( $breadcrumb_term ); 
  499.                       foreach ( $parent_terms as $parent_term ) { 
  500.                           $this->add_term_crumb( $parent_term ); 
  501.                       } 
  502.                   } 
  503.  
  504.                   $this->add_term_crumb( $breadcrumb_term ); 
  505.               } 
  506.           } 
  507.       } 
  508.   }

  592.       $this->add_predefined_crumb( 
  593.           $this->options['breadcrumbs-archiveprefix'] . ' ' . esc_html( get_query_var( 'year' ) ),  
  594.           null,  
  595.           true 
  596.  );

  621.   /** 
  622.    * Take the crumbs array and convert each crumb to a single breadcrumb string. 
  623.    * 
  624.    * @link http://support.google.com/webmasters/bin/answer.py?hl=en&answer=185417 Google documentation on RDFA 
  625.    */ 
  626.   private function prepare_links() { 
  627.       if ( ! is_array( $this->crumbs ) || $this->crumbs === array() ) { 
  628.           return; 
  629.       } 
  630.  
  631.       foreach ( $this->crumbs as $i => $crumb ) { 
  632.           $link_info = $crumb; // Keep pre-set url/text combis. 
  633.  
  634.           if ( isset( $crumb['id'] ) ) { 
  635.               $link_info = $this->get_link_info_for_id( $crumb['id'] ); 
  636.           } 
  637.           if ( isset( $crumb['term'] ) ) { 
  638.               $link_info = $this->get_link_info_for_term( $crumb['term'] ); 
  639.           } 
  640.           if ( isset( $crumb['ptarchive'] ) ) { 
  641.               $link_info = $this->get_link_info_for_ptarchive( $crumb['ptarchive'] ); 
  642.           } 
  643.  
  644.           $this->links[] = $this->crumb_to_link( $link_info, $i ); 
  645.       } 
  646.   }

  648.   /** 
  649.    * Retrieve link url and text based on post id 
  650.    * 
  651.    * @param int $id Post ID. 
  652.    * 
  653.    * @return array Array of link text and url 
  654.    */ 
  655.   private function get_link_info_for_id( $id ) { 
  656.       $link = array(); 
  657.  
  658.       $link['url'] = get_permalink( $id ); 
  659.       $link['text'] = WPSEO_Meta::get_value( 'bctitle', $id ); 
  660.       if ( $link['text'] === '' ) { 
  661.           $link['text'] = strip_tags( get_the_title( $id ) ); 
  662.       } 
  663.  
  664.       /** 
  665.        * Filter: 'wp_seo_get_bc_title' - Allow developer to filter the Yoast SEO Breadcrumb title. 
  666.        * 
  667.        * @api string $link_text The Breadcrumb title text 
  668.        * 
  669.        * @param int $link_id The post ID 
  670.        */ 
  671.       $link['text'] = apply_filters( 'wp_seo_get_bc_title', $link['text'], $id ); 
  672.  
  673.       return $link; 
  674.   }

  677.    * Retrieve link url and text based on term object 
  678.    * 
  679.    * @param object $term Term object. 
  680.    * 
  681.    * @return array Array of link text and url 
  682.    */ 
  683.   private function get_link_info_for_term( $term ) { 
  684.       $link = array(); 
  685.  
  686.       $bctitle = WPSEO_Taxonomy_Meta::get_term_meta( $term, $term->taxonomy, 'bctitle' ); 
  687.       if ( ! is_string( $bctitle ) || $bctitle === '' ) { 
  688.           $bctitle = $term->name;

  697.   /** 
  698.    * Retrieve link url and text based on post type 
  699.    * 
  700.    * @param string $pt Post type. 
  701.    * 
  702.    * @return array Array of link text and url 
  703.    */ 
  704.   private function get_link_info_for_ptarchive( $pt ) { 
  705.       $link = array(); 
  706.       $archive_title = ''; 
  707.  
  708.       if ( isset( $this->options[ 'bctitle-ptarchive-' . $pt ] ) && $this->options[ 'bctitle-ptarchive-' . $pt ] !== '' ) { 
  709.  
  710.           $archive_title = $this->options[ 'bctitle-ptarchive-' . $pt ]; 
  711.       } 
  712.       else { 
  713.           $post_type_obj = get_post_type_object( $pt ); 
  714.           if ( is_object( $post_type_obj ) ) { 
  715.               if ( isset( $post_type_obj->label ) && $post_type_obj->label !== '' ) { 
  716.                   $archive_title = $post_type_obj->label; 
  717.               } 
  718.               elseif ( isset( $post_type_obj->labels->menu_name ) && $post_type_obj->labels->menu_name !== '' ) { 
  719.                   $archive_title = $post_type_obj->labels->menu_name; 
  720.               } 
  721.               else { 
  722.                   $archive_title = $post_type_obj->name; 
  723.               } 
  724.           } 
  725.       } 
  726.  
  727.       $link['url'] = get_post_type_archive_link( $pt ); 
  728.       $link['text'] = $archive_title; 
  729.  
  730.       return $link; 
  731.   }

  734.   /** 
  735.    * Create a breadcrumb element string 
  736.    * 
  737.    * @todo The `$paged` variable only works for archives, not for paged articles, so this does not work 
  738.    * for paged article at this moment. 
  739.    * 
  740.    * @param  array $link Link info array containing the keys: 
  741.    *                     'text' => (string) link text 
  742.    *                     'url' => (string) link url 
  743.    *                     (optional) 'title' => (string) link title attribute text 
  744.    *                     (optional) 'allow_html' => (bool) whether to (not) escape html in the link text 
  745.    *                     This prevents html stripping from the text strings set in the 
  746.    *                     WPSEO -> Internal Links options page. 
  747.    * @param  int   $i    Index for the current breadcrumb. 
  748.    * 
  749.    * @return string 
  750.    */ 
  751.   private function crumb_to_link( $link, $i ) { 
  752.       $link_output = ''; 
  753.  
  754.       if ( isset( $link['text'] ) && ( is_string( $link['text'] ) && $link['text'] !== '' ) ) { 
  755.  
  756.           $link['text'] = trim( $link['text'] ); 
  757.           if ( ! isset( $link['allow_html'] ) || $link['allow_html'] !== true ) { 
  758.               $link['text'] = esc_html( $link['text'] ); 
  759.           } 
  760.  
  761.           $inner_elm = 'span'; 
  762.           if ( $this->options['breadcrumbs-boldlast'] === true && $i === ( $this->crumb_count - 1 ) ) { 
  763.               $inner_elm = 'strong'; 
  764.           } 
  765.  
  766.           if ( ( isset( $link['url'] ) && ( is_string( $link['url'] ) && $link['url'] !== '' ) ) && 
  767.                ( $i < ( $this->crumb_count - 1 ) ) 
  768.  ) { 
  769.               if ( $i === 0 ) { 
  770.                   $link_output .= '<' . $this->element . ' typeof="v:Breadcrumb">'; 
  771.               } 
  772.               else { 
  773.                   $link_output .= '<' . $this->element . ' rel="v:child" typeof="v:Breadcrumb">'; 
  774.               } 
  775.               $title_attr = isset( $link['title'] ) ? ' title="' . esc_attr( $link['title'] ) . '"' : ''; 
  776.               $link_output .= '<a href="' . esc_url( $link['url'] ) . '" rel="v:url" property="v:title"' . $title_attr . '>' . $link['text'] . '</a>'; 
  777.           } 
  778.           else { 
  779.               $link_output .= '<' . $inner_elm . ' class="breadcrumb_last">' . $link['text'] . '</' . $inner_elm . '>'; 
  780.               // This is the last element, now close all previous elements. 
  781.               while ( $i > 0 ) { 
  782.                   $link_output .= '</' . $this->element . '>'; 
  783.                   $i--; 
  784.               } 
  785.           } 
  786.       } 
  787.  
  788.       /** 
  789.        * Filter: 'wpseo_breadcrumb_single_link' - Allow changing of each link being put out by the Yoast SEO breadcrumbs class 
  790.        * 
  791.        * @api string $link_output The output string 
  792.        * 
  793.        * @param array $link The link array. 
  794.        */ 
  795.  
  796.       return apply_filters( 'wpseo_breadcrumb_single_link', $link_output, $link ); 
  797.   }

  800.   /** 
  801.    * Create a complete breadcrumb string from an array of breadcrumb element strings 
  802.    */ 
  803.   private function links_to_string() { 
  804.       if ( is_array( $this->links ) && $this->links !== array() ) { 
  805.           // Remove any effectively empty links. 
  806.           $links = array_map( 'trim', $this->links ); 
  807.           $links = array_filter( $links ); 
  808.  
  809.           $this->output = implode( $this->separator, $links ); 
  810.       } 
  811.   }

  813.   /** 
  814.    * Wrap a complete breadcrumb string in a Breadcrumb RDFA wrapper 
  815.    */ 
  816.   private function wrap_breadcrumb() { 
  817.       if ( is_string( $this->output ) && $this->output !== '' ) { 
  818.           $output = '<' . $this->wrapper . $this->get_output_id() . $this->get_output_class() . ' xmlns:v="http://rdf.data-vocabulary.org/#">' . $this->output . '</' . $this->wrapper . '>'; 
  819.  
  820.           /** 
  821.            * Filter: 'wpseo_breadcrumb_output' - Allow changing the HTML output of the Yoast SEO breadcrumbs class 
  822.            * 
  823.            * @api string $unsigned HTML output 
  824.            */ 
  825.           $output = apply_filters( 'wpseo_breadcrumb_output', $output ); 
  826.  
  827.           if ( $this->options['breadcrumbs-prefix'] !== '' ) { 
  828.               $output = "\t" . $this->options['breadcrumbs-prefix'] . "\n" . $output; 
  829.           } 
  830.  
  831.           $this->output = $output; 
  832.       } 
  833.   }
