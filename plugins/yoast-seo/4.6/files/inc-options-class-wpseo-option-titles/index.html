  256.   protected function validate_option( $dirty, $clean, $old ) { 
  257.       foreach ( $clean as $key => $value ) { 
  258.           $switch_key = $this->get_switch_key( $key ); 
  259.  
  260.           switch ( $switch_key ) { 
  261.               /** 
  262.               Text fields 
  263.               */ 
  264.  
  265.               /** 
  266.               Covers: 
  267.                      'title-home-wpseo', 'title-author-wpseo', 'title-archive-wpseo',  
  268.                      'title-search-wpseo', 'title-404-wpseo' 
  269.                      'title-' . $pt->name 
  270.                      'title-ptarchive-' . $pt->name 
  271.                      'title-tax-' . $tax->name 
  272.               */ 
  273.               case 'title-': 
  274.                   if ( isset( $dirty[ $key ] ) ) { 
  275.                       $clean[ $key ] = WPSEO_Utils::sanitize_text_field( $dirty[ $key ] ); 
  276.                   } 
  277.                   break; 
  278.  
  279.               /** 
  280.               Covers: 
  281.                      'metadesc-home-wpseo', 'metadesc-author-wpseo', 'metadesc-archive-wpseo' 
  282.                      'metadesc-' . $pt->name 
  283.                      'metadesc-ptarchive-' . $pt->name 
  284.                      'metadesc-tax-' . $tax->name 
  285.               */ 
  286.               case 'metadesc-': 
  287.                   /** 
  288.                   Covers: 
  289.                            'metakey-home-wpseo', 'metakey-author-wpseo' 
  290.                            'metakey-' . $pt->name 
  291.                            'metakey-ptarchive-' . $pt->name 
  292.                            'metakey-tax-' . $tax->name 
  293.                   */ 
  294.               case 'metakey-': 
  295.                   /** 
  296.                   Covers: 
  297.                            ''bctitle-ptarchive-' . $pt->name 
  298.                   */ 
  299.               case 'bctitle-ptarchive-': 
  300.                   if ( isset( $dirty[ $key ] ) && $dirty[ $key ] !== '' ) { 
  301.                       $clean[ $key ] = WPSEO_Utils::sanitize_text_field( $dirty[ $key ] ); 
  302.                   } 
  303.                   break; 
  304.  
  305.  
  306.               /** integer field - not in form*/ 
  307.               case 'title_test': 
  308.                   if ( isset( $dirty[ $key ] ) ) { 
  309.                       $int = WPSEO_Utils::validate_int( $dirty[ $key ] ); 
  310.                       if ( $int !== false && $int >= 0 ) { 
  311.                           $clean[ $key ] = $int; 
  312.                       } 
  313.                   } 
  314.                   elseif ( isset( $old[ $key ] ) ) { 
  315.                       $int = WPSEO_Utils::validate_int( $old[ $key ] ); 
  316.                       if ( $int !== false && $int >= 0 ) { 
  317.                           $clean[ $key ] = $int; 
  318.                       } 
  319.                   } 
  320.                   break; 
  321.  
  322.               /** Separator field - Radio */ 
  323.               case 'separator': 
  324.                   if ( isset( $dirty[ $key ] ) && $dirty[ $key ] !== '' ) { 
  325.  
  326.                       // Get separator fields. 
  327.                       $separator_fields = $this->get_separator_options(); 
  328.  
  329.                       // Check if the given separator is exists. 
  330.                       if ( isset( $separator_fields[ $dirty[ $key ] ] ) ) { 
  331.                           $clean[ $key ] = $dirty[ $key ]; 
  332.                       } 
  333.                   } 
  334.                   break; 
  335.  
  336.               /** 
  337.               Boolean fields 
  338.               */ 
  339.  
  340.               /** 
  341.               Covers: 
  342.                *        'noindex-subpages-wpseo', 'noindex-author-wpseo', 'noindex-archive-wpseo' 
  343.                *        'noindex-' . $pt->name 
  344.                *        'noindex-ptarchive-' . $pt->name 
  345.                *        'noindex-tax-' . $tax->name 
  346.                *        'forcerewritetitle': 
  347.                *        'usemetakeywords': 
  348.                *        'noodp': 
  349.                *        'noydir': 
  350.                *        'disable-author': 
  351.                *        'disable-date': 
  352.                *        'disable-post_format'; 
  353.                *        'noindex-' 
  354.                *        'showdate-' 
  355.                *        'showdate-'. $pt->name 
  356.                *        'hideeditbox-' 
  357.                *         'hideeditbox-'. $pt->name 
  358.                *        'hideeditbox-tax-' . $tax->name 
  359.                */ 
  360.               default: 
  361.                   $clean[ $key ] = ( isset( $dirty[ $key ] ) ? WPSEO_Utils::validate_bool( $dirty[ $key ] ) : false );

  363.           } 
  364.       } 
  365.  
  366.       return $clean;

  376.    *                                       version specific upgrades will be disregarded.

  384.  
  385.       // Double-run this function to ensure renaming of the taxonomy options will work.

  394.

  401.       $old_option = null;

  405.               $old_option = $all_old_option_values['wpseo_indexation'];

  411.       if ( is_array( $old_option ) && $old_option !== array() ) { 
  412.           $move = array( 
  413.               'noindexauthor' => 'noindex-author',  
  414.               'disableauthor' => 'disable-author',  
  415.               'noindexdate' => 'noindex-archive',  
  416.               'noindexcat' => 'noindex-category',

  421.           foreach ( $move as $old => $new ) { 
  422.               if ( isset( $old_option[ $old ] ) && ! isset( $option_value[ $new ] ) ) {

  425.           } 
  426.           unset( $move, $old, $new ); 
  427.       } 
  428.       unset( $old_option ); 
  429.  
  430.

  436.

  440.              'home', 'author', 'archive', 'search', '404', 'subpages' 
  441.   
  442.              Similarly, renaming the tax options to avoid a custom post type and a taxonomy 
  443.              with the same name occupying the same option

  445.       $rename = array( 
  446.           'title-home' => 'title-home-wpseo',  
  447.           'title-author' => 'title-author-wpseo',  
  448.           'title-archive' => 'title-archive-wpseo',  
  449.           'title-search' => 'title-search-wpseo',  
  450.           'title-404' => 'title-404-wpseo',  
  451.           'metadesc-home' => 'metadesc-home-wpseo',  
  452.           'metadesc-author' => 'metadesc-author-wpseo',  
  453.           'metadesc-archive' => 'metadesc-archive-wpseo',  
  454.           'metakey-home' => 'metakey-home-wpseo',  
  455.           'metakey-author' => 'metakey-author-wpseo',  
  456.           'noindex-subpages' => 'noindex-subpages-wpseo',  
  457.           'noindex-author' => 'noindex-author-wpseo',  
  458.           'noindex-archive' => 'noindex-archive-wpseo',  
  459.  ); 
  460.       foreach ( $rename as $old => $new ) { 
  461.           if ( isset( $option_value[ $old ] ) && ! isset( $option_value[ $new ] ) ) { 
  462.               $option_value[ $new ] = $option_value[ $old ]; 
  463.               unset( $option_value[ $old ] ); 
  464.           } 
  465.       } 
  466.       unset( $rename, $old, $new ); 
  467.  
  468.  
  469.       /** 
  470.        * @internal This clean-up action can only be done effectively once the taxonomies and post_types 
  471.        * have been registered, i.e. at the end of the init action. 
  472.        */ 
  473.       if ( isset( $original ) && current_filter() === 'wpseo_double_clean_titles' || did_action( 'wpseo_double_clean_titles' ) > 0 ) { 
  474.           $rename = array( 
  475.               'title-' => 'title-tax-',  
  476.               'metadesc-' => 'metadesc-tax-',  
  477.               'metakey-' => 'metakey-tax-',  
  478.               'noindex-' => 'noindex-tax-',  
  479.               'tax-hideeditbox-' => 'hideeditbox-tax-',  
  480.  
  481.  ); 
  482.           $taxonomy_names = get_taxonomies( array( 'public' => true ), 'names' ); 
  483.           $post_type_names = get_post_types( array( 'public' => true ), 'names' ); 
  484.           $defaults = $this->get_defaults(); 
  485.           if ( $taxonomy_names !== array() ) { 
  486.               foreach ( $taxonomy_names as $tax ) { 
  487.                   foreach ( $rename as $old_prefix => $new_prefix ) { 
  488.                       if ( 
  489.                           ( isset( $original[ $old_prefix . $tax ] ) && ! isset( $original[ $new_prefix . $tax ] ) )

  491.                                || ( isset( $option_value[ $new_prefix . $tax ] ) 
  492.                                     && $option_value[ $new_prefix . $tax ] === $defaults[ $new_prefix . $tax ] ) ) 
  493.  ) { 
  494.                           $option_value[ $new_prefix . $tax ] = $original[ $old_prefix . $tax ]; 
  495.  
  496.                           /** 
  497.                           Check if there is a cpt with the same name as the tax,  
  498.                                  if so, we should make sure that the old setting hasn't been removed 
  499.                           */ 
  500.                           if ( ! isset( $post_type_names[ $tax ] ) && isset( $option_value[ $old_prefix . $tax ] ) ) {
