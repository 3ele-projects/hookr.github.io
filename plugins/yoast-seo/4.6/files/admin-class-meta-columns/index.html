  471.   /** 
  472.    * Hacky way to get round the limitation that you can only have AND *or* OR relationship between 
  473.    * meta key clauses and not a combination - which is what we need. 
  474.    * 
  475.    * @deprecated 3.5 Unnecessary with nested meta queries in core. 
  476.    * @codeCoverageIgnore 
  477.    * 
  478.    * @param    string $where Where clause. 
  479.    * 
  480.    * @return    string 
  481.    */ 
  482.   public function seo_score_posts_where( $where ) { 
  483.  
  484.       _deprecated_function( __METHOD__, '3.5' ); 
  485.  
  486.       global $wpdb; 
  487.  
  488.       /** Find the two mutually exclusive noindex clauses which should be changed from AND to OR relation */ 
  489.       $find = '`([\s]+AND[\s]+)((?:' . $wpdb->prefix . 'postmeta|mt[0-9]|mt1)\.post_id IS NULL[\s]+)AND([\s]+\([\s]*(?:' . $wpdb->prefix . 'postmeta|mt[0-9])\.meta_key = \'' . WPSEO_Meta::$meta_prefix . 'meta-robots-noindex\' AND CAST\([^\)]+\)[^\)]+\))`'; 
  490.  
  491.       $replace = '$1( $2OR$3 )'; 
  492.  
  493.       $new_where = preg_replace( $find, $replace, $where ); 
  494.  
  495.       if ( $new_where ) { 
  496.           return $new_where; 
  497.       } 
  498.       return $where; 
  499.   }
