  665.     // PaperG - Function to convert the text from one character set to another if the two sets are not the same. 
  666.     function convert_text($text) 
  667.     { 
  668.         global $debug_object; 
  669.         if (is_object($debug_object)) {$debug_object->debug_log_entry(1);} 
  670.  
  671.         $converted_text = $text; 
  672.  
  673.         $sourceCharset = ""; 
  674.         $targetCharset = ""; 
  675.  
  676.         if ($this->dom) 
  677.         { 
  678.             $sourceCharset = strtoupper($this->dom->_charset); 
  679.             $targetCharset = strtoupper($this->dom->_target_charset); 
  680.         } 
  681.         if (is_object($debug_object)) {$debug_object->debug_log(3, "source charset: " . $sourceCharset . " target charaset: " . $targetCharset);} 
  682.  
  683.         if (!empty($sourceCharset) && !empty($targetCharset) && (strcasecmp($sourceCharset, $targetCharset) != 0)) 
  684.         { 
  685.             // Check if the reported encoding could have been incorrect and the text is actually already UTF-8 
  686.             if ((strcasecmp($targetCharset, 'UTF-8') == 0) && ($this->is_utf8($text))) 
  687.             { 
  688.                 $converted_text = $text; 
  689.             } 
  690.             else 
  691.             { 
  692.                 $converted_text = iconv($sourceCharset, $targetCharset, $text); 
  693.             } 
  694.         } 
  695.  
  696.         // Lets make sure that we don't have that silly BOM issue with any of the utf-8 text we output. 
  697.         if ($targetCharset == 'UTF-8') 
  698.         { 
  699.             if (substr($converted_text, 0, 3) == "\xef\xbb\xbf") 
  700.             { 
  701.                 $converted_text = substr($converted_text, 3); 
  702.             } 
  703.             if (substr($converted_text, -3) == "\xef\xbb\xbf") 
  704.             { 
  705.                 $converted_text = substr($converted_text, 0, -3); 
  706.             } 
  707.         } 
  708.  
  709.         return $converted_text; 
  710.     }

  712.     /** 
  713.     * Returns true if $string is valid UTF-8 and false otherwise. 
  714.     * 
  715.     * @param mixed $str String to be tested 
  716.     * @return boolean 
  717.     */ 
  718.     static function is_utf8($str) 
  719.     { 
  720.         $c=0; $b=0; 
  721.         $bits=0; 
  722.         $len=strlen($str); 
  723.         for($i=0; $i<$len; $i++) 
  724.         { 
  725.             $c=ord($str[$i]); 
  726.             if($c > 128) 
  727.             { 
  728.                 if(($c >= 254)) return false; 
  729.                 elseif($c >= 252) $bits=6; 
  730.                 elseif($c >= 248) $bits=5; 
  731.                 elseif($c >= 240) $bits=4; 
  732.                 elseif($c >= 224) $bits=3; 
  733.                 elseif($c >= 192) $bits=2; 
  734.                 else return false; 
  735.                 if(($i+$bits) > $len) return false; 
  736.                 while($bits > 1) 
  737.                 { 
  738.                     $i++; 
  739.                     $b=ord($str[$i]); 
  740.                     if($b < 128 || $b > 191) return false; 
  741.                     $bits--; 
  742.                 } 
  743.             } 
  744.         } 
  745.         return true; 
  746.     }

  755.     /** 
  756.      * Function to try a few tricks to determine the displayed size of an img on the page. 
  757.      * NOTE: This will ONLY work on an IMG tag. Returns FALSE on all other tag types. 
  758.      * 
  759.      * @author John Schlick 
  760.      * @version April 19 2012 
  761.      * @return array an array containing the 'height' and 'width' of the image on the page or -1 if we can't figure it out. 
  762.      */ 
  763.     function get_display_size() 
  764.     { 
  765.         global $debug_object; 
  766.  
  767.         $width = -1; 
  768.         $height = -1; 
  769.  
  770.         if ($this->tag !== 'img') 
  771.         { 
  772.             return false; 
  773.         } 
  774.  
  775.         // See if there is aheight or width attribute in the tag itself. 
  776.         if (isset($this->attr['width'])) 
  777.         { 
  778.             $width = $this->attr['width']; 
  779.         } 
  780.  
  781.         if (isset($this->attr['height'])) 
  782.         { 
  783.             $height = $this->attr['height']; 
  784.         } 
  785.  
  786.         // Now look for an inline style. 
  787.         if (isset($this->attr['style'])) 
  788.         { 
  789.             // Thanks to user gnarf from stackoverflow for this regular expression. 
  790.             $attributes = array(); 
  791.             preg_match_all("/([\w-]+)\s*:\s*([^;]+)\s*;?/", $this->attr['style'], $matches, PREG_SET_ORDER); 
  792.             foreach ($matches as $match) { 
  793.               $attributes[$match[1]] = $match[2]; 
  794.             } 
  795.  
  796.             // If there is a width in the style attributes: 
  797.             if (isset($attributes['width']) && $width == -1) 
  798.             { 
  799.                 // check that the last two characters are px (pixels) 
  800.                 if (strtolower(substr($attributes['width'], -2)) == 'px') 
  801.                 { 
  802.                     $proposed_width = substr($attributes['width'], 0, -2); 
  803.                     // Now make sure that it's an integer and not something stupid. 
  804.                     if (filter_var($proposed_width, FILTER_VALIDATE_INT)) 
  805.                     { 
  806.                         $width = $proposed_width; 
  807.                     } 
  808.                 } 
  809.             } 
  810.  
  811.             // If there is a width in the style attributes: 
  812.             if (isset($attributes['height']) && $height == -1) 
  813.             { 
  814.                 // check that the last two characters are px (pixels) 
  815.                 if (strtolower(substr($attributes['height'], -2)) == 'px') 
  816.                 { 
  817.                     $proposed_height = substr($attributes['height'], 0, -2); 
  818.                     // Now make sure that it's an integer and not something stupid. 
  819.                     if (filter_var($proposed_height, FILTER_VALIDATE_INT)) 
  820.                     { 
  821.                         $height = $proposed_height; 
  822.                     } 
  823.                 } 
  824.             } 
  825.  
  826.         } 
  827.  
  828.         // Future enhancement: 
  829.         // Look in the tag to see if there is a class or id specified that has a height or width attribute to it. 
  830.  
  831.         // Far future enhancement 
  832.         // Look at all the parent tags of this image to see if they specify a class or id that has an img selector that specifies a height or width 
  833.         // Note that in this case, the class or id will have the img subselector for it to apply to the image. 
  834.  
  835.         // ridiculously far future development 
  836.         // If the class or id is specified in a SEPARATE css file thats not on the page, go get it and do what we were just doing for the ones on the page. 
  837.  
  838.         $result = array('height' => $height,  
  839.                         'width' => $width); 
  840.         return $result; 
  841.     }
