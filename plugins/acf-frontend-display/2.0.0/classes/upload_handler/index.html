  621.             $_SERVER['HTTP_CONTENT_DESCRIPTION'] : null;

  623.         // Content-Range: bytes 0-524287/2000000 
  624.         $content_range = isset($_SERVER['HTTP_CONTENT_RANGE']) ? 
  625.             split('[^0-9]+', $_SERVER['HTTP_CONTENT_RANGE']) : null; 
  626.         $size =  $content_range ? $content_range[3] : null; 
  627.         $info = array(); 
  628.         if ($upload && is_array($upload['tmp_name'])) { 
  629.             // param_name is an array identifier like "files[]",

  637.                     $upload['error'][$index],

  639.                     $content_range 
  640.  ); 
  641.             }
