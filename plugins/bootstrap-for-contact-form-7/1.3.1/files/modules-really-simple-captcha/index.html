  144. function cf7bs_captcha_ajax_refill( $items ) { 
  145.   if ( ! is_array( $items ) ) { 
  146.       return $items; 
  147.   } 
  148.  
  149.   $fes = wpcf7_scan_shortcode( array( 'type' => 'captchar' ) ); 
  150.  
  151.   if ( empty( $fes ) ) { 
  152.       return $items; 
  153.   } 
  154.  
  155.   $refill = array(); 
  156.  
  157.   foreach ( $fes as $fe ) { 
  158.       if ( cf7bs_captchar_has_captchac( $fe ) ) { 
  159.           $fe = cf7bs_captchar_to_captchac( $fe ); 
  160.  
  161.           $name = $fe['name']; 
  162.           $options = $fe['options']; 
  163.  
  164.           if ( empty( $name ) ) { 
  165.               continue; 
  166.           } 
  167.  
  168.           $op = wpcf7_captchac_options( $options ); 
  169.           if ( $filename = wpcf7_generate_captcha( $op ) ) { 
  170.               $captcha_url = wpcf7_captcha_url( $filename ); 
  171.               $refill[ $name ] = $captcha_url; 
  172.           } 
  173.       } 
  174.   } 
  175.  
  176.   if ( count( $refill ) > 0 ) { 
  177.       if ( ! isset( $items['captcha'] ) ) { 
  178.           $items['captcha'] = $refill; 
  179.       } else { 
  180.           $items['captcha'] = array_merge( $items['captcha'], $refill ); 
  181.       } 
  182.   } 
  183.  
  184.   return $items; 
  185. }
