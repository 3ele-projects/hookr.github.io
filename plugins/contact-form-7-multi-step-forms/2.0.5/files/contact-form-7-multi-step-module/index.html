  209. function cf7msm_step_2($cf7) { 
  210.   $has_wpcf7_class = class_exists( 'WPCF7_ContactForm' ) && method_exists($cf7, 'prop'); 
  211.   $form_id = ''; 
  212.   if ( $has_wpcf7_class ) { 
  213.       $formstring = $cf7->prop('form'); 
  214.       $form_id = $cf7->id(); 
  215.   } 
  216.   else { 
  217.       $formstring = $cf7->form; 
  218.       $form_id = $cf7->id; 
  219.   } 
  220.  
  221.   //check if form has a step field 
  222.   if (!is_admin() && 
  223.       ( preg_match('/\[multistep "(\d+)-(\d+)-?(.*)"\]/', $formstring, $matches) || preg_match('/\[hidden step "(\d+)-(\d+)"\]/', $formstring, $matches) ) ) { 
  224.       $step = cf7msm_get('step'); 
  225.       if ( !isset($matches[1]) 
  226.           || ($matches[1] != 1 && empty($step) ) 
  227.           || ($matches[1] != 1 && ((int) $step) + 1 < $matches[1]) ) { 
  228.           if ( $has_wpcf7_class ) { 
  229.               $cf7->set_properties( array( 
  230.                   'form' => apply_filters('wh_hide_cf7_step_message', __( 'Please fill out the form on the previous page', 'contact-form-7-multi-step-module' ), $form_id ) 
  231.  ) ); 
  232.           } 
  233.           else { 
  234.               $cf7->form = apply_filters('wh_hide_cf7_step_message', __( 'Please fill out the form on the previous page', 'contact-form-7-multi-step-module' ), $form_id ); 
  235.           } 
  236.       } 
  237.       if (count($matches) >= 3 && $matches[1] != $matches[2]) { 
  238.           add_filter('wpcf7_ajax_json_echo', 'cf7msm_clear_success_message', 10, 2); 
  239.       } 
  240.   } 
  241.  
  242.   return $cf7; 
  243. }

  249. function cf7msm_store_data_steps_filter( $cf7_posted_data ) { 
  250.   if (isset($cf7_posted_data['step'])) { 
  251.       if (preg_match('/(\d+)-(\d+)/', $cf7_posted_data['step'], $matches)) { 
  252.           $curr_step = $matches[1]; 
  253.           $last_step = $matches[2]; 
  254.       } 
  255.  
  256.       //for back button 
  257.       $prev_urls = cf7msm_get('cf7msm_prev_urls'); 
  258.       if ( empty($prev_urls) ) { 
  259.           $prev_urls = array(); 
  260.       } 
  261.       //example: 
  262.       // on step 1,  
  263.       //    prev url {"2-3":"page-2"} will be set. 
  264.       //    back button is pressed, key "1-3" is looked up and returns undefined 
  265.       // on step 2,  
  266.       //    prev url {"3-3":"page-2"} will be set. 
  267.       //    back button is pressed, key "2-3" is looked up and returns "page-1" 
  268.       // on step 3,  
  269.       //    prev url {"4-3":"page-3"} will be set. - not used 
  270.       //    back button is pressed, key "3-3" is looked up and returns "page-2" 
  271.       // step 
  272.       $prev_urls[ ($curr_step+1) . '-' . $last_step] = cf7msm_current_url(); 
  273.       cf7msm_set('cf7msm_prev_urls', $prev_urls); 
  274.  
  275.       $prev_data = cf7msm_get('cf7msm_posted_data', '' ); 
  276.       if (!is_array($prev_data)) { 
  277.           $prev_data = array(); 
  278.       } 
  279.       //remove empty [form] tags from posted_data so $prev_data can be stored. 
  280.       $fes = wpcf7_scan_shortcode(); 
  281.       foreach ( $fes as $fe ) { 
  282.           if ( empty( $fe['name'] ) || ( $fe['type'] != 'form' && $fe['type'] != 'multiform' ) ) 
  283.               continue; 
  284.           unset($cf7_posted_data[$fe['name']]); 
  285.       } 
  286.       $free_text_keys = array(); 
  287.       foreach ( $prev_data as $key => $value ) { 
  288.           if ( strpos( $key, CF7MSM_FREE_TEXT_PREFIX_RADIO ) === 0 ) { 
  289.               $free_text_keys[$key] = str_replace(CF7MSM_FREE_TEXT_PREFIX_RADIO, '', $key); 
  290.           } 
  291.           else if ( strpos( $key, CF7MSM_FREE_TEXT_PREFIX_CHECKBOX ) === 0 ) { 
  292.               $free_text_keys[$key] = str_replace( CF7MSM_FREE_TEXT_PREFIX_CHECKBOX, '', $key ); 
  293.           } 
  294.       } 
  295.       //if original key is set and not free text, remove free text to reflect posted data. 
  296.       foreach( $free_text_keys as $free_text_key => $original_key ) { 
  297.           if ( isset( $cf7_posted_data[$original_key] ) && !isset( $cf7_posted_data[$free_text_key] ) ) { 
  298.               unset( $prev_data[$free_text_key] ); 
  299.           } 
  300.       } 
  301.       if ($curr_step != $last_step) { 
  302.           $wpcf7 = WPCF7_ContactForm::get_current(); 
  303.           $wpcf7->skip_mail = true; 
  304.           $posted_data = array_merge($prev_data, $cf7_posted_data); 
  305.           cf7msm_set('cf7msm_posted_data', $posted_data ); 
  306.       } 
  307.       else { 
  308.           $cf7_posted_data = array_merge($prev_data, $cf7_posted_data); 
  309.       } 
  310.   } 
  311.   return $cf7_posted_data; 
  312. } 
  313. add_filter( 'wpcf7_posted_data', 'cf7msm_store_data_steps_filter', 9 );

  318. function cf7msm_set_step( $result, $tags ) { 
  319.   if ( !empty( $_POST['step'] ) ) { 
  320.       $step = $_POST['step']; 
  321.       if (preg_match('/(\d+)-(\d+)/', $step, $matches)) { 
  322.           $curr_step = $matches[1]; 
  323.           $last_step = $matches[2]; 
  324.           if ( $result->is_valid() ) { 
  325.               if ($curr_step != $last_step) { 
  326.                   cf7msm_set('step', $curr_step); 
  327.               } 
  328.           } 
  329.           else { 
  330.               $stored_step = cf7msm_get('step'); 
  331.               if ( $stored_step >= $curr_step ) { 
  332.                   //reduce it so user cannot move onto next step. 
  333.                   cf7msm_set('step', intval( $curr_step ) - 1); 
  334.               } 
  335.           } 
  336.       } 
  337.   } 
  338.   return $result; 
  339. }

  354.           cf7msm_remove('cf7msm_posted_data'); 
  355.           cf7msm_remove('cf7msm_prev_urls'); 
  356.       } 
  357.       else { 
  358.           $wpcf7 = WPCF7_ContactForm::get_current(); 
  359.           $formstring = $wpcf7->prop('form'); 
  360.           // redirect when ajax is disabled. 
  361.           if ( !wpcf7_load_js() ) { 
  362.               //get url from saved form, not $_POST.  be safe. 
  363.               $redirect_url = parse_form_for_multistep_url( $wpcf7 ); 
  364.               if ( empty( $redirect_url ) ) { 
  365.                   // if using old additional_settings way 
  366.                   $subject = $wpcf7->prop('additional_settings'); 
  367.                   $pattern = '/location\.replace\(\'([^\']*)\'\);/'; 
  368.                   preg_match( $pattern, $subject, $matches ); 
  369.                   if ( count( $matches ) == 2 ) { 
  370.                       $redirect_url = $matches[1]; 
  371.                   } 
  372.               } 
  373.               $redirect_url = apply_filters( 'cf7msm_redirect_url', $redirect_url, $wpcf7->id() ); 
  374.               if ( !empty( $redirect_url ) ) { 
  375.                   wp_redirect( esc_url( $redirect_url ) ); 
  376.                   exit; 
  377.               } 
  378.           } 
  379.       } 
  380.   } 
  381. }
