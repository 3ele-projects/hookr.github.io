  59.   $validation_error = wpcf7_get_validation_error( $tag->name ); 
  60.   $class = wpcf7_form_controls_class( $tag->type, 'cf7msm-multistep' ); 
  61.   if ( $validation_error ) { 
  62.       $class .= ' wpcf7-not-valid'; 
  63.   } 
  64.   $class .= ' cf7msm-multistep'; 
  65.   if ( 'multistep*' === $tag->type ) { 
  66.       $class .= ' wpcf7-validates-as-required'; 
  67.   } 
  68.   $value = (string) reset( $tag->values ); 
  69.  
  70.   $multistep_values = cf7msm_format_multistep_value( $value ); 
  71.   $step_value = $multistep_values['curr_step'] . '-' . $multistep_values['total_steps']; 
  72.  
  73.   $atts = array(

  82.   //populate rest of form in hidden tags. 
  83.   $submission = WPCF7_Submission::get_instance(); 
  84.  
  85.   //get all posted data 
  86.   foreach ($_POST as $name => $value) { 
  87.       //add hidden elements for any not in current form. 
  88.  
  89.       //if multistep posted value is greater than current step, populate elements. 
  90.  
  91.       //print hidden elements 
  92.   } 
  93.  
  94.   //$wpcf7 = WPCF7_ContactForm::get_current(); 
  95.  
  96.   return $html;

  171. /** 
  172.  * Return the step value and next url in an array.  URL may be empty. 
  173.  */ 
  174. function cf7msm_format_multistep_value( $valueString ) { 
  175.   $no_url = false; 
  176.   $next_url = ''; 
  177.  
  178.   $i = stripos( $valueString, '-' ); 
  179.   $curr_step = substr( $valueString, 0, $i ); 
  180.   $j = stripos( $valueString, '-', $i+1 ); 
  181.   if ( $j === FALSE ) { 
  182.       $j = strlen( $valueString ); 
  183.       $no_url = true; 
  184.   } 
  185.   $total_steps = substr( $valueString, $i+1, $j-($i+1) ); 
  186.   if ( !$no_url ) { 
  187.       $next_url = substr( $valueString, $j+1 ); 
  188.   } 
  189.  
  190.   return array( 
  191.       'curr_step' => $curr_step,  
  192.       'total_steps' => $total_steps,  
  193.       'next_url' => $next_url 
  194.  ); 
  195. }
