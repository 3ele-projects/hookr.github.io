  142.                           'profile_data' =>    $data[2]

  144.                       array( 
  145.                           '%d', '%s', '%s', '%s' 
  146.  ) 
  147.  ) ) { 
  148.               gde_dx_log("Profile $id creation failed"); 
  149.               return 0; 
  150.           } else { 
  151.               gde_dx_log("Profile $id created"); 
  152.               return 1; 
  153.           }

  170.           } 
  171.            
  172.           if ( ! empty( $data[2] ) && ( $data[2] !== serialize( $olddata ) ) ) { 
  173.               // overwrite data 
  174.               $newdata['profile_data'] = $data[2]; 
  175.           } 
  176.            
  177.           if ( isset( $newdata ) ) { 
  178.               if ( ! $wpdb->update( 
  179.                           $table,  
  180.                           $newdata,  
  181.                           array( 'profile_id' => $id ),  
  182.                           array( 
  183.                               '%s', '%s', '%s' 
  184.  ) 
  185.  ) ) {

  192.               }

  667. /** 
  668.  * Add upload support for natively unsupported mimetypes used by this plugin 
  669.  * 
  670.  * @since   2.4.0.1 
  671.  * @return  array Updated array of allowed upload types 
  672.  */ 
  673. function gde_upload_mimes( $existing_mimes = array() ) { 
  674.   $supported_exts = gde_supported_types(); 
  675.    
  676.   foreach ( $supported_exts as $ext => $mimetype ) { 
  677.       if ( ! array_key_exists( $ext, gde_mimes_expanded( $existing_mimes ) ) ) { 
  678.           $existing_mimes[$ext] = $mimetype; 
  679.       } 
  680.   } 
  681.   return gde_mimes_collapsed( $existing_mimes ); 
  682. } 
  683.  
  684. function gde_mimes_expanded( array $types ) { 
  685.   // expand the supported mime types so that every ext is its own key 
  686.   foreach ( $types as $k => $v ) { 
  687.       if ( strpos( "|", $k ) ) { 
  688.           $subtypes = explode( "|", $k ); 
  689.           foreach ( $subtypes as $type ) { 
  690.               $newtypes[$type] = $v; 
  691.               unset( $types[$k] ); 
  692.           } 
  693.           $types = array_merge( $types, $newtypes ); 
  694.       } 
  695.   } 
  696.   return $types; 
  697. } 
  698.  
  699. function gde_mimes_collapsed( $types ) { 
  700.   // collapes the supported mime types so that each mime is listed once with combined key (default) 
  701.   $newtypes = array(); 
  702.    
  703.   foreach ( $types as $k => $v ) { 
  704.       if ( isset( $newtypes[$v] ) ) { 
  705.           $newtypes[$v] .= '|' . $k; 
  706.       } else { 
  707.           $newtypes[$v] = $k; 
  708.       } 
  709.   } 
  710.   return array_flip( $newtypes ); 
  711. }

  847. /** 
  848.  * Check to see if a beta is available (generally or to this install's API key) 
  849.  * 
  850.  * @since   2.5.0.1 
  851.  * @return  bool Whether or not a new beta is available 
  852.  */ 
  853. function gde_beta_available() { 
  854.   global $gdeoptions, $pdata; 
  855.    
  856.   $key = 'gde_beta_version'; 
  857.    
  858.   if ( $avail_version = get_site_transient( $key ) ) { 
  859.       // transient already set - compare versions 
  860.       if ( version_compare( $pdata['Version'], $avail_version, '>=' ) ) { 
  861.           // installed version is same or newer, don't do anything 
  862.           return false; 
  863.       } else { 
  864.           // transient is newer, get beta info (no version check necessary) 
  865.           return true; 
  866.       } 
  867.   } else { 
  868.       // beta status unknown - attempt to fetch 
  869.       $api_url = GDE_BETA_API . "versions/gde"; 
  870.        
  871.       if ( ! empty ( $gdeoptions['api_key'] ) ) { 
  872.           $api_url .= '?api_key=' . $gdeoptions['api_key']; 
  873.       } 
  874.        
  875.       gde_dx_log("Performing remote beta check"); 
  876.       $response = wp_remote_get( $api_url ); 
  877.        
  878.       // set checking interval lower if currently running a beta 
  879.       if ( gde_is_beta() ) { 
  880.           $hours = 3; 
  881.       } else { 
  882.           $hours = 24; 
  883.       } 
  884.        
  885.       if ( ! is_wp_error( $response ) ) { 
  886.           if ( $json = json_decode( wp_remote_retrieve_body( $response ) ) ) { 
  887.               if ( isset( $json->beta->version ) ) { 
  888.                   $ver = $json->beta->version; 
  889.                   gde_dx_log("Beta detected: ".$ver); 
  890.               } 
  891.               if ( ! empty( $ver ) ) { 
  892.                   gde_dx_log("Beta detected, don't check again for $hours hours"); 
  893.                   set_site_transient( $key, $ver, 60*60*$hours ); 
  894.                    
  895.                   // there is a beta available, let the checker decide if it's relevant 
  896.                   return true; 
  897.               } else { 
  898.                   // no beta available - don't check again for x hours 
  899.                   gde_dx_log("No beta detected, check again in $hours hours"); 
  900.                   set_site_transient( $key, $pdata['Version'], 60*60*24 ); 
  901.                   return false; 
  902.               } 
  903.           } 
  904.       } 
  905.        
  906.       // otherwise (in case of retrieve failure) 
  907.       return false; 
  908.   }     
  909. }
