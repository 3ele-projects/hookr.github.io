  115.       // validation skipped due to service failure

  118.       if ( is_array( $result ) ) { 
  119.           return $result; 
  120.       } else {

  181. function gde_valid_url( $url, $method = "head", $stop = 0 ) { 
  182.  
  183.   if ( $method == "head" ) { 
  184.       $result = wp_remote_head( $url ); 
  185.   } elseif ( $stop == 0 ) { 
  186.       $stop++; 
  187.       $result = wp_remote_get( $url ); 
  188.   } else { 
  189.       gde_dx_log("can't get URL to test; skipping"); 
  190.       return false; 
  191.   } 
  192.    
  193.   if ( is_array( $result ) ) { 
  194.       $code = $result['response']['code']; 
  195.       if ( ! empty( $code ) && ( $code == "301" || $code == "302" ) ) { 
  196.           // HEAD requests don't redirect. Probably a file/directory with spaces in it... 
  197.           return gde_valid_url( $url, 'get', $stop ); 
  198.       } else { 
  199.           // capture file size if determined 
  200.           if ( isset( $result['headers']['content-length'] ) ) { 
  201.               $result['response']['fsize'] = $result['headers']['content-length']; 
  202.           } else { 
  203.               $result['response']['fsize'] = ''; 
  204.           } 
  205.           return $result['response']; 
  206.       } 
  207.   } elseif ( is_wp_error( $result ) ) { 
  208.       // unable to get head 
  209.       $error = $result->get_error_message(); 
  210.       gde_dx_log("bypassing URL check; cant validate URL $url: $error"); 
  211.       return false; 
  212.   } else { 
  213.       gde_dx_log("cant determine URL validity; skipping");

  229. function gde_format_bytes( $bytes, $precision = 2 ) { 
  230.   if ( ! is_numeric( $bytes ) || $bytes < 1 ) { 
  231.       return __('Unknown', 'google-document-embedder'); 
  232.   } else { 
  233.       $units = array( 'B', 'KB', __('MB', 'google-document-embedder'), 'GB', 'TB' ); 
  234.        
  235.       $bytes = max( $bytes, 0 ); 
  236.       $pow = floor( ( $bytes ? log( $bytes ) : 0 ) / log( 1024 ) ); 
  237.       $pow = min( $pow, count( $units ) - 1 ); 
  238.        
  239.       $bytes /= pow( 1024, $pow ); 
  240.        
  241.       if ( $units[$pow] == "KB" ) { 
  242.           // less precision for small file sizes 
  243.           return round( $bytes ) . "KB"; 
  244.       } else { 
  245.           return round( $bytes, $precision ) . $units[$pow]; 
  246.       } 
  247.   } 
  248. }
