  246.     private function populate_orders() {             
  247.         $this->orders = array(); 
  248.          
  249.         // Get the orders 
  250.         $args = array( 
  251.             'posts_per_page' => -1,  
  252.             'post_type' => 'shop_order',  
  253.             'post_status' => 'publish',  
  254.             'post__in' => $this->order_ids,  
  255.             'orderby' => 'post__in' 
  256.  ); 
  257.         $posts = get_posts( $args ); 
  258.          
  259.         // All orders should exist 
  260.         if( count( $posts ) !== count( $this->order_ids ) ) { 
  261.             $this->orders = null; 
  262.             return false; 
  263.         } 
  264.          
  265.         // Check permissons of the user to determine  
  266.         // if the orders should be populated. 
  267.         foreach( $posts as $post ) { 
  268.             $order = new WC_Order( $post->ID ); 
  269.              
  270.             // Logged in users             
  271.             if( is_user_logged_in() && ( !current_user_can( 'edit_shop_orders' ) && !current_user_can( 'view_order', $order->id ) ) ) { 
  272.                 $this->orders = null; 
  273.                 return false; 
  274.             }  
  275.  
  276.             // An email is required for not logged in users   
  277.             if( !is_user_logged_in() && ( empty( $this->order_email ) || strtolower( $order->billing_email ) != $this->order_email ) ) { 
  278.                 $this->orders = null; 
  279.                 return false; 
  280.             } 
  281.              
  282.             // Save the order to get it without an additional database call 
  283.             $this->orders[$post->ID] = $order; 
  284.         } 
  285.         return true; 
  286.     }
