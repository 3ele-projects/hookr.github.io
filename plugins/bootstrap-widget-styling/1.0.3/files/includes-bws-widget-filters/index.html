  28.   $widget_key = 'disable_' . $type_of_widget . '_widget' ; 
  29.   if ( ( isset( $options[ $widget_key ] ) ) && ( '1' === $options[ $widget_key ] ) ) { 
  30.       return false ; 
  31.   } 
  32.   return true ; 
  33. }

  39.   } 
  40.   else {

  42.   } 
  43. }
