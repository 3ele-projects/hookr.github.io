  376.               -----------------------------------------------------

  494.                       $search_items = buddyboos_global_search_items(); 
  495.                        
  496.                       foreach( $ordered_items_group as $type=>&$items ) { 
  497.                           //now prepend html (opening tags) to first item of each 
  498.  
  499.                           $label = isset( $search_items[$type] ) ? $search_items[$type] : $type; 
  500.                           $first_item = reset($items); 
  501.                           $start_html = "<div class='results-group results-group-{$type} ". apply_filters( 'bboss_global_search_class_search_wrap', 'bboss-results-wrap', $label ) ."'>" 
  502.                                   .    "<h2 class='results-group-title'><span>" . apply_filters( 'bboss_global_search_label_search_type', $label ) . "</span></h2>" 
  503.                                   .    "<ul id='{$type}-stream' class='item-list {$type}-list ". apply_filters( 'bboss_global_search_class_search_list', 'bboss-results-list', $label ) ."'>"; 
  504.                                    
  505.                           $group_start_html = apply_filters( "bboss_global_search_results_group_start_html", $start_html, $type ); 
  506.                            
  507.                           $first_item['html'] = $group_start_html . $first_item['html']; 
  508.                           $items[$first_item['id']] = $first_item; 
  509.                            
  510.                           //and append html (closing tags) to last item of each type 
  511.                           $last_item = end($items); 
  512.                           $end_html = "</ul></div>"; 
  513.                                    
  514.                           $group_end_html = apply_filters( "bboss_global_search_results_group_end_html", $end_html, $type ); 
  515.                            
  516.                           $last_item['html'] = $last_item['html'] . $group_end_html; 
  517.                           $items[$last_item['id']] = $last_item; 
  518.                       } 
  519.                        
  520.                       //replace orginal items with this new, grouped set of items 
  521.                       $this->search_results['all']['items'] = array(); 
  522.                       foreach( $ordered_items_group as $type=>$grouped_items ) { 
  523.                           foreach( $grouped_items as $item_id=>$item ) { 
  524.                               $this->search_results['all']['items'][$item_id] = $item; 
  525.                           } 
  526.                       } 
  527.                       /** ________________________________ */ 
  528.                   } 
  529.               } 
  530.           } else { 
  531.               //if subset not in searchable items, bail out. 
  532.               if( !in_array( $args['search_subset'], $args['searchable_items'] ) ) 
  533.                   return; 
  534.                
  535.               if( !isset($this->search_helpers[$args['search_subset']])) 
  536.                   return; 
  537.                
  538.               /** 
  539.                * 1. Search top top 20( $args['per_page'] ) item( posts|members|..) 
  540.                * 2. Generate html for each of them 
  541.                */ 
  542.                
  543.               $obj = $this->search_helpers[$args['search_subset']]; 
  544.               $pre_search_query = $obj->union_sql( $args['search_term'] ) . " ORDER BY relevance DESC, entry_date DESC "; 
  545.                
  546.               if( $args['per_page']> 0 ) { 
  547.                   $offset = ( $args['current_page'] * $args['per_page'] ) - $args['per_page']; 
  548.                   $pre_search_query .= " LIMIT {$offset}, {$args['per_page']} "; 
  549.               } 
  550.                
  551.               $results = $wpdb->get_results( $pre_search_query ); 
  552.                
  553.               /** $results will have a structure like below */ 
  554.               /** 
  555.               id | type | relevance | entry_date 
  556.               45 | groups | 1 | 2014-10-28 17:05:18 
  557.               40 | posts | 1 | 2014-10-26 13:52:06 
  558.               4 | groups | 0 | 2014-10-21 15:15:36 
  559.               */ 
  560.               if( !empty( $results ) ) { 
  561.                   $obj = $this->search_helpers[$args['search_subset']]; 
  562.                   $this->search_results[$args['search_subset']] = array( 'total_match_count' => 0, 'items' => array() ); 
  563.                   //segregate items of a type together and pass it to corresponsing search handler, so that an aggregate query can be done 
  564.                   //e.g one single wordpress loop can be done for all posts 
  565.                   foreach( $results as $item ) { 
  566.                       $obj->add_search_item( $item->id ); 
  567.                   } 
  568.                    
  569.                   //now get html for each item 
  570.                   foreach( $results as $item ) { 
  571.                       $html = $obj->get_html( $item->id, $args['template_type'] ); 
  572.                        
  573.                       $result = array( 
  574.                           'id' => $item->id,  
  575.                           'type' => $args['search_subset'],  
  576.                           'html' => $obj->get_html( $item->id, $args['template_type'] ),  
  577.                           'title' => $obj->get_title( $item->id ),  
  578.  ); 
  579.                        
  580.                       $this->search_results[$args['search_subset']]['items'][$item->id] = $result; 
  581.                   } 
  582.                    
  583.                   //now prepend html (opening tags) to first item of each type 
  584.                   $first_item = reset($this->search_results[$args['search_subset']]['items']); 
  585.                   $start_html = "<div class='results-group results-group-{$args['search_subset']} ". apply_filters( 'bboss_global_search_class_search_wrap', 'bboss-results-wrap', $args['search_subset'] ) ."'>" 
  586.                           .    "<ul id='{$args['search_subset']}-stream' class='item-list {$args['search_subset']}-list ". apply_filters( 'bboss_global_search_class_search_list', 'bboss-results-list', $args['search_subset'] ) ."'>"; 
  587.  
  588.                   $group_start_html = apply_filters( "bboss_global_search_results_group_start_html", $start_html, $args['search_subset'] ); 
  589.  
  590.                   $first_item['html'] = $group_start_html . $first_item['html']; 
  591.                   $this->search_results[$args['search_subset']]['items'][$first_item['id']] = $first_item; 
  592.  
  593.                   //and append html (closing tags) to last item of each type 
  594.                   $last_item = end($this->search_results[$args['search_subset']]['items']); 
  595.                   $end_html = "</ul></div>"; 
  596.  
  597.                   $group_end_html = apply_filters( "bboss_global_search_results_group_end_html", $end_html, $args['search_subset'] ); 
  598.  
  599.                   $last_item['html'] = $last_item['html'] . $group_end_html; 
  600.                   $this->search_results[$args['search_subset']]['items'][$last_item['id']] = $last_item; 
  601.               } 
  602.           } 
  603.            
  604.           //html for search results is generated. 
  605.           //now, lets calculate the total number of search results, for all different types 
  606.           if( $args['count_total'] ) { 
  607.               $all_items_count = 0; 
  608.               foreach( $args['searchable_items'] as $search_type ) { 
  609.                   if( !isset($this->search_helpers[$search_type])) 
  610.                       continue; 
  611.  
  612.                   $obj = $this->search_helpers[$search_type]; 
  613.                   $total_match_count = $obj->get_total_match_count( $this->search_args['search_term'] ); 
  614.                   $this->search_results[$search_type]['total_match_count'] = $total_match_count; 
  615.                    
  616.                   $all_items_count += $total_match_count; 
  617.               } 
  618.  
  619.               $this->search_results['all']['total_match_count'] = $all_items_count; 
  620.           } 
  621.       }
