  110.  * Find a certain word in a string, and then wrap around it

  116. function buddyboss_global_search_result_match( $in, $wordToFind, $numWordsToWrap = 10 ) { 
  117.  
  118.   $words = preg_split('/\s+/', $in); 
  119.   $wordsToFind = preg_split('/\s+/', $wordToFind); 
  120.  
  121.   foreach ( $wordsToFind as $key => $value ) { 
  122.       $found_words =   preg_grep("/".$value.".*/i", $words); 
  123.       $found_pos =   array_keys($found_words); 
  124.  
  125.       if( count( $found_pos ) ) { 
  126.           $pos = $found_pos[0]; 
  127.           break; 
  128.       } 
  129.   } 
  130.  
  131.   if (isset($pos)) { 
  132.  
  133.       $start = ($pos - $numWordsToWrap > 0) ? $pos - $numWordsToWrap : 0; 
  134.       $length = (($pos + ($numWordsToWrap + 1) < count($words)) ? $pos + ($numWordsToWrap + 1) : count($words)) - $start; 
  135.       $slice = array_slice($words, $start, $length); 
  136.  
  137.       $pre_start =   ($start > 0) ? "...":""; 
  138.  
  139.       $post_end =   ($pos + ($numWordsToWrap + 1) < count($words)) ? "...":""; 
  140.  
  141.       $out = $pre_start.implode(' ', $slice).$post_end; 
  142.       return $out; 
  143.   } 
  144.  
  145.   return $in;

  169.   function emi_generate_paging_param($total_items, $items_per_page, $curr_paged, $slug, $links_on_each_side = 2, $hashlink = "", $param_key = "list") {

  178.       } 
  179.       /** $elements : an array of arrays; each child array will have following structure

  183.        */

  188.           $elements[] = array('←', $curr_paged - 1, 'link'); 
  189.       } 
  190.       //generating $s(2) links before the current one 
  191.       if ($curr_paged > 1) { 
  192.           $rev_array = array(); //paged in reverse order 
  193.           $i = $curr_paged - 1; 
  194.           $counter = 0;

  203.           }

  205.               $elements[] = array($el, $el, 'link'); 
  206.           }
