  609. /** 
  610.  * Filter our user count to take into account spam members. 
  611.  * 
  612.  * @param string $count Total active users. 
  613.  * @return mixed|null|string 
  614.  */ 
  615. function bp_registration_options_remove_moderated_count( $count ) { 
  616.  
  617.   $pending_count = bp_registration_get_pending_user_count(); 
  618.  
  619.   if ( '0' === $pending_count ) { 
  620.       return $count; 
  621.   } 
  622.  
  623.   global $wpdb; 
  624.  
  625.   $total_count = get_transient( 'bpro_total_user_count' ); 
  626.  
  627.   if ( false === $total_count ) { 
  628.  
  629.       $status_sql = 'user_status = 0'; 
  630.  
  631.       if ( is_multisite() ) { 
  632.           $status_sql = 'spam = 0 AND deleted = 0 AND user_status = 0'; 
  633.       } 
  634.  
  635.       $total_count = $wpdb->get_var( "SELECT COUNT(ID) FROM {$wpdb->users} WHERE {$status_sql}" ); 
  636.  
  637.       set_transient( 'bpro_total_user_count', $total_count, HOUR_IN_SECONDS ); 
  638.   } 
  639.  
  640.   $final_count = ( $total_count - $pending_count ); 
  641.  
  642.   return ( $final_count > 0 ) ? $final_count : $count; 
  643.  
  644. } 
  645. add_filter( 'bp_get_total_member_count', 'bp_registration_options_remove_moderated_count' );
