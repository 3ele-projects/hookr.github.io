  371.     function split_params( $response ) { 
  372.         $params = array(); 
  373.         $param_pairs = explode( '&', $response ); 
  374.         foreach ( $param_pairs as $param_pair ) { 
  375.             if ( trim( $param_pair ) == '' ) { 
  376.                 continue; 
  377.             } 
  378.             list( $key, $value ) = explode( '=', $param_pair ); 
  379.             $params[$key] = urldecode( $value ); 
  380.         } 
  381.  
  382.         return $params;
