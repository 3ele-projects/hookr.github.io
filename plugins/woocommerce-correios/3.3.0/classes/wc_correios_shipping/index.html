  424.      * Calculates the shipping rate.

  428.     public function calculate_shipping( $package = array() ) { 
  429.         // Check if valid to be calculeted. 
  430.         if ( '' === $package['destination']['postcode'] || 'BR' !== $package['destination']['country'] ) { 
  431.             return; 
  432.         } 
  433.  
  434.         // Check for shipping classes. 
  435.         if ( ! $this->has_only_selected_shipping_class( $package ) ) { 
  436.             return; 
  437.         } 
  438.  
  439.         $shipping = $this->get_rate( $package ); 
  440.  
  441.         if ( ! isset( $shipping->Erro ) ) { 
  442.             return; 
  443.         } 
  444.  
  445.         $error_number = (string) $shipping->Erro; 
  446.  
  447.         // Exit if have errors. 
  448.         if ( ! in_array( $error_number, $this->get_accepted_error_codes(), true ) ) { 
  449.             return; 
  450.         } 
  451.  
  452.         // Display Correios errors. 
  453.         $error_message = wc_correios_get_error_message( $error_number ); 
  454.         if ( '' !== $error_message && is_cart() ) { 
  455.             $notice_type = ( '010' === $error_number ) ? 'notice' : 'error'; 
  456.             $notice = '<strong>' . $this->title . ':</strong> ' . esc_html( $error_message ); 
  457.             wc_add_notice( $notice, $notice_type ); 
  458.         } 
  459.  
  460.         // Set the shipping rates. 
  461.         $label = $this->get_shipping_method_label( (int) $shipping->PrazoEntrega, $package ); 
  462.         $cost = wc_correios_normalize_price( esc_attr( (string) $shipping->Valor ) ); 
  463.  
  464.         // Exit if don't have price. 
  465.         if ( 0 === intval( $cost ) ) { 
  466.             return; 
  467.         } 
  468.  
  469.         // Apply fees. 
  470.         $fee = $this->get_fee( $this->fee, $cost ); 
  471.  
  472.         // Create the rate and apply filters. 
  473.         $rate = apply_filters( 'woocommerce_correios_' . $this->id . '_rate', array( 
  474.             'id' => $this->id . $this->instance_id,  
  475.             'label' => $label,  
  476.             'cost' => (float) $cost + (float) $fee,  
  477.  ), $this->instance_id, $package ); 
  478.  
  479.         // Deprecated filter. 
  480.         $rates = apply_filters( 'woocommerce_correios_shipping_methods', array( $rate ), $package ); 
  481.  
  482.         // Add rate to WooCommerce. 
  483.         $this->add_rate( $rates[0] ); 
  484.     }
