  327.    * bp_group_documents_check_ext() 
  328.    * 
  329.    * checks whether the passed filename ends in an extension 
  330.    * that is allowed by the site admin 
  331.    */ 
  332.   function bp_group_documents_check_ext($filename) { 
  333.  
  334.       if ( !$filename ) { 
  335.           return false; 
  336.       } 
  337.  
  338.       $valid_formats_string = get_option('bp_group_documents_valid_file_formats'); 
  339.       $valid_formats_array = explode(', ' , $valid_formats_string); 
  340.  
  341.       $extension = substr($filename , (strrpos($filename , ".") + 1)); 
  342.       $extension = strtolower($extension); 
  343.  
  344.       if ( in_array($extension , $valid_formats_array) ) { 
  345.           return true;

  350.   /** 
  351.    * get_file_size() 
  352.    * 
  353.    * returns a human-readable file-size for the passed file 
  354.    * adapted from a function in the PHP manual comments 
  355.    */ 
  356.   function get_file_size($document , $precision = 1) { 
  357.  
  358.       $units = array('b' , 'k' , 'm' , 'g'); 
  359.  
  360.       $bytes = file_exists($document->get_path(1)) ? filesize($document->get_path(1)) : 0; 
  361.       $bytes = max($bytes , 0); 
  362.       $pow = floor(($bytes ? log($bytes) : 0) / log(1024)); 
  363.       $pow = min($pow , count($units) - 1); 
  364.  
  365.       $bytes /= pow(1024 , $pow); 
  366.  
  367.       return round($bytes , $precision) . $units[$pow]; 
  368.   }

  370.   /** 
  371.    * return_bytes() 
  372.    * 
  373.    * taken from the PHP manual examples.  Returns the number of bites 
  374.    * when given an abrevition (eg, max_upload_size) 
  375.    */ 
  376.   function return_bytes($val) { 
  377.       $val = trim($val); 
  378.       $last = strtolower($val[strlen($val) - 1]); 
  379.       switch ($last) { 
  380.           // The 'G' modifier is available since PHP 5.1.0 
  381.           case 'g': 
  382.               $val *= 1024; 
  383.           case 'm': 
  384.               $val *= 1024; 
  385.           case 'k': 
  386.               $val *= 1024; 
  387.       } 
  388.  
  389.       return $val; 
  390.   }
