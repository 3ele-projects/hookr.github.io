  440.                 /**

  447.                 $transient_expiration = apply_filters( 'bfa_api_transient_expiration', 12 * HOUR_IN_SECONDS );

  619.             // Log the CSS fetch error.

  621.                 $this->set_error( 'css', $response->get_error_code(), $response->get_error_message() . " (URL: $url)" ); 
  622.             } 
  623.  
  624.             // Use the local fallback CSS. 
  625.             $response = $this->fallback_data['css']; 
  626.  
  627.             // Update the version string to match the fallback version. 
  628.             $this->font_awesome_version = $this->fallback_data['version']; 
  629.  
  630.             // Update the stylesheet URL to match the fallback version. 
  631.             $this->stylesheet_url = $this->fallback_data['url']; 
  632.         } 
  633.  
  634.         return $response;

  654.     /** 
  655.      * Get the CSS from the remote jsDelivr CDN. 
  656.      * 
  657.      * @since   1.0.0 
  658.      * 
  659.      * @param   string           $url       URL for the remote stylesheet. 
  660.      * @param   string           $version   Font Awesome version to fetch. 
  661.      * 
  662.      * @return  string|WP_ERROR  $response  Remote CSS, or WP_ERROR if 
  663.      *                                      wp_remote_get() fails. 
  664.      */ 
  665.     private function get_remote_css( $url, $version ) { 
  666.  
  667.         // Get the remote Font Awesome CSS. 
  668.         $url = set_url_scheme( $url ); 
  669.         $response = wp_remote_get( $url, $this->wp_remote_get_args ); 
  670.  
  671.         /** 
  672.          * If fetch was successful, return the remote CSS, and set the CSS 
  673.          * transient for this version. Otherwise, return a WP_Error object. 
  674.          */ 
  675.         if ( 200 == wp_remote_retrieve_response_code( $response ) ) { 
  676.  
  677.             $response = wp_remote_retrieve_body( $response ); 
  678.             $this->set_css_transient( $version, $response ); 
  679.  
  680.         } elseif ( is_wp_error( $response ) ) { // Check for faulty wp_remote_get() 
  681.             $response = $response; 
  682.         } elseif ( isset( $response['response'] ) ) { // Check for 404 and other non-WP_ERROR codes 
  683.             $response = new WP_Error( $response['response']['code'], $response['response']['message'] . " (URL: $url)" ); 
  684.         } else { // Total failsafe 
  685.             $response = ''; 
  686.         } 
  687.  
  688.         return $response; 
  689.  
  690.     }

  737.     /**

  746.     private function setup_icon_array( $css ) { 
  747.  
  748.         $icons = array(); 
  749.         $hex_codes = array();

  849.     /** 
  850.      * Render [icon] shortcode. 
  851.      * 
  852.      * Usage: 
  853.      * [icon name="flag" class="fw 2x spin" unprefixed_class="custom_class"] 
  854.      * 
  855.      * @param   array   $atts    Shortcode attributes. 
  856.      * @return  string  $output  Icon HTML (e.g. <i class="fa fa-car"></i>). 
  857.      */ 
  858.     public function render_shortcode( $atts ) { 
  859.  
  860.         extract( shortcode_atts( array( 
  861.                     'name' => '',  
  862.                     'class' => '',  
  863.                     'unprefixed_class' => '',  
  864.                     'title' => '', /** For compatibility with other plugins */ 
  865.                     'size' => '', /** For compatibility with other plugins */ 
  866.                     'space' => '',  
  867.  ), $atts ) 
  868.  ); 
  869.  
  870.         /** 
  871.          * Include for backwards compatibility with Font Awesome More Icons plugin. 
  872.          * 
  873.          * @see https://wordpress.org/plugins/font-awesome-more-icons/ 
  874.          */ 
  875.         $title = $title ? 'title="' . $title . '" ' : ''; 
  876.         $space = 'true' == $space ? ' ' : ''; 
  877.         $size = $size ? ' '. $this->prefix . '-' . $size : ''; 
  878.  
  879.         // Remove "icon-" and "fa-" from name 
  880.         // This helps both: 
  881.         //  1. Incorrect shortcodes (when user includes full class name including prefix) 
  882.         //  2. Old shortcodes from other plugins that required prefixes 
  883.  
  884.         /** 
  885.          * Strip 'icon-' and 'fa-' from the BEGINNING of $name. 
  886.          * 
  887.          * This corrects for: 
  888.          *     1. Incorrect shortcodes (when user includes full class name including prefix) 
  889.          *  2. Old shortcodes from other plugins that required prefixes 
  890.          */ 
  891.         $prefixes = array( 'icon-', 'fa-' ); 
  892.         foreach ( $prefixes as $prefix ) { 
  893.  
  894.             if ( substr( $name, 0, strlen( $prefix ) ) == $prefix ) { 
  895.                 $name = substr( $name, strlen( $prefix ) ); 
  896.             } 
  897.  
  898.         } 
  899.  
  900.         $name = str_replace( 'fa-', '', $name ); 
  901.  
  902.         // Add the version-specific prefix back on to $name, if it exists. 
  903.         $icon_name = $this->prefix ? $this->prefix . '-' . $name : $name; 
  904.  
  905.         // Remove "icon-" and "fa-" from the icon class. 
  906.         $class = str_replace( 'icon-', '', $class ); 
  907.         $class = str_replace( 'fa-', '', $class ); 
  908.  
  909.         // Remove extra spaces from the icon class. 
  910.         $class = trim( $class ); 
  911.         $class = preg_replace( '/\s{3, }/', ' ', $class ); 
  912.  
  913.         // Add the version-specific prefix back on to each class. 
  914.         $class_array = explode( ' ', $class ); 
  915.         foreach ( $class_array as $index => $class ) { 
  916.             $class_array[ $index ] = $this->prefix ? $this->prefix . '-' . $class : $class; 
  917.         } 
  918.         $class = implode( ' ', $class_array ); 
  919.  
  920.         // Add unprefixed classes. 
  921.         $class .= $unprefixed_class ? ' ' . $unprefixed_class : ''; 
  922.  
  923.         /** 
  924.          * Filter the icon class. 
  925.          * 
  926.          * @since  1.0.0 
  927.          * 
  928.          * @param  string  $class  Classes attached to the icon. 
  929.          */ 
  930.         $class = apply_filters( 'bfa_icon_class', $class, $name ); 
  931.  
  932.         // Generate the HTML <i> icon element output. 
  933.         $output = sprintf( '<i class="%s %s %s %s" %s>%s</i>',  
  934.             $this->prefix,  
  935.             $icon_name,  
  936.             $class,  
  937.             $size,  
  938.             $title,  
  939.             $space 
  940.  ); 
  941.  
  942.         /** 
  943.          * Filter the icon output. 
  944.          * 
  945.          * @since  1.0.0 
  946.          * 
  947.          * @param  string  $output  Icon output. 
  948.          */ 
  949.         return apply_filters( 'bfa_icon', $output ); 
  950.  
  951.     }
