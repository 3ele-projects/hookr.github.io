  112.      * 
  113.      * @throws \RuntimeException

  115.      */

  117.     { 
  118.         if (!$context instanceof Context) { 
  119.             $context = new Context($context); 
  120.         } 
  121.         $topTree = end($this->_stack); // never pop a value from stack 
  122.         list($index, $tree, $stop) = $topTree; 
  123.  
  124.         $buffer = ''; 
  125.         $rTrim = false; 
  126.         while (array_key_exists($index, $tree)) { 
  127.             $current = $tree[$index]; 
  128.             $index++; 
  129.             //if the section is exactly like waitFor 
  130.             if (is_string($stop) 
  131.                 && $current[Tokenizer::TYPE] == Tokenizer::T_ESCAPED 
  132.                 && $current[Tokenizer::NAME] === $stop 
  133.  ) { 
  134.                 break; 
  135.             } 
  136.             if (isset($current[Tokenizer::TRIM_LEFT]) && $current[Tokenizer::TRIM_LEFT]) { 
  137.                 $buffer = rtrim($buffer); 
  138.             } 
  139.  
  140.             $tmp = $this->_renderInternal($current, $context); 
  141.  
  142.             if (isset($current[Tokenizer::TRIM_LEFT]) && $current[Tokenizer::TRIM_LEFT]) { 
  143.                 $tmp = rtrim($tmp); 
  144.             } 
  145.  
  146.             if ($rTrim  || (isset($current[Tokenizer::TRIM_RIGHT]) && $current[Tokenizer::TRIM_RIGHT])) { 
  147.                 $tmp = ltrim($tmp); 
  148.             } 
  149.  
  150.             $buffer .= $tmp; 
  151.             // Some time, there is more than one string token (first is empty),  
  152.             //so we need to trim all of them in one shot 
  153.  
  154.             $rTrim = (empty($tmp) && $rTrim) || 
  155.                 isset($current[Tokenizer::TRIM_RIGHT]) && $current[Tokenizer::TRIM_RIGHT]; 
  156.         } 
  157.         if ($stop) { 
  158.             //Ok break here, the helper should be aware of this. 
  159.             $newStack = array_pop($this->_stack); 
  160.             $newStack[0] = $index; 
  161.             $newStack[2] = false; //No stop token from now on 
  162.             array_push($this->_stack, $newStack); 
  163.         } 
  164.  
  165.         return $buffer; 
  166.     }

  168.     /** 
  169.      * Render tokens base on type of tokens 
  170.      * 
  171.      * @param array $current current token 
  172.      * @param mixed $context current context 
  173.      * 
  174.      * @return string 
  175.      */ 
  176.     private function _renderInternal($current, $context) 
  177.     { 
  178.         $result = ''; 
  179.         switch ($current[Tokenizer::TYPE]) { 
  180.         case Tokenizer::T_END_SECTION: 
  181.             break; // Its here just for handling whitespace trim. 
  182.         case Tokenizer::T_SECTION : 
  183.             $newStack = isset($current[Tokenizer::NODES]) 
  184.                 ? $current[Tokenizer::NODES] : array(); 
  185.             array_push($this->_stack, array(0, $newStack, false)); 
  186.             $result = $this->_section($context, $current); 
  187.             array_pop($this->_stack); 
  188.             break; 
  189.         case Tokenizer::T_INVERTED : 
  190.             $newStack = isset($current[Tokenizer::NODES]) ? 
  191.                 $current[Tokenizer::NODES] : array(); 
  192.             array_push($this->_stack, array(0, $newStack, false)); 
  193.             $result = $this->_inverted($context, $current); 
  194.             array_pop($this->_stack); 
  195.             break; 
  196.         case Tokenizer::T_COMMENT : 
  197.             $result = ''; 
  198.             break; 
  199.         case Tokenizer::T_PARTIAL: 
  200.         case Tokenizer::T_PARTIAL_2: 
  201.             $result = $this->_partial($context, $current); 
  202.             break; 
  203.         case Tokenizer::T_UNESCAPED: 
  204.         case Tokenizer::T_UNESCAPED_2: 
  205.             $result = $this->_get($context, $current, false); 
  206.             break; 
  207.         case Tokenizer::T_ESCAPED: 
  208.             $result = $this->_get($context, $current, true); 
  209.             break; 
  210.         case Tokenizer::T_TEXT: 
  211.             $result = $current[Tokenizer::VALUE]; 
  212.             break; 
  213.             /** How we could have another type of token? this part of code 
  214.             is not used at all. 
  215.             default: 
  216.                 throw new \RuntimeException( 
  217.                     'Invalid node type : ' . json_encode($current) 
  218.  ); 
  219.             */ 
  220.         } 
  221.  
  222.         return $result; 
  223.     }

  268.     /** 
  269.      * Process handlebars section style 
  270.      * 
  271.      * @param Context $context current context 
  272.      * @param array   $current section node data 
  273.      * 
  274.      * @return mixed|string 
  275.      */ 
  276.     private function _handlebarsStyleSection(Context $context, $current) 
  277.     { 
  278.         $helpers = $this->handlebars->getHelpers(); 
  279.         $sectionName = $current[Tokenizer::NAME]; 
  280.  
  281.         if (isset($current[Tokenizer::END])) { 
  282.             $source = substr( 
  283.                 $this->getSource(),  
  284.                 $current[Tokenizer::INDEX],  
  285.                 $current[Tokenizer::END] - $current[Tokenizer::INDEX] 
  286.  ); 
  287.         } else { 
  288.             $source = ''; 
  289.         } 
  290.  
  291.         // subexpression parsing loop 
  292.         $subexprs = array(); // will contain all subexpressions inside outermost brackets 
  293.         $insideOf = array( 'single' => false, 'double' => false ); 
  294.         $lvl = 0; 
  295.         $cur_start = 0; 
  296.         for ($i=0; $i < strlen($current[Tokenizer::ARGS]); $i++) { 
  297.             $cur = substr($current[Tokenizer::ARGS], $i, 1); 
  298.             if ($cur == "'" ) { 
  299.                 $insideOf['single'] = ! $insideOf['single']; 
  300.             } 
  301.             if ($cur == '"' ) { 
  302.                 $insideOf['double'] = ! $insideOf['double']; 
  303.             } 
  304.             if ($cur == '(' && ! $insideOf['single'] && ! $insideOf['double']) { 
  305.                 if ($lvl == 0) { 
  306.                     $cur_start = $i+1; 
  307.                 } 
  308.                 $lvl++; 
  309.                 continue; 
  310.             } 
  311.             if ($cur == ')' && ! $insideOf['single'] && ! $insideOf['double']) { 
  312.                 $lvl--; 
  313.                 if ($lvl == 0) { 
  314.                     $subexprs[] = substr($current[Tokenizer::ARGS], $cur_start, $i - $cur_start); 
  315.                 } 
  316.  
  317.             } 
  318.         } 
  319.  
  320.         if (! empty($subexprs)) { 
  321.             foreach ($subexprs as $expr) { 
  322.                 $cmd = explode(" ", $expr); 
  323.                 $name = trim($cmd[0]); 
  324.                 // construct artificial section node 
  325.                 $section_node = array( 
  326.                     Tokenizer::TYPE => Tokenizer::T_ESCAPED,  
  327.                     Tokenizer::NAME => $name,  
  328.                     Tokenizer::OTAG => $current[Tokenizer::OTAG],  
  329.                     Tokenizer::CTAG => $current[Tokenizer::CTAG],  
  330.                     Tokenizer::INDEX => $current[Tokenizer::INDEX],  
  331.                     Tokenizer::ARGS => implode(" ", array_slice($cmd, 1)) 
  332.  ); 
  333.                 // resolve the node recursively 
  334.                 $resolved = addcslashes($this->_handlebarsStyleSection($context, $section_node), '"'); 
  335.                 // replace original subexpression with result 
  336.                 $current[Tokenizer::ARGS] = str_replace('('.$expr.')', '"' . $resolved . '"', $current[Tokenizer::ARGS]); 
  337.             } 
  338.         } 
  339.  
  340.         $return = $helpers->call($sectionName, $this, $context, $current[Tokenizer::ARGS], $source); 
  341.  
  342.         if ($return instanceof String) { 
  343.             return $this->handlebars->loadString($return)->render($context); 
  344.         } else { 
  345.             return $return; 
  346.         } 
  347.     }

  349.     /** 
  350.      * Process Mustache section style 
  351.      * 
  352.      * @param Context $context current context 
  353.      * @param array   $current section node data 
  354.      * 
  355.      * @throws \RuntimeException 
  356.      * @return mixed|string 
  357.      */ 
  358.     private function _mustacheStyleSection(Context $context, $current) 
  359.     { 
  360.         $sectionName = $current[Tokenizer::NAME]; 
  361.  
  362.         // fallback to mustache style each/with/for just if there is 
  363.         // no argument at all. 
  364.         try { 
  365.             $sectionVar = $context->get($sectionName, true); 
  366.         } catch (\InvalidArgumentException $e) { 
  367.             throw new \RuntimeException( 
  368.                 $sectionName . ' is not registered as a helper' 
  369.  ); 
  370.         } 
  371.         $buffer = ''; 
  372.         if (is_array($sectionVar) || $sectionVar instanceof \Traversable) { 
  373.             $isList = is_array($sectionVar) && 
  374.                 (array_keys($sectionVar) === range(0, count($sectionVar) - 1)); 
  375.             $index = 0; 
  376.             $lastIndex = $isList ? (count($sectionVar) - 1) : false; 
  377.  
  378.             foreach ($sectionVar as $key => $d) { 
  379.                 $specialVariables = array( 
  380.                     '@index' => $index,  
  381.                     '@first' => ($index === 0),  
  382.                     '@last' => ($index === $lastIndex),  
  383.  ); 
  384.                 if (!$isList) { 
  385.                     $specialVariables['@key'] = $key; 
  386.                 } 
  387.                 $context->pushSpecialVariables($specialVariables); 
  388.                 $context->push($d); 
  389.                 $buffer .= $this->render($context); 
  390.                 $context->pop(); 
  391.                 $context->popSpecialVariables(); 
  392.                 $index++; 
  393.             } 
  394.         } elseif (is_object($sectionVar)) { 
  395.             //Act like with 
  396.             $context->push($sectionVar); 
  397.             $buffer = $this->render($context); 
  398.             $context->pop(); 
  399.         } elseif ($sectionVar) { 
  400.             $buffer = $this->render($context); 
  401.         } 
  402.  
  403.         return $buffer; 
  404.     }
