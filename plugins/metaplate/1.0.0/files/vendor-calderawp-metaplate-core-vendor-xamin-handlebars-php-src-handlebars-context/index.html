  44.   /** 
  45.    * List of charcters that cannot be used in identifiers.

  76.   /**

  85.       }

  88.   /**

  144.   }

  158.   /** 
  159.    * Change the current context to one of current context members 
  160.    * 
  161.    * @param string $variableName name of variable or a callable on current context 
  162.    * 
  163.    * @return mixed actual value 
  164.    */ 
  165.   public function with($variableName) 
  166.   { 
  167.       $value = $this->get($variableName); 
  168.       $this->push($value); 
  169.  
  170.       return $value; 
  171.   } 
  172.  
  173.   /** 
  174.    * Get a available from current context 
  175.    * Supported types : 
  176.    * variable , ../variable , variable.variable , variable.[variable] , . 
  177.    * 
  178.    * @param string  $variableName variable name to get from current context 
  179.    * @param boolean $strict       strict search? if not found then throw exception 
  180.    * 
  181.    * @throws \InvalidArgumentException in strict mode and variable not found 
  182.    * @throws \RuntimeException if supplied argument is a malformed quoted string  
  183.    * @throws \InvalidArgumentException if variable name is invalid 
  184.    * @return mixed 
  185.    */ 
  186.   public function get($variableName, $strict = false) 
  187.   { 
  188.       if ($variableName instanceof \Handlebars\String) { 
  189.           return (string)$variableName; 
  190.       } 
  191.       $variableName = trim($variableName); 
  192.       $level = 0; 
  193.       while (substr($variableName, 0, 3) == '../') { 
  194.           $variableName = trim(substr($variableName, 3)); 
  195.           $level++; 
  196.       } 
  197.       if (count($this->stack) < $level) { 
  198.           if ($strict) { 
  199.               throw new \InvalidArgumentException( 
  200.                   'can not find variable in context' 
  201.  ); 
  202.           } 
  203.  
  204.           return ''; 
  205.       } 
  206.       end($this->stack); 
  207.       while ($level) { 
  208.           prev($this->stack); 
  209.           $level--; 
  210.       } 
  211.       $current = current($this->stack); 
  212.       if (!$variableName) { 
  213.           if ($strict) { 
  214.               throw new \InvalidArgumentException( 
  215.                   'can not find variable in context' 
  216.  ); 
  217.           } 
  218.           return ''; 
  219.       } elseif ($variableName == '.' || $variableName == 'this') { 
  220.           return $current; 
  221.       } elseif ($variableName[0] == '@') { 
  222.           $specialVariables = $this->lastSpecialVariables(); 
  223.           if (isset($specialVariables[$variableName])) { 
  224.               return $specialVariables[$variableName]; 
  225.           } elseif ($strict) { 
  226.               throw new \InvalidArgumentException( 
  227.                   'can not find variable in context' 
  228.  ); 
  229.           } else { 
  230.               return ''; 
  231.           } 
  232.       } else { 
  233.           $chunks = $this->_splitVariableName($variableName); 
  234.           foreach ($chunks as $chunk) { 
  235.               if (is_string($current) and $current == '') { 
  236.                   return $current; 
  237.               } 
  238.               $current = $this->_findVariableInContext($current, $chunk, $strict); 
  239.           } 
  240.       } 
  241.       return $current; 
  242.   } 
  243.  
  244.   /** 
  245.    * Check if $variable->$inside is available 
  246.    * 
  247.    * @param mixed   $variable variable to check 
  248.    * @param string  $inside   property/method to check 
  249.    * @param boolean $strict   strict search? if not found then throw exception 
  250.    * 
  251.    * @throws \InvalidArgumentException in strict mode and variable not found 
  252.    * @return boolean true if exist 
  253.    */ 
  254.   private function _findVariableInContext($variable, $inside, $strict = false) 
  255.   { 
  256.       $value = null; 
  257.       if (($inside !== '0' && empty($inside)) || ($inside == 'this')) { 
  258.           return $variable; 
  259.       } elseif (is_array($variable)) { 
  260.           if (isset($variable[$inside]) || array_key_exists($inside, $variable)) { 
  261.               return $variable[$inside]; 
  262.           } elseif ($inside == "length") { 
  263.               return count($variable); 
  264.           } 
  265.       } elseif (is_object($variable)) { 
  266.           if (isset($variable->$inside)) { 
  267.               return $variable->$inside; 
  268.           } elseif (is_callable(array($variable, $inside))) { 
  269.               return call_user_func(array($variable, $inside)); 
  270.           } 
  271.       } 
  272.  
  273.       if ($strict) { 
  274.           throw new \InvalidArgumentException('can not find variable in context'); 
  275.       } 
  276.  
  277.       return $value; 
  278.   } 
  279.  
  280.   /** 
  281.    * Splits variable name to chunks. 
  282.    * 
  283.    * @param string $variableName Fully qualified name of a variable. 
  284.    * 
  285.    * @throws \InvalidArgumentException if variable name is invalid. 
  286.    * @return array 
  287.    */ 
  288.   private function _splitVariableName($variableName) 
  289.   { 
  290.       $bad_chars = preg_quote(self::NOT_VALID_NAME_CHARS, '/'); 
  291.       $bad_seg_chars = preg_quote(self::NOT_VALID_SEGMENT_NAME_CHARS, '/'); 
  292.  
  293.       $name_pattern = "(?:[^" . $bad_chars . "\s]+)|(?:\[[^" . $bad_seg_chars . "]+\])"; 
  294.       $check_pattern = "/^((" . $name_pattern . ")\.)*(" . $name_pattern  . ")\.?$/"; 
  295.       $get_pattern = "/(?:" . $name_pattern . ")/"; 
  296.  
  297.       if (!preg_match($check_pattern, $variableName)) { 
  298.           throw new \InvalidArgumentException('variable name is invalid'); 
  299.       } 
  300.  
  301.       preg_match_all($get_pattern, $variableName, $matches); 
  302.  
  303.       $chunks = array(); 
  304.       foreach ($matches[0] as $chunk) { 
  305.           // Remove wrapper braces if needed 
  306.           if ($chunk[0] == '[') { 
  307.               $chunk = substr($chunk, 1, -1); 
  308.           } 
  309.           $chunks[] = $chunk; 
  310.       } 
  311.  
  312.       return $chunks; 
  313.   } 
  314.  
  315. }
