  361.   public function limit_words($str='', $len=100, $more=true) { 
  362.       if (trim($len) == '' || $len < 0) $len = 100; 
  363.      if ( $str=="" || $str==NULL ) return $str; 
  364.      if ( is_array($str) ) return $str; 
  365.      $str = trim($str); 
  366.      $str = strip_tags($str); 
  367.      if ( strlen($str) <= $len ) return $str; 
  368.      $str = substr($str, 0, $len); 
  369.      if ( $str != "" ) { 
  370.           if ( !substr_count($str, " ") ) { 
  371.                     if ( $more ) $str .= " ..."; 
  372.                   return $str; 
  373.           } 
  374.           while( strlen($str) && ($str[strlen($str)-1] != " ") ) { 
  375.                   $str = substr($str, 0, -1); 
  376.           } 
  377.           $str = substr($str, 0, -1); 
  378.           if ( $more ) $str .= " ..."; 
  379.           } 
  380.           return $str; 
  381.   }
