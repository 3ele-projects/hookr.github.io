  422.     public function limit_words($str='', $len=100, $more=true) { 
  423.         if (trim($len) == '' || $len < 0) $len = 100; 
  424.        if ( $str=="" || $str==NULL ) return $str; 
  425.        if ( is_array($str) ) return $str; 
  426.        $str = trim($str); 
  427.        $str = strip_tags($str); 
  428.        if ( strlen($str) <= $len ) return $str; 
  429.        $str = substr($str, 0, $len); 
  430.        if ( $str != "" ) { 
  431.             if ( !substr_count($str, " ") ) { 
  432.                       if ( $more ) $str .= " ..."; 
  433.                     return $str; 
  434.             } 
  435.             while( strlen($str) && ($str[strlen($str)-1] != " ") ) { 
  436.                     $str = substr($str, 0, -1); 
  437.             } 
  438.             $str = substr($str, 0, -1); 
  439.             if ( $more ) $str .= " ..."; 
  440.             } 
  441.             return $str; 
  442.     }
