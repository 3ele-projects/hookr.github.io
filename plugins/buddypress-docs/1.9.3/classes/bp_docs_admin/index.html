  198.     function replace_recent_comments_dashboard_widget() { 
  199.         global $wp_meta_boxes; 
  200.  
  201.         // Find the recent comments widget 
  202.         foreach ( $wp_meta_boxes['dashboard'] as $context => $widgets ) { 
  203.             if ( !empty( $widgets ) && !empty( $widgets['core'] ) && is_array( $widgets['core'] ) && array_key_exists( 'dashboard_recent_comments', $widgets['core'] ) ) { 
  204.                 // Take note of the context for when we add our widget 
  205.                 $drc_widget_context = $context; 
  206.  
  207.                 // Store the widget so that we have access to its information 
  208.                 $drc_widget = $widgets['core']['dashboard_recent_comments']; 
  209.  
  210.                 // Store the array keys, so that we can reorder things later 
  211.                 $widget_order = array_keys( $widgets['core'] ); 
  212.  
  213.                 // Remove the core widget 
  214.                 remove_meta_box( 'dashboard_recent_comments', 'dashboard', $drc_widget_context ); 
  215.  
  216.                 // No need to continue the loop 
  217.                 break; 
  218.             } 
  219.         } 
  220.  
  221.         // If we couldn't find the recent comments widget, it must have been removed. We'll 
  222.         // assume this means we shouldn't add our own 
  223.         if ( empty( $drc_widget ) ) 
  224.             return; 
  225.  
  226.         // Set up and add our widget 
  227.         $recent_comments_title = __( 'Recent Comments', 'bp-docs' ); 
  228.  
  229.         // Add our widget in the same location 
  230.         wp_add_dashboard_widget( 'dashboard_recent_comments_bp_docs', $recent_comments_title, array( $this, 'wp_dashboard_recent_comments' ), 'wp_dashboard_recent_comments_control' ); 
  231.  
  232.         // Restore the previous widget order. File this under "good citizenship" 
  233.         $wp_meta_boxes['dashboard'][$context]['core']['dashboard_recent_comments'] = $wp_meta_boxes['dashboard'][$context]['core']['dashboard_recent_comments_bp_docs']; 
  234.  
  235.         unset( $wp_meta_boxes['dashboard'][$context]['core']['dashboard_recent_comments_bp_docs'] ); 
  236.  
  237.         // In order to inherit the styles, we're going to spoof the widget ID. Sadness 
  238.         $wp_meta_boxes['dashboard'][$context]['core']['dashboard_recent_comments']['id'] = 'dashboard_recent_comments'; 
  239.     }

  246.     function wp_dashboard_recent_comments() { 
  247.         global $wpdb, $bp; 
  248.  
  249.         if ( current_user_can('edit_posts') ) 
  250.             $allowed_states = array('0', '1'); 
  251.         else 
  252.             $allowed_states = array('1'); 
  253.  
  254.         // Select all comment types and filter out spam later for better query performance. 
  255.         $comments = array(); 
  256.         $start = 0; 
  257.  
  258.         $widgets = get_option( 'dashboard_widget_options' ); 
  259.         $total_items = isset( $widgets['dashboard_recent_comments'] ) && isset( $widgets['dashboard_recent_comments']['items'] ) 
  260.             ? absint( $widgets['dashboard_recent_comments']['items'] ) : 5; 
  261.  
  262.         while ( count( $comments ) < $total_items && $possible = $wpdb->get_results( "SELECT c.*, p.post_type AS comment_post_post_type FROM $wpdb->comments c LEFT JOIN $wpdb->posts p ON c.comment_post_ID = p.ID WHERE p.post_status != 'trash' ORDER BY c.comment_date_gmt DESC LIMIT $start, 50" ) ) { 
  263.  
  264.             foreach ( $possible as $comment ) { 
  265.                 if ( count( $comments ) >= $total_items ) 
  266.                     break; 
  267.  
  268.                 // Is the user allowed to read this doc? 
  269.                 if ( $bp->bp_docs->post_type_name == $comment->comment_post_post_type && !bp_docs_user_can( 'read', get_current_user_ID(), $comment->comment_post_ID ) ) 
  270.                     continue; 
  271.  
  272.                 if ( in_array( $comment->comment_approved, $allowed_states ) && current_user_can( 'read_post', $comment->comment_post_ID ) ) 
  273.                     $comments[] = $comment; 
  274.             } 
  275.  
  276.             $start = $start + 50; 
  277.         } 
  278.  
  279.         if ( $comments ) : 
  280.     ?> 
  281.  
  282.             <div id="the-comment-list" class="list:comment"> 
  283.     <?php 
  284.             foreach ( $comments as $comment ) 
  285.                 _wp_dashboard_recent_comments_row( $comment ); 
  286.     ?> 
  287.  
  288.             </div> 
  289.  
  290.     <?php 
  291.             if ( current_user_can('edit_posts') ) { ?> 
  292.                 <?php _get_list_table('WP_Comments_List_Table')->views(); ?> 
  293.     <?php    } 
  294.  
  295.             wp_comment_reply( -1, false, 'dashboard', false ); 
  296.             wp_comment_trashnotice(); 
  297.  
  298.         else : 
  299.     ?> 
  300.  
  301.         <p><?php _e( 'No comments yet.', 'bp-docs' ); ?></p>
