  467.             return; 
  468.  
  469.         // Bail if in admin 
  470.         if ( is_admin() ) 
  471.             return; 
  472.  
  473.         // Don't query for any posts on /docs/create/ 
  474.         if ( $posts_query->get( BP_DOCS_CREATE_SLUG ) ) { 
  475.             $posts_query->is_404 = false; 
  476.             $posts_query->set( 'p', -1 ); 
  477.         } 
  478.  
  479.         // Fall back on archive template on /docs/my-groups/ 
  480.         if ( $posts_query->get( BP_DOCS_MY_GROUPS_SLUG ) ) { 
  481.             $posts_query->is_404 = false; 
  482.         } 
  483.  
  484.         // For single Doc views, allow access to 'deleted' items 
  485.         // that the current user is the admin of 
  486.         if ( $posts_query->is_single && bp_docs_get_post_type_name() === $posts_query->get( 'post_type' ) ) { 
  487.  
  488.             $doc_slug = $posts_query->get( 'name' ); 
  489.  
  490.             // Direct query, because barf 
  491.             global $wpdb; 
  492.             $author_id = $wpdb->get_var( $wpdb->prepare( 
  493.                 "SELECT post_author FROM {$wpdb->posts} WHERE post_type = %s AND post_name = %s",  
  494.                 bp_docs_get_post_type_name(),  
  495.                 $doc_slug 
  496.  ) ); 
  497.  
  498.             // Post author or mod can visit it 
  499.             if ( $author_id && ( $author_id == get_current_user_id() || current_user_can( 'bp_moderate' ) ) ) { 
  500.                 $posts_query->set( 'post_status', array( 'publish', 'trash' ) ); 
  501.  
  502.                 // Make the 'trash' post status public 
  503.                 add_filter( 'posts_request', array( $this, 'make_trash_public' ) ); 
  504.  
  505.                 // ... and undo that when we're done 
  506.                 add_filter( 'the_posts', array( $this, 'remove_make_trash_public' ) ); 
  507.             } 
  508.         } 
  509.     } 
  510.  
  511.     /** 
  512.      * Make the 'trash' post status public. 
  513.      * 
  514.      * This is an unavoidable hack for cases where we want a trashed doc 
  515.      * to be visitable by the post author or by an admin. 
  516.      * 
  517.      * Access is public because it needs to be accessible by 
  518.      * call_user_func(), but should *not* be called directly. 
  519.      * 
  520.      * @since BuddyPress 1.5.5 
  521.      * 
  522.      * @param $request Passthrough. 
  523.      */ 
  524.     public function make_trash_public( $request ) { 
  525.         global $wp_post_statuses; 
  526.         $wp_post_statuses['trash']->public = true; 
  527.         return $request; 
  528.     } 
  529.  
  530.     /** 
  531.      * Reverse the public-trash hack applied in self::make_trash_public() 
  532.      * 
  533.      * @since BuddyPress 1.5.5 
  534.      * 
  535.      * @param $posts Passthrough. 
  536.      */ 
  537.     public function remove_make_trash_public( $posts ) { 
  538.         global $wp_post_statuses; 
  539.         $wp_post_statuses['trash']->public = false; 
  540.         return $posts; 
  541.     }

  543.     /** 
  544.      * Protects group docs from unauthorized access 
  545.      * 
  546.      * @since 1.2 
  547.      */ 
  548.     function protect_doc_access() { 
  549.         // What is the user trying to do? 
  550.         if ( bp_docs_is_doc_read() ) { 
  551.             $action = 'bp_docs_read'; 
  552.         } else if ( bp_docs_is_doc_create() ) { 
  553.             $action = 'bp_docs_create'; 
  554.         } else if ( bp_docs_is_doc_edit() ) { 
  555.             $action = 'bp_docs_edit'; 
  556.         } else if ( bp_docs_is_doc_history() ) { 
  557.             $action = 'bp_docs_view_history'; 
  558.         } 
  559.  
  560.         if ( ! isset( $action ) ) { 
  561.             return; 
  562.         } 
  563.  
  564.         if ( ! current_user_can( $action ) ) { 
  565.             $redirect_to = bp_docs_get_doc_link(); 
  566.  
  567.             bp_core_no_access( array( 
  568.                 'mode' => 2,  
  569.                 'redirect' => $redirect_to,  
  570.  ) ); 
  571.         } 
  572.     }

  574.     function flush_rewrite_rules() { 
  575.         if ( ! is_admin() ) { 
  576.             return; 
  577.         } 
  578.  
  579.         if ( ! is_super_admin() ) { 
  580.             return; 
  581.         } 
  582.  
  583.         global $wp_rewrite; 
  584.  
  585.         // Check to see whether our rules have been registered yet, by 
  586.         // finding a Docs rule and then comparing it to the registered rules 
  587.         $test_rewrite = null; 
  588.         foreach ( $wp_rewrite->extra_rules_top as $rewrite => $rule ) { 
  589.             if ( 0 === strpos( $rewrite, bp_docs_get_docs_slug() ) ) { 
  590.                 $test_rewrite = $rewrite; 
  591.                 $test_rule = $rule; 
  592.                 break; 
  593.             } 
  594.         } 
  595.         $registered_rules = get_option( 'rewrite_rules' ); 
  596.  
  597.         if ( $test_rewrite && is_array( $registered_rules ) && ( ! isset( $registered_rules[ $test_rewrite ] ) || $test_rule !== $registered_rules[ $test_rewrite ] ) ) { 
  598.             flush_rewrite_rules(); 
  599.         } 
  600.     }
