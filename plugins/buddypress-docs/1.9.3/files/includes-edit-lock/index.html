  10.  * BP Docs heartbeat interval. 
  11.  * 
  12.  * @since 1.6.0 
  13.  * 
  14.  * @return int 
  15.  */ 
  16. function bp_docs_heartbeat_pulse() { 
  17.   // Check whether a global heartbeat already exists 
  18.   $heartbeat_settings = apply_filters( 'heartbeat_settings', array() ); 
  19.   if ( ! empty( $heartbeat_settings['interval'] ) ) { 
  20.       if ( 'fast' === $heartbeat_settings['interval'] ) { 
  21.           $pulse = 5; 
  22.       } else { 
  23.           $pulse = intval( $heartbeat_settings['interval'] ); 
  24.       } 
  25.   } 
  26.  
  27.   // Fallback 
  28.   if ( empty( $pulse ) ) { 
  29.       $pulse = 15; 
  30.   } 
  31.  
  32.   // Filter here to specify a Docs-specific pulse frequency 
  33.   $pulse = intval( apply_filters( 'bp_docs_activity_pulse', $pulse ) ); 
  34.  
  35.   return $pulse; 
  36. }

  38. /** 
  39.  * Handle heartbeat 
  40.  * 
  41.  * @since 1.6.0 
  42.  */ 
  43. function bp_docs_heartbeat_callback( $response, $data ) { 
  44.   if ( empty( $data['doc_id'] ) ) { 
  45.       return $response; 
  46.   } 
  47.  
  48.   $doc_id = intval( $data['doc_id'] ); 
  49.  
  50.   if ( ! $doc_id ) { 
  51.       return $response; 
  52.   } 
  53.  
  54.   $uid = bp_loggedin_user_id(); 
  55.  
  56.   $lock = bp_docs_check_post_lock( $doc_id ); 
  57.  
  58.   // No lock, or belongs to the current user 
  59.   if ( empty( $lock ) || $lock == bp_loggedin_user_id() ) { 
  60.       $time = time(); 
  61.       $lstring = "$time:$uid"; 
  62.       update_post_meta( $doc_id, '_bp_docs_last_pinged', $lstring ); 
  63.  
  64.   // Someone else is editing, so bounce 
  65.   } else { 
  66.       $bounce = bp_docs_get_doc_link( $doc_id ); 
  67.       $by = new WP_User( $lock ); 
  68.       if ( ! empty( $by->user_nicename ) ) { 
  69.           $bounce = add_query_arg( 'by', $by->user_nicename, $bounce ); 
  70.       } 
  71.       $response['bp_docs_bounce'] = $bounce; 
  72.   } 
  73.  
  74.   return $response; 
  75. } 
  76. add_filter( 'heartbeat_received', 'bp_docs_heartbeat_callback', 10, 2 );

  99. /** 
  100.  * Check to see if the post is currently being edited by another user. 
  101.  * 
  102.  * This is a verbatim copy of wp_check_post_lock(), which is only available 
  103.  * in the admin 
  104.  * 
  105.  * @since 1.2.8 
  106.  * 
  107.  * @param int $post_id ID of the post to check for editing 
  108.  * @return bool|int False: not locked or locked by current user. Int: user ID of user with lock. 
  109.  */ 
  110. function bp_docs_check_post_lock( $post_id ) { 
  111.   if ( !$post = get_post( $post_id ) ) 
  112.       return false; 
  113.  
  114.   if ( !$lock = get_post_meta( $post->ID, '_bp_docs_last_pinged', true ) ) 
  115.       return false; 
  116.  
  117.   $lock = explode( ':', $lock ); 
  118.   $time = $lock[0]; 
  119.   $user = isset( $lock[1] ) ? $lock[1] : get_post_meta( $post->ID, '_edit_last', true ); 
  120.  
  121.   $heartbeat_interval = bp_docs_heartbeat_pulse(); 
  122.  
  123.   // Bail out of the lock if four pings have been missed (one minute, by default) 
  124.   $time_window = apply_filters( 'bp_docs_post_lock_interval', $heartbeat_interval * 4 ); 
  125.  
  126.   if ( $time && $time > time() - $time_window && $user != get_current_user_id() ) { 
  127.       return $user; 
  128.   } 
  129.  
  130.   return false; 
  131. }

  133. /** 
  134.  * Get the lock status of a doc 
  135.  * 
  136.  * The function first tries to get the lock status out of $bp. If it has to look it up, it 
  137.  * stores the data in $bp for future use. 
  138.  * 
  139.  * @since 1.0-beta-2 
  140.  * 
  141.  * @param int $doc_id Optional. Defaults to the doc currently being viewed 
  142.  * @return int Returns 0 if there is no lock, otherwise returns the user_id of the locker 
  143.  */ 
  144. function bp_docs_is_doc_edit_locked( $doc_id = false ) { 
  145.   global $bp, $post; 
  146.  
  147.   // Try to get the lock out of $bp first 
  148.   if ( isset( $bp->bp_docs->current_doc_lock ) ) { 
  149.       $is_edit_locked = $bp->bp_docs->current_doc_lock; 
  150.   } else { 
  151.       $is_edit_locked = 0; 
  152.  
  153.       if ( empty( $doc_id ) ) 
  154.           $doc_id = !empty( $post->ID ) ? $post->ID : false; 
  155.  
  156.       if ( $doc_id ) { 
  157.           $is_edit_locked = bp_docs_check_post_lock( $doc_id ); 
  158.       } 
  159.  
  160.       // Put into the $bp global to avoid extra lookups 
  161.       $bp->bp_docs->current_doc_lock = $is_edit_locked; 
  162.   } 
  163.  
  164.   return apply_filters( 'bp_docs_is_doc_edit_locked', $is_edit_locked, $doc_id ); 
  165. }
