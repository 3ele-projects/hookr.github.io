  260.   public static function limits_filter( $filtered_posts ) { 
  261.       global $wpdb; 
  262.  
  263.       if ( @ $_POST['limits'] ) { 
  264.           $matched_products = false; 
  265.  
  266.           foreach ( $_POST['limits'] as $v ) { 
  267.               $matched_products_query = $wpdb->get_results( $wpdb->prepare(" 
  268.                   SELECT DISTINCT ID, post_parent, post_type FROM $wpdb->posts 
  269.                   INNER JOIN $wpdb->term_relationships as tr ON ID = tr.object_id 
  270.                   INNER JOIN $wpdb->term_taxonomy as tt ON tt.term_taxonomy_id = tr.term_taxonomy_id 
  271.                   INNER JOIN $wpdb->terms as t ON t.term_id = tt.term_id 
  272.                   WHERE post_type IN ( 'product', 'product_variation' ) AND post_status = 'publish' 
  273.                   AND tt.taxonomy = %s AND t.slug BETWEEN %d AND %d 
  274.               ", $v[0], $v[1], $v[2] ), OBJECT_K ); 
  275.  
  276.               if ( $matched_products_query ) { 
  277.                   if ( $matched_products === false ) { 
  278.                       $matched_products = array( 0 ); 
  279.                       foreach ( $matched_products_query as $product ) { 
  280.                           if ( $product->post_type == 'product' ) 
  281.                               $matched_products[] = $product->ID; 
  282.                           if ( $product->post_parent > 0 && ! in_array( $product->post_parent, $matched_products ) ) 
  283.                               $matched_products[] = $product->post_parent; 
  284.                       } 
  285.                   } else { 
  286.                       $new_products = array( 0 ); 
  287.                       foreach ( $matched_products_query as $product ) { 
  288.                           if ( $product->post_type == 'product' && in_array($product->ID, $matched_products)) 
  289.                               $new_products[] = $product->ID; 
  290.                           if ( $product->post_parent > 0 && ! in_array( $product->post_parent, $matched_products ) && in_array($product->post_parent, $matched_products)) 
  291.                               $new_products[] = $product->post_parent; 
  292.                       } 
  293.                       $matched_products = $new_products; 
  294.                   } 
  295.               } 
  296.           } 
  297.  
  298.           if ( $matched_products === false ) { 
  299.               $matched_products = array( 0 ); 
  300.           } 
  301.  
  302.           $matched_products = @ array_unique( $matched_products ); 
  303.  
  304.           // Filter the id's 
  305.           if ( sizeof( $filtered_posts ) == 0 ) { 
  306.               $filtered_posts = $matched_products; 
  307.           } else { 
  308.               $filtered_posts = array_intersect( $filtered_posts, $matched_products );
