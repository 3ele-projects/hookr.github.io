  552.  );

  568.  
  569.  ); 
  570.                   } 
  571.               } 
  572.  
  573.               $translated_product_tabs_updated = true; 
  574.           }

  634.   public function sync_product_tabs( $post_id, $post ) { 
  635.  
  636.       $override_tab_layout = get_post_meta( $post_id, '_override_tab_layout', true ); 
  637.  
  638.       if ( $override_tab_layout && $this->woocommerce_wpml->products->is_original_product( $post_id ) ) { 
  639.  
  640.           $original_product_tabs = $this->get_product_tabs( $post_id ); 
  641.  
  642.           $trid = $this->sitepress->get_element_trid( $post_id, 'post_'  . $post->post_type ); 
  643.           $translations = $this->sitepress->get_element_translations( $trid, 'post_'  . $post->post_type, true ); 
  644.  
  645.  
  646.  
  647.           foreach( $translations as $language => $translation ) { 
  648.  
  649.               if( empty( $translation->original ) ) { 
  650.  
  651.                   $translated_product_tabs = $this->get_product_tabs( $translation->element_id ); 
  652.  
  653.                   // sync tab positions for product tabs 
  654.                   foreach( $original_product_tabs as $tab ) { 
  655.                       if( $tab['type'] == 'product' ) { 
  656.                           $translated_tab_product_id = apply_filters( 'translate_object_id', $tab['id'], 'wc_product_tab', false, $language ); 
  657.                           if( $translated_tab_product_id && is_array( $translated_product_tabs['product_tab_' . $translated_tab_product_id ] ) ) { 
  658.                               $translated_product_tabs['product_tab_' . $translated_tab_product_id ]['position'] = $tab['position']; 
  659.                           } 
  660.                       } 
  661.                   } 
  662.  
  663.                   // sync translated core tabs with original tabs 
  664.                   foreach( $translated_product_tabs as $tab_key => $tab ) { 
  665.                       if( $tab['type'] === 'core' && !isset( $original_product_tabs[$tab_key] ) ) { 
  666.                           unset( $translated_product_tabs[$tab_key] ); 
  667.                       } 
  668.                   } 
  669.  
  670.                   update_post_meta( $translation->element_id, '_product_tabs', $translated_product_tabs ); 
  671.  
  672.               } 
  673.  
  674.           } 
  675.  
  676.       } 
  677.  
  678.   }
