  182.   public function init_currencies() { 
  183.       global $sitepress; 
  184.  
  185.       $this->default_currency = get_option( 'woocommerce_currency' ); 
  186.       $this->currencies =& $this->woocommerce_wpml->settings['currency_options']; 
  187.  
  188.       // Add default currency if missing (set when MC is off) 
  189.       if( !empty( $this->default_currency ) && !isset( $this->currencies[ $this->default_currency ] ) ) { 
  190.           $this->currencies[ $this->default_currency ] = array(); 
  191.       } 
  192.  
  193.       $save_to_db = false; 
  194.  
  195.       $active_languages = $sitepress->get_active_languages(); 
  196.  
  197.       $currency_defaults = array( 
  198.           'rate' => 0,  
  199.           'position' => 'left',  
  200.           'thousand_sep' => ', ',  
  201.           'decimal_sep' => '.',  
  202.           'num_decimals' => 2,  
  203.           'rounding' => 'disabled',  
  204.           'rounding_increment' => 1,  
  205.           'auto_subtract' => 0 
  206.  ); 
  207.  
  208.       foreach($this->currencies as $code => $currency) { 
  209.           foreach($currency_defaults as $key => $val) { 
  210.               if(!isset($currency[$key])) { 
  211.                   $this->currencies[$code][$key] = $val; 
  212.                   $save_to_db = true; 
  213.               } 
  214.           } 
  215.  
  216.           foreach($active_languages as $language) { 
  217.               if(!isset($currency['languages'][$language['code']])) { 
  218.                   $this->currencies[$code]['languages'][$language['code']] = 1; 
  219.                   $save_to_db = true; 
  220.               } 
  221.           } 
  222.       } 
  223.  
  224.       $this->currency_codes = array_keys($this->currencies); 
  225.  
  226.       // default language currencies 
  227.       foreach($active_languages as $language) { 
  228.           if(!isset($this->woocommerce_wpml->settings['default_currencies'][$language['code']])) { 
  229.               $this->woocommerce_wpml->settings['default_currencies'][$language['code']] = 0; 
  230.               $save_to_db = true; 
  231.           } 
  232.       } 
  233.  
  234.       // sanity check 
  235.       if(isset($this->woocommerce_wpml->settings['default_currencies'])) { 
  236.           foreach($this->woocommerce_wpml->settings['default_currencies'] as $language => $value) { 
  237.               if(!isset($active_languages[$language])) { 
  238.                   unset($this->woocommerce_wpml->settings['default_currencies'][$language]); 
  239.                   $save_to_db = true; 
  240.               } 
  241.               if(!empty($value) && !in_array($value, $this->currency_codes)) { 
  242.                   $this->woocommerce_wpml->settings['default_currencies'][$language] = 0; 
  243.                   $save_to_db = true; 
  244.               } 
  245.           } 
  246.       } 
  247.  
  248.       // add missing currencies to currencies_order 
  249.       if(isset($this->woocommerce_wpml->settings['currencies_order'])) { 
  250.           foreach ($this->currency_codes as $currency) { 
  251.               if (!in_array($currency, $this->woocommerce_wpml->settings['currencies_order'])) { 
  252.                   $this->woocommerce_wpml->settings['currencies_order'][] = $currency; 
  253.                   $save_to_db = true; 
  254.               } 
  255.           } 
  256.       } 
  257.  
  258.       if($save_to_db) { 
  259.           $this->woocommerce_wpml->update_settings(); 
  260.       } 
  261.  
  262.       // force disable multi-currency when the default currency is empty 
  263.       if( empty( $this->default_currency ) ) { 
  264.           $this->woocommerce_wpml->settings['enable_multi_currency'] = WCML_MULTI_CURRENCIES_DISABLED; 
  265.       } 
  266.  
  267.   }

  343.   public function get_client_currency() { 
  344.       global $woocommerce, $sitepress, $wpdb; 
  345.  
  346.       $default_currencies = $this->woocommerce_wpml->settings['default_currencies']; 
  347.       $current_language = $sitepress->get_current_language(); 
  348.       $current_language = ( $current_language != 'all' && !is_null( $current_language ) ) ? $current_language : $sitepress->get_default_language(); 
  349.  
  350.       if( is_product() && empty( $this->client_currency ) && 
  351.           isset($this->woocommerce_wpml->settings['display_custom_prices']) && 
  352.           $this->woocommerce_wpml->settings['display_custom_prices'] ) {

  354.           $product_obj = wc_get_product();

  356.           $original_product_language = $this->woocommerce_wpml->products->get_original_product_language( $current_product_id ); 
  357.           $default = false; 
  358.  
  359.           if( WooCommerce_Functions_Wrapper::get_product_type ($current_product_id ) === 'variable' ) { 
  360.               foreach( $product_obj->get_children() as $child ) { 
  361.                   if( !get_post_meta( apply_filters( 'translate_object_id', $child , get_post_type( $child ), true, $original_product_language ), '_wcml_custom_prices_status', true ) ) { 
  362.                       $default = true; 
  363.                       break; 
  364.                   } 
  365.               } 
  366.           }elseif( !get_post_meta( apply_filters( 'translate_object_id', $current_product_id , get_post_type( $current_product_id ), true, $original_product_language ), '_wcml_custom_prices_status', true ) ) { 
  367.               $default = true; 
  368.           } 
  369.  
  370.           if( $default ) { 
  371.               $this->client_currency = get_option('woocommerce_currency'); 
  372.           } 
  373.  
  374.       } 
  375.  
  376.       if( isset($_GET['pay_for_order']) && $_GET['pay_for_order'] == true && isset($_GET['key']) ) { 
  377.           $order_id = $wpdb->get_var( $wpdb->prepare( "SELECT post_id FROM $wpdb->postmeta WHERE meta_key = '_order_key' AND meta_value = %s", sanitize_text_field( $_GET['key'] ) ) ); 
  378.           if( $order_id ) { 
  379.               $this->client_currency = get_post_meta( $order_id, '_order_currency', true ); 
  380.           } 
  381.       } 
  382.  
  383.       if( 
  384.           isset( $_POST[ 'action' ] ) && 
  385.           $_POST[ 'action' ] == 'wcml_switch_currency' && 
  386.           !empty( $_POST[ 'currency' ] ) && 
  387.           isset( $_POST[ 'force_switch' ] ) && 
  388.           !$_POST[ 'force_switch' ] && 
  389.           ( 
  390.               $this->woocommerce_wpml->settings[ 'cart_sync' ][ 'currency_switch' ] == WCML_CART_SYNC || 
  391.               $this->woocommerce_wpml->settings[ 'cart_sync' ][ 'currency_switch' ] == WCML_CART_SYNC 
  392.  )

  397.       if( isset( $_GET['action'] ) && 
  398.           $_GET['action'] === 'woocommerce_json_search_products_and_variations' && 
  399.           !empty( $_COOKIE['_wcml_order_currency'] ) ) { 
  400.           $this->client_currency = $_COOKIE['_wcml_order_currency'];

  402.  
  403.       if( is_null($this->client_currency) && 
  404.           isset($default_currencies[$current_language]) &&

  486.       } 
  487.  
  488.       do_action('wcml_switch_currency', $currency );
