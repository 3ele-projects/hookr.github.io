  232.   public function duplicate_variation_data( $original_variation_id, $variation_id, $data, $lang, $trbl ) { 
  233.       global $iclTranslationManagement; 
  234.  
  235.       if( $this->woocommerce_wpml->sync_product_data->check_if_product_fields_sync_needed( $original_variation_id, $variation_id, 'postmeta_fields' ) || $trbl ) { 
  236.           // custom fields 
  237.           $settings = $iclTranslationManagement->settings[ 'custom_fields_translation' ]; 
  238.           $all_meta = get_post_custom( $original_variation_id ); 
  239.  
  240.           $post_fields = null; 
  241.           foreach( $all_meta as $meta_key => $meta ) { 
  242.  
  243.               foreach ( $meta as $meta_value ) { 
  244.                   // update current post variations meta 
  245.  
  246.                   if( ( substr( $meta_key, 0, 10 ) == 'attribute_' || isset( $settings[ $meta_key ] ) && $settings[ $meta_key ] == WPML_COPY_CUSTOM_FIELD ) ) { 
  247.  
  248.                       // adjust the global attribute slug in the custom field 
  249.                       $attid = null; 
  250.                       if( substr( $meta_key, 0, 10 ) == 'attribute_' ) { 
  251.                           $trn_post_meta = $this->woocommerce_wpml->attributes->get_translated_variation_attribute_post_meta( $meta_value, $meta_key, $original_variation_id, $variation_id, $lang );

  253.                           $meta_key = $trn_post_meta['meta_key']; 
  254.                       } 
  255.  
  256.                       if( $meta_key == '_stock') { 
  257.                           $this->update_stock_quantity( $variation_id, $meta_value ); 
  258.                       }else{ 
  259.                           update_post_meta( $variation_id, $meta_key, maybe_unserialize( $meta_value ) ); 
  260.                       } 
  261.                   }elseif ( !isset( $settings[ $meta_key ] ) || $settings[ $meta_key ] == WPML_IGNORE_CUSTOM_FIELD ) { 
  262.                       continue; 
  263.                   } 
  264.  
  265.                   //sync variation prices 
  266.                   if( 
  267.                       ( $this->woocommerce_wpml->settings[ 'enable_multi_currency' ] == WCML_MULTI_CURRENCIES_INDEPENDENT || $trbl ) && 
  268.                       in_array( $meta_key, array( '_sale_price', '_regular_price', '_price' ) ) 
  269.  ) { 
  270.                       $meta_value = get_post_meta( $original_variation_id, $meta_key, true ); 
  271.                       update_post_meta( $variation_id, $meta_key, $meta_value ); 
  272.                   } 
  273.  
  274.                   if( isset( $settings[ $meta_key ] ) && $settings[ $meta_key ] == WPML_TRANSLATE_CUSTOM_FIELD ) { 
  275.                       //sync custom fields 
  276.                       $post_fields = $this->woocommerce_wpml->sync_product_data->sync_custom_field_value( $meta_key, $data, $variation_id, $post_fields, $original_variation_id, true );
