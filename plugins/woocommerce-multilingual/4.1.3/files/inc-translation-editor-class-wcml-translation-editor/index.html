  289.   public function auto_generate_slug() { 
  290.       global $wpdb; 
  291.  
  292.       $title = filter_input( INPUT_POST, 'title'); 
  293.  
  294.       $post_name = urldecode( sanitize_title( $title ) ); 
  295.       $job_id = intval( $_POST['job_id'] ); 
  296.  
  297.       $post_id_sql = " 
  298.           SELECT t.element_id 
  299.           FROM {$wpdb->prefix}icl_translations t 
  300.               LEFT JOIN {$wpdb->prefix}icl_translation_status s ON t.translation_id = s.translation_id 
  301.               LEFT JOIN {$wpdb->prefix}icl_translate_job j ON s.rid = j.rid 
  302.           WHERE j.job_id = %d 
  303.       "; 
  304.       $post_id = $wpdb->get_var( $wpdb->prepare( $post_id_sql, $job_id ) ); 
  305.  
  306.       if( $post_id ) { 
  307.           $slug = wp_unique_post_slug( $post_name, $post_id, 'publish', 'product', 0 ); 
  308.       }else{ 
  309.           // handle the special case when the post for this slug has not been created yet 
  310.           $lang_sql = " 
  311.               SELECT t.language_code 
  312.               FROM {$wpdb->prefix}icl_translations t 
  313.                   LEFT JOIN {$wpdb->prefix}icl_translation_status s ON t.translation_id = s.translation_id 
  314.                   LEFT JOIN {$wpdb->prefix}icl_translate_job j ON s.rid = j.rid 
  315.               WHERE j.job_id = %d 
  316.           "; 
  317.           $lang = $wpdb->get_var( $wpdb->prepare( $lang_sql, $job_id ) ); 
  318.  
  319.           $check_sql = " 
  320.               SELECT p.post_name  
  321.               FROM {$wpdb->posts} p  
  322.                   LEFT JOIN {$wpdb->prefix}icl_translations t  
  323.                       ON t.element_id = p.ID AND t.element_type='post_product'    
  324.               WHERE p.post_name = %s AND t.language_code = %s LIMIT 1 
  325.           "; 
  326.           $post_name_check = $wpdb->get_var( $wpdb->prepare( $check_sql, $post_name, $lang) ); 
  327.  
  328.           if( $post_name_check ) { 
  329.               $suffix = 2; 
  330.               do { 
  331.  
  332.                   $alt_post_name = _truncate_post_slug( $post_name, 200 - ( strlen( $suffix ) + 1 ) ) . "-$suffix"; 
  333.                   $post_name_check = $wpdb->get_var( $wpdb->prepare( $check_sql, $alt_post_name, $lang ) ); 
  334.                   $suffix++; 
  335.  
  336.               } while ( $post_name_check ); 
  337.               $slug = $alt_post_name; 
  338.           } else{ 
  339.               $slug = $post_name; 
  340.           } 
  341.  
  342.       } 
  343.  
  344.       echo json_encode( array('slug' => $slug) ); 
  345.       exit; 
  346.  
  347.   }
