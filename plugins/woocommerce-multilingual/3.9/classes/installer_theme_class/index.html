  745.     /** Upgrade theme check */ 
  746.     private function installer_theme_upgrade_theme_check() { 
  747.  
  748.         // Step1-> we get all installed themes in clients local themes directory 
  749.         $installed_themes = wp_get_themes(); 
  750.  
  751.         // Step2: We need to loop through each repository with themes 
  752.         foreach ( $this->installer_repo_with_themes as $k => $repo_slug ) { 
  753.  
  754.             // We then need to retrieved the products URL for each of this repo 
  755.             $products_url = $this->repository_theme_products [$repo_slug]; 
  756.  
  757.             // Step3-> we get all available themes in our repository via API based on this URL 
  758.             $available_themes = $this->installer_theme_get_themes( $products_url, $repo_slug ); 
  759.  
  760.             if ( !($available_themes) ) { 
  761.  
  762.                 // API is not available as of the moment, return.. 
  763.                 return; 
  764.             } else { 
  765.  
  766.                 // We have available themes here... 
  767.                 // Step4->let's simplify available themes data by slugs 
  768.                 $simplified_available_themes = $this->installer_theme_savethemes_by_slug( $available_themes, true ); 
  769.  
  770.                 // Step5->Let's loop through installed themes 
  771.                 if ( (is_array( $installed_themes )) && (!(empty ($installed_themes))) ) { 
  772.                     $otgs_theme_updates_available = array(); 
  773.                     foreach ( $installed_themes as $theme_slug => $theme_object ) { 
  774.                         if ( array_key_exists( $theme_slug, $simplified_available_themes ) ) { 
  775.  
  776.                             // This is our theme 
  777.                             // Step6->Let's get version of the local theme installed 
  778.                             $local_version = $theme_object->get( 'Version' ); 
  779.  
  780.                             // Step7->Let's get the latest version of this theme, page URL and download URL from our repository 
  781.                             $repository_version = $simplified_available_themes [$theme_slug] ['version']; 
  782.                             $theme_page_url = $simplified_available_themes [$theme_slug] ['theme_page_url']; 
  783.                             $theme_download_url = $simplified_available_themes [$theme_slug] ['url']; 
  784.  
  785.                             // Step8->Let's compare the version 
  786.                             if ( version_compare( $repository_version, $local_version, '>' ) ) { 
  787.  
  788.                                 // Update available for this theme 
  789.                                 // Step9-> Define download URL with site key 
  790.                                 $package_url = WP_Installer()->append_site_key_to_download_url( $theme_download_url, $this->installer_site_key [$repo_slug], $repo_slug ); 
  791.  
  792.                                 //Step10-> Assign to updates array for later accessing. 
  793.                                 $otgs_theme_updates_available[$theme_slug] = array( 
  794.                                     'theme' => $theme_slug,  
  795.                                     'new_version' => $repository_version,  
  796.                                     'url' => $theme_page_url,  
  797.                                     'package' => $package_url 
  798.  ); 
  799.                             } 
  800.                         } 
  801.                     } 
  802.                     //Exited the upgrade loop for this specific theme repository 
  803.                     if ( !(empty($otgs_theme_updates_available)) ) { 
  804.                         //Has updates 
  805.                         update_option( $this->installer_themes_available_updates[$repo_slug], $otgs_theme_updates_available ); 
  806.                     } else { 
  807.                         //No updates 
  808.                         delete_option( $this->installer_themes_available_updates[$repo_slug] ); 
  809.                     } 
  810.  
  811.                 } 
  812.             } 
  813.         } 
  814.     }
