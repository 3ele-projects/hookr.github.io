  420.      * Verifies the id token, returns the verified token contents. 
  421.      * 
  422.      * @param $jwt string 
  423.      *            the token 
  424.      * @param $certs array 
  425.      *            of certificates 
  426.      * @param $required_audience string 
  427.      *            the expected consumer of the token 
  428.      * @param 
  429.      *            [$issuer] the expected issues, defaults to Google 
  430.      * @param 
  431.      *            [$max_expiry] the max lifetime of a token, defaults to MAX_TOKEN_LIFETIME_SECS 
  432.      * @throws Google_Auth_Exception 
  433.      * @return mixed token information if valid, false if not 
  434.      */ 
  435.     public function verifySignedJwtWithCerts($jwt, $certs, $required_audience, $issuer = null, $max_expiry = null) 
  436.     { 
  437.         if (! $max_expiry) { 
  438.             // Set the maximum time we will accept a token for. 
  439.             $max_expiry = self::MAX_TOKEN_LIFETIME_SECS; 
  440.         } 
  441.         $segments = explode(".", $jwt); 
  442.         if (count($segments) != 3) { 
  443.             throw new Google_Auth_Exception("Wrong number of segments in token: $jwt"); 
  444.         } 
  445.         $signed = $segments[0] . "." . $segments[1]; 
  446.         $signature = Google_Utils::urlSafeB64Decode($segments[2]); 
  447.         // Parse envelope. 
  448.         $envelope = json_decode(Google_Utils::urlSafeB64Decode($segments[0]), true); 
  449.         if (! $envelope) { 
  450.             throw new Google_Auth_Exception("Can't parse token envelope: " . $segments[0]); 
  451.         } 
  452.         // Parse token 
  453.         $json_body = Google_Utils::urlSafeB64Decode($segments[1]); 
  454.         $payload = json_decode($json_body, true); 
  455.         if (! $payload) { 
  456.             throw new Google_Auth_Exception("Can't parse token payload: " . $segments[1]); 
  457.         } 
  458.         // Check signature 
  459.         $verified = false; 
  460.         foreach ($certs as $keyName => $pem) { 
  461.             $public_key = new Google_Verifier_Pem($pem); 
  462.             if ($public_key->verify($signed, $signature)) { 
  463.                 $verified = true; 
  464.                 break; 
  465.             } 
  466.         } 
  467.         if (! $verified) { 
  468.             throw new Google_Auth_Exception("Invalid token signature: $jwt"); 
  469.         } 
  470.         // Check issued-at timestamp 
  471.         $iat = 0; 
  472.         if (array_key_exists("iat", $payload)) { 
  473.             $iat = $payload["iat"]; 
  474.         } 
  475.         if (! $iat) { 
  476.             throw new Google_Auth_Exception("No issue time in token: $json_body"); 
  477.         } 
  478.         $earliest = $iat - self::CLOCK_SKEW_SECS; 
  479.         // Check expiration timestamp 
  480.         $now = time(); 
  481.         $exp = 0; 
  482.         if (array_key_exists("exp", $payload)) { 
  483.             $exp = $payload["exp"]; 
  484.         } 
  485.         if (! $exp) { 
  486.             throw new Google_Auth_Exception("No expiration time in token: $json_body"); 
  487.         } 
  488.         if ($exp >= $now + $max_expiry) { 
  489.             throw new Google_Auth_Exception(sprintf("Expiration time too far in future: %s", $json_body)); 
  490.         } 
  491.         $latest = $exp + self::CLOCK_SKEW_SECS; 
  492.         if ($now < $earliest) { 
  493.             throw new Google_Auth_Exception(sprintf("Token used too early, %s < %s: %s", $now, $earliest, $json_body)); 
  494.         } 
  495.         if ($now > $latest) { 
  496.             throw new Google_Auth_Exception(sprintf("Token used too late, %s > %s: %s", $now, $latest, $json_body)); 
  497.         } 
  498.         $iss = $payload['iss']; 
  499.         if ($issuer && $iss != $issuer) { 
  500.             throw new Google_Auth_Exception(sprintf("Invalid issuer, %s != %s: %s", $iss, $issuer, $json_body)); 
  501.         } 
  502.         // Check audience 
  503.         $aud = $payload["aud"]; 
  504.         if ($aud != $required_audience) { 
  505.             throw new Google_Auth_Exception(sprintf("Wrong recipient, %s != %s:", $aud, $required_audience, $json_body)); 
  506.         } 
  507.         // All good. 
  508.         return new Google_Auth_LoginTicket($envelope, $payload); 
  509.     }
