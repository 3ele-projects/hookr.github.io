  251.         $field = array_merge($this->defaults, $field);

  415.     function get_js_locale($locale) { 
  416.         $dir_path = $this->settings['path'] . 'js/localization/'; 
  417.         $exclude_list = array(".", ".."); 
  418.         $languages = $this->ps_preg_filter("/jquery-ui-timepicker-(.*?)\.js/", "$1", array_diff(scandir($dir_path), $exclude_list)); 
  419.  
  420.         $locale = strtolower(str_replace("_", "-", $locale)); 
  421.  
  422.         if (false !== strpos($locale, '-')) { 
  423.             $l = explode("-", $locale); 
  424.             $pattern = array('/' .  $locale . '/', '/' . $l[0] . '/', '/' . $l[1]  . '/'); 
  425.         } else { 
  426.             $pattern = array('/' . $locale . '/'); 
  427.         } 
  428.         $res = $this->ps_preg_filter($pattern, "$0", $languages, -1, $count); 
  429.  
  430.         return ($count) ? implode("", $res) : 'en'; 
  431.     }

  434.     function ps_preg_filter ($pattern, $replace, $subject, $limit = -1, &$count = 0) { 
  435.         if (function_exists('preg_filter')) 
  436.             return preg_filter($pattern, $replace, $subject, $limit, $count); 
  437.         else 
  438.             return  array_diff(preg_replace($pattern, $replace, $subject, $limit, $count), $subject); 
  439.     }
