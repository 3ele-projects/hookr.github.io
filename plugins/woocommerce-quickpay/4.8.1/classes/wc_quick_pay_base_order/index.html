  520.

  524.             } 
  525.         } 
  526.         return $order_number; 
  527.     }

  577.      * Returns custom variables chosen in the gateway settings. This information will 
  578.      * be sent to QuickPay and stored with the transaction. 
  579.      * 
  580.      * @access public 
  581.      * @return array 
  582.      */ 
  583.     public function get_custom_variables() 
  584.     { 
  585.         $custom_vars_settings = (array)WC_QP()->s('quickpay_custom_variables'); 
  586.         $custom_vars = array(); 
  587.  
  588.         // Single: Order Email 
  589.         if (in_array('customer_email', $custom_vars_settings)) {

  591.         } 
  592.  
  593.         // Single: Order Phone 
  594.         if (in_array('customer_phone', $custom_vars_settings)) { 
  595.             $custom_vars[__('Customer Phone', 'woo-quickpay')] = version_compare( WC_VERSION, '3.0', '<' ) ? $this->billing_phone : $this->get_billing_phone(); 
  596.         } 
  597.  
  598.         // Single: Browser User Agent 
  599.         if (in_array('browser_useragent', $custom_vars_settings)) { 
  600.             $custom_vars[__('User Agent', 'woo-quickpay')] = version_compare( WC_VERSION, '3.0', '<' ) ? $this->customer_user_agent : $this->get_customer_user_agent(); 
  601.         } 
  602.  
  603.         // Single: Shipping Method 
  604.         if (in_array('shipping_method', $custom_vars_settings)) { 
  605.             $custom_vars[__('Shipping Method', 'woo-quickpay')] = $this->get_shipping_method(); 
  606.         } 
  607.  
  608.         // Save a POST ID reference on the transaction 
  609.         $custom_vars['order_post_id'] = version_compare( WC_VERSION, '3.0', '<' ) ? $this->id : $this->get_id(); 
  610.  
  611.         // Get the correct order_post_id. We want to fetch the ID of the subscription to store data on subscription (if available). 
  612.         // But only on the first attempt. In case of failed auto capture on the initial order, we dont want to add the subscription ID. 
  613.         // If we are handlong a product switch, we will not need this ID as we are making a regular payment. 
  614.         if ( ! $this->order_contains_switch() ) { 
  615.             $subscription_id = WC_QuickPay_Subscription::get_subscription_id($this); 
  616.             if ($subscription_id) { 
  617.                 $custom_vars['subscription_post_id'] = $subscription_id; 
  618.             } 
  619.         } 
  620.  
  621.         if ($this->is_request_to_change_payment()) { 
  622.             $custom_vars['change_payment'] = TRUE; 
  623.         } 
  624.  
  625.         $custom_vars = apply_filters('woocommerce_quickpay_transaction_params_variables', $custom_vars); 
  626.  
  627.         ksort($custom_vars); 
  628.  
  629.         return array('variables' => $custom_vars);

  702.     /** 
  703.      * Determine if we should enable autocapture on the order. This is based on both the 
  704.      * plugin configuration and the product types. If the order contains both virtual 
  705.      * and non-virtual products, we will default to the 'quickpay_autocapture'-setting. 
  706.      */ 
  707.     public function get_autocapture_setting() 
  708.     { 
  709.         // Get the autocapture settings 
  710.         $autocapture_default = WC_QP()->s('quickpay_autocapture'); 
  711.         $autocapture_virtual = WC_QP()->s('quickpay_autocapture_virtual'); 
  712.  
  713.         $has_virtual_products = FALSE; 
  714.         $has_nonvirtual_products = FALSE; 
  715.  
  716.         // If the two options are the same, return immediately. 
  717.         if ($autocapture_default === $autocapture_virtual) { 
  718.             return $autocapture_default; 
  719.         } 
  720.  
  721.         // Check order items type. 
  722.         $order_items = $this->get_items('line_item'); 
  723.  
  724.         // Loop through the order items 
  725.         foreach ($order_items as $order_item) { 
  726.             // Get the product 
  727.             $product = $this->get_product_from_item($order_item); 
  728.  
  729.             // Is this product virtual? 
  730.             if ($product->is_virtual()) { 
  731.                 $has_virtual_products = TRUE; 
  732.             } // This was a non-virtual product. 
  733.             else { 
  734.                 $has_nonvirtual_products = TRUE; 
  735.             } 
  736.         } 
  737.  
  738.         // If the order contains both virtual and nonvirtual products,  
  739.         // we use the 'quickpay_autopay' as the option of choice. 
  740.         if ($has_virtual_products AND $has_nonvirtual_products) { 
  741.             return $autocapture_default; 
  742.         } // Or check if the order contains virtual products only 
  743.         else if ($has_virtual_products) { 
  744.             return $autocapture_virtual; 
  745.         } // Or default 
  746.         else { 
  747.             return $autocapture_default; 
  748.         } 
  749.     }
