  233.   protected function process_webservice_payment( $order ) { 
  234.       $payment_url = ''; 
  235.       $card_number = isset( $_POST['cielo_debit_number'] ) ? sanitize_text_field( $_POST['cielo_debit_number'] ) : ''; 
  236.       $card_brand = $this->api->get_card_brand( $card_number ); 
  237.  
  238.       // Validate credit card brand. 
  239.       if ( 'mastercard' == $card_brand ) { 
  240.           $_card_brand = 'maestro'; 
  241.       } else if ( 'visa' == $card_brand ) { 
  242.           $_card_brand = 'visaelectron'; 
  243.       } else { 
  244.           $_card_brand = $card_brand; 
  245.       } 
  246.       $valid = $this->validate_credit_brand( $_card_brand ); 
  247.  
  248.       // Test the card fields. 
  249.       if ( $valid ) { 
  250.           $valid = $this->validate_card_fields( $_POST ); 
  251.       } 
  252.  
  253.       if ( $valid ) { 
  254.           $card_brand = ( 'maestro' == $card_brand ) ? 'mastercard' : $card_brand; 
  255.           $card_data = array( 
  256.               'name_on_card' => $_POST['cielo_debit_holder_name'],  
  257.               'card_number' => $_POST['cielo_debit_number'],  
  258.               'card_expiration' => $_POST['cielo_debit_expiry'],  
  259.               'card_cvv' => $_POST['cielo_debit_cvc'] 
  260.  ); 
  261.  
  262.           $response = $this->api->do_transaction( $order, $order->id . '-' . time(), $card_brand, 0, $card_data, true ); 
  263.  
  264.           // Set the error alert. 
  265.           if ( isset( $response->mensagem ) && ! empty( $response->mensagem ) ) { 
  266.               $this->add_error( (string) $response->mensagem ); 
  267.               $valid = false; 
  268.           } 
  269.  
  270.           // Save the tid. 
  271.           if ( isset( $response->tid ) && ! empty( $response->tid ) ) { 
  272.               update_post_meta( $order->id, '_transaction_id', (string) $response->tid ); 
  273.           } 
  274.  
  275.           // Set the transaction URL. 
  276.           if ( isset( $response->{'url-autenticacao'} ) && ! empty( $response->{'url-autenticacao'} ) ) { 
  277.               $payment_url = (string) $response->{'url-autenticacao'}; 
  278.           } else { 
  279.               $payment_url = str_replace( '&', '&', urldecode( $this->get_api_return_url( $order ) ) ); 
  280.           } 
  281.  
  282.           // Save payment data. 
  283.           update_post_meta( $order->id, '_wc_cielo_card_brand', $card_brand ); 
  284.       } 
  285.  
  286.       if ( $valid && $payment_url ) { 
  287.           return array( 
  288.               'result' => 'success',  
  289.               'redirect' => $payment_url 
  290.  ); 
  291.       } else { 
  292.           return array( 
  293.               'result' => 'fail',  
  294.               'redirect' => '' 
  295.  ); 
  296.       } 
  297.   }
