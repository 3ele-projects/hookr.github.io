  634.     /** 
  635.      * Return handler. 
  636.      * 
  637.      * @param WC_Order $order Order data. 
  638.      */ 
  639.     public function return_handler( $order ) { 
  640.         global $woocommerce; 
  641.  
  642.         $tid = get_post_meta( $order->id, '_transaction_id', true ); 
  643.  
  644.         if ( '' != $tid ) { 
  645.             $response = $this->api->get_transaction_data( $order, $tid, $order->id . '-' . time() ); 
  646.  
  647.             // Set the error alert. 
  648.             if ( isset( $response->mensagem ) && ! empty( $response->mensagem ) ) { 
  649.                 if ( 'yes' == $this->debug ) { 
  650.                     $this->log->add( $this->id, 'Cielo payment error: ' . print_r( $response->mensagem, true ) ); 
  651.                 } 
  652.  
  653.                 $this->helper->add_error( (string) $response->mensagem ); 
  654.             } 
  655.  
  656.             // Update the order status. 
  657.             $status = ( isset( $response->status ) && ! empty( $response->status ) ) ? intval( $response->status ) : -1; 
  658.             $order_note = "\n"; 
  659.  
  660.             if ( 'yes' == $this->debug ) { 
  661.                 $this->log->add( $this->id, 'Cielo payment status: ' . $status ); 
  662.             } 
  663.  
  664.             // For backward compatibility! 
  665.             if ( defined( 'WC_VERSION' ) && version_compare( WC_VERSION, '2.1.12', '<=' ) ) { 
  666.                 $order_note = "\n" . 'TID: ' . $tid . '.'; 
  667.             } 
  668.  
  669.             if ( isset( $response->{'forma-pagamento'} ) ) { 
  670.                 $payment_method = $response->{'forma-pagamento'}; 
  671.  
  672.                 $order_note .= "\n"; 
  673.                 $order_note .= __( 'Paid with', 'cielo-woocommerce' ); 
  674.                 $order_note .= ' '; 
  675.                 $order_note .= $this->get_payment_method_name( (string) $payment_method->bandeira ); 
  676.                 $order_note .= ' '; 
  677.  
  678.                 if ( 'A' == $payment_method->produto ) { 
  679.                     $order_note .=  __( 'debit', 'cielo-woocommerce' ); 
  680.                 } elseif ( '1' == $payment_method->produto ) { 
  681.                     $order_note .= __( 'credit at sight', 'cielo-woocommerce' ); 
  682.                 } else { 
  683.                     $order_note .= sprintf( __( 'credit %dx', 'cielo-woocommerce' ), $payment_method->parcelas ); 
  684.                 } 
  685.  
  686.                 $order_note .= '.'; 
  687.             } 
  688.             $this->process_order_status( $order, $status, $order_note ); 
  689.  
  690.             if ( defined( 'WC_VERSION' ) && version_compare( WC_VERSION, '2.1', '>=' ) ) { 
  691.                 $return_url = $this->get_return_url( $order ); 
  692.             } else { 
  693.                 $return_url = add_query_arg( 'order', $order->id, add_query_arg( 'key', $order->order_key, get_permalink( woocommerce_get_page_id( 'thanks' ) ) ) ); 
  694.             } 
  695.  
  696.             // Order cancelled. 
  697.             if ( 9 == $status ) { 
  698.                 $message = __( 'Order canceled successfully.', 'cielo-woocommerce' ); 
  699.                 if ( function_exists( 'wc_add_notice' ) ) { 
  700.                     wc_add_notice( $message ); 
  701.                 } else { 
  702.                     $woocommerce->add_message( $message); 
  703.                 } 
  704.  
  705.                 if ( function_exists( 'wc_get_page_id' ) ) { 
  706.                     $return_url = get_permalink( wc_get_page_id( 'shop' ) ); 
  707.                 } else { 
  708.                     $return_url = get_permalink( woocommerce_get_page_id( 'shop' ) ); 
  709.                 } 
  710.             } 
  711.  
  712.             wp_redirect( esc_url_raw( $return_url ) ); 
  713.             exit; 
  714.         } else { 
  715.             if ( function_exists( 'wc_get_page_id' ) ) { 
  716.                 $cart_url = get_permalink( wc_get_page_id( 'cart' ) ); 
  717.             } else { 
  718.                 $cart_url = get_permalink( woocommerce_get_page_id( 'cart' ) );

  726.     /** 
  727.      * Process a refund in WooCommerce 2.2 or later. 
  728.      * 
  729.      * @param  int    $order_id 
  730.      * @param  float  $amount 
  731.      * @param  string $reason 
  732.      * 
  733.      * @return bool|WP_Error True or false based on success, or a WP_Error object. 
  734.      */ 
  735.     public function process_refund( $order_id, $amount = null, $reason = '' ) { 
  736.         $order = new WC_Order( $order_id ); 
  737.  
  738.         if ( ! $order || ! $order->get_transaction_id() ) { 
  739.             return false; 
  740.         } 
  741.  
  742.         $diff = ( strtotime( $order->order_date ) - strtotime( current_time( 'mysql' ) ) ); 
  743.         $days = absint( $diff / ( 60 * 60 * 24 ) ); 
  744.         $limit = 120; 
  745.  
  746.         if ( $limit > $days ) { 
  747.             $tid = $order->get_transaction_id(); 
  748.             $amount = wc_format_decimal( $amount ); 
  749.             $response = $this->api->do_transaction_cancellation( $order, $tid, $order->id . '-' . time(), $amount ); 
  750.  
  751.             // Already canceled. 
  752.             if ( isset( $response->mensagem ) && ! empty( $response->mensagem ) ) { 
  753.                 $order->add_order_note( __( 'Cielo', 'cielo-woocommerce' ) . ': ' . sanitize_text_field( $response->mensagem ) ); 
  754.  
  755.                 return new WP_Error( 'cielo_refund_error', sanitize_text_field( $response->mensagem ) ); 
  756.             } else { 
  757.                 if ( isset( $response->cancelamentos->cancelamento ) ) { 
  758.                     $order->add_order_note( sprintf( __( 'Cielo: %s - Refunded amount: %s.', 'cielo-woocommerce' ), sanitize_text_field( $response->cancelamentos->cancelamento->mensagem ), wc_price( $response->cancelamentos->cancelamento->valor / 100 ) ) ); 
  759.                 } 
  760.  
  761.                 return true; 
  762.             } 
  763.  
  764.         } else { 
  765.             return new WP_Error( 'cielo_refund_error', sprintf( __( 'This transaction has been made **more than %s days and therefore it can not be canceled', 'cielo-woocommerce' ), $limit ) ); 
  766.         } 
  767.  
  768.         return false; 
  769.     }
