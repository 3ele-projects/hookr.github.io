  438.   private function notify() {

  440.       $sisow = new Sisow($this->settings['merchantid'], $this->settings['merchantkey'], $this->settings['shopid']);

  443.        
  444.       if(empty($trxid)) 
  445.           $trxid = $this->trxid; 
  446.  
  447.       if (($order->status != 'processing' && $order->status != 'completed') || $sisow->status == Sisow::statusReversed || $sisow->status == Sisow::statusRefunded) { 
  448.           if (($ex = $sisow->StatusRequest($trxid)) < 0) { 
  449.               echo 'fail' . $ex; 
  450.               exit; 
  451.           } else { 
  452.               if(($sisow->status == 'Pending' || $sisow->status == 'Open') && ($this->paymentcode != 'overboeking' && $this->paymentcode != 'ebill')) 
  453.                   exit('Order still pending'); 
  454.

  566. function sisow_payment_fee($cart) { 
  567.   global $woocommerce; 
  568.    
  569.   if (isset($_POST['payment_method'])) { 
  570.       $gw = $_POST['payment_method']; 
  571.   } elseif (isset($_POST['post_data'])) { 
  572.       $parsed = array(); 
  573.       parse_str($_POST['post_data'], $parsed); 
  574.  
  575.       if (isset($parsed['payment_method'])) { 
  576.           $gw = $parsed['payment_method']; 
  577.       } 
  578.   } elseif (isset($woocommerce->checkout->posted) && !empty($woocommerce->checkout->posted)) { 
  579.       $gw = $woocommerce->checkout->posted['payment_method']; 
  580.   } 
  581.        
  582.   if (isset($gw) && strpos($gw, 'sisow') !== false) { 
  583.       $class = 'WC_Sisow_' . str_replace('sisow', '', $gw); 
  584.  
  585.       if (!class_exists($class)) 
  586.           return; 
  587.  
  588.       $payment = new $class; 
  589.  
  590.       $settings = $payment->settings; 
  591.       $paymentFee = 0; 
  592.       $cart_total = $cart->subtotal + $cart->shipping_total; 
  593.       $add_note = false; 
  594.  
  595.       $paymentFee = calculate_fee_for($settings, $cart_total); 
  596.  
  597.       if ($paymentFee > 0) { 
  598.           $tax_amount = calculate_tax($settings, $paymentFee); 
  599.  
  600.           if (get_option('woocommerce_tax_display_cart') == 'incl') { 
  601.               $paymentFee += $tax_amount; 
  602.           } 
  603.  
  604.           if ($tax_amount > 0) { 
  605.  
  606.               if (get_option('woocommerce_prices_include_tax') == 'yes') { 
  607.                   $paymentFee -= $tax_amount; 
  608.  
  609.                   if (get_option('woocommerce_tax_display_cart') == 'excl') { 
  610.                       $add_note = true; 
  611.                   } 
  612.               } 
  613.  
  614.               if (get_option('woocommerce_tax_display_cart') == 'excl') { 
  615.                   $cart_taxes = $cart->taxes; 
  616.                   $keys = array_keys($cart_taxes); 
  617.  
  618.                   if (isset($keys[0])) { 
  619.                       $cart_taxes[$keys[0]] += $tax_amount; 
  620.                   } else { 
  621.                       $cart_taxes[0] = $tax_amount; 
  622.                   } 
  623.  
  624.                   $cart->taxes = $cart_taxes; 
  625.                   $cart->tax_total += $tax_amount; 
  626.               } 
  627.           } 
  628.  
  629.           if (function_exists('get_product')) { 
  630.               $cart->fee_total = $cart->fee_total + $paymentFee; 
  631.           } 
  632.            
  633.           if (!function_exists('get_product')) { 
  634.               $cart->cart_contents_total = $cart->cart_contents_total + $paymentFee; 
  635.           } 
  636.  
  637.           $paymentfeelabel = (isset($settings['paymentfeelabel']) && $settings['paymentfeelabel'] != '') ? $settings['paymentfeelabel'] : 'Payment Fee'; 
  638.  
  639.           if (method_exists($woocommerce->cart, 'add_fee')) { 
  640.               $cart->add_fee($paymentfeelabel, $paymentFee, true, $settings['paymentfeetax']); 
  641.           } 
  642.  
  643.           if (isset($woocommerce->checkout->posted) && !empty($woocommerce->checkout->posted)) { 
  644.               $woocommerce->checkout->posted['processing_fee_label'] = $paymentfeelabel; 
  645.               $woocommerce->checkout->posted['processing_fee'] = $paymentFee; 
  646.               $woocommerce->checkout->posted['processing_fee_tax'] = $tax_amount; 
  647.           } 
  648.       } 
  649.   }

  737. function process_shop_order($post_id, $post) { 
  738.   $old_fee = get_post_meta($post_id, '_processing_fee', true); 
  739.   $new_fee = stripslashes($_POST['_processing_fee']); 
  740.   $order_total = get_post_meta($post_id, '_order_total', true); 
  741.  
  742.   if ($old_fee != '') { 
  743.       if ($new_fee == 0 || $new_fee == '') { 
  744.           // if there is a processing fee which is now being removed 
  745.           $order_total -= $old_fee; 
  746.       } else { 
  747.           // adding or subtracting of fee 
  748.           if ($old_fee > $new_fee) { 
  749.               // fee lessened 
  750.               $diff = $old_fee - $new_fee; 
  751.               $order_total -= $diff; 
  752.           } elseif ($old_fee < $new_fee) { 
  753.               // fee added 
  754.               $diff = $new_fee - $old_fee; 
  755.               $order_total += $diff; 
  756.           } 
  757.       } 
  758.  
  759.       update_post_meta($post_id, '_order_total', $order_total); 
  760.       update_post_meta($post_id, '_processing_fee', $new_fee); 
  761.   } 
  762. }
