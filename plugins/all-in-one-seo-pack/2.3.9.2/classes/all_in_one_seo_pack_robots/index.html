  472.                     'strict' => false,  
  473.  ); 
  474.             } 
  475.         } 
  476.  
  477.         return $rules;

  480.     function parse_annotated_robots( $robots ) { 
  481.         $state = 0; 
  482.         $rules = Array(); 
  483.         $opts = Array( 'sitemap', 'crawl-delay', 'user-agent', 'allow', 'disallow', 'comment' ); 
  484.         $rule = Array(); 
  485.         foreach ( $opts as $o ) { 
  486.             $rule[ $o ] = Array(); 
  487.         } 
  488.         $blank_rule = $rule; 
  489.         foreach ( $robots as $l ) { 
  490.             switch ( $l['type'] ) { 
  491.                 case 'blank': 
  492.                     if ( $state >= 1 ) { 
  493.                         if ( ( $state === 1 ) && ( empty( $rule['user-agent'] ) ) ) { 
  494.                             $rule['user-agent'] = Array( null ); 
  495.                         } 
  496.                         $rules[] = $rule; 
  497.                         $rule = $blank_rule; 
  498.                     } 
  499.                     continue; 
  500.                 case 'comment': 
  501.                     $rule['comment'][] = $l['content']; 
  502.                     continue; 
  503.                 case 'sitemap': 
  504.                     $rule['sitemap'][] = trim( substr( $l['content'], 8 ) ); 
  505.                     break; 
  506.                 case 'crawl-delay': 
  507.                     $rule['crawl-delay'][] = trim( substr( $l['content'], 12 ) ); 
  508.                     break; 
  509.                 case 'user-agent': 
  510.                     if ( $l['strict'] ) { 
  511.                         $ua = trim( substr( $l['content'], 11 ) ); 
  512.                     } else { 
  513.                         $ua = trim( substr( $l['content'], 10 ) ); 
  514.                     } 
  515.                     $rule['user-agent'][] = $ua; 
  516.                     break; 
  517.                 case 'disallow': 
  518.                     if ( $l['valid'] ) { 
  519.                         $rule['disallow'][] = trim( substr( $l['content'], 9 ) ); 
  520.                         break; 
  521.                     } 
  522.                     continue; 
  523.                 case 'allow': 
  524.                     if ( $l['valid'] ) { 
  525.                         $rule['allow'][] = trim( substr( $l['content'], 6 ) ); 
  526.                         break; 
  527.                     } 
  528.                     continue; 
  529.                 case 'unknown': 
  530.                 default: 
  531.             } 
  532.             $state = $l['state']; 
  533.         } 
  534.         if ( ( $state === 1 ) && ( empty( $rule['user-agent'] ) ) ) { 
  535.             $rule['user-agent'] = Array( null ); 
  536.         } 
  537.         if ( $state >= 1 ) { 
  538.             $rules[] = $rule; 
  539.         } 
  540.  
  541.         return $rules; 
  542.     }

  548.     function get_robot_user_agents( $rules ) { 
  549.         $opts = Array( 'sitemap', 'crawl-delay', 'user-agent', 'allow', 'disallow', 'comment' ); 
  550.         $user_agents = Array(); 
  551.         foreach ( $rules as $r ) { 
  552.             if ( ! empty( $r['sitemap'] ) && empty( $r['user-agent'] ) ) { 
  553.                 $r['user-agent'] = Array( null ); 
  554.             } 
  555.             foreach ( $r['user-agent'] as $ua ) { 
  556.                 if ( ! isset( $user_agents[ $ua ] ) ) { 
  557.                     $user_agents[ $ua ] = Array(); 
  558.                 } 
  559.                 foreach ( $opts as $o ) { 
  560.                     if ( ! isset( $user_agents[ $ua ][ $o ] ) ) { 
  561.                         $user_agents[ $ua ][ $o ] = $r[ $o ]; 
  562.                     } else { 
  563.                         $user_agents[ $ua ][ $o ] = array_merge( $user_agents[ $ua ][ $o ], $r[ $o ] ); 
  564.                     } 
  565.                 } 
  566.             } 
  567.         } 
  568.  
  569.         return $user_agents; 
  570.     }

  582.     function opt_robot_rule( $dis ) { 
  583.         if ( is_array( $dis ) ) { // unique rules only 
  584.             $dis = array_unique( $dis, SORT_STRING ); 
  585.             $pd = null; 
  586.             foreach ( $dis as $k => $d ) { 
  587.                 $d = trim( $d ); 
  588.                 if ( ! empty( $pd ) && ! empty( $d ) ) { 
  589.                     if ( strpos( $d, $pd ) === 0 ) { 
  590.                         unset( $dis[ $k ] ); 
  591.                         continue; // get rid of subpaths of $pd 
  592.                     } 
  593.                 } 
  594.                 $l = strlen( $d ); 
  595.                 if ( ( $l > 0 ) && ( $d[ $l - 1 ] !== '/' ) ) { 
  596.                     continue; 
  597.                 } 
  598.                 $pd = $d; // only allow directory paths for $pd 
  599.             } 
  600.         } 
  601.  
  602.         return $dis; 
  603.     }
