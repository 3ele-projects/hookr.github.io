  411.     public function parse_readme() { 
  412.         $file = file_exists( NF_PLUGIN_DIR . 'readme.txt' ) ? NF_PLUGIN_DIR . 'readme.txt' : null; 
  413.  
  414.         if ( ! $file ) { 
  415.             $readme = '<p>' . __( 'No valid changelog was found.', 'ninja-forms' ) . '</p>'; 
  416.         } else { 
  417.             $readme = file_get_contents( $file ); 
  418.             $readme = nl2br( esc_html( $readme ) ); 
  419.  
  420.             $readme = explode( '== Changelog ==', $readme ); 
  421.                         $readme = end( $readme ); 
  422.  
  423.             $readme = preg_replace( '/`(.*?)`/', '<code>\\1</code>', $readme ); 
  424.             $readme = preg_replace( '/[\040]\*\*(.*?)\*\*/', ' <strong>\\1</strong>', $readme ); 
  425.             $readme = preg_replace( '/[\040]\*(.*?)\*/', ' <em>\\1</em>', $readme ); 
  426.             $readme = preg_replace( '/= (.*?) =/', '<h4>\\1</h4>', $readme ); 
  427.             $readme = preg_replace( '/\[(.*?)\]\((.*?)\)/', '<a href="\\2">\\1</a>', $readme );
