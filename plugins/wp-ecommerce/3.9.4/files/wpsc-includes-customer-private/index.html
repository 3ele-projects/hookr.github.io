  156. /**

  166. function _wpsc_create_customer_id_cookie( $id, $fake_it = false ) { 
  167.  
  168.   do_action( '_wpsc_create_customer_id_cookie', $id, $fake_it ); 
  169.  
  170.   $expire = time() + WPSC_CUSTOMER_DATA_EXPIRATION; // valid for 48 hours

  177.

  180.       $_COOKIE[ WPSC_CUSTOMER_COOKIE ] = $cookie; 
  181.   } else { 
  182.       _wpsc_set_customer_cookie( $cookie, $expire );

  187. /** 
  188.  * Compute the security hash for a WPeC visitor based on the given parameters 
  189.  * 
  190.  * @access private 
  191.  * 
  192.  * @since 3.8.14 
  193.  * 
  194.  * @param int        $visitor_id             id of the visitor 
  195.  * @param int        $expiration                Unix timestamp being attached to the visitor 
  196.  * @param string    $visitor_security_key    random string generated at the time the visitor was created, and stored with the visitor profile 
  197.  * 
  198.  * @return string                      security hash 
  199.  */ 
  200. function _wpsc_get_visitor_hash( $visitor_id, $expiration, $visitor_security_key ) { 
  201.   $second_hash_op_key = hash_hmac( 'md5', $visitor_security_key, WPSC_SALT ); 
  202.   $visitor_hash = hash_hmac( 'md5', $visitor_id . $expiration , $second_hash_op_key ); 
  203.   return $visitor_hash; 
  204. }

  206. /** 
  207.  * Make sure the customer cookie is not compromised. 
  208.  * 
  209.  * @access public 
  210.  * @since 3.8.9 
  211.  * @return mixed Return the customer ID if the cookie is valid, false if otherwise. 
  212.  */ 
  213. function _wpsc_validate_customer_cookie() { 
  214.  
  215.   do_action( '_wpsc_validate_customer_cookie' ); 
  216.  
  217.   if ( ! isset( $_COOKIE[ WPSC_CUSTOMER_COOKIE ] ) ) { 
  218.       return false; 
  219.   } 
  220.  
  221.   $cookie = $_COOKIE[ WPSC_CUSTOMER_COOKIE ]; 
  222.  
  223.   list( $id, $expire, $visitor_hash_from_cookie ) = $x = explode( '|', $cookie ); 
  224.  
  225.   // check to see if the ID is valid, it must be an integer, empty test is because old versions of php 
  226.   // can return true on empty string 
  227.   if ( ! empty( $id ) &&  is_numeric( $id ) ) { 
  228.       $id = intval( $id ); 
  229.       $security_key = _wpsc_visitor_security_key( $id ); 
  230.  
  231.       // if a user is found keep checking, user not found clear the cookie and return invalid 
  232.       if ( ! empty( $security_key ) ) { 
  233.           $recomputed_visitor_hash = _wpsc_get_visitor_hash( $id, $expire, $security_key ); 
  234.  
  235.           // integrity check 
  236.           if ( $visitor_hash_from_cookie === $recomputed_visitor_hash ) { 
  237.               return $id; 
  238.           } 
  239.       } 
  240.   } 
  241.  
  242.   // if we get to here the cookie or user is not valid 
  243.   return _wpsc_unset_customer_cookie(); 
  244. }

  425. /** 
  426.  * Is the user an automata not worthy of a WPEC profile to hold shopping cart and other info 
  427.  * 
  428.  * @access private 
  429.  * @since  3.8.13 
  430.  */ 
  431. function _wpsc_is_bot_user() { 
  432.  
  433.   $is_bot = false; 
  434.  
  435.   // if the customer cookie is invalid, unset it 
  436.   $visitor_id_from_cookie = _wpsc_validate_customer_cookie(); 
  437.   if ( $visitor_id_from_cookie ) { 
  438.       return $visitor_id_from_cookie === WPSC_BOT_VISITOR_ID; 
  439.   } 
  440.  
  441.   if ( ! is_user_logged_in() ) { 
  442.  
  443.       // check for WordPress detected 404 or feed request 
  444.       if ( did_action( 'posts_selection' ) ) { 
  445.           if ( is_feed() ) { 
  446.               $is_bot = true; 
  447.           } 
  448.  
  449.           if ( is_404() ) { 
  450.               $is_bot = true; 
  451.           } 
  452.       } 
  453.  
  454.       // Check for non WPEC ajax request, no reason to create a visitor profile if this is the case 
  455.       // Although the AJAX request may not have originated from a bot, from WPeC's perspective we 
  456.       // treat it as such to avoid creating a WPeC visitor profile for an AJAX rerquest that doesn't need 
  457.       // WPeC functionality. 
  458.       if ( ! $is_bot && ( defined( 'DOING_AJAX' ) && DOING_AJAX ) && ! _wpsc_doing_wpsc_ajax_request() ) { 
  459.           $is_bot = true; 
  460.       } 
  461.  
  462.       if ( ! $is_bot && ( strpos( $_SERVER['REQUEST_URI'], '?wpsc_action=rss' ) ) ) { 
  463.           $is_bot = true; 
  464.       } 
  465.  
  466.       // Cron jobs are not flesh originated 
  467.       if ( ! $is_bot && ( defined( 'DOING_CRON' ) && DOING_CRON ) ) { 
  468.           $is_bot = true; 
  469.       } 
  470.  
  471.       // XML RPC requests are probably from cybernetic beasts 
  472.       if ( ! $is_bot && ( defined( 'XMLRPC_REQUEST' ) && XMLRPC_REQUEST ) ) { 
  473.           $is_bot = true; 
  474.       } 
  475.  
  476.       // coming to login first, after the user logs in we know they are a live being, until then they are something else 
  477.       if ( ! $is_bot && ( strpos( $_SERVER['PHP_SELF'], 'wp-login' ) || strpos( $_SERVER['PHP_SELF'], 'wp-register' ) ) ) { 
  478.           $is_bot = true; 
  479.       } 
  480.  
  481.       if ( ! $is_bot && ( ! empty( $_SERVER['HTTP_USER_AGENT']) ) ) { 
  482.  
  483.           // the user agent could be google bot, bing bot or some other bot, one would hope real user agents do not have the 
  484.           // string 'bot|spider|crawler|preview' in them, there are bots that don't do us the kindness of identifying themselves as such,  
  485.           // check for the user being logged in in a real user is using a bot to access content from our site 
  486.           $bot_agent_strings = array( 'robot', 'bot', 'crawler', 'spider', 'preview', 'WordPress', ); 
  487.           $bot_agent_strings = apply_filters( 'wpsc_bot_user_agents', $bot_agent_strings ); 
  488.  
  489.           foreach ( $bot_agent_strings as $bot_agent_string ) { 
  490.               if ( stripos( $_SERVER['HTTP_USER_AGENT'], $bot_agent_string ) !== false ) { 
  491.                   $is_bot = true; 
  492.                   break; 
  493.               } 
  494.           } 
  495.       } 
  496.   } 
  497.  
  498.   $is_bot = apply_filters( 'wpsc_is_bot_user', $is_bot ); 
  499.  
  500.   return $is_bot; 
  501. }
