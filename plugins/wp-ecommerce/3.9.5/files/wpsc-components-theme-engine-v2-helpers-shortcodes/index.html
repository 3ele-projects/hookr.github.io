  7. /** 
  8.  * Cart shortcode. 
  9.  * 
  10.  * Usage example: [wpsc-cart type="widget"] 
  11.  * 
  12.  * Available attributes: 
  13.  *     - 'type': Can be set to 'form', 'widget', 'table'. 
  14.  *               'form' would display similarly to the Shopping Cart page 
  15.  *               'widget' would display similarly to the Cart widget 
  16.  *               'table' would display a plain table of items without form 
  17.  * 
  18.  * @param  array $atts    Attributes 
  19.  * @return string         Shortcode output 
  20.  */ 
  21. function _wpsc_shortcode_cart( $atts ) { 
  22.   global $wpsc_cart; 
  23.  
  24.   ob_start(); 
  25.  
  26.   $defaults = array( 
  27.       'type' => 'form' 
  28.  ); 
  29.  
  30.   $atts = shortcode_atts( $defaults, $atts, 'wpsc-cart' ); 
  31.  
  32.   require_once( WPSC_TE_V2_CLASSES_PATH . '/cart-item-table-widget-form.php' ); 
  33.  
  34.   if ( ! count( $wpsc_cart->cart_items ) ) { 
  35.       return '<p>' . __( 'No items in cart.', 'wpsc' ) . '</p>'; 
  36.   } 
  37.  
  38.   switch ( $atts['type'] ) { 
  39.       case 'form' : 
  40.           $table = new WPSC_Cart_Item_Table_Form(); 
  41.           break; 
  42.       case 'widget' : 
  43.           $table = new WPSC_Cart_Item_Table_Widget_Form(); 
  44.           break; 
  45.       case 'table' : 
  46.           $table = new WPSC_Cart_Item_Table(); 
  47.           break; 
  48.   } 
  49.  
  50.   $table->display(); 
  51.   return ob_get_clean(); 
  52.  
  53. }

  55. /** 
  56.  * Products shortcode 
  57.  * 
  58.  * Usage example: 
  59.  *     - [wpsc-products id="12"] 
  60.  *     - [wpsc-products id="12, 15"] 
  61.  *     - [wpsc-products display="sale"] 
  62.  * 
  63.  * Available attributes: 
  64.  * 
  65.  *     - 'id': A single, or a comma separated list of product IDs. Set to '0' will 
  66.  *             display all products matching remaining criteria. Defaults to '0'. 
  67.  *     - 'display': Defaults to 'all' 
  68.  *         + display="all" will display all products 
  69.  *         + display="sale" will only display products on sale 
  70.  *         + display="not-sale" will only display products not on sale 
  71.  *     - 'per_page': Set to -1 to disable pagination. Defaults to -1. 
  72.  *     - 'paged': Current page number 
  73.  *     - 'offset': Skip how many products at the beginning of the found results 
  74.  *     - 'template_part': Which template part to include to display this shortcode. 
  75.  *                        Defaults to 'loop-products' 
  76.  *     - 'category_in': Comma separated list of categories whose products you want to display 
  77.  *     - 'category_not_in': Comma separated list of categories you want to exclude 
  78.  *     - 'tag_in': Comma separated list of tags whose products you want to display 
  79.  *     - 'tag_not_in': Comma separated list of tags whose products you want to exclude 
  80.  * 
  81.  * @param  array $atts Attributes 
  82.  * @return string      Shortcode output 
  83.  */ 
  84. function _wpsc_shortcode_products( $atts ) { 
  85.   // Default and allowed attributes 
  86.   $defaults = array( 
  87.       'id' => 0,  
  88.       'display' => 'all',  
  89.       'per_page' => -1,  
  90.       'paged' => 1,  
  91.       'offset' => 0,  
  92.       'template_part' => 'loop-products',  
  93.       'category_in' => '',  
  94.       'category_not_in' => '',  
  95.       'tag_in' => '',  
  96.       'tag_not_in' => '',  
  97.  ); 
  98.  
  99.   $args = array( 'post_type' => 'wpsc-product' ); 
  100.  
  101.   $atts = shortcode_atts( $defaults, $atts, 'wpsc-products' ); 
  102.  
  103.   // Query by post ID 
  104.   $atts['id'] = str_replace( ' ', '', $atts['id'] ); 
  105.   $ids =   empty( $atts['id'] ) 
  106.          ? array() 
  107.          : array_map( 'absint', explode( ', ', $atts['id'] ) ); 
  108.  
  109.   if ( ! empty( $ids ) ) { 
  110.       $args['post__in'] = $ids; 
  111.   } 
  112.  
  113.   // Meta query for whether to select products on sale or not 
  114.   switch ( $atts['display'] ) { 
  115.       case 'sale': 
  116.           $args['meta_query'] = array( 
  117.               array( 
  118.                   'key' => '_wpsc_special_price',  
  119.                   'value' => array( '', '0' ),  
  120.                   'compare' => 'NOT IN',  
  121.  ) 
  122.  ); 
  123.           break; 
  124.  
  125.       case 'not-sale': 
  126.           $args['meta_query'] = array( 
  127.               array( 
  128.                   'key' => '_wpsc_special_price',  
  129.                   'value' => array( '', '0' ),  
  130.                   'compare' => 'IN',  
  131.  ) 
  132.  ); 
  133.           break; 
  134.   } 
  135.  
  136.   // Pagination 
  137.   if ( $atts['per_page'] !== -1 ) { 
  138.       $args['posts_per_page'] = absint( $atts['per_page'] ); 
  139.       $args['paged'] = absint( $atts['paged'] ); 
  140.       $args['offset'] = absint( $atts['offset'] ); 
  141.   } else { 
  142.       $args['nopaging'] = true; 
  143.   } 
  144.  
  145.   // Taxonomy queries 
  146.   $atts['category_in'] = str_replace( ' ', '', $atts['category_in'] ); 
  147.   $atts['category_not_in'] = str_replace( ' ', '', $atts['category_not_in'] ); 
  148.   $atts['tag_in'] = str_replace( ' ', '', $atts['tag_in'] ); 
  149.   $atts['tag_not_in'] = str_replace( ' ', '', $atts['tag_not_in'] ); 
  150.  
  151.   $atts['category_in'] =   empty( $atts['category_in'] ) 
  152.                              ? array() 
  153.                              : array_map( 'absint', explode( ', ', $atts['category_in'] ) ); 
  154.  
  155.   $atts['category_not_in'] =   empty( $atts['category_not_in'] ) 
  156.                              ? array() 
  157.                              : array_map( 'absint', explode( ', ', $atts['category_not_in'] ) ); 
  158.  
  159.   $atts['tag_in'] =   empty( $atts['tag_in'] ) 
  160.                              ? array() 
  161.                              : array_map( 'absint', explode( ', ', $atts['tag_in'] ) ); 
  162.  
  163.   $atts['tag_not_in'] =   empty( $atts['tag_not_in'] ) 
  164.                              ? array() 
  165.                              : array_map( 'absint', explode( ', ', $atts['tag_not_in'] ) ); 
  166.  
  167.   $args['tax_query'] = array(); 
  168.  
  169.   // Category slug in 
  170.   if ( ! empty( $atts['category_in'] ) ) { 
  171.       $args['tax_query'][] = array( 
  172.           'taxonomy' => 'wpsc_product_category',  
  173.           'field' => 'slug',  
  174.           'terms' => $atts['category_in'],  
  175.           'operator' => 'IN',  
  176.  ); 
  177.   } 
  178.  
  179.   // Category slug not in 
  180.   if ( ! empty( $atts['category_not_in'] ) ) { 
  181.       $args['tax_query'][] = array( 
  182.           'taxonomy' => 'wpsc_product_category',  
  183.           'field' => 'slug',  
  184.           'terms' => $atts['category_not_in'],  
  185.           'operator' => 'NOT IN',  
  186.  ); 
  187.   } 
  188.  
  189.   // Product tag in 
  190.   if ( ! empty( $atts['tag_in'] ) ) { 
  191.       $args['tax_query'][] = array( 
  192.           'taxonomy' => 'product_tag',  
  193.           'field' => 'slug',  
  194.           'terms' => $atts['tag_in'],  
  195.           'operator' => 'IN',  
  196.  ); 
  197.   } 
  198.  
  199.   // Product tag not in 
  200.   if ( ! empty( $atts['tag_not_in'] ) ) { 
  201.       $args['tax_query'][] = array( 
  202.           'taxonomy' => 'product_tag',  
  203.           'field' => 'slug',  
  204.           'terms' => $atts['tag_not_in'],  
  205.           'operator' => 'NOT IN',  
  206.  ); 
  207.   } 
  208.  
  209.   // I don't like query posts either but we need to preserve the ability 
  210.   // to use the_post() from within the templates, without having to resort 
  211.   // to tags like $query->the_post() 
  212.   query_posts( $args ); 
  213.  
  214.   ob_start(); 
  215.  
  216.   wpsc_get_template_part( $atts['template_part'] ); 
  217.  
  218.   $output = ob_get_clean(); 
  219.  
  220.   wp_reset_query(); 
  221.  
  222.   return $output; 
  223. }

  225. /** 
  226.  * Add to cart shortcode 
  227.  * 
  228.  * Usage example: [wpsc-add-to-cart id="35"] 
  229.  * 
  230.  * Available attributes: 
  231.  *  - id: Post ID of the product for which you want to display the add to cart form 
  232.  * @param  array $atts Attributes 
  233.  * @return string      Output 
  234.  */ 
  235. function _wpsc_shortcode_add_to_cart( $atts ) { 
  236.   $defaults = array( 'id' => 0 ); 
  237.   $atts = array_map( 'absint', shortcode_atts( $defaults, $atts ) ); 
  238.  
  239.   if ( empty( $atts['id'] ) ) { 
  240.       return ''; 
  241.   } 
  242.  
  243.   return wpsc_get_add_to_cart_form( $atts['id'] ); 
  244. }
