  212. class wpsc_custom_meta { 
  213.   // Custom meta values 
  214.   var $custom_meta; 
  215.   var $custom_meta_count = 0; 
  216.   var $current_custom_meta = -1; 
  217.   var $custom_meta_values; 
  218.  
  219.   function __construct( $post_id ) { 
  220.  
  221.       $cleaned_metas = array(); 
  222.  
  223.       if ( ! empty( $post_id ) ) { 
  224.           $meta_values = get_post_meta( $post_id ); 
  225.  
  226.           foreach ( $meta_values as $key => $values ) { 
  227.               if ( ! is_protected_meta( $key, 'wpsc-product' ) ) { 
  228.                   if ( is_array( $values ) ) { 
  229.                       foreach ( $values as $value ) { 
  230.                           $cleaned_metas[] = array( 'meta_key' => $key, 'meta_value' => $value ); 
  231.                       } 
  232.                   } 
  233.               } 
  234.           } 
  235.       } 
  236.  
  237.       $this->custom_meta = $cleaned_metas; 
  238.       $this->custom_meta_count = count( $this->custom_meta ); 
  239.   } 
  240.  
  241.   function have_custom_meta() { 
  242.       if (($this->current_custom_meta + 1) < $this->custom_meta_count) { 
  243.           return true; 
  244.       } else if ($this->current_custom_meta + 1 == $this->custom_meta_count && $this->custom_meta_count > 0) { 
  245.           $this->rewind_custom_meta(); 
  246.       } 
  247.       return false; 
  248.   } 
  249.  
  250.   /** 
  251.    * Custom Meta Loop Code Starts here 
  252.   */ 
  253.   function next_custom_meta() { 
  254.       $this->current_custom_meta++; 
  255.       $this->custom_meta_values = $this->custom_meta[$this->current_custom_meta]; 
  256.       return $this->custom_meta_values; 
  257.   } 
  258.  
  259.  
  260.   function the_custom_meta() { 
  261.       $this->custom_meta_values = $this->next_custom_meta(); 
  262.       return $this->custom_meta_values; 
  263.   } 
  264.  
  265.   function rewind_custom_meta() { 
  266.       if ($this->custom_meta_count > 0) {

  268.       } 
  269.   }
