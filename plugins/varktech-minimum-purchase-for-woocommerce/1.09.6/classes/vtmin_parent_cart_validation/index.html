  298.   /** ************************************************ 
  299.   **   Application - get current page url 
  300.   *        
  301.   *       The code checking for 'www.' is included since 
  302.   *       some server configurations do not respond with the 
  303.   *       actual info, as to whether 'www.' is part of the  
  304.   *       URL.  The additional code balances out the currURL,  
  305.   *       relative to the Parent Plugin's recorded URLs            
  306.   *************************************************** */  
  307.  public  function vtmin_currPageURL() { 
  308.      global $vtmin_info; 
  309.      $currPageURL = $this->vtmin_get_currPageURL(); 
  310.      $www = 'www.'; 
  311.       
  312.      $curr_has_www = 'no'; 
  313.      if (strpos($currPageURL, $www )) { 
  314.          $curr_has_www = 'yes'; 
  315.      } 
  316.       
  317.      //use checkout URL as an example of all setup URLs 
  318.      $checkout_has_www = 'no'; 
  319.      if (strpos($vtmin_info['woo_checkout_url'], $www )) { 
  320.          $checkout_has_www = 'yes'; 
  321.      }      
  322.           
  323.      switch( true ) { 
  324.         case ( ($curr_has_www == 'yes') && ($checkout_has_www == 'yes') ): 
  325.         case ( ($curr_has_www == 'no')  && ($checkout_has_www == 'no') ):  
  326.             //all good, no action necessary 
  327.           break; 
  328.         case ( ($curr_has_www == 'no') && ($checkout_has_www == 'yes') ): 
  329.             //reconstruct the URL with 'www.' included. 
  330.             $currPageURL = $this->vtmin_get_currPageURL($www);  
  331.           break; 
  332.         case ( ($curr_has_www == 'yes') && ($checkout_has_www == 'no') ):  
  333.             //all of the woo URLs have no 'www.', and curr has it, so remove the string  
  334.             $currPageURL = str_replace($www, "", $currPageURL); 
  335.           break; 
  336.      }  
  337.   
  338.      return $currPageURL; 
  339.   }
