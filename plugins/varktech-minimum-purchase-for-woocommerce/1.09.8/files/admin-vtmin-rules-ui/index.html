  637. public function vtmin_build_checkbox_contents ($taxonomy, $tax_class, $checked_list = NULL) { 
  638.       global $wpdb, $vtmin_info;          
  639.       $sql = "SELECT terms.`term_id`, terms.`name`  FROM `" . $wpdb->prefix . "terms` as terms, `" . $wpdb->prefix . "term_taxonomy` as term_taxonomy WHERE terms.`term_id` = term_taxonomy.`term_id` AND term_taxonomy.`taxonomy` = '" . $taxonomy . "' ORDER BY terms.`term_id` ASC";                          
  640.           $categories = $wpdb->get_results($sql, ARRAY_A) ; 
  641.

  647.           $output  .= 'value="'.$category['term_id'].'" '; 
  648.           if ($checked_list) { 
  649.               if (in_array($category['term_id'], $checked_list)) {   //if cat_id is in previously checked_list   
  650.                  $output  .= 'checked="checked"';

  658.           $output  .= '</label>';             
  659.           $output  .= '</li>'; 
  660.             echo $output ; 
  661.        }

  663.   }
