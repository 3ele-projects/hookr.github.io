  516.

  522.             } else {

  524.                 $selectorsMultiplied = array(); 
  525.  
  526.                 // merge the current list of non parent selector elements 
  527.                 // on to the current list of selectors to add 
  528.                 if( $currentElements ) { 
  529.                     $this->mergeElementsOnToSelectors( $currentElements, $newSelectors); 
  530.                 } 
  531.  
  532.                 // loop through our current selectors 
  533.                 foreach($newSelectors as $sel) { 
  534.  
  535.                     // if we don't have any parent paths, the & might be in a mixin so that it can be used 
  536.                     // whether there are parents or not

  539.                         // it is not lost

  597.  
  598.                 // our new selectors has been multiplied, so reset the state 
  599.                 $newSelectors = $selectorsMultiplied; 
  600.                 $currentElements = array(); 
  601.             } 
  602.         } 
  603.  
  604.         // if we have any elements left over (e.g. .a& .b == .b) 
  605.         // add them on to all the current selectors 
  606.         if( $currentElements ) { 
  607.             $this->mergeElementsOnToSelectors($currentElements, $newSelectors); 
  608.         } 
  609.         foreach( $newSelectors as $new_sel) { 
  610.             if( $new_sel ) { 
  611.                 $paths[] = $new_sel; 
  612.             } 
  613.         } 
  614.     }

  616.     function mergeElementsOnToSelectors( $elements, &$selectors) { 
  617.  
  618.         if( !$selectors ) { 
  619.             $selectors[] = array( new Less_Tree_Selector($elements) ); 
  620.             return; 
  621.         } 
  622.  
  623.  
  624.         foreach( $selectors as &$sel) { 
  625.  
  626.             // if the previous thing in sel is a parent this needs to join on to it 
  627.             if( $sel ) { 
  628.                 $last = count($sel)-1; 
  629.                 $sel[$last] = $sel[$last]->createDerived( array_merge($sel[$last]->elements, $elements) ); 
  630.             }else{ 
  631.                 $sel[] = new Less_Tree_Selector( $elements ); 
  632.             } 
  633.         }
