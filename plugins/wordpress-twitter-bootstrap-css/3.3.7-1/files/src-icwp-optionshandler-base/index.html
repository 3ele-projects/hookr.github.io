  687.       protected function convertIpListForDisplay( $inaIpList = array() ) { 
  688.  
  689.           $aDisplay = array(); 
  690.           if ( empty( $inaIpList ) || empty( $inaIpList['ips'] ) ) { 
  691.               return $aDisplay; 
  692.           } 
  693.           foreach( $inaIpList['ips'] as $sAddress ) { 
  694.               // offset=1 in the case that it's a range and the first number is negative on 32-bit systems 
  695.               $mPos = strpos( $sAddress, '-', 1 ); 
  696.  
  697.               if ( $mPos === false ) { //plain IP address 
  698.                   $sDisplayText = long2ip( $sAddress ); 
  699.               } 
  700.               else { 
  701.                   //we remove the first character in case this is '-' 
  702.                   $aParts = array( substr( $sAddress, 0, 1 ), substr( $sAddress, 1 ) ); 
  703.                   list( $nStart, $nEnd ) = explode( '-', $aParts[1], 2 ); 
  704.                   $sDisplayText = long2ip( $aParts[0].$nStart ) .'-'. long2ip( $nEnd ); 
  705.               } 
  706.               $sLabel = $inaIpList['meta'][ md5($sAddress) ]; 
  707.               $sLabel = trim( $sLabel, '()' );

  761.       /** 
  762.        * @param string $sAllOptionsInput - comma separated list of all the input keys to be processed from the $_POST 
  763.        * @return void|boolean 
  764.        */ 
  765.       public function updatePluginOptionsFromSubmit( $sAllOptionsInput ) { 
  766.           if ( empty( $sAllOptionsInput ) ) { 
  767.               return; 
  768.           } 
  769.           $oDp = $this->loadDataProcessor(); 
  770.           $this->loadStoredOptionsValues(); 
  771.  
  772.           $aAllInputOptions = explode( self::CollateSeparator, $sAllOptionsInput ); 
  773.           foreach ( $aAllInputOptions as $sInputKey ) { 
  774.               $aInput = explode( ':', $sInputKey ); 
  775.               list( $sOptionType, $sOptionKey ) = $aInput; 
  776.  
  777.               if ( !$this->getIsOptionKey( $sOptionKey ) ) { 
  778.                   continue; 
  779.               } 
  780.  
  781.  
  782.               $sOptionValue = $oDp->FetchPost( $this->prefixOptionKey( $sOptionKey ) ); 
  783.               if ( is_null( $sOptionValue ) ) { 
  784.  
  785.                   if ( $sOptionType == 'text' || $sOptionType == 'email' ) { //if it was a text box, and it's null, don't update anything 
  786.                       continue; 
  787.                   } 
  788.                   else if ( $sOptionType == 'checkbox' ) { //if it was a checkbox, and it's null, it means 'N' 
  789.                       $sOptionValue = 'N'; 
  790.                   } 
  791.                   else if ( $sOptionType == 'integer' ) { //if it was a integer, and it's null, it means '0' 
  792.                       $sOptionValue = 0; 
  793.                   } 
  794.               } 
  795.               else { //handle any pre-processing we need to. 
  796.  
  797.                   if ( $sOptionType == 'text' || $sOptionType == 'email' ) { 
  798.                       $sOptionValue = trim( $sOptionValue ); 
  799.                   } 
  800.                   if ( $sOptionType == 'integer' ) { 
  801.                       $sOptionValue = intval( $sOptionValue ); 
  802.                   } 
  803.                   else if ( $sOptionType == 'password' && $this->hasEncryptOption() ) { //md5 any password fields 
  804.                       $sTempValue = trim( $sOptionValue ); 
  805.                       if ( empty( $sTempValue ) ) { 
  806.                           continue; 
  807.                       } 
  808.                       $sOptionValue = md5( $sTempValue ); 
  809.                   } 
  810.                   else if ( $sOptionType == 'ip_addresses' ) { //ip addresses are textareas, where each is separated by newline 
  811.                       $sOptionValue = $oDp->ExtractIpAddresses( $sOptionValue ); 
  812.                   } 
  813.                   else if ( $sOptionType == 'yubikey_unique_keys' ) { //ip addresses are textareas, where each is separated by newline and are 12 chars long 
  814.                       $sOptionValue = $oDp->CleanYubikeyUniqueKeys( $sOptionValue ); 
  815.                   } 
  816.                   else if ( $sOptionType == 'email' && function_exists( 'is_email' ) && !is_email( $sOptionValue ) ) { 
  817.                       $sOptionValue = ''; 
  818.                   } 
  819.                   else if ( $sOptionType == 'comma_separated_lists' ) { 
  820.                       $sOptionValue = $oDp->ExtractCommaSeparatedList( $sOptionValue ); 
  821.                   } 
  822.                   else if ( $sOptionType == 'multiple_select' ) { 
  823.                   } 
  824.               } 
  825.               $this->setOpt( $sOptionKey, $sOptionValue ); 
  826.           } 
  827.           return $this->savePluginOptions(); 
  828.       }
