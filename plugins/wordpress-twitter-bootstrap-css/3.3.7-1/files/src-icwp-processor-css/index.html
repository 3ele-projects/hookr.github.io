  162.   protected function getCssIncludeUrls() { 
  163.       /** @var ICWP_WPTB_FeatureHandler_Css $oFO */ 
  164.       $oFO = $this->getFeatureOptions(); 
  165.  
  166.       // We've cached the inclusions list so we don't work it out every page load. 
  167.       $aIncludesList = $oFO->getIncludesCache(); 
  168.       if ( !empty( $aIncludesList ) && is_array( $aIncludesList ) ) { 
  169.           return $aIncludesList; 
  170.       } 
  171.       $aIncludesList = array(); 
  172.  
  173.       // An unsupported option, so just return the custom CSS. 
  174.       $aPossibleIncludeOptions = array( 'twitter', 'twitter-legacy', 'twitter-less', 'yahoo-reset', 'yahoo-reset-3', 'normalize' ); 
  175.       $sIncludeOption = $this->getOption( 'option' ); 
  176.       if ( in_array( $sIncludeOption, $aPossibleIncludeOptions ) ) { 
  177.  
  178.           // 'twitter', 'twitter-legacy', 'yahoo-reset', 'yahoo-reset-3', 'normalize' 
  179.           switch ( $sIncludeOption ) { 
  180.               case 'normalize': 
  181.                   $aIncludesList = array( 'normalize' => $this->getUrl_NormalizeCss() ); 
  182.                   break; 
  183.               case 'yahoo-reset': 
  184.                   $aIncludesList = array( 'yahoo-reset-290' => $this->getUrl_YahooCss2() ); 
  185.                   break; 
  186.               case 'yahoo-reset-3': 
  187.                   $aIncludesList = array( 'yahoo-reset-3' => $this->getUrl_YahooCss() ); 
  188.                   break; 
  189.               default: //twitter 
  190.                   $aIncludesList = $this->getTwitterCssUrls(); 
  191.                   break; 
  192.           } 
  193.  
  194.           // At this point $aIncludesList should be an array of all the URLs to be included with their labels. 
  195.           // Now add Custom/Reset CSS. 
  196.       } 
  197.  
  198.       // finally add the custom CSS 
  199.       $sCustomCssUrl = $this->getCustomCssLink(); 
  200.       if ( !empty( $sCustomCssUrl ) ) { 
  201.           $aIncludesList[ 'custom-reset' ] = $sCustomCssUrl; 
  202.       } 
  203.  
  204.       // cache it 
  205.       $oFO->updateIncludesCache( $aIncludesList ); 
  206.       return $aIncludesList; 
  207.   }

  214.   protected function getTwitterCssUrls() { 
  215.       /** @var ICWP_WPTB_FeatureHandler_Css $oFO */ 
  216.       $oFO = $this->getFeatureOptions(); 
  217.  
  218.       $oWpFs = $this->loadFileSystemProcessor(); 
  219.       $sCssFileExtension = $this->getIsOption( 'use_minified_css', 'Y' )? '.min.css' : '.css'; 
  220.       $aUrls = array(); 
  221.  
  222.       // link to the Twitter LESS-compiled CSS (only if the files exists) 
  223.       if ( $this->getIsOption( 'option', 'twitter-less' ) ) { 
  224.           $sLessStemUrl = 'bootstrap.less'.$sCssFileExtension; 
  225.           if ( $oWpFs->exists( $this->oFeatureOptions->getPathToCss( $sLessStemUrl ) ) ) { 
  226.               $aUrls[ 'twitter-bootstrap-less' ] = $this->getCssUrl( $sLessStemUrl ); 
  227.               return $aUrls; 
  228.           } 
  229.       } 
  230.  
  231.       $sTwitterStem = $this->getBootstrapUrl( 'css/bootstrap' ).$sCssFileExtension; // default is to serve it "local" 
  232.       // Determine the Twitter URL stem based on local or if CDNJS selected 
  233.       if ( $this->getIsOption( 'use_cdnjs', 'Y' ) ) { 
  234.           $sTwitterCdnStem = '%stwitter-bootstrap/%s/css/bootstrap%s'; 
  235.           $sTwitterCdnStem = sprintf( $sTwitterCdnStem, self::CdnjsStem, $oFO->getTwitterBootstrapVersion(), $sCssFileExtension ); 
  236.           if ( $oWpFs->getIsUrlValid( 'http:'.$sTwitterCdnStem ) ) { 
  237.               $sTwitterStem = $sTwitterCdnStem; 
  238.           } 
  239.       } 
  240.       $aUrls[ 'twitter-bootstrap' ] = $sTwitterStem; 
  241.  
  242.       if ( $this->getIsOption( 'inc_responsive_css', 'Y' ) && $this->getIsOption( 'option', 'twitter-legacy' ) ) { 
  243.  
  244.           $sTwitterStem = $this->getBootstrapUrl( 'css/bootstrap-responsive' ).$sCssFileExtension; // default is to serve it "local" 
  245.           if ( $this->getIsOption( 'use_cdnjs', 'Y' ) ) { 
  246.               $sTwitterCdnStem = '%stwitter-bootstrap/%s/css/bootstrap-responsive%s'; 
  247.               $sTwitterCdnStem = sprintf( $sTwitterCdnStem, self::CdnjsStem, $oFO->getTwitterBootstrapVersion(), $sCssFileExtension ); 
  248.               if ( $oWpFs->getIsUrlValid( 'http:'.$sTwitterCdnStem ) ) { 
  249.                   $sTwitterStem = $sTwitterCdnStem; 
  250.               } 
  251.           } 
  252.           $aUrls[ 'twitter-bootstrap-responsive' ] = $sTwitterStem; 
  253.       }

  269.   public function doEnqueueScripts() { 
  270.       /** @var ICWP_WPTB_FeatureHandler_Css $oFO */ 
  271.       $oFO = $this->getFeatureOptions(); 
  272.  
  273.       $fJsInFooter = ($this->getIsOption( 'js_head', 'Y' ) )? false : true ; 
  274.  
  275.       if ( $this->getIsUseTwitter() && $this->getIsOption( 'all_js', 'Y' ) ) { 
  276.  
  277.           $sExtension = $this->getIsOption( 'use_minified_css', 'Y' )? '.min.js' : '.js'; 
  278.  
  279.           if ( $this->getIsOption( 'use_cdnjs', 'Y' ) ) { 
  280.               //cdnjs.cloudflare.com/ajax/libs/twitter-bootstrap/2.2.2/bootstrap.min.js 
  281.               //Since version 2.3.0, now changed to: 
  282.               //cdnjs.cloudflare.com/ajax/libs/twitter-bootstrap/2.3.1/js/bootstrap.min.js 
  283.               $sUrlBootstrapJs = self::CdnjsStem.'twitter-bootstrap/%s/js/bootstrap'.$sExtension; 
  284.               $sUrlBootstrapJs = sprintf( $sUrlBootstrapJs, $oFO->getTwitterBootstrapVersion() ); 
  285.           } 
  286.           else { 
  287.               $sUrlBootstrapJs = $this->getBootstrapUrl( 'js/bootstrap'.$sExtension ); 
  288.           } 
  289.  
  290.           if ( $this->getIsOption( 'replace_jquery_cdn', 'Y' ) ) { 
  291.               //cdnjs.cloudflare.com/ajax/libs/jquery/1.8.3/jquery.min.js 
  292.               $sJqueryCdnUri = self::CdnjsStem.'jquery/'.self::CdnJqueryVersion.'/jquery'.$sExtension; 
  293.               wp_deregister_script('jquery'); 
  294.               wp_register_script( 'jquery', $sJqueryCdnUri, '', self::CdnJqueryVersion, false ); 
  295.           } 
  296.  
  297.           wp_enqueue_script( 'jquery' ); 
  298.           wp_register_script( 'bootstrap-all-min', $sUrlBootstrapJs, array('jquery'), $oFO->getVersion(), $fJsInFooter ); 
  299.           wp_enqueue_script( 'bootstrap-all-min' ); 
  300.       } 
  301.  
  302.       if ( $this->getIsOption( 'prettify', 'Y' ) ) { 
  303.           $sUrl = $this->getJsUrl( 'google-code-prettify/prettify.js' ); 
  304.           wp_register_script( 'prettify_script', $sUrl, false, $oFO->getVersion(), $fJsInFooter ); 
  305.           wp_enqueue_script( 'prettify_script' ); 
  306.       } 
  307.   }
