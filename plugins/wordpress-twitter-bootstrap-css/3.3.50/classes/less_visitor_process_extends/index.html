  23.     private function doExtendChaining( $extendsList, $extendsListTarget, $iterationCount = 0) { 
  24.         // 
  25.         // chaining is different from normal extension.. if we extend an extend then we are not just copying, altering and pasting 
  26.         // the selector we would do normally, but we are also adding an extend with the same target selector 
  27.         // this means this new extend can then go and alter other extends 
  28.         // 
  29.         // this method deals with all the chaining work - without it, extend is flat and doesn't work on other extend selectors 
  30.         // this is also the most expensive.. and a match on one selector can cause an extension of a selector we had already processed if 
  31.         // we look at each selector at a time, as is done in visitRuleset 
  32.  
  33.         $extendsToAdd = array(); 
  34.  
  35.  
  36.         //loop through comparing every extend with every target extend. 
  37.         // a target extend is the one on the ruleset we are looking at copy/edit/pasting in place 
  38.         // e.g. .a:extend(.b) {} and .b:extend(.c) {} then the first extend extends the second one 
  39.         // and the second is the target. 
  40.         // the seperation into two lists allows us to process a subset of chains with a bigger set, as is the 
  41.         // case when processing media queries 
  42.         for( $extendIndex = 0, $extendsList_len = count($extendsList); $extendIndex < $extendsList_len; $extendIndex++ ) { 
  43.             for( $targetExtendIndex = 0; $targetExtendIndex < count($extendsListTarget); $targetExtendIndex++ ) { 
  44.  
  45.                 $extend = $extendsList[$extendIndex]; 
  46.                 $targetExtend = $extendsListTarget[$targetExtendIndex]; 
  47.  
  48.                 // look for circular references 
  49.                 if( in_array($targetExtend->object_id, $extend->parent_ids, true) ) { 
  50.                     continue; 
  51.                 } 
  52.  
  53.                 // find a match in the target extends self selector (the bit before :extend) 
  54.                 $selectorPath = array( $targetExtend->selfSelectors[0] ); 
  55.                 $matches = $this->findMatch( $extend, $selectorPath); 
  56.  
  57.  
  58.                 if( $matches ) { 
  59.  
  60.                     // we found a match, so for each self selector.. 
  61.                     foreach($extend->selfSelectors as $selfSelector ) { 
  62.  
  63.  
  64.                         // process the extend as usual 
  65.                         $newSelector = $this->extendSelector( $matches, $selectorPath, $selfSelector); 
  66.  
  67.                         // but now we create a new extend from it 
  68.                         $newExtend = new Less_Tree_Extend( $targetExtend->selector, $targetExtend->option, 0);

  70.  
  71.                         // add the extend onto the list of extends for that selector 
  72.                         end($newSelector)->extendList = array($newExtend); 
  73.                         //$newSelector[ count($newSelector)-1]->extendList = array($newExtend); 
  74.  
  75.                         // record that we need to add it. 
  76.                         $extendsToAdd[] = $newExtend; 
  77.                         $newExtend->ruleset = $targetExtend->ruleset; 
  78.  
  79.                         //remember its parents for circular references 
  80.                         $newExtend->parent_ids = array_merge($newExtend->parent_ids, $targetExtend->parent_ids, $extend->parent_ids); 
  81.  
  82.                         // only process the selector once.. if we have :extend(.a, .b) then multiple 
  83.                         // extends will look at the same selector path, so when extending 
  84.                         // we know that any others will be duplicates in terms of what is added to the css 
  85.                         if( $targetExtend->firstExtendOnThisSelectorPath ) { 
  86.                             $newExtend->firstExtendOnThisSelectorPath = true; 
  87.                             $targetExtend->ruleset->paths[] = $newSelector;

  89.                     } 
  90.                 } 
  91.             } 
  92.         } 
  93.  
  94.         if( $extendsToAdd ) { 
  95.             // try to detect circular references to stop a stack overflow. 
  96.             // may no longer be needed.            $this->extendChainCount++; 
  97.             if( $iterationCount > 100) { 
  98.  
  99.                 try{

  101.                     $selectorTwo = $extendsToAdd[0]->selector->toCSS();

  104.                     $selectorTwo = "{unable to calculate}"; 
  105.                 } 
  106.  
  107.                 throw new Less_Exception_Parser("extend circular reference detected. One of the circular extends is currently:"+$selectorOne+":extend(" + $selectorTwo+")"); 
  108.             } 
  109.  
  110.             // now process the new extends on the existing rules so that we can handle a extending b extending c ectending d extending e... 
  111.             $extendsToAdd = $this->doExtendChaining( $extendsToAdd, $extendsListTarget, $iterationCount+1); 
  112.         } 
  113.  
  114.         return array_merge($extendsList, $extendsToAdd);

  174.     private function findMatch($extend, $haystackSelectorPath ) { 
  175.  
  176.  
  177.         if( !$this->HasMatches($extend, $haystackSelectorPath) ) { 
  178.             return false; 
  179.         } 
  180.  
  181.  
  182.         // 
  183.         // look through the haystack selector path to try and find the needle - extend.selector 
  184.         // returns an array of selector matches that can then be replaced 
  185.         // 
  186.         $needleElements = $extend->selector->elements; 
  187.         $potentialMatches = array(); 
  188.         $potentialMatches_len = 0; 
  189.         $potentialMatch = null; 
  190.         $matches = array(); 
  191.  
  192.  
  193.  
  194.         // loop through the haystack elements 
  195.         $haystack_path_len = count($haystackSelectorPath); 
  196.         for($haystackSelectorIndex = 0; $haystackSelectorIndex < $haystack_path_len; $haystackSelectorIndex++ ) { 
  197.             $hackstackSelector = $haystackSelectorPath[$haystackSelectorIndex]; 
  198.  
  199.             $haystack_elements_len = count($hackstackSelector->elements); 
  200.             for($hackstackElementIndex = 0; $hackstackElementIndex < $haystack_elements_len; $hackstackElementIndex++ ) { 
  201.  
  202.                 $haystackElement = $hackstackSelector->elements[$hackstackElementIndex]; 
  203.  
  204.                 // if we allow elements before our match we can add a potential match every time. otherwise only at the first element. 
  205.                 if( $extend->allowBefore || ($haystackSelectorIndex === 0 && $hackstackElementIndex === 0) ) { 
  206.                     $potentialMatches[] = array('pathIndex'=> $haystackSelectorIndex, 'index'=> $hackstackElementIndex, 'matched'=> 0, 'initialCombinator'=> $haystackElement->combinator); 
  207.                     $potentialMatches_len++; 
  208.                 } 
  209.  
  210.                 for($i = 0; $i < $potentialMatches_len; $i++ ) { 
  211.  
  212.                     $potentialMatch = &$potentialMatches[$i]; 
  213.                     $potentialMatch = $this->PotentialMatch( $potentialMatch, $needleElements, $haystackElement, $hackstackElementIndex ); 
  214.  
  215.  
  216.                     // if we are still valid and have finished, test whether we have elements after and whether these are allowed 
  217.                     if( $potentialMatch && $potentialMatch['matched'] === $extend->selector->elements_len ) { 
  218.                         $potentialMatch['finished'] = true; 
  219.  
  220.                         if( !$extend->allowAfter && ($hackstackElementIndex+1 < $haystack_elements_len || $haystackSelectorIndex+1 < $haystack_path_len) ) { 
  221.                             $potentialMatch = null; 
  222.                         } 
  223.                     } 
  224.  
  225.                     // if null we remove, if not, we are still valid, so either push as a valid match or continue 
  226.                     if( $potentialMatch ) { 
  227.                         if( $potentialMatch['finished'] ) { 
  228.                             $potentialMatch['length'] = $extend->selector->elements_len; 
  229.                             $potentialMatch['endPathIndex'] = $haystackSelectorIndex; 
  230.                             $potentialMatch['endPathElementIndex'] = $hackstackElementIndex + 1; // index after end of match 
  231.                             $potentialMatches = array(); // we don't allow matches to overlap, so start matching again 
  232.                             $potentialMatches_len = 0; 
  233.                             $matches[] = $potentialMatch; 
  234.                         } 
  235.                         continue; 
  236.                     } 
  237.  
  238.                     array_splice($potentialMatches, $i, 1); 
  239.                     $potentialMatches_len--; 
  240.                     $i--; 
  241.                 } 
  242.             } 
  243.         } 
  244.  
  245.         return $matches; 
  246.     }

  276.     private function PotentialMatch( $potentialMatch, $needleElements, $haystackElement, $hackstackElementIndex ) { 
  277.  
  278.  
  279.         if( $potentialMatch['matched'] > 0 ) { 
  280.  
  281.             // selectors add " " onto the first element. When we use & it joins the selectors together, but if we don't 
  282.             // then each selector in haystackSelectorPath has a space before it added in the toCSS phase. so we need to work out 
  283.             // what the resulting combinator will be 
  284.             $targetCombinator = $haystackElement->combinator; 
  285.             if( $targetCombinator === '' && $hackstackElementIndex === 0 ) { 
  286.                 $targetCombinator = ' '; 
  287.             } 
  288.  
  289.             if( $needleElements[ $potentialMatch['matched'] ]->combinator !== $targetCombinator ) { 
  290.                 return null; 
  291.             } 
  292.         } 
  293.  
  294.         // if we don't match, null our match to indicate failure 
  295.         if( !$this->isElementValuesEqual( $needleElements[$potentialMatch['matched'] ]->value, $haystackElement->value) ) { 
  296.             return null; 
  297.         } 
  298.  
  299.         $potentialMatch['finished'] = false; 
  300.         $potentialMatch['matched']++; 
  301.  
  302.         return $potentialMatch; 
  303.     }

  306.     private function isElementValuesEqual( $elementValue1, $elementValue2 ) { 
  307.  
  308.         if( $elementValue1 === $elementValue2 ) { 
  309.             return true; 
  310.         } 
  311.  
  312.         if( is_string($elementValue1) || is_string($elementValue2) ) { 
  313.             return false; 
  314.         } 
  315.  
  316.         if( $elementValue1 instanceof Less_Tree_Attribute ) { 
  317.             return $this->isAttributeValuesEqual( $elementValue1, $elementValue2 ); 
  318.         } 
  319.  
  320.         $elementValue1 = $elementValue1->value; 
  321.         if( $elementValue1 instanceof Less_Tree_Selector ) { 
  322.             return $this->isSelectorValuesEqual( $elementValue1, $elementValue2 ); 
  323.         }

  333.

  346.  
  347.             if( !$this->isElementValuesEqual($elementValue1->elements[$i]->value, $elementValue2->elements[$i]->value) ) { 
  348.                 return false; 
  349.             } 
  350.         }

  352.         return true;

  381.         //for a set of matches, replace each match with the replacement selector 
  382.

  389.  
  390.  
  391.             $match = $matches[$matchIndex]; 
  392.             $selector = $selectorPath[ $match['pathIndex'] ]; 
  393.  
  394.             $firstElement = new Less_Tree_Element( 
  395.                 $match['initialCombinator'],  
  396.                 $replacementSelector->elements[0]->value,  
  397.                 $replacementSelector->elements[0]->index,  
  398.                 $replacementSelector->elements[0]->currentFileInfo 
  399.  ); 
  400.  
  401.             if( $match['pathIndex'] > $currentSelectorPathIndex && $currentSelectorPathElementIndex > 0 ) { 
  402.                 $last_path = end($path); 
  403.                 $last_path->elements = array_merge( $last_path->elements, array_slice( $selectorPath[$currentSelectorPathIndex]->elements, $currentSelectorPathElementIndex)); 
  404.                 $currentSelectorPathElementIndex = 0; 
  405.                 $currentSelectorPathIndex++; 
  406.             } 
  407.  
  408.             $newElements = array_merge( 
  409.                 array_slice($selector->elements, $currentSelectorPathElementIndex, ($match['index'] - $currentSelectorPathElementIndex) ) // last parameter of array_slice is different than the last parameter of javascript's slice 
  410.  , array($firstElement)

  413.
