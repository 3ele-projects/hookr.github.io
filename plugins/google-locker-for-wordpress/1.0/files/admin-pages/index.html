  425. function gl_plugin_options_page( ) { 
  426.   // Options page 
  427.   $tab = isset( $_GET[ 'tab' ] ) ? $_GET[ 'tab' ] : gl_get_general_settings_key(); 
  428.    
  429.   // Get options value after user press "Save Settings" button 
  430.   $update_settings = isset( $_GET[ "settings-updated" ] ) ? $_GET[ "settings-updated" ] : ""; 
  431.   if ( $update_settings == "true" ) {     
  432.       $general_settings = gl_get_general_settings_values(); 
  433.       $parts = explode('_', $general_settings[ 'language' ]); 
  434.       $general_settings = array_merge( $general_settings, array( 
  435.               'short_language' => $parts[0] 
  436.  ) ); 
  437.            
  438.       delete_option( gl_get_general_settings_key() ); 
  439.       update_option( gl_get_general_settings_key(), $general_settings ); 
  440.   } 
  441.    
  442.   // reset settings 
  443.   $reset_settings = isset( $_GET[ "reset" ] ) ? $_GET[ "reset" ] : ""; 
  444.   if ( $reset_settings == "true" ) {         
  445.       if ( $tab == gl_get_general_settings_key() ) {

  463.
