  241.   if ( $type == 'social' && $_COOKIE["g_locker_social"] == 'true') {

  389. /** 
  390.  * Function convert number to short value with M, B, k 
  391.  */ 
  392. function gl_get_number_format($value) { 
  393.   if ( $value > 1000000000 ) return round( ($value/1000000000), 1 ).'B'; 
  394.   else if ( $value > 1000000 ) return round( ($value/1000000), 1 ).'M'; 
  395.   else if ( $value > 1000 ) return round( ($value/1000), 1 ).'k'; 
  396.  
  397.   return $value; 
  398. }

  400. /** 
  401.  * Function get Subscriber Counter for Youtube 
  402.  */  
  403. function gl_get_counter_youtube($user) { 
  404.    
  405.   $CHECK_URL_PREFIX = 'https://gdata.youtube.com/feeds/api/users/'; 
  406.    
  407.   $check_url = $CHECK_URL_PREFIX . $user . '?alt=json'; 
  408.    
  409.   try { 
  410.       $response = file_get_contents( $check_url ); 
  411.       $result = json_decode( $response, true ); 
  412.        
  413.       if (isset( $result['entry']['yt$statistics']['subscriberCount'] )) { 
  414.           return gl_get_number_format( $result['entry']['yt$statistics']['subscriberCount'] ); 
  415.       } 
  416.       else { 
  417.           return "0"; 
  418.       } 
  419.   } catch ( Exception $e ) { 
  420.       return "0"; 
  421.   }
