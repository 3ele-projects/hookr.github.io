  77.       'body' => array( 
  78.           'action' => 'edd_test_ajax' 
  79.  )

  107.   }

  530.  
  531.   $items = $wpdb->get_results( $prepared_statement );

  533.   if( $items ) { 
  534.  
  535.       foreach( $items as $item ) {
