  783.

  785.  
  786.         } elseif ( $variable_pricing && $price_id === false ) {

  789.             $highest_price = (float) edd_get_highest_price_option( $this->ID );

  823.      * Updates a single meta entry for the download 
  824.      * 
  825.      * @since  2.3 
  826.      * @access private 
  827.      * @param  string $meta_key   The meta_key to update 
  828.      * @param  string|array|object $meta_value The value to put into the meta 
  829.      * @return bool             The result of the update query 
  830.      */ 
  831.     private function update_meta( $meta_key = '', $meta_value = '' ) { 
  832.  
  833.         global $wpdb; 
  834.  
  835.         if ( empty( $meta_key ) || empty( $meta_value ) ) { 
  836.             return false; 
  837.         } 
  838.  
  839.         // Make sure if it needs to be serialized, we do 
  840.         $meta_value = maybe_serialize( $meta_value ); 
  841.  
  842.         if ( is_numeric( $meta_value ) ) { 
  843.             $value_type = is_float( $meta_value ) ? '%f' : '%d'; 
  844.         } else { 
  845.             $value_type = "'%s'"; 
  846.         } 
  847.  
  848.         $sql = $wpdb->prepare( "UPDATE $wpdb->postmeta SET meta_value = $value_type WHERE post_id = $this->ID AND meta_key = '%s'", $meta_value, $meta_key ); 
  849.  
  850.         if ( $wpdb->query( $sql ) ) { 
  851.  
  852.             clean_post_cache( $this->ID );

  855.         } 
  856.  
  857.         return false;
