  664.                             /** *

  666.                              * are distributed across all cart items. The discount amount is divided by the number 
  667.                              * of items in the cart and then a portion is evenly applied to each cart item

  689.                         } else { 
  690.                             $discounted_price -= $price - edd_get_discounted_amount( $discount, $price );

  692.                     } 
  693.                 }

  695.                 if ( $discounted_price < 0 ) {

  701.  
  702.             if ( 'flat' !== edd_get_discount_type( $code_id ) ) {

  704.             } 
  705.         } 
  706.  
  707.         return $amount; 
  708.     }
