  323.      * @access public

  327.     public function load_textdomain() { 
  328.         global $wp_version; 
  329.  
  330.         /** 
  331.          * Due to the introduction of language packs through translate.wordpress.org, loading our textdomain is complex. 
  332.          * 
  333.          * In v2.4.6, our textdomain changed from "edd" to "easy-digital-downloads". 
  334.          * 
  335.          * To support existing translation files from before the change, we must look for translation files in several places and under several names. 
  336.          * 
  337.          * - wp-content/languages/plugins/easy-digital-downloads (introduced with language packs) 
  338.          * - wp-content/languages/edd/ (custom folder we have supported since 1.4) 
  339.          * - wp-content/plugins/easy-digital-downloads/languages/ 
  340.          * 
  341.          * In wp-content/languages/edd/ we must look for "easy-digital-downloads-{lang}_{country}.mo" 
  342.          * In wp-content/languages/edd/ we must look for "edd-{lang}_{country}.mo" as that was the old file naming convention 
  343.          * In wp-content/languages/plugins/easy-digital-downloads/ we only need to look for "easy-digital-downloads-{lang}_{country}.mo" as that is the new structure 
  344.          * In wp-content/plugins/easy-digital-downloads/languages/, we must look for both naming conventions. This is done by filtering "load_textdomain_mofile" 
  345.          * 
  346.          */ 
  347.  
  348.         add_filter( 'load_textdomain_mofile', array( $this, 'load_old_textdomain' ), 10, 2 ); 
  349.  
  350.         // Set filter for plugin's languages directory. 
  351.         $edd_lang_dir = dirname( plugin_basename( EDD_PLUGIN_FILE ) ) . '/languages/'; 
  352.         $edd_lang_dir = apply_filters( 'edd_languages_directory', $edd_lang_dir ); 
  353.  
  354.         // Traditional WordPress plugin locale filter. 
  355.  
  356.         $get_locale = get_locale(); 
  357.  
  358.         if ( $wp_version >= 4.7 ) { 
  359.  
  360.             $get_locale = get_user_locale(); 
  361.         } 
  362.  
  363.         /** 
  364.          * Defines the plugin language locale used in AffiliateWP. 
  365.          * 
  366.          * @var $get_locale The locale to use. Uses get_user_locale()` in WordPress 4.7 or greater,  
  367.          *                  otherwise uses `get_locale()`. 
  368.          */ 
  369.         $locale = apply_filters( 'plugin_locale', $get_locale, 'easy-digital-downloads' ); 
  370.         $mofile = sprintf( '%1$s-%2$s.mo', 'easy-digital-downloads', $locale ); 
  371.  
  372.         // Look for wp-content/languages/edd/easy-digital-downloads-{lang}_{country}.mo 
  373.         $mofile_global1 = WP_LANG_DIR . '/edd/easy-digital-downloads-' . $locale . '.mo'; 
  374.  
  375.         // Look for wp-content/languages/edd/edd-{lang}_{country}.mo 
  376.         $mofile_global2 = WP_LANG_DIR . '/edd/edd-' . $locale . '.mo'; 
  377.  
  378.         // Look in wp-content/languages/plugins/easy-digital-downloads 
  379.         $mofile_global3 = WP_LANG_DIR . '/plugins/easy-digital-downloads/' . $mofile; 
  380.  
  381.         if ( file_exists( $mofile_global1 ) ) { 
  382.  
  383.             load_textdomain( 'easy-digital-downloads', $mofile_global1 ); 
  384.  
  385.         } elseif ( file_exists( $mofile_global2 ) ) { 
  386.  
  387.             load_textdomain( 'easy-digital-downloads', $mofile_global2 ); 
  388.  
  389.         } elseif ( file_exists( $mofile_global3 ) ) { 
  390.  
  391.             load_textdomain( 'easy-digital-downloads', $mofile_global3 ); 
  392.  
  393.         } else { 
  394.  
  395.             // Load the default language files. 
  396.             load_plugin_textdomain( 'easy-digital-downloads', false, $edd_lang_dir ); 
  397.         } 
  398.  
  399.     }

  401.     /** 
  402.      * Load a .mo file for the old textdomain if one exists. 
  403.      * 
  404.      * h/t: https://github.com/10up/grunt-wp-plugin/issues/21#issuecomment-62003284 
  405.      */ 
  406.     function load_old_textdomain( $mofile, $textdomain ) { 
  407.  
  408.         if ( $textdomain === 'easy-digital-downloads' && ! file_exists( $mofile ) ) { 
  409.             $mofile = dirname( $mofile ) . DIRECTORY_SEPARATOR . str_replace( $textdomain, 'edd', basename( $mofile ) ); 
  410.         } 
  411.  
  412.         return $mofile; 
  413.     }
