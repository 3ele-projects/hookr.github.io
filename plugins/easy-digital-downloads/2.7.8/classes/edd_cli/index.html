  426.             WP_CLI::line( sprintf( __( 'Start Date: %s', 'easy-digital-downloads' ), ( empty( $discount['start_date'] ) ? __( 'No Start Date', 'easy-digital-downloads' ) : $discount['start_date'] ) ) ); 
  427.             WP_CLI::line( sprintf( __( 'Expiration Date: %s', 'easy-digital-downloads' ), ( empty( $discount['exp_date'] ) ? __( 'No Expiration', 'easy-digital-downloads' ) : $discount['exp_date'] ) ) ); 
  428.             WP_CLI::line( sprintf( __( 'Status: %s', 'easy-digital-downloads' ), ucwords( $discount['status'] ) ) ); 
  429.  
  430.             WP_CLI::line( '' ); 
  431.  
  432.             if( array_key_exists( 0, $discount['product_requirements'] ) ) { 
  433.                 WP_CLI::line( __( 'Product Requirements:', 'easy-digital-downloads' ) ); 
  434.  
  435.                 foreach( $discount['product_requirements'] as $req => $req_id ) { 
  436.                     WP_CLI::line( sprintf( __( '  Product: %s', 'easy-digital-downloads' ), $req_id ) ); 
  437.                 } 
  438.             } 
  439.  
  440.             WP_CLI::line( '' ); 
  441.  
  442.             WP_CLI::line( sprintf( __( 'Global Discount: %s', 'easy-digital-downloads' ), ( empty( $discount['global_discount'] ) ? 'False' : 'True' ) ) ); 
  443.             WP_CLI::line( sprintf( __( 'Single Use: %s', 'easy-digital-downloads' ), ( empty( $discount['single_use'] ) ? 'False' : 'True' ) ) ); 
  444.  
  445.             WP_CLI::line( '' ); 
  446.         } 
  447.     } 
  448.  
  449.  
  450.     /** 
  451.      * Create sample purchase data for your EDD site 
  452.      * 
  453.      * ## OPTIONS 
  454.      * 
  455.      * --number: The number of purchases to create 
  456.      * --status=<status>: The status to create purchases as 
  457.      * --id=<product_id>: A specific product to create purchase data for 
  458.      * --price_id=<price_id>: A price ID of the specified product 
  459.      * 
  460.      * ## EXAMPLES 
  461.      * 
  462.      * wp edd payments create --number=10 --status=completed 
  463.      * wp edd payments create --number=10 --id=103 
  464.      */ 
  465.     public function payments( $args, $assoc_args ) { 
  466.  
  467.         $error = false; 
  468.  
  469.         // At some point we'll likely add another action for payments 
  470.         if( ! isset( $args ) ||  count( $args ) == 0 ) { 
  471.             $error = __( 'No action specified, did you mean', 'easy-digital-downloads' ); 
  472.         } elseif( isset( $args ) && ! in_array( 'create', $args ) ) { 
  473.             $error = __( 'Invalid action specified, did you mean', 'easy-digital-downloads' ); 
  474.         } 
  475.  
  476.         if( $error ) { 
  477.             foreach( $assoc_args as $key => $value ) { 
  478.                 $query .= ' --' . $key . '=' . $value; 
  479.             } 
  480.  
  481.             WP_CLI::error( 
  482.                 sprintf( $error . ' %s?', 'wp edd payments create' . $query ) 
  483.  ); 
  484.  
  485.             return; 
  486.         } 
  487.  
  488.  
  489.         // Setup some defaults 
  490.         $number = 1; 
  491.         $status = 'complete'; 
  492.         $id = false; 
  493.         $price_id = false; 
  494.         $tax = 0; 
  495.         $email = 'guest@local.dev'; 
  496.         $fname = 'Pippin'; 
  497.         $lname = 'Williamson'; 
  498.         $date = false; 
  499.         $range = 30; 
  500.  
  501.         if( count( $assoc_args ) > 0 ) { 
  502.             $number = ( array_key_exists( 'number', $assoc_args ) )   ? absint( $assoc_args['number'] )             : $number; 
  503.             $id = ( array_key_exists( 'id', $assoc_args ) )       ? absint( $assoc_args['id'] )                 : $id; 
  504.             $price_id = ( array_key_exists( 'price_id', $assoc_args ) ) ? absint( $assoc_args['id'] )                 : $price_id; 
  505.             $tax = ( array_key_exists( 'tax', $assoc_args ) )      ? floatval( $assoc_args['tax'] )              : $tax; 
  506.             $email = ( array_key_exists( 'email', $assoc_args ) )    ? sanitize_email( $assoc_args['email'] )      : $email; 
  507.             $fname = ( array_key_exists( 'fname', $assoc_args ) )    ? sanitize_text_field( $assoc_args['fname'] ) : $fname; 
  508.             $lname = ( array_key_exists( 'lname', $assoc_args ) )    ? sanitize_text_field( $assoc_args['lname'] ) : $lname; 
  509.             $date = ( array_key_exists( 'date', $assoc_args ) )     ? sanitize_text_field( $assoc_args['date'] )  : $date; 
  510.             $range = ( array_key_exists( 'range', $assoc_args ) )    ? absint( $assoc_args['range'] )              : $range; 
  511.  
  512.             // Status requires a bit more validation 
  513.             if( array_key_exists( 'status', $assoc_args ) ) { 
  514.                 $stati = array( 
  515.                     'publish',  
  516.                     'complete',  
  517.                     'pending',  
  518.                     'refunded',  
  519.                     'revoked',  
  520.                     'failed',  
  521.                     'abandoned',  
  522.                     'preapproval',  
  523.                     'cancelled' 
  524.  ); 
  525.  
  526.                 if( in_array( $assoc_args['status'], $stati ) ) { 
  527.                     $status = ( $assoc_args['status'] == 'complete' ) ? 'publish' : $assoc_args['status']; 
  528.                 } else { 
  529.                     WP_CLI::warning( sprintf( 
  530.                         __( "Invalid status '%s', defaulting to 'complete'", 'easy-digital-downloads' ),  
  531.                         $assoc_args['status'] 
  532.  ) ); 
  533.                 } 
  534.             } 
  535.         } 
  536.  
  537.         // Build the user info array 
  538.         $user_info = array( 
  539.             'id' => 0,  
  540.             'email' => $email,  
  541.             'first_name' => $fname,  
  542.             'last_name' => $lname,  
  543.             'discount' => 'none' 
  544.  ); 
  545.  
  546.         for( $i = 0; $i < $number; $i++ ) { 
  547.  
  548.             $products = array(); 
  549.             $total = 0; 
  550.  
  551.             // No specified product 
  552.             if( ! $id ) { 
  553.  
  554.                 $products = get_posts( array( 
  555.                     'post_type' => 'download',  
  556.                     'orderby' => 'rand',  
  557.                     'order' => 'ASC',  
  558.                     'posts_per_page'=> 1 
  559.  ) ); 
  560.  
  561.             } else { 
  562.  
  563.                 $product = get_post( $id ); 
  564.  
  565.                 if( $product->post_type != 'download' ) { 
  566.                     WP_CLI::error( __( 'Specified ID is not a product', 'easy-digital-downloads' ) ); 
  567.                     return; 
  568.                 } 
  569.  
  570.                 $products[] = $product; 
  571.  
  572.             } 
  573.  
  574.             $cart_details = array(); 
  575.  
  576.             // Create the purchases 
  577.             foreach( $products as $key => $download ) { 
  578.  
  579.                 if( ! is_a( $download, 'WP_Post' ) ) { 
  580.                     continue; 
  581.                 } 
  582.  
  583.                 $options = array(); 
  584.                 $final_downloads = array(); 
  585.  
  586.                 // Deal with variable pricing 
  587.                 if( edd_has_variable_prices( $download->ID ) ) { 
  588.  
  589.                     $prices = edd_get_variable_prices( $download->ID ); 
  590.  
  591.                     if( false === $price_id || ! array_key_exists( $price_id, (array) $prices ) ) { 
  592.                         $price_id = rand( 0, count( $prices ) - 1 ); 
  593.                     } 
  594.  
  595.                     $item_price = $prices[ $price_id ]['amount']; 
  596.                     $options['price_id'] = $price_id; 
  597.  
  598.                 } else { 
  599.  
  600.                     $item_price = edd_get_download_price( $download->ID ); 
  601.  
  602.                 } 
  603.  
  604.                 $item_number = array( 
  605.                     'id' => $download->ID,  
  606.                     'quantity' => 1,  
  607.                     'options' => $options 
  608.  ); 
  609.  
  610.                 $cart_details[$key] = array( 
  611.                     'name' => $download->post_title,  
  612.                     'id' => $download->ID,  
  613.                     'item_number' => $item_number,  
  614.                     'item_price' => edd_sanitize_amount( $item_price ),  
  615.                     'subtotal' => edd_sanitize_amount( $item_price ),  
  616.                     'price' => edd_sanitize_amount( $item_price ),  
  617.                     'quantity' => 1,  
  618.                     'discount' => 0,  
  619.                     'tax' => $tax 
  620.  ); 
  621.  
  622.                 $final_downloads[$key] = $item_number; 
  623.  
  624.                 $total += $item_price; 
  625.  
  626.             } 
  627.  
  628.             $purchase_data = array( 
  629.                 'price' => edd_sanitize_amount( $total ),  
  630.                 'tax' => 0,  
  631.                 'purchase_key' => strtolower( md5( uniqid() ) ),  
  632.                 'user_email' => $email,  
  633.                 'user_info' => $user_info,  
  634.                 'currency' => edd_get_currency(),  
  635.                 'downloads' => $final_downloads,  
  636.                 'cart_details' => $cart_details,  
  637.                 'status' => 'pending' 
  638.  ); 
  639.  
  640.             $payment_id = edd_insert_payment( $purchase_data ); 
  641.  
  642.             remove_action( 'edd_complete_purchase', 'edd_trigger_purchase_receipt', 999 ); 
  643.  
  644.             if( $status != 'pending' ) { 
  645.                 edd_update_payment_status( $payment_id, $status ); 
  646.             } 
  647.  
  648.             if ( 'random' === $date ) { 
  649.                 // Randomly grab a date from the current past 30 days 
  650.                 $oldest_time = strtotime( '-' . $range . ' days', current_time( 'timestamp') ); 
  651.                 $newest_time = current_time( 'timestamp' ); 
  652.  
  653.                 $timestamp = rand( $oldest_time, $newest_time ); 
  654.                 $timestring = date( "Y-m-d H:i:s", $timestamp ); 
  655.             } elseif ( empty( $date ) ) { 
  656.                 $timestring = false; 
  657.             } else { 
  658.                 if ( is_numeric( $date ) ) { 
  659.                     $timestring = date( "Y-m-d H:i:s", $date ); 
  660.                 } else { 
  661.                     $parsed_time = strtotime( $date ); 
  662.                     $timestring = date( "Y-m-d H:i:s", $parsed_time ); 
  663.                 } 
  664.             } 
  665.  
  666.             if ( ! empty( $timestring ) ) { 
  667.                 $payment = new EDD_Payment( $payment_id ); 
  668.                 $payment->date = $timestring; 
  669.                 $payment->completed_date = $timestring; 
  670.                 $payment->save(); 
  671.             } 
  672.  
  673.         } 
  674.  
  675.         WP_CLI::success( sprintf( __( 'Created %s payments', 'easy-digital-downloads' ), $number ) ); 
  676.         return; 
  677.     } 
  678. }
