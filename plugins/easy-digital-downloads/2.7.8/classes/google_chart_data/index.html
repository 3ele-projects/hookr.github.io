  618.     /** 
  619.     * linear regression function 
  620.     * 
  621.     * @param $data array Points to calculate 
  622.     * @returns array() m=>slope, b=>intercept 
  623.     */ 
  624.     static public function calculateLinearRegression($data) 
  625.     { 
  626.         // Calculate number points 
  627.         $n = count($data); 
  628.  
  629.         $x = array_keys($data); 
  630.         $y = array_values($data); 
  631.  
  632.         // Calculate sums 
  633.         $x_sum = array_sum($x); 
  634.         $y_sum = array_sum($y); 
  635.  
  636.         $xx_sum = 0; 
  637.         $xy_sum = 0; 
  638.  
  639.         for($i = 0; $i < $n; $i++) { 
  640.  
  641.             $xy_sum+=($x[$i]*$y[$i]); 
  642.             $xx_sum+=($x[$i]*$x[$i]); 
  643.  
  644.         } 
  645.  
  646.         // Calculate slope 
  647.         $m = (($n * $xy_sum) - ($x_sum * $y_sum)) / (($n * $xx_sum) - ($x_sum * $x_sum)); 
  648.  
  649.         // Calculate intercept 
  650.         $b = ($y_sum - ($m * $x_sum)) / $n; 
  651.  
  652.         // Return result 
  653.         return array($m, $b); 
  654.     }
