  15. /** 
  16.  * Sanitize Amount 
  17.  * 
  18.  * Returns a sanitized amount by stripping out thousands separators. 
  19.  * 
  20.  * @since 1.0 
  21.  * @param string $amount Price amount to format 
  22.  * @return string $amount Newly sanitized amount 
  23.  */ 
  24. function edd_sanitize_amount( $amount ) { 
  25.   $is_negative = false; 
  26.   $thousands_sep = edd_get_option( 'thousands_separator', ', ' ); 
  27.   $decimal_sep = edd_get_option( 'decimal_separator', '.' ); 
  28.  
  29.   // Sanitize the amount 
  30.   if ( $decimal_sep == ', ' && false !== ( $found = strpos( $amount, $decimal_sep ) ) ) { 
  31.       if ( ( $thousands_sep == '.' || $thousands_sep == ' ' ) && false !== ( $found = strpos( $amount, $thousands_sep ) ) ) { 
  32.           $amount = str_replace( $thousands_sep, '', $amount ); 
  33.       } elseif( empty( $thousands_sep ) && false !== ( $found = strpos( $amount, '.' ) ) ) { 
  34.           $amount = str_replace( '.', '', $amount ); 
  35.       } 
  36.  
  37.       $amount = str_replace( $decimal_sep, '.', $amount ); 
  38.   } elseif( $thousands_sep == ', ' && false !== ( $found = strpos( $amount, $thousands_sep ) ) ) { 
  39.       $amount = str_replace( $thousands_sep, '', $amount ); 
  40.   } 
  41.  
  42.   if( $amount < 0 ) { 
  43.       $is_negative = true; 
  44.   } 
  45.  
  46.   $amount = preg_replace( '/[^0-9\.]/', '', $amount ); 
  47.  
  48.   /** 
  49.    * Filter number of decimals to use for prices 
  50.    * 
  51.    * @since unknown 
  52.    * 
  53.    * @param int $number Number of decimals 
  54.    * @param int|string $amount Price 
  55.    */ 
  56.   $decimals = apply_filters( 'edd_sanitize_amount_decimals', 2, $amount ); 
  57.   $amount = number_format( (double) $amount, $decimals, '.', '' ); 
  58.  
  59.   if( $is_negative ) { 
  60.       $amount *= -1; 
  61.   } 
  62.  
  63.   /** 
  64.    * Filter the sanitized price before returning 
  65.    * 
  66.    * @since unknown 
  67.    * 
  68.    * @param string $amount Price 
  69.    */ 
  70.   return apply_filters( 'edd_sanitize_amount', $amount ); 
  71. }

  73. /** 
  74.  * Returns a nicely formatted amount. 
  75.  * 
  76.  * @since 1.0 
  77.  * 
  78.  * @param string $amount   Price amount to format 
  79.  * @param string $decimals Whether or not to use decimals.  Useful when set to false for non-currency numbers. 
  80.  * 
  81.  * @return string $amount Newly formatted amount or Price Not Available 
  82.  */ 
  83. function edd_format_amount( $amount, $decimals = true ) { 
  84.   $thousands_sep = edd_get_option( 'thousands_separator', ', ' ); 
  85.   $decimal_sep = edd_get_option( 'decimal_separator', '.' ); 
  86.  
  87.   // Format the amount 
  88.   if ( $decimal_sep == ', ' && false !== ( $sep_found = strpos( $amount, $decimal_sep ) ) ) { 
  89.       $whole = substr( $amount, 0, $sep_found ); 
  90.       $part = substr( $amount, $sep_found + 1, ( strlen( $amount ) - 1 ) ); 
  91.       $amount = $whole . '.' . $part; 
  92.   } 
  93.  
  94.   // Strip , from the amount (if set as the thousands separator) 
  95.   if ( $thousands_sep == ', ' && false !== ( $found = strpos( $amount, $thousands_sep ) ) ) { 
  96.       $amount = str_replace( ', ', '', $amount ); 
  97.   } 
  98.  
  99.   // Strip ' ' from the amount (if set as the thousands separator) 
  100.   if ( $thousands_sep == ' ' && false !== ( $found = strpos( $amount, $thousands_sep ) ) ) { 
  101.       $amount = str_replace( ' ', '', $amount ); 
  102.   } 
  103.  
  104.   if ( empty( $amount ) ) { 
  105.       $amount = 0; 
  106.   } 
  107.  
  108.   $decimals = apply_filters( 'edd_format_amount_decimals', $decimals ? 2 : 0, $amount ); 
  109.   $formatted = number_format( $amount, $decimals, $decimal_sep, $thousands_sep ); 
  110.  
  111.   return apply_filters( 'edd_format_amount', $formatted, $amount, $decimals, $decimal_sep, $thousands_sep ); 
  112. }

  115. /** 
  116.  * Formats the currency display 
  117.  * 
  118.  * @since 1.0 
  119.  * @param string $price Price 
  120.  * @return array $currency Currencies displayed correctly 
  121.  */ 
  122. function edd_currency_filter( $price = '', $currency = '' ) { 
  123.   if( empty( $currency ) ) { 
  124.  
  125.       $currency = edd_get_currency(); 
  126.  
  127.   } 
  128.  
  129.   $position = edd_get_option( 'currency_position', 'before' ); 
  130.  
  131.   $negative = $price < 0; 
  132.  
  133.   if( $negative ) { 
  134.       $price = substr( $price, 1 ); // Remove proceeding "-" - 
  135.   } 
  136.  
  137.   $symbol = edd_currency_symbol( $currency ); 
  138.  
  139.   if ( $position == 'before' ): 
  140.       switch ( $currency ): 
  141.           case "GBP" : 
  142.           case "BRL" : 
  143.           case "EUR" : 
  144.           case "USD" : 
  145.           case "AUD" : 
  146.           case "CAD" : 
  147.           case "HKD" : 
  148.           case "MXN" : 
  149.           case "NZD" : 
  150.           case "SGD" : 
  151.           case "JPY" : 
  152.               $formatted = $symbol . $price; 
  153.               break; 
  154.           default : 
  155.               $formatted = $currency . ' ' . $price; 
  156.               break; 
  157.       endswitch; 
  158.       $formatted = apply_filters( 'edd_' . strtolower( $currency ) . '_currency_filter_before', $formatted, $currency, $price ); 
  159.   else : 
  160.       switch ( $currency ) : 
  161.           case "GBP" : 
  162.           case "BRL" : 
  163.           case "EUR" : 
  164.           case "USD" : 
  165.           case "AUD" : 
  166.           case "CAD" : 
  167.           case "HKD" : 
  168.           case "MXN" : 
  169.           case "SGD" : 
  170.           case "JPY" : 
  171.               $formatted = $price . $symbol; 
  172.               break; 
  173.           default : 
  174.               $formatted = $price . ' ' . $currency; 
  175.               break; 
  176.       endswitch; 
  177.       $formatted = apply_filters( 'edd_' . strtolower( $currency ) . '_currency_filter_after', $formatted, $currency, $price ); 
  178.   endif; 
  179.  
  180.   if( $negative ) { 
  181.       // Prepend the mins sign before the currency sign 
  182.       $formatted = '-' . $formatted; 
  183.   } 
  184.  
  185.   return $formatted; 
  186. }
