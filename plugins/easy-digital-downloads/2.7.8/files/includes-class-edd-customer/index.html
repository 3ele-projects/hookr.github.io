  836.   /** 
  837.    * Sanitize the data for update/create 
  838.    * 
  839.    * @since  2.3 
  840.    * @param  array $data The data to sanitize 
  841.    * @return array       The sanitized data, based off column defaults 
  842.    */ 
  843.   private function sanitize_columns( $data ) { 
  844.  
  845.       $columns = $this->db->get_columns(); 
  846.       $default_values = $this->db->get_column_defaults(); 
  847.  
  848.       foreach ( $columns as $key => $type ) { 
  849.  
  850.           // Only sanitize data that we were provided 
  851.           if ( ! array_key_exists( $key, $data ) ) { 
  852.               continue; 
  853.           } 
  854.  
  855.           switch( $type ) { 
  856.  
  857.               case '%s': 
  858.                   if ( 'email' == $key ) { 
  859.                       $data[$key] = sanitize_email( $data[$key] ); 
  860.                   } elseif ( 'notes' == $key ) { 
  861.                       $data[$key] = strip_tags( $data[$key] ); 
  862.                   } else { 
  863.                       $data[$key] = sanitize_text_field( $data[$key] ); 
  864.                   } 
  865.                   break; 
  866.  
  867.               case '%d': 
  868.                   if ( ! is_numeric( $data[$key] ) || (int) $data[$key] !== absint( $data[$key] ) ) { 
  869.                       $data[$key] = $default_values[$key]; 
  870.                   } else { 
  871.                       $data[$key] = absint( $data[$key] ); 
  872.                   } 
  873.                   break; 
  874.  
  875.               case '%f': 
  876.                   // Convert what was given to a float 
  877.                   $value = floatval( $data[$key] ); 
  878.  
  879.                   if ( ! is_float( $value ) ) { 
  880.                       $data[$key] = $default_values[$key]; 
  881.                   } else { 
  882.                       $data[$key] = $value; 
  883.                   } 
  884.                   break; 
  885.  
  886.               default: 
  887.                   $data[$key] = sanitize_text_field( $data[$key] ); 
  888.                   break; 
  889.  
  890.           } 
  891.  
  892.       } 
  893.  
  894.       return $data; 
  895.   }
