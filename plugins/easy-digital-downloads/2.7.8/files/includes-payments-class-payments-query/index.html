  156.  
  157.       $custom_output = array( 
  158.           'payments',  
  159.           'edd_payments',  
  160.  ); 
  161.  
  162.       if ( ! in_array( $this->args['output'], $custom_output ) ) { 
  163.           return $query->posts; 
  164.       } 
  165.  
  166.       if ( $query->have_posts() ) { 
  167.           while ( $query->have_posts() ) { 
  168.               $query->the_post(); 
  169.  
  170.               $payment_id = get_post()->ID; 
  171.               $payment = edd_get_payment( $payment_id ); 
  172.  
  173.               if ( edd_get_option( 'enable_sequential' ) ) { 
  174.                   // Backwards Compatibility, needs to set `payment_number` attribute 
  175.                   $payment->payment_number = $payment->number; 
  176.               } 
  177.  
  178.               $this->payments[] = apply_filters( 'edd_payment', $payment, $payment_id, $this ); 
  179.           } 
  180.  
  181.           wp_reset_postdata(); 
  182.       } 
  183.  
  184.       add_action( 'edd_post_get_payments', array( $this, 'date_filter_post' ) ); 
  185.  
  186.       do_action( 'edd_post_get_payments', $this ); 
  187.  
  188.       return $this->payments; 
  189.   }

  357.    * Search 
  358.    * 
  359.    * @access public 
  360.    * @since 1.8 
  361.    * @return void

  363.   public function search() { 
  364.  
  365.       if( ! isset( $this->args['s'] ) ) { 
  366.           return; 
  367.       } 
  368.  
  369.       $search = trim( $this->args['s'] ); 
  370.  
  371.       if( empty( $search ) ) { 
  372.           return; 
  373.       } 
  374.  
  375.       $is_email = is_email( $search ) || strpos( $search, '@' ) !== false; 
  376.       $is_user = strpos( $search, strtolower( 'user:' ) ) !== false; 
  377.  
  378.       if ( ! empty( $this->args['search_in_notes'] ) ) { 
  379.  
  380.           $notes = edd_get_payment_notes( 0, $search ); 
  381.  
  382.           if( ! empty( $notes ) ) { 
  383.  
  384.               $payment_ids = wp_list_pluck( (array) $notes, 'comment_post_ID' ); 
  385.  
  386.               $this->__set( 'post__in', $payment_ids ); 
  387.           } 
  388.  
  389.           $this->__unset( 's' ); 
  390.  
  391.       } elseif ( $is_email || strlen( $search ) == 32 ) { 
  392.  
  393.           $key = $is_email ? '_edd_payment_user_email' : '_edd_payment_purchase_key'; 
  394.           $search_meta = array( 
  395.               'key' => $key,  
  396.               'value' => $search,  
  397.               'compare' => 'LIKE' 
  398.  ); 
  399.  
  400.           $this->__set( 'meta_query', $search_meta ); 
  401.           $this->__unset( 's' ); 
  402.  
  403.       } elseif ( $is_user ) { 
  404.  
  405.           $search_meta = array( 
  406.               'key' => '_edd_payment_user_id',  
  407.               'value' => trim( str_replace( 'user:', '', strtolower( $search ) ) ) 
  408.  ); 
  409.  
  410.           $this->__set( 'meta_query', $search_meta ); 
  411.  
  412.           if( edd_get_option( 'enable_sequential' ) ) { 
  413.  
  414.               $search_meta = array( 
  415.                   'key' => '_edd_payment_number',  
  416.                   'value' => $search,  
  417.                   'compare' => 'LIKE' 
  418.  ); 
  419.  
  420.               $this->__set( 'meta_query', $search_meta ); 
  421.  
  422.               $this->args['meta_query']['relation'] = 'OR'; 
  423.  
  424.           } 
  425.  
  426.           $this->__unset( 's' ); 
  427.  
  428.       } elseif ( 
  429.           edd_get_option( 'enable_sequential' ) && 
  430.           ( 
  431.               false !== strpos( $search, edd_get_option( 'sequential_prefix' ) ) || 
  432.               false !== strpos( $search, edd_get_option( 'sequential_postfix' ) ) 
  433.  ) 
  434.  ) { 
  435.  
  436.           $search_meta = array( 
  437.               'key' => '_edd_payment_number',  
  438.               'value' => $search,  
  439.               'compare' => 'LIKE' 
  440.  ); 
  441.  
  442.           $this->__set( 'meta_query', $search_meta ); 
  443.           $this->__unset( 's' ); 
  444.  
  445.       } elseif ( is_numeric( $search ) ) { 
  446.  
  447.           $post = get_post( $search ); 
  448.  
  449.           if( is_object( $post ) && $post->post_type == 'edd_payment' ) { 
  450.  
  451.               $arr = array(); 
  452.               $arr[] = $search; 
  453.               $this->__set( 'post__in', $arr ); 
  454.               $this->__unset( 's' ); 
  455.           } 
  456.  
  457.       } elseif ( '#' == substr( $search, 0, 1 ) ) { 
  458.  
  459.           $search = str_replace( '#:', '', $search ); 
  460.           $search = str_replace( '#', '', $search ); 
  461.           $this->__set( 'download', $search ); 
  462.           $this->__unset( 's' ); 
  463.  
  464.       } elseif ( 0 === strpos( $search, 'discount:' ) ) { 
  465.  
  466.           $search = trim( str_replace( 'discount:', '', $search ) ); 
  467.           $search = 'discount.*' . $search; 
  468.  
  469.           $search_meta = array( 
  470.               'key' => '_edd_payment_meta',  
  471.               'value' => $search,  
  472.               'compare' => 'REGEXP',  
  473.  ); 
  474.  
  475.           $this->__set( 'meta_query', $search_meta ); 
  476.           $this->__unset( 's' ); 
  477.  
  478.       } else { 
  479.           $this->__set( 's', $search ); 
  480.       } 
  481.  
  482.   }
