  676. /** 
  677.  * Retrieve the name of the highest priority template file that exists. 
  678.  * 
  679.  * Searches in the STYLESHEETPATH before TEMPLATEPATH so that themes which 
  680.  * inherit from a parent theme can just overload one file. If the template is 
  681.  * not found in either of those, it looks in the theme-compat folder last. 
  682.  * 
  683.  * Taken from bbPress 
  684.  * 
  685.  * @since 1.2 
  686.  * 
  687.  * @param string|array $template_names Template file(s) to search for, in order. 
  688.  * @param bool $load If true the template file will be loaded if it is found. 
  689.  * @param bool $require_once Whether to require_once or require. Default true. 
  690.  *   Has no effect if $load is false. 
  691.  * @return string The template filename if one is located. 
  692.  */ 
  693. function edd_locate_template( $template_names, $load = false, $require_once = true ) { 
  694.   // No file found yet 
  695.   $located = false; 
  696.  
  697.   // Try to find a template file 
  698.   foreach ( (array) $template_names as $template_name ) { 
  699.  
  700.       // Continue if template is empty 
  701.       if ( empty( $template_name ) ) 
  702.           continue; 
  703.  
  704.       // Trim off any slashes from the template name 
  705.       $template_name = ltrim( $template_name, '/' ); 
  706.  
  707.       // try locating this template file by looping through the template paths 
  708.       foreach( edd_get_theme_template_paths() as $template_path ) { 
  709.  
  710.           if( file_exists( $template_path . $template_name ) ) { 
  711.               $located = $template_path . $template_name; 
  712.               break; 
  713.           } 
  714.       } 
  715.  
  716.       if( $located ) { 
  717.           break; 
  718.       } 
  719.   } 
  720.  
  721.   if ( ( true == $load ) && ! empty( $located ) ) 
  722.       load_template( $located, $require_once ); 
  723.  
  724.   return $located; 
  725. }

  733. function edd_get_theme_template_paths() {

  738.       1 => trailingslashit( get_stylesheet_directory() ) . $template_dir,
