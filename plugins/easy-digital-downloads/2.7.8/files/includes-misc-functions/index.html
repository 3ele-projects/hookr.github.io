  690. /**

  693.  * @since 1.6 
  694.  * @return string $timezone The timezone ID

  696. function edd_get_timezone_id() { 
  697.  
  698.   // if site timezone string exists, return it 
  699.   if ( $timezone = get_option( 'timezone_string' ) ) 
  700.       return $timezone; 
  701.  
  702.   // get UTC offset, if it isn't set return UTC 
  703.   if ( ! ( $utc_offset = 3600 * get_option( 'gmt_offset', 0 ) ) ) 
  704.       return 'UTC'; 
  705.  
  706.   // attempt to guess the timezone string from the UTC offset 
  707.   $timezone = timezone_name_from_abbr( '', $utc_offset ); 
  708.  
  709.   // last try, guess timezone string manually 
  710.   if ( $timezone === false ) { 
  711.  
  712.       $is_dst = date( 'I' ); 
  713.  
  714.       foreach ( timezone_abbreviations_list() as $abbr ) { 
  715.           foreach ( $abbr as $city ) { 
  716.               if ( $city['dst'] == $is_dst &&  $city['offset'] == $utc_offset ) 
  717.                   return $city['timezone_id']; 
  718.           } 
  719.       } 
  720.   } 
  721.  
  722.   // fallback 
  723.   return 'UTC'; 
  724. }

  726. /**

  729.  * @since    1.7 
  730.  * @internal Updated in 2.6 
  731.  * @param    object|array $object An object or an array of objects 
  732.  * @return   array                An array or array of arrays, converted from the provided object(s) 
  733.  */ 
  734. function edd_object_to_array( $object = array() ) { 
  735.  
  736.   if ( empty( $object ) || ( ! is_object( $object ) && ! is_array( $object ) ) ) { 
  737.       return $object; 
  738.   } 
  739.  
  740.   if ( is_array( $object ) ) { 
  741.       $return = array(); 
  742.       foreach ( $object as $item ) { 
  743.           if ( is_a( $object, 'EDD_Payment' ) ) { 
  744.               $return[] = $object->array_convert(); 
  745.           } else { 
  746.               $return[] = edd_object_to_array( $item ); 
  747.           } 
  748.  
  749.       } 
  750.   } else { 
  751.       if ( is_a( $object, 'EDD_Payment' ) ) { 
  752.           $return = $object->array_convert(); 
  753.       } else {

  756.           // Now look at the items that came back and convert any nested objects to arrays
