  19.  * Triggered by the edd_update_payment_status() function.

  27. function edd_complete_purchase( $payment_id, $new_status, $old_status ) { 
  28.   if ( $old_status == 'publish' || $old_status == 'complete' ) { 
  29.       return; // Make sure that payments are only completed once 
  30.   } 
  31.  
  32.   // Make sure the payment completion is only processed when new status is complete 
  33.   if ( $new_status != 'publish' && $new_status != 'complete' ) { 
  34.       return; 
  35.   } 
  36.  
  37.   $payment = new EDD_Payment( $payment_id ); 
  38.  
  39.   $creation_date = get_post_field( 'post_date', $payment_id, 'raw' ); 
  40.   $completed_date = $payment->completed_date; 
  41.   $user_info = $payment->user_info; 
  42.   $customer_id = $payment->customer_id; 
  43.   $amount = $payment->total; 
  44.   $cart_details = $payment->cart_details; 
  45.  
  46.   do_action( 'edd_pre_complete_purchase', $payment_id ); 
  47.  
  48.   if ( is_array( $cart_details ) ) { 
  49.  
  50.       // Increase purchase count and earnings 
  51.       foreach ( $cart_details as $cart_index => $download ) { 
  52.  
  53.           // "bundle" or "default" 
  54.           $download_type = edd_get_download_type( $download['id'] ); 
  55.           $price_id = isset( $download['item_number']['options']['price_id'] ) ? (int) $download['item_number']['options']['price_id'] : false; 
  56.           // Increase earnings and fire actions once per quantity number 
  57.           for( $i = 0; $i < $download['quantity']; $i++ ) { 
  58.  
  59.               // Ensure these actions only run once, ever 
  60.               if ( empty( $completed_date ) ) { 
  61.  
  62.                   edd_record_sale_in_log( $download['id'], $payment_id, $price_id, $creation_date ); 
  63.                   do_action( 'edd_complete_download_purchase', $download['id'], $payment_id, $download_type, $download, $cart_index ); 
  64.  
  65.               } 
  66.  
  67.           } 
  68.  
  69.           $increase_earnings = $download['price']; 
  70.           if ( ! empty( $download['fees'] ) ) { 
  71.               foreach ( $download['fees'] as $fee ) { 
  72.                   if ( $fee['amount'] > 0 ) { 
  73.                       continue; 
  74.                   } 
  75.                   $increase_earnings += $fee['amount']; 
  76.               } 
  77.           } 
  78.  
  79.           // Increase the earnings for this download ID 
  80.           edd_increase_earnings( $download['id'], $increase_earnings ); 
  81.           edd_increase_purchase_count( $download['id'], $download['quantity'] ); 
  82.  
  83.       } 
  84.  
  85.       // Clear the total earnings cache 
  86.       delete_transient( 'edd_earnings_total' ); 
  87.       // Clear the This Month earnings (this_monththis_month is NOT a typo) 
  88.       delete_transient( md5( 'edd_earnings_this_monththis_month' ) ); 
  89.       delete_transient( md5( 'edd_earnings_todaytoday' ) ); 
  90.   } 
  91.  
  92.  
  93.   // Increase the customer's purchase stats 
  94.   $customer = new EDD_Customer( $customer_id ); 
  95.   $customer->increase_purchase_count(); 
  96.   $customer->increase_value( $amount ); 
  97.  
  98.   edd_increase_total_earnings( $amount ); 
  99.  
  100.   // Check for discount codes and increment their use counts 
  101.   if ( ! empty( $user_info['discount'] ) && $user_info['discount'] !== 'none' ) { 
  102.  
  103.       $discounts = array_map( 'trim', explode( ', ', $user_info['discount'] ) ); 
  104.  
  105.       if( ! empty( $discounts ) ) { 
  106.  
  107.           foreach( $discounts as $code ) { 
  108.  
  109.               edd_increase_discount_usage( $code ); 
  110.  
  111.           } 
  112.  
  113.       } 
  114.   } 
  115.  
  116.  
  117.   // Ensure this action only runs once ever 
  118.   if( empty( $completed_date ) ) { 
  119.  
  120.       // Save the completed date 
  121.       $payment->completed_date = current_time( 'mysql' ); 
  122.       $payment->save(); 
  123.  
  124.       do_action( 'edd_complete_purchase', $payment_id ); 
  125.  
  126.   } 
  127.  
  128.   // Empty the shopping cart 
  129.   edd_empty_cart(); 
  130. } 
  131. add_action( 'edd_update_payment_status', 'edd_complete_purchase', 100, 3 );

  245. /** 
  246.  * Listens to the updated_postmeta hook for our backwards compatible payment_meta updates, and runs through them 
  247.  * 
  248.  * @since  2.3 
  249.  * @param  int $meta_id    The Meta ID that was updated 
  250.  * @param  int $object_id  The Object ID that was updated (post ID) 
  251.  * @param  string $meta_key   The Meta key that was updated 
  252.  * @param  string|int|float $meta_value The Value being updated 
  253.  * @return bool|int             If successful the number of rows updated, if it fails, false 
  254.  */ 
  255. function edd_update_payment_backwards_compat( $meta_id, $object_id, $meta_key, $meta_value ) { 
  256.  
  257.   $meta_keys = array( '_edd_payment_meta', '_edd_payment_tax' ); 
  258.  
  259.   if ( ! in_array( $meta_key, $meta_keys ) ) { 
  260.       return; 
  261.   } 
  262.  
  263.   global $wpdb; 
  264.   switch( $meta_key ) { 
  265.  
  266.       case '_edd_payment_meta': 
  267.           $meta_value = maybe_unserialize( $meta_value ); 
  268.  
  269.           if( ! isset( $meta_value['tax'] ) ) { 
  270.               return; 
  271.           } 
  272.  
  273.           $tax_value = $meta_value['tax']; 
  274.  
  275.           $data = array( 'meta_value' => $tax_value ); 
  276.           $where = array( 'post_id' => $object_id, 'meta_key' => '_edd_payment_tax' ); 
  277.           $data_format = array( '%f' ); 
  278.           $where_format = array( '%d', '%s' ); 
  279.           break; 
  280.  
  281.       case '_edd_payment_tax': 
  282.           $tax_value = ! empty( $meta_value ) ? $meta_value : 0; 
  283.           $current_meta = edd_get_payment_meta( $object_id, '_edd_payment_meta', true ); 
  284.  
  285.           $current_meta['tax'] = $tax_value; 
  286.           $new_meta = maybe_serialize( $current_meta ); 
  287.  
  288.           $data = array( 'meta_value' => $new_meta ); 
  289.           $where = array( 'post_id' => $object_id, 'meta_key' => '_edd_payment_meta' ); 
  290.           $data_format = array( '%s' ); 
  291.           $where_format = array( '%d', '%s' ); 
  292.  
  293.           break; 
  294.  
  295.   } 
  296.  
  297.   $updated = $wpdb->update( $wpdb->postmeta, $data, $where, $data_format, $where_format ); 
  298.  
  299.   if ( ! empty( $updated ) ) { 
  300.       // Since we did a direct DB query, clear the postmeta cache. 
  301.       wp_cache_delete( $object_id, 'post_meta' ); 
  302.   } 
  303.  
  304.   return $updated; 
  305.  
  306.  
  307. } 
  308. add_action( 'updated_postmeta', 'edd_update_payment_backwards_compat', 10, 4 );

  352. /** 
  353.  * Process an attempt to complete a recoverable payment. 
  354.  * 
  355.  * @since  2.7 
  356.  * @return void 
  357.  */ 
  358. function edd_recover_payment() { 
  359.   if ( empty( $_GET['payment_id'] ) ) { 
  360.       return; 
  361.   } 
  362.  
  363.   $payment = new EDD_Payment( $_GET['payment_id'] ); 
  364.   if ( $payment->ID !== (int) $_GET['payment_id'] ) { 
  365.       return; 
  366.   } 
  367.  
  368.   if ( ! $payment->is_recoverable() ) { 
  369.       return; 
  370.   } 
  371.  
  372.   if ( is_user_logged_in() && $payment->user_id != get_current_user_id() ) { 
  373.       $redirect = get_permalink( edd_get_option( 'purchase_history_page' ) ); 
  374.       edd_set_error( 'edd-payment-recovery-user-mismatch', __( 'Error resuming payment.', 'easy-digital-downloads' ) ); 
  375.       wp_redirect( $redirect ); 
  376.   } 
  377.  
  378.   $payment->add_note( __( 'Payment recovery triggered URL', 'easy-digital-downloads' ) ); 
  379.  
  380.   // Empty out the cart. 
  381.   EDD()->cart->empty_cart(); 
  382.  
  383.   // Recover any downloads. 
  384.   foreach ( $payment->cart_details as $download ) { 
  385.       edd_add_to_cart( $download['id'], $download['item_number']['options'] ); 
  386.  
  387.       // Recover any item specific fees. 
  388.       if ( ! empty( $download['fees'] ) ) { 
  389.           foreach ( $download['fees'] as $fee ) { 
  390.               EDD()->fees->add_fee( $fee ); 
  391.           } 
  392.       } 
  393.   } 
  394.  
  395.   // Recover any global fees. 
  396.   foreach ( $payment->fees as $fee ) { 
  397.       EDD()->fees->add_fee( $fee ); 
  398.   } 
  399.  
  400.   // Recover any discounts. 
  401.   if ( 'none' !== $payment->discounts && ! empty( $payment->discounts ) ) { 
  402.       $discounts = ! is_array( $payment->discounts ) ? explode( ', ', $payment->discounts ) : $payment->discounts; 
  403.  
  404.       foreach ( $discounts as $discount ) { 
  405.           edd_set_cart_discount( $discount ); 
  406.       } 
  407.   } 
  408.  
  409.   EDD()->session->set( 'edd_resume_payment', $payment->ID ); 
  410.  
  411.   $redirect_args = array( 'payment-mode' => $payment->gateway ); 
  412.   $redirect = add_query_arg( $redirect_args, edd_get_checkout_uri() ); 
  413.   wp_redirect( $redirect ); 
  414.   exit; 
  415. } 
  416. add_action( 'edd_recover_payment', 'edd_recover_payment' );
