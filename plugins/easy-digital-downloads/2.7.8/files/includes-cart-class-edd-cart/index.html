  374.   public function add( $download_id, $options = array() ) { 
  375.       $download = new EDD_Download( $download_id ); 
  376.  
  377.       if ( empty( $download->ID ) ) { 
  378.           return; // Not a download product 
  379.       } 
  380.  
  381.       if ( ! $download->can_purchase() ) { 
  382.           return; // Do not allow draft/pending to be purchased if can't edit. Fixes #1056 
  383.       } 
  384.  
  385.       do_action( 'edd_pre_add_to_cart', $download_id, $options ); 
  386.  
  387.       $this->contents = apply_filters( 'edd_pre_add_to_cart_contents', $this->contents ); 
  388.  
  389.       $quantities_enabled = edd_item_quantities_enabled() && ! edd_download_quantities_disabled( $download_id ); 
  390.  
  391.       if ( $download->has_variable_prices()  && ! isset( $options['price_id'] ) ) { 
  392.           // Forces to the default price ID if none is specified and download has variable prices 
  393.           $options['price_id'] = get_post_meta( $download->ID, '_edd_default_price_id', true ); 
  394.       } 
  395.  
  396.       if ( isset( $options['quantity'] ) ) { 
  397.           if ( is_array( $options['quantity'] ) ) { 
  398.               $quantity = array(); 
  399.               foreach ( $options['quantity'] as $q ) { 
  400.                   $quantity[] = $quantities_enabled ? absint( preg_replace( '/[^0-9\.]/', '', $q ) ) : 1; 
  401.               } 
  402.           } else { 
  403.               $quantity = $quantities_enabled ? absint( preg_replace( '/[^0-9\.]/', '', $options['quantity'] ) ) : 1; 
  404.           } 
  405.  
  406.           unset( $options['quantity'] ); 
  407.       } else { 
  408.           $quantity = 1; 
  409.       } 
  410.  
  411.       // If the price IDs are a string and is a coma separated list, make it an array (allows custom add to cart URLs) 
  412.       if ( isset( $options['price_id'] ) && ! is_array( $options['price_id'] ) && false !== strpos( $options['price_id'], ', ' ) ) { 
  413.           $options['price_id'] = explode( ', ', $options['price_id'] ); 
  414.       } 
  415.  
  416.       $items = array(); 
  417.  
  418.       if ( isset( $options['price_id'] ) && is_array( $options['price_id'] ) ) { 
  419.           // Process multiple price options at once 
  420.           foreach ( $options['price_id'] as $key => $price ) { 
  421.               $items[] = array( 
  422.                   'id' => $download_id,  
  423.                   'options' => array( 
  424.                       'price_id' => preg_replace( '/[^0-9\.-]/', '', $price ) 
  425.  ),  
  426.                   'quantity' => $quantity[ $key ],  
  427.  ); 
  428.           } 
  429.       } else { 
  430.           // Sanitize price IDs 
  431.           foreach( $options as $key => $option ) { 
  432.               if ( 'price_id' == $key ) { 
  433.                   $options[ $key ] = preg_replace( '/[^0-9\.-]/', '', $option ); 
  434.               } 
  435.           } 
  436.  
  437.           // Add a single item 
  438.           $items[] = array( 
  439.               'id' => $download_id,  
  440.               'options' => $options,  
  441.               'quantity' => $quantity 
  442.  ); 
  443.       } 
  444.  
  445.       foreach ( $items as &$item ) { 
  446.           $item = apply_filters( 'edd_add_to_cart_item', $item ); 
  447.           $to_add = $item; 
  448.  
  449.           if ( ! is_array( $to_add ) ) { 
  450.               return; 
  451.           } 
  452.  
  453.           if ( ! isset( $to_add['id'] ) || empty( $to_add['id'] ) ) { 
  454.               return; 
  455.           } 
  456.  
  457.           if ( edd_item_in_cart( $to_add['id'], $to_add['options'] ) && edd_item_quantities_enabled() ) { 
  458.               $key = edd_get_item_position_in_cart( $to_add['id'], $to_add['options'] ); 
  459.  
  460.               if ( is_array( $quantity ) ) { 
  461.                   $this->contents[ $key ]['quantity'] += $quantity[ $key ]; 
  462.               } else { 
  463.                   $this->contents[ $key ]['quantity'] += $quantity; 
  464.               } 
  465.           } else { 
  466.               $this->contents[] = $to_add; 
  467.           } 
  468.       } 
  469.  
  470.       unset( $item ); 
  471.  
  472.       $this->update_cart(); 
  473.  
  474.       do_action( 'edd_post_add_to_cart', $download_id, $options, $items ); 
  475.  
  476.       // Clear all the checkout errors, if any 
  477.       edd_clear_errors(); 
  478.  
  479.       return count( $this->contents ) - 1; 
  480.   }

  630.    * @access public 
  631.    * 
  632.    * @param array       $item     Cart item. 
  633.    * @param bool|string $discount False to use the cart discounts or a string to check with a discount code. 
  634.    * @return float The discounted amount 
  635.    */ 
  636.   public function get_item_discount_amount( $item = array(), $discount = false ) { 
  637.       global $edd_is_last_cart_item, $edd_flat_discount_total; 
  638.  
  639.       // If we're not meeting the requirements of the $item array, return or set them 
  640.       if ( empty( $item ) || empty( $item['id'] ) ) { 
  641.           return 0; 
  642.       } 
  643.  
  644.       // Quantity is a requirement of the cart options array to determine the discounted price 
  645.       if ( empty( $item['quantity'] ) ) { 
  646.           return 0; 
  647.       } 
  648.  
  649.       if ( ! isset( $item['options'] ) ) { 
  650.           $item['options'] = array(); 
  651.       } 
  652.  
  653.       $amount = 0; 
  654.       $price = $this->get_item_price( $item['id'], $item['options'] ); 
  655.       $discounted_price = $price; 
  656.  
  657.       $discounts = false === $discount ? $this->get_discounts() : array( $discount ); 
  658.  
  659.       if ( $discounts ) { 
  660.           foreach ( $discounts as $discount ) { 
  661.               $code_id = edd_get_discount_id_by_code( $discount ); 
  662.  
  663.               // Check discount exists 
  664.               if( ! $code_id ) { 
  665.                   continue; 
  666.               } 
  667.  
  668.               $reqs = edd_get_discount_product_reqs( $code_id ); 
  669.               $excluded_products = edd_get_discount_excluded_products( $code_id ); 
  670.  
  671.               // Make sure requirements are set and that this discount shouldn't apply to the whole cart 
  672.               if ( ! empty( $reqs ) && edd_is_discount_not_global( $code_id ) ) { 
  673.                   // This is a product(s) specific discount 
  674.                   foreach ( $reqs as $download_id ) { 
  675.                       if ( $download_id == $item['id'] && ! in_array( $item['id'], $excluded_products ) ) { 
  676.                           $discounted_price -= $price - edd_get_discounted_amount( $discount, $price ); 
  677.                       } 
  678.                   } 
  679.               } else { 
  680.                   // This is a global cart discount 
  681.                   if( ! in_array( $item['id'], $excluded_products ) ) { 
  682.                       if( 'flat' === edd_get_discount_type( $code_id ) ) { 
  683.                           /** * 
  684.                            * In order to correctly record individual item amounts, global flat rate discounts 
  685.                            * are distributed across all cart items. The discount amount is divided by the number 
  686.                            * of items in the cart and then a portion is evenly applied to each cart item 
  687.                            */ 
  688.                           $items_subtotal = 0.00; 
  689.                           $cart_items = $this->get_contents(); 
  690.                           foreach ( $cart_items as $cart_item ) { 
  691.                               if ( ! in_array( $cart_item['id'], $excluded_products ) ) { 
  692.                                   $item_price = $this->get_item_price( $cart_item['id'], $cart_item['options'] ); 
  693.                                   $items_subtotal += $item_price * $cart_item['quantity']; 
  694.                               } 
  695.                           } 
  696.  
  697.                           $subtotal_percent = ( ( $price * $item['quantity'] ) / $items_subtotal ); 
  698.                           $code_amount = edd_get_discount_amount( $code_id ); 
  699.                           $discounted_amount = $code_amount * $subtotal_percent; 
  700.                           $discounted_price -= $discounted_amount; 
  701.  
  702.                           $edd_flat_discount_total += round( $discounted_amount, edd_currency_decimal_filter() ); 
  703.  
  704.                           if ( $edd_is_last_cart_item && $edd_flat_discount_total < $code_amount ) { 
  705.                               $adjustment = $code_amount - $edd_flat_discount_total; 
  706.                               $discounted_price -= $adjustment; 
  707.                           } 
  708.                       } else { 
  709.                           $discounted_price -= $price - edd_get_discounted_amount( $discount, $price ); 
  710.                       } 
  711.                   } 
  712.               } 
  713.  
  714.               if ( $discounted_price < 0 ) { 
  715.                   $discounted_price = 0; 
  716.               } 
  717.           } 
  718.  
  719.           $amount = ( $price - apply_filters( 'edd_get_cart_item_discounted_amount', $discounted_price, $discounts, $item, $price ) ); 
  720.  
  721.           if ( 'flat' !== edd_get_discount_type( $code_id ) ) { 
  722.               $amount = $amount * $item['quantity']; 
  723.           } 
  724.       } 
  725.  
  726.       return $amount; 
  727.   }
