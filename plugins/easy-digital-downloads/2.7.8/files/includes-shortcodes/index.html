  719.  ), $atts, 'edd_receipt' ); 
  720.  
  721.   $session = edd_get_purchase_session(); 
  722.   if ( isset( $_GET['payment_key'] ) ) { 
  723.       $payment_key = urldecode( $_GET['payment_key'] ); 
  724.   } else if ( $session ) { 
  725.       $payment_key = $session['purchase_key'];

  736.   $user_can_view = edd_can_view_receipt( $payment_key ); 
  737.  
  738.   // Key was provided, but user is logged out. Offer them the ability to login and view the receipt 
  739.   if ( ! $user_can_view && ! empty( $payment_key ) && ! is_user_logged_in() && ! edd_is_guest_payment( $payment_id ) ) {

  824. /** 
  825.  * Process Profile Updater Form 
  826.  * 
  827.  * Processes the profile updater form by updating the necessary fields 
  828.  * 
  829.  * @since 1.4 
  830.  * @author Sunny Ratilal 
  831.  * @param array $data Data sent from the profile editor 
  832.  * @return void 
  833.  */ 
  834. function edd_process_profile_editor_updates( $data ) { 
  835.   // Profile field change request 
  836.   if ( empty( $_POST['edd_profile_editor_submit'] ) && !is_user_logged_in() ) { 
  837.       return false; 
  838.   } 
  839.  
  840.   // Pending users can't edit their profile 
  841.   if ( edd_user_pending_verification() ) { 
  842.       return false; 
  843.   } 
  844.  
  845.   // Nonce security 
  846.   if ( ! wp_verify_nonce( $data['edd_profile_editor_nonce'], 'edd-profile-editor-nonce' ) ) { 
  847.       return false; 
  848.   } 
  849.  
  850.   $user_id = get_current_user_id(); 
  851.   $old_user_data = get_userdata( $user_id ); 
  852.  
  853.   $display_name = isset( $data['edd_display_name'] )    ? sanitize_text_field( $data['edd_display_name'] )    : $old_user_data->display_name; 
  854.   $first_name = isset( $data['edd_first_name'] )      ? sanitize_text_field( $data['edd_first_name'] )      : $old_user_data->first_name; 
  855.   $last_name = isset( $data['edd_last_name'] )       ? sanitize_text_field( $data['edd_last_name'] )       : $old_user_data->last_name; 
  856.   $email = isset( $data['edd_email'] )           ? sanitize_email( $data['edd_email'] )                : $old_user_data->user_email; 
  857.   $line1 = isset( $data['edd_address_line1'] )   ? sanitize_text_field( $data['edd_address_line1'] )   : ''; 
  858.   $line2 = isset( $data['edd_address_line2'] )   ? sanitize_text_field( $data['edd_address_line2'] )   : ''; 
  859.   $city = isset( $data['edd_address_city'] )    ? sanitize_text_field( $data['edd_address_city'] )    : ''; 
  860.   $state = isset( $data['edd_address_state'] )   ? sanitize_text_field( $data['edd_address_state'] )   : ''; 
  861.   $zip = isset( $data['edd_address_zip'] )     ? sanitize_text_field( $data['edd_address_zip'] )     : ''; 
  862.   $country = isset( $data['edd_address_country'] ) ? sanitize_text_field( $data['edd_address_country'] ) : ''; 
  863.  
  864.   $userdata = array( 
  865.       'ID' => $user_id,  
  866.       'first_name' => $first_name,  
  867.       'last_name' => $last_name,  
  868.       'display_name' => $display_name,  
  869.       'user_email' => $email 
  870.  ); 
  871.  
  872.  
  873.   $address = array( 
  874.       'line1' => $line1,  
  875.       'line2' => $line2,  
  876.       'city' => $city,  
  877.       'state' => $state,  
  878.       'zip' => $zip,  
  879.       'country' => $country 
  880.  ); 
  881.  
  882.   do_action( 'edd_pre_update_user_profile', $user_id, $userdata ); 
  883.  
  884.   // New password 
  885.   if ( ! empty( $data['edd_new_user_pass1'] ) ) { 
  886.       if ( $data['edd_new_user_pass1'] !== $data['edd_new_user_pass2'] ) { 
  887.           edd_set_error( 'password_mismatch', __( 'The passwords you entered do not match. Please try again.', 'easy-digital-downloads' ) ); 
  888.       } else { 
  889.           $userdata['user_pass'] = $data['edd_new_user_pass1']; 
  890.       } 
  891.   } 
  892.  
  893.   // Make sure the new email doesn't belong to another user 
  894.   if( $email != $old_user_data->user_email ) { 
  895.       // Make sure the new email is valid 
  896.       if( ! is_email( $email ) ) { 
  897.           edd_set_error( 'email_invalid', __( 'The email you entered is invalid. Please enter a valid email.', 'easy-digital-downloads' ) ); 
  898.       } 
  899.  
  900.       // Make sure the new email doesn't belong to another user 
  901.       if( email_exists( $email ) ) { 
  902.           edd_set_error( 'email_exists', __( 'The email you entered belongs to another user. Please use another.', 'easy-digital-downloads' ) ); 
  903.       } 
  904.   } 
  905.  
  906.   // Check for errors 
  907.   $errors = edd_get_errors(); 
  908.

  911.       wp_redirect( $data['edd_redirect'] ); 
  912.       edd_die(); 
  913.   } 
  914.  
  915.   // Update the user

  917.   $updated = wp_update_user( $userdata ); 
  918.  
  919.   // Possibly update the customer 
  920.   $customer = new EDD_Customer( $user_id, true ); 
  921.   if ( $customer->email === $email || ( is_array( $customer->emails ) && in_array( $email, $customer->emails ) ) ) { 
  922.       $customer->set_primary_email( $email ); 
  923.   }; 
  924.  
  925.   if ( $customer->id > 0 ) { 
  926.       $update_args = array( 
  927.           'name' => $first_name . ' ' . $last_name,  
  928.  ); 
  929.  
  930.       $customer->update( $update_args ); 
  931.   } 
  932.  
  933.   if ( $updated ) {
