  44. function edd_process_paypal_purchase( $purchase_data ) { 
  45.   global $edd_options; 
  46.  
  47.   /**  
  48.   * purchase data comes in like this 
  49.   * 
  50.   $purchase_data = array( 
  51.       'downloads' => array of download IDs,  
  52.       'price' => total price of cart contents,  
  53.       'purchase_key' =>  // random key 
  54.       'user_email' => $user_email,  
  55.       'date' => date('Y-m-d H:i:s'),  
  56.       'user_id' => $user_id,  
  57.       'post_data' => $_POST,  
  58.       'user_info' => array of user's information and used discount code 
  59.       'cart_details' => array of cart details,  
  60.  ); 
  61.   */ 
  62.  
  63.   $payment_data = array(  
  64.       'price' => $purchase_data['price'],  
  65.       'date' => $purchase_data['date'],  
  66.       'user_email' => $purchase_data['user_email'],  
  67.       'purchase_key' => $purchase_data['purchase_key'],  
  68.       'currency' => $edd_options['currency'],  
  69.       'downloads' => $purchase_data['downloads'],  
  70.       'user_info' => $purchase_data['user_info'],  
  71.       'cart_details' => $purchase_data['cart_details'],  
  72.       'status' => 'pending' 
  73.  ); 
  74.  
  75.   // record the pending payment 
  76.   $payment = edd_insert_payment( $payment_data ); 
  77.    
  78.   if( $payment ) { 
  79.       // only send to paypal if the pending payment is created successfully 
  80.       $listener_url = trailingslashit( home_url() ).'?edd-listener=IPN'; 
  81.       $return_url = add_query_arg( 'payment-confirmation', 'paypal', get_permalink( $edd_options['success_page'] ) ); 
  82.       $cart_summary = edd_get_purchase_summary( $purchase_data, false ); 
  83.  
  84.       // one time payment 
  85.       if( edd_is_test_mode() ) { 
  86.           $paypal_redirect = 'https://www.sandbox.paypal.com/cgi-bin/webscr/?'; 
  87.       } else { 
  88.           $paypal_redirect = 'https://www.paypal.com/cgi-bin/webscr/?'; 
  89.       } 
  90.       $paypal_args = array( 
  91.           'cmd' => '_xclick',  
  92.           'amount' => $purchase_data['price'],  
  93.           'business' => $edd_options['paypal_email'],  
  94.           'item_name' => $cart_summary,  
  95.           'email' => $purchase_data['user_email'],  
  96.           'no_shipping' => '1',  
  97.           'no_note' => '1',  
  98.           'currency_code' => $edd_options['currency'],  
  99.           'item_number' => $purchase_data['purchase_key'],  
  100.           'charset' => 'UTF-8',  
  101.           'custom' => $payment,  
  102.           'rm' => '2',  
  103.           'return' => $return_url,  
  104.           'notify_url' => $listener_url 
  105.  ); 
  106.       //var_dump(http_build_query($paypal_args)); exit; 
  107.       $paypal_redirect .= http_build_query($paypal_args); 
  108.  
  109.       //var_dump(urldecode($paypal_redirect)); exit; 
  110.  
  111.       // get rid of cart contents 
  112.       edd_empty_cart(); 
  113.  
  114.       // Redirect to paypal 
  115.       wp_redirect( $paypal_redirect ); 
  116.       exit; 
  117.        
  118.   } else { 
  119.       // if errors are present, send the user back to the purchase page so they can be corrected

  126. /** 
  127.  * Listen For PayPal IPN 
  128.  * 
  129.  * Listens for a PayPal IPN requests and then sends to the processing function. 
  130.  * 
  131.  * @access      private 
  132.  * @since       1.0  
  133.  * @return      void

  135.  
  136. function edd_listen_for_paypal_ipn() { 
  137.   global $edd_options; 
  138.    
  139.   // regular PayPal IPN 
  140.   if( !isset( $edd_options['paypal_alternate_verification'] ) ) { 
  141.        
  142.       if( isset( $_GET['edd-listener'] ) && $_GET['edd-listener'] == 'IPN' ) { 
  143.           do_action( 'edd_verify_paypal_ipn' ); 
  144.       } 
  145.        
  146.   // alternate purchase verification     
  147.   } else {  
  148.       if( isset( $_GET['tx'] ) && isset( $_GET['st'] ) && isset( $_GET['amt'] ) && isset( $_GET['cc'] ) && isset( $_GET['cm'] ) && isset( $_GET['item_number'] ) ) { 
  149.           // we are using the alternate method of verifying PayPal purchases 
  150.           // setup each of the variables from PayPal 
  151.           $payment_status = $_GET['st']; 
  152.           $paypal_amount = $_GET['amt']; 
  153.           $payment_id = $_GET['cm']; 
  154.           $purchase_key = $_GET['item_number']; 
  155.           $currency = $_GET['cc']; 
  156.  
  157.           // retrieve the meta info for this payment 
  158.           $payment_meta = get_post_meta( $payment_id, '_edd_payment_meta', true ); 
  159.           $payment_amount = edd_format_amount( $payment_meta['amount'] ); 
  160.           if( $currency != $edd_options['currency'] ) { 
  161.               return; // the currency code is invalid 
  162.           } 
  163.           if( $paypal_amount != $payment_amount ) { 
  164.               return; // the prices don't match 
  165.           } 
  166.           if( $purchase_key != $payment_meta['key'] ) { 
  167.               return; // purchase keys don't match 
  168.           } 
  169.           if( strtolower( $payment_status ) != 'completed' || edd_is_test_mode() ) { 
  170.               return; // payment wasn't completed 
  171.           } 
  172.  
  173.           // everything has been verified, update the payment to "complete" 
  174.           edd_update_payment_status( $payment_id, 'publish' ); 
  175.       } 
  176.   } 
  177. } 
  178. add_action( 'init', 'edd_listen_for_paypal_ipn' );

  181. /** 
  182.  * Process PayPal IPN

  186.  * @return      void

  188.  
  189. function edd_process_paypal_ipn() { 
  190.   global $edd_options; 
  191.  
  192.   // instantiate the IpnListener class 
  193.   if( !class_exists( 'IpnListener' ) ) { 
  194.       include_once( EDD_PLUGIN_DIR . 'includes/gateways/libraries/paypal/ipnlistener.php' ); 
  195.   } 
  196.  
  197.   $listener = new IpnListener(); 
  198.  
  199.   if( edd_is_test_mode() ) { 
  200.       $listener->use_sandbox = true; 
  201.   } 
  202.  
  203.   if( isset( $edd_options['ssl'] ) ) { 
  204.       $listener->use_ssl = false; 
  205.   } 
  206.   // to post using the fsockopen() function rather than cURL, use: 
  207.   if( isset( $edd_options['paypal_disable_curl'] ) ) { 
  208.       $listener->use_curl = false; 
  209.   } 
  210.    
  211.   try { 
  212.       $listener->requirePostMethod(); 
  213.       $verified = $listener->processIpn(); 
  214.   } catch( Exception $e ) { 
  215.       wp_mail( get_bloginfo('admin_email'), 'IPN Error', $e->getMessage() ); 
  216.       exit(0); 
  217.   } 
  218.  
  219.   if( $verified ) { 
  220.       $payment_id = $_POST['custom']; 
  221.       $purchase_key = $_POST['item_number']; 
  222.       $paypal_amount = $_POST['mc_gross']; 
  223.       $payment_status = $_POST['payment_status']; 
  224.       $currency_code = strtolower( $_POST['mc_currency'] ); 
  225.  
  226.       // retrieve the meta info for this payment 
  227.       $payment_meta = get_post_meta( $payment_id, '_edd_payment_meta', true ); 
  228.       $payment_amount = edd_format_amount( $payment_meta['amount'] ); 
  229.  
  230.       if( $currency_code != strtolower( $edd_options['currency'] ) ) { 
  231.           return; // the currency code is invalid 
  232.       } 
  233.       if( $paypal_amount != $payment_amount ) { 
  234.           return; // the prices don't match 
  235.       } 
  236.       if( $purchase_key != $payment_meta['key'] ) { 
  237.           return; // purchase keys don't match 
  238.       } 
  239.  
  240.       if( isset( $_POST['txn_type'] ) && $_POST['txn_type'] == 'web_accept' ) { 
  241.  
  242.           $status = strtolower( $payment_status ); 
  243.  
  244.           if( $status == 'completed' || edd_is_test_mode()) { 
  245.  
  246.               // set the payment to complete. This also sends the emails 
  247.               edd_update_payment_status( $payment_id, 'publish' ); 
  248.  
  249.           } else if( $status == 'refunded' ) { 
  250.  
  251.               // this refund process doesn't work yet 
  252.               $payment_data = get_post_meta( $payment_id, '_edd_payment_meta', true ); 
  253.               $downloads = maybe_unserialize( $payment_data['downloads'] ); 
  254.  
  255.               if( is_array( $downloads ) ) { 
  256.                   foreach( $downloads as $download ) { 
  257.                       edd_undo_purchase( $download['id'], $payment_id );         
  258.                   } 
  259.               } 
  260.  
  261.               wp_update_post( array( 'ID' => $payment_id, 'post_status' => 'refunded' ) ); 
  262.  
  263.           } 
  264.       } 
  265.  
  266.   } else { 
  267.       wp_mail( get_bloginfo('admin_email'), __( 'Invalid IPN', 'edd' ), $listener->getTextReport() ); 
  268.   } 
  269. } 
  270. add_action( 'edd_verify_paypal_ipn', 'edd_process_paypal_ipn' );
