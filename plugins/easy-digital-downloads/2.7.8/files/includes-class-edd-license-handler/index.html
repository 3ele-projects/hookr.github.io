  209.    * Activate the license key 
  210.    * 
  211.    * @access  public 
  212.    * @return  void 
  213.    */ 
  214.   public function activate_license() { 
  215.  
  216.       if ( ! isset( $_POST['edd_settings'] ) ) { 
  217.           return; 
  218.       } 
  219.  
  220.       if ( ! isset( $_REQUEST[ $this->item_shortname . '_license_key-nonce'] ) || ! wp_verify_nonce( $_REQUEST[ $this->item_shortname . '_license_key-nonce'], $this->item_shortname . '_license_key-nonce' ) ) { 
  221.  
  222.           return; 
  223.  
  224.       } 
  225.  
  226.       if ( ! current_user_can( 'manage_shop_settings' ) ) { 
  227.           return; 
  228.       } 
  229.  
  230.       if ( empty( $_POST['edd_settings'][ $this->item_shortname . '_license_key'] ) ) { 
  231.  
  232.           delete_option( $this->item_shortname . '_license_active' ); 
  233.  
  234.           return; 
  235.  
  236.       } 
  237.  
  238.       foreach ( $_POST as $key => $value ) { 
  239.           if( false !== strpos( $key, 'license_key_deactivate' ) ) { 
  240.               // Don't activate a key when deactivating a different key 
  241.               return; 
  242.           } 
  243.       } 
  244.  
  245.       $details = get_option( $this->item_shortname . '_license_active' ); 
  246.  
  247.       if ( is_object( $details ) && 'valid' === $details->license ) { 
  248.           return; 
  249.       } 
  250.  
  251.       $license = sanitize_text_field( $_POST['edd_settings'][ $this->item_shortname . '_license_key'] ); 
  252.  
  253.       if( empty( $license ) ) { 
  254.           return; 
  255.       } 
  256.  
  257.       // Data to send to the API 
  258.       $api_params = array( 
  259.           'edd_action' => 'activate_license',  
  260.           'license' => $license,  
  261.           'item_name' => urlencode( $this->item_name ),  
  262.           'url' => home_url() 
  263.  ); 
  264.  
  265.       // Call the API 
  266.       $response = wp_remote_post( 
  267.           $this->api_url,  
  268.           array( 
  269.               'timeout' => 15,  
  270.               'sslverify' => false,  
  271.               'body' => $api_params 
  272.  ) 
  273.  ); 
  274.  
  275.       // Make sure there are no errors 
  276.       if ( is_wp_error( $response ) ) { 
  277.           return; 
  278.       } 
  279.  
  280.       // Tell WordPress to look for updates 
  281.       set_site_transient( 'update_plugins', null ); 
  282.  
  283.       // Decode license data 
  284.       $license_data = json_decode( wp_remote_retrieve_body( $response ) ); 
  285.  
  286.       update_option( $this->item_shortname . '_license_active', $license_data ); 
  287.  
  288.   }
