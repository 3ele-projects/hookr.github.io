  282. /** 
  283.  * Converts old sale and file download logs to new logging system 
  284.  *

  286.  * @uses WP_Query 
  287.  * @uses EDD_Logging 
  288.  * @return void

  290. function edd_v131_upgrades() { 
  291.   if ( get_option( 'edd_logs_upgraded' ) )

  293.  
  294.   if ( version_compare( get_option( 'edd_version' ), '1.3', '>=' ) ) 
  295.       return;

  298.

  300.       set_time_limit( 0 ); 
  301.  
  302.   $args = array( 
  303.       'post_type' => 'download',  
  304.       'posts_per_page' => -1,  
  305.       'post_status' => 'publish' 
  306.  ); 
  307.  
  308.   $query = new WP_Query( $args ); 
  309.   $downloads = $query->get_posts(); 
  310.  
  311.   if ( $downloads ) {

  314.           // Convert sale logs 
  315.           $sale_logs = edd_get_download_sales_log( $download->ID, false ); 
  316.  
  317.           if ( $sale_logs ) { 
  318.               foreach ( $sale_logs['sales'] as $sale ) { 
  319.                   $log_data = array( 
  320.                       'post_parent' => $download->ID,  
  321.                       'post_date' => $sale['date'],  
  322.                       'log_type' => 'sale' 
  323.  ); 
  324.  
  325.                   $log_meta = array( 
  326.                       'payment_id'=> $sale['payment_id'] 
  327.  ); 
  328.  
  329.                   $log = $edd_log->insert_log( $log_data, $log_meta ); 
  330.               } 
  331.           } 
  332.  
  333.           // Convert file download logs 
  334.           $file_logs = edd_get_file_download_log( $download->ID, false ); 
  335.  
  336.           if ( $file_logs ) { 
  337.               foreach ( $file_logs['downloads'] as $log ) { 
  338.                   $log_data = array( 
  339.                       'post_parent' => $download->ID,  
  340.                       'post_date' => $log['date'],  
  341.                       'log_type' => 'file_download' 
  342.  
  343.  ); 
  344.  
  345.                   $log_meta = array( 
  346.                       'user_info' => $log['user_info'],  
  347.                       'file_id' => $log['file_id'],  
  348.                       'ip' => $log['ip'] 
  349.  ); 
  350.  
  351.                   $log = $edd_log->insert_log( $log_data, $log_meta ); 
  352.               } 
  353.           } 
  354.       } 
  355.   } 
  356.   add_option( 'edd_logs_upgraded', '1' ); 
  357. }
