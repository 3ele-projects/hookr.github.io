  53.   public function get_data() { 
  54.       global $edd_logs, $wpdb; 
  55.  
  56.       $totals = $this->get_stored_data( 'edd_temp_recount_all_stats' ); 
  57.       $payment_items = $this->get_stored_data( 'edd_temp_payment_items' ); 
  58.       $processed_payments = $this->get_stored_data( 'edd_temp_processed_payments' ); 
  59.       $accepted_statuses = apply_filters( 'edd_recount_accepted_statuses', array( 'publish', 'revoked' ) ); 
  60.  
  61.       if ( false === $totals ) { 
  62.           $totals = array(); 
  63.       } 
  64.  
  65.       if ( false === $payment_items ) { 
  66.           $payment_items = array(); 
  67.       } 
  68.  
  69.       if ( false === $processed_payments ) { 
  70.           $processed_payments = array(); 
  71.       } 
  72.  
  73.       $all_downloads = $this->get_stored_data( 'edd_temp_download_ids' ); 
  74.  
  75.       $args = apply_filters( 'edd_recount_download_stats_args', array( 
  76.           'post_parent__in' => $all_downloads,  
  77.           'post_type' => 'edd_log',  
  78.           'posts_per_page' => $this->per_step,  
  79.           'post_status' => 'publish',  
  80.           'paged' => $this->step,  
  81.           'log_type' => 'sale',  
  82.           'fields' => 'ids',  
  83.  ) ); 
  84.  
  85.       $log_ids = $edd_logs->get_connected_logs( $args, 'sale' ); 
  86.  
  87.       if ( $log_ids ) { 
  88.           $log_ids = implode( ', ', $log_ids ); 
  89.           $payment_ids = $wpdb->get_col( "SELECT meta_value FROM $wpdb->postmeta WHERE meta_key='_edd_log_payment_id' AND post_id IN ($log_ids)" ); 
  90.           unset( $log_ids ); 
  91.  
  92.           $payment_ids = implode( ', ', $payment_ids ); 
  93.           $payments = $wpdb->get_results( "SELECT ID, post_status FROM $wpdb->posts WHERE ID IN (" . $payment_ids . ")" ); 
  94.           unset( $payment_ids ); 
  95.  
  96.           foreach ( $payments as $payment ) { 
  97.  
  98.               // Prevent payments that have all ready been retrieved from a previous sales log from counting again. 
  99.               if ( in_array( $payment->ID, $processed_payments ) ) { 
  100.                   continue; 
  101.               } 
  102.  
  103.               if ( ! in_array( $payment->post_status, $accepted_statuses ) ) { 
  104.                   $processed_payments[] = $payment->ID; 
  105.                   continue; 
  106.               } 
  107.  
  108.               $items = $payment_items[ $payment->ID ]; 
  109.  
  110.               foreach ( $items as $item ) { 
  111.                   $download_id = $item['id']; 
  112.  
  113.                   if ( ! in_array( $download_id, $all_downloads ) ) { 
  114.                       continue; 
  115.                   } 
  116.  
  117.                   if ( ! array_key_exists( $download_id, $totals ) ) { 
  118.                       $totals[ $download_id ] = array( 
  119.                           'sales' => (int) 0,  
  120.                           'earnings' => (float) 0,  
  121.  ); 
  122.                   } 
  123.  
  124.                   $amount = $item['price']; 
  125.                   if ( ! empty( $item['fees'] ) ) { 
  126.                       foreach ( $item['fees'] as $fee ) { 
  127.                           // Only let negative fees affect earnings 
  128.                           if ( $fee['amount'] > 0 ) { 
  129.                               continue; 
  130.                           } 
  131.  
  132.                           $amount += $fee['amount']; 
  133.                       } 
  134.                   } 
  135.  
  136.                   $totals[ $download_id ]['sales']++; 
  137.                   $totals[ $download_id ]['earnings'] += $amount; 
  138.  
  139.               } 
  140.  
  141.               $processed_payments[] = $payment->ID; 
  142.           } 
  143.  
  144.           $this->store_data( 'edd_temp_processed_payments', $processed_payments ); 
  145.           $this->store_data( 'edd_temp_recount_all_stats', $totals ); 
  146.  
  147.           return true; 
  148.       } 
  149.  
  150.       foreach ( $totals as $key => $stats ) { 
  151.           update_post_meta( $key, '_edd_download_sales' , $stats['sales'] ); 
  152.           update_post_meta( $key, '_edd_download_earnings', $stats['earnings'] ); 
  153.       } 
  154.  
  155.       return false; 
  156.  
  157.   }

  167.       $total = $this->get_stored_data( 'edd_recount_all_total', false ); 
  168.  
  169.       if ( false === $total ) {

  259.       $accepted_statuses = apply_filters( 'edd_recount_accepted_statuses', array( 'publish', 'revoked' ) );

  261.

  308.               $payment_ids = implode( ', ', $payment_ids );

  335.   /**
