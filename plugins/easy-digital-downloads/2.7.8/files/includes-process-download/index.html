  26.   if( ! isset( $_GET['download_id'] ) && isset( $_GET['download'] ) ) { 
  27.       $_GET['download_id'] = $_GET['download']; 
  28.   } 
  29.  
  30.   $args = apply_filters( 'edd_process_download_args', array( 
  31.       'download' => ( isset( $_GET['download_id'] ) )  ? (int) $_GET['download_id']                       : '',

  55.  
  56.       // Validate a legacy URL without a token

  86.  
  87.           if( 'pdf' === strtolower( edd_get_file_extension( $requested_file ) ) ) { 
  88.               // Do not ever grab the thumbnail for PDFs. See https://github.com/easydigitaldownloads/easy-digital-downloads/issues/5491

  90.           } 
  91.

  93.

  95.                   $attached_file = wp_get_attachment_image_url( $attachment_id, $thumbnail_size, false ); 
  96.               } else { 
  97.                   $attached_file = wp_get_attachment_url( $attachment_id ); 
  98.               } 
  99.  
  100.           } else { 
  101.  
  102.               if ( $thumbnail_size ) { 
  103.  
  104.                   $attachment_data = wp_get_attachment_image_src( $attachment_id, $thumbnail_size, false ); 
  105.

  107.                       $attached_file = $attachment_data['0']; 
  108.                       $attached_file = str_replace( site_url(), '', $attached_file ); 
  109.                       $attached_file = realpath( ABSPATH . $attached_file ); 
  110.                   } 
  111.  
  112.               } 
  113.  
  114.               if ( empty( $attached_file ) ) { 
  115.                   $attached_file = get_attached_file( $attachment_id, false ); 
  116.               } 
  117.  
  118.               // Confirm the file exists 
  119.               if( ! file_exists( $attached_file ) ) { 
  120.                   $attached_file = false; 
  121.               } 
  122.  
  123.           } 
  124.  
  125.           if( $attached_file ) { 
  126.  
  127.               $requested_file = $attached_file;

  129.           } 
  130.  
  131.       }

  135.  
  136.       if( 'x_sendfile' == $method && ( ! function_exists( 'apache_get_modules' ) || ! in_array( 'mod_xsendfile', apache_get_modules() ) ) ) { 
  137.           // If X-Sendfile is selected but is not supported, fallback to Direct 
  138.           $method = 'direct'; 
  139.       } 
  140.  
  141.       $file_details = parse_url( $requested_file ); 
  142.       $schemes = array( 'http', 'https' ); // Direct URL schemes 
  143.  
  144.       if ( ( ! isset( $file_details['scheme'] ) || ! in_array( $file_details['scheme'], $schemes ) ) && isset( $file_details['path'] ) && file_exists( $requested_file ) ) { 
  145.  
  146.           /** 
  147.            * Download method is seto to Redirect in settings but an absolute path was provided 
  148.            * We need to switch to a direct download in order for the file to download properly 
  149.            */

  151.  
  152.       }

  154.       /** 
  155.        * Allow extensions to run actions prior to recording the file download log entry

  157.        * @since 2.6.14

  160.  
  161.       // Record this file download in the log

  175.       if ( ! edd_is_func_disabled( 'set_time_limit' ) ) {

  179.           set_magic_quotes_runtime(0);

  181.  
  182.       @session_write_close();

  185.       } 
  186.       @ini_set( 'zlib.output_compression', 'Off' ); 
  187.  
  188.       do_action( 'edd_process_download_headers', $requested_file, $args['download'], $args['email'], $args['payment'] );

  190.       nocache_headers(); 
  191.       header("Robots: none"); 
  192.       header("Content-Type: " . $ctype . "");

  195.       header("Content-Transfer-Encoding: binary"); 
  196.  
  197.       // If the file isn't locally hosted, process the redirect 
  198.       if ( filter_var( $requested_file, FILTER_VALIDATE_URL ) && ! edd_is_local_file( $requested_file ) ) { 
  199.           edd_deliver_download( $requested_file, true ); 
  200.           exit; 
  201.       } 
  202.  
  203.       switch( $method ) : 
  204.  
  205.           case 'redirect' : 
  206.  
  207.               // Redirect straight to the file 
  208.               edd_deliver_download( $requested_file, true ); 
  209.               break; 
  210.  
  211.           case 'direct' : 
  212.           default: 
  213.  
  214.               $direct = false; 
  215.               $file_path = $requested_file; 
  216.  
  217.               if ( ( ! isset( $file_details['scheme'] ) || ! in_array( $file_details['scheme'], $schemes ) ) && isset( $file_details['path'] ) && file_exists( $requested_file ) ) { 
  218.  
  219.                   /** This is an absolute path */ 
  220.                   $direct = true; 
  221.                   $file_path = $requested_file; 
  222.  
  223.               } else if( defined( 'UPLOADS' ) && strpos( $requested_file, UPLOADS ) !== false ) { 
  224.  
  225.                   /** 
  226.                    * This is a local file given by URL so we need to figure out the path 
  227.                    * UPLOADS is always relative to ABSPATH 
  228.                    * site_url() is the URL to where WordPress is installed 
  229.                    */ 
  230.                   $file_path = str_replace( site_url(), '', $requested_file ); 
  231.                   $file_path = realpath( ABSPATH . $file_path );

  233.  
  234.               } else if( strpos( $requested_file, content_url() ) !== false ) { 
  235.  
  236.                   /** This is a local file given by URL so we need to figure out the path */ 
  237.                   $file_path = str_replace( content_url(), WP_CONTENT_DIR, $requested_file ); 
  238.                   $file_path = realpath( $file_path ); 
  239.                   $direct = true; 
  240.  
  241.               } else if( strpos( $requested_file, set_url_scheme( content_url(), 'https' ) ) !== false ) { 
  242.  
  243.                   /** This is a local file given by an HTTPS URL so we need to figure out the path */ 
  244.                   $file_path = str_replace( set_url_scheme( content_url(), 'https' ), WP_CONTENT_DIR, $requested_file ); 
  245.                   $file_path = realpath( $file_path ); 
  246.                   $direct = true; 
  247.  
  248.               } 
  249.  
  250.               // Set the file size header 
  251.               header( "Content-Length: " . @filesize( $file_path ) ); 
  252.  
  253.               // Now deliver the file based on the kind of software the server is running / has enabled 
  254.               if ( stristr( getenv( 'SERVER_SOFTWARE' ), 'lighttpd' ) ) { 
  255.  
  256.                   header( "X-LIGHTTPD-send-file: $file_path" ); 
  257.  
  258.               } elseif ( $direct && ( stristr( getenv( 'SERVER_SOFTWARE' ), 'nginx' ) || stristr( getenv( 'SERVER_SOFTWARE' ), 'cherokee' ) ) ) { 
  259.  
  260.                   // We need a path relative to the domain 
  261.                   $file_path = str_ireplace( realpath( $_SERVER['DOCUMENT_ROOT'] ), '', $file_path ); 
  262.                   header( "X-Accel-Redirect: /$file_path" ); 
  263.  
  264.               } 
  265.  
  266.               if( $direct ) { 
  267.  
  268.                   edd_deliver_download( $file_path ); 
  269.  
  270.               } else { 
  271.  
  272.                   // The file supplied does not have a discoverable absolute path 
  273.                   edd_deliver_download( $requested_file, true ); 
  274.  
  275.               } 
  276.  
  277.               break; 
  278.  
  279.       endswitch; 
  280.  
  281.       edd_die(); 
  282.   } else { 
  283.       $error_message = __( 'You do not have permission to download this file', 'easy-digital-downloads' ); 
  284.       wp_die( apply_filters( 'edd_deny_download_message', $error_message, __( 'Purchase Verification Failed', 'easy-digital-downloads' ) ), __( 'Error', 'easy-digital-downloads' ), array( 'response' => 403 ) ); 
  285.   } 
  286.  
  287.   exit;

  291. /** 
  292.  * Deliver the download file 
  293.  * 
  294.  * If enabled, the file is symlinked to better support large file downloads 
  295.  * 
  296.  * @access   public 
  297.  * @param    string    $file 
  298.  * @param    bool      $redirect True if we should perform a header redirect instead of calling edd_readfile_chunked() 
  299.  * @return   void 
  300.  */ 
  301. function edd_deliver_download( $file = '', $redirect = false ) { 
  302.  
  303.   /** 
  304.    * If symlinks are enabled, a link to the file will be created 
  305.    * This symlink is used to hide the true location of the file, even when the file URL is revealed 
  306.    * The symlink is deleted after it is used 
  307.    */ 
  308.   if( edd_symlink_file_downloads() && edd_is_local_file( $file ) ) { 
  309.  
  310.       $file = edd_get_local_path_from_url( $file ); 
  311.  
  312.       // Generate a symbolic link 
  313.       $ext = edd_get_file_extension( $file ); 
  314.       $parts = explode( '.', $file ); 
  315.       $name = basename( $parts[0] ); 
  316.       $md5 = md5( $file ); 
  317.       $file_name = $name . '_' . substr( $md5, 0, -15 ) . '.' . $ext; 
  318.       $path = edd_get_symlink_dir() . '/' . $file_name; 
  319.       $url = edd_get_symlink_url() . '/' . $file_name; 
  320.  
  321.       // Set a transient to ensure this symlink is not deleted before it can be used 
  322.       set_transient( md5( $file_name ), '1', 30 ); 
  323.  
  324.       // Schedule deletion of the symlink 
  325.       if ( ! wp_next_scheduled( 'edd_cleanup_file_symlinks' ) ) { 
  326.           wp_schedule_single_event( current_time( 'timestamp' ) + 60, 'edd_cleanup_file_symlinks' ); 
  327.       } 
  328.  
  329.       // Make sure the symlink doesn't already exist before we create it 
  330.       if( ! file_exists( $path ) ) { 
  331.           $link = @symlink( realpath( $file ), $path ); 
  332.       } else { 
  333.           $link = true; 
  334.       } 
  335.  
  336.       if( $link ) { 
  337.           // Send the browser to the file 
  338.           header( 'Location: ' . $url ); 
  339.       } else { 
  340.           edd_readfile_chunked( $file ); 
  341.       } 
  342.  
  343.   } elseif( $redirect ) { 
  344.  
  345.       header( 'Location: ' . $file ); 
  346.  
  347.   } else { 
  348.  
  349.       // Read the file and deliver it in chunks 
  350.       edd_readfile_chunked( $file ); 
  351.  
  352.   } 
  353.  
  354. }

  356. /** 
  357.  * Determine if the file being requested is hosted locally or not 
  358.  * 
  359.  * @since  2.5.10 
  360.  * @param  string $requested_file The file being requested 
  361.  * @return bool                   If the file is hosted locally or not 
  362.  */ 
  363. function edd_is_local_file( $requested_file ) { 
  364.   $home_url = preg_replace('#^https?://#', '', home_url() ); 
  365.   $requested_file = preg_replace('#^(https?|file)://#', '', $requested_file ); 
  366.  
  367.   $is_local_url = strpos( $requested_file, $home_url ) === 0; 
  368.   $is_local_path = strpos( $requested_file, '/' ) === 0; 
  369.  
  370.   return ( $is_local_url || $is_local_path ); 
  371. }

  373. /** 
  374.  * Given the URL to a file, determine it's local path

  376.  * Used during the symlink process to determine where to make the symlink point to

  380.  * @return string      If found to be locally hosted, the path to the file 
  381.  */ 
  382. function edd_get_local_path_from_url( $url ) { 
  383.  
  384.   $file = $url; 
  385.   $upload_dir = wp_upload_dir(); 
  386.   $upload_url = $upload_dir['baseurl'] . '/edd'; 
  387.  
  388.   if( defined( 'UPLOADS' ) && strpos( $file, UPLOADS ) !== false ) { 
  389.  
  390.       /** 
  391.        * This is a local file given by URL so we need to figure out the path 
  392.        * UPLOADS is always relative to ABSPATH 
  393.        * site_url() is the URL to where WordPress is installed 
  394.        */ 
  395.       $file = str_replace( site_url(), '', $file ); 
  396.  
  397.   } else if( strpos( $file, $upload_url ) !== false ) { 
  398.  
  399.       /** This is a local file given by URL so we need to figure out the path */ 
  400.       $file = str_replace( $upload_url, edd_get_upload_dir(), $file ); 
  401.  
  402.   } else if( strpos( $file, set_url_scheme( $upload_url, 'https' ) ) !== false ) { 
  403.  
  404.       /** This is a local file given by an HTTPS URL so we need to figure out the path */ 
  405.       $file = str_replace( set_url_scheme( $upload_url, 'https' ), edd_get_upload_dir(), $file ); 
  406.  
  407.   } elseif( strpos( $file, content_url() ) !== false ) { 
  408.  
  409.       $file = str_replace( content_url(), WP_CONTENT_DIR, $file ); 
  410.  
  411.   } 
  412.  
  413.   return $file; 
  414.  
  415. }

  726. /** 
  727.  * Reads file in chunks so big downloads are possible without changing PHP.INI 
  728.  * See http://codeigniter.com/wiki/Download_helper_for_large_files/ 
  729.  * 
  730.  * @access   public 
  731.  * @param    string  $file      The file 
  732.  * @param    boolean $retbytes  Return the bytes of file 
  733.  * @return   bool|string        If string, $status || $cnt 
  734.  */ 
  735. function edd_readfile_chunked( $file, $retbytes = true ) { 
  736.  
  737.   $chunksize = 1024 * 1024; 
  738.   $buffer = ''; 
  739.   $cnt = 0; 
  740.   $handle = @fopen( $file, 'r' ); 
  741.  
  742.   if ( $size = @filesize( $file ) ) { 
  743.       header("Content-Length: " . $size ); 
  744.   } 
  745.  
  746.   if ( false === $handle ) { 
  747.       return false; 
  748.   } 
  749.  
  750.   while ( ! @feof( $handle ) ) { 
  751.       $buffer = @fread( $handle, $chunksize ); 
  752.       echo $buffer; 
  753.  
  754.       if ( $retbytes ) { 
  755.           $cnt += strlen( $buffer ); 
  756.       } 
  757.   } 
  758.  
  759.   $status = @fclose( $handle ); 
  760.  
  761.   if ( $retbytes && $status ) { 
  762.       return $cnt; 
  763.   } 
  764.  
  765.   return $status; 
  766. }

  881. function edd_check_file_url_head( $requested_file, $args, $method ) { 
  882.  
  883.   // If this is a file URL (not a path), perform a head request to determine if it's valid 
  884.   if( filter_var( $requested_file, FILTER_VALIDATE_URL ) && ! edd_is_local_file( $requested_file ) ) { 
  885.

  894.  
  895.       } 
  896.  
  897.       if( 404 === wp_remote_retrieve_response_code( $request ) ) { 
  898.

  901.           $title = __( 'File not found', 'easy-digital-downloads' ); 
  902.  
  903.       } 
  904.  
  905.       if( ! $valid ) { 
  906.

  908.           wp_die( $message, $title, array( 'response' => 403 ) );

  910.       }

  912.   }
