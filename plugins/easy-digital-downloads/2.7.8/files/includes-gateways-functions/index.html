  38. /** 
  39.  * Returns a list of all enabled gateways. 
  40.  * 
  41.  * @since 1.0 
  42.  * @param  bool $sort If true, the default gateway will be first 
  43.  * @return array $gateway_list All the available gateways 
  44. */ 
  45. function edd_get_enabled_payment_gateways( $sort = false ) { 
  46.   $gateways = edd_get_payment_gateways(); 
  47.   $enabled = (array) edd_get_option( 'gateways', false ); 
  48.  
  49.   $gateway_list = array(); 
  50.  
  51.   foreach ( $gateways as $key => $gateway ) { 
  52.       if ( isset( $enabled[ $key ] ) && $enabled[ $key ] == 1 ) { 
  53.           $gateway_list[ $key ] = $gateway; 
  54.       } 
  55.   } 
  56.  
  57.   if ( true === $sort ) { 
  58.       // Reorder our gateways so the default is first 
  59.       $default_gateway_id = edd_get_default_gateway(); 
  60.  
  61.       if( edd_is_gateway_active( $default_gateway_id ) ) { 
  62.  
  63.           $default_gateway = array( $default_gateway_id => $gateway_list[ $default_gateway_id ] ); 
  64.           unset( $gateway_list[ $default_gateway_id ] ); 
  65.  
  66.           $gateway_list = array_merge( $default_gateway, $gateway_list ); 
  67.  
  68.       } 
  69.  
  70.   } 
  71.  
  72.   return apply_filters( 'edd_enabled_payment_gateways', $gateway_list ); 
  73. }

  203. function edd_build_straight_to_gateway_data( $download_id = 0, $options = array(), $quantity = 1 ) { 
  204.  
  205.   $price_options = array();

  207.   if( empty( $options ) || ! edd_has_variable_prices( $download_id ) ) { 
  208.       $price = edd_get_download_price( $download_id ); 
  209.   } else { 
  210.  
  211.       if( is_array( $options['price_id'] ) ) { 
  212.           $price_id = $options['price_id'][0]; 
  213.       } else { 
  214.           $price_id = $options['price_id']; 
  215.       } 
  216.  
  217.       $prices = edd_get_variable_prices( $download_id ); 
  218.  
  219.       // Make sure a valid price ID was supplied 
  220.       if( ! isset( $prices[ $price_id ] ) ) { 
  221.           wp_die( __( 'The requested price ID does not exist.', 'easy-digital-downloads' ), __( 'Error', 'easy-digital-downloads' ), array( 'response' => 404 ) ); 
  222.       } 
  223.  
  224.       $price_options = array( 
  225.           'price_id' => $price_id,  
  226.           'amount' => $prices[ $price_id ]['amount'] 
  227.  ); 
  228.       $price = $prices[ $price_id ]['amount']; 
  229.   } 
  230.  
  231.   // Set up Downloads array

  233.       array( 
  234.           'id' => $download_id,  
  235.           'options' => $price_options 
  236.  ) 
  237.  ); 
  238.  
  239.   // Set up Cart Details array

  241.       array( 
  242.           'name' => get_the_title( $download_id ),

  246.               'options' => $price_options 
  247.  ),  
  248.           'tax' => 0,  
  249.           'discount' => 0,  
  250.           'item_price' => $price,  
  251.           'subtotal' => ( $price * $quantity ),

  253.           'quantity' => $quantity,  
  254.  )
