  965.  * Constructs a secure file download url for a specific file. 
  966.  * 
  967.  * @since 1.0 
  968.  * 
  969.  * @param string    $key Payment key. Use edd_get_payment_key() to get key. 
  970.  * @param string    $email Customer email address. Use edd_get_payment_user_email() to get user email. 
  971.  * @param int       $filekey Index of array of files returned by edd_get_download_files() that this download link is for. 
  972.  * @param int       $download_id Optional. ID of download this download link is for. Default is 0. 
  973.  * @param bool|int  $price_id Optional. Price ID when using variable prices. Default is false. 
  974.  * 
  975.  * @return string A secure download URL 
  976.  */ 
  977. function edd_get_download_file_url( $key, $email, $filekey, $download_id = 0, $price_id = false ) { 
  978.  
  979.   $hours = absint( edd_get_option( 'download_link_expiration', 24 ) ); 
  980.  
  981.   if ( ! ( $date = strtotime( '+' . $hours . 'hours', current_time( 'timestamp') ) ) ) { 
  982.       $date = 2147472000; // Highest possible date, January 19, 2038 
  983.   } 
  984.  
  985.   // Leaving in this array and the filter for backwards compatibility now 
  986.   $old_args = array( 
  987.       'download_key' => rawurlencode( $key ),  
  988.       'email' => rawurlencode( $email ),  
  989.       'file' => rawurlencode( $filekey ),  
  990.       'price_id' => (int) $price_id,  
  991.       'download_id' => $download_id,  
  992.       'expire' => rawurlencode( $date ) 
  993.  ); 
  994.  
  995.   $params = apply_filters( 'edd_download_file_url_args', $old_args ); 
  996.   $payment = edd_get_payment_by( 'key', $params['download_key'] ); 
  997.  
  998.   if ( ! $payment ) { 
  999.       return false; 
  1000.   } 
  1001.  
  1002.   $args = array(); 
  1003.  
  1004.   if ( ! empty( $payment->ID ) ) { 
  1005.  
  1006.       // Simply the URL by concatenating required data using a colon as a delimiter. 
  1007.       $args = array( 
  1008.           'eddfile' => rawurlencode( sprintf( '%d:%d:%d:%d', $payment->ID, $params['download_id'], $params['file'], $price_id ) ) 
  1009.  ); 
  1010.  
  1011.       if ( isset( $params['expire'] ) ) { 
  1012.           $args['ttl'] = $params['expire']; 
  1013.       } 
  1014.  
  1015.       // Ensure all custom args registered with extensions through edd_download_file_url_args get added to the URL, but without adding all the old args 
  1016.       $args = array_merge( $args, array_diff_key( $params, $old_args ) ); 
  1017.  
  1018.       $args = apply_filters( 'edd_get_download_file_url_args', $args, $payment->ID, $params ); 
  1019.  
  1020.       $args['file'] = $params['file']; 
  1021.       $args['token'] = edd_get_download_token( add_query_arg( $args, untrailingslashit( site_url() ) ) ); 
  1022.   } 
  1023.  
  1024.   $download_url = add_query_arg( $args, site_url( 'index.php' ) ); 
  1025.  
  1026.   return $download_url; 
  1027. }

  1123. /** 
  1124.  * Generates a token for a given URL. 
  1125.  * 
  1126.  * An 'o' query parameter on a URL can include optional variables to test 
  1127.  * against when verifying a token without passing those variables around in 
  1128.  * the URL. For example, downloads can be limited to the IP that the URL was 
  1129.  * generated for by adding 'o=ip' to the query string. 
  1130.  * 
  1131.  * Or suppose when WordPress requested a URL for automatic updates, the user 
  1132.  * agent could be tested to ensure the URL is only valid for requests from 
  1133.  * that user agent. 
  1134.  * 
  1135.  * @since 2.3 
  1136.  * 
  1137.  * @param string $url The URL to generate a token for. 
  1138.  * @return string The token for the URL. 
  1139.  */ 
  1140. function edd_get_download_token( $url = '' ) { 
  1141.  
  1142.   $args = array(); 
  1143.   $hash = apply_filters( 'edd_get_url_token_algorithm', 'sha256' ); 
  1144.   $secret = apply_filters( 'edd_get_url_token_secret', hash( $hash, wp_salt() ) ); 
  1145.  
  1146.   /** 
  1147.    * Add additional args to the URL for generating the token. 
  1148.    * Allows for restricting access to IP and/or user agent. 
  1149.    */ 
  1150.   $parts = parse_url( $url ); 
  1151.   $options = array(); 
  1152.  
  1153.   if ( isset( $parts['query'] ) ) {
