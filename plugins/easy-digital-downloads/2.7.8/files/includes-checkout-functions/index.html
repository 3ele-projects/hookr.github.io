  407. /** 
  408.  * Given a number and algorithem, determine if we have a valid credit card format 
  409.  * 
  410.  * @since  2.4 
  411.  * @param  integer $number The Credit Card Number to validate 
  412.  * @return bool            If the card number provided matches a specific format of a valid card 
  413.  */ 
  414. function edd_validate_card_number_format( $number = 0 ) { 
  415.  
  416.   $number = trim( $number ); 
  417.   if ( empty( $number ) ) { 
  418.       return false; 
  419.   } 
  420.  
  421.   if ( ! is_numeric( $number ) ) { 
  422.       return false; 
  423.   } 
  424.  
  425.   $is_valid_format = false; 
  426.  
  427.   // First check if it passes with the passed method, Luhn by default 
  428.   $is_valid_format = edd_validate_card_number_format_luhn( $number ); 
  429.  
  430.   // Run additional checks before we start the regexing and looping by type 
  431.   $is_valid_format = apply_filters( 'edd_valiate_card_format_pre_type', $is_valid_format, $number ); 
  432.  
  433.   if ( true === $is_valid_format ) { 
  434.       // We've passed our method check, onto card specific checks 
  435.       $card_type = edd_detect_cc_type( $number ); 
  436.       $is_valid_format = ! empty( $card_type ) ? true : false; 
  437.   } 
  438.  
  439.   return apply_filters( 'edd_cc_is_valid_format', $is_valid_format, $number ); 
  440. }

  442. /** 
  443.  * Validate credit card number based on the luhn algorithm 
  444.  * 
  445.  * @since  2.4 
  446.  * @param string $number 
  447.  * @return bool 
  448.  */ 
  449. function edd_validate_card_number_format_luhn( $number ) { 
  450.  
  451.   // Strip any non-digits (useful for credit card numbers with spaces and hyphens) 
  452.   $number = preg_replace( '/\D/', '', $number ); 
  453.  
  454.   // Set the string length and parity 
  455.   $length = strlen( $number ); 
  456.   $parity = $length % 2; 
  457.  
  458.   // Loop through each digit and do the math 
  459.   $total = 0; 
  460.   for ( $i = 0; $i < $length; $i++ ) { 
  461.       $digit = $number[ $i ]; 
  462.  
  463.       // Multiply alternate digits by two 
  464.       if ( $i % 2 == $parity ) { 
  465.           $digit *= 2; 
  466.  
  467.           // If the sum is two digits, add them together (in effect) 
  468.           if ( $digit > 9 ) { 
  469.               $digit -= 9; 
  470.           } 
  471.       } 
  472.  
  473.       // Total up the digits 
  474.       $total += $digit; 
  475.   } 
  476.  
  477.   // If the total mod 10 equals 0, the number is valid 
  478.   return ( $total % 10 == 0 ) ? true : false; 
  479.  
  480. }
