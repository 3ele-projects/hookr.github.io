  16. /** 
  17.  * Get Discounts 
  18.  * 
  19.  * Retrieves an array of all available discount codes. 
  20.  * 
  21.  * @since 1.0 
  22.  * @param array $args Query arguments 
  23.  * @return mixed array if discounts exist, false otherwise 
  24.  */ 
  25. function edd_get_discounts( $args = array() ) { 
  26.   $defaults = array( 
  27.       'post_type' => 'edd_discount',  
  28.       'posts_per_page' => 30,  
  29.       'paged' => null,  
  30.       'post_status' => array( 'active', 'inactive', 'expired' ) 
  31.  ); 
  32.  
  33.   $args = wp_parse_args( $args, $defaults ); 
  34.  
  35.   $discounts = get_posts( $args ); 
  36.  
  37.   if ( $discounts ) { 
  38.       return $discounts; 
  39.   } 
  40.  
  41.   if( ! $discounts && ! empty( $args['s'] ) ) { 
  42.       // If no discounts are found and we are searching, re-query with a meta key to find discounts by code 
  43.       $args['meta_key'] = '_edd_discount_code'; 
  44.       $args['meta_value'] = $args['s']; 
  45.       $args['meta_compare'] = 'LIKE'; 
  46.       unset( $args['s'] ); 
  47.       $discounts = get_posts( $args ); 
  48.   } 
  49.  
  50.   if( $discounts ) { 
  51.       return $discounts; 
  52.   } 
  53.  
  54.   return false; 
  55. }

  57. /**

  61.  * 
  62.  * @since 1.0 
  63.  * @return bool 
  64.  */ 
  65. function edd_has_active_discounts() { 
  66.   $discounts = edd_get_discounts( 
  67.       array( 
  68.           'post_status' => 'active',  
  69.           'posts_per_page' => 100,  
  70.           'fields' => 'ids' 
  71.  ) 
  72.  ); 
  73.  
  74.   // When there are no discounts found anymore there are no active ones. 
  75.   if ( ! is_array( $discounts ) || array() === $discounts ) { 
  76.       return false; 
  77.   } 
  78.  
  79.   foreach ( $discounts as $discount ) { 
  80.       // If we catch an active one, we can quit and return true. 
  81.       if ( edd_is_discount_active( $discount, false ) ) { 
  82.           return true; 
  83.       } 
  84.   } 
  85.  
  86.   return false; 
  87. }

  146.   if( ! is_string( $field ) ) {

  164.           return false; 
  165.   } 
  166.

  168.       return $discount;

  171.   return false;

  946. /** 
  947.  * Updates discounts that are expired or at max use (that are not already marked as so) as inactive or expired 
  948.  * 
  949.  * @since 2.6 
  950.  * @return void 
  951. */ 
  952. function edd_discount_status_cleanup() { 
  953.   global $wpdb; 
  954.  
  955.   // We only want to get 25 active discounts to check their status per step here 
  956.   $cron_discount_number = apply_filters( 'edd_discount_status_cleanup_count', 25 ); 
  957.   $discount_ids_to_update = array(); 
  958.   $needs_inactive_meta = array(); 
  959.   $needs_expired_meta = array(); 
  960.  
  961.   // start by getting the last 25 that hit their maximum usage 
  962.   $args = array( 
  963.       'suppress_filters' => false,  
  964.       'post_status' => array( 'active' ),  
  965.       'posts_per_page' => $cron_discount_number,  
  966.       'order' => 'ASC',  
  967.       'meta_query' => array( 
  968.           'relation' => 'AND',  
  969.           array( 
  970.               'key' => '_edd_discount_uses',  
  971.               'value' => 'mt1.meta_value',  
  972.               'compare' => '>=',  
  973.               'type' => 'NUMERIC',  
  974.  ),  
  975.           array( 
  976.               'key' => '_edd_discount_max_uses',  
  977.               'value' => array( '', 0 ),  
  978.               'compare' => 'NOT IN',  
  979.  ),  
  980.           array( 
  981.               'key' => '_edd_discount_max_uses',  
  982.               'compare' => 'EXISTS',  
  983.  ),  
  984.  ),  
  985.  ); 
  986.  
  987.   add_filter( 'posts_request', 'edd_filter_discount_code_cleanup' ); 
  988.   $discounts = edd_get_discounts( $args ); 
  989.   remove_filter( 'posts_request', 'edd_filter_discount_code_cleanup' ); 
  990.  
  991.   if ( $discounts ) { 
  992.       foreach ( $discounts as $discount ) { 
  993.  
  994.           $discount_ids_to_update[] = (int) $discount->ID; 
  995.           $needs_inactive_meta[] = (int) $discount->ID; 
  996.  
  997.       } 
  998.   } 
  999.  
  1000.   // Now lets look at the last 25 that hit their expiration without hitting their limit 
  1001.   $args = array( 
  1002.       'post_status' => array( 'active' ),  
  1003.       'posts_per_page' => $cron_discount_number,  
  1004.       'order' => 'ASC',  
  1005.       'meta_query' => array( 
  1006.           'relation' => 'AND',  
  1007.           array( 
  1008.               'key' => '_edd_discount_expiration',  
  1009.               'value' => '',  
  1010.               'compare' => '!=',  
  1011.  ),  
  1012.           array( 
  1013.               'key' => '_edd_discount_expiration',  
  1014.               'value' => date( 'm/d/Y H:i:s', current_time( 'timestamp' ) ),  
  1015.               'compare' => '<',  
  1016.  ),  
  1017.  ),  
  1018.  ); 
  1019.  
  1020.   $discounts = edd_get_discounts( $args ); 
  1021.  
  1022.   if ( $discounts ) { 
  1023.       foreach ( $discounts as $discount ) { 
  1024.  
  1025.           $discount_ids_to_update[] = (int) $discount->ID; 
  1026.           if ( ! in_array( $discount->ID, $needs_inactive_meta ) ) { 
  1027.               $needs_expired_meta[] = (int) $discount->ID; 
  1028.           } 
  1029.  
  1030.       } 
  1031.   } 
  1032.  
  1033.   $discount_ids_to_update = array_unique( $discount_ids_to_update ); 
  1034.   if ( ! empty ( $discount_ids_to_update ) ) { 
  1035.       $discount_ids_string = "'" . implode( "', '", $discount_ids_to_update ) . "'"; 
  1036.       $sql = "UPDATE $wpdb->posts SET post_status = 'inactive' WHERE ID IN ($discount_ids_string)"; 
  1037.       $wpdb->query( $sql ); 
  1038.   } 
  1039.  
  1040.   $needs_inactive_meta = array_unique( $needs_inactive_meta ); 
  1041.   if ( ! empty( $needs_inactive_meta ) ) { 
  1042.       $inactive_ids = "'" . implode( "', '", $needs_inactive_meta ) . "'"; 
  1043.       $sql = "UPDATE $wpdb->postmeta SET meta_value = 'inactive' WHERE meta_key = '_edd_discount_status' AND post_id IN ($inactive_ids)"; 
  1044.       $wpdb->query( $sql ); 
  1045.   } 
  1046.  
  1047.   $needs_expired_meta = array_unique( $needs_expired_meta ); 
  1048.   if ( ! empty( $needs_expired_meta ) ) { 
  1049.       $expired_ids = "'" . implode( "', '", $needs_expired_meta ) . "'"; 
  1050.       $sql = "UPDATE $wpdb->postmeta SET meta_value = 'inactive' WHERE meta_key = '_edd_discount_status' AND post_id IN ($expired_ids)"; 
  1051.       $wpdb->query( $sql ); 
  1052.   } 
  1053.  
  1054. }
