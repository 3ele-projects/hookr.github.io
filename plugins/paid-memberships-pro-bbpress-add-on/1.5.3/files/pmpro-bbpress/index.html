  38. function pmprobbp_check_forum() { 
  39.   //make sure pmpro and bbpress are active 
  40.   if ( !defined('PMPRO_VERSION') || !class_exists('bbPress') ) 
  41.       return; 
  42.    
  43.   global $current_user; 
  44.  
  45.   $forum_id = bbp_get_forum_id(); 
  46.    
  47.   // Is this even a forum page at all? 
  48.   if( ! bbp_is_forum_archive() && ! empty( $forum_id ) && pmpro_bbp_is_forum() ) { 
  49.       // The current user does not have access to this forum, re-direct them away 
  50.       if( ! pmpro_has_membership_access( $forum_id ) ) {

  60. /** 
  61.  * Function to tell if the current forum, topic, or reply is a subpost of the forum_id passed. 
  62.  * If no forum_id is passed, it will return true if it is any forum, topic, or reply. 
  63.  */ 
  64. function pmpro_bbp_is_forum( $forum_id = NULL ) { 
  65.   global $post; 
  66.    
  67.   //avoid notices on 404 pages 
  68.   if(!$post) 
  69.       return false; 
  70.  
  71.   //if bbPress is not active, no forum 
  72.   if(!function_exists('bbp_is_forum')) 
  73.       return false; 
  74.  
  75.   //check bbpress tests 
  76.   if(bbp_is_forum($post->ID)) 
  77.   {         
  78.       if(!empty($forum_id) && $post->ID == $forum_id) 
  79.           return true; 
  80.       elseif(empty($forum_id)) 
  81.           return true; 
  82.       else 
  83.           return false; 
  84.   } 
  85.   elseif(bbp_is_topic($post->ID)) 
  86.   {         
  87.       if(!empty($forum_id) && $post->post_parent == $forum_id) 
  88.           return true; 
  89.       elseif(empty($forum_id)) 
  90.           return true; 
  91.       else 
  92.           return false; 
  93.   } 
  94.   elseif(bbp_is_reply($post->ID)) 
  95.   {         
  96.       if(!empty($forum_id) && in_array($forum_id, $post->ancestors)) 
  97.           return true; 
  98.       elseif(empty($forum_id)) 
  99.           return true; 
  100.       else 
  101.           return false; 
  102.   } 
  103.   else 
  104.       return false; 
  105. }

  122. function pmprobb_pre_get_posts($query) { 
  123.  
  124.   global $wpdb; 
  125.        
  126.   //only filter front end searches 
  127.   if(is_admin() || !$query->is_search || bbp_is_single_topic()) 
  128.       return $query; 
  129.    
  130.   //get all member forums 
  131.   $sqlQuery = "SELECT ID FROM $wpdb->posts WHERE post_type LIKE 'forum'"; 
  132.   $all_forums = $wpdb->get_col($sqlQuery); 
  133.    
  134.   //no forums? 
  135.   if(empty($all_forums)) 
  136.       return $query; 
  137.    
  138.   //add restricted forums to array 
  139.   $restricted_forum_ids = array(); 
  140.   foreach($all_forums as $forum_id) { 
  141.       if(!pmpro_has_membership_access($forum_id)) 
  142.           $restricted_forum_ids[] = $forum_id; 
  143.   } 
  144.  
  145.   //if there are restricted forums, find topics and exclude them all from searches 
  146.   if(!empty($restricted_forum_ids)) 
  147.   {     
  148.       //get topics belonging to restricted forums 
  149.       $sqlQuery = "SELECT post_id FROM $wpdb->postmeta WHERE meta_key LIKE '_bbp_forum_id' AND meta_value IN(" . implode(', ', $restricted_forum_ids) . ")"; 
  150.       $restricted_topic_ids = $wpdb->get_col($sqlQuery); 
  151.  
  152.       //exclude restricted topics and posts 
  153.       $query->set('post__not_in', array_merge($query->get('post__not_in'), $restricted_topic_ids, $restricted_forum_ids));         
  154.   } 
  155.    
  156.   return $query; 
  157. }
