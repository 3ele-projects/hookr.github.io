  562.         /** 
  563.          * Lock uptransactionstatus from database and updateorderstatus 
  564.          * called by 'valid_mpay24_standard_request' action 
  565.          * 
  566.          * @param array $posted - request parameters 
  567.          * @access public 
  568.          * @return void 
  569.          */ 
  570.         public function successful_request( $posted ) { 
  571.             global $wpdb; 
  572.  
  573.             if ( ! empty( $posted['TID'] ) ) { 
  574.  
  575.                 $order = new WC_Order( (int) $posted['TID'] ); 
  576.                 $table_name = $wpdb->prefix . GATEWAY_MPAY24_TABLE_NAME; 
  577.                 $transaction_db = $wpdb->get_row( $wpdb->prepare( "SELECT * FROM $table_name WHERE tid = %s", $posted['TID'] ) ); 
  578.  
  579.                 //mpay24transactionstatus: BILLED, RESERVED, ERROR, SUSPENDED, CREDITED, REVERSED 
  580.                 // Lowercase 
  581.                 $tstatus = strtolower( $transaction_db->tstatus ); 
  582.  
  583.                 if ( 'yes' == $this->debug ) { 
  584.                     $this->log->add( 'mpay24', 'Transaction status: ' . $tstatus ); 
  585.                 } 
  586.  
  587.                 // We are here so lets check status and do actions 
  588.                 switch ( $tstatus ) { 
  589.                     case 'billed': 
  590.                         // Checkordernot already completed 
  591.                         if ( 'completed' == $order->status ) { 
  592.                             if ( 'yes' == $this->debug ) { 
  593.                                 $this->log->add( 'mpay24', 'Aborting,Order#' . $order->id . ' is already complete.' ); 
  594.                             } 
  595.                             exit; 
  596.                         } 
  597.  
  598.                         // Payment completed 
  599.                         $order->add_order_note( __( 'mPAY24 payment completed', 'wc-mpay24' ) ); 
  600.                         $order->payment_complete(); 
  601.  
  602.                         if ( 'yes' == $this->debug ) { 
  603.                             $this->log->add( 'mpay24', 'Payment complete.' ); 
  604.                         } 
  605.                     break; 
  606.                     case 'suspended': 
  607.                     case 'reserved': 
  608.                         //Orderpending 
  609.                         $order->update_status( 'on-hold', sprintf( __( 'Payment %s via mPAY24.', 'wc-mpay24' ), $tstatus ) ); 
  610.                     break; 
  611.                     case error 
  612.                         //Orderfailed 
  613.                         $order->update_status( 'failed', sprintf( __( 'Payment %s via mPAY24.', 'wc-mpay24' ), $tstatus ) ); 
  614.                         if( version_compare( $woocommerce->version, '2.1', '>=' ) ) { 
  615.                             wc_add_notice( __( 'Payment failed via mPAY24.', 'wc-mpay24' ), error ); 
  616.                         } 
  617.                     break; 
  618.                     case 'credited': 
  619.                         $order->update_status( 'refunded', sprintf(__('Payment %s via mPAY24.', 'wc-mpay24' ), $tstatus ) ); 
  620.                     break; 
  621.                     case 'reversed': 
  622.                         //Ordercancelled 
  623.                         $order->cancel_order( sprintf( __( 'Payment %s via mPAY24.', 'wc-mpay24' ), $tstatus ) ); 
  624.                     break; 
  625.  
  626.                     default: 
  627.                     // No action 
  628.                     break; 
  629.                 } 
  630.                 exit; 
  631.             } 
  632.         } 
  633.     }
