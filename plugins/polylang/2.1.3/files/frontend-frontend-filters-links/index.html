  177.           }

  179.           else {

  187.   /** 
  188.    * Outputs references to translated pages ( if exists ) in the html head section 
  189.    * 
  190.    * @since 0.1 
  191.    */ 
  192.   public function wp_head() { 
  193.       // Don't output anything on paged archives: see https://wordpress.org/support/topic/hreflang-on-page2 
  194.       // Don't output anything on paged pages and paged posts 
  195.       if ( is_paged() || ( is_singular() && ( $page = get_query_var( 'page' ) ) && $page > 1 ) ) { 
  196.           return; 
  197.       } 
  198.  
  199.       // Google recommends to include self link https://support.google.com/webmasters/answer/189077?hl=en 
  200.       foreach ( $this->model->get_languages_list() as $language ) { 
  201.           if ( $url = $this->links->get_translation_url( $language ) ) { 
  202.               $urls[ $language->get_locale( 'display' ) ] = $url; 
  203.           } 
  204.       } 
  205.  
  206.       // Ouptputs the section only if there are translations ( $urls always contains self link ) 
  207.       if ( ! empty( $urls ) && count( $urls ) > 1 ) { 
  208.  
  209.           // Prepare the list of languages to remove the country code 
  210.           foreach ( array_keys( $urls ) as $locale ) { 
  211.               $split = explode( '-', $locale ); 
  212.               $languages[ $locale ] = reset( $split ); 
  213.           } 
  214.  
  215.           $count = array_count_values( $languages ); 
  216.  
  217.           foreach ( $urls as $locale => $url ) { 
  218.               $lang = $count[ $languages[ $locale ] ] > 1 ? $locale : $languages[ $locale ]; // Output the country code only when necessary 
  219.               $hreflangs[ $lang ] = $url; 
  220.           } 
  221.  
  222.           // Adds the site root url when the default language code is not hidden 
  223.           // See https://wordpress.org/support/topic/implementation-of-hreflangx-default 
  224.           if ( is_front_page() && ! $this->options['hide_default'] && $this->options['force_lang'] < 3 ) { 
  225.               $hreflangs['x-default'] = home_url( '/' ); 
  226.           } 
  227.  
  228.           /** 
  229.            * Filters the list of rel hreflang attributes 
  230.            * 
  231.            * @since 2.1 
  232.            * 
  233.            * @param array $hreflangs Array of urls with language codes as keys 
  234.            */ 
  235.           $hreflangs = apply_filters( 'pll_rel_hreflang_attributes', $hreflangs ); 
  236.  
  237.           foreach ( $hreflangs as $lang => $url ) { 
  238.               printf( '<link rel="alternate" href="%s" hreflang="%s" />'."\n", esc_url( $url ), esc_attr( $lang ) ); 
  239.           } 
  240.       } 
  241.   }

  243.   /** 
  244.    * Filters the home url to get the right language 
  245.    * 
  246.    * @since 0.4 
  247.    * 
  248.    * @param string $url 
  249.    * @param string $path 
  250.    * @return string 
  251.    */ 
  252.   public function home_url( $url, $path ) { 
  253.       if ( ! ( did_action( 'template_redirect' ) || did_action( 'login_init' ) ) || rtrim( $url, '/' ) != $this->links_model->home ) { 
  254.           return $url; 
  255.       } 
  256.  
  257.       static $white_list, $black_list; // Avoid evaluating this at each function call 
  258.  
  259.       // We *want* to filter the home url in these cases 
  260.       if ( empty( $white_list ) ) { 
  261.           // On Windows get_theme_root() mixes / and \ 
  262.           // We want only \ for the comparison with debug_backtrace 
  263.           $theme_root = get_theme_root(); 
  264.           $theme_root = ( false === strpos( $theme_root, '\\' ) ) ? $theme_root : str_replace( '/', '\\', $theme_root ); 
  265.  
  266.           /** 
  267.            * Filter the white list of the Polylang 'home_url' filter 
  268.            * The $args contains an array of arrays each of them having 
  269.            * a 'file' key and/or a 'function' key to decide which functions in 
  270.            * which files using home_url() calls must be filtered 
  271.            * 
  272.            * @since 1.1.2 
  273.            * 
  274.            * @param array $args 
  275.            */ 
  276.           $white_list = apply_filters( 'pll_home_url_white_list', array( 
  277.               array( 'file' => $theme_root ),  
  278.               array( 'function' => 'wp_nav_menu' ),  
  279.               array( 'function' => 'login_footer' ),  
  280.               array( 'function' => 'get_custom_logo' ),  
  281.  ) ); 
  282.       } 
  283.  
  284.       // We don't want to filter the home url in these cases 
  285.       if ( empty( $black_list ) ) { 
  286.  
  287.           /** 
  288.            * Filter the black list of the Polylang 'home_url' filter 
  289.            * The $args contains an array of arrays each of them having 
  290.            * a 'file' key and/or a 'function' key to decide which functions in 
  291.            * which files using home_url() calls must be filtered 
  292.            * 
  293.            * @since 1.1.2 
  294.            * 
  295.            * @param array $args 
  296.            */ 
  297.           $black_list = apply_filters( 'pll_home_url_black_list', array( 
  298.               array( 'file' => 'searchform.php' ), // Since WP 3.6 searchform.php is passed through get_search_form 
  299.               array( 'function' => 'get_search_form' ),  
  300.  ) ); 
  301.       } 
  302.  
  303.       $traces = version_compare( PHP_VERSION, '5.2.5', '>=' ) ? debug_backtrace( false ) : debug_backtrace(); 
  304.       unset( $traces[0], $traces[1] ); // We don't need the last 2 calls: this function + call_user_func_array (or apply_filters on PHP7+) 
  305.  
  306.       foreach ( $traces as $trace ) { 
  307.           // Black list first 
  308.           foreach ( $black_list as $v ) { 
  309.               if ( ( isset( $trace['file'], $v['file'] ) && false !== strpos( $trace['file'], $v['file'] ) ) || ( isset( $trace['function'], $v['function'] ) && $trace['function'] == $v['function'] ) ) { 
  310.                   return $url; 
  311.               } 
  312.           } 
  313.  
  314.           foreach ( $white_list as $v ) { 
  315.               if ( ( isset( $trace['function'], $v['function'] ) && $trace['function'] == $v['function'] ) || 
  316.                   ( isset( $trace['file'], $v['file'] ) && false !== strpos( $trace['file'], $v['file'] ) && in_array( $trace['function'], array( 'home_url', 'get_home_url', 'bloginfo', 'get_bloginfo' ) ) ) ) { 
  317.                   $ok = true; 
  318.               } 
  319.           } 
  320.       } 
  321.  
  322.       return empty( $ok ) ? $url : ( empty( $path ) ? rtrim( $this->links->get_home_url( $this->curlang ), '/' ) : $this->links->get_home_url( $this->curlang ) ); 
  323.   }

  338.   /** 
  339.    * If the language code is not in agreement with the language of the content 
  340.    * redirects incoming links to the proper URL to avoid duplicate content 
  341.    * 
  342.    * @since 0.9.6 
  343.    * 
  344.    * @param string $requested_url optional 
  345.    * @param bool   $do_redirect   optional, whether to perform the redirection or not 
  346.    * @return string if redirect is not performed 
  347.    */ 
  348.   public function check_canonical_url( $requested_url = '', $do_redirect = true ) { 
  349.       global $wp_query, $post, $is_IIS; 
  350.  
  351.       // Don't redirect in same cases as WP 
  352.       if ( is_trackback() || is_search() || is_admin() || is_preview() || is_robots() || ( $is_IIS && ! iis7_supports_permalinks() ) ) { 
  353.           return; 
  354.       } 
  355.  
  356.       // Don't redirect mysite.com/?attachment_id= to mysite.com/en/?attachment_id= 
  357.       if ( 1 == $this->options['force_lang'] && is_attachment() && isset( $_GET['attachment_id'] ) ) { 
  358.           return; 
  359.       } 
  360.  
  361.       // If the default language code is not hidden and the static front page url contains the page name 
  362.       // the customizer lands here and the code below would redirect to the list of posts 
  363.       if ( is_customize_preview() ) { 
  364.           return; 
  365.       } 
  366.  
  367.       if ( empty( $requested_url ) ) { 
  368.           $requested_url = ( is_ssl() ? 'https://' : 'http://' ) . $_SERVER['HTTP_HOST'] . $_SERVER['REQUEST_URI']; 
  369.       } 
  370.  
  371.       if ( is_single() || is_page() ) { 
  372.           if ( isset( $post->ID ) && $this->model->is_translated_post_type( $post->post_type ) ) { 
  373.               $language = $this->model->post->get_language( (int) $post->ID ); 
  374.           } 
  375.       } 
  376.  
  377.       elseif ( is_category() || is_tag() || is_tax() ) { 
  378.           $obj = $wp_query->get_queried_object(); 
  379.           if ( $this->model->is_translated_taxonomy( $obj->taxonomy ) ) { 
  380.               $language = $this->model->term->get_language( (int) $obj->term_id ); 
  381.           } 
  382.       } 
  383.  
  384.       elseif ( $wp_query->is_posts_page ) { 
  385.           $obj = $wp_query->get_queried_object(); 
  386.           $language = $this->model->post->get_language( (int) $obj->ID ); 
  387.       } 
  388.  
  389.       elseif ( is_404() && ! empty( $wp_query->query['page_id'] ) && $id = get_query_var( 'page_id' ) ) { 
  390.           // Special case for page shortlinks when using subdomains or multiple domains 
  391.           // Needed because redirect_canonical doesn't accept to change the domain name 
  392.           $language = $this->model->post->get_language( (int) $id ); 
  393.       } 
  394.  
  395.       if ( empty( $language ) ) { 
  396.           $language = $this->curlang; 
  397.           $redirect_url = $requested_url; 
  398.       } else { 
  399.           // First get the canonical url evaluated by WP 
  400.           // Workaround a WP bug wich removes the port for some urls and get it back at second call to redirect_canonical 
  401.           $_redirect_url = ( ! $_redirect_url = redirect_canonical( $requested_url, false ) ) ? $requested_url : $_redirect_url; 
  402.           $redirect_url = ( ! $redirect_url = redirect_canonical( $_redirect_url, false ) ) ? $_redirect_url : $redirect_url; 
  403.  
  404.           // Then get the right language code in url 
  405.           $redirect_url = $this->options['force_lang'] ? 
  406.               $this->links_model->switch_language_in_link( $redirect_url, $language ) : 
  407.               $this->links_model->remove_language_from_link( $redirect_url ); // Works only for default permalinks 
  408.       } 
  409.  
  410.       /** 
  411.        * Filters the canonical url detected by Polylang 
  412.        * 
  413.        * @since 1.6 
  414.        * 
  415.        * @param bool|string $redirect_url false or the url to redirect to 
  416.        * @param object      $language the language detected 
  417.        */ 
  418.       $redirect_url = apply_filters( 'pll_check_canonical_url', $redirect_url, $language ); 
  419.  
  420.       // The language is not correctly set so let's redirect to the correct url for this object 
  421.       if ( $do_redirect && $redirect_url && $requested_url != $redirect_url ) { 
  422.           wp_redirect( $redirect_url, 301 ); 
  423.           exit; 
  424.       } 
  425.  
  426.       return $redirect_url; 
  427.   } 
  428. }
