  107.     public function get_preferred_browser_language() { 
  108.         $accept_langs = array(); 
  109.  
  110.         if ( isset( $_SERVER['HTTP_ACCEPT_LANGUAGE'] ) ) { 
  111.             // break up string into pieces ( languages and q factors ) 
  112.             preg_match_all( '/([a-z]{1, 8}(-[a-z]{1, 8})?)\s*(;\s*q\s*=\s*( 1|0\.[0-9]+))?/i', $_SERVER['HTTP_ACCEPT_LANGUAGE'], $lang_parse ); 
  113.  
  114.             $k = $lang_parse[1]; 
  115.             $v = $lang_parse[4]; 
  116.  
  117.             if ( $n = count( $k ) ) { 
  118.                 // set default to 1 for any without q factor 
  119.                 foreach ( $v as $key => $val ) { 
  120.                     if ( '' === $val ) { 
  121.                         $v[ $key ] = 1; 
  122.                     } 
  123.                 } 
  124.  
  125.                 // bubble sort ( need a stable sort for Android, so can't use a PHP sort function ) 
  126.                 if ( $n > 1 ) { 
  127.                     for ( $i = 2; $i <= $n; $i++ ) { 
  128.                         for ( $j = 0; $j <= $n - 2; $j++ ) { 
  129.                             if ( $v[ $j ] < $v[ $j + 1 ] ) { 
  130.                                 // swap values 
  131.                                 $temp = $v[ $j ]; 
  132.                                 $v[ $j ] = $v[ $j + 1 ]; 
  133.                                 $v[ $j + 1 ] = $temp; 
  134.                                 //swap keys 
  135.                                 $temp = $k[ $j ]; 
  136.                                 $k[ $j ] = $k[ $j + 1 ]; 
  137.                                 $k[ $j + 1 ] = $temp; 
  138.                             } 
  139.                         } 
  140.                     } 
  141.                 } 
  142.                 $accept_langs = array_combine( $k, $v ); 
  143.             } 
  144.         } 
  145.  
  146.         $languages = $this->model->get_languages_list( array( 'hide_empty' => true ) ); // hides languages with no post 
  147.  
  148.         /** 
  149.          * Filter the list of languages to use to match the browser preferences 
  150.          * 
  151.          * @since 1.9.3 
  152.          * 
  153.          * @param array $languages array of PLL_Language objects 
  154.          */ 
  155.         $languages = apply_filters( 'pll_languages_for_browser_preferences', $languages ); 
  156.  
  157.         // looks through sorted list and use first one that matches our language list 
  158.         foreach ( array_keys( $accept_langs ) as $accept_lang ) { 
  159.             // first loop to match the exact locale 
  160.             foreach ( $languages as $language ) { 
  161.                 if ( 0 === strcasecmp( $accept_lang, $language->get_locale( 'display' ) ) ) { 
  162.                     return $language->slug; 
  163.                 } 
  164.             } 
  165.  
  166.             // second loop to match the language set 
  167.             foreach ( $languages as $language ) { 
  168.                 if ( 0 === stripos( $accept_lang, $language->slug ) || 0 === stripos( $language->get_locale( 'display' ), $accept_lang ) ) { 
  169.                     return $language->slug; 
  170.                 } 
  171.             } 
  172.         } 
  173.         return false;
