  124.     public function copy_taxonomies( $from, $to, $lang, $sync = false ) { 
  125.         // Get taxonomies to sync for this post type 
  126.         $taxonomies = array_intersect( get_post_taxonomies( $from ), $this->get_taxonomies_to_copy( $sync, $from, $to, $lang ) ); 
  127.  
  128.         // Update the term cache to reduce the number of queries in the loop 
  129.         update_object_term_cache( $sync ? array( $from, $to ) : $from, get_post_type( $from ) ); 
  130.  
  131.         // Copy or synchronize terms 
  132.         // FIXME quite a lot of query in foreach 
  133.         foreach ( $taxonomies as $tax ) { 
  134.             $terms = get_the_terms( $from, $tax ); 
  135.  
  136.             // Translated taxonomy 
  137.             if ( $this->model->is_translated_taxonomy( $tax ) ) { 
  138.                 $newterms = array(); 
  139.                 if ( is_array( $terms ) ) { 
  140.                     foreach ( $terms as $term ) { 
  141.                         if ( $term_id = $this->model->term->get_translation( $term->term_id, $lang ) ) { 
  142.                             $newterms[] = (int) $term_id; // Cast is important otherwise we get 'numeric' tags 
  143.                         } 
  144.                     } 
  145.                 } 
  146.  
  147.                 // For some reasons, the user may have untranslated terms in the translation. don't forget them. 
  148.                 if ( $sync ) { 
  149.                     $tr_terms = get_the_terms( $to, $tax ); 
  150.                     if ( is_array( $tr_terms ) ) { 
  151.                         foreach ( $tr_terms as $term ) { 
  152.                             if ( ! $this->model->term->get_translation( $term->term_id, $this->model->post->get_language( $from ) ) ) { 
  153.                                 $newterms[] = (int) $term->term_id; 
  154.                             } 
  155.                         } 
  156.                     } 
  157.                 } 
  158.  
  159.                 if ( ! empty( $newterms ) || $sync ) { 
  160.                     wp_set_object_terms( $to, $newterms, $tax ); // replace terms in translation 
  161.                 } 
  162.             } 
  163.  
  164.             // Untranslated taxonomy ( post format ) 
  165.             // Don't use simple get_post_format / set_post_format to generalize the case to other taxonomies 
  166.             else { 
  167.                 wp_set_object_terms( $to, is_array( $terms ) ? array_map( 'intval', wp_list_pluck( $terms, 'term_id' ) ) : null, $tax ); 
  168.             }

  184.         $metas = get_post_custom( $from ); 
  185.         $keys = array(); 
  186.  
  187.         // Get public meta keys ( including from translated post in case we just deleted a custom field ) 
  188.         if ( ! $sync || in_array( 'post_meta', $this->options['sync'] ) ) { 
  189.             foreach ( $keys = array_unique( array_merge( array_keys( $metas ), array_keys( get_post_custom( $to ) ) ) ) as $k => $meta_key ) { 
  190.                 if ( is_protected_meta( $meta_key ) ) { 
  191.                     unset( $keys[ $k ] ); 
  192.                 } 
  193.             } 
  194.         } 
  195.  
  196.         // Add page template and featured image 
  197.         foreach ( array( '_wp_page_template', '_thumbnail_id' ) as $meta ) { 
  198.             if ( ! $sync || in_array( $meta, $this->options['sync'] ) ) { 
  199.                 $keys[] = $meta; 
  200.             } 
  201.         } 
  202.  
  203.         /** 
  204.          * Filter the custom fields to copy or synchronize 
  205.          * 
  206.          * @since 0.6

  213.          * @param string $lang language slug

  223.                     // Thanks to goncalveshugo http://wordpress.org/support/topic/plugin-polylang-pll_copy_post_meta

  227.                 } 
  228.             } 
  229.         }

  241.     public function pll_save_post( $post_id, $post, $translations ) { 
  242.         global $wpdb; 
  243.  
  244.         // Prepare properties to synchronize 
  245.         foreach ( array( 'comment_status', 'ping_status', 'menu_order' ) as $property ) { 
  246.             if ( in_array( $property, $this->options['sync'] ) ) { 
  247.                 $postarr[ $property ] = $post->$property; 
  248.             } 
  249.         } 
  250.  
  251.         if ( in_array( 'post_date', $this->options['sync'] ) ) { 
  252.             // For new drafts, save the date now otherwise it is overriden by WP. Thanks to JoryHogeveen. See #32. 
  253.             if ( 'post-new.php' === $GLOBALS['pagenow'] && isset( $_GET['from_post'], $_GET['new_lang'] ) ) { 
  254.                 $original = get_post( (int) $_GET['from_post'] ); 
  255.                 $wpdb->update( 
  256.                     $wpdb->posts, array( 
  257.                         'post_date' => $original->post_date,  
  258.                         'post_date_gmt' => $original->post_date_gmt,  
  259.  ),  
  260.                     array( 'ID' => $post_id ) 
  261.  ); 
  262.             } else { 
  263.                 $postarr['post_date'] = $post->post_date; 
  264.                 $postarr['post_date_gmt'] = $post->post_date_gmt; 
  265.             } 
  266.         } 
  267.  
  268.         // Synchronize terms and metas in translations 
  269.         foreach ( $translations as $lang => $tr_id ) { 
  270.             if ( ! $tr_id || $tr_id === $post_id ) { 
  271.                 continue; 
  272.             } 
  273.  
  274.             // Synchronize terms and metas 
  275.             $this->copy_taxonomies( $post_id, $tr_id, $lang, true ); 
  276.             $this->copy_post_metas( $post_id, $tr_id, $lang, true ); 
  277.  
  278.             // Sticky posts 
  279.             if ( in_array( 'sticky_posts', $this->options['sync'] ) ) { 
  280.                 isset( $_REQUEST['sticky'] ) && 'sticky' === $_REQUEST['sticky'] ? stick_post( $tr_id ) : unstick_post( $tr_id ); 
  281.             } 
  282.  
  283.             // Add comment status, ping status, menu order... to synchronization 
  284.             $tr_arr = empty( $postarr ) ? array() : $postarr; 
  285.  
  286.             if ( isset( $GLOBALS['post_type'] ) ) { 
  287.                 $post_type =  $GLOBALS['post_type']; 
  288.             } elseif ( isset( $_REQUEST['post_type'] ) ) { 
  289.                 $post_type = $_REQUEST['post_type']; // 2nd case for quick edit 
  290.             } 
  291.  
  292.             // Add post parent to synchronization 
  293.             // Make sure not to impact media translations when creating them at the same time as post 
  294.             // Do not udpate the translation parent if the user set a parent with no translation 
  295.             if ( in_array( 'post_parent', $this->options['sync'] ) && isset( $post_type ) && $post_type === $post->post_type ) { 
  296.                 $post_parent = ( $parent_id = wp_get_post_parent_id( $post_id ) ) ? $this->model->post->get_translation( $parent_id, $lang ) : 0; 
  297.                 if ( ! ( $parent_id && ! $post_parent ) ) { 
  298.                     $tr_arr['post_parent'] = $post_parent; 
  299.                 } 
  300.             } 
  301.  
  302.             // Update all the row at once 
  303.             // Don't use wp_update_post to avoid infinite loop 
  304.             if ( ! empty( $tr_arr ) ) { 
  305.                 $wpdb->update( $wpdb->posts, $tr_arr, array( 'ID' => $tr_id ) ); 
  306.                 clean_post_cache( $tr_id ); 
  307.             } 
  308.         }

  324.         } 
  325.  
  326.         // Get all posts associated to this term 
  327.         $posts = get_posts( array( 
  328.             'numberposts' => -1,  
  329.             'nopaging' => true,  
  330.             'post_type' => 'any',  
  331.             'post_status' => 'any',  
  332.             'fields' => 'ids',  
  333.             'tax_query' => array( array( 
  334.                 'taxonomy' => $taxonomy,

  336.                 'terms' => array_merge( array( $term_id ), array_values( $translations ) ),  
  337.                 'include_children' => false,  
  338.  ) ),  
  339.  ) );

  342.         // FIXME quite a lot of query in foreach 
  343.         foreach ( $this->model->get_languages_list() as $language ) {
