  261.         // Validate locale with the same pattern as WP 4.3. See #28303

  269.         } 
  270.  
  271.         // Validate slug is unique 
  272.         if ( $this->get_language( $args['slug'] ) && ( null === $lang || ( isset( $lang ) && $lang->slug != $args['slug'] ) ) ) { 
  273.             add_settings_error( 'general', 'pll_non_unique_slug', __( 'The language code must be unique', 'polylang' ) ); 
  274.         } 
  275.  
  276.         // Validate name
