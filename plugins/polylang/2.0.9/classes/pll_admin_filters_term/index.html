  560.     protected function get_queried_language( $taxonomies, $args ) {

  562.         // Since WP 4.7, make sure not to filter wp_get_object_terms()

  590.     public function get_terms_args( $args, $taxonomies ) {

  678.     /** 
  679.      * Updates the translations term ids when splitting a shared term 
  680.      * Splits translations if these are shared terms too 
  681.      * 
  682.      * @since 1.7 
  683.      * 
  684.      * @param int $term_id shared term_id 
  685.      * @param int $new_term_id 
  686.      * @param int $term_taxonomy_id 
  687.      * @param string $taxonomy 
  688.      */ 
  689.     public function split_shared_term( $term_id, $new_term_id, $term_taxonomy_id, $taxonomy ) { 
  690.         if ( ! $this->model->is_translated_taxonomy( $taxonomy ) ) { 
  691.             return; 
  692.         } 
  693.  
  694.         // Avoid recursion 
  695.         static $avoid_recursion = false; 
  696.         if ( $avoid_recursion ) { 
  697.             return; 
  698.         } 
  699.  
  700.         $avoid_recursion = true; 
  701.         $lang = $this->model->term->get_language( $term_id ); 
  702.  
  703.         foreach ( $this->model->term->get_translations( $term_id ) as $key => $tr_id ) { 
  704.             if ( $lang->slug == $key ) { 
  705.                 $translations[ $key ] = $new_term_id; 
  706.             } 
  707.             else { 
  708.                 $tr_term = get_term( $tr_id, $taxonomy ); 
  709.                 $translations[ $key ] = _split_shared_term( $tr_id, $tr_term->term_taxonomy_id ); 
  710.  
  711.                 // Hack translation ids sent by the form to avoid overwrite in PLL_Admin_Filters_Term::save_translations 
  712.                 if ( isset( $_POST['term_tr_lang'][ $key ] ) && $_POST['term_tr_lang'][ $key ] == $tr_id ) { 
  713.                     $_POST['term_tr_lang'][ $key ] = $translations[ $key ]; 
  714.                 } 
  715.             } 
  716.             $this->model->term->set_language( $translations[ $key ], $key ); 
  717.         } 
  718.  
  719.         $this->model->term->save_translations( $new_term_id, $translations ); 
  720.         $avoid_recursion = false; 
  721.     }
