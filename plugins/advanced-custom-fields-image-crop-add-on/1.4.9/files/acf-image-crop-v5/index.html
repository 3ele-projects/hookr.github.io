  385.   * Parses the field value into a consistent data object

  387.   function get_image_data($field) { 
  388.       $imageData = new stdClass(); 
  389.       $imageData->original_image = ''; 
  390.       $imageData->original_image_width = ''; 
  391.       $imageData->original_image_height = ''; 
  392.       $imageData->cropped_image = ''; 
  393.       $imageData->original_image_url = ''; 
  394.       $imageData->preview_image_url = ''; 
  395.       $imageData->image_url = ''; 
  396.  
  397.       if($field['value'] == '') { 
  398.           // Field has not yet been saved or is an empty image field 
  399.           return $imageData; 
  400.       } 
  401.  
  402.       $data = json_decode($field['value']); 
  403.  
  404.       if(! is_object($data)) { 
  405.           // Field was saved as a regular image field 
  406.           $imageAtts = wp_get_attachment_image_src($field['value'], 'full'); 
  407.           $imageData->original_image = $field['value']; 
  408.           $imageData->original_image_width = $imageAtts[1]; 
  409.           $imageData->original_image_height = $imageAtts[2]; 
  410.           $imageData->preview_image_url = $this->get_image_src($field['value'], $field['preview_size']); 
  411.           $imageData->image_url = $this->get_image_src($field['value'], 'full'); 
  412.           $imageData->original_image_url = $imageData->image_url; 
  413.           return $imageData; 
  414.       } 
  415.  
  416.       if( !is_numeric($data->original_image) ) 
  417.       { 
  418.           // The field has been saved, but has no image 
  419.           return $imageData; 
  420.       } 
  421.  
  422.       // By now, we have at least a saved original image 
  423.       $imageAtts = wp_get_attachment_image_src($data->original_image, 'full'); 
  424.       $imageData->original_image = $data->original_image; 
  425.       $imageData->original_image_width = $imageAtts[1]; 
  426.       $imageData->original_image_height = $imageAtts[2]; 
  427.       $imageData->original_image_url = $this->get_image_src($data->original_image, 'full'); 
  428.  
  429.       // Set defaults to original image 
  430.       $imageData->image_url = $this->get_image_src($data->original_image, 'full'); 
  431.       $imageData->preview_image_url = $this->get_image_src($data->original_image, $field['preview_size']); 
  432.  
  433.       // Check if there is a cropped version and set appropriate attributes 
  434.       if(is_numeric($data->cropped_image)) { 
  435.           // Cropped image was saved to media library ans has an ID 
  436.           $imageData->cropped_image = $data->cropped_image; 
  437.           $imageData->image_url = $this->get_image_src($data->cropped_image, 'full'); 
  438.           $imageData->preview_image_url = $this->get_image_src($data->cropped_image, $field['preview_size']); 
  439.       } 
  440.       elseif(is_object($data->cropped_image)) { 
  441.           // Cropped image was not saved to media library and is only stored by its URL 
  442.           $imageData->cropped_image = $data->cropped_image; 
  443.  
  444.           // Generate appropriate URLs 
  445.           $mediaDir = wp_upload_dir(); 
  446.           $imageData->image_url = $mediaDir['baseurl'] . '/' .  $data->cropped_image->image; 
  447.           $imageData->preview_image_url = $mediaDir['baseurl'] . '/' . $data->cropped_image->preview; 
  448.       } 
  449.       return $imageData; 
  450.   }

  516.   function generate_cropped_image($id, $x1, $x2, $y1, $y2, $targetW, $targetH, $saveToMediaLibrary, $previewSize) {//$id, $x1, $x2, $y$, $y2, $targetW, $targetH) { 
  517.       require_once ABSPATH . "/wp-admin/includes/file.php"; 
  518.       require_once ABSPATH . "/wp-admin/includes/image.php"; 
  519.  
  520.       // Create the variable that will hold the new image data 
  521.       $imageData = array(); 
  522.  
  523.       // Fetch media library info 
  524.       $mediaDir = wp_upload_dir(); 
  525.  
  526.       // Get original image info 
  527.       $originalImageData = wp_get_attachment_metadata($id); 
  528.  
  529.       // Get image editor from original image path to crop the image 
  530.       $image = wp_get_image_editor( $mediaDir['basedir'] . '/' . $originalImageData['file'] ); 
  531.  
  532.       // Set quality 
  533.       $image->set_quality( apply_filters('acf-image-crop/image-quality', 100) ); 
  534.  
  535.       if(! is_wp_error( $image ) ) { 
  536.  
  537.           // Crop the image using the provided measurements 
  538.           $image->crop($x1, $y1, $x2 - $x1, $y2 - $y1, $targetW, $targetH); 
  539.  
  540.           // Retrieve original filename and seperate it from its file extension 
  541.           $originalFileName = explode('.', basename($originalImageData['file'])); 
  542.  
  543.           // Retrieve and remove file extension from array 
  544.           $originalFileExtension = array_pop($originalFileName); 
  545.  
  546.           // Generate new base filename 
  547.           $targetFileName = implode('.', $originalFileName) . '_' . $targetW . 'x' . $targetH . apply_filters('acf-image-crop/filename_postfix', '_acf_cropped')  . '.' . $originalFileExtension; 
  548.  
  549.  
  550.           // Generate target path new file using existing media library 
  551.           $targetFilePath = $mediaDir['path'] . '/' . wp_unique_filename( $mediaDir['path'], $targetFileName); 
  552.  
  553.           // Get the relative path to save as the actual image url 
  554.           $targetRelativePath = str_replace($mediaDir['basedir'] . '/', '', $targetFilePath); 
  555.  
  556.           // Save the image to the target path 
  557.           if(is_wp_error($image->save($targetFilePath))) { 
  558.               // There was an error saving the image 
  559.               //TODO handle it 
  560.           } 
  561.  
  562.           // If file should be saved to media library, create an attachment for it at get the new attachment ID 
  563.           if($saveToMediaLibrary) { 
  564.               // Generate attachment from created file 
  565.  
  566.               // Get the filetype 
  567.               $wp_filetype = wp_check_filetype(basename($targetFilePath), null ); 
  568.               $attachment = array( 
  569.                    'guid' => $mediaDir['url'] . '/' . basename($targetFilePath),  
  570.                    'post_mime_type' => $wp_filetype['type'],  
  571.                    'post_title' => preg_replace('/\.[^.]+$/', '', basename($targetFilePath)),  
  572.                    'post_content' => '',  
  573.                    'post_status' => 'inherit' 
  574.  ); 
  575.               $attachmentId = wp_insert_attachment( $attachment, $targetFilePath); 
  576.               $attachmentData = wp_generate_attachment_metadata( $attachmentId, $targetFilePath ); 
  577.               wp_update_attachment_metadata( $attachmentId, $attachmentData ); 
  578.               add_post_meta($attachmentId, 'acf_is_cropped', 'true', true); 
  579.  
  580.               // Add the id to the imageData-array 
  581.               $imageData['value'] = $attachmentId; 
  582.  
  583.               // Add the image url 
  584.               $imageUrlObject = wp_get_attachment_image_src( $attachmentId, 'full'); 
  585.               $imageData['url'] = $imageUrlObject[0]; 
  586.  
  587.               // Add the preview url as well 
  588.               $previewUrlObject = wp_get_attachment_image_src( $attachmentId, $previewSize); 
  589.               $imageData['preview_url'] = $previewUrlObject[0]; 
  590.           } 
  591.           // Else we need to return the actual path of the cropped image 
  592.           else{ 
  593.               // Add the image url to the imageData-array 
  594.               $imageData['value'] = array('image' => $targetRelativePath); 
  595.               $imageData['url'] = $mediaDir['baseurl'] . '/' . $targetRelativePath; 
  596.  
  597.               // Get preview size dimensions 
  598.               global $_wp_additional_image_sizes; 
  599.               $previewWidth = 0; 
  600.               $previewHeight = 0; 
  601.               $crop = 0; 
  602.               if (isset($_wp_additional_image_sizes[$previewSize])) { 
  603.                   $previewWidth = intval($_wp_additional_image_sizes[$previewSize]['width']); 
  604.                   $previewHeight = intval($_wp_additional_image_sizes[$previewSize]['height']); 
  605.                   $crop = $_wp_additional_image_sizes[$previewSize]['crop']; 
  606.               } else { 
  607.                   $previewWidth = get_option($previewSize.'_size_w'); 
  608.                   $previewHeight = get_option($previewSize.'_size_h'); 
  609.                   $crop = get_option($previewSize.'_crop'); 
  610.               } 
  611.  
  612.               // Generate preview file path 
  613.               $previewFilePath = $mediaDir['path'] . '/' . wp_unique_filename( $mediaDir['path'], 'preview_' . $targetFileName); 
  614.               $previewRelativePath = str_replace($mediaDir['basedir'] . '/', '', $previewFilePath); 
  615.  
  616.               // Get image editor from cropped image 
  617.               $croppedImage = wp_get_image_editor( $targetFilePath ); 
  618.               $croppedImage->resize($previewWidth, $previewHeight, $crop); 
  619.  
  620.               // Save the preview 
  621.               $croppedImage->save($previewFilePath); 
  622.  
  623.               // Add the preview url 
  624.               $imageData['preview_url'] = $mediaDir['baseurl'] . '/' . $previewRelativePath; 
  625.               $imageData['value']['preview'] = $previewRelativePath; 
  626.           } 
  627.           $imageData['success'] = true; 
  628.           return $imageData; 
  629.       } 
  630.       else{ 
  631.           // Handle WP_ERROR 
  632.           $response = array(); 
  633.           $response['success'] = false; 
  634.           $response['error_message'] = ''; 
  635.           foreach($image->error_data as $code => $message) { 
  636.               $response['error_message'] .= '<p><strong>' . $code . '</strong>: ' . $message . '</p>'; 
  637.           } 
  638.           return $response; 
  639.       } 
  640.   }
