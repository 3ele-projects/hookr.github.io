  679.   function get_image_data($field) { 
  680.       $imageData = new stdClass(); 
  681.       $imageData->original_image = ''; 
  682.       $imageData->original_image_width = ''; 
  683.       $imageData->original_image_height = ''; 
  684.       $imageData->cropped_image = ''; 
  685.       $imageData->original_image_url = ''; 
  686.       $imageData->preview_image_url = ''; 
  687.       $imageData->image_url = ''; 
  688.  
  689.       if($field['value'] == '') { 
  690.           // Field has not yet been saved or is an empty image field 
  691.           return $imageData; 
  692.       } 
  693.  
  694.       $data = json_decode($field['value']); 
  695.  
  696.       if(! is_object($data)) { 
  697.           // Field was saved as a regular image field 
  698.           $imageAtts = wp_get_attachment_image_src($field['value'], 'full'); 
  699.           $imageData->original_image = $field['value']; 
  700.           $imageData->original_image_width = $imageAtts[1]; 
  701.           $imageData->original_image_height = $imageAtts[2]; 
  702.           $imageData->preview_image_url = $this->get_image_src($field['value'], $field['preview_size']); 
  703.           $imageData->image_url = $this->get_image_src($field['value'], 'full'); 
  704.           $imageData->original_image_url = $imageData->image_url; 
  705.           return $imageData; 
  706.       } 
  707.  
  708.       if( !is_numeric($data->original_image) ) 
  709.       { 
  710.           // The field has been saved, but has no image 
  711.           return $imageData; 
  712.       } 
  713.  
  714.       // By now, we have at least a saved original image 
  715.       $imageAtts = wp_get_attachment_image_src($data->original_image, 'full'); 
  716.       $imageData->original_image = $data->original_image; 
  717.       $imageData->original_image_width = $imageAtts[1]; 
  718.       $imageData->original_image_height = $imageAtts[2]; 
  719.       $imageData->original_image_url = $this->get_image_src($data->original_image, 'full'); 
  720.  
  721.       // Set defaults to original image 
  722.       $imageData->image_url = $this->get_image_src($data->original_image, 'full'); 
  723.       $imageData->preview_image_url = $this->get_image_src($data->original_image, $field['preview_size']); 
  724.  
  725.       // Check if there is a cropped version and set appropriate attributes 
  726.       if(is_numeric($data->cropped_image)) { 
  727.           // Cropped image was saved to media library ans has an ID

  730.           $imageData->preview_image_url = $this->get_image_src($data->cropped_image, $field['preview_size']); 
  731.       } 
  732.       elseif(is_object($data->cropped_image)) {

  734.           $imageData->cropped_image = $data->cropped_image; 
  735.  
  736.           // Generate appropriate URLs 
  737.           $mediaDir = wp_upload_dir(); 
  738.           $imageData->image_url = $mediaDir['baseurl'] . '/' .  $data->cropped_image->image; 
  739.           $imageData->preview_image_url = $mediaDir['baseurl'] . '/' . $data->cropped_image->preview; 
  740.       } 
  741.       return $imageData; 
  742.   }

  755.   function generate_cropped_image($id, $x1, $x2, $y1, $y2, $targetW, $targetH, $saveToMediaLibrary, $previewSize) {//$id, $x1, $x2, $y$, $y2, $targetW, $targetH) { 
  756.       require_once ABSPATH . "/wp-admin/includes/file.php"; 
  757.       require_once ABSPATH . "/wp-admin/includes/image.php"; 
  758.  
  759.       // Create the variable that will hold the new image data 
  760.       $imageData = array(); 
  761.  
  762.       // Fetch media library info 
  763.       $mediaDir = wp_upload_dir(); 
  764.  
  765.       // Get original image info 
  766.       $originalImageData = wp_get_attachment_metadata($id); 
  767.  
  768.       // Get image editor from original image path to crop the image 
  769.       $image = wp_get_image_editor( $mediaDir['basedir'] . '/' . $originalImageData['file'] ); 
  770.  
  771.       if(! is_wp_error( $image ) ) { 
  772.  
  773.           // Crop the image using the provided measurements 
  774.           $image->crop($x1, $y1, $x2 - $x1, $y2 - $y1, $targetW, $targetH); 
  775.  
  776.           // Retrieve original filename and seperate it from its file extension 
  777.           $originalFileName = explode('.', basename($originalImageData['file'])); 
  778.  
  779.           // Retrieve and remove file extension from array 
  780.           $originalFileExtension = array_pop($originalFileName); 
  781.  
  782.           // Generate new base filename 
  783.           $targetFileName = implode('.', $originalFileName) . '_' . $targetW . 'x' . $targetH . apply_filters('acf-image-crop/filename_postfix', '_acf_cropped')  . '.' . $originalFileExtension; 
  784.  
  785.           // Generate target path new file using existing media library 
  786.           $targetFilePath = $mediaDir['path'] . '/' . wp_unique_filename( $mediaDir['path'], $targetFileName); 
  787.  
  788.           // Get the relative path to save as the actual image url 
  789.           $targetRelativePath = str_replace($mediaDir['basedir'] . '/', '', $targetFilePath); 
  790.  
  791.           // Save the image to the target path 
  792.           if(is_wp_error($image->save($targetFilePath))) { 
  793.               // There was an error saving the image 
  794.               //TODO handle it 
  795.           } 
  796.  
  797.           // If file should be saved to media library, create an attachment for it at get the new attachment ID 
  798.           if($saveToMediaLibrary) { 
  799.               // Generate attachment from created file 
  800.  
  801.               // Get the filetype 
  802.               $wp_filetype = wp_check_filetype(basename($targetFilePath), null ); 
  803.               $attachment = array( 
  804.                    'guid' => $mediaDir['url'] . '/' . basename($targetFilePath),  
  805.                    'post_mime_type' => $wp_filetype['type'],  
  806.                    'post_title' => preg_replace('/\.[^.]+$/', '', basename($targetFilePath)),  
  807.                    'post_content' => '',  
  808.                    'post_status' => 'inherit' 
  809.  ); 
  810.               $attachmentId = wp_insert_attachment( $attachment, $targetFilePath); 
  811.               $attachmentData = wp_generate_attachment_metadata( $attachmentId, $targetFilePath ); 
  812.               wp_update_attachment_metadata( $attachmentId, $attachmentData ); 
  813.               add_post_meta($attachmentId, 'acf_is_cropped', 'true', true); 
  814.  
  815.               // Add the id to the imageData-array 
  816.               $imageData['value'] = $attachmentId; 
  817.  
  818.               // Add the image url 
  819.               $imageUrlObject = wp_get_attachment_image_src( $attachmentId, 'full'); 
  820.               $imageData['url'] = $imageUrlObject[0]; 
  821.  
  822.               // Add the preview url as well 
  823.               $previewUrlObject = wp_get_attachment_image_src( $attachmentId, $previewSize); 
  824.               $imageData['preview_url'] = $previewUrlObject[0]; 
  825.           } 
  826.           // Else we need to return the actual path of the cropped image 
  827.           else{ 
  828.               // Add the image url to the imageData-array 
  829.               $imageData['value'] = array('image' => $targetRelativePath); 
  830.               $imageData['url'] = $mediaDir['baseurl'] . '/' . $targetRelativePath; 
  831.  
  832.               // Get preview size dimensions 
  833.               global $_wp_additional_image_sizes; 
  834.               $previewWidth = 0; 
  835.               $previewHeight = 0; 
  836.               $crop = 0; 
  837.               if (isset($_wp_additional_image_sizes[$previewSize])) { 
  838.                   $previewWidth = intval($_wp_additional_image_sizes[$previewSize]['width']); 
  839.                   $previewHeight = intval($_wp_additional_image_sizes[$previewSize]['height']); 
  840.                   $crop = $_wp_additional_image_sizes[$previewSize]['crop']; 
  841.               } else { 
  842.                   $previewWidth = get_option($previewSize.'_size_w'); 
  843.                   $previewHeight = get_option($previewSize.'_size_h'); 
  844.                   $crop = get_option($previewSize.'_crop'); 
  845.               } 
  846.  
  847.               // Generate preview file path 
  848.               $previewFilePath = $mediaDir['path'] . '/' . wp_unique_filename( $mediaDir['path'], 'preview_' . $targetFileName); 
  849.               $previewRelativePath = str_replace($mediaDir['basedir'] . '/', '', $previewFilePath); 
  850.  
  851.               // Get image editor from cropped image 
  852.               $croppedImage = wp_get_image_editor( $targetFilePath ); 
  853.               $croppedImage->resize($previewWidth, $previewHeight, $crop); 
  854.  
  855.               // Save the preview 
  856.               $croppedImage->save($previewFilePath); 
  857.  
  858.               // Add the preview url 
  859.               $imageData['preview_url'] = $mediaDir['baseurl'] . '/' . $previewRelativePath; 
  860.               $imageData['value']['preview'] = $previewRelativePath; 
  861.           } 
  862.           $imageData['success'] = true; 
  863.           return $imageData; 
  864.       } 
  865.       else{ 
  866.            // Handle WP_ERROR 
  867.           $response = array(); 
  868.           $response['success'] = false; 
  869.           $response['error_message'] = ''; 
  870.           foreach($image->error_data as $code => $message) { 
  871.               $response['error_message'] .= '<p><strong>' . $code . '</strong>: ' . $message . '</p>'; 
  872.           } 
  873.           return $response; 
  874.       }
