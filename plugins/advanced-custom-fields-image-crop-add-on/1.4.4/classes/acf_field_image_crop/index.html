  384.     function get_image_data($field) { 
  385.         $imageData = new stdClass(); 
  386.         $imageData->original_image = ''; 
  387.         $imageData->original_image_width = ''; 
  388.         $imageData->original_image_height = ''; 
  389.         $imageData->cropped_image = ''; 
  390.         $imageData->original_image_url = ''; 
  391.         $imageData->preview_image_url = ''; 
  392.         $imageData->image_url = ''; 
  393.  
  394.         if($field['value'] == '') { 
  395.             // Field has not yet been saved or is an empty image field 
  396.             return $imageData; 
  397.         } 
  398.  
  399.         $data = json_decode($field['value']); 
  400.  
  401.         if(! is_object($data)) { 
  402.             // Field was saved as a regular image field 
  403.             $imageAtts = wp_get_attachment_image_src($field['value'], 'full'); 
  404.             $imageData->original_image = $field['value']; 
  405.             $imageData->original_image_width = $imageAtts[1]; 
  406.             $imageData->original_image_height = $imageAtts[2]; 
  407.             $imageData->preview_image_url = $this->get_image_src($field['value'], $field['preview_size']); 
  408.             $imageData->image_url = $this->get_image_src($field['value'], 'full'); 
  409.             $imageData->original_image_url = $imageData->image_url; 
  410.             return $imageData; 
  411.         } 
  412.  
  413.         if( !is_numeric($data->original_image) ) 
  414.         { 
  415.             // The field has been saved, but has no image 
  416.             return $imageData; 
  417.         } 
  418.  
  419.         // By now, we have at least a saved original image 
  420.         $imageAtts = wp_get_attachment_image_src($data->original_image, 'full'); 
  421.         $imageData->original_image = $data->original_image; 
  422.         $imageData->original_image_width = $imageAtts[1]; 
  423.         $imageData->original_image_height = $imageAtts[2]; 
  424.         $imageData->original_image_url = $this->get_image_src($data->original_image, 'full'); 
  425.  
  426.         // Set defaults to original image 
  427.         $imageData->image_url = $this->get_image_src($data->original_image, 'full'); 
  428.         $imageData->preview_image_url = $this->get_image_src($data->original_image, $field['preview_size']); 
  429.  
  430.         // Check if there is a cropped version and set appropriate attributes 
  431.         if(is_numeric($data->cropped_image)) { 
  432.             // Cropped image was saved to media library ans has an ID 
  433.             $imageData->cropped_image = $data->cropped_image; 
  434.             $imageData->image_url = $this->get_image_src($data->cropped_image, 'full'); 
  435.             $imageData->preview_image_url = $this->get_image_src($data->cropped_image, $field['preview_size']); 
  436.         } 
  437.         elseif(is_object($data->cropped_image)) { 
  438.             // Cropped image was not saved to media library and is only stored by its URL 
  439.             $imageData->cropped_image = $data->cropped_image; 
  440.  
  441.             // Generate appropriate URLs 
  442.             $mediaDir = wp_upload_dir(); 
  443.             $imageData->image_url = $mediaDir['baseurl'] . '/' .  $data->cropped_image->image; 
  444.             $imageData->preview_image_url = $mediaDir['baseurl'] . '/' . $data->cropped_image->preview; 
  445.         } 
  446.         return $imageData; 
  447.     }

  514.     function generate_cropped_image($id, $x1, $x2, $y1, $y2, $targetW, $targetH, $saveToMediaLibrary, $previewSize) {//$id, $x1, $x2, $y$, $y2, $targetW, $targetH) { 
  515.         require_once ABSPATH . "/wp-admin/includes/file.php"; 
  516.         require_once ABSPATH . "/wp-admin/includes/image.php"; 
  517.  
  518.         // Create the variable that will hold the new image data 
  519.         $imageData = array(); 
  520.  
  521.         // Fetch media library info 
  522.         $mediaDir = wp_upload_dir(); 
  523.  
  524.         // Get original image info 
  525.         $originalImageData = wp_get_attachment_metadata($id); 
  526.  
  527.         // Get image editor from original image path to crop the image 
  528.         $image = wp_get_image_editor( $mediaDir['basedir'] . '/' . $originalImageData['file'] ); 
  529.  
  530.         if(! is_wp_error( $image ) ) { 
  531.  
  532.             // Crop the image using the provided measurements 
  533.             $image->crop($x1, $y1, $x2 - $x1, $y2 - $y1, $targetW, $targetH); 
  534.  
  535.             // Retrieve original filename and seperate it from its file extension 
  536.             $originalFileName = explode('.', basename($originalImageData['file'])); 
  537.  
  538.             // Retrieve and remove file extension from array 
  539.             $originalFileExtension = array_pop($originalFileName); 
  540.  
  541.             // Generate new base filename 
  542.             $targetFileName = implode('.', $originalFileName) . '_' . $targetW . 'x' . $targetH . '_acf_cropped'  . '.' . $originalFileExtension; 
  543.  
  544.             // Generate target path new file using existing media library 
  545.             $targetFilePath = $mediaDir['path'] . '/' . wp_unique_filename( $mediaDir['path'], $targetFileName); 
  546.  
  547.             // Get the relative path to save as the actual image url 
  548.             $targetRelativePath = str_replace($mediaDir['basedir'] . '/', '', $targetFilePath); 
  549.  
  550.             // Save the image to the target path 
  551.             if(is_wp_error($image->save($targetFilePath))) { 
  552.                 // There was an error saving the image 
  553.                 //TODO handle it 
  554.             } 
  555.  
  556.             // If file should be saved to media library, create an attachment for it at get the new attachment ID 
  557.             if($saveToMediaLibrary) { 
  558.                 // Generate attachment from created file 
  559.  
  560.                 // Get the filetype 
  561.                 $wp_filetype = wp_check_filetype(basename($targetFilePath), null ); 
  562.                 $attachment = array( 
  563.                      'guid' => $mediaDir['url'] . '/' . basename($targetFilePath),  
  564.                      'post_mime_type' => $wp_filetype['type'],  
  565.                      'post_title' => preg_replace('/\.[^.]+$/', '', basename($targetFilePath)),  
  566.                      'post_content' => '',  
  567.                      'post_status' => 'inherit' 
  568.  ); 
  569.                 $attachmentId = wp_insert_attachment( $attachment, $targetFilePath); 
  570.                 $attachmentData = wp_generate_attachment_metadata( $attachmentId, $targetFilePath ); 
  571.                 wp_update_attachment_metadata( $attachmentId, $attachmentData ); 
  572.                 add_post_meta($attachmentId, 'acf_is_cropped', 'true', true); 
  573.  
  574.                 // Add the id to the imageData-array 
  575.                 $imageData['value'] = $attachmentId; 
  576.  
  577.                 // Add the image url 
  578.                 $imageUrlObject = wp_get_attachment_image_src( $attachmentId, 'full'); 
  579.                 $imageData['url'] = $imageUrlObject[0]; 
  580.  
  581.                 // Add the preview url as well 
  582.                 $previewUrlObject = wp_get_attachment_image_src( $attachmentId, $previewSize); 
  583.                 $imageData['preview_url'] = $previewUrlObject[0]; 
  584.             } 
  585.             // Else we need to return the actual path of the cropped image 
  586.             else{ 
  587.                 // Add the image url to the imageData-array 
  588.                 $imageData['value'] = array('image' => $targetRelativePath); 
  589.                 $imageData['url'] = $mediaDir['baseurl'] . '/' . $targetRelativePath; 
  590.  
  591.                 // Get preview size dimensions 
  592.                 global $_wp_additional_image_sizes; 
  593.                 $previewWidth = 0; 
  594.                 $previewHeight = 0; 
  595.                 $crop = 0; 
  596.                 if (isset($_wp_additional_image_sizes[$previewSize])) { 
  597.                     $previewWidth = intval($_wp_additional_image_sizes[$previewSize]['width']); 
  598.                     $previewHeight = intval($_wp_additional_image_sizes[$previewSize]['height']); 
  599.                     $crop = $_wp_additional_image_sizes[$previewSize]['crop']; 
  600.                 } else { 
  601.                     $previewWidth = get_option($previewSize.'_size_w'); 
  602.                     $previewHeight = get_option($previewSize.'_size_h'); 
  603.                     $crop = get_option($previewSize.'_crop'); 
  604.                 } 
  605.  
  606.                 // Generate preview file path 
  607.                 $previewFilePath = $mediaDir['path'] . '/' . wp_unique_filename( $mediaDir['path'], 'preview_' . $targetFileName); 
  608.                 $previewRelativePath = str_replace($mediaDir['basedir'] . '/', '', $previewFilePath); 
  609.  
  610.                 // Get image editor from cropped image 
  611.                 $croppedImage = wp_get_image_editor( $targetFilePath ); 
  612.                 $croppedImage->resize($previewWidth, $previewHeight, $crop); 
  613.  
  614.                 // Save the preview 
  615.                 $croppedImage->save($previewFilePath); 
  616.  
  617.                 // Add the preview url 
  618.                 $imageData['preview_url'] = $mediaDir['baseurl'] . '/' . $previewRelativePath; 
  619.                 $imageData['value']['preview'] = $previewRelativePath; 
  620.             } 
  621.             $imageData['success'] = true; 
  622.             return $imageData; 
  623.         } 
  624.         else{ 
  625.             // Handle WP_ERROR 
  626.             $response = array(); 
  627.             $response['success'] = false; 
  628.             $response['error_message'] = ''; 
  629.             foreach($image->error_data as $code => $message) { 
  630.                 $response['error_message'] .= '<p><strong>' . $code . '</strong>: ' . $message . '</p>'; 
  631.             } 
  632.             return $response; 
  633.         } 
  634.     }
