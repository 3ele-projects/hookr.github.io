  193. function bpge_get_nav_order() { 
  194.   global $bp, $bpge; 
  195.  
  196.   if ( empty( $bpge ) ) { 
  197.       $bpge = bpge_get_options(); 
  198.   } 
  199.  
  200.   if ( bp_is_group() && bp_is_single_item() ) { 
  201.       $order = groups_get_groupmeta( $bp->groups->current_group->id, 'bpge_nav_order' ); 
  202.  
  203.       if ( ! empty( $order ) && is_array( $order ) ) { 
  204.           foreach ( $order as $slug => $position ) { 
  205.               if ( bpge_is_bp_26() ) { 
  206.                   buddypress()->groups->nav->edit_nav( array( 'position' => $position ), $slug, bp_current_item() ); 
  207.               } else { 
  208.                   if ( isset( $bp->bp_options_nav[ $bp->groups->current_group->slug ][ $slug ] ) ) { 
  209.                       $bp->bp_options_nav[ $bp->groups->current_group->slug ][ $slug ]['position'] = $position; 
  210.                   } 
  211.               } 
  212.           } 
  213.       } 
  214.  
  215.       do_action( 'bpge_get_nav_order' ); 
  216.  
  217.       return bpge_get_group_nav(); 
  218.   } 
  219.  
  220.   return false; 
  221. }

  232. function bpge_landing_page( $old_slug ) {

  235.   $new_slug = $old_slug; 
  236.  
  237.   if ( bp_is_group() && bp_is_single_item() && 
  238.        ( 
  239.            ( is_array( $bpge['groups'] ) && in_array( $bp->groups->current_group->id, $bpge['groups'] ) ) 
  240.            || 
  241.            ( is_string( $bpge['groups'] ) && $bpge['groups'] == 'all' ) 
  242.  ) 
  243.  ) { 
  244.       // get all pages - take the first 
  245.       $order = groups_get_groupmeta( $bp->groups->current_group->id, 'bpge_nav_order' ); 
  246.  
  247.       if ( is_array( $order ) && ! empty( $order ) ) { 
  248.           $flipped = array_flip( $order ); 
  249.           $new_slug = reset( $flipped ); 
  250.       } 
  251.   } 
  252.  
  253.   return apply_filters( 'bpge_landing_page', $new_slug ); 
  254. }
