  491.       public function save_post( $post_id, $post = false ) { 
  492.  
  493.           $post_type = $post ? $post->post_type : get_post_type( $post_id ); 
  494.  
  495.           // check permissions 
  496.           if ( 
  497.               // check nonce 
  498.               ! isset( $_POST['wp_meta_box_nonce'] ) 
  499.               || ! wp_verify_nonce( $_POST['wp_meta_box_nonce'], self::nonce() ) 
  500.               // check if autosave 
  501.               || defined('DOING_AUTOSAVE' ) && DOING_AUTOSAVE 
  502.               // check user editing permissions 
  503.               || ( 'page' == $_POST['post_type'] && ! current_user_can( 'edit_page', $post_id ) ) 
  504.               || ! current_user_can( 'edit_post', $post_id ) 
  505.               // get the metabox post_types & compare it to this post_type 
  506.               || ! in_array( $post_type, $this->_meta_box['pages'] ) 
  507.  ) 
  508.               return $post_id; 
  509.  
  510.           self::save_fields( $this->_meta_box, $post_id, 'post' ); 
  511.       }
